23/* Codes in a variable definition saying where the definition came from.
24 Increasing numeric values signify less-overridable definitions. */
25enum variable_origin
26 {
27 o_default, /* Variable from the default set. */
28 o_env, /* Variable from environment. */
29 o_file, /* Variable given in a makefile. */
30 o_env_override, /* Variable from environment, if -e. */
31 o_command, /* Variable given by user. */
32 o_override, /* Variable from an 'override' directive. */
33 o_automatic, /* Automatic variable -- cannot be set. */
34 o_debugger, /* Set inside debugger. */
35 o_invalid /* Core dump time. */
36 };
37
38enum variable_flavor
39 {
40 f_bogus, /* Bogus (error) */
41 f_simple, /* Simple definition (:= or ::=) */
42 f_recursive, /* Recursive definition (=) */
43 f_append, /* Appending definition (+=) */
44 f_conditional, /* Conditional definition (?=) */
45 f_shell /* Shell assignment (!=) */
46 };

48/* Structure that represents one variable definition.
49 Each bucket of the hash table is a chain of these,
50 chained through 'next'. */
51
52#define EXP_COUNT_BITS 15 /* This gets all the bitfields into 32 bits */
53#define EXP_COUNT_MAX ((1<<EXP_COUNT_BITS)-1)
54
55struct variable
56 {
57 char *name; /* Variable name. */
58 char *value; /* Variable value. */
59 gmk_floc fileinfo; /* Where the variable was defined. */
60 int length; /* strlen (name) */
61 unsigned int recursive:1; /* Gets recursively re-evaluated. */
62 unsigned int append:1; /* Nonzero if an appending target-specific
63 variable. */
64 unsigned int conditional:1; /* Nonzero if set with a ?=. */
65 unsigned int per_target:1; /* Nonzero if a target-specific variable. */
66 unsigned int special:1; /* Nonzero if this is a special variable. */
67 unsigned int exportable:1; /* Nonzero if the variable _could_ be
68 exported. */
69 unsigned int expanding:1; /* Nonzero if currently being expanded. */
70 unsigned int private_var:1; /* Nonzero avoids inheritance of this
71 target-specific variable. */
72 unsigned int exp_count:EXP_COUNT_BITS;
73 /* If >1, allow this many self-referential
74 expansions. */
75 enum variable_flavor
76 flavor ENUM_BITFIELD (4); /* Variable flavor. */
77 enum variable_origin
78 origin ENUM_BITFIELD (4); /* Variable origin. */
79 enum variable_export
80 {
81 v_export, /* Export this variable. */
82 v_noexport, /* Don't export this variable. */
83 v_ifset, /* Export it if it has a non-default value. */
84 v_default /* Decide in target_environment. */
85 } export ENUM_BITFIELD (2);
86 };

88/* Structure that represents a variable set. */
89
90struct variable_set
91 {
92 struct hash_table table; /* Hash table of variables. */
93 };
94
95/* Structure that represents a list of variable sets. */
96
97struct variable_set_list
98 {
99 struct variable_set_list *next; /* Link in the chain. */
100 struct variable_set *set; /* Variable set. */
101 int next_is_parent; /* True if next is a parent target. */
102 };
103
104/* Structure used for pattern-specific variables. */
105
106struct pattern_var
107 {
108 struct pattern_var *next;
109 const char *suffix;
110 const char *target;
111 unsigned int len;
112 struct variable variable;
113 };
