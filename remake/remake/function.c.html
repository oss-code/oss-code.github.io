69/* Store into VARIABLE_BUFFER at O the result of scanning TEXT and replacing
70 each occurrence of SUBST with REPLACE. TEXT is null-terminated. SLEN is
71 the length of SUBST and RLEN is the length of REPLACE. If BY_WORD is
72 nonzero, substitutions are done only on matches which are complete
73 whitespace-delimited words. */
74
75char *
76subst_expand (char *o, const char *text, const char *subst, const char *replace,
77 unsigned int slen, unsigned int rlen, int by_word)
78{
79 const char *t = text;
80 const char *p;
81
82 if (slen == 0 && !by_word)
83 {
84 /* The first occurrence of "" in any string is its end. */
85 o = variable_buffer_output (o, t, strlen (t));
86 if (rlen > 0)
87 o = variable_buffer_output (o, replace, rlen);
88 return o;
89 }
90
91 do
92 {
93 if (by_word && slen == 0)
94 /* When matching by words, the empty string should match
95 the end of each word, rather than the end of the whole text. */
96 p = end_of_token (next_token (t));
97 else
98 {
99 p = strstr (t, subst);
100 if (p == 0)
101 {
102 /* No more matches. Output everything left on the end. */
103 o = variable_buffer_output (o, t, strlen (t));
104 return o;
105 }
106 }
107
108 /* Output everything before this occurrence of the string to replace. */
109 if (p > t)
110 o = variable_buffer_output (o, t, p - t);
111
112 /* If we're substituting only by fully matched words,
113 or only at the ends of words, check that this case qualifies. */
114 if (by_word
115 && ((p > text && !ISSPACE (p[-1]))
116 || ! STOP_SET (p[slen], MAP_SPACE|MAP_NUL)))
117 /* Struck out. Output the rest of the string that is
118 no longer to be replaced. */
119 o = variable_buffer_output (o, subst, slen);
120 else if (rlen > 0)
121 /* Output the replacement string. */
122 o = variable_buffer_output (o, replace, rlen);
123
124 /* Advance T past the string to be replaced. */
125 t = p + slen;
126 } while (*t != '\0');
127
128 return o;
129}

132/* Store into VARIABLE_BUFFER at O the result of scanning TEXT
133 and replacing strings matching PATTERN with REPLACE.
134 If PATTERN_PERCENT is not nil, PATTERN has already been
135 run through find_percent, and PATTERN_PERCENT is the result.
136 If REPLACE_PERCENT is not nil, REPLACE has already been
137 run through find_percent, and REPLACE_PERCENT is the result.
138 Note that we expect PATTERN_PERCENT and REPLACE_PERCENT to point to the
139 character _AFTER_ the %, not to the % itself.
140*/
141
142char *
143patsubst_expand_pat (char *o, const char *text,
144 const char *pattern, const char *replace,
145 const char *pattern_percent, const char *replace_percent)
146{
147 unsigned int pattern_prepercent_len, pattern_postpercent_len;
148 unsigned int replace_prepercent_len, replace_postpercent_len;
149 const char *t;
150 unsigned int len;
151 int doneany = 0;
152
153 /* Record the length of REPLACE before and after the % so we don't have to
154 compute these lengths more than once. */
155 if (replace_percent)
156 {
157 replace_prepercent_len = replace_percent - replace - 1;
158 replace_postpercent_len = strlen (replace_percent);
159 }
160 else
161 {
162 replace_prepercent_len = strlen (replace);
163 replace_postpercent_len = 0;
164 }
165
166 if (!pattern_percent)
167 /* With no % in the pattern, this is just a simple substitution. */
168 return subst_expand (o, text, pattern, replace,
169 strlen (pattern), strlen (replace), 1);
170
171 /* Record the length of PATTERN before and after the %
172 so we don't have to compute it more than once. */
173 pattern_prepercent_len = pattern_percent - pattern - 1;
174 pattern_postpercent_len = strlen (pattern_percent);
175
176 while ((t = find_next_token (&text, &len)) != 0)
177 {
178 int fail = 0;
179
180 /* Is it big enough to match? */
181 if (len < pattern_prepercent_len + pattern_postpercent_len)
182 fail = 1;
183
184 /* Does the prefix match? */
185 if (!fail && pattern_prepercent_len > 0
186 && (*t != *pattern
187 || t[pattern_prepercent_len - 1] != pattern_percent[-2]
188 || !strneq (t + 1, pattern + 1, pattern_prepercent_len - 1)))
189 fail = 1;
190
191 /* Does the suffix match? */
192 if (!fail && pattern_postpercent_len > 0
193 && (t[len - 1] != pattern_percent[pattern_postpercent_len - 1]
194 || t[len - pattern_postpercent_len] != *pattern_percent
195 || !strneq (&t[len - pattern_postpercent_len],
196 pattern_percent, pattern_postpercent_len - 1)))
197 fail = 1;
198
199 if (fail)
200 /* It didn't match. Output the string. */
201 o = variable_buffer_output (o, t, len);
202 else
203 {
204 /* It matched. Output the replacement. */
205
206 /* Output the part of the replacement before the %. */
207 o = variable_buffer_output (o, replace, replace_prepercent_len);
208
209 if (replace_percent != 0)
210 {
211 /* Output the part of the matched string that
212 matched the % in the pattern. */
213 o = variable_buffer_output (o, t + pattern_prepercent_len,
214 len - (pattern_prepercent_len
215 + pattern_postpercent_len));
216 /* Output the part of the replacement after the %. */
217 o = variable_buffer_output (o, replace_percent,
218 replace_postpercent_len);
219 }
220 }
221
222 /* Output a space, but not if the replacement is "". */
223 if (fail || replace_prepercent_len > 0
224 || (replace_percent != 0 && len + replace_postpercent_len > 0))
225 {
226 o = variable_buffer_output (o, " ", 1);
227 doneany = 1;
228 }
229 }
230 if (doneany)
231 /* Kill the last space. */
232 --o;
233
234 return o;
235}

237/* Store into VARIABLE_BUFFER at O the result of scanning TEXT
238 and replacing strings matching PATTERN with REPLACE.
239 If PATTERN_PERCENT is not nil, PATTERN has already been
240 run through find_percent, and PATTERN_PERCENT is the result.
241 If REPLACE_PERCENT is not nil, REPLACE has already been
242 run through find_percent, and REPLACE_PERCENT is the result.
243 Note that we expect PATTERN_PERCENT and REPLACE_PERCENT to point to the
244 character _AFTER_ the %, not to the % itself.
245*/
246
247char *
248patsubst_expand (char *o, const char *text, char *pattern, char *replace)
249{
250 const char *pattern_percent = find_percent (pattern);
251 const char *replace_percent = find_percent (replace);
252
253 /* If there's a percent in the pattern or replacement skip it. */
254 if (replace_percent)
255 ++replace_percent;
256 if (pattern_percent)
257 ++pattern_percent;
258
259 return patsubst_expand_pat (o, text, pattern, replace,
260 pattern_percent, replace_percent);
261}

266static const struct function_table_entry *
267lookup_function (const char *s)
268{
269 struct function_table_entry function_table_entry_key;
270 const char *e = s;
271
272 while (STOP_SET (*e, MAP_USERFUNC))
273 e++;
274
275 if (e == s || !STOP_SET(*e, MAP_NUL|MAP_SPACE))
276 return NULL;
277
278 function_table_entry_key.name = s;
279 function_table_entry_key.len = e - s;
280
281 return hash_find_item (&function_table, &function_table_entry_key);
282}

285/* Return 1 if PATTERN matches STR, 0 if not. */
286
287int
288pattern_matches (const char *pattern, const char *percent, const char *str)
289{
290 unsigned int sfxlen, strlength;
291
292 if (percent == 0)
293 {
294 unsigned int len = strlen (pattern) + 1;
295 char *new_chars = alloca (len);
296 memcpy (new_chars, pattern, len);
297 percent = find_percent (new_chars);
298 if (percent == 0)
299 return streq (new_chars, str);
300 pattern = new_chars;
301 }
302
303 sfxlen = strlen (percent + 1);
304 strlength = strlen (str);
305
306 if (strlength < (percent - pattern) + sfxlen
307 || !strneq (pattern, str, percent - pattern))
308 return 0;
309
310 return !strcmp (percent + 1, str + (strlength - sfxlen));
311}

314/* Find the next comma or ENDPAREN (counting nested STARTPAREN and
315 ENDPARENtheses), starting at PTR before END. Return a pointer to
316 next character.
317
318 If no next argument is found, return NULL.
319*/
320
321static char *
322find_next_argument (char startparen, char endparen,
323 const char *ptr, const char *end)
324{
325 int count = 0;
326
327 for (; ptr < end; ++ptr)
328 if (*ptr == startparen)
329 ++count;
330
331 else if (*ptr == endparen)
332 {
333 --count;
334 if (count < 0)
335 return NULL;
336 }
337
338 else if (*ptr == ',' && !count)
339 return (char *)ptr;
340
341 /* We didn't find anything. */
342 return NULL;
343}

346/* Glob-expand LINE. The returned pointer is
347 only good until the next call to string_glob. */
348
349static char *
350string_glob (char *line)
351{
352 static char *result = 0;
353 static unsigned int length;
354 struct nameseq *chain;
355 unsigned int idx;
356
357 chain = PARSE_FILE_SEQ (&line, struct nameseq, MAP_NUL, NULL,
358 /* We do not want parse_file_seq to strip './'s.
359 That would break examples like:
360 $(patsubst ./%.c,obj/%.o,$(wildcard ./?*.c)). */
361 PARSEFS_NOSTRIP|PARSEFS_NOCACHE|PARSEFS_EXISTS);
362
363 if (result == 0)
364 {
365 length = 100;
366 result = xmalloc (100);
367 }
368
369 idx = 0;
370 while (chain != 0)
371 {
372 struct nameseq *next = chain->next;
373 unsigned int len = strlen (chain->name);
374
375 if (idx + len + 1 > length)
376 {
377 length += (len + 1) * 2;
378 result = xrealloc (result, length);
379 }
380 memcpy (&result[idx], chain->name, len);
381 idx += len;
382 result[idx++] = ' ';
383
384 /* Because we used PARSEFS_NOCACHE above, we have to free() NAME. */
385 free ((char *)chain->name);
386 free (chain);
387 chain = next;
388 }
389
390 /* Kill the last space and terminate the string. */
391 if (idx == 0)
392 result[0] = '\0';
393 else
394 result[idx - 1] = '\0';
395
396 return result;
397}

411static char *
412func_join (char *o, char **argv, const char *funcname UNUSED)
413{
414 int doneany = 0;
415
416 /* Write each word of the first argument directly followed
417 by the corresponding word of the second argument.
418 If the two arguments have a different number of words,
419 the excess words are just output separated by blanks. */
420 const char *tp;
421 const char *pp;
422 const char *list1_iterator = argv[0];
423 const char *list2_iterator = argv[1];
424 do
425 {
426 unsigned int len1, len2;
427
428 tp = find_next_token (&list1_iterator, &len1);
429 if (tp != 0)
430 o = variable_buffer_output (o, tp, len1);
431
432 pp = find_next_token (&list2_iterator, &len2);
433 if (pp != 0)
434 o = variable_buffer_output (o, pp, len2);
435
436 if (tp != 0 || pp != 0)
437 {
438 o = variable_buffer_output (o, " ", 1);
439 doneany = 1;
440 }
441 }
442 while (tp != 0 || pp != 0);
443 if (doneany)
444 /* Kill the last blank. */
445 --o;
446
447 return o;
448}

508static char *
509func_notdir_suffix (char *o, char **argv, const char *funcname)
510{
511 /* Expand the argument. */
512 const char *list_iterator = argv[0];
513 const char *p2;
514 int doneany =0;
515 unsigned int len=0;
516
517 int is_suffix = funcname[0] == 's';
518 int is_notdir = !is_suffix;
519 int stop = MAP_DIRSEP | (is_suffix ? MAP_DOT : 0);
520#ifdef VMS
521 /* For VMS list_iterator points to a comma separated list. To use the common
522 [find_]next_token, create a local copy and replace the commas with
523 spaces. Obviously, there is a problem if there is a ',' in the VMS filename
524 (can only happen on ODS5), the same problem as with spaces in filenames,
525 which seems to be present in make on all platforms. */
526 char *vms_list_iterator = alloca(strlen(list_iterator) + 1);
527 int i;
528 for (i = 0; list_iterator[i]; i++)
529 if (list_iterator[i] == ',')
530 vms_list_iterator[i] = ' ';
531 else
532 vms_list_iterator[i] = list_iterator[i];
533 vms_list_iterator[i] = list_iterator[i];
534 while ((p2 = find_next_token((const char**) &vms_list_iterator, &len)) != 0)
535#else
536 while ((p2 = find_next_token (&list_iterator, &len)) != 0)
537#endif
538 {
539 const char *p = p2 + len - 1;
540
541 while (p >= p2 && ! STOP_SET (*p, stop))
542 --p;
543
544 if (p >= p2)
545 {
546 if (is_notdir)
547 ++p;
548 else if (*p != '.')
549 continue;
550 o = variable_buffer_output (o, p, len - (p - p2));
551 }
552#ifdef HAVE_DOS_PATHS
553 /* Handle the case of "d:foo/bar". */
554 else if (is_notdir && p2[0] && p2[1] == ':')
555 {
556 p = p2 + 2;
557 o = variable_buffer_output (o, p, len - (p - p2));
558 }
559#endif
560 else if (is_notdir)
561 o = variable_buffer_output (o, p2, len);
562
563 if (is_notdir || p >= p2)
564 {
565#ifdef VMS
566 if (vms_comma_separator)
567 o = variable_buffer_output (o, ",", 1);
568 else
569#endif
570 o = variable_buffer_output (o, " ", 1);
571
572 doneany = 1;
573 }
574 }
575
576 if (doneany)
577 /* Kill last space. */
578 --o;
579
580 return o;
581}

584static char *
585func_basename_dir (char *o, char **argv, const char *funcname)
586{
587 /* Expand the argument. */
588 const char *p3 = argv[0];
589 const char *p2;
590 int doneany = 0;
591 unsigned int len = 0;
592
593 int is_basename = funcname[0] == 'b';
594 int is_dir = !is_basename;
595 int stop = MAP_DIRSEP | (is_basename ? MAP_DOT : 0) | MAP_NUL;
596#ifdef VMS
597 /* As in func_notdir_suffix ... */
598 char *vms_p3 = alloca (strlen(p3) + 1);
599 int i;
600 for (i = 0; p3[i]; i++)
601 if (p3[i] == ',')
602 vms_p3[i] = ' ';
603 else
604 vms_p3[i] = p3[i];
605 vms_p3[i] = p3[i];
606 while ((p2 = find_next_token((const char**) &vms_p3, &len)) != 0)
607#else
608 while ((p2 = find_next_token (&p3, &len)) != 0)
609#endif
610 {
611 const char *p = p2 + len - 1;
612 while (p >= p2 && ! STOP_SET (*p, stop))
613 --p;
614
615 if (p >= p2 && (is_dir))
616 o = variable_buffer_output (o, p2, ++p - p2);
617 else if (p >= p2 && (*p == '.'))
618 o = variable_buffer_output (o, p2, p - p2);
619#ifdef HAVE_DOS_PATHS
620 /* Handle the "d:foobar" case */
621 else if (p2[0] && p2[1] == ':' && is_dir)
622 o = variable_buffer_output (o, p2, 2);
623#endif
624 else if (is_dir)
625#ifdef VMS
626 {
627 extern int vms_report_unix_paths;
628 if (vms_report_unix_paths)
629 o = variable_buffer_output (o, "./", 2);
630 else
631 o = variable_buffer_output (o, "[]", 2);
632 }
633#else
634#ifndef _AMIGA
635 o = variable_buffer_output (o, "./", 2);
636#else
637 ; /* Just a nop... */
638#endif /* AMIGA */
639#endif /* !VMS */
640 else
641 /* The entire name is the basename. */
642 o = variable_buffer_output (o, p2, len);
643
644#ifdef VMS
645 if (vms_comma_separator)
646 o = variable_buffer_output (o, ",", 1);
647 else
648#endif
649 o = variable_buffer_output (o, " ", 1);
650
651 doneany = 1;
652 }
653
654 if (doneany)
655 /* Kill last space. */
656 --o;
657
658 return o;
659}

747/* Set begpp to point to the first non-whitespace character of the string,
748 * and endpp to point to the last non-whitespace character of the string.
749 * If the string is empty or contains nothing but whitespace, endpp will be
750 * begpp-1.
751 */
752char *
753strip_whitespace (const char **begpp, const char **endpp)
754{
755 while (*begpp <= *endpp && ISSPACE (**begpp))
756 (*begpp) ++;
757 while (*endpp >= *begpp && ISSPACE (**endpp))
758 (*endpp) --;
759 return (char *)*begpp;
760}

762static void
763check_numeric (const char *s, const char *msg)
764{
765 const char *end = s + strlen (s) - 1;
766 const char *beg = s;
767 strip_whitespace (&s, &end);
768
769 for (; s <= end; ++s)
770 if (!ISDIGIT (*s)) /* ISDIGIT only evals its arg once: see makeint.h. */
771 break;
772
773 if (s <= end || end - beg < 0)
774 OSS (fatal, *expanding_var, "%s: '%s'", msg, beg);
775}

779static char *
780func_word (char *o, char **argv, const char *funcname UNUSED)
781{
782 const char *end_p;
783 const char *p;
784 int i;
785

943static char *
944func_filter_filterout (char *o, char **argv, const char *funcname)
945{
946 struct a_word *wordhead;
947 struct a_word **wordtail;
948 struct a_word *wp;
949 struct a_pattern *pathead;
950 struct a_pattern **pattail;
951 struct a_pattern *pp;
952
953 struct hash_table a_word_table;
954 int is_filter = funcname[CSTRLEN ("filter")] == '\0';
955 const char *pat_iterator = argv[0];
956 const char *word_iterator = argv[1];
957 int literals = 0;
958 int words = 0;
959 int hashing = 0;
960 char *p;
961 unsigned int len;
962
963 /* Chop ARGV[0] up into patterns to match against the words.
964 We don't need to preserve it because our caller frees all the
965 argument memory anyway. */
966
967 pattail = &pathead;
968 while ((p = find_next_token (&pat_iterator, &len)) != 0)
969 {
970 struct a_pattern *pat = alloca (sizeof (struct a_pattern));
971
972 *pattail = pat;
973 pattail = &pat->next;
974
975 if (*pat_iterator != '\0')
976 ++pat_iterator;
977
978 pat->str = p;
979 p[len] = '\0';
980 pat->percent = find_percent (p);
981 if (pat->percent == 0)
982 literals++;
983
984 /* find_percent() might shorten the string so LEN is wrong. */
985 pat->length = strlen (pat->str);
986 }
987 *pattail = 0;
988
989 /* Chop ARGV[1] up into words to match against the patterns. */
990
991 wordtail = &wordhead;
992 while ((p = find_next_token (&word_iterator, &len)) != 0)
993 {
994 struct a_word *word = alloca (sizeof (struct a_word));
995
996 *wordtail = word;
997 wordtail = &word->next;
998
999 if (*word_iterator != '\0')
1000 ++word_iterator;
1001
1002 p[len] = '\0';
1003 word->str = p;
1004 word->length = len;
1005 word->matched = 0;
1006 word->chain = 0;
1007 words++;
1008 }
1009 *wordtail = 0;
1010
1011 /* Only use a hash table if arg list lengths justifies the cost. */
1012 hashing = (literals >= 2 && (literals * words) >= 10);
1013 if (hashing)
1014 {
1015 hash_init (&a_word_table, words, a_word_hash_1, a_word_hash_2,
1016 a_word_hash_cmp);
1017 for (wp = wordhead; wp != 0; wp = wp->next)
1018 {
1019 struct a_word *owp = hash_insert (&a_word_table, wp);
1020 if (owp)
1021 wp->chain = owp;
1022 }
1023 }
1024
1025 if (words)
1026 {
1027 int doneany = 0;
1028
1029 /* Run each pattern through the words, killing words. */
1030 for (pp = pathead; pp != 0; pp = pp->next)
1031 {
1032 if (pp->percent)
1033 for (wp = wordhead; wp != 0; wp = wp->next)
1034 wp->matched |= pattern_matches (pp->str, pp->percent, wp->str);
1035 else if (hashing)
1036 {
1037 struct a_word a_word_key;
1038 a_word_key.str = pp->str;
1039 a_word_key.length = pp->length;
1040 wp = hash_find_item (&a_word_table, &a_word_key);
1041 while (wp)
1042 {
1043 wp->matched |= 1;
1044 wp = wp->chain;
1045 }
1046 }
1047 else
1048 for (wp = wordhead; wp != 0; wp = wp->next)
1049 wp->matched |= (wp->length == pp->length
1050 && strneq (pp->str, wp->str, wp->length));
1051 }
1052
1053 /* Output the words that matched (or didn't, for filter-out). */
1054 for (wp = wordhead; wp != 0; wp = wp->next)
1055 if (is_filter ? wp->matched : !wp->matched)
1056 {
1057 o = variable_buffer_output (o, wp->str, strlen (wp->str));
1058 o = variable_buffer_output (o, " ", 1);
1059 doneany = 1;
1060 }
1061
1062 if (doneany)
1063 /* Kill the last space. */
1064 --o;
1065 }
1066
1067 if (hashing)
1068 hash_free (&a_word_table, 0);
1069
1070 return o;
1071}

1214/*
1215 $(if condition,true-part[,false-part])
1216
1217 CONDITION is false iff it evaluates to an empty string. White
1218 space before and after condition are stripped before evaluation.
1219
1220 If CONDITION is true, then TRUE-PART is evaluated, otherwise FALSE-PART is
1221 evaluated (if it exists). Because only one of the two PARTs is evaluated,
1222 you can use $(if ...) to create side-effects (with $(shell ...), for
1223 example).
1224*/

1226static char *
1227func_if (char *o, char **argv, const char *funcname UNUSED)
1228{
1229 const char *begp = argv[0];
1230 const char *endp = begp + strlen (argv[0]) - 1;
1231 int result = 0;
1232
1233 /* Find the result of the condition: if we have a value, and it's not
1234 empty, the condition is true. If we don't have a value, or it's the
1235 empty string, then it's false. */
1236
1237 strip_whitespace (&begp, &endp);
1238
1239 if (begp <= endp)
1240 {
1241 char *expansion = expand_argument (begp, endp+1);
1242
1243 result = strlen (expansion);
1244 free (expansion);
1245 }
1246
1247 /* If the result is true (1) we want to eval the first argument, and if
1248 it's false (0) we want to eval the second. If the argument doesn't
1249 exist we do nothing, otherwise expand it and add to the buffer. */
1250
1251 argv += 1 + !result;
1252
1253 if (*argv)
1254 {
1255 char *expansion = expand_argument (*argv, NULL);
1256
1257 o = variable_buffer_output (o, expansion, strlen (expansion));
1258

1260 }

1265/*
1266 $(or condition1[,condition2[,condition3[...]]])
1267
1268 A CONDITION is false iff it evaluates to an empty string. White
1269 space before and after CONDITION are stripped before evaluation.
1270
1271 CONDITION1 is evaluated. If it's true, then this is the result of
1272 expansion. If it's false, CONDITION2 is evaluated, and so on. If none of
1273 the conditions are true, the expansion is the empty string.
1274
1275 Once a CONDITION is true no further conditions are evaluated
1276 (short-circuiting).
1277*/
1278
1279static char *
1280func_or (char *o, char **argv, const char *funcname UNUSED)
1281{
1282 for ( ; *argv ; ++argv)
1283 {
1284 const char *begp = *argv;
1285 const char *endp = begp + strlen (*argv) - 1;
1286 char *expansion;
1287 int result = 0;
1288
1289 /* Find the result of the condition: if it's false keep going. */
1290
1291 strip_whitespace (&begp, &endp);
1292
1293 if (begp > endp)
1294 continue;
1295
1296 expansion = expand_argument (begp, endp+1);
1297 result = strlen (expansion);
1298
1299 /* If the result is false keep going. */
1300 if (!result)
1301 {
1302 free (expansion);
1303 continue;
1304 }
1305
1306 /* It's true! Keep this result and return. */
1307 o = variable_buffer_output (o, expansion, result);
1308 free (expansion);
1309 break;
1310 }
1311
1312 return o;
1313}

1315/*
1316 $(and condition1[,condition2[,condition3[...]]])
1317
1318 A CONDITION is false iff it evaluates to an empty string. White
1319 space before and after CONDITION are stripped before evaluation.
1320
1321 CONDITION1 is evaluated. If it's false, then this is the result of
1322 expansion. If it's true, CONDITION2 is evaluated, and so on. If all of
1323 the conditions are true, the expansion is the result of the last condition.
1324
1325 Once a CONDITION is false no further conditions are evaluated
1326 (short-circuiting).
1327*/
1328
1329static char *
1330func_and (char *o, char **argv, const char *funcname UNUSED)
1331{
1332 char *expansion;
1333
1334 while (1)
1335 {
1336 const char *begp = *argv;
1337 const char *endp = begp + strlen (*argv) - 1;
1338 int result;
1339
1340 /* An empty condition is always false. */
1341 strip_whitespace (&begp, &endp);
1342 if (begp > endp)
1343 return o;
1344
1345 expansion = expand_argument (begp, endp+1);
1346 result = strlen (expansion);
1347
1348 /* If the result is false, stop here: we're done. */
1349 if (!result)
1350 break;
1351
1352 /* Otherwise the result is true. If this is the last one, keep this
1353 result and quit. Otherwise go on to the next one! */
1354
1355 if (*(++argv))
1356 free (expansion);
1357 else
1358 {
1359 o = variable_buffer_output (o, expansion, result);
1360 break;
1361 }
1362 }
1363
1364 free (expansion);
1365
1366 return o;
1367}

2044/* Return the absolute name of file NAME which does not contain any '.',
2045 '..' components nor any repeated path separators ('/'). */
2046
2047static char *
2048abspath (const char *name, char *apath)
2049{
2050 char *dest;
2051 const char *start, *end, *apath_limit;
2052 unsigned long root_len = ROOT_LEN;
2053
2054 if (name[0] == '\0' || apath == NULL)
2055 return NULL;
2056
2057 apath_limit = apath + GET_PATH_MAX;
2058
2059 if (!IS_ABSOLUTE(name))
2060 {
2061 /* It is unlikely we would make it until here but just to make sure. */
2062 if (!starting_directory)
2063 return NULL;
2064
2065 strcpy (apath, starting_directory);
2066
2067#ifdef HAVE_DOS_PATHS
2068 if (STOP_SET (name[0], MAP_DIRSEP))
2069 {
2070 if (STOP_SET (name[1], MAP_DIRSEP))
2071 {
2072 /* A UNC. Don't prepend a drive letter. */
2073 apath[0] = name[0];
2074 apath[1] = name[1];
2075 root_len = 2;
2076 }
2077 /* We have /foo, an absolute file name except for the drive
2078 letter. Assume the missing drive letter is the current
2079 drive, which we can get if we remove from starting_directory
2080 everything past the root directory. */
2081 apath[root_len] = '\0';
2082 }
2083#endif
2084
2085 dest = strchr (apath, '\0');
2086 }
2087 else
2088 {
2089#if defined(__CYGWIN__) && defined(HAVE_DOS_PATHS)
2090 if (STOP_SET (name[0], MAP_DIRSEP))
2091 root_len = 1;
2092#endif
2093 strncpy (apath, name, root_len);
2094 apath[root_len] = '\0';
2095 dest = apath + root_len;
2096 /* Get past the root, since we already copied it. */
2097 name += root_len;
2098#ifdef HAVE_DOS_PATHS
2099 if (! STOP_SET (apath[root_len - 1], MAP_DIRSEP))
2100 {
2101 /* Convert d:foo into d:./foo and increase root_len. */
2102 apath[2] = '.';
2103 apath[3] = '/';
2104 dest++;
2105 root_len++;
2106 /* strncpy above copied one character too many. */
2107 name--;
2108 }
2109 else
2110 apath[root_len - 1] = '/'; /* make sure it's a forward slash */
2111#endif
2112 }
2113
2114 for (start = end = name; *start != '\0'; start = end)
2115 {
2116 unsigned long len;
2117
2118 /* Skip sequence of multiple path-separators. */
2119 while (STOP_SET (*start, MAP_DIRSEP))
2120 ++start;
2121
2122 /* Find end of path component. */

2124 ;
2125

2127
2128 if (len == 0)

2131 /* nothing */;
2132 else if (len == 2 && start[0] == '.' && start[1] == '.')
2133 {

2137 ;
2138 }

2437/* Check for a function invocation in *STRINGP. *STRINGP points at the
2438 opening ( or { and is not null-terminated. If a function invocation
2439 is found, expand it into the buffer at *OP, updating *OP, incrementing
2440 *STRINGP past the reference and returning nonzero. If not, return zero. */
2441
2442int
2443handle_function (char **op, const char **stringp)
2444{
2445 const struct function_table_entry *entry_p;
2446 char openparen = (*stringp)[0];
2447 char closeparen = openparen == '(' ? ')' : '}';
2448 const char *beg;
2449 const char *end;
2450 int count = 0;
2451 char *abeg = NULL;
2452 char **argv, **argvp;
2453 int nargs;
2454
2455 beg = *stringp + 1;
2456
2457 entry_p = lookup_function (beg);
2458
2459 if (!entry_p)
2460 return 0;
2461
2462 /* We found a builtin function. Find the beginning of its arguments (skip
2463 whitespace after the name). */
2464
2465 beg += entry_p->len;
2466 NEXT_TOKEN (beg);
2467
2468 /* Find the end of the function invocation, counting nested use of
2469 whichever kind of parens we use. Since we're looking, count commas
2470 to get a rough estimate of how many arguments we might have. The
2471 count might be high, but it'll never be low. */
2472
2473 for (nargs=1, end=beg; *end != '\0'; ++end)
2474 if (*end == ',')
2475 ++nargs;
2476 else if (*end == openparen)
2477 ++count;
2478 else if (*end == closeparen && --count < 0)
2479 break;
2480
2481 if (count >= 0)
2482 fatal (*expanding_var, strlen (entry_p->name),
2483 _("unterminated call to function '%s': missing '%c'"),
2484 entry_p->name, closeparen);
2485
2486 *stringp = end;
2487
2488 /* Get some memory to store the arg pointers. */
2489 argvp = argv = alloca (sizeof (char *) * (nargs + 2));
2490
2491 /* Chop the string into arguments, then a nul. As soon as we hit
2492 MAXIMUM_ARGS (if it's >0) assume the rest of the string is part of the
2493 last argument.
2494
2495 If we're expanding, store pointers to the expansion of each one. If
2496 not, make a duplicate of the string and point into that, nul-terminating
2497 each argument. */
2498
2499 if (entry_p->expand_args)
2500 {
2501 const char *p;
2502 for (p=beg, nargs=0; p <= end; ++argvp)
2503 {
2504 const char *next;
2505
2506 ++nargs;
2507
2508 if (nargs == entry_p->maximum_args
2509 || (! (next = find_next_argument (openparen, closeparen, p, end))))
2510 next = end;
2511
2512 *argvp = expand_argument (p, next);
2513 p = next + 1;
2514 }
2515 }
2516 else
2517 {
2518 int len = end - beg;
2519 char *p, *aend;
2520
2521 abeg = xmalloc (len+1);
2522 memcpy (abeg, beg, len);
2523 abeg[len] = '\0';
2524 aend = abeg + len;
2525
2526 for (p=abeg, nargs=0; p <= aend; ++argvp)
2527 {
2528 char *next;
2529
2530 ++nargs;
2531
2532 if (nargs == entry_p->maximum_args
2533 || (! (next = find_next_argument (openparen, closeparen, p, aend))))
2534 next = aend;
2535
2536 *argvp = p;
2537 *next = '\0';
2538 p = next + 1;
2539 }
2540 }
2541 *argvp = NULL;
2542
2543 /* Finally! Run the function... */
2544 *op = expand_builtin_function (*op, nargs, argv, entry_p);
2545
2546 /* Free memory. */
2547 if (entry_p->expand_args)
2548 for (argvp=argv; *argvp != 0; ++argvp)
2549 free (*argvp);
2550 else
2551 free (abeg);
2552
2553 return 1;
2554}

2557/* User-defined functions. Expand the first argument as either a builtin
2558 function or a make variable, in the context of the rest of the arguments
2559 assigned to $1, $2, ... $N. $0 is the name of the function. */
2560
2561static char *
2562func_call (char *o, char **argv, const char *funcname UNUSED)
2563{
2564 static int max_args = 0;
2565 char *fname;
2566 char *body;
2567 int flen;
2568 int i;
2569 int saved_args;
2570 const struct function_table_entry *entry_p;
2571 struct variable *v;
2572
2573 /* Clean up the name of the variable to be invoked. */
2574 fname = next_token (argv[0]);
2575 end_of_token (fname)[0] = '\0';
2576
2577 /* Calling nothing is a no-op */
2578 if (*fname == '\0')
2579 return o;
2580
2581 /* Are we invoking a builtin function? */
2582
2583 entry_p = lookup_function (fname);
2584 if (entry_p)
2585 {
2586 /* How many arguments do we have? */
2587 for (i=0; argv[i+1]; ++i)
2588 ;
2589 return expand_builtin_function (o, i, argv+1, entry_p);
2590 }
2591
2592 /* Not a builtin, so the first argument is the name of a variable to be
2593 expanded and interpreted as a function. Find it. */

2595
2596 v = lookup_variable (fname, flen);
2597
2598 if (v == 0)
2599 warn_undefined (fname, flen);
2600
2601 if (v == 0 || *v->value == '\0')
2602 return o;
2603
2604 body = alloca (flen + 4);
2605 body[0] = '$';
2606 body[1] = '(';
2607 memcpy (body + 2, fname, flen);
2608 body[flen+2] = ')';
2609 body[flen+3] = '\0';
2610
2611 /* Set up arguments $(1) .. $(N). $(0) is the function name. */
2612
2613 push_new_variable_scope ();
2614
2615 for (i=0; *argv; ++i, ++argv)
2616 {
2617 char num[11];
2618
2619 sprintf (num, "%d", i);
2620 define_variable (num, strlen (num), *argv, o_automatic, 0);
2621 }
2622
2623 /* If the number of arguments we have is < max_args, it means we're inside
2624 a recursive invocation of $(call ...). Fill in the remaining arguments
2625 in the new scope with the empty value, to hide them from this
2626 invocation. */
2627
2628 for (; i < max_args; ++i)
2629 {
2630 char num[11];
2631
2632 sprintf (num, "%d", i);
2633 define_variable (num, strlen (num), "", o_automatic, 0);
2634 }
2635
2636 /* Expand the body in the context of the arguments, adding the result to
2637 the variable buffer. */
2638
2639 v->exp_count = EXP_COUNT_MAX;
2640
2641 saved_args = max_args;
2642 max_args = i;
2643 o = variable_expand_string (o, body, flen+3);
2644 max_args = saved_args;
2645
2646 v->exp_count = 0;
2647
2648 pop_variable_scope ();
2649
2650 return o + strlen (o);
2651}

2654define_new_function (const gmk_floc *flocp, const char *name,
2655 unsigned int min, unsigned int max, unsigned int flags,
2656 gmk_func_ptr func)
2657{
2658 const char *e = name;
2659 struct function_table_entry *ent;
2660 size_t len;
2661
2662 while (STOP_SET (*e, MAP_USERFUNC))
2663 e++;
2664 len = e - name;
2665
2666 if (len == 0)
2667 O (fatal, flocp, _("Empty function name"));
2668 if (*name == '.' || *e != '\0')
2669 OS (fatal, flocp, _("Invalid function name: %s"), name);
2670 if (len > 255)
2671 OS (fatal, flocp, _("Function name too long: %s"), name);
2672 if (min > 255)
2673 ONS (fatal, flocp,
2674 _("Invalid minimum argument count (%u) for function %s"), min, name);
2675 if (max > 255 || (max && max < min))
2676 ONS (fatal, flocp,
2677 _("Invalid maximum argument count (%u) for function %s"), max, name);
2678
2679 ent = xmalloc (sizeof (struct function_table_entry));
2680 ent->name = name;
2681 ent->len = len;
2682 ent->minimum_args = min;
2683 ent->maximum_args = max;
2684 ent->expand_args = ANY_SET(flags, GMK_FUNC_NOEXPAND) ? 0 : 1;
2685 ent->alloc_fn = 1;
2686 ent->fptr.alloc_func_ptr = func;
