32/*!
33 Replace all carriage returns and linefeeds with spaces.

40extern void hash_free_function_table (void);

42/*! Set begpp to point to the first non-whitespace character of the string,
43 and endpp to point to the last non-whitespace character of the string.
44 If the string is empty or contains nothing but whitespace, endpp will be
45 begpp-1.
46 */
47extern char *strip_whitespace (const char **begpp, const char **endpp);

49/*!
50 $(eval *makefile-string*)

52 Always resolves to the empty string.
53
54 Treat the arguments as a segment of makefile, and parse them.
55*/
56extern char *func_eval (char *o, char **argv, const char *funcname UNUSED);
