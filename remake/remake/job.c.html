216/* This function creates a temporary file name with an extension specified
217 * by the unixy arg.
218 * Return an xmalloc'ed string of a newly created temp file and its
219 * file descriptor, or die. */
220static char *
221create_batch_file (char const *base, int unixy, int *fd)
222{
223 const char *const ext = unixy ? "sh" : "bat";
224 const char *error_string = NULL;
225 char temp_path[MAXPATHLEN]; /* need to know its length */
226 unsigned path_size = GetTempPath (sizeof temp_path, temp_path);
227 int path_is_dot = 0;
228 /* The following variable is static so we won't try to reuse a name
229 that was generated a little while ago, because that file might
230 not be on disk yet, since we use FILE_ATTRIBUTE_TEMPORARY below,
231 which tells the OS it doesn't need to flush the cache to disk.
232 If the file is not yet on disk, we might think the name is
233 available, while it really isn't. This happens in parallel
234 builds, where Make doesn't wait for one job to finish before it
235 launches the next one. */
236 static unsigned uniq = 0;
237 static int second_loop = 0;
238 const unsigned sizemax = strlen (base) + strlen (ext) + 10;
239
240 if (path_size == 0)
241 {
242 path_size = GetCurrentDirectory (sizeof temp_path, temp_path);
243 path_is_dot = 1;
244 }
245
246 ++uniq;
247 if (uniq >= 0x10000 && !second_loop)
248 {
249 /* If we already had 64K batch files in this
250 process, make a second loop through the numbers,
251 looking for free slots, i.e. files that were
252 deleted in the meantime. */
253 second_loop = 1;
254 uniq = 1;
255 }
256 while (path_size > 0 &&
257 path_size + sizemax < sizeof temp_path &&
258 !(uniq >= 0x10000 && second_loop))
259 {
260 unsigned size = sprintf (temp_path + path_size,
261 "%s%s-%x.%s",
262 temp_path[path_size - 1] == '\\' ? "" : "\\",
263 base, uniq, ext);
264 HANDLE h = CreateFile (temp_path, /* file name */
265 GENERIC_READ | GENERIC_WRITE, /* desired access */
266 0, /* no share mode */
267 NULL, /* default security attributes */
268 CREATE_NEW, /* creation disposition */
269 FILE_ATTRIBUTE_NORMAL | /* flags and attributes */
270 FILE_ATTRIBUTE_TEMPORARY, /* we'll delete it */
271 NULL); /* no template file */
272
273 if (h == INVALID_HANDLE_VALUE)
274 {
275 const DWORD er = GetLastError ();
276
277 if (er == ERROR_FILE_EXISTS || er == ERROR_ALREADY_EXISTS)
278 {
279 ++uniq;
280 if (uniq == 0x10000 && !second_loop)
281 {
282 second_loop = 1;
283 uniq = 1;
284 }
285 }
286
287 /* the temporary path is not guaranteed to exist */
288 else if (path_is_dot == 0)
289 {
290 path_size = GetCurrentDirectory (sizeof temp_path, temp_path);
291 path_is_dot = 1;
292 }
293
294 else
295 {
296 error_string = map_windows32_error_to_string (er);
297 break;
298 }
299 }
300 else
301 {
302 const unsigned final_size = path_size + size + 1;
303 char *const path = xmalloc (final_size);
304 memcpy (path, temp_path, final_size);
305 *fd = _open_osfhandle ((intptr_t)h, 0);
306 if (unixy)
307 {
308 char *p;
309 int ch;
310 for (p = path; (ch = *p) != 0; ++p)
311 if (ch == '\\')
312 *p = '/';
313 }
314 return path; /* good return */
315 }
316 }
317
318 *fd = -1;
319 if (error_string == NULL)
320 error_string = _("Cannot create a temporary file\n");
321 O (fatal, NILF, error_string);
322
323 /* not reached */
324 return NULL;
325}

328/* determines whether path looks to be a Bourne-like shell. */
329int
330is_bourne_compatible_shell (const char *path)
331{
332 /* List of known POSIX (or POSIX-ish) shells. */
333 static const char *unix_shells[] = {
334 "sh",
335 "bash",
336 "ksh",
337 "rksh",
338 "zsh",
339 "ash",
340 "dash",
341 NULL
342 };
343 const char **s;
344
345 /* find the rightmost '/' or '\\' */
346 const char *name = strrchr (path, '/');
347 char *p = strrchr (path, '\\');
348
349 if (name && p) /* take the max */
350 name = (name > p) ? name : p;
351 else if (p) /* name must be 0 */
352 name = p;
353 else if (!name) /* name and p must be 0 */
354 name = path;
355
356 if (*name == '/' || *name == '\\')
357 ++name;
358
359 /* this should be able to deal with extensions on Windows-like systems */
360 for (s = unix_shells; *s != NULL; ++s)
361 {
362#if defined(WINDOWS32) || defined(__MSDOS__)
363 unsigned int len = strlen (*s);
364 if ((strlen (name) >= len && STOP_SET (name[len], MAP_DOT|MAP_NUL))
365 && strncasecmp (name, *s, len) == 0)
366#else
367 if (strcmp (name, *s) == 0)
368#endif
369 return 1; /* a known unix-style shell */
370 }
371
372 /* if not on the list, assume it's not a Bourne-like shell */
373 return 0;
374}

377/* Write an error message describing the exit status given in
378 EXIT_CODE, EXIT_SIG, and COREDUMP, for the target TARGET_NAME.
379 Append "(ignored)" if IGNORED is nonzero. */
380
381static void
382child_error (child_t *p_child, target_stack_node_t *p_call_stack,
383 int exit_code, int exit_sig, int coredump, int ignored)
384{
385 const char *pre = "*** ";
386 const char *post = "";
387 const char *dump = "";
388 const struct file *f = p_child->file;
389
390 if (ignored && silent_flag)
391 return;
392
393 if (exit_sig && coredump)
394 dump = _(" (core dumped)");
395
396 if (ignored)
397 {
398 pre = "";
399 post = _(" (ignored)");
400 }
401
402 OUTPUT_SET (&p_child->output);
403
404 show_goal_error ();
405
406 if (exit_sig == 0)
407 err_with_stack(p_call_stack,
408 _("%s[%s] Error %d%s"),
409 pre, f->name, exit_code, post);
410 else
411 err_with_stack(p_call_stack, "%s[%s] %s%s%s",
412 pre, f->name, strsignal (exit_sig), dump, post);
413
414 OUTPUT_UNSET ();
415
416 /* If have enabled debugging but haven't entered the debugger above
417 because we haven't set to debug on error, enter the debugger now.
418 FIXME: Add be another variable/option to control entry here as
419 well?
420 */
421 if (! (debugger_on_error & DEBUGGER_ON_ERROR) && debugger_enabled )
422 enter_debugger(p_call_stack, p_child->file, exit_code, DEBUG_ERROR_HIT);
423}

426/* Handle a dead child. This handler may or may not ever be installed.
427
428 If we're using the jobserver feature without pselect(), we need it.
429 First, installing it ensures the read will interrupt on SIGCHLD. Second,
430 we close the dup'd read FD to ensure we don't enter another blocking read
431 without reaping all the dead children. In this case we don't need the
432 dead_children count.
433
434 If we don't have either waitpid or wait3, then make is unreliable, but we
435 use the dead_children count to reap children as best we can. */
436
437static unsigned int dead_children = 0;
438
439RETSIGTYPE
440child_handler (int sig UNUSED)
441{
442 ++dead_children;
443
444 jobserver_signal ();
445}

449/* Reap all dead children, storing the returned status and the new command
450 state ('cs_finished') in the 'file' member of the 'struct child' for the
451 dead child, and removing the child from the chain. In addition, if BLOCK
452 nonzero, we block in this function until we've reaped at least one
453 complete child, waiting for it to die if necessary. If ERR is nonzero,
454 print an error message first. */
455
456void
457reap_children (int block, int err, target_stack_node_t *p_call_stack)
458{
459#ifndef WINDOWS32
460 WAIT_T status;
461#endif
462 /* Initially, assume we have some. */
463 int reap_more = 1;
464
465#ifdef WAIT_NOHANG
466# define REAP_MORE reap_more
467#else
468# define REAP_MORE dead_children
469#endif
470
471 /* As long as:
472
473 We have at least one child outstanding OR a shell function in progress,
474 AND
475 We're blocking for a complete child OR there are more children to reap
476
477 we'll keep reaping children. */
478
479 while ((children != 0 || shell_function_pid != 0)
480 && (block || REAP_MORE))
481 {
482 unsigned int remote = 0;
483 pid_t pid;
484 int exit_code, exit_sig, coredump;
485 struct child *lastc, *c;
486 int child_failed;
487 int any_remote, any_local;
488 int dontcare;
489
490 if (err && block)
491 {
492 static int printed = 0;
493
494 /* We might block for a while, so let the user know why.
495 Only print this message once no matter how many jobs are left. */
496 fflush (stdout);
497 if (!printed)
498 O (error, NILF, _("*** Waiting for unfinished jobs...."));
499 printed = 1;
500 }
501
502 /* We have one less dead child to reap. As noted in
503 child_handler() above, this count is completely unimportant for
504 all modern, POSIX-y systems that support wait3() or waitpid().
505 The rest of this comment below applies only to early, broken
506 pre-POSIX systems. We keep the count only because... it's there...
507
508 The test and decrement are not atomic; if it is compiled into:
509 register = dead_children - 1;
510 dead_children = register;
511 a SIGCHLD could come between the two instructions.
512 child_handler increments dead_children.
513 The second instruction here would lose that increment. But the
514 only effect of dead_children being wrong is that we might wait
515 longer than necessary to reap a child, and lose some parallelism;
516 and we might print the "Waiting for unfinished jobs" message above
517 when not necessary. */
518
519 if (dead_children > 0)
520 --dead_children;
521
522 any_remote = 0;
523 any_local = shell_function_pid != 0;
524 for (c = children; c != 0; c = c->next)
525 {
526 any_remote |= c->remote;
527 any_local |= ! c->remote;
528 DB (DB_JOBS, (_("Live child %p (%s) PID %s %s\n"),
529 c, c->file->name, pid2str (c->pid),
530 c->remote ? _(" (remote)") : ""));
531 }
532
533 /* First, check for remote children. */
534 if (any_remote)
535 pid = remote_status (&exit_code, &exit_sig, &coredump, 0);
536 else
537 pid = 0;
538
539 if (pid > 0)
540 /* We got a remote child. */
541 remote = 1;
542 else if (pid < 0)
543 {
544 /* A remote status command failed miserably. Punt. */
545 remote_status_lose:
546 pfatal_with_name ("remote_status");
547 }
548 else
549 {
550 /* No remote children. Check for local children. */
551#if !defined(__MSDOS__) && !defined(WINDOWS32)
552 if (any_local)
553 {
554#ifdef WAIT_NOHANG
555 if (!block)
556 pid = WAIT_NOHANG (&status);
557 else
558#endif
559 EINTRLOOP (pid, wait (&status));
560 }
561 else
562 pid = 0;
563
564 if (pid < 0)
565 {
566 /* The wait*() failed miserably. Punt. */
567 pfatal_with_name ("wait");
568 }
569 else if (pid > 0)
570 {
571 /* We got a child exit; chop the status word up. */
572 exit_code = WEXITSTATUS (status);
573 exit_sig = WIFSIGNALED (status) ? WTERMSIG (status) : 0;
574 coredump = WCOREDUMP (status);
575
576 /* If we have started jobs in this second, remove one. */
577 if (job_counter)
578 --job_counter;
579 }
580 else
581 {
582 /* No local children are dead. */
583 reap_more = 0;
584
585 if (!block || !any_remote)
586 break;
587
588 /* Now try a blocking wait for a remote child. */
589 pid = remote_status (&exit_code, &exit_sig, &coredump, 1);
590 if (pid < 0)
591 goto remote_status_lose;
592 else if (pid == 0)
593 /* No remote children either. Finally give up. */
594 break;
595
596 /* We got a remote child. */
597 remote = 1;
598 }
599#endif /* !__MSDOS__, !Amiga, !WINDOWS32. */
600
601#ifdef __MSDOS__
602 /* Life is very different on MSDOS. */
603 pid = dos_pid - 1;
604 status = dos_status;
605 exit_code = WEXITSTATUS (status);
606 if (exit_code == 0xff)
607 exit_code = -1;
608 exit_sig = WIFSIGNALED (status) ? WTERMSIG (status) : 0;
609 coredump = 0;
610#endif /* __MSDOS__ */
611#ifdef WINDOWS32
612 {
613 HANDLE hPID;
614 HANDLE hcTID, hcPID;
615 DWORD dwWaitStatus = 0;
616 exit_code = 0;
617 exit_sig = 0;
618 coredump = 0;
619
620 /* Record the thread ID of the main process, so that we
621 could suspend it in the signal handler. */
622 if (!main_thread)
623 {
624 hcTID = GetCurrentThread ();
625 hcPID = GetCurrentProcess ();
626 if (!DuplicateHandle (hcPID, hcTID, hcPID, &main_thread, 0,
627 FALSE, DUPLICATE_SAME_ACCESS))
628 {
629 DWORD e = GetLastError ();
630 fprintf (stderr,
631 "Determine main thread ID (Error %ld: %s)\n",
632 e, map_windows32_error_to_string (e));
633 }
634 else
635 DB (DB_VERBOSE, ("Main thread handle = %p\n", main_thread));
636 }
637
638 /* wait for anything to finish */
639 hPID = process_wait_for_any (block, &dwWaitStatus);
640 if (hPID)
641 {
642 /* was an error found on this process? */
643 int werr = process_last_err (hPID);
644
645 /* get exit data */
646 exit_code = process_exit_code (hPID);
647
648 if (werr)
649 fprintf (stderr, "make (e=%d): %s", exit_code,
650 map_windows32_error_to_string (exit_code));
651
652 /* signal */
653 exit_sig = process_signal (hPID);
654
655 /* cleanup process */
656 process_cleanup (hPID);
657
658 coredump = 0;
659 }
660 else if (dwWaitStatus == WAIT_FAILED)
661 {
662 /* The WaitForMultipleObjects() failed miserably. Punt. */
663 pfatal_with_name ("WaitForMultipleObjects");
664 }
665 else if (dwWaitStatus == WAIT_TIMEOUT)
666 {
667 /* No child processes are finished. Give up waiting. */
668 reap_more = 0;
669 break;
670 }
671
672 pid = (pid_t) hPID;
673 }
674#endif /* WINDOWS32 */
675 }
676
677 /* Check if this is the child of the 'shell' function. */
678 if (!remote && pid == shell_function_pid)
679 {
680 shell_completed (exit_code, exit_sig);
681 break;
682 }
683
684 /* Search for a child matching the deceased one. */
685 lastc = 0;
686 for (c = children; c != 0; lastc = c, c = c->next)
687 if (c->pid == pid && c->remote == remote)
688 break;
689
690 if (c == 0)
691 /* An unknown child died.
692 Ignore it; it was inherited from our invoker. */
693 continue;
694
695 /* Determine the failure status: 0 for success, 1 for updating target in
696 question mode, 2 for anything else. */
697 if (exit_sig == 0 && exit_code == 0)
698 child_failed = MAKE_SUCCESS;
699 else if (exit_sig == 0 && exit_code == 1 && question_flag && c->recursive)
700 child_failed = MAKE_TROUBLE;
701 else
702 child_failed = MAKE_FAILURE;
703
704 DB (DB_JOBS, (child_failed
705 ? _("Reaping losing child %p PID %s %s\n")
706 : _("Reaping winning child %p PID %s %s\n"),
707 c, pid2str (c->pid), c->remote ? _(" (remote)") : ""));
708
709 if (c->sh_batch_file)
710 {
711 int rm_status;
712
713 DB (DB_JOBS, (_("Cleaning up temp batch file %s\n"),
714 c->sh_batch_file));
715
716 errno = 0;
717 rm_status = remove (c->sh_batch_file);
718 if (rm_status)
719 DB (DB_JOBS, (_("Cleaning up temp batch file %s failed (%d)\n"),
720 c->sh_batch_file, errno));
721
722 /* all done with memory */
723 free (c->sh_batch_file);
724 c->sh_batch_file = NULL;
725 }
726
727 /* If this child had the good stdin, say it is now free. */
728 if (c->good_stdin)
729 good_stdin_used = 0;
730
731 dontcare = c->dontcare;
732
733 if (exit_code == DEBUGGER_QUIT_RC && debugger_enabled) {
734 if (job_slots_used > 0) --job_slots_used;
735 c->file->update_status = 0;
736 free_child (c);
737 in_debugger = DEBUGGER_QUIT_RC;
738 die(DEBUGGER_QUIT_RC);
739 }
740
741 if (child_failed && !c->noerror && !ignore_errors_flag)
742 {
743 /* The commands failed. Write an error message,
744 delete non-precious targets, and abort. */
745 static int delete_on_error = -1;
746
747 if (!dontcare && child_failed == MAKE_FAILURE)
748 child_error (c, p_call_stack, exit_code, exit_sig, coredump, 0);
749
750 c->file->update_status = child_failed == MAKE_FAILURE ? us_failed : us_question;
751 if (delete_on_error == -1)
752 {
753 struct file *f = lookup_file (".DELETE_ON_ERROR");
754 delete_on_error = f != 0 && f->is_target;
755 }
756 if (exit_sig != 0 || delete_on_error)
757 delete_child_targets (c);
758 }
759 else
760 {
761 if (child_failed)
762 {
763 /* The commands failed, but we don't care. */
764 child_error (c, p_call_stack, exit_code, exit_sig, coredump, 1);
765 child_failed = 0;
766 }
767
768 /* If there are more commands to run, try to start them. */
769 if (job_next_command (c))
770 {
771 if (handling_fatal_signal)
772 {
773 /* Never start new commands while we are dying.
774 Since there are more commands that wanted to be run,
775 the target was not completely remade. So we treat
776 this as if a command had failed. */
777 c->file->update_status = us_failed;
778 }
779 else
780 {
781#ifndef NO_OUTPUT_SYNC
782 /* If we're sync'ing per line, write the previous line's
783 output before starting the next one. */
784 if (output_sync == OUTPUT_SYNC_LINE)
785 output_dump (&c->output);
786#endif
787 /* Check again whether to start remotely.
788 Whether or not we want to changes over time.
789 Also, start_remote_job may need state set up
790 by start_remote_job_p. */
791 c->remote = start_remote_job_p (0);
792 start_job_command (c, p_call_stack);
793 /* Fatal signals are left blocked in case we were
794 about to put that child on the chain. But it is
795 already there, so it is safe for a fatal signal to
796 arrive now; it will clean up this child's targets. */
797 unblock_sigs ();
798 if (c->file->command_state == cs_running)
799 /* We successfully started the new command.
800 Loop to reap more children. */
801 continue;
802 }
803
804 if (c->file->update_status != us_success)
805 /* We failed to start the commands. */
806 delete_child_targets (c);
807 }
808 else
809 /* There are no more commands. We got through them all
810 without an unignored error. Now the target has been
811 successfully updated. */
812 c->file->update_status = us_success;
813 }
814
815 /* When we get here, all the commands for c->file are finished. */
816
817#ifndef NO_OUTPUT_SYNC
818 /* Synchronize any remaining parallel output. */
819 output_dump (&c->output);
820#endif
821
822 /* At this point c->file->update_status is success or failed. But
823 c->file->command_state is still cs_running if all the commands
824 ran; notice_finish_file looks for cs_running to tell it that
825 it's interesting to check the file's modtime again now. */
826
827 if (! handling_fatal_signal)
828 /* Notice if the target of the commands has been changed.
829 This also propagates its values for command_state and
830 update_status to its also_make files. */
831 notice_finished_file (c->file);
832
833 DB (DB_JOBS, (_("Removing child %p PID %s%s from chain.\n"),
834 c, pid2str (c->pid), c->remote ? _(" (remote)") : ""));
835
836 /* Block fatal signals while frobnicating the list, so that
837 children and job_slots_used are always consistent. Otherwise
838 a fatal signal arriving after the child is off the chain and
839 before job_slots_used is decremented would believe a child was
840 live and call reap_children again. */
841 block_sigs ();
842
843 /* There is now another slot open. */
844 if (job_slots_used > 0)
845 --job_slots_used;
846
847 /* Remove the child from the chain and free it. */
848 if (lastc == 0)
849 children = c->next;
850 else
851 lastc->next = c->next;
852
853 {
854 /* Save file info in case we need to use it in the debugger
855 */
856 file_t file;
857
858 memcpy(&file, c->file, sizeof(file_t));
859
860 free_child (c);
861
862 unblock_sigs ();
863
864 /* Debugger "quit" takes precedence over --ignore-errors
865 --keep-going, etc.
866 */
867 if (exit_code == DEBUGGER_QUIT_RC && debugger_enabled) {
868 if (job_slots_used > 0) --job_slots_used;
869 in_debugger = DEBUGGER_QUIT_RC;
870 die(DEBUGGER_QUIT_RC);
871 }
872
873 /* If the job failed, and the -k flag was not given, die,
874 unless we are already in the process of dying. */
875 if (!err && child_failed && !dontcare && !keep_going_flag &&
876 /* fatal_error_signal will die with the right signal. */
877 !handling_fatal_signal) {
878 if ( (debugger_on_error & DEBUGGER_ON_FATAL)
879 || i_debugger_stepping || i_debugger_nexting )
880 enter_debugger(p_call_stack, &file, 2, DEBUG_ERROR_HIT);
881 die (child_failed);
882 }
883 }
884
885 /* Only block for one child. */
886 block = 0;
887 }
888
889 return;
890}

929 while (*ep != 0)

932 }
933
934 free (child);
935}

964/* Start a job to run the commands specified in CHILD.
965 CHILD is updated to reflect the commands and ID of the child process.
966
967 NOTE: On return fatal signals are blocked! The caller is responsible
968 for calling 'unblock_sigs', once the new child is safely on the chain so
969 it can be cleaned up in the event of a fatal signal. */
970
971static void
972start_job_command (child_t *child,
973 target_stack_node_t *p_call_stack)
974{
975 int flags;
976 char *p;
977#ifdef VMS
978 char *argv;
979#else
980 char **argv;
981#endif
982
983 /* If we have a completely empty commandset, stop now. */
984 if (!child->command_ptr)
985 goto next_command;
986
987 /* Combine the flags parsed for the line itself with
988 the flags specified globally for this target. */
989 flags = (child->file->command_flags
990 | child->file->cmds->lines_flags[child->command_line - 1]);
991
992 p = child->command_ptr;
993 child->noerror = ((flags & COMMANDS_NOERROR) != 0);
994
995 while (*p != '\0')
996 {
997 if (*p == '@')
998 flags |= COMMANDS_SILENT;
999 else if (*p == '+')
1000 flags |= COMMANDS_RECURSE;
1001 else if (*p == '-')
1002 child->noerror = 1;
1003 /* Don't skip newlines. */
1004 else if (!ISBLANK (*p))
1005 break;
1006 ++p;
1007 }
1008
1009 child->recursive = ((flags & COMMANDS_RECURSE) != 0);
1010
1011 /* Update the file's command flags with any new ones we found. We only
1012 keep the COMMANDS_RECURSE setting. Even this isn't 100% correct; we are
1013 now marking more commands recursive than should be in the case of
1014 multiline define/endef scripts where only one line is marked "+". In
1015 order to really fix this, we'll have to keep a lines_flags for every
1016 actual line, after expansion. */
1017 child->file->cmds->lines_flags[child->command_line - 1] |= flags & COMMANDS_RECURSE;
1018
1019 /* POSIX requires that a recipe prefix after a backslash-newline should
1020 be ignored. Remove it now so the output is correct. */
1021 {
1022 char prefix = child->file->cmds->recipe_prefix;
1023 char *p1, *p2;
1024 p1 = p2 = p;
1025 while (*p1 != '\0')
1026 {
1027 *(p2++) = *p1;
1028 if (p1[0] == '\n' && p1[1] == prefix)
1029 ++p1;
1030 ++p1;
1031 }
1032 *p2 = *p1;
1033 }
1034
1035 /* Figure out an argument list from this command line. */
1036 {
1037 char *end = 0;
1038#ifdef VMS
1039 /* Skip any leading whitespace */
1040 while (*p)
1041 {
1042 if (!ISSPACE (*p))
1043 {
1044 if (*p != '\\')
1045 break;
1046 if ((p[1] != '\n') && (p[1] != 'n') && (p[1] != 't'))
1047 break;
1048 }
1049 p++;
1050 }
1051
1052 argv = p;
1053 /* Although construct_command_argv contains some code for VMS, it was/is
1054 not called/used. Please note, for VMS argv is a string (not an array
1055 of strings) which contains the complete command line, which for
1056 multi-line variables still includes the newlines. So detect newlines
1057 and set 'end' (which is used for child->command_ptr) instead of
1058 (re-)writing construct_command_argv */
1059 if (!one_shell)
1060 {
1061 char *s = p;
1062 int instring = 0;
1063 while (*s)
1064 {
1065 if (*s == '"')
1066 instring = !instring;
1067 else if (*s == '\\' && !instring && *(s+1) != 0)
1068 s++;
1069 else if (*s == '\n' && !instring)
1070 {
1071 end = s;
1072 break;
1073 }
1074 ++s;
1075 }
1076 }
1077#else
1078 argv = construct_command_argv (p, &end, child->file,
1079 child->file->cmds->lines_flags[child->command_line - 1],
1080 &child->sh_batch_file);
1081#endif
1082 if (end == NULL)
1083 child->command_ptr = NULL;
1084 else
1085 {
1086 *end++ = '\0';
1087 child->command_ptr = end;
1088 }
1089 }
1090
1091 /* If -q was given, say that updating 'failed' if there was any text on the
1092 command line, or 'succeeded' otherwise. The exit status of 1 tells the
1093 user that -q is saying 'something to do'; the exit status for a random
1094 error is 2. */
1095 if (argv != 0 && question_flag && !(flags & COMMANDS_RECURSE))
1096 {
1097#ifndef VMS
1098 free (argv[0]);
1099 free (argv);
1100#endif
1101#ifdef VMS
1102 /* On VMS, argv[0] can be a null string here */
1103 if (argv[0] != 0)
1104 {
1105#endif
1106 child->file->update_status = us_question;
1107 notice_finished_file (child->file);
1108 return;
1109#ifdef VMS
1110 }
1111#endif
1112 }
1113
1114 if (touch_flag && !(flags & COMMANDS_RECURSE))
1115 {
1116 /* Go on to the next command. It might be the recursive one.
1117 We construct ARGV only to find the end of the command line. */
1118#ifndef VMS
1119 if (argv)
1120 {
1121 free (argv[0]);
1122 free (argv);
1123 }
1124#endif
1125 argv = 0;
1126 }
1127
1128 if (argv == 0)
1129 {
1130 next_command:
1131#ifdef __MSDOS__
1132 execute_by_shell = 0; /* in case construct_command_argv sets it */
1133#endif
1134 /* This line has no commands. Go to the next. */
1135 if (job_next_command (child))
1136 start_job_command (child, p_call_stack);
1137 else
1138 {
1139 /* No more commands. Make sure we're "running"; we might not be if
1140 (e.g.) all commands were skipped due to -n. */
1141 set_command_state (child->file, cs_running);
1142 child->file->update_status = us_success;
1143 notice_finished_file (child->file);
1144 }
1145
1146 OUTPUT_UNSET();
1147 return;
1148 }
1149
1150 /* Are we going to synchronize this command's output? Do so if either we're
1151 in SYNC_RECURSE mode or this command is not recursive. We'll also check
1152 output_sync separately below in case it changes due to error. */
1153 child->output.syncout = output_sync && (output_sync == OUTPUT_SYNC_RECURSE
1154 || !(flags & COMMANDS_RECURSE));
1155
1156 OUTPUT_SET (&child->output);
1157
1158#ifndef NO_OUTPUT_SYNC
1159 if (! child->output.syncout)
1160 /* We don't want to sync this command: to avoid misordered
1161 output ensure any already-synced content is written. */
1162 output_dump (&child->output);
1163#endif
1164
1165 /* Print the command if appropriate. */
1166 {
1167 bool print_it =
1168 (just_print_flag
1169 || (!(flags & COMMANDS_SILENT) && !silent_flag)
1170 || (db_level & DB_SHELL));
1171
1172 if (print_it) {
1173 if (db_level & DB_SHELL) {
1174 char pid_str[20] = ">>";
1175 if (job_slots != 1)
1176 snprintf(pid_str, sizeof(pid_str), "%d", child->pid);
1177 OS (message, 0, "##>>>>>>>>>>>>>>>>>>>>>>>>>>%s>>>>>>>>>>>>>>>>>>>>>>>>>>>>",
1178 pid_str);
1179 OS (message, 0, "%s", p);
1180 OS (message, 0, "##<<<<<<<<<<<<<<<<<<<<<<<<<<%s<<<<<<<<<<<<<<<<<<<<<<<<<<<<",
1181 job_slots != 1 ? pid_str : "<<");
1182 } else
1183 OS (message, 0, "%s", p);
1184 }
1185 }
1186
1187 /* Tell update_goal_chain that a command has been started on behalf of
1188 this target. It is important that this happens here and not in
1189 reap_children (where we used to do it), because reap_children might be
1190 reaping children from a different target. We want this increment to
1191 guaranteedly indicate that a command was started for the dependency
1192 chain (i.e., update_file recursion chain) we are processing. */
1193
1194 ++commands_started;
1195
1196 /* Optimize an empty command. People use this for timestamp rules,
1197 so avoid forking a useless shell. Do this after we increment
1198 commands_started so make still treats this special case as if it
1199 performed some action (makes a difference as to what messages are
1200 printed, etc. */
1201
1202 if (
1203#if defined __MSDOS__
1204 unixy_shell /* the test is complicated and we already did it */
1205#else
1206 (argv[0] && is_bourne_compatible_shell (argv[0]))
1207#endif
1208 && (argv[1] && argv[1][0] == '-'
1209 &&
1210 ((argv[1][1] == 'c' && argv[1][2] == '\0')
1211 ||
1212 (argv[1][1] == 'e' && argv[1][2] == 'c' && argv[1][3] == '\0')))
1213 && (argv[2] && argv[2][0] == ':' && argv[2][1] == '\0')
1214 && argv[3] == NULL)
1215 {
1216 free (argv[0]);
1217 free (argv);
1218 goto next_command;
1219 }
1220
1221 /* If -n was given, recurse to get the next line in the sequence. */
1222
1223 if (just_print_flag && !(flags & COMMANDS_RECURSE))
1224 {
1225 free (argv[0]);
1226 free (argv);
1227 goto next_command;
1228 }
1229
1230 /* We're sure we're going to invoke a command: set up the output. */
1231 output_start ();
1232
1233 p_stack_top = p_call_stack;
1234 if (i_debugger_stepping)
1235 enter_debugger(p_call_stack, child->file, 0, DEBUG_STEP_COMMAND);
1236
1237 /* Flush the output streams so they won't have things written twice. */
1238
1239 fflush (stdout);
1240 fflush (stderr);
1241
1242 /* Decide whether to give this child the 'good' standard input
1243 (one that points to the terminal or whatever), or the 'bad' one
1244 that points to the read side of a broken pipe. */
1245
1246 child->good_stdin = !good_stdin_used;
1247 if (child->good_stdin)
1248 good_stdin_used = 1;
1249
1250 child->deleted = 0;
1251
1252 /* Set up the environment for the child. */
1253 if (child->environment == 0)
1254 child->environment = target_environment (child->file);
1255
1256#if !defined(__MSDOS__) && !defined(WINDOWS32)
1257
1258 /* start_waiting_job has set CHILD->remote if we can start a remote job. */
1259 if (child->remote)
1260 {
1261 int is_remote, id, used_stdin;
1262 if (start_remote_job (argv, child->environment,
1263 child->good_stdin ? 0 : get_bad_stdin (),
1264 &is_remote, &id, &used_stdin))
1265 /* Don't give up; remote execution may fail for various reasons. If
1266 so, simply run the job locally. */
1267 goto run_local;
1268 else
1269 {
1270 if (child->good_stdin && !used_stdin)
1271 {
1272 child->good_stdin = 0;
1273 good_stdin_used = 0;
1274 }
1275 child->remote = is_remote;
1276 child->pid = id;
1277 }
1278 }
1279 else
1280 {
1281 /* Fork the child process. */
1282
1283 char **parent_environ;
1284
1285 run_local:
1286 block_sigs ();
1287
1288 child->remote = 0;
1289
1290 parent_environ = environ;
1291
1292 jobserver_pre_child (flags & COMMANDS_RECURSE);
1293
1294 child->pid = child_execute_job (&child->output, child->good_stdin, argv, child->environment);
1295
1296 environ = parent_environ; /* Restore value child may have clobbered. */
1297 jobserver_post_child (flags & COMMANDS_RECURSE);
1298
1299 if (child->pid < 0)
1300 {
1301 /* Fork failed! */
1302 unblock_sigs ();
1303 perror_with_name ("fork", "");
1304 goto error;
1305 }
1306 }
1307
1308#else /* __MSDOS__ or Amiga or WINDOWS32 */
1309#ifdef __MSDOS__
1310 {
1311 int proc_return;
1312
1313 block_sigs ();
1314 dos_status = 0;
1315
1316 /* We call 'system' to do the job of the SHELL, since stock DOS
1317 shell is too dumb. Our 'system' knows how to handle long
1318 command lines even if pipes/redirection is needed; it will only
1319 call COMMAND.COM when its internal commands are used. */
1320 if (execute_by_shell)
1321 {
1322 char *cmdline = argv[0];
1323 /* We don't have a way to pass environment to 'system',
1324 so we need to save and restore ours, sigh... */
1325 char **parent_environ = environ;
1326
1327 environ = child->environment;
1328
1329 /* If we have a *real* shell, tell 'system' to call
1330 it to do everything for us. */
1331 if (unixy_shell)
1332 {
1333 /* A *real* shell on MSDOS may not support long
1334 command lines the DJGPP way, so we must use 'system'. */
1335 cmdline = argv[2]; /* get past "shell -c" */
1336 }
1337
1338 dos_command_running = 1;
1339 proc_return = system (cmdline);
1340 environ = parent_environ;
1341 execute_by_shell = 0; /* for the next time */
1342 }
1343 else
1344 {
1345 dos_command_running = 1;
1346 proc_return = spawnvpe (P_WAIT, argv[0], argv, child->environment);
1347 }
1348
1349 /* Need to unblock signals before turning off
1350 dos_command_running, so that child's signals
1351 will be treated as such (see fatal_error_signal). */
1352 unblock_sigs ();
1353 dos_command_running = 0;
1354
1355 /* If the child got a signal, dos_status has its
1356 high 8 bits set, so be careful not to alter them. */
1357 if (proc_return == -1)
1358 dos_status |= 0xff;
1359 else
1360 dos_status |= (proc_return & 0xff);
1361 ++dead_children;
1362 child->pid = dos_pid++;
1363 }
1364#endif /* __MSDOS__ */
1365#ifdef WINDOWS32
1366 {
1367 HANDLE hPID;
1368 char* arg0;
1369 int outfd = FD_STDOUT;
1370 int errfd = FD_STDERR;
1371
1372 /* make UNC paths safe for CreateProcess -- backslash format */
1373 arg0 = argv[0];
1374 if (arg0 && arg0[0] == '/' && arg0[1] == '/')
1375 for ( ; arg0 && *arg0; arg0++)
1376 if (*arg0 == '/')
1377 *arg0 = '\\';
1378
1379 /* make sure CreateProcess() has Path it needs */
1380 sync_Path_environment ();
1381
1382#ifndef NO_OUTPUT_SYNC
1383 /* Divert child output if output_sync in use. */
1384 if (child->output.syncout)
1385 {
1386 if (child->output.out >= 0)
1387 outfd = child->output.out;
1388 if (child->output.err >= 0)
1389 errfd = child->output.err;
1390 }
1391#else
1392 outfd = errfd = -1;
1393#endif
1394 hPID = process_easy (argv, child->environment, outfd, errfd);
1395
1396 if (hPID != INVALID_HANDLE_VALUE)
1397 child->pid = (pid_t) hPID;
1398 else
1399 {
1400 int i;
1401 unblock_sigs ();
1402 fprintf (stderr,
1403 _("process_easy() failed to launch process (e=%ld)\n"),
1404 process_last_err (hPID));
1405 for (i = 0; argv[i]; i++)
1406 fprintf (stderr, "%s ", argv[i]);
1407 fprintf (stderr, _("\nCounted %d args in failed launch\n"), i);
1408 goto error;
1409 }
1410 }
1411#endif /* WINDOWS32 */
1412#endif /* __MSDOS__ or Amiga or WINDOWS32 */
1413
1414 /* Bump the number of jobs started in this second. */
1415 ++job_counter;
1416
1417 /* We are the parent side. Set the state to
1418 say the commands are running and return. */
1419
1420 set_command_state (child->file, cs_running);
1421
1422 OUTPUT_UNSET();
1423 return;
1424
1425 error:
1426 child->file->update_status = us_failed;
1427 notice_finished_file (child->file);
1428 OUTPUT_UNSET();
1429}

1431/* Try to start a child running.
1432 Returns nonzero if the child was started (and maybe finished), or zero if
1433 the load was too high and the child was put on the 'waiting_jobs' chain. */
1434
1435static int
1436start_waiting_job (struct child *c, target_stack_node_t *p_call_stack)
1437{
1438 struct file *f = c->file;
1439
1440 /* If we can start a job remotely, we always want to, and don't care about
1441 the local load average. We record that the job should be started
1442 remotely in C->remote for start_job_command to test. */
1443
1444 c->remote = start_remote_job_p (1);
1445
1446 /* If we are running at least one job already and the load average
1447 is too high, make this one wait. */
1448 if (!c->remote
1449 && ((job_slots_used > 0 && load_too_high ())
1450#ifdef WINDOWS32
1451 || (process_used_slots () >= MAXIMUM_WAIT_OBJECTS)
1452#endif
1453 ))
1454 {
1455 /* Put this child on the chain of children waiting for the load average
1456 to go down. */
1457 set_command_state (f, cs_running);
1458 c->next = waiting_jobs;
1459 waiting_jobs = c;
1460 return 0;
1461 }
1462
1463 /* Start the first command; reap_children will run later command lines. */
1464 start_job_command (c, p_call_stack);
1465
1466 switch (f->command_state)
1467 {
1468 case cs_running:
1469 c->next = children;
1470 DB (DB_JOBS, (_("Putting child %p (%s) PID %s%s on the chain.\n"),
1471 c, c->file->name, pid2str (c->pid),
1472 c->remote ? _(" (remote)") : ""));
1473 children = c;
1474 /* One more job slot is in use. */
1475 ++job_slots_used;
1476 unblock_sigs ();
1477 break;
1478
1479 case cs_not_started:
1480 /* All the command lines turned out to be empty. */
1481 f->update_status = us_success;
1482 /* FALLTHROUGH */
1483
1484 case cs_finished:
1485 notice_finished_file (f);
1486 free_child (c);
1487 break;
1488
1489 default:
1490 assert (f->command_state == cs_finished);
1491 break;
1492 }
1493
1494 return 1;
1495}

1497/* Create a 'struct child' for FILE and start its commands running. */
1498
1499void
1500new_job (struct file *file, target_stack_node_t *p_call_stack)
1501{
1502 struct commands *cmds = file->cmds;
1503 struct child *c;
1504 char **lines;
1505 unsigned int i;
1506
1507 /* Let any previously decided-upon jobs that are waiting
1508 for the load to go down start before this new one. */
1509 start_waiting_jobs (p_call_stack);
1510
1511 /* Reap any children that might have finished recently. */
1512 reap_children (0, 0, p_call_stack);
1513
1514 /* Chop the commands up into lines if they aren't already. */
1515 chop_commands (cmds);
1516
1517 /* Start the command sequence, record it in a new
1518 'struct child', and add that to the chain. */
1519
1520 c = xcalloc (sizeof (struct child));
1521 output_init (&c->output);
1522
1523 c->file = file;
1524 c->sh_batch_file = NULL;
1525
1526 /* Cache dontcare flag because file->dontcare can be changed once we
1527 return. Check dontcare inheritance mechanism for details. */
1528 c->dontcare = file->dontcare;
1529
1530 /* Start saving output in case the expansion uses $(info ...) etc. */
1531 OUTPUT_SET (&c->output);
1532
1533 /* Expand the command lines and store the results in LINES. */
1534 lines = xmalloc (cmds->ncommand_lines * sizeof (char *));
1535 for (i = 0; i < cmds->ncommand_lines; ++i)
1536 {
1537 /* Collapse backslash-newline combinations that are inside variable
1538 or function references. These are left alone by the parser so
1539 that they will appear in the echoing of commands (where they look
1540 nice); and collapsed by construct_command_argv when it tokenizes.
1541 But letting them survive inside function invocations loses because
1542 we don't want the functions to see them as part of the text. */
1543
1544 char *in, *out, *ref;
1545
1546 /* IN points to where in the line we are scanning.
1547 OUT points to where in the line we are writing.
1548 When we collapse a backslash-newline combination,
1549 IN gets ahead of OUT. */
1550
1551 in = out = cmds->command_lines[i];
1552 while ((ref = strchr (in, '$')) != 0)
1553 {
1554 ++ref; /* Move past the $. */
1555
1556 if (out != in)
1557 /* Copy the text between the end of the last chunk
1558 we processed (where IN points) and the new chunk
1559 we are about to process (where REF points). */
1560 memmove (out, in, ref - in);
1561
1562 /* Move both pointers past the boring stuff. */
1563 out += ref - in;
1564 in = ref;
1565
1566 if (*ref == '(' || *ref == '{')
1567 {
1568 char openparen = *ref;
1569 char closeparen = openparen == '(' ? ')' : '}';
1570 char *outref;
1571 int count;
1572 char *p;
1573
1574 *out++ = *in++; /* Copy OPENPAREN. */
1575 outref = out;
1576 /* IN now points past the opening paren or brace.
1577 Count parens or braces until it is matched. */
1578 count = 0;
1579 while (*in != '\0')
1580 {
1581 if (*in == closeparen && --count < 0)
1582 break;
1583 else if (*in == '\\' && in[1] == '\n')
1584 {
1585 /* We have found a backslash-newline inside a
1586 variable or function reference. Eat it and
1587 any following whitespace. */
1588
1589 int quoted = 0;
1590 for (p = in - 1; p > ref && *p == '\\'; --p)
1591 quoted = !quoted;
1592
1593 if (quoted)
1594 /* There were two or more backslashes, so this is
1595 not really a continuation line. We don't collapse
1596 the quoting backslashes here as is done in
1597 collapse_continuations, because the line will
1598 be collapsed again after expansion. */
1599 *out++ = *in++;
1600 else
1601 {
1602 /* Skip the backslash, newline, and whitespace. */
1603 in += 2;
1604 NEXT_TOKEN (in);
1605
1606 /* Discard any preceding whitespace that has
1607 already been written to the output. */
1608 while (out > outref && ISBLANK (out[-1]))
1609 --out;
1610
1611 /* Replace it all with a single space. */
1612 *out++ = ' ';
1613 }
1614 }
1615 else
1616 {
1617 if (*in == openparen)
1618 ++count;
1619
1620 *out++ = *in++;
1621 }
1622 }
1623 }
1624 }
1625
1626 /* There are no more references in this line to worry about.
1627 Copy the remaining uninteresting text to the output. */
1628 if (out != in)
1629 memmove (out, in, strlen (in) + 1);
1630
1631 /* Finally, expand the line, keeping track of the line number
1632 within that command block. */
1633 cmds->fileinfo.offset = i;
1634 if (p_call_stack && p_call_stack->p_target) {
1635 gmk_floc *p_floc = &(p_call_stack->p_target->floc);
1636 p_floc->offset = i;
1637 }
1638
1639 lines[i] = allocated_variable_expand_for_file (cmds->command_lines[i],
1640 file);
1641 }
1642
1643 cmds->fileinfo.offset = 0;
1644 c->command_lines = lines;
1645
1646 /* Fetch the first command line to be run. */
1647 job_next_command (c);
1648
1649 /* Wait for a job slot to be freed up. If we allow an infinite number
1650 don't bother; also job_slots will == 0 if we're using the jobserver. */
1651
1652 if (job_slots != 0)
1653 while (job_slots_used == job_slots)
1654 reap_children (1, 0, p_call_stack);
1655
1656#ifdef MAKE_JOBSERVER
1657 /* If we are controlling multiple jobs make sure we have a token before
1658 starting the child. */
1659
1660 /* This can be inefficient. There's a decent chance that this job won't
1661 actually have to run any subprocesses: the command script may be empty
1662 or otherwise optimized away. It would be nice if we could defer
1663 obtaining a token until just before we need it, in start_job_command.
1664 To do that we'd need to keep track of whether we'd already obtained a
1665 token (since start_job_command is called for each line of the job, not
1666 just once). Also more thought needs to go into the entire algorithm;
1667 this is where the old parallel job code waits, so... */
1668
1669 else if (jobserver_enabled ())
1670 while (1)
1671 {
1672 int got_token;
1673
1674 DB (DB_JOBS, ("Need a job token; we %shave children\n",
1675 children ? "" : "don't "));
1676
1677 /* If we don't already have a job started, use our "free" token. */
1678 if (!jobserver_tokens)
1679 break;
1680
1681 /* Prepare for jobserver token acquisition. */
1682 jobserver_pre_acquire ();
1683
1684 /* Reap anything that's currently waiting. */
1685 reap_children (0, 0, p_call_stack);
1686
1687 /* Kick off any jobs we have waiting for an opportunity that
1688 can run now (i.e., waiting for load). */
1689 start_waiting_jobs (p_call_stack);
1690
1691 /* If our "free" slot is available, use it; we don't need a token. */
1692 if (!jobserver_tokens)
1693 break;
1694
1695 /* There must be at least one child already, or we have no business
1696 waiting for a token. */
1697 if (!children)
1698 fatal_err(p_call_stack, "INTERNAL: no children as we go to sleep on read\n");
1699
1700 /* Get a token. */
1701 got_token = jobserver_acquire (waiting_jobs != NULL);
1702
1703 /* If we got one, we're done here. */
1704 if (got_token == 1)
1705 {
1706 DB (DB_JOBS, (_("Obtained token for child %p (%s).\n"),
1707 c, c->file->name));
1708 break;
1709 }
1710 }
1711#endif
1712
1713 ++jobserver_tokens;
1714
1715 /* Trace the build.
1716 Use message here so that changes to working directories are logged. */
1717 if (db_level & DB_SHELL)
1718 {
1719 char *newer = allocated_variable_expand_for_file ("$?", c->file);
1720 const char *nm;
1721
1722 if (! cmds->fileinfo.filenm)
1723 nm = _("<builtin>");
1724 else
1725 {
1726 char *n = alloca (strlen (cmds->fileinfo.filenm) + 1 + 11 + 1);
1727 sprintf (n, "%s:%lu", cmds->fileinfo.filenm, cmds->fileinfo.lineno);
1728 nm = n;
1729 }
1730
1731 if (newer[0] == '\0')
1732 OSS (message, 0,
1733 _("%s: target '%s' does not exist"), nm, c->file->name);
1734 else
1735 OSSS (message, 0,
1736 _("%s: update target '%s' due to: %s"), nm, c->file->name, newer);
1737
1738 free (newer);
1739 }
1740
1741 /* FIXME: The below is a sign that we need update location somewhere else
1742 */
1743 if (cmds->fileinfo.filenm) {
1744 if (!file->floc.filenm) {
1745 file->floc.filenm = cmds->fileinfo.filenm;
1746 file->floc.lineno = cmds->fileinfo.lineno - 1;
1747 if (!p_call_stack->p_target->floc.filenm) {
1748 p_call_stack->p_target->floc.filenm = file->floc.filenm;
1749 p_call_stack->p_target->floc.lineno = file->floc.lineno;
1750 }
1751 }
1752 }
1753
1754 /* The job is now primed. Start it running.
1755 (This will notice if there is in fact no recipe.) */
1756 (void) start_waiting_job (c, p_call_stack);
1757
1758 if (job_slots == 1 || not_parallel)
1759 /* Since there is only one job slot, make things run linearly.
1760 Wait for the child to die, setting the state to 'cs_finished'. */
1761 while (file->command_state == cs_running)
1762 reap_children (1, 0, p_call_stack);
1763
1764 OUTPUT_UNSET ();
1765 return;
1766}

1768/* Move CHILD's pointers to the next command for it to execute.
1769 Returns nonzero if there is another command. */

1771static int
1772job_next_command (struct child *child)
1773{
1774 while (child->command_ptr == 0 || *child->command_ptr == '\0')
1775 {

1777 if (child->command_line == child->file->cmds->ncommand_lines)

1793/* Determine if the load average on the system is too high to start a new job.
1794 The real system load average is only recomputed once a second. However, a
1795 very parallel make can easily start tens or even hundreds of jobs in a
1796 second, which brings the system to its knees for a while until that first
1797 batch of jobs clears out.
1798
1799 To avoid this we use a weighted algorithm to try to account for jobs which
1800 have been started since the last second, and guess what the load average
1801 would be now if it were computed.
1802
1803 This algorithm was provided by Thomas Riedl <thomas.riedl@siemens.com>,
1804 who writes:
1805
1806! calculate something load-oid and add to the observed sys.load,
1807! so that latter can catch up:
1808! - every job started increases jobctr;
1809! - every dying job decreases a positive jobctr;
1810! - the jobctr value gets zeroed every change of seconds,
1811! after its value*weight_b is stored into the 'backlog' value last_sec
1812! - weight_a times the sum of jobctr and last_sec gets
1813! added to the observed sys.load.
1814!
1815! The two weights have been tried out on 24 and 48 proc. Sun Solaris-9
1816! machines, using a several-thousand-jobs-mix of cpp, cc, cxx and smallish
1817! sub-shelled commands (rm, echo, sed...) for tests.
1818! lowering the 'direct influence' factor weight_a (e.g. to 0.1)
1819! resulted in significant excession of the load limit, raising it
1820! (e.g. to 0.5) took bad to small, fast-executing jobs and didn't
1821! reach the limit in most test cases.
1822!
1823! lowering the 'history influence' weight_b (e.g. to 0.1) resulted in
1824! exceeding the limit for longer-running stuff (compile jobs in
1825! the .5 to 1.5 sec. range),raising it (e.g. to 0.5) overrepresented
1826! small jobs' effects.
1827
1828 */
1829
1830#define LOAD_WEIGHT_A 0.25
1831#define LOAD_WEIGHT_B 0.25
1832
1833static int
1834load_too_high (void)
1835{
1836#if defined(__MSDOS__)
1837 return 1;
1838#else
1839 static double last_sec;
1840 static time_t last_now;
1841 double load, guess;
1842 time_t now;
1843
1844#ifdef WINDOWS32
1845 /* sub_proc.c cannot wait for more than MAXIMUM_WAIT_OBJECTS children */
1846 if (process_used_slots () >= MAXIMUM_WAIT_OBJECTS)
1847 return 1;
1848#endif
1849
1850 if (max_load_average < 0)
1851 return 0;
1852
1853 /* Find the real system load average. */
1854 make_access ();
1855 if (getloadavg (&load, 1) != 1)
1856 {
1857 static int lossage = -1;
1858 /* Complain only once for the same error. */
1859 if (lossage == -1 || errno != lossage)
1860 {
1861 if (errno == 0)
1862 /* An errno value of zero means getloadavg is just unsupported. */
1863 O (error, NILF,
1864 _("cannot enforce load limits on this operating system"));
1865 else
1866 perror_with_name (_("cannot enforce load limit: "), "getloadavg");
1867 }
1868 lossage = errno;
1869 load = 0;
1870 }
1871 user_access ();
1872
1873 /* If we're in a new second zero the counter and correct the backlog
1874 value. Only keep the backlog for one extra second; after that it's 0. */
1875 now = time (NULL);
1876 if (last_now < now)
1877 {
1878 if (last_now == now - 1)
1879 last_sec = LOAD_WEIGHT_B * job_counter;
1880 else
1881 last_sec = 0.0;
1882
1883 job_counter = 0;
1884 last_now = now;
1885 }
1886
1887 /* Try to guess what the load would be right now. */
1888 guess = load + (LOAD_WEIGHT_A * (job_counter + last_sec));
1889
1890 DB (DB_JOBS, ("Estimated system load = %f (actual = %f) (max requested = %f)\n",
1891 guess, load, max_load_average));
1892
1893 return guess >= max_load_average;
1894#endif
1895}

1897/* Start jobs that are waiting for the load to be lower. */
1898
1899void
1900start_waiting_jobs (target_stack_node_t *p_call_stack)
1901{
1902 struct child *job;
1903
1904 if (waiting_jobs == 0)
1905 return;
1906
1907 do
1908 {
1909 /* Check for recently deceased descendants. */
1910 reap_children (0, 0, NULL);
1911
1912 /* Take a job off the waiting list. */
1913 job = waiting_jobs;
1914 waiting_jobs = job->next;
1915
1916 /* Try to start that job. We break out of the loop as soon
1917 as start_waiting_job puts one back on the waiting list. */
1918 }
1919 while (start_waiting_job (job, p_call_stack) && waiting_jobs != 0);
1920
1921 return;
1922}
