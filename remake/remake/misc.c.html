32/* Compare strings *S1 and *S2.
33 Return negative if the first is less, positive if it is greater,
34 zero if they are equal. */
35
36int
37alpha_compare (const void *v1, const void *v2)
38{
39 const char *s1 = *((char **)v1);
40 const char *s2 = *((char **)v2);
41
42 if (*s1 != *s2)
43 return *s1 - *s2;
44 return strcmp (s1, s2);
45}

47/* Discard each backslash-newline combination from LINE.
48 Backslash-backslash-newline combinations become backslash-newlines.
49 This is done by copying the text at LINE into itself. */
50
51void
52collapse_continuations (char *line)
53{
54 char *in, *out, *p;
55
56 in = strchr (line, '\n');
57 if (in == 0)
58 return;
59
60 out = in;
61 while (out > line && out[-1] == '\\')
62 --out;
63
64 while (*in != '\0')
65 {
66 /* BS_WRITE gets the number of quoted backslashes at
67 the end just before IN, and BACKSLASH gets nonzero
68 if the next character is quoted. */
69 unsigned int backslash = 0;
70 unsigned int bs_write = 0;
71 for (p = in - 1; p >= line && *p == '\\'; --p)
72 {
73 if (backslash)
74 ++bs_write;
75 backslash = !backslash;
76
77 /* It should be impossible to go back this far without exiting,
78 but if we do, we can't get the right answer. */
79 if (in == out - 1)
80 abort ();
81 }
82
83 /* Output the appropriate number of backslashes. */
84 while (bs_write-- > 0)
85 *out++ = '\\';
86
87 /* Skip the newline. */
88 ++in;
89
90 if (backslash)
91 {
92 /* Backslash/newline handling:
93 In traditional GNU make all trailing whitespace, consecutive
94 backslash/newlines, and any leading non-newline whitespace on the
95 next line is reduced to a single space.
96 In POSIX, each backslash/newline and is replaced by a space. */
97 while (ISBLANK (*in))
98 ++in;
99 if (! posix_pedantic)
100 while (out > line && ISBLANK (out[-1]))
101 --out;
102 *out++ = ' ';
103 }
104 else
105 /* If the newline isn't quoted, put it in the output. */
106 *out++ = '\n';
107
108 /* Now copy the following line to the output.
109 Stop when we find backslashes followed by a newline. */
110 while (*in != '\0')
111 if (*in == '\\')
112 {
113 p = in + 1;
114 while (*p == '\\')
115 ++p;
116 if (*p == '\n')
117 {
118 in = p;
119 break;
120 }
121 while (in < p)
122 *out++ = *in++;
123 }
124 else
125 *out++ = *in++;
126 }
127
128 *out = '\0';
129}

141/* Return a string whose contents concatenate the NUM strings provided
142 This string lives in static, re-used memory. */
143
144const char *
145concat (unsigned int num, ...)
146{
147 static unsigned int rlen = 0;
148 static char *result = NULL;
149 unsigned int ri = 0;
150 va_list args;
151
152 va_start (args, num);
153
154 while (num-- > 0)
155 {
156 const char *s = va_arg (args, const char *);
157 unsigned int l = xstrlen (s);
158
159 if (l == 0)
160 continue;
161
162 if (ri + l > rlen)
163 {
164 rlen = ((rlen ? rlen : 60) + l) * 2;
165 result = xrealloc (result, rlen);
166 }
167
168 memcpy (result + ri, s, l);
169 ri += l;
170 }
171
172 va_end (args);
173
174 /* Get some more memory if we don't have enough space for the
175 terminating '\0'. */
176 if (ri == rlen)
177 {
178 rlen = (rlen ? rlen : 60) * 2;
179 result = xrealloc (result, rlen);
180 }
181
182 result[ri] = '\0';
183
184 return result;
185}

191strerror (int errnum)
192{

194#ifndef __DECC
195 extern char *sys_errlist[];
196#endif

198

208/* Limited INDEX:
209 Search through the string STRING, which ends at LIMIT, for the character C.
210 Returns a pointer to the first occurrence, or nil if none is found.
211 Like INDEX except that the string searched ends where specified
212 instead of at the first null. */
213
214char *
215lindex (const char *s, const char *limit, int c)
216{
217 while (s < limit)
218 if (*s++ == c)
219 return (char *)(s - 1);
220
221 return 0;
222}

242/* Find the next token in PTR; return the address of it, and store the length
243 of the token into *LENGTHPTR if LENGTHPTR is not nil. Set *PTR to the end
244 of the token, so this function can be called repeatedly in a loop. */
245
246char *
247find_next_token (const char **ptr, unsigned int *lengthptr)
248{
249 const char *p = next_token (*ptr);
250
251 if (*p == '\0')
252 return 0;
253
254 *ptr = end_of_token (p);
255 if (lengthptr != 0)
256 *lengthptr = *ptr - p;
257
258 return (char *)p;
259}

262/* Copy a chain of 'struct dep'. For 2nd expansion deps, dup the name. */
263
264struct dep *
265copy_dep_chain (const struct dep *d)
266{
267 struct dep *firstnew = 0;
268 struct dep *lastnew = 0;
269
270 while (d != 0)
271 {
272 struct dep *c = xmalloc (sizeof (struct dep));
273 memcpy (c, d, sizeof (struct dep));
274
275 if (c->need_2nd_expansion)
276 c->name = xstrdup (c->name);
277
278 c->next = 0;
279 if (firstnew == 0)
280 firstnew = lastnew = c;
281 else
282 lastnew = lastnew->next = c;
283
284 d = d->next;
285 }
286
287 return firstnew;
288}

305#if !HAVE_STRCASECMP && !HAVE_STRICMP && !HAVE_STRCMPI
306/* If we don't have strcasecmp() (from POSIX), or anything that can substitute
307 for it, define our own version. */
308
309int
310strcasecmp (const char *s1, const char *s2)
311{
312 while (1)
313 {
314 int c1 = (int) *(s1++);
315 int c2 = (int) *(s2++);
316
317 if (isalpha (c1))
318 c1 = tolower (c1);
319 if (isalpha (c2))
320 c2 = tolower (c2);
321
322 if (c1 != '\0' && c1 == c2)
323 continue;
324
325 return (c1 - c2);
326 }
327}
328#endif

330#if !HAVE_STRNCASECMP && !HAVE_STRNICMP && !HAVE_STRNCMPI

332 substitute for it, define our own version. */

335strncasecmp (const char *s1, const char *s2, int n)

346

348 continue;
349

351 }
352
353 return 0;

401/* Keep track of the user and group IDs for user- and make- access. */
402static int user_uid = -1, user_gid = -1, make_uid = -1, make_gid = -1;
403#define access_inited (user_uid != -1)
404static enum { make, user } current_access;

407/* Under -d, write a message describing the current IDs. */
408
409static void
410log_access (const char *flavor)
411{
412 if (! ISDB (DB_JOBS))
413 return;
414
415 /* All the other debugging messages go to stdout,
416 but we write this one to stderr because it might be
417 run in a child fork whose stdout is piped. */
418
419 fprintf (stderr, _("%s: user %lu (real %lu), group %lu (real %lu)\n"),
420 flavor, (unsigned long) geteuid (), (unsigned long) getuid (),
421 (unsigned long) getegid (), (unsigned long) getgid ());
422 fflush (stderr);
423}

426static void
427init_access (void)
428{
429#ifndef VMS
430 user_uid = getuid ();
431 user_gid = getgid ();
432
433 make_uid = geteuid ();
434 make_gid = getegid ();
435
436 /* Do these ever fail? */
437 if (user_uid == -1 || user_gid == -1 || make_uid == -1 || make_gid == -1)
438 pfatal_with_name ("get{e}[gu]id");

440 log_access (_("Initialized access"));
441

448/* Give the process appropriate permissions for access to
449 user data (i.e., to stat files, or to spawn a child process). */
450void
451user_access (void)
452{
453#ifdef GETLOADAVG_PRIVILEGED
454
455 if (!access_inited)
456 init_access ();
457
458 if (current_access == user)
459 return;
460
461 /* We are in "make access" mode. This means that the effective user and
462 group IDs are those of make (if it was installed setuid or setgid).
463 We now want to set the effective user and group IDs to the real IDs,
464 which are the IDs of the process that exec'd make. */
465
466#ifdef HAVE_SETEUID
467
468 /* Modern systems have the seteuid/setegid calls which set only the
469 effective IDs, which is ideal. */
470
471 if (seteuid (user_uid) < 0)
472 pfatal_with_name ("user_access: seteuid");
473
474#else /* Not HAVE_SETEUID. */
475
476#ifndef HAVE_SETREUID
477
478 /* System V has only the setuid/setgid calls to set user/group IDs.
479 There is an effective ID, which can be set by setuid/setgid.
480 It can be set (unless you are root) only to either what it already is
481 (returned by geteuid/getegid, now in make_uid/make_gid),
482 the real ID (return by getuid/getgid, now in user_uid/user_gid),
483 or the saved set ID (what the effective ID was before this set-ID
484 executable (make) was exec'd). */
485
486 if (setuid (user_uid) < 0)
487 pfatal_with_name ("user_access: setuid");
488
489#else /* HAVE_SETREUID. */
490
491 /* In 4BSD, the setreuid/setregid calls set both the real and effective IDs.
492 They may be set to themselves or each other. So you have two alternatives
493 at any one time. If you use setuid/setgid, the effective will be set to
494 the real, leaving only one alternative. Using setreuid/setregid, however,
495 you can toggle between your two alternatives by swapping the values in a
496 single setreuid or setregid call. */
497
498 if (setreuid (make_uid, user_uid) < 0)
499 pfatal_with_name ("user_access: setreuid");
500
501#endif /* Not HAVE_SETREUID. */
502#endif /* HAVE_SETEUID. */
503
504#ifdef HAVE_SETEGID
505 if (setegid (user_gid) < 0)
506 pfatal_with_name ("user_access: setegid");
507#else
508#ifndef HAVE_SETREGID
509 if (setgid (user_gid) < 0)
510 pfatal_with_name ("user_access: setgid");
511#else
512 if (setregid (make_gid, user_gid) < 0)
513 pfatal_with_name ("user_access: setregid");
514#endif
515#endif
516
517 current_access = user;
518
519 log_access (_("User access"));
520
521#endif /* GETLOADAVG_PRIVILEGED */
522}

524/* Give the process appropriate permissions for access to
525 make data (i.e., the load average). */
526void
527make_access (void)
528{
529#ifdef GETLOADAVG_PRIVILEGED
530
531 if (!access_inited)
532 init_access ();
533
534 if (current_access == make)
535 return;
536
537 /* See comments in user_access, above. */
538
539#ifdef HAVE_SETEUID
540 if (seteuid (make_uid) < 0)
541 pfatal_with_name ("make_access: seteuid");
542#else
543#ifndef HAVE_SETREUID
544 if (setuid (make_uid) < 0)
545 pfatal_with_name ("make_access: setuid");
546#else
547 if (setreuid (user_uid, make_uid) < 0)
548 pfatal_with_name ("make_access: setreuid");
549#endif
550#endif
551
552#ifdef HAVE_SETEGID
553 if (setegid (make_gid) < 0)
554 pfatal_with_name ("make_access: setegid");
555#else
556#ifndef HAVE_SETREGID
557 if (setgid (make_gid) < 0)
558 pfatal_with_name ("make_access: setgid");
559#else
560 if (setregid (user_gid, make_gid) < 0)
561 pfatal_with_name ("make_access: setregid");
562#endif
563#endif
564
565 current_access = make;
566
567 log_access (_("Make access"));
568
569#endif /* GETLOADAVG_PRIVILEGED */
570}

572/* Give the process appropriate permissions for a child process.
573 This is like user_access, but you can't get back to make_access. */
574void
575child_access (void)
576{
577#ifdef GETLOADAVG_PRIVILEGED
578
579 if (!access_inited)
580 abort ();
581
582 /* Set both the real and effective UID and GID to the user's.
583 They cannot be changed back to make's. */
584
585#ifndef HAVE_SETREUID
586 if (setuid (user_uid) < 0)
587 pfatal_with_name ("child_access: setuid");
588#else
589 if (setreuid (user_uid, user_uid) < 0)
590 pfatal_with_name ("child_access: setreuid");
591#endif
592
593#ifndef HAVE_SETREGID
594 if (setgid (user_gid) < 0)
595 pfatal_with_name ("child_access: setgid");
596#else
597 if (setregid (user_gid, user_gid) < 0)
598 pfatal_with_name ("child_access: setregid");
599#endif
600
601 log_access (_("Child access"));
602
603#endif /* GETLOADAVG_PRIVILEGED */
604}

625/* Map of possible stop characters for searching strings. */
626#ifndef UCHAR_MAX
627# define UCHAR_MAX 255
628#endif
629unsigned short stopchar_map[UCHAR_MAX + 1] = {0};
630
631/* This character map locate stop chars when parsing GNU makefiles.
632 Each element is true if we should stop parsing on that character. */
633
634extern void
635initialize_stopchar_map (void)
636{
637 int i;
638
639 stopchar_map[(int)'\0'] = MAP_NUL;
640 stopchar_map[(int)'#'] = MAP_COMMENT;
641 stopchar_map[(int)';'] = MAP_SEMI;
642 stopchar_map[(int)'='] = MAP_EQUALS;
643 stopchar_map[(int)':'] = MAP_COLON;
644 stopchar_map[(int)'%'] = MAP_PERCENT;
645 stopchar_map[(int)'|'] = MAP_PIPE;
646 stopchar_map[(int)'.'] = MAP_DOT | MAP_USERFUNC;
647 stopchar_map[(int)','] = MAP_COMMA;
648 stopchar_map[(int)'$'] = MAP_VARIABLE;
649
650 stopchar_map[(int)'-'] = MAP_USERFUNC;
651 stopchar_map[(int)'_'] = MAP_USERFUNC;
652
653 stopchar_map[(int)' '] = MAP_BLANK;
654 stopchar_map[(int)'\t'] = MAP_BLANK;
655
656 stopchar_map[(int)'/'] = MAP_DIRSEP;
657#if defined(HAVE_DOS_PATHS)
658 stopchar_map[(int)'\\'] = MAP_DIRSEP;
659#endif
660
661 for (i = 1; i <= UCHAR_MAX; ++i)
662 {
663 if (isspace (i) && NONE_SET (stopchar_map[i], MAP_BLANK))
664 /* Don't mark blank characters as newline characters. */
665 stopchar_map[i] |= MAP_NEWLINE;
666 else if (isalnum (i))
667 stopchar_map[i] |= MAP_USERFUNC;
668 }
669}
