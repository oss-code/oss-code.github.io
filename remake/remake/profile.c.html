35/*! \brief Node for an item in the target call stack */
36typedef struct profile_call {
37 const file_t *p_target;
38 struct profile_call *p_next;
39} profile_call_t;
40
41struct profile_entry {
42 uint64_t elapsed_time; /* runtime in milliseconds */
43 const char *name;
44 gmk_floc floc; /* location in Makefile - for tracing */
45 profile_call_t *calls; /* List of targets this target calls. */
46};
47typedef struct profile_entry profile_entry_t;
48
49static struct hash_table profile_table;
50
51static unsigned long
52profile_table_entry_hash_1 (const void *key)
53{
54 return_ISTRING_HASH_1 (((profile_entry_t *) key)->name);
55}
56
57static unsigned long
58profile_table_entry_hash_2 (const void *key)
59{
60 return_ISTRING_HASH_2 (((profile_entry_t *) key)->name);
61}
62
63static int
64profile_hash_cmp (const void *x, const void *y)
65{
66 return_ISTRING_COMPARE (((profile_entry_t *) x)->name,
67 ((profile_entry_t *) y)->name);
68}
69
70static profile_entry_t *
71add_profile_entry (const file_t *target)
72{
73 /* Look up the string in the hash. If it's there, return it. */
74
75 profile_entry_t *new;
76 profile_entry_t **slot;
77 profile_entry_t *profile_entry;
78
79 new = xcalloc (sizeof (profile_entry_t));
80 new->name = target->name;
81
82 slot = (profile_entry_t **) hash_find_slot (&profile_table, new);
83 profile_entry = *slot;
84 if (!HASH_VACANT (profile_entry)) {
85 free(new);
86 return profile_entry;
87 }
88
89 /* Not there yet. So finish initializing data and add it into the
90 profile hash table. */
91 new->calls = NULL;
92 memcpy(&(new->floc), &(target->floc), sizeof(gmk_floc));
93 hash_insert_at (&profile_table, new, slot);
94 return new;
95}

97#define CALLGRIND_FILE_PREFIX "callgrind.out."
98#define CALLGRIND_FILE_TEMPLATE CALLGRIND_FILE_PREFIX "%d"

100/* + 10 is more than enough since 2**64 ~= 10**9 */
101#define CALLGRIND_FILENAME_LEN sizeof(CALLGRIND_FILE_PREFIX) + 20
102
103char callgrind_fname[CALLGRIND_FILENAME_LEN];
104static FILE *callgrind_fd;
105
106#define CALLGRIND_PREAMBLE_TEMPLATE1 "version: 1\n\
107creator: %s\n"
108
109#define CALLGRIND_PREAMBLE_TEMPLATE2 "pid: %d\n\
110\n\
111desc: Trigger: %s\n\
112desc: Node: Targets\n\
113\n\
114positions: line\n\
115events: 100usec\n"

117#ifndef HAVE_GETTIMEOFDAY
118# error Somebody has to work out how to handle if getimeofday is not around
119#endif
120extern bool
121get_time(struct timeval *t) {
122 return (0 != gettimeofday(t, NULL));
123}
124
125static inline uint64_t
126time_in_100micro(struct timeval *time) {
127 return ((time->tv_sec * (uint64_t)10000) +
128 (uint64_t) (time->tv_usec / 100));
129}
130
131extern uint64_t
132time_diff(struct timeval *start_time, struct timeval *finish_time) {
133 struct timeval diff_time;
134 diff_time.tv_sec = finish_time->tv_sec - start_time->tv_sec;
135 diff_time.tv_usec = finish_time->tv_usec;
136 if (diff_time.tv_usec < start_time->tv_usec) {
137 diff_time.tv_usec += 1000000;
138 diff_time.tv_sec--;
139 }
140 diff_time.tv_usec -= start_time->tv_usec;
141 return time_in_100micro(&diff_time);
142}

144static struct timeval program_start_time;
145static struct timeval program_finish_time;
146static bool time_error;
147static pid_t callgrind_pid;
148
149extern bool
150init_callgrind(const char *creator, const char *const *argv) {
151 size_t len;
152 unsigned int i;
153
154 callgrind_pid = getpid();
155 len = sprintf(callgrind_fname, CALLGRIND_FILE_TEMPLATE, callgrind_pid);
156 time_error = get_time(&program_start_time);
157
158 if (len >= CALLGRIND_FILENAME_LEN) {
159 printf("Error in generating name\n");
160 return false;
161 }
162 callgrind_fd = fopen(callgrind_fname, "w");
163 if (NULL == callgrind_fd) {
164 printf("Error in opening callgrind file %s\n", callgrind_fname);
165 return false;
166 }
167 hash_init (&profile_table, 1000, profile_table_entry_hash_1,
168 profile_table_entry_hash_2, profile_hash_cmp);
169 fprintf(callgrind_fd, CALLGRIND_PREAMBLE_TEMPLATE1,
170 creator);
171 fprintf(callgrind_fd, "cmd:");
172 for (i = 0; argv[i]; i++) {
173 fprintf(callgrind_fd, " %s", argv[i]);
174 }
175 fprintf(callgrind_fd, "\n");
176 return true;
177}

185extern void
186add_target(file_t *target, file_t *prev) {
187 profile_entry_t *p = add_profile_entry(target);
188 p->elapsed_time = target->elapsed_time;
189 if (prev) {
190 profile_entry_t *q = add_profile_entry(prev);
191 if (q) {
192 profile_call_t *new = CALLOC(profile_call_t, 1);
193 new->p_target = target;
194 new->p_next = q->calls;
195 q->calls = new;
196 }
197 }
198}

200#ifdef DEBUG_PROFILE
201static void
202print_profile_entry (const void *item)
203{
204 const profile_entry_t *p = item;
205 profile_call_t *c;
206 printf("name: %s, file: %s, line: %" PRIu64 ", time: %" PRIu64 "\n",
207 p->name, p->floc.filenm, p->floc.lineno, p->elapsed_time);
208 for (c = p->calls; c; c = c->p_next) {
209 printf("calls: %s\n", c->p_target->name);
210 }
211}
212
213/* Print all profile entries */
214static void
215print_profile(struct hash_table *hash_table)
216{
217 hash_map (hash_table, print_profile_entry);
218}
219#endif
220
221static void
222callgrind_profile_entry (const void *item)
223{
224 const profile_entry_t *p = item;
225 profile_call_t *c;
226 if (p->floc.filenm) fprintf(callgrind_fd, "fl=%s\n\n", p->floc.filenm);
227 fprintf(callgrind_fd, "fn=%s\n", p->name);
228 fprintf(callgrind_fd, "%" PRIu64 " %" PRIu64 "\n",
229 (uint64_t) p->floc.lineno,
230 p->elapsed_time == 0 ? 1 : p->elapsed_time);
231 for (c = p->calls; c; c = c->p_next) {
232 if (c->p_target->floc.filenm)
233 fprintf(callgrind_fd, "cfi=%s\n", c->p_target->floc.filenm);
234 fprintf(callgrind_fd, "cfn=%s\n", c->p_target->name);
235 fprintf(callgrind_fd, "calls=1 %" PRIu64 "\n",
236 (uint64_t) p->floc.lineno);
237 fprintf(callgrind_fd, "%" PRIu64 " %" PRIu64 "\n",
238 (uint64_t) p->floc.lineno,
239 c->p_target->elapsed_time == 0 ? 1 : c->p_target->elapsed_time);
240 }
241 fprintf(callgrind_fd, "\n");
242}
243
244/* Print all profile entries */
245void
246callgrind_profile_data(struct hash_table *hash_table)
247{
248 hash_map (hash_table, callgrind_profile_entry);
249}

251extern void
252close_callgrind(const char *program_status) {
253 fprintf(callgrind_fd, CALLGRIND_PREAMBLE_TEMPLATE2, callgrind_pid,
254 program_status);
255 if (!time_error) {
256 time_error = time_error || get_time(&program_finish_time);
257 }
258
259 if (!time_error) {
260 fprintf(callgrind_fd, "summary: %" PRIu64 "\n\n",
261 time_diff(&program_start_time, &program_finish_time));
262 }
263#ifdef DEBUG_PROFILE
264 print_profile(&profile_table);
265#endif
266 callgrind_profile_data(&profile_table);
267 printf("Created callgrind profiling data file: %s\n", callgrind_fname);
268 fclose(callgrind_fd);
269}

272#include "config.h"

274int main(int argc, const char * const* argv) {
275 bool rc = init_callgrind(PACKAGE_TARNAME " " PACKAGE_VERSION, argv);
276 init_hash_files();
277 if (rc) {
278 file_t *target = enter_file("Makefile");
279 file_t *target2, *target3;
280 target->floc.filenm = "Makefile";
281
282
283 target2 = enter_file("all");
284 target2->floc.filenm = "Makefile";
285 target2->floc.lineno = 5;
286 target2->elapsed_time = 500;
287 add_target(target2, NULL);
288
289 target3 = enter_file("all-recursive");
290 target3->floc.filenm = "Makefile";
291 target3->floc.lineno = 5;
292 target3->elapsed_time = 1000;
293 add_target(target3, target2);
294
295 close_callgrind("Program termination");
296 }
297 return rc;
298}
