70/* The structure that describes an accepted command switch. */
71
72struct command_switch
73 {
74 int c; /* The switch character. */
75
76 enum /* Type of the value. */
77 {
78 flag, /* Turn int flag on. */
79 flag_off, /* Turn int flag off. */
80 string, /* One string per invocation. */
81 strlist, /* One string per switch. */
82 filename, /* A string containing a file name. */
83 positive_int, /* A positive integer. */
84 floating, /* A floating-point number (double). */
85 ignore /* Ignored. */
86 } type;
87
88 void *value_ptr; /* Pointer to the value-holding variable. */
89
90 unsigned int env:1; /* Can come from MAKEFLAGS. */
91 unsigned int toenv:1; /* Should be put in MAKEFLAGS. */
92 unsigned int no_makefile:1; /* Don't propagate when remaking makefiles. */
93
94 const void *noarg_value; /* Pointer to value used if no arg given. */
95 const void *default_value; /* Pointer to default value. */
96
97 const char *long_name; /* Long option name. */
98 };

562void
563decode_trace_flags (stringlist_t *ppsz_tracing_opts)
564{
565 if (ppsz_tracing_opts) {
566 const char **p;
567 db_level |= (DB_TRACE | DB_SHELL);
568 if (!ppsz_tracing_opts->list)
569 db_level |= (DB_BASIC);
570 else
571 for (p = ppsz_tracing_opts->list; *p != 0; ++p) {
572 if (0 == strcmp(*p, "command"))
573 ;
574 else if (0 == strcmp(*p, "full"))
575 db_level |= (DB_VERBOSE|DB_READ_MAKEFILES);
576 else if (0 == strcmp(*p, "normal"))
577 db_level |= DB_BASIC;
578 else if (0 == strcmp(*p, "noshell"))
579 db_level = DB_BASIC | DB_TRACE;
580 else if (0 == strcmp(*p, "read"))
581 db_level |= DB_READ_MAKEFILES;
582 }
583 }
584}

616/* This character map locate stop chars when parsing GNU makefiles.
617 Each element is true if we should stop parsing on that character. */
618
619static const char *
620expand_command_line_file (const char *name)
621{
622 const char *cp;
623 char *expanded = 0;
624
625 if (name[0] == '\0')
626 O (fatal, NILF, _("empty string invalid as file name"));
627
628 if (name[0] == '~')
629 {
630 expanded = remake_tilde_expand (name);
631 if (expanded && expanded[0] != '\0')
632 name = expanded;
633 }
634
635 /* This is also done in parse_file_seq, so this is redundant
636 for names read from makefiles. It is here for names passed
637 on the command line. */
638 while (name[0] == '.' && name[1] == '/')
639 {
640 name += 2;
641 while (name[0] == '/')
642 /* Skip following slashes: ".//foo" is "foo", not "/foo". */
643 ++name;
644 }
645
646 if (name[0] == '\0')
647 {
648 /* Nothing else but one or more "./", maybe plus slashes! */
649 name = "./";
650 }
651
652 cp = strcache_add (name);
653
654 free (expanded);
655
656 return cp;
657}

669static void
670decode_debug_flags (void)
671{
672 const char **pp;
673
674 if (debug_flag)
675 db_level = DB_ALL;
676
677 if (db_flags)
678 for (pp=db_flags->list; *pp; ++pp)
679 {
680 const char *p = *pp;
681
682 while (1)
683 {
684 switch (tolower (p[0]))

780/*
781 * HANDLE runtime exceptions by avoiding a requestor on the GUI. Capture
782 * exception and print it to stderr instead.
783 *
784 * If ! DB_VERBOSE, just print a simple message and exit.
785 * If DB_VERBOSE, print a more verbose message.
786 * If compiled for DEBUG, let exception pass through to GUI so that
787 * debuggers can attach.
788 */
789LONG WINAPI
790handle_runtime_exceptions (struct _EXCEPTION_POINTERS *exinfo)
791{
792 PEXCEPTION_RECORD exrec = exinfo->ExceptionRecord;
793 LPSTR cmdline = GetCommandLine ();
794 LPSTR prg = strtok (cmdline, " ");
795 CHAR errmsg[1024];
796#ifdef USE_EVENT_LOG
797 HANDLE hEventSource;
798 LPTSTR lpszStrings[1];
799#endif

801 if (! ISDB (DB_VERBOSE))
802 {
803 sprintf (errmsg,
804 _("%s: Interrupt/Exception caught (code = 0x%lx, addr = 0x%p)\n"),
805 prg, exrec->ExceptionCode, exrec->ExceptionAddress);
806 fprintf (stderr, errmsg);
807 exit (255);
808 }
809
810 sprintf (errmsg,
811 _("\nUnhandled exception filter called from program %s\nExceptionCode = %lx\nExceptionFlags = %lx\nExceptionAddress = 0x%p\n"),

813 exrec->ExceptionAddress);

816 && exrec->NumberParameters >= 2)
817 sprintf (&errmsg[strlen(errmsg)],
818 (exrec->ExceptionInformation[0]
819 ? _("Access violation: write operation at address 0x%p\n")

821 (PVOID)exrec->ExceptionInformation[1]);

824#ifdef USE_EVENT_LOG

831 EVENTLOG_ERROR_TYPE, /* event type */
832 0, /* event category */

834 NULL, /* current user's SID */
835 1, /* strings in lpszStrings */
836 0, /* no bytes of raw data */
837 lpszStrings, /* array of error strings */
838 NULL); /* no raw data */
839
840 (VOID) DeregisterEventSource (hEventSource);
841 }
842#endif
843
844 /* Write the error to stderr too */
845 fprintf (stderr, errmsg);
846
847#ifdef DEBUG
848 return EXCEPTION_CONTINUE_SEARCH;
849#else
850 exit (255);
851 return (255); /* not reached */
852#endif

855/*
856 * On WIN32 systems we don't have the luxury of a /bin directory that
857 * is mapped globally to every drive mounted to the system. Since make could
858 * be invoked from any drive, and we don't want to propagate /bin/sh
859 * to every single drive. Allow ourselves a chance to search for
860 * a value for default shell here (if the default path does not exist).
861 */
862
863int
864find_and_set_default_shell (const char *token)
865{
866 int sh_found = 0;
867 char *atoken = 0;
868 const char *search_token;
869 const char *tokend;
870 PATH_VAR(sh_path);
871 extern const char *default_shell;
872
873 if (!token)
874 search_token = default_shell;
875 else
876 search_token = atoken = xstrdup (token);
877
878 /* If the user explicitly requests the DOS cmd shell, obey that request.
879 However, make sure that's what they really want by requiring the value
880 of SHELL either equal, or have a final path element of, "cmd" or
881 "cmd.exe" case-insensitive. */
882 tokend = search_token + strlen (search_token) - 3;
883 if (((tokend == search_token
884 || (tokend > search_token
885 && (tokend[-1] == '/' || tokend[-1] == '\\')))
886 && !strcasecmp (tokend, "cmd"))
887 || ((tokend - 4 == search_token
888 || (tokend - 4 > search_token
889 && (tokend[-5] == '/' || tokend[-5] == '\\')))
890 && !strcasecmp (tokend - 4, "cmd.exe")))
891 {
892 batch_mode_shell = 1;
893 unixy_shell = 0;
894 sprintf (sh_path, "%s", search_token);
895 default_shell = xstrdup (w32ify (sh_path, 0));
896 DB (DB_VERBOSE, (_("find_and_set_shell() setting default_shell = %s\n"),
897 default_shell));
898 sh_found = 1;
899 }
900 else if (!no_default_sh_exe
901 && (token == NULL || !strcmp (search_token, default_shell)))
902 {
903 /* no new information, path already set or known */
904 sh_found = 1;
905 }
906 else if (_access (search_token, 0) == 0)
907 {
908 /* search token path was found */
909 sprintf (sh_path, "%s", search_token);
910 default_shell = xstrdup (w32ify (sh_path, 0));
911 DB (DB_VERBOSE, (_("find_and_set_shell() setting default_shell = %s\n"),
912 default_shell));
913 sh_found = 1;
914 }
915 else
916 {
917 char *p;
918 struct variable *v = lookup_variable (STRING_SIZE_TUPLE ("PATH"));
919
920 /* Search Path for shell */
921 if (v && v->value)
922 {
923 char *ep;
924
925 p = v->value;
926 ep = strchr (p, PATH_SEPARATOR_CHAR);
927
928 while (ep && *ep)
929 {
930 *ep = '\0';
931
932 sprintf (sh_path, "%s/%s", p, search_token);
933 if (_access (sh_path, 0) == 0)
934 {
935 default_shell = xstrdup (w32ify (sh_path, 0));
936 sh_found = 1;
937 *ep = PATH_SEPARATOR_CHAR;
938
939 /* terminate loop */
940 p += strlen (p);
941 }
942 else
943 {
944 *ep = PATH_SEPARATOR_CHAR;
945 p = ++ep;
946 }
947
948 ep = strchr (p, PATH_SEPARATOR_CHAR);
949 }
950
951 /* be sure to check last element of Path */
952 if (p && *p)
953 {
954 sprintf (sh_path, "%s/%s", p, search_token);
955 if (_access (sh_path, 0) == 0)
956 {
957 default_shell = xstrdup (w32ify (sh_path, 0));
958 sh_found = 1;
959 }
960 }
961
962 if (sh_found)
963 DB (DB_VERBOSE,
964 (_("find_and_set_shell() path search set default_shell = %s\n"),
965 default_shell));
966 }
967 }
968
969 /* naive test */
970 if (!unixy_shell && sh_found
971 && (strstr (default_shell, "sh") || strstr (default_shell, "SH")))
972 {
973 unixy_shell = 1;
974 batch_mode_shell = 0;
975 }
976
977#ifdef BATCH_MODE_ONLY_SHELL
978 batch_mode_shell = 1;
979#endif
980
981 free (atoken);
982
983 return (sh_found);
984}

2625{

2627
2628 if (arg[0] == '-' && arg[1] == '\0')
2629 /* Ignore plain '-' for compatibility. */

2632#ifdef VMS

2644 len = strlen(arg);

2650 new_arg[seg1 + seg2] = 0;

2752/* Decode switches from ARGC and ARGV.
2753 They came from the environment if ENV is nonzero. */
2754
2755static void
2756decode_switches (int argc, const char **argv, int env)
2757{
2758 int bad = 0;
2759 register const struct command_switch *cs;
2760 register struct stringlist *sl;
2761 register int c;
2762
2763 /* getopt does most of the parsing for us.
2764 First, get its vectors set up. */
2765
2766 init_switches ();
2767
2768 /* Let getopt produce error messages for the command line,
2769 but not for options from the environment. */
2770 opterr = !env;
2771 /* Reset getopt's state. */
2772 optind = 0;
2773
2774 while (optind < argc)
2775 {
2776 const char *coptarg;
2777
2778 /* Parse the next argument. */
2779 c = getopt_long (argc, (char*const*)argv, options, long_options, NULL);
2780 coptarg = optarg;
2781 if (c == EOF)
2782 /* End of arguments, or "--" marker seen. */
2783 break;
2784 else if (c == 1)
2785 /* An argument not starting with a dash. */
2786 handle_non_switch_argument (coptarg, env);
2787 else if (c == '?')
2788 /* Bad option. We will print a usage message and die later.
2789 But continue to parse the other options so the user can
2790 see all he did wrong. */
2791 bad = 1;
2792 else
2793 for (cs = switches; cs->c != '\0'; ++cs)
2794 if (cs->c == c)
2795 {
2796 /* Whether or not we will actually do anything with
2797 this switch. We test this individually inside the
2798 switch below rather than just once outside it, so that
2799 options which are to be ignored still consume args. */
2800 int doit = !env || cs->env;
2801
2802 switch (cs->type)
2803 {
2804 default:
2805 abort ();
2806
2807 case ignore:
2808 break;
2809
2810 case flag:
2811 case flag_off:
2812 if (doit)
2813 *(int *) cs->value_ptr = cs->type == flag;
2814 break;
2815
2816 case string:
2817 case strlist:
2818 case filename:
2819 if (!doit)
2820 break;
2821
2822 if (! coptarg)
2823 coptarg = xstrdup (cs->noarg_value);
2824 else if (*coptarg == '\0')
2825 {
2826 char opt[2] = "c";
2827 const char *op = opt;
2828
2829 if (short_option (cs->c))
2830 opt[0] = cs->c;
2831 else

2949/* Decode switches from environment variable ENVAR (which is LEN chars long).
2950 We do this by chopping the value into a vector of words, prepending a
2951 dash to the first word if it lacks one, and passing the vector to

3020 no magic by decode_env_switches; also double dollar signs to avoid
