20/** \file expand.h
21 *
22 * \brief Header for variable expansion functions for GNU Make.
23 */
24
25#ifndef REMAKE_EXPAND_H
26#define REMAKE_EXPAND_H
27
28#include "variable.h"
29#include "filedef.h"
30
31/*! Like variable_expand_for_file, but the returned string is malloc'd.
32 This function is called a lot. It wants to be efficient. */
33
34extern char *allocated_variable_expand_for_file(const char *psz_line,
35 file_t *p_file);
36
37
38/*! Expand an argument for an expansion function. The text starting
39 at STR and ending at END is variable-expanded into a
40 null-terminated string that is returned as the value. This is done
41 without clobbering `variable_buffer' or the current
42 variable-expansion that is in progress. */
43
44extern char *expand_argument(const char *str, const char *end);
45
46/*! Install a new variable_buffer context, returning the current one for
47 safe-keeping. */
48
49extern void install_variable_buffer (char **pp_buf, unsigned int *pi_len);
50
51/*! Restore a previously-saved variable_buffer setting (free the current one).
52 */
53
54#define recursively_expand(v) recursively_expand_for_file (v, NULL)
55
56/*! Recursively expand V. The returned string is malloc'd. */
57extern char *recursively_expand_for_file(variable_t *v, file_t *file);
58
59/*! Subroutine of variable_expand and friends:
60 The text to add is LENGTH chars starting at STRING to the variable_buffer.
61 The text is added to the buffer at PTR, and the updated pointer into
62 the buffer is returned as the value. Thus, the value returned by
63 each call to variable_buffer_output should be the first argument to
64 the following call. */
65
66extern void restore_variable_buffer(char *p_buf, unsigned int len);
67
68/** Expand PSZ_LINE. Expansion uses P_FILE_SET if it is not NULL. */
69extern char *variable_expand_set (char *psz_line,
70 variable_set_list_t *p_file_set);
71
72#endif /*REMAKE_EXPAND_H*/
