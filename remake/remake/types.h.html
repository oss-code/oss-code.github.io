54/** Debugger breakpoint type */
55typedef enum {
56 BRK_NONE = 0x00, /**< Mask when none of the below are set. */
57 BRK_BEFORE_PREREQ = 0x01, /**< Stop after prequisites checking done */
58 BRK_AFTER_PREREQ = 0x02, /**< Stop after prequisites checking done */
59 BRK_AFTER_CMD = 0x04, /**< Stop after running commands */
60 BRK_ALL = (BRK_BEFORE_PREREQ|BRK_AFTER_PREREQ|BRK_AFTER_CMD),
61 BRK_TEMP = 0x08, /**< Temporary or one-time breakpoint */
62} breakpoint_mask_t;
