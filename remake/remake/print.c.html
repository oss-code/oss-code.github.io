39/* Write a message indicating what directory we are in
40 and the make level number. */
41
42static void
43err_log_working_directory (void)
44{
45 static char *buf = NULL;
46 static unsigned int len = 0;
47 unsigned int need;
48 const char *fmt;
49 char *p;
50
51 /* Get enough space for the longest possible output. */
52 need = strlen (program) + INTSTR_LENGTH + 2 + 1;
53 if (starting_directory)
54 need += strlen (starting_directory);
55
56 /* Use entire sentences to give the translators a fighting chance. */
57 if (makelevel == 0)
58 if (starting_directory == 0)
59 fmt = _("%s\n");
60 else
61 fmt = _("%s: directory '%s'\n");
62 else
63 if (starting_directory == 0)
64 fmt = _("%s[%u]: unknown directory\n");
65 else
66 fmt = _("%s[%u]: directory '%s'\n");
67
68 need += strlen (fmt);
69
70 if (need > len)
71 {
72 buf = xrealloc (buf, need);
73 len = need;
74 }
75
76 p = buf;
77 if (print_data_base_flag)
78 {
79 *(p++) = '#';
80 *(p++) = ' ';
81 }
82
83 if (makelevel == 0)
84 if (starting_directory == 0)
85 sprintf (p, fmt , program);
86 else
87 sprintf (p, fmt, program, starting_directory);
88 else if (starting_directory == 0)
89 sprintf (p, fmt, program, makelevel);
90 else
91 sprintf (p, fmt, program, makelevel, starting_directory);
92
93 outputs (0, buf);
94}

97void
98err_with_stack (target_stack_node_t *p_call, const char *fmt, ...)
99{
100 va_list args;
101 gmk_floc *p_floc = NULL;
102 file_t *p_target = NULL;
103
104 if (p_call && p_call->p_target) {
105 p_target = p_call->p_target;
106 p_floc = &(p_target->floc);
107 }
108
109 if (p_floc) {
110 if (!p_floc->filenm)
111 fprintf (stderr, "<builtin>");
112 else
113 fprintf (stderr, "%s:%lu: ", p_floc->filenm, p_floc->lineno + p_floc->offset);
114 } else {
115 err_log_working_directory ();
116 if (makelevel == 0)
117 fprintf (stderr, "%s: ", program);
118 else
119 fprintf (stderr, "%s[%u]: ", program, makelevel);
120 }
121
122 va_start (args, fmt);
123 vfprintf (stderr, fmt, args);
124 va_end (args);
125
126 putc ('\n', stderr);
127 if (!no_extended_errors) {
128 if (p_call) {
129 putc ('\n', stdout);
130 print_target_stack(p_call, -1, MAX_STACK_SHOW);
131 } else if (p_stack_floc_top) {
132 putc ('\n', stdout);
133 print_floc_stack(-1, MAX_STACK_SHOW);
134 }
135 }
136 fflush (stdout);
137 fflush (stderr);
138 if (debugger_on_error & DEBUGGER_ON_ERROR)
139 enter_debugger(p_call, p_target, -1, DEBUG_ERROR_HIT);
140}

142/* Print an error message and exit. */
143void
144fatal_err (target_stack_node_t *p_call, const char *fmt, ...)
145{
146 va_list args;
147 gmk_floc *p_floc = NULL;
148 file_t *p_target = NULL;
149
150 err_log_working_directory ();
151
152 if (p_call && p_call->p_target) {
153 p_target = p_call->p_target;
154 p_floc = &(p_target->floc);
155 }
156
157 if (p_floc && p_floc->filenm)
158 fprintf (stderr, "%s:%lu: *** ", p_floc->filenm, p_floc->lineno);
159 else if (makelevel == 0)
160 fprintf (stderr, "%s: *** ", program);
161 else
162 fprintf (stderr, "%s[%u]: *** ", program, makelevel);
163
164 va_start (args, fmt);
165 vfprintf (stderr, fmt, args);
166 va_end (args);
167
168 fputs (_(". Stop.\n"), stderr);
169 if (!no_extended_errors) {
170 if (p_call)
171 print_target_stack(p_call, -1, MAX_STACK_SHOW);
172 else if (p_stack_floc_top)
173 print_floc_stack(-1, MAX_STACK_SHOW);
174 }
175 if ( (debugger_on_error & DEBUGGER_ON_FATAL) || debugger_enabled )
176 enter_debugger(p_call, p_target, 2, DEBUG_ERROR_HIT);
177 die (MAKE_FAILURE);
178}

184char *
185strerror (int errnum)
186{
187 extern int errno, sys_nerr;
188#ifndef __DECC
189 extern char *sys_errlist[];
190#endif
191 static char buf[] = "Unknown error 12345678901234567890";
192
193 if (errno < sys_nerr)
194 return sys_errlist[errnum];
195
196 sprintf (buf, _("Unknown error %d"), errnum);
197 return buf;
198}
199#endif

201/*! Under -d, write a message describing the current IDs. */
202
203void
204log_access (char *flavor)
205{
206 if (! ISDB (DB_JOBS))
207 return;
208
209 /* All the other debugging messages go to stdout,
210 but we write this one to stderr because it might be
211 run in a child fork whose stdout is piped. */
212
213 fprintf (stderr, _("%s: user %lu (real %lu), group %lu (real %lu)\n"),
214 flavor, (unsigned long) geteuid (), (unsigned long) getuid (),
215 (unsigned long) getegid (), (unsigned long) getgid ());
216 fflush (stderr);
217}

219/*! Display a variable and its value. */
220void
221print_variable (variable_t *p_v)
222{
223 if (p_v) {
224 const char *psz_origin = origin2str(p_v->origin);
225 if (NULL != p_v->fileinfo.filenm) {
226 printf(_("%s:%lu (origin: %s) %s = %s\n"),
227 p_v->fileinfo.filenm, p_v->fileinfo.lineno,
228 psz_origin,
229 p_v->name, p_v->value);
230 } else {
231 printf("(origin %s) %s = %s\n", psz_origin, p_v->name, p_v->value);
232 }
233 }
234}

236/*! Display a variable and its value with all substitutions included. */
237void
238print_variable_expand (variable_t *p_v)
239{
240 if (p_v) {
241 const char *psz_origin = origin2str(p_v->origin);
242 if (NULL != p_v->fileinfo.filenm) {
243 printf(_("%s:%lu (origin: %s) %s := %s\n"),
244 p_v->fileinfo.filenm, p_v->fileinfo.lineno,
245 psz_origin,
246 p_v->name, variable_expand(p_v->value));
247 } else {
248 printf("(origin %s) %s := %s\n", psz_origin,
249 p_v->name, variable_expand(p_v->value));
250 }
251 }

272/*! Show a command before executing it. */
273extern void
274print_floc_prefix (const gmk_floc *p_floc)
275{
276 if (!p_floc) return;
277 if (p_floc->filenm) {
278 if (!basename_filenames && strlen(p_floc->filenm)
279 && p_floc->filenm[0] != '/')
280 printf("%s/", starting_directory);
281 printf("%s:%lu", p_floc->filenm, p_floc->lineno);
282 } else {
283 if (!basename_filenames)
284 printf("%s/", starting_directory);
285 printf("??:%lu", p_floc->lineno);
286 }
287}

289/*! Show a command before executing it. */
290extern debug_return_t
291print_child_cmd (child_t *p_child, target_stack_node_t *p)
292{
293 debug_return_t rc = continue_execution;
294
295
296 if (!p_child) return continue_execution;
297
298 if (i_debugger_stepping || p_child->file->tracing) {
299 debug_enter_reason_t reason = DEBUG_STEP_HIT;
300 if (i_debugger_stepping)
301 reason = DEBUG_STEP_HIT;
302 else if (p_child->file->tracing & BRK_BEFORE_PREREQ)
303 reason = DEBUG_BRKPT_BEFORE_PREREQ;
304 else if (p_child->file->tracing & BRK_BEFORE_PREREQ)
305 reason = DEBUG_BRKPT_AFTER_PREREQ;
306
307 rc=enter_debugger(p, p_child->file, 0, reason);
308 }
309
310 return rc;
311}

313void
314print_target_stack_entry (const file_t *p_target, int i, int i_pos)
315{
316 gmk_floc floc;
317 const char *psz_target_name =
318 (p_target && p_target->name) ? p_target->name : "(null)";
319
320 /* If we don't have a line recorded for the target,
321 but we do have one for the commands it runs,
322 use that.
323 */
324 if (p_target->floc.filenm) {
325 memcpy(&floc, &(p_target->floc), sizeof(gmk_floc));
326 } else if (p_target->cmds) {
327 memcpy(&floc, &(p_target->cmds->fileinfo.filenm), sizeof(gmk_floc));
328 /* HACK: is it okay to assume that the target is on the line
329 before the first command? Or should we list the line
330 that the command starts on - so we know we've faked the location?
331 */
332 floc.lineno--;
333 } else {
334 floc.filenm = NULL;
335 }
336
337 if (floc.filenm) {
338 if (i_pos != -1) {
339 printf("%s", (i == i_pos) ? "=>" : " ");
340 }
341 printf ("#%d %s at ", i, psz_target_name);
342 print_floc_prefix(&floc);
343 } else {
344 if (i_pos != -1) {
345 printf("%s", (i == i_pos) ? "=>" : " ");
346 }
347 if (p_target->phony)
348 printf ("#%d %s (.PHONY target)", i, psz_target_name);
349 else
350 printf ("#%d %s at ??", i, psz_target_name);
351
352 }
353 printf ("\n");
354}

357/*! Display the target stack. i_pos is the position we are currently.
358 i_max is the maximum number of entries to show.
359 */
360extern void
361print_target_stack (target_stack_node_t *p, int i_pos, int i_max)
362{
363 int i=0;
364 for ( ; p && i < i_max ;
365 i++, p = p->p_parent ) {
366 print_target_stack_entry (p->p_target, i, i_pos);
367 }
368}

370/*! Display the Makefile read stack. i_pos is the position we are currently.
371 i_max is the maximum number of entries to show. */
372extern void
373print_floc_stack (int i_pos, int i_max)
374{
375 int i=0;
376 floc_stack_node_t *p;
377 printf("\n");
378 for ( p=p_stack_floc_top; p && i < i_max ;
379 i++, p = p->p_parent ) {
380 if (i_pos != -1) {
381 printf("%s", (i == i_pos) ? "=>" : " ");
382 }
383 printf ("#%d ", i);
384 if (p->p_floc->filenm) {
385 print_floc_prefix(p->p_floc);
386 }
387 printf ("\n");
388 }
389}

391/*! Print the file information. */
392void print_file (file_t *p_file)
393{
394 char buf[FILE_TIMESTAMP_PRINT_LEN_BOUND + 1];
395 printf("File %s:\n", p_file->name);
396 file_timestamp_sprintf (buf, p_file->last_mtime);
397 printf("\tLast modified: %s\n", buf);
398 if (p_file->mtime_before_update != p_file->last_mtime) {
399 file_timestamp_sprintf (buf, p_file->mtime_before_update);
400 printf("\tBefore update: %s\n", buf);
401 }
402 printf("\tNumber of lines: %u\n", p_file->nlines);
403}

405/*! Print the list makefiles read by read_makefiles(). */
406bool print_read_makefiles(const char *psz_filename)
407{
408 goaldep_t *p_dep;
409 if (!read_makefiles) return false;
410 if (NULL == psz_filename) {
411 for (p_dep = read_makefiles; p_dep; p_dep = p_dep->next) {
412 if (p_dep->file) {
413 print_file(p_dep->file);
414 }
415 }
416 return true;
417 } else {
418 for (p_dep = read_makefiles; p_dep; p_dep = p_dep->next) {
419 if (p_dep->file && 0 == strcmp(p_dep->file->name, psz_filename)) {
420 print_file(p_dep->file);
421 return true;
422 }
423 }
424 }
425 return false;
426}

428/*! Print the command line used to invoke Make. */
429void print_cmdline (void)
430{
431 unsigned int i;
432 printf(_("Command-line invocation:"));
433 printf("\n\t\"");
434 if (global_argv[1]) {
435 printf("%s", argv0);
436 for (i = 1; global_argv[i]; i++) {
437 printf(" %s", global_argv[i]);
438 }
439 printf("\"");
440 } else {
441 printf(_(" none"));
442 }
443 printf("\n");
444}
