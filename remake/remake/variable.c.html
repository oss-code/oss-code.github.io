44/*!
45 Return a string describing origin.
46 */
47const char *
48origin2str(variable_origin_t origin)
49{
50 switch (origin)
51 {
52 case o_default:
53 return _("default");
54 break;
55 case o_env:
56 return _("environment");
57 break;
58 case o_file:
59 return _("makefile");
60 break;
61 case o_env_override:
62 return _("environment under -e");
63 break;
64 case o_command:
65 return _("command line");
66 break;
67 case o_override:
68 return _("`override' directive");
69 break;
70 case o_automatic:
71 return _("automatic");
72 break;
73 case o_debugger:
74 return _("debugger");
75 break;
76 case o_invalid:
77 default:
78 return _("invalid");
79 }
80}

82/* Create a new pattern-specific variable struct. The new variable is
83 inserted into the PATTERN_VARS list in the shortest patterns first
84 order to support the shortest stem matching (the variables are
85 matched in the reverse order so the ones with the longest pattern
86 will be considered first). Variables with the same pattern length
87 are inserted in the definition order. */
88
89struct pattern_var *
90create_pattern_var (const char *target, const char *suffix)
91{
92 register unsigned int len = strlen (target);
93 register struct pattern_var *p = xmalloc (sizeof (struct pattern_var));
94
95 if (pattern_vars != 0)
96 {
97 if (len < 256 && last_pattern_vars[len] != 0)
98 {
99 p->next = last_pattern_vars[len]->next;
100 last_pattern_vars[len]->next = p;
101 }
102 else
103 {
104 /* Find the position where we can insert this variable. */
105 register struct pattern_var **v;
106
107 for (v = &pattern_vars; ; v = &(*v)->next)
108 {
109 /* Insert at the end of the pack so that patterns with the
110 same length appear in the order they were defined .*/
111
112 if (*v == 0 || (*v)->len > len)
113 {
114 p->next = *v;
115 *v = p;
116 break;
117 }
118 }
119 }
120 }
121 else
122 {
123 pattern_vars = p;
124 p->next = 0;
125 }
126
127 p->target = target;
128 p->len = len;
129 p->suffix = suffix + 1;
130
131 if (len < 256)
132 last_pattern_vars[len] = p;
133
134 return p;
135}
136
137/* Look up a target in the pattern-specific variable list. */
138
139static struct pattern_var *
140lookup_pattern_var (struct pattern_var *start, const char *target)
141{
142 struct pattern_var *p;
143 unsigned int targlen = strlen (target);
144
145 for (p = start ? start->next : pattern_vars; p != 0; p = p->next)
146 {
147 const char *stem;
148 unsigned int stemlen;
149
150 if (p->len > targlen)
151 /* It can't possibly match. */
152 continue;
153
154 /* From the lengths of the filename and the pattern parts,
155 find the stem: the part of the filename that matches the %. */
156 stem = target + (p->suffix - p->target - 1);
157 stemlen = targlen - p->len + 1;
158
159 /* Compare the text in the pattern before the stem, if any. */
160 if (stem > target && !strneq (p->target, target, stem - target))
161 continue;
162
163 /* Compare the text in the pattern after the stem, if any.
164 We could test simply using streq, but this way we compare the
165 first two characters immediately. This saves time in the very
166 common case where the first character matches because it is a
167 period. */
168 if (*p->suffix == stem[stemlen]
169 && (*p->suffix == '\0' || streq (&p->suffix[1], &stem[stemlen+1])))
170 break;
171 }
172
173 return p;
174}

176/* Hash table of all global variable definitions. */
177
178static unsigned long
179variable_hash_1 (const void *keyv)
180{
181 struct variable const *key = (struct variable const *) keyv;
182 return_STRING_N_HASH_1 (key->name, key->length);
183}
184
185static unsigned long
186variable_hash_2 (const void *keyv)
187{
188 struct variable const *key = (struct variable const *) keyv;
189 return_STRING_N_HASH_2 (key->name, key->length);
190}
191
192static int
193variable_hash_cmp (const void *xv, const void *yv)
194{
195 struct variable const *x = (struct variable const *) xv;
196 struct variable const *y = (struct variable const *) yv;
197 int result = x->length - y->length;
198 if (result)
199 return result;
200 return_STRING_N_COMPARE (x->name, y->name, x->length);
201}

227/* Define variable named NAME with value VALUE in SET. VALUE is copied.
228 LENGTH is the length of NAME, which does not need to be null-terminated.
229 ORIGIN specifies the origin of the variable (makefile, command line
230 or environment).
231 If RECURSIVE is nonzero a flag is set in the variable saying
232 that it should be recursively re-expanded. */
233
234struct variable *
235define_variable_in_set (const char *name, unsigned int length,
236 const char *value, enum variable_origin origin,
237 int recursive, struct variable_set *set,
238 const gmk_floc *flocp)
239{
240 struct variable *v;
241 struct variable **var_slot;
242 struct variable var_key;
243
244 if (set == NULL)
245 set = &global_variable_set;
246
247 var_key.name = (char *) name;
248 var_key.length = length;
249 var_slot = (struct variable **) hash_find_slot (&set->table, &var_key);
250 v = *var_slot;
251
252#ifdef VMS
253 /* VMS does not populate envp[] with DCL symbols and logical names which
254 historically are mapped to environent variables.
255 If the variable is not yet defined, then we need to check if getenv()
256 can find it. Do not do this for origin == o_env to avoid infinte
257 recursion */
258 if (HASH_VACANT (v) && (origin != o_env))
259 {
260 struct variable * vms_variable;
261 char * vname = alloca (length + 1);
262 char * vvalue;
263
264 strncpy (vname, name, length);
265 vvalue = getenv(vname);
266
267 /* Values starting with '$' are probably foreign commands.
268 We want to treat them as Shell aliases and not look them up here */
269 if ((vvalue != NULL) && (vvalue[0] != '$'))
270 {
271 vms_variable = lookup_variable(name, length);
272 /* Refresh the slot */
273 var_slot = (struct variable **) hash_find_slot (&set->table,
274 &var_key);
275 v = *var_slot;
276 }
277 }
278#endif
279
280 if (env_overrides && origin == o_env)
281 origin = o_env_override;
282
283 if (! HASH_VACANT (v))
284 {
285 if (env_overrides && v->origin == o_env)
286 /* V came from in the environment. Since it was defined
287 before the switches were parsed, it wasn't affected by -e. */
288 v->origin = o_env_override;
289
290 /* A variable of this name is already defined.
291 If the old definition is from a stronger source
292 than this one, don't redefine it. */
293 if ((int) origin >= (int) v->origin)
294 {
295 free (v->value);
296 v->value = xstrdup (value);
297 if (flocp != 0)
298 v->fileinfo = *flocp;
299 else
300 v->fileinfo.filenm = 0;
301 v->origin = origin;
302 v->recursive = recursive;
303 }
304 return v;
305 }
306
307 /* Create a new variable definition and add it to the hash table. */
308
309 v = xmalloc (sizeof (struct variable));
310 v->name = xstrndup (name, length);
311 v->length = length;
312 hash_insert_at (&set->table, v, var_slot);
313 if (set == &global_variable_set)
314 ++variable_changenum;
315
316 v->value = xstrdup (value);
317 if (flocp != 0)
318 v->fileinfo = *flocp;
319 else
320 v->fileinfo.filenm = 0;
321 v->origin = origin;
322 v->recursive = recursive;
323 v->special = 0;
324 v->expanding = 0;
325 v->exp_count = 0;
326 v->per_target = 0;
327 v->append = 0;
328 v->private_var = 0;
329 v->export = v_default;
330
331 v->exportable = 1;
332 if (*name != '_' && (*name < 'A' || *name > 'Z')
333 && (*name < 'a' || *name > 'z'))
334 v->exportable = 0;
335 else
336 {
337 for (++name; *name != '\0'; ++name)
338 if (*name != '_' && (*name < 'a' || *name > 'z')
339 && (*name < 'A' || *name > 'Z') && !ISDIGIT(*name))
340 break;
341
342 if (*name != '\0')
343 v->exportable = 0;
344 }
345
346 return v;
347}

371void
372undefine_variable_in_set (const char *name, unsigned int length,
373 enum variable_origin origin,
374 struct variable_set *set)
375{
376 struct variable *v;
377 struct variable **var_slot;
378 struct variable var_key;
379
380 if (set == NULL)
381 set = &global_variable_set;
382
383 var_key.name = (char *) name;
384 var_key.length = length;
385 var_slot = (struct variable **) hash_find_slot (&set->table, &var_key);
386
387 if (env_overrides && origin == o_env)
388 origin = o_env_override;
389
390 v = *var_slot;
391 if (! HASH_VACANT (v))
392 {
393 if (env_overrides && v->origin == o_env)
394 /* V came from in the environment. Since it was defined
395 before the switches were parsed, it wasn't affected by -e. */
396 v->origin = o_env_override;
397
398 /* Undefine only if this undefinition is from an equal or stronger
399 source than the variable definition. */
400 if ((int) origin >= (int) v->origin)
401 {
402 hash_delete_at (&set->table, var_slot);
403 free_variable_name_and_value (v);
404 free (v);
405 if (set == &global_variable_set)
406 ++variable_changenum;
407 }
408 }
409}

411/* If the variable passed in is "special", handle its special nature.
412 Currently there are two such variables, both used for introspection:
413 .VARIABLES expands to a list of all the variables defined in this instance
414 of make.
415 .TARGETS expands to a list of all the targets defined in this
416 instance of make.
417 Returns the variable reference passed in. */
418
419#define EXPANSION_INCREMENT(_l) ((((_l) / 500) + 1) * 500)
420
421static struct variable *
422lookup_special_var (struct variable *var)
423{
424 static unsigned long last_changenum = 0;
425
426
427 /* This one actually turns out to be very hard, due to the way the parser
428 records targets. The way it works is that target information is collected
429 internally until make knows the target is completely specified. It unitl
430 it sees that some new construct (a new target or variable) is defined that
431 it knows the previous one is done. In short, this means that if you do
432 this:
433
434 all:
435
436 TARGS := $(.TARGETS)
437
438 then $(TARGS) won't contain "all", because it's not until after the
439 variable is created that the previous target is completed.
440
441 Changing this would be a major pain. I think a less complex way to do it
442 would be to pre-define the target files as soon as the first line is
443 parsed, then come back and do the rest of the definition as now. That
444 would allow $(.TARGETS) to be correct without a major change to the way
445 the parser works.
446
447 if (streq (var->name, ".TARGETS"))
448 var->value = build_target_list (var->value);
449 else
450 */
451
452 if (variable_changenum != last_changenum && streq (var->name, ".VARIABLES"))
453 {
454 unsigned long max = EXPANSION_INCREMENT (strlen (var->value));
455 unsigned long len;
456 char *p;
457 struct variable **vp = (struct variable **) global_variable_set.table.ht_vec;
458 struct variable **end = &vp[global_variable_set.table.ht_size];
459
460 /* Make sure we have at least MAX bytes in the allocated buffer. */
461 var->value = xrealloc (var->value, max);
462
463 /* Walk through the hash of variables, constructing a list of names. */
464 p = var->value;
465 len = 0;
466 for (; vp < end; ++vp)
467 if (!HASH_VACANT (*vp))
468 {
469 struct variable *v = *vp;
470 int l = v->length;
471
472 len += l + 1;
473 if (len > max)
474 {
475 unsigned long off = p - var->value;
476
477 max += EXPANSION_INCREMENT (l + 1);
478 var->value = xrealloc (var->value, max);
479 p = &var->value[off];
480 }
481
482 memcpy (p, v->name, l);
483 p += l;
484 *(p++) = ' ';
485 }
486 *(p-1) = '\0';
487
488 /* Remember the current variable change number. */
489 last_changenum = variable_changenum;
490 }
491
492 return var;
493}

496/* Lookup a variable whose name is a string starting at NAME
497 and with LENGTH chars. NAME need not be null-terminated.
498 Returns address of the 'struct variable' containing all info
499 on the variable, or nil if no such variable is defined. */
500
501struct variable *
502lookup_variable (const char *name, unsigned int length)
503{
504 const struct variable_set_list *setlist;
505 struct variable var_key;
506 int is_parent = 0;
507
508 var_key.name = (char *) name;
509 var_key.length = length;
510
511 for (setlist = current_variable_set_list;
512 setlist != 0; setlist = setlist->next)
513 {
514 const struct variable_set *set = setlist->set;
515 struct variable *v;
516
517 v = (struct variable *) hash_find_item ((struct hash_table *) &set->table, &var_key);
518 if (v && (!is_parent || !v->private_var))
519 return v->special ? lookup_special_var (v) : v;
520
521 is_parent |= setlist->next_is_parent;
522 }
523
524#ifdef VMS
525 /* VMS does not populate envp[] with DCL symbols and logical names which
526 historically are mapped to enviroment varables and returned by getenv() */
527 {
528 char *vname = alloca (length + 1);
529 char *value;
530 strncpy (vname, name, length);
531 vname[length] = 0;
532 value = getenv (vname);
533 if (value != 0)
534 {
535 char *sptr;
536 int scnt;
537
538 sptr = value;
539 scnt = 0;
540
541 while ((sptr = strchr (sptr, '$')))
542 {
543 scnt++;
544 sptr++;
545 }
546
547 if (scnt > 0)
548 {
549 char *nvalue;
550 char *nptr;
551
552 nvalue = alloca (strlen (value) + scnt + 1);
553 sptr = value;
554 nptr = nvalue;
555
556 while (*sptr)
557 {
558 if (*sptr == '$')
559 {
560 *nptr++ = '$';
561 *nptr++ = '$';
562 }
563 else
564 {
565 *nptr++ = *sptr;
566 }
567 sptr++;
568 }
569
570 *nptr = '\0';
571 return define_variable (vname, length, nvalue, o_env, 1);
572
573 }
574
575 return define_variable (vname, length, value, o_env, 1);
576 }
577 }
578#endif /* VMS */
579
580 return 0;
581}

600/* Initialize FILE's variable set list. If FILE already has a variable set
601 list, the topmost variable set is left intact, but the the rest of the
602 chain is replaced with FILE->parent's setlist. If FILE is a double-colon
603 rule, then we will use the "root" double-colon target's variable set as the
604 parent of FILE's variable set.
605
606 If we're READING a makefile, don't do the pattern variable search now,
607 since the pattern variable might not have been defined yet. */
608
609void
610initialize_file_variables (struct file *file, int reading)
611{
612 struct variable_set_list *l = file->variables;
613
614 if (l == 0)
615 {
616 l = (struct variable_set_list *)
617 xmalloc (sizeof (struct variable_set_list));
618 l->set = xmalloc (sizeof (struct variable_set));
619 hash_init (&l->set->table, PERFILE_VARIABLE_BUCKETS,
620 variable_hash_1, variable_hash_2, variable_hash_cmp);
621 file->variables = l;
622 }
623
624 /* If this is a double-colon, then our "parent" is the "root" target for
625 this double-colon rule. Since that rule has the same name, parent,
626 etc. we can just use its variables as the "next" for ours. */
627
628 if (file->double_colon && file->double_colon != file)
629 {
630 initialize_file_variables (file->double_colon, reading);
631 l->next = file->double_colon->variables;
632 l->next_is_parent = 0;
633 return;
634 }
635
636 if (file->parent == 0)
637 l->next = &global_setlist;
638 else
639 {
640 initialize_file_variables (file->parent, reading);
641 l->next = file->parent->variables;
642 }
643 l->next_is_parent = 1;
644
645 /* If we're not reading makefiles and we haven't looked yet, see if
646 we can find pattern variables for this target. */
647
648 if (!reading && !file->pat_searched)
649 {
650 struct pattern_var *p;
651
652 p = lookup_pattern_var (0, file->name);
653 if (p != 0)
654 {
655 struct variable_set_list *global = current_variable_set_list;
656
657 /* We found at least one. Set up a new variable set to accumulate
658 all the pattern variables that match this target. */
659
660 file->pat_variables = create_new_variable_set ();
661 current_variable_set_list = file->pat_variables;
662
663 do
664 {
665 /* We found one, so insert it into the set. */
666
667 struct variable *v;
668
669 if (p->variable.flavor == f_simple)
670 {
671 v = define_variable_loc (
672 p->variable.name, strlen (p->variable.name),
673 p->variable.value, p->variable.origin,
674 0, &p->variable.fileinfo);
675
676 v->flavor = f_simple;
677 }
678 else
679 {
680 v = do_variable_definition (
681 &p->variable.fileinfo, p->variable.name,
682 p->variable.value, p->variable.origin,
683 p->variable.flavor, 1);
684 }
685
686 /* Also mark it as a per-target and copy export status. */
687 v->per_target = p->variable.per_target;
688 v->export = p->variable.export;
689 v->private_var = p->variable.private_var;
690 }
691 while ((p = lookup_pattern_var (p, file->name)) != 0);
692
693 current_variable_set_list = global;
694 }
695 file->pat_searched = 1;
696 }
697
698 /* If we have a pattern variable match, set it up. */
699
700 if (file->pat_variables != 0)
701 {
702 file->pat_variables->next = l->next;
703 file->pat_variables->next_is_parent = l->next_is_parent;
704 l->next = file->pat_variables;
705 l->next_is_parent = 0;
706 }
707}

709/* Pop the top set off the current variable set list,
710 and free all its storage. */
711
712struct variable_set_list *
713create_new_variable_set (void)
714{
715 register struct variable_set_list *setlist;
716 register struct variable_set *set;
717
718 set = xmalloc (sizeof (struct variable_set));
719 hash_init (&set->table, SMALL_SCOPE_VARIABLE_BUCKETS,
720 variable_hash_1, variable_hash_2, variable_hash_cmp);
721
722 setlist = (struct variable_set_list *)
723 xmalloc (sizeof (struct variable_set_list));
724 setlist->set = set;
725 setlist->next = current_variable_set_list;
726 setlist->next_is_parent = 0;
727
728 return setlist;
729}

731/* Create a new variable set and push it on the current setlist.
732 If we're pushing a global scope (that is, the current scope is the global
733 scope) then we need to "push" it the other way: file variable sets point
734 directly to the global_setlist so we need to replace that with the new one.
735 */
736
737struct variable_set_list *
738push_new_variable_scope (void)
739{
740 current_variable_set_list = create_new_variable_set ();
741 if (current_variable_set_list->next == &global_setlist)
742 {
743 /* It was the global, so instead of new -> &global we want to replace
744 &global with the new one and have &global -> new, with current still
745 pointing to &global */
746 struct variable_set *set = current_variable_set_list->set;
747 current_variable_set_list->set = global_setlist.set;
748 global_setlist.set = set;
749 current_variable_set_list->next = global_setlist.next;
750 global_setlist.next = current_variable_set_list;
751 current_variable_set_list = &global_setlist;
752 }
753 return (current_variable_set_list);
754}
755
756void
757pop_variable_scope (void)
758{
759 struct variable_set_list *setlist;
760 struct variable_set *set;
761
762 /* Can't call this if there's no scope to pop! */
763 assert (current_variable_set_list->next != NULL);
764
765 if (current_variable_set_list != &global_setlist)
766 {
767 /* We're not pointing to the global setlist, so pop this one. */
768 setlist = current_variable_set_list;
769 set = setlist->set;
770 current_variable_set_list = setlist->next;
771 }
772 else
773 {
774 /* This set is the one in the global_setlist, but there is another global
775 set beyond that. We want to copy that set to global_setlist, then
776 delete what used to be in global_setlist. */
777 setlist = global_setlist.next;
778 set = global_setlist.set;
779 global_setlist.set = setlist->set;
780 global_setlist.next = setlist->next;
781 global_setlist.next_is_parent = setlist->next_is_parent;
782 }
783
784 /* Free the one we no longer need. */
785 free (setlist);
786 hash_map (&set->table, free_variable_name_and_value);
787 hash_free (&set->table, 1);
788 free (set);
789}

791/* Merge FROM_SET into TO_SET, freeing unused storage in FROM_SET. */

793static void
794merge_variable_sets (struct variable_set *to_set,
795 struct variable_set *from_set)
796{
797 struct variable **from_var_slot = (struct variable **) from_set->table.ht_vec;
798 struct variable **from_var_end = from_var_slot + from_set->table.ht_size;
799
800 int inc = to_set == &global_variable_set ? 1 : 0;
801
802 for ( ; from_var_slot < from_var_end; from_var_slot++)
803 if (! HASH_VACANT (*from_var_slot))
804 {
805 struct variable *from_var = *from_var_slot;
806 struct variable **to_var_slot
807 = (struct variable **) hash_find_slot (&to_set->table, *from_var_slot);
808 if (HASH_VACANT (*to_var_slot))
809 {
810 hash_insert_at (&to_set->table, from_var, to_var_slot);
811 variable_changenum += inc;
812 }
813 else
814 {
815 /* GKM FIXME: delete in from_set->table */
816 free (from_var->value);
817 free (from_var);
818 }
819 }

822/* Merge SETLIST1 into SETLIST0, freeing unused storage in SETLIST1. */
823
824void
825merge_variable_set_lists (struct variable_set_list **setlist0,
826 struct variable_set_list *setlist1)
827{
828 struct variable_set_list *to = *setlist0;
829 struct variable_set_list *last0 = 0;
830
831 /* If there's nothing to merge, stop now. */
832 if (!setlist1)
833 return;
834
835 /* This loop relies on the fact that all setlists terminate with the global
836 setlist (before NULL). If that's not true, arguably we SHOULD die. */
837 if (to)
838 while (setlist1 != &global_setlist && to != &global_setlist)
839 {
840 struct variable_set_list *from = setlist1;
841 setlist1 = setlist1->next;
842
843 merge_variable_sets (to->set, from->set);
844
845 last0 = to;
846 to = to->next;
847 }
848
849 if (setlist1 != &global_setlist)
850 {
851 if (last0 == 0)
852 *setlist0 = setlist1;
853 else
854 last0->next = setlist1;
855 }
856}

1015/* Create a new environment for FILE's commands.
1016 If FILE is nil, this is for the 'shell' function.
1017 The child's MAKELEVEL variable is incremented. */
1018
1019char **
1020target_environment (struct file *file)
1021{
1022 struct variable_set_list *set_list;
1023 register struct variable_set_list *s;
1024 struct hash_table table;
1025 struct variable **v_slot;
1026 struct variable **v_end;
1027 struct variable makelevel_key;
1028 char **result_0;
1029 char **result;
1030
1031 if (file == 0)
1032 set_list = current_variable_set_list;
1033 else
1034 set_list = file->variables;
1035
1036 hash_init (&table, VARIABLE_BUCKETS,
1037 variable_hash_1, variable_hash_2, variable_hash_cmp);
1038
1039 /* Run through all the variable sets in the list,
1040 accumulating variables in TABLE. */
1041 for (s = set_list; s != 0; s = s->next)
1042 {
1043 struct variable_set *set = s->set;
1044 v_slot = (struct variable **) set->table.ht_vec;
1045 v_end = v_slot + set->table.ht_size;
1046 for ( ; v_slot < v_end; v_slot++)
1047 if (! HASH_VACANT (*v_slot))
1048 {
1049 struct variable **new_slot;
1050 struct variable *v = *v_slot;
1051
1052 /* If this is a per-target variable and it hasn't been touched
1053 already then look up the global version and take its export
1054 value. */
1055 if (v->per_target && v->export == v_default)
1056 {
1057 struct variable *gv;
1058
1059 gv = lookup_variable_in_set (v->name, strlen (v->name),
1060 &global_variable_set);
1061 if (gv)
1062 v->export = gv->export;
1063 }
1064
1065 switch (v->export)
1066 {
1067 case v_default:
1068 if (v->origin == o_default || v->origin == o_automatic)
1069 /* Only export default variables by explicit request. */
1070 continue;
1071
1072 /* The variable doesn't have a name that can be exported. */
1073 if (! v->exportable)
1074 continue;
1075
1076 if (! export_all_variables
1077 && v->origin != o_command
1078 && v->origin != o_env && v->origin != o_env_override)
1079 continue;
1080 break;
1081
1082 case v_export:
1083 break;
1084
1085 case v_noexport:
1086 {
1087 /* If this is the SHELL variable and it's not exported,
1088 then add the value from our original environment, if
1089 the original environment defined a value for SHELL. */
1090 if (streq (v->name, "SHELL") && shell_var.value)
1091 {
1092 v = &shell_var;
1093 break;
1094 }
1095 continue;
1096 }
1097
1098 case v_ifset:
1099 if (v->origin == o_default)
1100 continue;
1101 break;
1102 }
1103
1104 new_slot = (struct variable **) hash_find_slot (&table, v);
1105 if (HASH_VACANT (*new_slot))
1106 hash_insert_at (&table, v, new_slot);
1107 }
1108 }
1109
1110 makelevel_key.name = (char *)MAKELEVEL_NAME;
1111 makelevel_key.length = MAKELEVEL_LENGTH;
1112 hash_delete (&table, &makelevel_key);
1113
1114 result = result_0 = xmalloc ((table.ht_fill + 2) * sizeof (char *));
1115
1116 v_slot = (struct variable **) table.ht_vec;
1117 v_end = v_slot + table.ht_size;
1118 for ( ; v_slot < v_end; v_slot++)
1119 if (! HASH_VACANT (*v_slot))
1120 {
1121 struct variable *v = *v_slot;
1122
1123 /* If V is recursively expanded and didn't come from the environment,
1124 expand its value. If it came from the environment, it should
1125 go back into the environment unchanged. */
1126 if (v->recursive
1127 && v->origin != o_env && v->origin != o_env_override)
1128 {
1129 char *value = recursively_expand_for_file (v, file);
1130#ifdef WINDOWS32
1131 if (strcmp (v->name, "Path") == 0 ||
1132 strcmp (v->name, "PATH") == 0)
1133 convert_Path_to_windows32 (value, ';');
1134#endif
1135 *result++ = xstrdup (concat (3, v->name, "=", value));
1136 free (value);
1137 }
1138 else
1139 {
1140#ifdef WINDOWS32
1141 if (strcmp (v->name, "Path") == 0 ||
1142 strcmp (v->name, "PATH") == 0)
1143 convert_Path_to_windows32 (v->value, ';');
1144#endif
1145 *result++ = xstrdup (concat (3, v->name, "=", v->value));
1146 }
1147 }
1148
1149 *result = xmalloc (100);
1150 sprintf (*result, "%s=%u", MAKELEVEL_NAME, makelevel + 1);
1151 *++result = 0;
1152
1153 hash_free (&table, 0);
1154
1155 return result_0;
1156}

1195/* Given a variable, a value, and a flavor, define the variable.
1196 See the try_variable_definition() function for details on the parameters. */
1197
1198struct variable *
1199do_variable_definition (const gmk_floc *flocp, const char *varname,
1200 const char *value, enum variable_origin origin,
1201 enum variable_flavor flavor, int target_var)
1202{
1203 const char *p;
1204 char *alloc_value = NULL;
1205 struct variable *v;
1206 int append = 0;
1207 int conditional = 0;
1208
1209 /* Calculate the variable's new value in VALUE. */
1210
1211 switch (flavor)
1212 {
1213 default:
1214 case f_bogus:
1215 /* Should not be possible. */
1216 abort ();
1217 case f_simple:
1218 /* A simple variable definition "var := value". Expand the value.
1219 We have to allocate memory since otherwise it'll clobber the
1220 variable buffer, and we may still need that if we're looking at a
1221 target-specific variable. */
1222 p = alloc_value = allocated_variable_expand (value);
1223 break;
1224 case f_shell:
1225 {
1226 /* A shell definition "var != value". Expand value, pass it to
1227 the shell, and store the result in recursively-expanded var. */
1228 char *q = allocated_variable_expand (value);
1229 p = alloc_value = shell_result (q);
1230 free (q);
1231 flavor = f_recursive;
1232 break;
1233 }
1234 case f_conditional:
1235 /* A conditional variable definition "var ?= value".
1236 The value is set IFF the variable is not defined yet. */
1237 v = lookup_variable (varname, strlen (varname));
1238 if (v)
1239 return v->special ? set_special_var (v) : v;
1240
1241 conditional = 1;
1242 flavor = f_recursive;
1243 /* FALLTHROUGH */
1244 case f_recursive:
1245 /* A recursive variable definition "var = value".
1246 The value is used verbatim. */
1247 p = value;
1248 break;
1249 case f_append:
1250 {
1251 /* If we have += but we're in a target variable context, we want to
1252 append only with other variables in the context of this target. */
1253 if (target_var)
1254 {
1255 append = 1;
1256 v = lookup_variable_in_set (varname, strlen (varname),
1257 current_variable_set_list->set);
1258
1259 /* Don't append from the global set if a previous non-appending
1260 target-specific variable definition exists. */
1261 if (v && !v->append)
1262 append = 0;
1263 }
1264 else
1265 v = lookup_variable (varname, strlen (varname));
1266
1267 if (v == 0)
1268 {
1269 /* There was no old value.
1270 This becomes a normal recursive definition. */
1271 p = value;
1272 flavor = f_recursive;
1273 }
1274 else
1275 {
1276 /* Paste the old and new values together in VALUE. */
1277
1278 unsigned int oldlen, vallen;
1279 const char *val;
1280 char *tp = NULL;
1281
1282 val = value;
1283 if (v->recursive)
1284 /* The previous definition of the variable was recursive.
1285 The new value is the unexpanded old and new values. */
1286 flavor = f_recursive;
1287 else
1288 /* The previous definition of the variable was simple.
1289 The new value comes from the old value, which was expanded
1290 when it was set; and from the expanded new value. Allocate
1291 memory for the expansion as we may still need the rest of the
1292 buffer if we're looking at a target-specific variable. */
1293 val = tp = allocated_variable_expand (val);
1294
1295 oldlen = strlen (v->value);
1296 vallen = strlen (val);
1297 p = alloc_value = xmalloc (oldlen + 1 + vallen + 1);
1298 memcpy (alloc_value, v->value, oldlen);
1299 alloc_value[oldlen] = ' ';
1300 memcpy (&alloc_value[oldlen + 1], val, vallen + 1);
1301
1302 free (tp);
1303 }
1304 }
1305 }
1306
1307#ifdef __MSDOS__
1308 /* Many Unix Makefiles include a line saying "SHELL=/bin/sh", but
1309 non-Unix systems don't conform to this default configuration (in
1310 fact, most of them don't even have '/bin'). On the other hand,
1311 $SHELL in the environment, if set, points to the real pathname of
1312 the shell.
1313 Therefore, we generally won't let lines like "SHELL=/bin/sh" from
1314 the Makefile override $SHELL from the environment. But first, we
1315 look for the basename of the shell in the directory where SHELL=
1316 points, and along the $PATH; if it is found in any of these places,
1317 we define $SHELL to be the actual pathname of the shell. Thus, if
1318 you have bash.exe installed as d:/unix/bash.exe, and d:/unix is on
1319 your $PATH, then SHELL=/usr/local/bin/bash will have the effect of
1320 defining SHELL to be "d:/unix/bash.exe". */
1321 if ((origin == o_file || origin == o_override)
1322 && strcmp (varname, "SHELL") == 0)
1323 {
1324 PATH_VAR (shellpath);
1325 extern char * __dosexec_find_on_path (const char *, char *[], char *);
1326
1327 /* See if we can find "/bin/sh.exe", "/bin/sh.com", etc. */
1328 if (__dosexec_find_on_path (p, NULL, shellpath))
1329 {
1330 char *tp;
1331
1332 for (tp = shellpath; *tp; tp++)
1333 if (*tp == '\\')
1334 *tp = '/';
1335
1336 v = define_variable_loc (varname, strlen (varname),
1337 shellpath, origin, flavor == f_recursive,
1338 flocp);
1339 }
1340 else
1341 {
1342 const char *shellbase, *bslash;
1343 struct variable *pathv = lookup_variable ("PATH", 4);
1344 char *path_string;
1345 char *fake_env[2];
1346 size_t pathlen = 0;
1347
1348 shellbase = strrchr (p, '/');
1349 bslash = strrchr (p, '\\');
1350 if (!shellbase || bslash > shellbase)
1351 shellbase = bslash;
1352 if (!shellbase && p[1] == ':')
1353 shellbase = p + 1;
1354 if (shellbase)
1355 shellbase++;
1356 else
1357 shellbase = p;
1358
1359 /* Search for the basename of the shell (with standard
1360 executable extensions) along the $PATH. */
1361 if (pathv)
1362 pathlen = strlen (pathv->value);
1363 path_string = xmalloc (5 + pathlen + 2 + 1);
1364 /* On MSDOS, current directory is considered as part of $PATH. */
1365 sprintf (path_string, "PATH=.;%s", pathv ? pathv->value : "");
1366 fake_env[0] = path_string;
1367 fake_env[1] = 0;
1368 if (__dosexec_find_on_path (shellbase, fake_env, shellpath))
1369 {
1370 char *tp;
1371
1372 for (tp = shellpath; *tp; tp++)
1373 if (*tp == '\\')
1374 *tp = '/';
1375
1376 v = define_variable_loc (varname, strlen (varname),
1377 shellpath, origin,
1378 flavor == f_recursive, flocp);
1379 }
1380 else
1381 v = lookup_variable (varname, strlen (varname));
1382
1383 free (path_string);
1384 }
1385 }
1386 else
1387#endif /* __MSDOS__ */
1388#ifdef WINDOWS32
1389 if ((origin == o_file || origin == o_override || origin == o_command)
1390 && streq (varname, "SHELL"))
1391 {
1392 extern const char *default_shell;
1393
1394 /* Call shell locator function. If it returns TRUE, then
1395 set no_default_sh_exe to indicate sh was found and
1396 set new value for SHELL variable. */
1397
1398 if (find_and_set_default_shell (p))
1399 {
1400 v = define_variable_in_set (varname, strlen (varname), default_shell,
1401 origin, flavor == f_recursive,
1402 (target_var
1403 ? current_variable_set_list->set
1404 : NULL),
1405 flocp);
1406 no_default_sh_exe = 0;
1407 }
1408 else
1409 {
1410 char *tp = alloc_value;
1411
1412 alloc_value = allocated_variable_expand (p);
1413
1414 if (find_and_set_default_shell (alloc_value))
1415 {
1416 v = define_variable_in_set (varname, strlen (varname), p,
1417 origin, flavor == f_recursive,
1418 (target_var
1419 ? current_variable_set_list->set
1420 : NULL),
1421 flocp);
1422 no_default_sh_exe = 0;
1423 }
1424 else
1425 v = lookup_variable (varname, strlen (varname));
1426
1427 free (tp);
1428 }
1429 }
1430 else
1431#endif
1432
1433 /* If we are defining variables inside an $(eval ...), we might have a
1434 different variable context pushed, not the global context (maybe we're
1435 inside a $(call ...) or something. Since this function is only ever
1436 invoked in places where we want to define globally visible variables,
1437 make sure we define this variable in the global set. */
1438
1439 v = define_variable_in_set (varname, strlen (varname), p,
1440 origin, flavor == f_recursive,
1441 (target_var
1442 ? current_variable_set_list->set : NULL),
1443 flocp);
1444 v->append = append;
1445 v->conditional = conditional;
1446
1447 free (alloc_value);
1448
1449 return v->special ? set_special_var (v) : v;
1450}

1452/* Parse P (a null-terminated string) as a variable definition.
1453
1454 If it is not a variable definition, return NULL and the contents of *VAR
1455 are undefined, except NAME is set to the first non-space character or NIL.
1456
1457 If it is a variable definition, return a pointer to the char after the
1458 assignment token and set the following fields (only) of *VAR:
1459 name : name of the variable (ALWAYS SET) (NOT NUL-TERMINATED!)
1460 length : length of the variable name
1461 value : value of the variable (nul-terminated)
1462 flavor : flavor of the variable
1463 Other values in *VAR are unchanged.
1464 */
1465
1466char *
1467parse_variable_definition (const char *p, struct variable *var)
1468{
1469 int wspace = 0;
1470 const char *e = NULL;
1471
1472 NEXT_TOKEN (p);
1473 var->name = (char *)p;
1474 var->length = 0;
1475
1476 while (1)
1477 {
1478 int c = *p++;
1479
1480 /* If we find a comment or EOS, it's not a variable definition. */
1481 if (STOP_SET (c, MAP_COMMENT|MAP_NUL))
1482 return NULL;
1483
1484 if (c == '$')
1485 {
1486 /* This begins a variable expansion reference. Make sure we don't
1487 treat chars inside the reference as assignment tokens. */
1488 char closeparen;
1489 unsigned int count;
1490
1491 c = *p++;
1492 if (c == '(')
1493 closeparen = ')';
1494 else if (c == '{')
1495 closeparen = '}';
1496 else if (c == '\0')
1497 return NULL;
1498 else
1499 /* '$$' or '$X'. Either way, nothing special to do here. */
1500 continue;
1501
1502 /* P now points past the opening paren or brace.
1503 Count parens or braces until it is matched. */
1504 for (count = 1; *p != '\0'; ++p)
1505 {
1506 if (*p == closeparen && --count == 0)
1507 {
1508 ++p;
1509 break;
1510 }
1511 if (*p == c)
1512 ++count;
1513 }
1514 continue;
1515 }
1516
1517 /* If we find whitespace skip it, and remember we found it. */
1518 if (ISBLANK (c))
1519 {
1520 wspace = 1;
1521 e = p - 1;
1522 NEXT_TOKEN (p);
1523 c = *p;
1524 if (c == '\0')
1525 return NULL;
1526 ++p;
1527 }
1528
1529
1530 if (c == '=')
1531 {
1532 var->flavor = f_recursive;
1533 if (! e)
1534 e = p - 1;
1535 break;
1536 }
1537
1538 /* Match assignment variants (:=, +=, ?=, !=) */
1539 if (*p == '=')
1540 {
1541 switch (c)
1542 {
1543 case ':':
1544 var->flavor = f_simple;
1545 break;
1546 case '+':
1547 var->flavor = f_append;
1548 break;
1549 case '?':
1550 var->flavor = f_conditional;
1551 break;
1552 case '!':
1553 var->flavor = f_shell;
1554 break;
1555 default:
1556 /* If we skipped whitespace, non-assignments means no var. */
1557 if (wspace)
1558 return NULL;
1559
1560 /* Might be assignment, or might be $= or #=. Check. */
1561 continue;
1562 }
1563 if (! e)
1564 e = p - 1;
1565 ++p;
1566 break;
1567 }
1568
1569 /* Check for POSIX ::= syntax */
1570 if (c == ':')
1571 {
1572 /* A colon other than :=/::= is not a variable defn. */
1573 if (*p != ':' || p[1] != '=')
1574 return NULL;
1575
1576 /* POSIX allows ::= to be the same as GNU make's := */
1577 var->flavor = f_simple;
1578 if (! e)
1579 e = p - 1;
1580 p += 2;
1581 break;
1582 }
1583
1584 /* If we skipped whitespace, non-assignments means no var. */
1585 if (wspace)
1586 return NULL;
1587 }
1588
1589 var->length = e - var->name;
1590 var->value = next_token (p);
1591 return (char *)p;
1592}
1593
1594/* Try to interpret LINE (a null-terminated string) as a variable definition.
1595
1596 If LINE was recognized as a variable definition, a pointer to its 'struct
1597 variable' is returned. If LINE is not a variable definition, NULL is
1598 returned. */
1599
1600struct variable *
1601assign_variable_definition (struct variable *v, const char *line)
1602{
1603 char *name;
1604
1605 if (!parse_variable_definition (line, v))
1606 return NULL;
1607
1608 /* Expand the name, so "$(foo)bar = baz" works. */
1609 name = alloca (v->length + 1);
1610 memcpy (name, v->name, v->length);
1611 name[v->length] = '\0';
1612 v->name = allocated_variable_expand (name);
1613
1614 if (v->name[0] == '\0')
1615 O (fatal, &v->fileinfo, _("empty variable name"));
1616
1617 return v;
1618}
1619
1620/* Try to interpret LINE (a null-terminated string) as a variable definition.
1621
1622 ORIGIN may be o_file, o_override, o_env, o_env_override,
1623 or o_command specifying that the variable definition comes
1624 from a makefile, an override directive, the environment with
1625 or without the -e switch, or the command line.
1626
1627 See the comments for assign_variable_definition().
1628
1629 If LINE was recognized as a variable definition, a pointer to its 'struct
1630 variable' is returned. If LINE is not a variable definition, NULL is
1631 returned. */
1632
1633struct variable *
1634try_variable_definition (const gmk_floc *flocp, const char *line,
1635 enum variable_origin origin, int target_var)
1636{
1637 struct variable v;
1638 struct variable *vp;
1639
1640 if (flocp != 0)
1641 v.fileinfo = *flocp;
1642 else
1643 v.fileinfo.filenm = 0;
1644
1645 if (!assign_variable_definition (&v, line))
1646 return 0;
1647
1648 vp = do_variable_definition (flocp, v.name, v.value,
1649 origin, v.flavor, target_var);
1650
1651 free (v.name);
1652
1653 return vp;
1654}

1656/* Print information for variable V, prefixing it with PREFIX. */
1657
1658static void
1659print_variable (const void *item, void *arg)
1660{
1661 const struct variable *v = item;
1662 const char *prefix = arg;
1663 const char *origin;
1664
1665 switch (v->origin)
1666 {
1667 case o_automatic:
1668 origin = _("automatic");
1669 break;
1670 case o_default:
1671 origin = _("default");
1672 break;
1673 case o_env:
1674 origin = _("environment");
1675 break;
1676 case o_file:
1677 origin = _("makefile");
1678 break;
1679 case o_env_override:
1680 origin = _("environment under -e");
1681 break;
1682 case o_command:
1683 origin = _("command line");
1684 break;
1685 case o_override:
1686 origin = _("'override' directive");
1687 break;
1688 case o_invalid:
1689 default:
1690 abort ();
1691 }
1692 fputs ("# ", stdout);
1693 fputs (origin, stdout);
1694 if (v->private_var)
1695 fputs (" private", stdout);
1696 if (v->fileinfo.filenm)
1697 printf (_(" (from '%s', line %lu)"),
1698 v->fileinfo.filenm, v->fileinfo.lineno + v->fileinfo.offset);
1699 putchar ('\n');
1700 fputs (prefix, stdout);
1701
1702 /* Is this a 'define'? */
1703 if (v->recursive && strchr (v->value, '\n') != 0)
1704 printf ("define %s\n%s\nendef\n", v->name, v->value);
1705 else
1706 {
1707 char *p;
1708
1709 printf ("%s %s= ", v->name, v->recursive ? v->append ? "+" : "" : ":");
1710
1711 /* Check if the value is just whitespace. */
1712 p = next_token (v->value);
1713 if (p != v->value && *p == '\0')
1714 /* All whitespace. */
1715 printf ("$(subst ,,%s)", v->value);
1716 else if (v->recursive)
1717 fputs (v->value, stdout);
1718 else
1719 /* Double up dollar signs. */
1720 for (p = v->value; *p != '\0'; ++p)
1721 {
1722 if (*p == '$')
1723 putchar ('$');
1724 putchar (*p);
1725 }
1726 putchar ('\n');
1727 }
1728}
