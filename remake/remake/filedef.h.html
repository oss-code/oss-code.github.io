19 that the makefile says how to make.
20 All of these are chained together through 'next'. */

29struct file
30 {
31 const char *name;
32 const char *hname; /* Hashed filename */
33 const char *vpath; /* VPATH/vpath pathname */
34 gmk_floc floc; /* location in Makefile - for tracing */
35 uint64_t elapsed_time; /* Runtime in 100microsec to build target */
36 unsigned int nlines; /* Number of lines in file - for debugging. */
37
38 const char *description; /* Description of target taken from comment.
39 Part after #: */
40 struct dep *deps; /* all dependencies, including duplicates */
41 struct commands *cmds; /* Commands to execute for this target. */
42 const char *stem; /* Implicit stem, if an implicit
43 rule has been used */
44 struct dep *also_make; /* Targets that are made by making this. */
45 struct file *prev; /* Previous entry for same file name;
46 used when there are multiple double-colon
47 entries for the same file. */
48 struct file *last; /* Last entry for the same file name. */
49
50 /* File that this file was renamed to. After any time that a
51 file could be renamed, call 'check_renamed' (below). */
52 struct file *renamed;
53
54 /* List of variable sets used for this file. */
55 struct variable_set_list *variables;
56
57 /* Pattern-specific variable reference for this target, or null if there
58 isn't one. Also see the pat_searched flag, below. */
59 struct variable_set_list *pat_variables;
60
61 /* Immediate dependent that caused this target to be remade,
62 or nil if there isn't one. */
63 struct file *parent;
64
65 /* For a double-colon entry, this is the first double-colon entry for
66 the same file. Otherwise this is null. */
67 struct file *double_colon;
68
69 FILE_TIMESTAMP last_mtime; /* File's modtime, if already known. */
70 FILE_TIMESTAMP mtime_before_update; /* File's modtime before any updating
71 has been performed. */
72 unsigned int considered; /* equal to 'considered' if file has been
73 considered on current scan of goal chain */
74 int command_flags; /* Flags OR'd in for cmds; see commands.h. */
75 enum update_status /* Status of the last attempt to update. */
76 {
77 us_success = 0, /* Successfully updated. Must be 0! */
78 us_none, /* No attempt to update has been made. */
79 us_question, /* Needs to be updated (-q is is set). */
80 us_failed /* Update failed. */
81 } update_status ENUM_BITFIELD (2);
82 enum cmd_state /* State of the commands. */
83 {
84 cs_not_started = 0, /* Not yet started. Must be 0! */
85 cs_deps_running, /* Dep commands running. */
86 cs_running, /* Commands running. */
87 cs_finished /* Commands finished. */
88 } command_state ENUM_BITFIELD (2);
89
90 breakpoint_mask_t tracing; /* breakpoint status of target. */
91
92 unsigned int builtin:1; /* True if the file is a builtin rule. */
93 unsigned int precious:1; /* Non-0 means don't delete file on quit */
94 unsigned int loaded:1; /* True if the file is a loaded object. */
95 unsigned int low_resolution_time:1; /* Nonzero if this file's time stamp
96 has only one-second resolution. */
97 unsigned int tried_implicit:1; /* Nonzero if have searched
98 for implicit rule for making
99 this file; don't search again. */
100 unsigned int updating:1; /* Nonzero while updating deps of this file */
101 unsigned int updated:1; /* Nonzero if this file has been remade. */
102 unsigned int is_target:1; /* Nonzero if file is described as target. */
103 unsigned int cmd_target:1; /* Nonzero if file was given on cmd line. */
104 unsigned int phony:1; /* Nonzero if this is a phony file
105 i.e., a prerequisite of .PHONY. */
106 unsigned int intermediate:1;/* Nonzero if this is an intermediate file. */
107 unsigned int secondary:1; /* Nonzero means remove_intermediates should
108 not delete it. */
109 unsigned int dontcare:1; /* Nonzero if no complaint is to be made if
110 this target cannot be remade. */
111 unsigned int ignore_vpath:1;/* Nonzero if we threw out VPATH name. */
112 unsigned int pat_searched:1;/* Nonzero if we already searched for
113 pattern-specific variables. */
114 unsigned int no_diag:1; /* True if the file failed to update and no
115 diagnostics has been issued (dontcare). */
116 unsigned int file_profiled:1; /* True if --profile has been set and
117 we have emitted a callgrind file line. */
118 };

150/* If FILE_TIMESTAMP is 64 bits (or more), use nanosecond resolution.
151 (Multiply by 2**30 instead of by 10**9 to save time at the cost of
152 slightly decreasing the number of available timestamps.) With
153 64-bit FILE_TIMESTAMP, this stops working on 2514-05-30 01:53:04
154 UTC, but by then uintmax_t should be larger than 64 bits. */
155#define FILE_TIMESTAMPS_PER_S (FILE_TIMESTAMP_HI_RES ? 1000000000 : 1)
156#define FILE_TIMESTAMP_LO_BITS (FILE_TIMESTAMP_HI_RES ? 30 : 0)
157
158#define FILE_TIMESTAMP_S(ts) (((ts) - ORDINARY_MTIME_MIN) \
159 >> FILE_TIMESTAMP_LO_BITS)
160#define FILE_TIMESTAMP_NS(ts) ((int) (((ts) - ORDINARY_MTIME_MIN) \
161 & ((1 << FILE_TIMESTAMP_LO_BITS) - 1)))
162
163/* Upper bound on length of string "YYYY-MM-DD HH:MM:SS.NNNNNNNNN"
164 representing a file timestamp. The upper bound is not necessarily 29,
165 since the year might be less than -999 or greater than 9999.
166
167 Subtract one for the sign bit if in case file timestamps can be negative;
168 subtract FLOOR_LOG2_SECONDS_PER_YEAR to yield an upper bound on how many
169 file timestamp bits might affect the year;
170 302 / 1000 is log10 (2) rounded up;
171 add one for integer division truncation;
172 add one more for a minus sign if file timestamps can be negative;
173 add 4 to allow for any 4-digit epoch year (e.g. 1970);
174 add 25 to allow for "-MM-DD HH:MM:SS.NNNNNNNNN". */
175#define FLOOR_LOG2_SECONDS_PER_YEAR 24
176#define FILE_TIMESTAMP_PRINT_LEN_BOUND \
177 (((sizeof (FILE_TIMESTAMP) * CHAR_BIT - 1 - FLOOR_LOG2_SECONDS_PER_YEAR) \
178 * 302 / 1000) \
179 + 1 + 1 + 4 + 25)

198 The value is NONEXISTENT_MTIME if the file does not exist. */
199#define file_mtime(f) file_mtime_1 ((f), 1)
200/* Return the mtime of file F (a struct file *), caching it.
201 Don't search using vpath for the file--if it doesn't actually exist,
202 we don't find it.
203 The value is NONEXISTENT_MTIME if the file does not exist. */
204#define file_mtime_no_search(f) file_mtime_1 ((f), 0)

209/* Special timestamp values. */
210
211/* The file's timestamp is not yet known. */
212#define UNKNOWN_MTIME 0
213
214/* The file does not exist. */
215#define NONEXISTENT_MTIME 1
216
217/* The file does not exist, and we assume that it is older than any
218 actual file. */
219#define OLD_MTIME 2
220
221/* The smallest and largest ordinary timestamps. */
222#define ORDINARY_MTIME_MIN (OLD_MTIME + 1)
223#define ORDINARY_MTIME_MAX ((FILE_TIMESTAMP_S (NEW_MTIME) \
224 << FILE_TIMESTAMP_LO_BITS) \
225 + ORDINARY_MTIME_MIN + FILE_TIMESTAMPS_PER_S - 1)
226
227/* Modtime value to use for 'infinitely new'. We used to get the current time
228 from the system and use that whenever we wanted 'new'. But that causes
229 trouble when the machine running make and the machine holding a file have
230 different ideas about what time it is; and can also lose for 'force'
231 targets, which need to be considered newer than anything that depends on
232 them, even if said dependents' modtimes are in the future. */
233#define NEW_MTIME INTEGER_TYPE_MAXIMUM (FILE_TIMESTAMP)
