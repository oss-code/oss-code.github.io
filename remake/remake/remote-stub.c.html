45/* Start a remote job running the command in ARGV,
46 with environment from ENVP. It gets standard input from STDIN_FD. On
47 failure, return nonzero. On success, return zero, and set *USED_STDIN
48 to nonzero if it will actually use STDIN_FD, zero if not, set *ID_PTR to
49 a unique identification, and set *IS_REMOTE to zero if the job is local,
50 nonzero if it is remote (meaning *ID_PTR is a process ID). */
51
52int
53start_remote_job (char **argv UNUSED, char **envp UNUSED, int stdin_fd UNUSED,
54 int *is_remote UNUSED, int *id_ptr UNUSED,
55 int *used_stdin UNUSED)
56{
57 return -1;
58}

60/* Get the status of a dead remote child. Block waiting for one to die
61 if BLOCK is nonzero. Set *EXIT_CODE_PTR to the exit status, *SIGNAL_PTR
62 to the termination signal or zero if it exited normally, and *COREDUMP_PTR
63 nonzero if it dumped core. Return the ID of the child that died,
64 0 if we would have to block and !BLOCK, or < 0 if there were none. */
65
66int
67remote_status (int *exit_code_ptr UNUSED, int *signal_ptr UNUSED,
68 int *coredump_ptr UNUSED, int block UNUSED)
69{
70 errno = ECHILD;
71 return -1;
72}
