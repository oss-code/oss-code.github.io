56/* Write a string to the current STDOUT or STDERR. */
57static void
58_outputs (struct output *out, int is_err, const char *msg)
59{
60 if (! out || ! out->syncout)
61 {
62 FILE *f = is_err ? stderr : stdout;
63 fputs (msg, f);
64 fflush (f);
65 }
66 else
67 {
68 int fd = is_err ? out->err : out->out;
69 int len = strlen (msg);
70 int r;
71
72 EINTRLOOP (r, lseek (fd, 0, SEEK_END));
73 while (1)
74 {
75 EINTRLOOP (r, write (fd, msg, len));
76 if (r == len || r <= 0)
77 break;
78 len -= r;
79 msg += r;
80 }
81 }
82}

84/* Write a message indicating that we've just entered or
85 left (according to ENTERING) the current directory. */
86
87int
88log_working_directory (int entering)
89{
90 static char *buf = NULL;
91 static unsigned int len = 0;
92 unsigned int need;
93 const char *fmt;
94 char *p;
95
96 /* Get enough space for the longest possible output. */
97 need = strlen (program) + INTSTR_LENGTH + 2 + 1;
98 if (starting_directory)
99 need += strlen (starting_directory);
100
101 /* Use entire sentences to give the translators a fighting chance. */
102 if (makelevel == 0)
103 if (starting_directory == 0)
104 if (entering)
105 fmt = _("%s: Entering an unknown directory\n");
106 else
107 fmt = _("%s: Leaving an unknown directory\n");
108 else
109 if (entering)
110 fmt = _("%s: Entering directory '%s'\n");
111 else
112 fmt = _("%s: Leaving directory '%s'\n");
113 else
114 if (starting_directory == 0)
115 if (entering)
116 fmt = _("%s[%u]: Entering an unknown directory\n");
117 else
118 fmt = _("%s[%u]: Leaving an unknown directory\n");
119 else
120 if (entering)
121 fmt = _("%s[%u]: Entering directory '%s'\n");
122 else
123 fmt = _("%s[%u]: Leaving directory '%s'\n");
124
125 need += strlen (fmt);
126
127 if (need > len)
128 {
129 buf = xrealloc (buf, need);
130 len = need;
131 }
132
133 p = buf;
134 if (print_data_base_flag)
135 {
136 *(p++) = '#';
137 *(p++) = ' ';
138 }
139
140 if (makelevel == 0)
141 if (starting_directory == 0)
142 sprintf (p, fmt , program);
143 else
144 sprintf (p, fmt, program, starting_directory);
145 else if (starting_directory == 0)
146 sprintf (p, fmt, program, makelevel);
147 else
148 sprintf (p, fmt, program, makelevel, starting_directory);
149
150 _outputs (NULL, 0, buf);
151
152 return 1;
153}

155/* Set a file descriptor to be in O_APPEND mode.
156 If it fails, just ignore it. */
157
158static void
159set_append_mode (int fd)
160{
161#if defined(F_GETFL) && defined(F_SETFL) && defined(O_APPEND)
162 int flags = fcntl (fd, F_GETFL, 0);
163 if (flags >= 0)
164 fcntl (fd, F_SETFL, flags | O_APPEND);
165#endif
166}

171/* Semaphore for use in -j mode with output_sync. */
172static sync_handle_t sync_handle = -1;
173
174#define FD_NOT_EMPTY(_f) ((_f) != OUTPUT_NONE && lseek ((_f), 0, SEEK_END) > 0)
175
176/* Set up the sync handle. Disables output_sync on error. */
177static int
178sync_init (void)
179{
180 int combined_output = 0;
181
182#ifdef WINDOWS32
183 if ((!STREAM_OK (stdout) && !STREAM_OK (stderr))
184 || (sync_handle = create_mutex ()) == -1)
185 {
186 perror_with_name ("output-sync suppressed: ", "stderr");
187 output_sync = 0;
188 }
189 else
190 {
191 combined_output = same_stream (stdout, stderr);
192 prepare_mutex_handle_string (sync_handle);
193 }
194
195#else
196 if (STREAM_OK (stdout))
197 {
198 struct stat stbuf_o, stbuf_e;
199
200 sync_handle = fileno (stdout);
201 combined_output = (fstat (fileno (stdout), &stbuf_o) == 0
202 && fstat (fileno (stderr), &stbuf_e) == 0
203 && stbuf_o.st_dev == stbuf_e.st_dev
204 && stbuf_o.st_ino == stbuf_e.st_ino);
205 }
206 else if (STREAM_OK (stderr))
207 sync_handle = fileno (stderr);
208 else
209 {
210 perror_with_name ("output-sync suppressed: ", "stderr");
211 output_sync = 0;
212 }
213#endif
214
215 return combined_output;
216}

218/* Support routine for output_sync() */
219static void
220pump_from_tmp (int from, FILE *to)
221{
222 static char buffer[8192];
223
224#ifdef WINDOWS32
225 int prev_mode;
226
227 /* "from" is opened by open_tmpfd, which does it in binary mode, so
228 we need the mode of "to" to match that. */
229 prev_mode = _setmode (fileno (to), _O_BINARY);
230#endif
231
232 if (lseek (from, 0, SEEK_SET) == -1)
233 perror ("lseek()");
234
235 while (1)
236 {
237 int len;
238 EINTRLOOP (len, read (from, buffer, sizeof (buffer)));
239 if (len < 0)
240 perror ("read()");
241 if (len <= 0)
242 break;
243 if (fwrite (buffer, len, 1, to) < 1)
244 {
245 perror ("fwrite()");
246 break;
247 }
248 fflush (to);
249 }
250
251#ifdef WINDOWS32
252 /* Switch "to" back to its original mode, so that log messages by
253 Make have the same EOL format as without --output-sync. */
254 _setmode (fileno (to), prev_mode);
255#endif
256}

258/* Obtain the lock for writing output. */
259static void *
260acquire_semaphore (void)
261{
262 static struct flock fl;
263
264 fl.l_type = F_WRLCK;
265 fl.l_whence = SEEK_SET;
266 fl.l_start = 0;
267 fl.l_len = 1;
268 if (fcntl (sync_handle, F_SETLKW, &fl) != -1)
269 return &fl;
270 perror ("fcntl()");
271 return NULL;
272}
273
274/* Release the lock for writing output. */
275static void
276release_semaphore (void *sem)
277{
278 struct flock *flp = (struct flock *)sem;
279 flp->l_type = F_UNLCK;
280 if (fcntl (sync_handle, F_SETLKW, flp) == -1)
281 perror ("fcntl()");
282}

284/* Returns a file descriptor to a temporary file. The file is automatically
285 closed/deleted on exit. Don't use a FILE* stream. */
286int
287output_tmpfd (void)
288{
289 int fd = -1;
290 FILE *tfile = tmpfile ();
291
292 if (! tfile)
293 pfatal_with_name ("tmpfile");
294
295 /* Create a duplicate so we can close the stream. */
296 fd = dup (fileno (tfile));
297 if (fd < 0)
298 pfatal_with_name ("dup");
299
300 fclose (tfile);
301
302 set_append_mode (fd);
303
304 return fd;
305}

307/* Adds file descriptors to the child structure to support output_sync; one
308 for stdout and one for stderr as long as they are open. If stdout and
309 stderr share a device they can share a temp file too.
310 Will reset output_sync on error. */
311static void
312setup_tmpfile (struct output *out)
313{
314 /* Is make's stdout going to the same place as stderr? */
315 static int combined_output = -1;
316
317 if (combined_output < 0)
318 combined_output = sync_init ();
319
320 if (STREAM_OK (stdout))
321 {
322 int fd = output_tmpfd ();
323 if (fd < 0)
324 goto error;
325 CLOSE_ON_EXEC (fd);
326 out->out = fd;
327 }
328
329 if (STREAM_OK (stderr))
330 {
331 if (out->out != OUTPUT_NONE && combined_output)
332 out->err = out->out;
333 else
334 {
335 int fd = output_tmpfd ();
336 if (fd < 0)
337 goto error;
338 CLOSE_ON_EXEC (fd);
339 out->err = fd;
340 }
341 }
342
343 return;
344
345 /* If we failed to create a temp file, disable output sync going forward. */
346 error:
347 output_close (out);
348 output_sync = OUTPUT_SYNC_NONE;
349}

351/* Synchronize the output of jobs in -j mode to keep the results of
352 each job together. This is done by holding the results in temp files,
353 one for stdout and potentially another for stderr, and only releasing
354 them to "real" stdout/stderr when a semaphore can be obtained. */
355
356void
357output_dump (struct output *out)
358{
359 int outfd_not_empty = FD_NOT_EMPTY (out->out);
360 int errfd_not_empty = FD_NOT_EMPTY (out->err);
361
362 if (outfd_not_empty || errfd_not_empty)
363 {
364 int traced = 0;
365
366 /* Try to acquire the semaphore. If it fails, dump the output
367 unsynchronized; still better than silently discarding it.
368 We want to keep this lock for as little time as possible. */
369 void *sem = acquire_semaphore ();
370
371 /* Log the working directory for this dump. */
372 if (print_directory_flag && output_sync != OUTPUT_SYNC_RECURSE)
373 traced = log_working_directory (1);
374
375 if (outfd_not_empty)
376 pump_from_tmp (out->out, stdout);
377 if (errfd_not_empty && out->err != out->out)
378 pump_from_tmp (out->err, stderr);
379
380 if (traced)
381 log_working_directory (0);
382
383 /* Exit the critical section. */
384 if (sem)
385 release_semaphore (sem);
386
387 /* Truncate and reset the output, in case we use it again. */
388 if (out->out != OUTPUT_NONE)
389 {
390 int e;
391 lseek (out->out, 0, SEEK_SET);
392 EINTRLOOP (e, ftruncate (out->out, 0));
393 }
394 if (out->err != OUTPUT_NONE && out->err != out->out)
395 {
396 int e;
397 lseek (out->err, 0, SEEK_SET);
398 EINTRLOOP (e, ftruncate (out->err, 0));
399 }
400 }
401}
402#endif /* NO_OUTPUT_SYNC */

407#ifndef HAVE_STDLIB_H
408# ifdef HAVE_MKSTEMP
409int mkstemp (char *template);
410# else
411char *mktemp (char *template);
412# endif
413#endif
414
415FILE *
416output_tmpfile (char **name, const char *template)
417{
418#ifdef HAVE_FDOPEN
419 int fd;
420#endif
421
422#if defined HAVE_MKSTEMP || defined HAVE_MKTEMP
423# define TEMPLATE_LEN strlen (template)
424#else
425# define TEMPLATE_LEN L_tmpnam
426#endif
427 *name = xmalloc (TEMPLATE_LEN + 1);
428 strcpy (*name, template);
429
430#if defined HAVE_MKSTEMP && defined HAVE_FDOPEN
431 /* It's safest to use mkstemp(), if we can. */
432 fd = mkstemp (*name);
433 if (fd == -1)
434 return 0;
435 return fdopen (fd, "w");
436#else
437# ifdef HAVE_MKTEMP
438 (void) mktemp (*name);
439# else
440 (void) tmpnam (*name);
441# endif
442
443# ifdef HAVE_FDOPEN
444 /* Can't use mkstemp(), but guard against a race condition. */
445 EINTRLOOP (fd, open (*name, O_CREAT|O_EXCL|O_WRONLY, 0600));
446 if (fd == -1)
447 return 0;
448 return fdopen (fd, "w");
449# else
450 /* Not secure, but what can we do? */
451 return fopen (*name, "w");
452# endif
453#endif
454}

457/* This code is stolen from gnulib.
458 If/when we abandon the requirement to work with K&R compilers, we can
459 remove this (and perhaps other parts of GNU make!) and migrate to using
460 gnulib directly.
461
462 This is called only through atexit(), which means die() has already been
463 invoked. So, call exit() here directly. Apparently that works...?
464*/
465
466/* Close standard output, exiting with status 'exit_failure' on failure.
467 If a program writes *anything* to stdout, that program should close
468 stdout and make sure that it succeeds before exiting. Otherwise,
469 suppose that you go to the extreme of checking the return status
470 of every function that does an explicit write to stdout. The last
471 printf can succeed in writing to the internal stream buffer, and yet
472 the fclose(stdout) could still fail (due e.g., to a disk full error)
473 when it tries to write out that buffered data. Thus, you would be
474 left with an incomplete output file and the offending program would
475 exit successfully. Even calling fflush is not always sufficient,
476 since some file systems (NFS and CODA) buffer written/flushed data
477 until an actual close call.
478
479 Besides, it's wasteful to check the return value from every call
480 that writes to stdout -- just let the internal stream state record
481 the failure. That's what the ferror test is checking below.
482
483 It's important to detect such failures and exit nonzero because many
484 tools (most notably 'make' and other build-management systems) depend
485 on being able to detect failure in other tools via their exit status. */
486
487static void
488close_stdout (void)
489{
490 int prev_fail = ferror (stdout);
491 int fclose_fail = fclose (stdout);
492
493 if (prev_fail || fclose_fail)
494 {
495 if (fclose_fail)
496 perror_with_name (_("write error: stdout"), "");
497 else
498 O (error, NILF, _("write error: stdout"));
499 exit (MAKE_TROUBLE);
500 }
501}

504void
505output_init (struct output *out)
506{
507 if (out)
508 {
509 out->out = out->err = OUTPUT_NONE;
510 out->syncout = !!output_sync;
511 return;
512 }
513
514 /* Configure this instance of make. Be sure stdout is line-buffered. */
515
516#ifdef HAVE_SETVBUF
517# ifdef SETVBUF_REVERSED
518 setvbuf (stdout, _IOLBF, xmalloc (BUFSIZ), BUFSIZ);
519# else /* setvbuf not reversed. */
520 /* Some buggy systems lose if we pass 0 instead of allocating ourselves. */
521 setvbuf (stdout, 0, _IOLBF, BUFSIZ);
522# endif /* setvbuf reversed. */
523#elif HAVE_SETLINEBUF
524 setlinebuf (stdout);
525#endif /* setlinebuf missing. */
526
527 /* Force stdout/stderr into append mode. This ensures parallel jobs won't
528 lose output due to overlapping writes. */
529 set_append_mode (fileno (stdout));
530 set_append_mode (fileno (stderr));
531
532#ifdef HAVE_ATEXIT
533 if (STREAM_OK (stdout))
534 atexit (close_stdout);
535#endif
536}

538void
539output_close (struct output *out)
540{
541 if (! out)
542 {
543 if (stdio_traced)
544 log_working_directory (0);
545 return;
546 }
547
548#ifndef NO_OUTPUT_SYNC
549 output_dump (out);
550#endif
551
552 if (out->out >= 0)
553 close (out->out);
554 if (out->err >= 0 && out->err != out->out)
555 close (out->err);
556
557 output_init (out);
558}

560/* We're about to generate output: be sure it's set up. */
561void
562output_start (void)
563{
564#ifndef NO_OUTPUT_SYNC
565 /* If we're syncing output make sure the temporary file is set up. */
566 if (output_context && output_context->syncout)
567 if (! OUTPUT_ISSET(output_context))
568 setup_tmpfile (output_context);
569#endif
570
571 /* If we're not syncing this output per-line or per-target, make sure we emit
572 the "Entering..." message where appropriate. */
573 if (output_sync == OUTPUT_SYNC_NONE || output_sync == OUTPUT_SYNC_RECURSE)
574 if (! stdio_traced && print_directory_flag)
575 stdio_traced = log_working_directory (1);
576}
577
578void
579outputs (int is_err, const char *msg)
580{
581 if (! msg || *msg == '\0')
582 return;
583
584 output_start ();
585
586 _outputs (output_context, is_err, msg);
587}

590static struct fmtstring
591 {
592 char *buffer;
593 size_t size;
594 } fmtbuf = { NULL, 0 };
595
596static char *
597get_buffer (size_t need)
598{
599 /* Make sure we have room. NEED includes space for \0. */
600 if (need > fmtbuf.size)
601 {
602 fmtbuf.size += need * 2;
603 fmtbuf.buffer = xrealloc (fmtbuf.buffer, fmtbuf.size);
604 }
605
606 fmtbuf.buffer[need-1] = '\0';
607
608 return fmtbuf.buffer;
609}

611/* Print a message on stdout. */
612
613void
614message (int prefix, size_t len, const char *fmt, ...)
615{
616 va_list args;
617 char *p;
618
619 len += strlen (fmt) + strlen (program) + INTSTR_LENGTH + 4 + 1 + 1;
620 p = get_buffer (len);
621
622 if (prefix)
623 {
624 if (makelevel == 0)
625 sprintf (p, "%s: ", program);
626 else
627 sprintf (p, "%s[%u]: ", program, makelevel);
628 p += strlen (p);
629 }
630
631 va_start (args, fmt);
632 vsprintf (p, fmt, args);
633 va_end (args);
634
635 strcat (p, "\n");
636
637 assert (fmtbuf.buffer[len-1] == '\0');
638 outputs (0, fmtbuf.buffer);
639}

641/* Print an error message. */
642
643void
644error (const gmk_floc *flocp, size_t len, const char *fmt, ...)
645{
646 va_list args;
647 char *p;
648
649 len += (strlen (fmt) + strlen (program)
650 + (flocp && flocp->filenm ? strlen (flocp->filenm) : 0)
651 + INTSTR_LENGTH + 4 + 1 + 1);
652 p = get_buffer (len);
653
654 if (flocp && flocp->filenm)
655 sprintf (p, "%s:%lu: ", flocp->filenm, flocp->lineno + flocp->offset);
656 else if (makelevel == 0)
657 sprintf (p, "%s: ", program);
658 else
659 sprintf (p, "%s[%u]: ", program, makelevel);
660 p += strlen (p);
661
662 va_start (args, fmt);
663 vsprintf (p, fmt, args);
664 va_end (args);
665
666 strcat (p, "\n");
667
668 assert (fmtbuf.buffer[len-1] == '\0');
669 outputs (1, fmtbuf.buffer);
670}

672/* Print an error message and exit. */
673
674void
675fatal (const gmk_floc *flocp, size_t len, const char *fmt, ...)
676{
677 va_list args;
678 const char *stop = _(". Stop.\n");
679 char *p;
680
681 len += (strlen (fmt) + strlen (program)
682 + (flocp && flocp->filenm ? strlen (flocp->filenm) : 0)
683 + INTSTR_LENGTH + 8 + strlen (stop) + 1);
684 p = get_buffer (len);
685
686 if (flocp && flocp->filenm)
687 sprintf (p, "%s:%lu: *** ", flocp->filenm, flocp->lineno + flocp->offset);
688 else if (makelevel == 0)
689 sprintf (p, "%s: *** ", program);
690 else
691 sprintf (p, "%s[%u]: *** ", program, makelevel);
692 p += strlen (p);
693
694 va_start (args, fmt);
695 vsprintf (p, fmt, args);
696 va_end (args);
697
698 strcat (p, stop);
699
700 assert (fmtbuf.buffer[len-1] == '\0');
701 outputs (1, fmtbuf.buffer);
702
703 die (MAKE_FAILURE);
704}
