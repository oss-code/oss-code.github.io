20/** \file dbg_break.h
21 *
22 * \brief debugger command beakpoint routines.
23 */
24
25#ifndef DBG_BREAK_H
26#define DBG_BREAK_H
27
28#include "types.h"
29
30/*! Opaque type definition for an item in the breakpoint list. */
31typedef struct breakpoint_node breakpoint_node_t;
32
33/** Pointers to top of current breakpoint list. */
34extern breakpoint_node_t *p_breakpoint_top;
35
36/** Pointers to bottom of current breakpoint list. */
37extern breakpoint_node_t *p_breakpoint_bottom;
38
39/** The largest breakpoint number previously given. When a new
40 breakpoint is set it will be i_breakpoints+1. */
41extern unsigned int i_breakpoints;
42
43/*! Add "p_target" to the list of breakpoints. Return true if
44 there were no errors
45*/
46extern bool add_breakpoint (file_t *p_target, unsigned int brkp_mask);
47
48/*! Remove breakpoint i from the list of breakpoints. Return true if
49 there were no errors
50*/
51extern bool remove_breakpoint (unsigned int i);
52
53/*! List breakpoints.*/
54extern void list_breakpoints (void);
55
56#endif /* DBG_BREAK_H */
