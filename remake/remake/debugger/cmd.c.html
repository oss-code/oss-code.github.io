147short_cmd_t short_command[256] = { { NULL,
148 (const char *) '\0',
149 (const char *) '\0' }, };

151/* Look up NAME as the name of a command, and return a pointer to that
152 command. Return a NULL pointer if NAME isn't a command name. */
153static short_cmd_t *
154find_command (const char *psz_name)
155{
156 unsigned int i;
157
158 for (i = 0; aliases[i].alias; i++) {
159 const int cmp = strcmp (psz_name, aliases[i].alias);
160 if ( 0 == cmp ) {
161 psz_name = aliases[i].command;
162 break;
163 } else
164 /* Words should be in alphabetic order by alias name.
165 Have we gone too far? */
166 if (cmp < 0) break;
167 }
168
169 for (i = 0; commands[i].long_name; i++) {
170 const int cmp = strcmp (psz_name, commands[i].long_name);
171 if ( 0 == cmp ) {
172 return (&short_command[(uint8_t) commands[i].short_name]);
173 } else
174 /* Words should be in alphabetic order by command name.
175 Have we gone too far? */
176 if (cmp < 0) break;
177 }
178
179 return ((short_cmd_t *)NULL);
180}

252/* Execute a command line. */
253extern debug_return_t
254execute_line (char *psz_line)
255{
256 unsigned int i = 0;
257 short_cmd_t *command;
258 char *psz_word = get_word(&psz_line);
259
260 if (1 == strlen(psz_word)) {
261 if ( NULL != short_command[(uint8_t) psz_word[0]].func )
262 command = &short_command[(uint8_t) psz_word[0]];
263 else
264 command = NULL;
265 } else {
266 command = find_command (psz_word);
267 }
268 if (!command)
269 {
270 dbg_errmsg(_("No such debugger command: %s."), psz_word);
271 return debug_readloop;
272 }
273
274 /* Get argument to command, if any. */
275 while (whitespace (psz_line[i]))
276 i++;
277
278 psz_debugger_args = psz_line + i;
279
280 /* Call the function. */
281 return ((*(command->func)) (psz_debugger_args));
282}

306/* Set a variable. Set "expand' to 1 if you want variable
307 definitions inside the value getting passed in to be expanded
308 before assigment. */
309static debug_return_t dbg_cmd_set_var (char *psz_args, int expand)
310{
311 if (!psz_args || 0==strlen(psz_args)) {
312 dbg_msg(_("You need to supply a variable name."));
313 } else {
314 variable_t *p_v;
315 char *psz_varname = get_word(&psz_args);
316 unsigned int u_len = strlen(psz_varname);
317
318 while (*psz_args && whitespace (*psz_args))
319 *psz_args +=1;
320
321 p_v = lookup_variable (psz_varname, u_len);
322
323 if (p_v) {
324 char *psz_value = expand ? variable_expand(psz_args) : psz_args;
325
326 define_variable_in_set(p_v->name, u_len, psz_value,
327 o_debugger, 0, NULL,
328 &(p_v->fileinfo));
329 dbg_msg(_("Variable %s now has value '%s'"), psz_varname,
330 psz_value);
331 } else {
332 p_v = try_without_dollar(psz_varname);
333 if (p_v) {
334 char *psz_value = expand ? variable_expand(psz_args) : psz_args;
335 define_variable_in_set(p_v->name, u_len, psz_value,
336 o_debugger, 0, NULL,
337 &(p_v->fileinfo));
338 dbg_msg(_("Variable %s now has value '%s'"), psz_varname,
339 psz_value);
340 }
341 }
342 }
343 return debug_readloop;
344}

348#define PROMPT_LENGTH 300
349
350#include <setjmp.h>
351jmp_buf debugger_loop;

353/* Should be less that PROMPT_LENGTH / 2 - strlen("remake ") + log(history)
354 We will make it much less that since people can't count more than
355 10 or so nested <<<<>>>>'s easily.
356*/
357#define MAX_NEST_DEPTH 10
358
359debug_return_t enter_debugger (target_stack_node_t *p,
360 file_t *p_target, int errcode,
361 debug_enter_reason_t reason)
362{
363 volatile debug_return_t debug_return = debug_readloop;
364 static bool b_init = false;
365 static bool b_readline_init = false;
366 char open_depth[MAX_NEST_DEPTH];
367 char close_depth[MAX_NEST_DEPTH];
368 unsigned int i = 0;
369
370 last_stop_reason = reason;
371
372 if ( in_debugger == DEBUGGER_QUIT_RC ) {
373 return continue_execution;
374 }
375
376 if ( i_debugger_stepping > 1 || i_debugger_nexting > 1 ) {
377 /* Don't stop unless we are here from a breakpoint. But
378 do decrement the step count. */
379 if (i_debugger_stepping) i_debugger_stepping--;
380 if (i_debugger_nexting) i_debugger_nexting--;
381 if (!p_target->tracing) return continue_execution;
382 } else if ( !debugger_on_error
383 && !(i_debugger_stepping || i_debugger_nexting)
384 && p_target && !p_target->tracing && -2 != errcode )
385 return continue_execution;
386
387 /* Clear temporary breakpoints. */
388 if (p_target && p_target->tracing & BRK_TEMP)
389 switch(last_stop_reason)
390 {
391 case DEBUG_BRKPT_AFTER_CMD:
392 case DEBUG_BRKPT_BEFORE_PREREQ:
393 case DEBUG_BRKPT_AFTER_PREREQ:
394 p_target->tracing = BRK_NONE;
395 default:
396 ;
397 }
398
399#ifdef HAVE_LIBREADLINE
400 if (use_readline_flag && !b_readline_init) {
401 rl_initialize ();
402 using_history ();
403 add_history ("");
404 b_readline_init = true;
405 }
406#endif
407 if (!b_init) {
408 cmd_initialize();
409 file2lines.ht_size = 0;
410 b_init = true;
411 }
412
413
414 /* Set initial frame position reporting area: 0 is bottom. */
415 p_target_loc = NULL;
416 psz_target_name = (char *) "";
417 i_stack_pos = 0;
418
419 p_stack = p_stack_top = p;
420 p_floc_stack = p_stack_floc_top;
421
422 /* Get the target name either from the stack top (preferred) or
423 the passed in target.
424 */
425 if (p && p->p_target) {
426 p_target_loc = &(p->p_target->floc);
427 psz_target_name = (char *) p->p_target->name;
428 } else if (p_target) {
429 p_target_loc = &(p_target->floc);
430 psz_target_name = (char *) p_target->name;
431 }
432
433 for (i=0; i<=makelevel && i < MAX_NEST_DEPTH-5; i++) {
434 open_depth[i] = '<';
435 close_depth[i] = '>';
436 }
437
438 if ( MAX_NEST_DEPTH - 5 == i ) {
439 close_depth[i] = open_depth[i] = '.'; i++;
440 close_depth[i] = open_depth[i] = '.'; i++;
441 close_depth[i] = open_depth[i] = '.'; i++;
442 }
443
444 open_depth[i] = close_depth[i] = '\0';
445
446 in_debugger = true;
447
448 if (errcode) {
449 if (-1 == errcode) {
450 printf("\n***Entering debugger because we encountered an error.\n");
451 } else if (-2 == errcode) {
452 if (0 == makelevel) {
453 printf("\nMakefile terminated.\n");
454 dbg_msg("Use q to quit or R to restart");
455 } else {
456 printf("\nMakefile finished at level %u. Use R to restart\n",
457 makelevel);
458 dbg_msg("the makefile at this level or 's', 'n', or 'F' to continue "
459 "in parent");
460 in_debugger = DEBUGGER_QUIT_RC;
461 }
462 } else {
463 printf("\n***Entering debugger because we encountered a fatal error.\n");
464 dbg_errmsg("Exiting the debugger will exit make with exit code %d.",
465 errcode);
466 }
467 }
468
469 print_debugger_location(p_target, reason, NULL);
470
471 /* Loop reading and executing lines until the user quits. */
472 for ( debug_return = debug_readloop;
473 debug_return == debug_readloop || debug_return == debug_cmd_error; ) {
474 char prompt[PROMPT_LENGTH];
475 char *line=NULL;
476 char *s;
477
478 if (setjmp(debugger_loop))
479 dbg_errmsg("Internal error jumped back to debugger loop");
480 else {
481
482#ifdef HAVE_LIBREADLINE
483 if (use_readline_flag) {
484 snprintf(prompt, PROMPT_LENGTH, "remake%s%d%s ",
485 open_depth, where_history(), close_depth);
486
487 line = readline (prompt);
488 } else
489#endif
490 {
491 snprintf(prompt, PROMPT_LENGTH, "remake%s0%s ", open_depth,
492 close_depth);
493 printf("%s", prompt);
494 if (line == NULL) line = calloc(1, 2048);
495 line = fgets(line, 2048, stdin);
496 if (NULL != line) chomp(line);
497 }
498
499 if ( line ) {
500 if ( *(s=stripwhite(line)) ) {
501 add_history (s);
502 debug_return=execute_line(s);
503 } else {
504 add_history ("step");
505 debug_return=dbg_cmd_step((char *) "");
506 }
507 free (line);
508 } else {
509 dbg_cmd_quit(NULL);
510 }
511 }
512 }
513
514 if (in_debugger != DEBUGGER_QUIT_RC)
515 in_debugger=false;
516
517 return debug_return;
518}
