23static debug_return_t
24dbg_cmd_write(char *psz_args)
25{
26 file_t *p_target = NULL;
27 const char *psz_target = NULL;
28 int b_stdout = 0;
29
30 p_target = get_target(&psz_args, &psz_target);
31 if (p_target) {
32 variable_t *p_v = lookup_variable ("SHELL", strlen ("SHELL"));
33 char *psz_filename = NULL;
34 FILE *outfd;
35 char *s;
36
37 if (! p_target->cmds || ! p_target->cmds->commands) {
38 printf(_("Target \"%s\" doesn't have commands associated with it.\n"),
39 psz_target);
40 return debug_readloop;
41 }
42
43 s = p_target->cmds->commands;
44
45 /* FIXME: should get directory from a variable, e.g. TMPDIR */
46
47 if (psz_args && *psz_args) {
48 if (strcmp (psz_args, "here") == 0)
49 b_stdout = 1;
50 else
51 psz_filename = strdup(psz_args);
52 } else {
53 /* Create target from the basename of the target name. */
54 char *psz_target_basename = strrchr(psz_target, '/');
55 if (!psz_target_basename)
56 psz_target_basename = (char *) psz_target;
57 else
58 psz_target_basename++; /* Skip delimiter */
59 psz_filename = CALLOC(char, strlen(psz_target_basename) + 10);
60 snprintf(psz_filename, MAX_FILE_LENGTH, "/tmp/%s.sh",
61 psz_target_basename);
62 }
63
64 /* Skip leading space, MAKE's command prefixes:
65 echo suppression @,
66 ignore-error -,
67 and recursion +
68 */
69 while (*s != '\0')
70 {
71 switch (*s) {
72 case '@':
73 case '+':
74 case '-':
75 ++s;
76 break;
77 default:
78 if (!isblank ((unsigned char)*s))
79 goto found_begin;
80 ++s;
81 }
82 }
83
84 found_begin:
85 if ( '\0' == *s ) {
86 printf(_("Null command string parsed\n"));
87 } else {
88 dep_t *d;
89 dep_t *ood = 0;
90
91 if (b_stdout)
92 outfd = stdout;
93 else if (!(outfd = fopen (psz_filename, "w"))) {
94 perror ("write target");
95 free(psz_filename);
96 return debug_readloop;
97 }
98
99 if (p_v) {
100 fprintf(outfd, "#!%s\n", variable_expand(p_v->value));
101 }
102
103 if (!p_target->floc.filenm && p_target->cmds->fileinfo.filenm) {
104 /* Fake the location based on the commands - it's better than
105 nothing...
106 */
107 fprintf(outfd, "## %s/%s:%lu\n", starting_directory,
108 p_target->cmds->fileinfo.filenm,
109 p_target->cmds->fileinfo.lineno-1);
110 } else {
111 fprintf(outfd, "## %s/%s:%lu\n", starting_directory,
112 p_target->floc.filenm, p_target->floc.lineno);
113 }
114
115 /* FIXME: this code duplicaes some code in file.c: print_target_props. DRY. */
116 fprintf (outfd,
117 "## %s:%s", p_target->name,
118 p_target->double_colon ? ":" : "");
119
120 /* Print all normal dependencies; note any order-only deps. */
121 for (d = p_target->deps; d != 0; d = d->next)
122 if (! d->ignore_mtime) {
123 fprintf (outfd, " %s", dep_name (d));
124 } else if (! ood)
125 ood = d;
126
127 /* Print order-only deps, if we have any. */
128 if (ood)
129 {
130 fprintf (outfd, " | %s", dep_name (ood));
131 for (d = ood->next; d != 0; d = d->next)
132 if (d->ignore_mtime)
133 fprintf (outfd, " %s", dep_name (d));
134 }
135
136 fprintf (outfd, "\n");
137
138 {
139 char wd[300];
140 if (getcwd (wd, sizeof(wd))) {
141 fprintf(outfd, "\n#cd %s\n", wd);
142 }
143 }
144
145 initialize_file_variables (p_target, 0);
146 set_file_variables (p_target);
147
148 {
149#if 0
150 commands_t cmds;
151 char **lines;
152 unsigned int i;
153 memcpy(&cmds, p_target->cmds, sizeof(cmds));
154 cmds.command_lines = NULL;
155 chop_commands (&cmds);
156 lines = xmalloc (cmds.ncommand_lines * sizeof (char *));
157 expand_command_lines(&cmds, lines, p_target);
158 for (i = 0; i < cmds.ncommand_lines; ++i)
159 {
160 fprintf (outfd, "%s\n", lines[i]);
161 free (lines[i]);
162 }
163 free (lines);
164#else
165 char *line = allocated_variable_expand_for_file (s, p_target);
166 fprintf (outfd, "%s\n", line);
167 free(line);
168#endif
169 }
170
171 if (!b_stdout) {
172 struct stat buf;
173 if (0 == fstat(fileno(outfd), &buf)) {
174 mode_t mode = buf.st_mode;
175 if (buf.st_mode & S_IRUSR) mode |= S_IXUSR;
176 if (buf.st_mode & S_IRGRP) mode |= S_IXGRP;
177 if (buf.st_mode & S_IROTH) mode |= S_IXOTH;
178 if (0 != fchmod(fileno(outfd), mode)) {
179 printf(_("Can't set execute mode on \"%s\".\n"), psz_filename);
180 }
181 }
182 fclose(outfd);
183 printf(_("File \"%s\" written.\n"), psz_filename);
184 free(psz_filename);
185 }
186 }
187 }
188 return debug_readloop;
189}
190
191static void
192dbg_cmd_write_init(unsigned int c)
193{
194 short_command[c].func = &dbg_cmd_write;
195 short_command[c].use = _("write [TARGET [FILENAME]]");
196 short_command[c].doc =
197 _("writes the commands associated of a target to a file with MAKE\n"
198 "variables expanded. If no target given, the basename of the current\n"
199 "is used. If a filename is supplied it is used. If it is the string\n"
200 "\"here\", we write the output to stdout. If no filename is\n"
201 "given then create the filename by prepending a directory name to\n"
202 "the target name and then append \".sh\".");
203}
