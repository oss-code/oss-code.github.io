1/* Continue until the next command to be executed. */
2#define DEPENDS_COMMANDS " depends commands"
3static debug_return_t
4dbg_cmd_list(char *psz_arg)
5{
6 const char *psz_target = NULL;
7 char *target_cmd = NULL;
8 file_t *p_target;
9
10 if (psz_arg && 0 == strcmp(psz_arg, "-")) {
11 /* Show info for parent target. */
12 if (p_stack_top) {
13 /* We have a target stack */
14 target_stack_node_t *p=p_stack;
15
16 if (!p) {
17 dbg_errmsg(_("We don't seem to have a target to get parent of."));
18 return debug_cmd_error;
19 }
20
21 p = p->p_parent;
22 if (!p) {
23 dbg_errmsg(_("We don't seem to have a parent target."));
24 return debug_cmd_error;
25 }
26 p_target = p->p_target;
27 psz_target = p_target->name;
28 } else {
29 dbg_errmsg(_("We don't seem to have a target stack to get parent of."));
30 return debug_cmd_error;
31 }
32 } else {
33 unsigned int u_lineno=0;
34 f2l_entry_t entry_type;
35 if (get_uint(psz_arg, &u_lineno, false)) {
36 if (p_stack) {
37 p_target = target_for_file_and_line(p_stack->p_target->floc.filenm,
38 u_lineno, &entry_type);
39 if (!p_target) {
40 dbg_errmsg("Can't find target or pattern on line %s.\n"
41 "Use 'info lines' to get a list of and pattern lines.",
42 psz_arg);
43 return debug_cmd_error;
44 }
45 psz_target = p_target->name;
46 } else {
47 dbg_errmsg(_("We don't seem to have a target stack to get parent of."));
48 return debug_cmd_error;
49 }
50 } else
51 p_target = get_target(&psz_arg, &psz_target);
52 }
53
54 if (!p_target) {
55 dbg_errmsg(_("Trouble getting a target name for %s."), psz_target);
56 return debug_cmd_error;
57 }
58 print_floc_prefix(&p_target->floc);
59 target_cmd = CALLOC(char, strlen(psz_target) + 1 + strlen(DEPENDS_COMMANDS));
60 sprintf(target_cmd, "%s%s", psz_target, DEPENDS_COMMANDS);
61 return dbg_cmd_target(target_cmd);
62}
63
64static void
65dbg_cmd_list_init(unsigned int c)
66{
67 short_command[c].func = &dbg_cmd_list;
68 short_command[c].use = _("list [TARGET|LINE-NUMBER]");
69 short_command[c].doc =
70 _("List target dependencies and commands for TARGET or LINE NUMBER.\n"
71"Without a target name or line number, use the current target.\n"
72"A target name of '-' will use the parent target on the target stack.\n"
73 );
74}
