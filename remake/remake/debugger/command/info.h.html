131/*! Show target information: location, name, and description. */
132static void
133dbg_cmd_info_target_entry (const file_t *p_target,
134 info_target_output_mask_t output_mask)
135{
136 const gmk_floc *p_floc = &p_target->floc;
137 if (p_floc) {
138 if ((p_floc->filenm) && (output_mask & INFO_TARGET_POSITION)) {
139 printf("%s:%lu", p_floc->filenm, p_floc->lineno);
140 if (output_mask & INFO_TARGET_NAME)
141 printf(":\n");
142 else
143 printf("\n");
144 }
145 }
146 if (output_mask & INFO_TARGET_NAME) {
147 printf("\t%s", p_target->name);
148 if (p_target->description)
149 printf("\t# %s", p_target->description);
150 printf("\n");
151 } else if
152 ( (output_mask & INFO_TARGET_TASKS
153 && (p_target->cmds || p_target->phony)
154 && p_floc->filenm)
155 || (output_mask & INFO_TARGET_NAME) ) {
156 if (p_target->description) {
157 printf("%s", p_target->name);
158 printf("\t# %s\n", p_target->description);
159 } else if (!(output_mask & INFO_TARGET_TASK_COMMENT)) {
160 printf("%s\n", p_target->name);
161 }
162 }
163}

165int
166dbg_target_compare(const void *p1, const void *p2)
167{
168 const struct file *p_target1 = *(const file_t **) p1;
169 const struct file *p_target2 = *(const file_t **) p2;
170 return strcmp(p_target1->name, p_target2->name);
171}
172
173void
174dbg_cmd_info_targets(info_target_output_mask_t output_mask)
175{
176 struct file **file_slot_0 = (struct file **) hash_dump (&files, 0,
177 dbg_target_compare);
178 struct file **file_end = file_slot_0 + files.ht_fill;
179 struct file **pp_file_slot;
180 struct file *p_target;
181
182 for (pp_file_slot = file_slot_0; pp_file_slot < file_end; pp_file_slot++) {
183 if ((p_target = *pp_file_slot) != NULL)
184 dbg_cmd_info_target_entry(p_target, output_mask);
185 }
186}

188/* Show line information. We want output to be compatible with gdb output.*/
189void
190dbg_cmd_info_line()
191{
192 if (p_stack_top && p_stack_top->p_target &&
193 p_stack_top->p_target->floc.filenm) {
194 const gmk_floc *p_floc = &p_stack_top->p_target->floc;
195 if (!basename_filenames && strlen(p_floc->filenm)
196 && p_floc->filenm[0] != '/')
197 dbg_msg("Line %lu of \"%s/%s\"",
198 p_floc->lineno, starting_directory,
199 p_floc->filenm);
200 else
201 dbg_msg("Line %lu of \"%s\"", p_floc->lineno, p_floc->filenm);
202 } else {
203 dbg_msg("No line number info recorded.\n");
204 }
205}

207void
208dbg_cmd_info_program()
209{
210 printf(_("Starting directory `%s'\n"), starting_directory);
211 printf(_("Program invocation:\n"));
212 printf("\t");
213 dbg_print_invocation();
214 printf(_("Recursion level: %u\n"), makelevel);
215 dbg_cmd_info_line();
216 switch (last_stop_reason)
217 {
218 case DEBUG_BRKPT_AFTER_CMD:
219 printf(_("Program is stopped after running rule command(s).\n"));
220 break;
221 case DEBUG_BRKPT_BEFORE_PREREQ:
222 printf(_("Program stopped before rule-prequisite checking.\n"));
223 break;
224 case DEBUG_BRKPT_AFTER_PREREQ:
225 printf(_("Program is stopped after rule-prequisite checking.\n"));
226 break;
227 case DEBUG_GOAL_UPDATED_HIT:
228 printf(_("Program stopped for updating a goal.\n"));
229 printf("\n");
230 break;
231 case DEBUG_READ_HIT:
232 printf(_("Program stopped for reading a file.\n"));
233 printf("\n");
234 break;
235 case DEBUG_ERROR_HIT:
236 printf(_("Program stopped after an error encountered.\n"));
237 printf("\n");
238 break;
239 case DEBUG_STEP_HIT:
240 printf(_("Program stopped in stepping.\n"));
241 printf("\n");
242 break;
243 case DEBUG_STEP_COMMAND:
244 printf(_("Program stopped in stepping before running rule command(s).\n"));
245 printf("\n");
246 break;
247 case DEBUG_EXPLICIT_CALL:
248 printf(_("Program stopped from explicit debugger function call.\n"));
249 printf("\n");
250 break;
251 case DEBUG_NOT_GIVEN:
252 printf(_("Reason not given.\n"));
253 break;
254 case DEBUG_STACK_CHANGING:
255 /* Should not happen? */
256 break;
257 }
258}

261/* Give some info regarding the running program. */
262debug_return_t
263dbg_cmd_info(char *psz_args)
264{
265 if (!psz_args || 0==strlen(psz_args)) {
266 unsigned int i;
267 for (i = 0; info_subcommands[i].name; i++) {
268 dbg_help_subcmd_entry("info", "%-10s -- %s",
269 &(info_subcommands[i]), false);
270 }
271 return debug_readloop;
272 } else {
273 char *psz_subcmd = get_word(&psz_args);
274 if (0 == strcmp(psz_subcmd, "lines")) {
275 file2lines_dump();
276 } else if (is_abbrev_of (psz_subcmd, "line", 2)) {
277 dbg_cmd_info_line();
278 } else if (is_abbrev_of (psz_subcmd, "locals", 2)) {
279 const char *psz_target = NULL;
280 char *psz_subcmds = NULL;
281 file_t *p_target = get_target(&psz_subcmds, &psz_target);
282
283 if (p_target) {
284 if (!p_target->variables) {
285 initialize_file_variables (p_target, 0);
286 set_file_variables (p_target);
287 if (!p_target->variables) {
288 printf("Can't get variable information for target %s\n",
289 psz_target);
290 return debug_readloop;
291 }
292 }
293 } else {
294 dbg_errmsg("No target information for %s.", psz_target);
295 return debug_cmd_error;
296 }
297 hash_map_arg (&p_target->variables->set->table,
298 print_variable_info, NULL);
299 } else if (is_abbrev_of (psz_subcmd, "breakpoints", 1)) {
300 list_breakpoints();
301 } else if (is_abbrev_of (psz_subcmd, "makefiles", 1) ||
302 is_abbrev_of (psz_subcmd, "files", 2)) {
303 if (0 == strlen(psz_args))
304 print_read_makefiles(NULL);
305 else {
306 if (!print_read_makefiles(psz_args))
307 dbg_errmsg("File %s not in list of read-in files.", psz_args);
308 }
309 } else if (is_abbrev_of (psz_subcmd, "frame", 2)) {
310 dbg_cmd_where(psz_args);
311 } else if (is_abbrev_of (psz_subcmd, "program", 1)) {
312 dbg_cmd_info_program();
313 } else if (is_abbrev_of (psz_subcmd, "rules", 1)) {
314 if (0 == strlen(psz_args))
315 print_rule_data_base (false);
316 else if (0 == strcmp(psz_args, "verbose"))
317 print_rule_data_base (true);
318 else
319 {
320 rule_t *r = find_rule(psz_args);
321 if (r)
322 print_rule(r, true);
323 else
324 dbg_errmsg(_("Rule %s not found.\n"), psz_args);
325 }
326 } else if (is_abbrev_of (psz_subcmd, "stack", 1)) {
327 print_target_stack(p_stack_top, i_stack_pos, MAX_STACK_SHOW);
328 } else if (0 == strcmp(psz_subcmd, "targets")) {
329 /* Note: "targets" has to come before "target" */
330 info_target_output_mask_t output_type;
331 if (0 == strlen(psz_args))
332 output_type = INFO_TARGET_POSITION_AND_NAME;
333 else if (is_abbrev_of (psz_args, "all", 1))
334 output_type = INFO_TARGET_POSITION_AND_NAME;
335 else if (is_abbrev_of (psz_args, "positions", 1))
336 output_type = INFO_TARGET_POSITION;
337 else if (is_abbrev_of (psz_args, "names", 1))
338 output_type = INFO_TARGET_NAME;
339 else if (is_abbrev_of (psz_args, "tasks", 1))
340 output_type = INFO_TARGET_TASKS;
341 else {
342 printf("Expecting 'all', 'positions', 'names', 'tasks', or nothing; got %s.\n",
343 psz_args);
344 return debug_cmd_error;
345 }
346 dbg_cmd_info_targets(output_type);
347
348 } else if (is_abbrev_of (psz_subcmd, "target", 1)) {
349 if (0 == strlen(psz_args)) {
350 if (p_stack_top && p_stack_top->p_target &&
351 p_stack_top->p_target->name)
352 printf("target: %s\n", p_stack_top->p_target->name);
353 else
354 {
355 printf("target unknown\n");
356 }
357 }
358 else
359 dbg_cmd_target(psz_args);
360
361 } else if (is_abbrev_of (psz_subcmd, "variables", 1)) {
362 print_variable_data_base();
363 } else if (is_abbrev_of (psz_subcmd, "vpath", 1)) {
364 print_vpath_data_base ();
365 } else if (is_abbrev_of (psz_subcmd, "warranty", 1)) {
366 printf("%s", WARRANTY);
367 } else {
368 dbg_errmsg(_("Undefined command \"%s\". Try \"help info\"."),
369 psz_subcmd);
370 }
371 }
372
373 return debug_readloop;
374}

376static void
377dbg_cmd_info_init(unsigned int c)
378{
379 short_command[c].func = &dbg_cmd_info;
380 short_command[c].use = _("info [SUBCOMMAND]");
381 short_command[c].doc =
382 _("Show program information regarding SUBCOMMAND.\n"
383 "If SUBCOMMAND is not specified, give list of \"info\" subcommands.");
384}
