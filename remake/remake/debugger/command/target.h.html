20debug_return_t
21dbg_cmd_target(char *psz_args)
22{
23 const char *psz_target;
24 file_t *p_target = get_target(&psz_args, &psz_target);
25
26 if (p_target) {
27 print_target_mask_t i_mask = 0;
28 char *psz_word;
29
30 while( (psz_word = get_word(&psz_args))) {
31 if (!*psz_word) {
32 break;
33 } else if (is_abbrev_of(psz_word, "depends", 1)) {
34 i_mask |= PRINT_TARGET_DEPEND;
35 } else if (is_abbrev_of(psz_word, "order", 1)) {
36 i_mask |= PRINT_TARGET_ORDER;
37 } else if (is_abbrev_of(psz_word, "nonorder", 1)) {
38 i_mask |= PRINT_TARGET_NONORDER;
39 } else if (is_abbrev_of(psz_word, "attributes", 1)) {
40 i_mask |= PRINT_TARGET_ATTRS;
41 } else if (is_abbrev_of(psz_word, "state", 1)) {
42 i_mask |= PRINT_TARGET_STATE;
43 } else if (is_abbrev_of(psz_word, "time", 1)) {
44 i_mask |= PRINT_TARGET_TIME;
45 } else if (is_abbrev_of(psz_word, "variables", 1)) {
46 i_mask |= PRINT_TARGET_VARS;
47 } else if (is_abbrev_of(psz_word, "commands", 1)) {
48 i_mask |= PRINT_TARGET_CMDS;
49 } else if (is_abbrev_of(psz_word, "expand", 1)) {
50 i_mask |= (PRINT_TARGET_CMDS_EXP);
51 } else if (is_abbrev_of(psz_word, "previous", 1)) {
52 i_mask |= PRINT_TARGET_PREV;
53 } else {
54 printf("Don't understand attribute '%s'\n", psz_word);
55 return debug_readloop;
56 }
57 }
58
59 if (0 == i_mask) i_mask = PRINT_TARGET_ALL & (~PRINT_TARGET_VARS_HASH);
60
61 if (i_mask & PRINT_TARGET_VARS) {
62 initialize_file_variables (p_target, 0);
63 set_file_variables (p_target);
64 }
65
66 print_target_props(p_target, i_mask);
67 }
68 return debug_readloop;
69}
70
71static void
72dbg_cmd_target_init(unsigned int c)
73{
74 short_command[c].func = &dbg_cmd_target;
75 short_command[c].use = _("target [TARGET-NAME] [info1 [info2...]]");
76 short_command[c].doc =
77 _("Show information about a target.\n"
78"\nThe following attributes names can be given after a target name:\n"
79" attributes - rule attributes: precious, rule search, and pattern stem\n"
80" commands - shell commands that are run to update the target\n"
81" expand - like 'commands', but Makefile variables are expanded\n"
82" nonorder - non-order dependencies\n"
83" order - order dependencies\n"
84" depends - all target dependencies, i.e. order and non-order\n"
85" previous - previous target name hwen there are multiple double-colons\n"
86" state - shell command state\n"
87" time - last modification time and whether file has been updated\n"
88" variables - automatically set variables such as @ or <\n"
89"\n"
90"TARGET-NAME can be a variable like `@' (current target) or `<'\n"
91"(first dependency). If no variable or target name is supplied\n"
92"we to use the current target name.\n"
93 );
94}
