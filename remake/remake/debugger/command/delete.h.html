19/*
20 Delete some breakpoints. Arguments are breakpoint numbers with spaces
21 in between."To delete all breakpoints, give no argument.
22*/
23static debug_return_t
24dbg_cmd_delete(char *psz_args)
25{
26 int i_brkpt;
27 char *psz_word;
28
29 if (!psz_args || !*psz_args) {
30 while(i_breakpoints)
31 remove_breakpoint(1);
32 return debug_readloop;
33 }
34
35 psz_word = get_word(&psz_args);
36 while ( psz_word && *psz_word ) {
37 if (get_int(psz_word, &i_brkpt, true)) {
38 remove_breakpoint(i_brkpt);
39 }
40 psz_word = get_word(&psz_args);
41 }
42
43 return debug_readloop;
44}
45
46static void
47dbg_cmd_delete_init(unsigned int c)
48{
49 short_command[c].func = &dbg_cmd_delete;
50 short_command[c].use = _("delete breakpoint numbers..");
51 short_command[c].doc = _("Delete some breakpoints\n."
52"Arguments are breakpoint numbers with spaces in between.\n"
53"To delete all breakpoints, give no argument.");
54}
