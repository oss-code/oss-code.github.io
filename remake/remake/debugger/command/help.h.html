23void
24dbg_help_subcmd_entry(const char *psz_subcmd_name, const char *psz_fmt,
25 subcommand_var_info_t *p_subcmd, bool full_info)
26{
27 if (full_info) {
28 const char *doc = p_subcmd->doc ?
29 p_subcmd->doc : p_subcmd->short_doc;
30 printf("%s ", psz_subcmd_name);
31 printf("%s%s", p_subcmd->name, doc);
32 } else {
33 printf("%s ", psz_subcmd_name);
34 printf(psz_fmt, p_subcmd->name, p_subcmd->short_doc);
35 if (p_subcmd->var) {
36 if (p_subcmd->b_onoff)
37 printf(" is %s.",
38 var_to_on_off(* (int *) p_subcmd->var));
39 else
40 printf(" is %d.", *(int *)(p_subcmd->var));
41 }
42 }
43
44 printf("\n");
45}
46
47debug_return_t
48dbg_help_subcmd(const char *psz_subcmd_name,
49 short_cmd_t *p_command, const char *psz_args,
50 subcommand_var_info_t *subcommands)
51{
52 unsigned int i;
53 if (!psz_args || !*psz_args) {
54 printf("%s\n\n%s\n", p_command->use, p_command->doc);
55 } else {
56 for (i = 0; subcommands[i].name; i++) {
57 if (is_abbrev_of(psz_args, subcommands[i].name,
58 subcommands[i].min_abbrev)) {
59 dbg_help_subcmd_entry(psz_subcmd_name, "%s%s",
60 &(subcommands[i]), true);
61 return debug_readloop;
62 }
63 }
64 printf("There is no \"%s %s\" command.\n", psz_subcmd_name, psz_args);
65 }
66 return debug_readloop;
67}

69debug_return_t
70dbg_cmd_help(char *psz_args)
71{
72 unsigned int i;
73
74 if (!psz_args || !*psz_args) {
75 printf (" Command Short Name Aliases\n");
76 printf (" ---------------------- ---------- ---------\n");
77 for (i = 0; commands[i].long_name; i++) {
78 unsigned int j;
79 bool b_alias = false;
80 uint8_t s=commands[i].short_name;
81 printf(" %-31s (%c)",
82 short_command[s].use, commands[i].short_name);
83 for (j = 0; aliases[j].alias; j++) {
84 if (strcmp (commands[i].long_name, aliases[j].command) == 0) {
85 if (!b_alias) {
86 printf(" %s", aliases[j].alias);
87 b_alias = true;
88 } else {
89 printf(", %s", aliases[j].alias);
90 }
91 }
92 }
93 printf("\n");
94 }
95
96 printf("\nReadline command line editing (emacs/vi mode) is available.\n"
97"For more detailed help, type 'help COMAMND-NAME' or consult\n"
98"the online-documentation.\n");
99
100 } else {
101 short_cmd_t *p_command;
102 const char *psz_command = "";
103
104 if (1 == strlen(psz_args)) {
105 if ( NULL != short_command[(uint8_t)psz_args[0]].func )
106 p_command = &short_command[(uint8_t)psz_args[0]];
107 else
108 p_command = NULL;
109 } else {
110 psz_command = get_word(&psz_args);
111 p_command = find_command (psz_command);
112 }
113 if (p_command) {
114 if ( p_command->func == &dbg_cmd_info ) {
115 return dbg_help_subcmd("info", p_command, psz_args, info_subcommands);
116 } else if ( p_command->func == &dbg_cmd_show ) {
117 return dbg_help_subcmd("show", p_command, psz_args, show_subcommands);
118 } else if ( p_command->func == &dbg_cmd_set ) {
119 return dbg_help_subcmd("set", p_command, psz_args, set_subcommands);
120 } else {
121 printf("%s\n\n", p_command->use);
122 printf("%s\n", p_command->doc);
123 }
124 } else {
125 printf("Undefined command `%s'. Try help for a list of commands.\n",
126 psz_command);
127 }
128 }
129
130 return debug_readloop;
131}
132
133static void
134dbg_cmd_help_init(unsigned int c)
135{
136 short_command[c].func = &dbg_cmd_help;
137 short_command[c].use = _("help [COMMAND]");
138 short_command[c].doc =
139 _("Display list of commands (i.e. this help text.)\n" \
140 "\twith an command name, give only the help for that command.");
141}
