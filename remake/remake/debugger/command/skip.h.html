19/* Skip over next command or action. */
20static debug_return_t
21dbg_cmd_skip (char *psz_arg)
22{
23 UNUSED_ARGUMENT(psz_arg);
24 return skip_execution;
25}
26
27static void
28dbg_cmd_skip_init(unsigned int c)
29{
30 short_command[c].func = &dbg_cmd_skip;
31 short_command[c].use = _("skip");
32 short_command[c].doc =
33 _("Skip execution of next command or action." );
34}
