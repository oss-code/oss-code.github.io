75debug_return_t
76dbg_cmd_show(char *psz_arg)
77{
78 if (!psz_arg || 0==strlen(psz_arg)) {
79 unsigned int i;
80 for (i=0; show_subcommands[i].name; i++) {
81 if ( 0 == strcmp(show_subcommands[i].name, "warranty") ||
82 0 == strcmp(show_subcommands[i].name, "history"))
83 continue;
84 dbg_help_subcmd_entry("show", "%-10s -- %s",
85 &(show_subcommands[i]), false);
86 }
87 } else {
88 if (is_abbrev_of (psz_arg, "args", 3)) {
89 print_cmdline();
90 } else if (is_abbrev_of (psz_arg, "basename", 4)) {
91 printf("basename is %s.\n", var_to_on_off(basename_filenames));
92 } else if (is_abbrev_of (psz_arg, "debug", 3)) {
93 printf("debug is 0x%x.\n", (unsigned int) db_level);
94 print_db_level(db_level);
95 } else if (is_abbrev_of (psz_arg, "commands", 3)) {
96 dbg_cmd_show_command(psz_arg);
97 } else if (is_abbrev_of (psz_arg, "ignore-errors", 3)) {
98 printf("ignore-errors is %s.\n", var_to_on_off(ignore_errors_flag));
99 } else if (is_abbrev_of (psz_arg, "keep-going", 4)) {
100 printf("keep-going is %s.\n", var_to_on_off(keep_going_flag));
101 } else if (is_abbrev_of (psz_arg, "silent", 3)) {
102 printf("silent is %s.\n", var_to_on_off(silent_flag));
103 } else if (is_abbrev_of (psz_arg, "version", 3)) {
104 printf("version: %s\n", version_string);
105 } else if (is_abbrev_of (psz_arg, "warranty", 3)) {
106 printf("warranty: ");
107 printf("%s", WARRANTY);
108 } else {
109 printf("Undefined command \"%s\". Try \"help show\"\n", psz_arg);
110 }
111 }
112
113 return debug_readloop;
114}
115
116static void
117dbg_cmd_show_init(unsigned int c)
118{
119 short_command[c].func = &dbg_cmd_show;
120 short_command[c].use = _("show [SUBCOMMAND]");
121 short_command[c].doc =
122 _("Show debugger information regarding SUBCOMMAND.\n"
123 "If no SUBCOMMAND is specified, give a list of \"show\" subcommands.");
124}
