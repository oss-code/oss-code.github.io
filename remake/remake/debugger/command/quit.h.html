20/* Terminate execution. */
21static debug_return_t
22dbg_cmd_quit(char *psz_arg)
23{
24 if (!psz_arg || !*psz_arg) {
25 in_debugger = DEBUGGER_QUIT_RC;
26 dbg_msg("remake: That's all, folks...");
27 die(DEBUGGER_QUIT_RC);
28 } else {
29 int rc=0;
30 if (get_int(psz_arg, &rc, true)) {
31 dbg_msg("remake: That's all, folks...");
32 die(rc);
33 }
34 }
35 return debug_readloop;
36}
37
38static void
39dbg_cmd_quit_init(unsigned int c)
40{
41 short_command[c].func = &dbg_cmd_quit;
42 short_command[c].use = _("quit [exit-status]");
43 short_command[c].doc =
44 _("Exit make. If a numeric argument is given, it will be the exit\n"
45 "status reported back. A status of 77 in a nested make will signals\n"
46 "termination in the parent. So if no numeric argument is given and\n"
47 "MAKELEVEL is 0, then status 0 is set; otherwise it is 77."
48 );
49}
