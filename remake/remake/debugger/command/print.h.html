21static debug_return_t
22dbg_cmd_print(char *psz_args)
23{
24 char *psz_name;
25 static char *psz_last_name = NULL;
26
27 if (!psz_args || 0==strlen(psz_args)) {
28 /* Use last value */
29 if (psz_last_name)
30 psz_name = psz_last_name;
31 else {
32 printf("No current variable - must supply something to print\n");
33 return debug_readloop;
34 }
35 } else {
36 psz_name = get_word(&psz_args);
37 }
38
39 if (dbg_cmd_show_exp(psz_name, false)) {
40 if (psz_last_name) free(psz_last_name);
41 psz_last_name = strdup(psz_name);
42 }
43
44 return debug_readloop;
45}
46
47static void
48dbg_cmd_print_init(unsigned int c)
49{
50 short_command[c].func = &dbg_cmd_print;
51 short_command[c].use = _("print {VARIABLE [attrs...]}");
52 short_command[c].doc =
53 _("Show a variable definition.\n"
54 "The value is shown with embedded\n"
55 "variable-references unexpanded. Don't include $ before a variable\n"
56 "name. See also \"examine\".\n\n"
57 "If no variable is supplied, we try to use the\n"
58 "last value given."
59 );
60}
