20debug_return_t
21dbg_cmd_frame_down(char *psz_amount)
22{
23 int i_amount = 1;
24
25 if (!psz_amount || !*psz_amount) {
26 i_amount = 1;
27 } else if (!get_int(psz_amount, &i_amount, true)) {
28 return debug_readloop;
29 }
30 return dbg_adjust_frame(-i_amount, false);
31}
32
33static void
34dbg_cmd_down_init(unsigned int c)
35{
36 short_command[c].func = &dbg_cmd_frame_down;
37 short_command[c].use = _("down [AMOUNT]");
38 short_command[c].doc =
39 _("Select and print the target this one caused to be examined.\n"
40 "\tAn argument says how many targets down to go.");
41}
