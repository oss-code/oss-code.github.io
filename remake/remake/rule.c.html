63/* Compute the maximum dependency length and maximum number of
64 dependencies of all implicit rules. Also sets the subdir
65 flag for a rule when appropriate, possibly removing the rule
66 completely when appropriate. */
67
68void
69count_implicit_rule_limits (void)
70{
71 char *name;
72 int namelen;
73 struct rule *rule;
74
75 num_pattern_rules = max_pattern_targets = max_pattern_deps = 0;
76 max_pattern_dep_length = 0;
77
78 name = 0;
79 namelen = 0;
80 rule = pattern_rules;
81 while (rule != 0)
82 {
83 unsigned int ndeps = 0;
84 struct dep *dep;
85 struct rule *next = rule->next;
86
87 ++num_pattern_rules;
88
89 if (rule->num > max_pattern_targets)
90 max_pattern_targets = rule->num;
91
92 for (dep = rule->deps; dep != 0; dep = dep->next)
93 {
94 const char *dname = dep_name (dep);
95 unsigned int len = strlen (dname);
96
97#ifdef VMS
98 const char *p = strrchr (dname, ']');
99 const char *p2;
100 if (p == 0)
101 p = strrchr (dname, ':');
102 p2 = p != 0 ? strchr (dname, '%') : 0;
103#else
104 const char *p = strrchr (dname, '/');
105 const char *p2 = p != 0 ? strchr (dname, '%') : 0;
106#endif
107 ndeps++;
108
109 if (len > max_pattern_dep_length)
110 max_pattern_dep_length = len;
111
112 if (p != 0 && p2 > p)
113 {
114 /* There is a slash before the % in the dep name.
115 Extract the directory name. */
116 if (p == dname)
117 ++p;
118 if (p - dname > namelen)
119 {
120 namelen = p - dname;
121 name = xrealloc (name, namelen + 1);
122 }
123 memcpy (name, dname, p - dname);
124 name[p - dname] = '\0';
125
126 /* In the deps of an implicit rule the 'changed' flag
127 actually indicates that the dependency is in a
128 nonexistent subdirectory. */
129
130 dep->changed = !dir_file_exists_p (name, "");
131 }
132 else
133 /* This dependency does not reside in a subdirectory. */
134 dep->changed = 0;
135 }
136
137 if (ndeps > max_pattern_deps)
138 max_pattern_deps = ndeps;
139
140 rule = next;
141 }
142
143 free (name);
144}

146/* Create a pattern rule from a suffix rule.
147 TARGET is the target suffix; SOURCE is the source suffix.
148 CMDS are the commands.
149 If TARGET is nil, it means the target pattern should be '(%.o)'.
150 If SOURCE is nil, it means there should be no deps. */
151
152static void
153convert_suffix_rule (const char *target, const char *source,
154 struct commands *cmds)
155{
156 const char **names, **percents;
157 struct dep *deps;
158
159 names = xmalloc (sizeof (const char *));
160 percents = xmalloc (sizeof (const char *));
161
162 if (target == 0)
163 {
164 /* Special case: TARGET being nil means we are defining a '.X.a' suffix
165 rule; the target pattern is always '(%.o)'. */
166#ifdef VMS
167 *names = strcache_add_len ("(%.obj)", 7);
168#else
169 *names = strcache_add_len ("(%.o)", 5);
170#endif
171 *percents = *names + 1;
172 }
173 else
174 {
175 /* Construct the target name. */
176 unsigned int len = strlen (target);
177 char *p = alloca (1 + len + 1);
178 p[0] = '%';
179 memcpy (p + 1, target, len + 1);
180 *names = strcache_add_len (p, len + 1);
181 *percents = *names;
182 }
183
184 if (source == 0)
185 deps = 0;
186 else
187 {
188 /* Construct the dependency name. */
189 unsigned int len = strlen (source);
190 char *p = alloca (1 + len + 1);
191 p[0] = '%';
192 memcpy (p + 1, source, len + 1);
193 deps = alloc_dep ();
194 deps->name = strcache_add_len (p, len + 1);
195 }
196
197 create_pattern_rule (names, percents, 1, 0, deps, cmds, 0);
198}

200/* Convert old-style suffix rules to pattern rules.
201 All rules for the suffixes on the .SUFFIXES list are converted and added to
202 the chain of pattern rules. */
203
204void
205convert_to_pattern (void)
206{
207 struct dep *d, *d2;
208 char *rulename;
209
210 /* We will compute every potential suffix rule (.x.y) from the list of
211 suffixes in the .SUFFIXES target's dependencies and see if it exists.
212 First find the longest of the suffixes. */
213
214 maxsuffix = 0;
215 for (d = suffix_file->deps; d != 0; d = d->next)
216 {
217 unsigned int l = strlen (dep_name (d));
218 if (l > maxsuffix)
219 maxsuffix = l;
220 }
221
222 /* Space to construct the suffix rule target name. */
223 rulename = alloca ((maxsuffix * 2) + 1);
224
225 for (d = suffix_file->deps; d != 0; d = d->next)
226 {
227 unsigned int slen;
228
229 /* Make a rule that is just the suffix, with no deps or commands.
230 This rule exists solely to disqualify match-anything rules. */
231 convert_suffix_rule (dep_name (d), 0, 0);
232
233 if (d->file->cmds != 0)
234 /* Record a pattern for this suffix's null-suffix rule. */
235 convert_suffix_rule ("", dep_name (d), d->file->cmds);
236
237 /* Add every other suffix to this one and see if it exists as a
238 two-suffix rule. */
239 slen = strlen (dep_name (d));
240 memcpy (rulename, dep_name (d), slen);
241
242 for (d2 = suffix_file->deps; d2 != 0; d2 = d2->next)
243 {
244 struct file *f;
245 unsigned int s2len;
246
247 s2len = strlen (dep_name (d2));
248
249 /* Can't build something from itself. */
250 if (slen == s2len && streq (dep_name (d), dep_name (d2)))
251 continue;
252
253 memcpy (rulename + slen, dep_name (d2), s2len + 1);
254 f = lookup_file (rulename);
255 if (f == 0 || f->cmds == 0)
256 continue;
257
258 if (s2len == 2 && rulename[slen] == '.' && rulename[slen + 1] == 'a')
259 /* A suffix rule '.X.a:' generates the pattern rule '(%.o): %.X'.
260 It also generates a normal '%.a: %.X' rule below. */
261 convert_suffix_rule (NULL, /* Indicates '(%.o)'. */
262 dep_name (d),
263 f->cmds);
264
265 /* The suffix rule '.X.Y:' is converted
266 to the pattern rule '%.Y: %.X'. */
267 convert_suffix_rule (dep_name (d2), dep_name (d), f->cmds);
268 }
269 }
270}

273/* Install the pattern rule RULE (whose fields have been filled in) at the end
274 of the list (so that any rules previously defined will take precedence).
275 If this rule duplicates a previous one (identical target and dependencies),
276 the old one is replaced if OVERRIDE is nonzero, otherwise this new one is
277 thrown out. When an old rule is replaced, the new one is put at the end of
278 the list. Return nonzero if RULE is used; zero if not. */
279
280static int
281new_pattern_rule (struct rule *rule, int override)
282{
283 struct rule *r, *lastrule;
284 unsigned int i, j;
285
286 rule->in_use = 0;
287 rule->terminal = 0;
288
289 rule->next = 0;
290
291 /* Search for an identical rule. */
292 lastrule = 0;
293 for (r = pattern_rules; r != 0; lastrule = r, r = r->next)
294 for (i = 0; i < rule->num; ++i)
295 {
296 for (j = 0; j < r->num; ++j)
297 if (!streq (rule->targets[i], r->targets[j]))
298 break;
299 /* If all the targets matched... */
300 if (j == r->num)
301 {
302 struct dep *d, *d2;
303 for (d = rule->deps, d2 = r->deps;
304 d != 0 && d2 != 0; d = d->next, d2 = d2->next)
305 if (!streq (dep_name (d), dep_name (d2)))
306 break;
307 if (d == 0 && d2 == 0)
308 {
309 /* All the dependencies matched. */
310 if (override)
311 {
312 /* Remove the old rule. */
313 freerule (r, lastrule);
314 /* Install the new one. */
315 if (pattern_rules == 0)
316 pattern_rules = rule;
317 else
318 last_pattern_rule->next = rule;
319 last_pattern_rule = rule;
320
321 /* We got one. Stop looking. */
322 goto matched;
323 }
324 else
325 {
326 /* The old rule stays intact. Destroy the new one. */
327 freerule (rule, (struct rule *) 0);
328 return 0;
329 }
330 }
331 }
332 }
333
334 matched:;
335
336 if (r == 0)
337 {
338 /* There was no rule to replace. */
339 if (pattern_rules == 0)
340 pattern_rules = rule;
341 else
342 last_pattern_rule->next = rule;
343 last_pattern_rule = rule;
344 }
345
346 return 1;
347}

350/* Install an implicit pattern rule based on the three text strings
351 in the structure P points to. These strings come from one of
352 the arrays of default implicit pattern rules.
353 TERMINAL specifies what the 'terminal' field of the rule should be. */
354
355void
356install_pattern_rule (struct pspec *p, int terminal)
357{
358 struct rule *r;
359 const char *ptr;
360
361 r = xmalloc (sizeof (struct rule));
362
363 r->num = 1;
364 r->targets = xmalloc (sizeof (const char *));
365 r->suffixes = xmalloc (sizeof (const char *));
366 r->lens = xmalloc (sizeof (unsigned int));
367
368 r->lens[0] = strlen (p->target);
369 r->targets[0] = p->target;
370 r->suffixes[0] = find_percent_cached (&r->targets[0]);
371 r->floc.filenm = NULL;
372 r->floc.lineno = 0;
373 assert (r->suffixes[0] != NULL);
374 ++r->suffixes[0];
375
376 ptr = p->dep;
377 r->deps = PARSE_SIMPLE_SEQ ((char **)&ptr, struct dep);
378
379 if (new_pattern_rule (r, 0))
380 {
381 r->terminal = terminal;
382 r->cmds = xmalloc (sizeof (struct commands));
383 r->cmds->fileinfo.filenm = 0;
384 r->cmds->fileinfo.lineno = 0;
385 r->cmds->fileinfo.offset = 0;
386 /* These will all be string literals, but we malloc space for them
387 anyway because somebody might want to free them later. */
388 r->cmds->commands = xstrdup (p->commands);
389 r->cmds->command_lines = 0;
390 r->cmds->recipe_prefix = RECIPEPREFIX_DEFAULT;
391 }
392}

395/* Free all the storage used in RULE and take it out of the
396 pattern_rules chain. LASTRULE is the rule whose next pointer
397 points to RULE. */
398
399static void
400freerule (struct rule *rule, struct rule *lastrule)
401{
402 struct rule *next = rule->next;
403
404 free_dep_chain (rule->deps);
405
406 /* MSVC erroneously warns without a cast here. */
407 free ((void *)rule->targets);
408 free ((void *)rule->suffixes);
409 free (rule->lens);
410
411 /* We can't free the storage for the commands because there
412 are ways that they could be in more than one place:
413 * If the commands came from a suffix rule, they could also be in
414 the 'struct file's for other suffix rules or plain targets given
415 on the same makefile line.
416 * If two suffixes that together make a two-suffix rule were each
417 given twice in the .SUFFIXES list, and in the proper order, two
418 identical pattern rules would be created and the second one would
419 be discarded here, but both would contain the same 'struct commands'
420 pointer from the 'struct file' for the suffix rule. */
421
422 free (rule);
423
424 if (pattern_rules == rule)
425 if (lastrule != 0)
426 abort ();
427 else
428 pattern_rules = next;
429 else if (lastrule != 0)
430 lastrule->next = next;
431 if (last_pattern_rule == rule)
432 last_pattern_rule = lastrule;
433}

435/* Create a new pattern rule with the targets in the nil-terminated array
436 TARGETS. TARGET_PERCENTS is an array of pointers to the % in each element
437 of TARGETS. N is the number of items in the array (not counting the nil
438 element). The new rule has dependencies DEPS and commands from COMMANDS.
439 It is a terminal rule if TERMINAL is nonzero. This rule overrides
440 identical rules with different commands if OVERRIDE is nonzero.
441
442 The storage for TARGETS and its elements and TARGET_PERCENTS is used and
443 must not be freed until the rule is destroyed. */
444
445void
446create_pattern_rule (const char **targets, const char **target_percents,
447 unsigned int n, int terminal, struct dep *deps,
448 struct commands *commands, int override)
449{
450 unsigned int i;
451 struct rule *r = xmalloc (sizeof (struct rule));
452
453 r->num = n;
454 r->cmds = commands;
455 r->deps = deps;
456 r->targets = targets;
457 r->suffixes = target_percents;
458 r->lens = xmalloc (n * sizeof (unsigned int));
459
460 r->tracing = BRK_NONE;
461 if (commands) {
462 r->floc.filenm = commands->fileinfo.filenm;
463 r->floc.lineno = commands->fileinfo.lineno - 1;
464 } else {
465 r->floc.filenm = NULL;
466 r->floc.lineno = 0;
467 }
468
469 for (i = 0; i < n; ++i)
470 {
471 r->lens[i] = strlen (targets[i]);
472 assert (r->suffixes[i] != NULL);
473 ++r->suffixes[i];
474 }
475
476 if (new_pattern_rule (r, override))
477 r->terminal = terminal;
478}

480/*! Show information about a given rule. Useful from the debugger or gdb. */
481void
482print_rule (rule_t *r, bool b_verbose)
483{
484 unsigned int i;
485
486 for (i = 0; i < r->num; ++i)
487 {
488 fputs (r->targets[i], stdout);
489 putchar ((i + 1 == r->num) ? ':' : ' ');
490 }
491 if (r->terminal)
492 putchar (':');
493
494 print_prereqs (r->deps);
495
496 putchar ('\n');
497 if (!b_verbose) return;
498 if (r->cmds != 0)
499 print_commands (NULL, r->cmds, false);
500}

502/* Print the data base of rules. */
503void
504print_rule_data_base (bool b_verbose)
505{
506 unsigned int rules, terminal;
507 struct rule *r;
508
509 puts (_("\n# Implicit Rules"));
510
511 rules = terminal = 0;
512 for (r = pattern_rules; r != 0; r = r->next)
513 {
514 ++rules;
515
516 print_rule (r, b_verbose);
517
518 if (r->terminal)
519 ++terminal;
520 }
521
522 if (!b_verbose) {
523 printf("\n");
524 return;
525 }
526
527 if (rules == 0)
528 puts (_("\n# No implicit rules."));
529 else
530 {
531 printf (_("\n# %u implicit rules, %u"), rules, terminal);
532#ifndef NO_FLOAT
533 printf (" (%.1f%%)", (double) terminal / (double) rules * 100.0);
534#else
535 {
536 int f = (terminal * 1000 + 5) / rules;
537 printf (" (%d.%d%%)", f/10, f%10);
538 }
539#endif
540 puts (_(" terminal."));
541 }
542
543 if (num_pattern_rules != rules)
544 {
545 /* This can happen if a fatal error was detected while reading the
546 makefiles and thus count_implicit_rule_limits wasn't called yet. */
547 if (num_pattern_rules != 0)
548 ONN (fatal, NILF, _("BUG: num_pattern_rules is wrong! %u != %u"),
549 num_pattern_rules, rules);
550 }
