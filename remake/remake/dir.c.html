71static const char *
72dosify (const char *filename)
73{
74 static char dos_filename[14];
75 char *df;
76 int i;
77
78 if (filename == 0 || _USE_LFN)
79 return filename;
80
81 /* FIXME: what about filenames which violate
82 8+3 constraints, like "config.h.in", or ".emacs"? */
83 if (strpbrk (filename, "\"*+,;<=>?[\\]|") != 0)
84 return filename;
85
86 df = dos_filename;
87
88 /* First, transform the name part. */
89 for (i = 0; i < 8 && ! STOP_SET (*filename, MAP_DOT|MAP_NUL); ++i)
90 *df++ = tolower ((unsigned char)*filename++);
91
92 /* Now skip to the next dot. */
93 while (! STOP_SET (*filename, MAP_DOT|MAP_NUL))
94 ++filename;
95 if (*filename != '\0')
96 {
97 *df++ = *filename++;
98 for (i = 0; i < 3 && ! STOP_SET (*filename, MAP_DOT|MAP_NUL); ++i)
99 *df++ = tolower ((unsigned char)*filename++);
100 }
101
102 /* Look for more dots. */
103 while (! STOP_SET (*filename, MAP_DOT|MAP_NUL))
104 ++filename;
105 if (*filename == '.')
106 return filename;
107 *df = 0;
108 return dos_filename;
109}
110#endif /* __MSDOS__ */

121static const char *
122downcase (const char *filename)
123{
124 static PATH_VAR (new_filename);
125 char *df;
126
127 if (filename == 0)
128 return 0;
129
130 df = new_filename;
131 while (*filename != '\0')
132 {
133 *df++ = tolower ((unsigned char)*filename);
134 ++filename;
135 }
136
137 *df = 0;
138
139 return new_filename;
140}
141#endif /* HAVE_CASE_INSENSITIVE_FS */

171static int
172vms_hash (const char *name)
173{
174 int h = 0;
175
176 while (*name)
177 {
178 unsigned char uc = *name;
179 int g;
180#ifdef HAVE_CASE_INSENSITIVE_FS
181 h = (h << 4) + (isupper (uc) ? tolower (uc) : uc);
182#else
183 h = (h << 4) + uc;
184#endif
185 name++;
186 g = h & 0xf0000000;
187 if (g)
188 {
189 h = h ^ (g >> 24);
190 h = h ^ g;
191 }
192 }
193 return h;
194}

196/* fake stat entry for a directory */
197static int
198vmsstat_dir (const char *name, struct stat *st)
199{
200 char *s;
201 int h;
202 DIR *dir;
203
204 dir = opendir (name);
205 if (dir == 0)
206 return -1;
207 closedir (dir);
208 s = strchr (name, ':'); /* find device */
209 if (s)
210 {
211 /* to keep the compiler happy we said "const char *name", now we cheat */
212 *s++ = 0;
213 st->st_dev = (char *)vms_hash (name);
214 h = vms_hash (s);
215 *(s-1) = ':';
216 }
217 else
218 {
219 st->st_dev = 0;
220 h = vms_hash (name);
221 }
222
223 st->st_ino[0] = h & 0xff;
224 st->st_ino[1] = h & 0xff00;
225 st->st_ino[2] = h >> 16;
226
227 return 0;
228}

241struct directory_contents
242 {
243 dev_t dev; /* Device and inode numbers of this dir. */
244#ifdef WINDOWS32
245 /* Inode means nothing on WINDOWS32. Even file key information is
246 * unreliable because it is random per file open and undefined for remote
247 * filesystems. The most unique attribute I can come up with is the fully
248 * qualified name of the directory. Beware though, this is also
249 * unreliable. I'm open to suggestion on a better way to emulate inode. */
250 char *path_key;
251 time_t ctime;
252 time_t mtime; /* controls check for stale directory cache */
253 int fs_flags; /* FS_FAT, FS_NTFS, ... */
254# define FS_FAT 0x1
255# define FS_NTFS 0x2
256# define FS_UNKNOWN 0x4
257#else
258# ifdef VMS_INO_T
259 ino_t ino[3];
260# else
261 ino_t ino;
262# endif
263#endif /* WINDOWS32 */
264 struct hash_table dirfiles; /* Files in this directory. */
265 DIR *dirstream; /* Stream reading this directory. */
266 };

268static unsigned long
269directory_contents_hash_1 (const void *key_0)
270{
271 const struct directory_contents *key = key_0;
272 unsigned long hash;
273
274#ifdef WINDOWS32
275 hash = 0;
276 ISTRING_HASH_1 (key->path_key, hash);
277 hash ^= ((unsigned int) key->dev << 4) ^ (unsigned int) key->ctime;
278#else
279# ifdef VMS_INO_T
280 hash = (((unsigned int) key->dev << 4)
281 ^ ((unsigned int) key->ino[0]
282 + (unsigned int) key->ino[1]
283 + (unsigned int) key->ino[2]));
284# else
285 hash = ((unsigned int) key->dev << 4) ^ (unsigned int) key->ino;
286# endif
287#endif /* WINDOWS32 */
288 return hash;
289}

291static unsigned long
292directory_contents_hash_2 (const void *key_0)
293{
294 const struct directory_contents *key = key_0;
295 unsigned long hash;
296
297#ifdef WINDOWS32
298 hash = 0;
299 ISTRING_HASH_2 (key->path_key, hash);
300 hash ^= ((unsigned int) key->dev << 4) ^ (unsigned int) ~key->ctime;
301#else
302# ifdef VMS_INO_T
303 hash = (((unsigned int) key->dev << 4)
304 ^ ~((unsigned int) key->ino[0]
305 + (unsigned int) key->ino[1]
306 + (unsigned int) key->ino[2]));
307# else
308 hash = ((unsigned int) key->dev << 4) ^ (unsigned int) ~key->ino;
309# endif
310#endif /* WINDOWS32 */
311
312 return hash;
313}

315/* Sometimes it's OK to use subtraction to get this value:
316 result = X - Y;
317 But, if we're not sure of the type of X and Y they may be too large for an
318 int (on a 64-bit system for example). So, use ?: instead.
319 See Savannah bug #15534.
320
321 NOTE! This macro has side-effects!
322*/
323
324#define MAKECMP(_x,_y) ((_x)<(_y)?-1:((_x)==(_y)?0:1))
325
326static int
327directory_contents_hash_cmp (const void *xv, const void *yv)
328{
329 const struct directory_contents *x = xv;
330 const struct directory_contents *y = yv;
331 int result;
332
333#ifdef WINDOWS32
334 ISTRING_COMPARE (x->path_key, y->path_key, result);
335 if (result)
336 return result;
337 result = MAKECMP(x->ctime, y->ctime);
338 if (result)
339 return result;
340#else
341# ifdef VMS_INO_T
342 result = MAKECMP(x->ino[0], y->ino[0]);
343 if (result)
344 return result;
345 result = MAKECMP(x->ino[1], y->ino[1]);
346 if (result)
347 return result;
348 result = MAKECMP(x->ino[2], y->ino[2]);
349 if (result)
350 return result;
351# else
352 result = MAKECMP(x->ino, y->ino);
353 if (result)
354 return result;
355# endif
356#endif /* WINDOWS32 */
357
358 return MAKECMP(x->dev, y->dev);
359}

425dirfile_hash_cmp (const void *xv, const void *yv)
426{
427 const struct dirfile *x = xv;

443/* Find the directory named NAME and return its 'struct directory'. */
444
445static struct directory *
446find_directory (const char *name)
447{
448 struct directory *dir;
449 struct directory **dir_slot;
450 struct directory dir_key;
451
452 dir_key.name = name;
453 dir_slot = (struct directory **) hash_find_slot (&directories, &dir_key);
454 dir = *dir_slot;
455
456 if (HASH_VACANT (dir))
457 {
458 /* The directory was not found. Create a new entry for it. */
459 const char *p = name + strlen (name);
460 struct stat st;
461 int r;
462
463 dir = xmalloc (sizeof (struct directory));
464#if defined(HAVE_CASE_INSENSITIVE_FS) && defined(VMS)
465 /* Todo: Why is this only needed on VMS? */
466 {
467 char *lname = downcase_inplace (xstrdup (name));
468 dir->name = strcache_add_len (lname, p - name);
469 free (lname);
470 }
471#else
472 dir->name = strcache_add_len (name, p - name);
473#endif
474 hash_insert_at (&directories, dir, dir_slot);
475 /* The directory is not in the name hash table.
476 Find its device and inode numbers, and look it up by them. */
477
478#if defined(WINDOWS32)
479 {
480 char tem[MAXPATHLEN], *tstart, *tend;
481
482 /* Remove any trailing slashes. Windows32 stat fails even on
483 valid directories if they end in a slash. */
484 memcpy (tem, name, p - name + 1);
485 tstart = tem;
486 if (tstart[1] == ':')
487 tstart += 2;
488 for (tend = tem + (p - name - 1);
489 tend > tstart && (*tend == '/' || *tend == '\\');
490 tend--)
491 *tend = '\0';
492
493 r = stat (tem, &st);
494 }
495#else
496 EINTRLOOP (r, stat (name, &st));
497#endif
498
499 if (r < 0)
500 {
501 /* Couldn't stat the directory. Mark this by
502 setting the 'contents' member to a nil pointer. */
503 dir->contents = 0;
504 }
505 else
506 {
507 /* Search the contents hash table; device and inode are the key. */
508
509#ifdef WINDOWS32
510 char *w32_path;
511#endif
512 struct directory_contents *dc;
513 struct directory_contents **dc_slot;
514 struct directory_contents dc_key;
515
516 dc_key.dev = st.st_dev;
517#ifdef WINDOWS32
518 dc_key.path_key = w32_path = w32ify (name, 1);
519 dc_key.ctime = st.st_ctime;
520#else
521# ifdef VMS_INO_T
522 dc_key.ino[0] = st.st_ino[0];
523 dc_key.ino[1] = st.st_ino[1];
524 dc_key.ino[2] = st.st_ino[2];
525# else
526 dc_key.ino = st.st_ino;
527# endif
528#endif
529 dc_slot = (struct directory_contents **) hash_find_slot (&directory_contents, &dc_key);
530 dc = *dc_slot;
531
532 if (HASH_VACANT (dc))
533 {
534 /* Nope; this really is a directory we haven't seen before. */
535#ifdef WINDOWS32
536 char fs_label[BUFSIZ];
537 char fs_type[BUFSIZ];
538 unsigned long fs_serno;
539 unsigned long fs_flags;
540 unsigned long fs_len;
541#endif
542 dc = (struct directory_contents *)
543 xmalloc (sizeof (struct directory_contents));
544
545 /* Enter it in the contents hash table. */
546 dc->dev = st.st_dev;
547#ifdef WINDOWS32
548 dc->path_key = xstrdup (w32_path);
549 dc->ctime = st.st_ctime;
550 dc->mtime = st.st_mtime;
551
552 /* NTFS is the only WINDOWS32 filesystem that bumps mtime on a
553 directory when files are added/deleted from a directory. */
554 w32_path[3] = '\0';
555 if (GetVolumeInformation (w32_path, fs_label, sizeof (fs_label),
556 &fs_serno, &fs_len, &fs_flags, fs_type,
557 sizeof (fs_type)) == FALSE)
558 dc->fs_flags = FS_UNKNOWN;
559 else if (!strcmp (fs_type, "FAT"))
560 dc->fs_flags = FS_FAT;
561 else if (!strcmp (fs_type, "NTFS"))
562 dc->fs_flags = FS_NTFS;
563 else
564 dc->fs_flags = FS_UNKNOWN;
565#else
566# ifdef VMS_INO_T
567 dc->ino[0] = st.st_ino[0];
568 dc->ino[1] = st.st_ino[1];
569 dc->ino[2] = st.st_ino[2];
570# else
571 dc->ino = st.st_ino;
572# endif
573#endif /* WINDOWS32 */
574 hash_insert_at (&directory_contents, dc, dc_slot);
575 ENULLLOOP (dc->dirstream, opendir (name));
576 if (dc->dirstream == 0)
577 /* Couldn't open the directory. Mark this by setting the
578 'files' member to a nil pointer. */
579 dc->dirfiles.ht_vec = 0;
580 else
581 {
582 hash_init (&dc->dirfiles, DIRFILE_BUCKETS,
583 dirfile_hash_1, dirfile_hash_2, dirfile_hash_cmp);
584 /* Keep track of how many directories are open. */
585 ++open_directories;
586 if (open_directories == MAX_OPEN_DIRECTORIES)
587 /* We have too many directories open already.
588 Read the entire directory and then close it. */
589 dir_contents_file_exists_p (dc, 0);
590 }
591 }
592
593 /* Point the name-hashed entry for DIR at its contents data. */
594 dir->contents = dc;
595 }
596 }
597
598 return dir;
599}

601/* Return 1 if the name FILENAME is entered in DIR's hash table.
602 FILENAME must contain no slashes. */
603
604static int
605dir_contents_file_exists_p (struct directory_contents *dir,
606 const char *filename)
607{
608 struct dirfile *df;
609 struct dirent *d;
610#ifdef WINDOWS32
611 struct stat st;
612 int rehash = 0;
613#endif
614
615 if (dir == 0 || dir->dirfiles.ht_vec == 0)
616 /* The directory could not be stat'd or opened. */
617 return 0;
618
619#ifdef __MSDOS__
620 filename = dosify (filename);
621#endif
622
623#ifdef HAVE_CASE_INSENSITIVE_FS
624 filename = downcase (filename);
625#endif
626
627#ifdef __EMX__
628 if (filename != 0)
629 _fnlwr (filename); /* lower case for FAT drives */
630#endif
631 if (filename != 0)
632 {
633 struct dirfile dirfile_key;
634
635 if (*filename == '\0')
636 {
637 /* Checking if the directory exists. */
638 return 1;
639 }
640 dirfile_key.name = filename;
641 dirfile_key.length = strlen (filename);
642 df = hash_find_item (&dir->dirfiles, &dirfile_key);
643 if (df)
644 return !df->impossible;
645 }
646
647 /* The file was not found in the hashed list.
648 Try to read the directory further. */
649
650 if (dir->dirstream == 0)
651 {
652#ifdef WINDOWS32
653 /*
654 * Check to see if directory has changed since last read. FAT
655 * filesystems force a rehash always as mtime does not change
656 * on directories (ugh!).
657 */
658 if (dir->path_key)
659 {
660 if ((dir->fs_flags & FS_FAT) != 0)
661 {
662 dir->mtime = time ((time_t *) 0);
663 rehash = 1;
664 }
665 else if (stat (dir->path_key, &st) == 0 && st.st_mtime > dir->mtime)
666 {
667 /* reset date stamp to show most recent re-process. */
668 dir->mtime = st.st_mtime;
669 rehash = 1;
670 }
671
672 /* If it has been already read in, all done. */
673 if (!rehash)
674 return 0;
675
676 /* make sure directory can still be opened; if not return. */
677 dir->dirstream = opendir (dir->path_key);
678 if (!dir->dirstream)
679 return 0;
680 }
681 else
682#endif
683 /* The directory has been all read in. */
684 return 0;
685 }
686
687 while (1)
688 {
689 /* Enter the file in the hash table. */
690 unsigned int len;
691 struct dirfile dirfile_key;
692 struct dirfile **dirfile_slot;
693
694 ENULLLOOP (d, readdir (dir->dirstream));
695 if (d == 0)
696 {
697 if (errno)
698 pfatal_with_name ("INTERNAL: readdir");
699 break;
700 }
701
702#if defined(VMS) && defined(HAVE_DIRENT_H)
703 /* In VMS we get file versions too, which have to be stripped off.
704 Some versions of VMS return versions on Unix files even when
705 the feature option to strip them is set. */
706 {
707 char *p = strrchr (d->d_name, ';');
708 if (p)
709 *p = '\0';
710 }
711#endif
712 if (!REAL_DIR_ENTRY (d))
713 continue;
714
715 len = NAMLEN (d);
716 dirfile_key.name = d->d_name;
717 dirfile_key.length = len;
718 dirfile_slot = (struct dirfile **) hash_find_slot (&dir->dirfiles, &dirfile_key);
719#ifdef WINDOWS32
720 /*
721 * If re-reading a directory, don't cache files that have
722 * already been discovered.
723 */
724 if (! rehash || HASH_VACANT (*dirfile_slot))
725#endif
726 {
727 df = xmalloc (sizeof (struct dirfile));
728#if defined(HAVE_CASE_INSENSITIVE_FS) && defined(VMS)
729 /* TODO: Why is this only needed on VMS? */
730 df->name = strcache_add_len (downcase_inplace (d->d_name), len);
731#else
732 df->name = strcache_add_len (d->d_name, len);
733#endif
734 df->length = len;
735 df->impossible = 0;
736 hash_insert_at (&dir->dirfiles, df, dirfile_slot);
737 }
738 /* Check if the name matches the one we're searching for. */
739 if (filename != 0 && patheq (d->d_name, filename))
740 return 1;
741 }
742
743 /* If the directory has been completely read in,
744 close the stream and reset the pointer to nil. */
745 if (d == 0)
746 {
747 --open_directories;
748 closedir (dir->dirstream);
749 dir->dirstream = 0;
750 }
751 return 0;
752}

754/* Return 1 if the name FILENAME in directory DIRNAME
755 is entered in the dir hash table.
756 FILENAME must contain no slashes. */
757
758int
759dir_file_exists_p (const char *dirname, const char *filename)
760{
761#ifdef VMS
762 if ((filename != NULL) && (dirname != NULL))
763 {
764 int want_vmsify;
765 want_vmsify = (strpbrk (dirname, ":<[") != NULL);
766 if (want_vmsify)
767 filename = vmsify (filename, 0);
768 }
769#endif
770 return dir_contents_file_exists_p (find_directory (dirname)->contents,
771 filename);
772}

774/* Return 1 if the file named NAME exists. */
775
776int
777file_exists_p (const char *name)
778{
779 const char *dirend;
780 const char *dirname;
781 const char *slash;
782
783#ifndef NO_ARCHIVES
784 {
785 time_t member_date;
786 if (ar_name (name))
787 return ar_member_date (name, &member_date);
788 }
789#endif
790
791 dirend = strrchr (name, '/');
792#ifdef VMS
793 if (dirend == 0)
794 {
795 dirend = strrchr (name, ']');
796 dirend == NULL ? dirend : dirend++;
797 }
798 if (dirend == 0)
799 {
800 dirend = strrchr (name, '>');
801 dirend == NULL ? dirend : dirend++;
802 }
803 if (dirend == 0)
804 {
805 dirend = strrchr (name, ':');
806 dirend == NULL ? dirend : dirend++;
807 }
808#endif /* VMS */
809#ifdef HAVE_DOS_PATHS
810 /* Forward and backslashes might be mixed. We need the rightmost one. */
811 {
812 const char *bslash = strrchr (name, '\\');
813 if (!dirend || bslash > dirend)
814 dirend = bslash;
815 /* The case of "d:file". */
816 if (!dirend && name[0] && name[1] == ':')
817 dirend = name + 1;
818 }
819#endif /* HAVE_DOS_PATHS */
820 if (dirend == 0)
821#ifndef _AMIGA
822 return dir_file_exists_p (".", name);
823#else /* !AMIGA */
824 return dir_file_exists_p ("", name);
825#endif /* AMIGA */
826
827 slash = dirend;
828 if (dirend == name)
829 dirname = "/";
830 else
831 {
832 char *p;
833#ifdef HAVE_DOS_PATHS
834 /* d:/ and d: are *very* different... */
835 if (dirend < name + 3 && name[1] == ':' &&
836 (*dirend == '/' || *dirend == '\\' || *dirend == ':'))
837 dirend++;
838#endif
839 p = alloca (dirend - name + 1);
840 memcpy (p, name, dirend - name);
841 p[dirend - name] = '\0';
842 dirname = p;
843 }
844#ifdef VMS
845 if (*slash == '/')
846 slash++;
847#else
848 slash++;
849#endif
850 return dir_file_exists_p (dirname, slash);
851}

853/* Mark FILENAME as 'impossible' for 'file_impossible_p'.
854 This means an attempt has been made to search for FILENAME
855 as an intermediate file, and it has failed. */
856
857void
858file_impossible (const char *filename)
859{
860 const char *dirend;
861 const char *p = filename;
862 struct directory *dir;
863 struct dirfile *new;
864
865 dirend = strrchr (p, '/');
866#ifdef VMS
867 if (dirend == NULL)
868 {
869 dirend = strrchr (p, ']');
870 dirend == NULL ? dirend : dirend++;
871 }
872 if (dirend == NULL)
873 {
874 dirend = strrchr (p, '>');
875 dirend == NULL ? dirend : dirend++;
876 }
877 if (dirend == NULL)
878 {
879 dirend = strrchr (p, ':');
880 dirend == NULL ? dirend : dirend++;
881 }
882#endif
883#ifdef HAVE_DOS_PATHS
884 /* Forward and backslashes might be mixed. We need the rightmost one. */
885 {
886 const char *bslash = strrchr (p, '\\');
887 if (!dirend || bslash > dirend)
888 dirend = bslash;
889 /* The case of "d:file". */
890 if (!dirend && p[0] && p[1] == ':')
891 dirend = p + 1;
892 }
893#endif /* HAVE_DOS_PATHS */
894 if (dirend == 0)
895#ifdef _AMIGA
896 dir = find_directory ("");
897#else /* !AMIGA */
898 dir = find_directory (".");
899#endif /* AMIGA */
900 else
901 {
902 const char *dirname;
903 const char *slash = dirend;
904 if (dirend == p)
905 dirname = "/";
906 else
907 {
908 char *cp;
909#ifdef HAVE_DOS_PATHS
910 /* d:/ and d: are *very* different... */
911 if (dirend < p + 3 && p[1] == ':' &&
912 (*dirend == '/' || *dirend == '\\' || *dirend == ':'))
913 dirend++;
914#endif
915 cp = alloca (dirend - p + 1);
916 memcpy (cp, p, dirend - p);
917 cp[dirend - p] = '\0';
918 dirname = cp;
919 }
920 dir = find_directory (dirname);
921#ifdef VMS
922 if (*slash == '/')
923 filename = p = slash + 1;
924 else
925 filename = p = slash;
926#else
927 filename = p = slash + 1;
928#endif
929 }
930
931 if (dir->contents == 0)
932 /* The directory could not be stat'd. We allocate a contents
933 structure for it, but leave it out of the contents hash table. */
934 dir->contents = xcalloc (sizeof (struct directory_contents));
935
936 if (dir->contents->dirfiles.ht_vec == 0)
937 {
938 hash_init (&dir->contents->dirfiles, DIRFILE_BUCKETS,
939 dirfile_hash_1, dirfile_hash_2, dirfile_hash_cmp);
940 }
941
942 /* Make a new entry and put it in the table. */
943
944 new = xmalloc (sizeof (struct dirfile));
945 new->length = strlen (filename);
946#if defined(HAVE_CASE_INSENSITIVE_FS) && defined(VMS)
947 /* todo: Why is this only needed on VMS? */
948 new->name = strcache_add_len (downcase (filename), new->length);
949#else
950 new->name = strcache_add_len (filename, new->length);
951#endif
952 new->impossible = 1;
953 hash_insert (&dir->contents->dirfiles, new);
954}

956/* Return nonzero if FILENAME has been marked impossible. */
957
958int
959file_impossible_p (const char *filename)
960{
961 const char *dirend;
962 struct directory_contents *dir;
963 struct dirfile *dirfile;
964 struct dirfile dirfile_key;
965#ifdef VMS
966 int want_vmsify = 0;
967#endif
968
969 dirend = strrchr (filename, '/');
970#ifdef VMS
971 if (dirend == NULL)
972 {
973 want_vmsify = (strpbrk (filename, "]>:^") != NULL);
974 dirend = strrchr (filename, ']');
975 }
976 if (dirend == NULL && want_vmsify)
977 dirend = strrchr (filename, '>');
978 if (dirend == NULL && want_vmsify)
979 dirend = strrchr (filename, ':');
980#endif
981#ifdef HAVE_DOS_PATHS
982 /* Forward and backslashes might be mixed. We need the rightmost one. */
983 {
984 const char *bslash = strrchr (filename, '\\');
985 if (!dirend || bslash > dirend)
986 dirend = bslash;
987 /* The case of "d:file". */
988 if (!dirend && filename[0] && filename[1] == ':')
989 dirend = filename + 1;
990 }
991#endif /* HAVE_DOS_PATHS */
992 if (dirend == 0)
993#ifdef _AMIGA
994 dir = find_directory ("")->contents;
995#else /* !AMIGA */
996 dir = find_directory (".")->contents;
997#endif /* AMIGA */
998 else
999 {
1000 const char *dirname;
1001 const char *slash = dirend;
1002 if (dirend == filename)
1003 dirname = "/";
1004 else
1005 {
1006 char *cp;
1007#ifdef HAVE_DOS_PATHS
1008 /* d:/ and d: are *very* different... */
1009 if (dirend < filename + 3 && filename[1] == ':' &&
1010 (*dirend == '/' || *dirend == '\\' || *dirend == ':'))
1011 dirend++;
1012#endif
1013 cp = alloca (dirend - filename + 1);
1014 memcpy (cp, filename, dirend - filename);
1015 cp[dirend - filename] = '\0';
1016 dirname = cp;
1017 }
1018 dir = find_directory (dirname)->contents;
1019#ifdef VMS
1020 if (*slash == '/')
1021 filename = slash + 1;
1022 else
1023 filename = slash;
1024#else
1025 filename = slash + 1;
1026#endif
1027 }
1028
1029 if (dir == 0 || dir->dirfiles.ht_vec == 0)
1030 /* There are no files entered for this directory. */
1031 return 0;
1032
1033#ifdef __MSDOS__
1034 filename = dosify (filename);
1035#endif
1036#ifdef HAVE_CASE_INSENSITIVE_FS
1037 filename = downcase (filename);
1038#endif
1039#ifdef VMS
1040 if (want_vmsify)
1041 filename = vmsify (filename, 1);
1042#endif
1043
1044 dirfile_key.name = filename;
1045 dirfile_key.length = strlen (filename);
1046 dirfile = hash_find_item (&dir->dirfiles, &dirfile_key);
1047 if (dirfile)
1048 return dirfile->impossible;
1049
1050 return 0;
1051}

1101#endif
1102#endif /* WINDOWS32 */

1177/* Structure describing state of iterating through a directory hash table. */
1178
1179struct dirstream
1180 {
1181 struct directory_contents *contents; /* The directory being read. */
1182 struct dirfile **dirfile_slot; /* Current slot in table. */
1183 };
1184
1185/* Forward declarations. */
1186static __ptr_t open_dirstream (const char *);
1187static struct dirent *read_dirstream (__ptr_t);
1188
1189static __ptr_t
1190open_dirstream (const char *directory)
1191{
1192 struct dirstream *new;
1193 struct directory *dir = find_directory (directory);
1194
1195 if (dir->contents == 0 || dir->contents->dirfiles.ht_vec == 0)
1196 /* DIR->contents is nil if the directory could not be stat'd.
1197 DIR->contents->dirfiles is nil if it could not be opened. */
1198 return 0;
1199
1200 /* Read all the contents of the directory now. There is no benefit
1201 in being lazy, since glob will want to see every file anyway. */
1202
1203 dir_contents_file_exists_p (dir->contents, 0);
1204
1205 new = xmalloc (sizeof (struct dirstream));
1206 new->contents = dir->contents;
1207 new->dirfile_slot = (struct dirfile **) new->contents->dirfiles.ht_vec;
1208
1209 return (__ptr_t) new;
1210}
1211
1212static struct dirent *
1213read_dirstream (__ptr_t stream)
1214{
1215 static char *buf;
1216 static unsigned int bufsz;
1217
1218 struct dirstream *const ds = (struct dirstream *) stream;
1219 struct directory_contents *dc = ds->contents;
1220 struct dirfile **dirfile_end = (struct dirfile **) dc->dirfiles.ht_vec + dc->dirfiles.ht_size;
1221
1222 while (ds->dirfile_slot < dirfile_end)
1223 {
1224 struct dirfile *df = *ds->dirfile_slot++;
1225 if (! HASH_VACANT (df) && !df->impossible)
1226 {
1227 /* The glob interface wants a 'struct dirent', so mock one up. */
1228 struct dirent *d;
1229 unsigned int len = df->length + 1;
1230 unsigned int sz = sizeof (*d) - sizeof (d->d_name) + len;
1231 if (sz > bufsz)
1232 {
1233 bufsz *= 2;
1234 if (sz > bufsz)
1235 bufsz = sz;
1236 buf = xrealloc (buf, bufsz);
1237 }
1238 d = (struct dirent *) buf;
1239#ifdef __MINGW32__
1240# if __MINGW32_MAJOR_VERSION < 3 || (__MINGW32_MAJOR_VERSION == 3 && \
1241 __MINGW32_MINOR_VERSION == 0)
1242 d->d_name = xmalloc (len);
1243# endif
1244#endif
1245 FAKE_DIR_ENTRY (d);
1246#ifdef _DIRENT_HAVE_D_NAMLEN
1247 d->d_namlen = len - 1;
1248#endif
1249#ifdef _DIRENT_HAVE_D_TYPE
1250 d->d_type = DT_UNKNOWN;
1251#endif
1252 memcpy (d->d_name, df->name, len);
1253 return d;
1254 }
1255 }
1256
1257 return 0;
1258}

1260/* On 64 bit ReliantUNIX (5.44 and above) in LFS mode, stat() is actually a
1261 * macro for stat64(). If stat is a macro, make a local wrapper function to
1262 * invoke it.
1263 *
1264 * On MS-Windows, stat() "succeeds" for foo/bar/. where foo/bar is a
1265 * regular file; fix that here.
1266 */
1267#if !defined(stat) && !defined(WINDOWS32) || defined(VMS)
1268# ifndef VMS
1269# ifndef HAVE_SYS_STAT_H
1270int stat (const char *path, struct stat *sbuf);
1271# endif
1272# else
1273 /* We are done with the fake stat. Go back to the real stat */
1274# ifdef stat
1275# undef stat
1276# endif
1277# endif
1278# define local_stat stat
1279#else
1280static int
1281local_stat (const char *path, struct stat *buf)
1282{
1283 int e;
1284#ifdef WINDOWS32
1285 size_t plen = strlen (path);
1286
1287 /* Make sure the parent of "." exists and is a directory, not a
1288 file. This is because 'stat' on Windows normalizes the argument
1289 foo/. => foo without checking first that foo is a directory. */
1290 if (plen > 1 && path[plen - 1] == '.'
1291 && (path[plen - 2] == '/' || path[plen - 2] == '\\'))
1292 {
1293 char parent[MAXPATHLEN];
1294
1295 strncpy (parent, path, plen - 2);
1296 parent[plen - 2] = '\0';
1297 if (stat (parent, buf) < 0 || !_S_ISDIR (buf->st_mode))
1298 return -1;
1299 }
1300#endif
1301
1302 EINTRLOOP (e, stat (path, buf));
1303 return e;
1304}
1305#endif
1306
1307/* Similarly for lstat. */
1308#if !defined(lstat) && !defined(WINDOWS32) || defined(VMS)
1309# ifndef VMS
1310# ifndef HAVE_SYS_STAT_H
1311int lstat (const char *path, struct stat *sbuf);
1312# endif
1313# else
1314 /* We are done with the fake lstat. Go back to the real lstat */
1315# ifdef lstat
1316# undef lstat
1317# endif
1318# endif
1319# define local_lstat lstat
1320#elif defined(WINDOWS32)
1321/* Windows doesn't support lstat(). */
1322# define local_lstat local_stat
1323#else
1324static int
1325local_lstat (const char *path, struct stat *buf)
1326{
1327 int e;
1328 EINTRLOOP (e, lstat (path, buf));
1329 return e;
1330}
1331#endif
