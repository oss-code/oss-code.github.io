17#ifndef REMAKE_OUTPUT_H
18#define REMAKE_OUTPUT_H
19
20struct output
21 {
22 int out;
23 int err;
24 unsigned int syncout:1; /* True if we want to synchronize output. */
25 };
26
27extern struct output *output_context;
28extern unsigned int stdio_traced;
29
30#define OUTPUT_SET(_new) do{ output_context = (_new)->syncout ? (_new) : NULL; }while(0)
31#define OUTPUT_UNSET() do{ output_context = NULL; }while(0)
32
33#define OUTPUT_TRACED() do{ stdio_traced = 1; }while(0)
34#define OUTPUT_IS_TRACED() (!!stdio_traced)
35
36FILE *output_tmpfile (char **, const char *);
37
38/* Initialize and close a child output structure: if NULL do this program's
39 output (this should only be done once). */
40void output_init (struct output *out);
41void output_close (struct output *out);
42
43/* In situations where output may be about to be displayed but we're not
44 sure if we've set it up yet, call this. */
45void output_start (void);
46
47/* Show a message on stdout or stderr. Will start the output if needed. */
48void outputs (int is_err, const char *msg);
49
50#ifndef NO_OUTPUT_SYNC
51int output_tmpfd (void);
52/* Dump any child output content to stdout, and reset it. */
53void output_dump (struct output *out);
54#endif
55
56#endif /*REMAKE_OUTPUT_H*/
