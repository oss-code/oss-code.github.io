27#ifndef REMAKE_READ_H
28#define REMAKE_READ_H
29
30#include "dep.h"
31
32extern stringlist_t *include_directories;
33
34/*! Search STRING for an unquoted STOPCHAR or blank (if BLANK is nonzero).
35 Backslashes quote STOPCHAR, blanks if BLANK is nonzero, and backslash.
36 Quoting backslashes are removed from STRING by compacting it into
37 itself. Returns a pointer to the first unquoted STOPCHAR if there is
38 one, or nil if there are none. */
39
40char *find_char_unquote (char *string, int stop1, int stop2,
41 int blank);
42
43/*! Search PATTERN for an unquoted %. */
44char *find_percent (char *pattern);
45
46/*! Expand ~ or ~USER at the beginning of NAME.
47 Return a newly malloc'd string or 0.
48*/
49char *make_tilde_expand (char *name);
50
51/*! Given a chain of struct nameseq's describing a sequence of filenames,
52 in reverse of the intended order, return a new chain describing the
53 result of globbing the filenames. The new chain is in forward order.
54 The links of the old chain are freed or used in the new chain.
55 Likewise for the names in the old chain.
56
57 SIZE is how big to construct chain elements.
58 This is useful if we want them actually to be other structures
59 that have room for additional info.
60*/
61nameseq_t * multi_glob (struct nameseq *chain, unsigned int size);
62
63/*! Read a Makefile and save information in our database about it.
64 */
65extern int eval_makefile (const char *filename, int flags);
66
67
68#endif /*REMAKE_READ_H*/
