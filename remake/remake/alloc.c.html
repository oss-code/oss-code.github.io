20/* Like malloc but get fatal error if memory is exhausted. */
21/* Don't bother if we're using dmalloc; it provides these for us. */
22
23#ifndef HAVE_DMALLOC_H
24
25#undef xmalloc
26#undef xcalloc
27#undef xrealloc
28#undef xstrdup
29
30void *
31xmalloc (unsigned int size)
32{
33 /* Make sure we don't allocate 0, for pre-ISO implementations. */
34 void *result = malloc (size ? size : 1);
35 if (result == 0)
36 OUT_OF_MEM();
37 return result;
38}
39
40
41void *
42xcalloc (unsigned int size)
43{
44 /* Make sure we don't allocate 0, for pre-ISO implementations. */
45 void *result = calloc (size ? size : 1, 1);
46 if (result == 0)
47 OUT_OF_MEM();
48 return result;
49}
50
51
52void *
53xrealloc (void *ptr, unsigned int size)
54{
55 void *result;
56
57 /* Some older implementations of realloc() don't conform to ISO. */
58 if (! size)
59 size = 1;
60 result = ptr ? realloc (ptr, size) : malloc (size);
61 if (result == 0)
62 OUT_OF_MEM();
63 return result;
64}

67char *
68xstrdup (const char *ptr)
69{
70 char *result;
71
72#ifdef HAVE_STRDUP
73 result = strdup (ptr);
74#else
75 result = malloc (strlen (ptr) + 1);
76#endif
77
78 if (result == 0)
79 OUT_OF_MEM();
80
81#ifdef HAVE_STRDUP
82 return result;
83#else
84 return strcpy (result, ptr);
85#endif
86}
87
88#endif /* HAVE_DMALLOC_H */
89
90char *
91xstrndup (const char *str, unsigned int length)
92{
93 char *result;
94
95#ifdef HAVE_STRNDUP
96 result = strndup (str, length);
97 if (result == 0)
98 OUT_OF_MEM();
99#else
100 result = xmalloc (length + 1);
101 if (length > 0)
102 strncpy (result, str, length);
103 result[length] = '\0';
104#endif
105
106 return result;
107}
