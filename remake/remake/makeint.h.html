174/* Nonzero if the integer type T is signed.

176#define INTEGER_TYPE_SIGNED(t) ((t) -1 <= 0)

178/* The minimum and maximum values for the integer type T.
179 Use ~ (t) 0, not -1, for portability to 1's complement hosts. */
180#define INTEGER_TYPE_MINIMUM(t) \
181 (! INTEGER_TYPE_SIGNED (t) ? (t) 0 : ~ (t) 0 << (sizeof (t) * CHAR_BIT - 1))
182#define INTEGER_TYPE_MAXIMUM(t) (~ (t) 0 - INTEGER_TYPE_MINIMUM (t))

286/* ISDIGIT offers the following features:
287 - Its arg may be any int or unsigned int; it need not be an unsigned char.
288 - It's guaranteed to evaluate its argument exactly once.
289 NOTE! Make relies on this behavior, don't change it!
290 - It's typically faster.
291 POSIX 1003.2-1992 section 2.5.2.1 page 50 lines 1556-1558 says that
292 only '0' through '9' are digits. Prefer ISDIGIT to isdigit() unless
293 it's important to use the locale's definition of 'digit' even when the
294 host does not conform to POSIX. */
295#define ISDIGIT(c) ((unsigned) (c) - '0' <= 9)

297/* Test if two strings are equal. Is this worthwhile? Should be profiled. */
298#define streq(a, b) \
299 ((a) == (b) || \
300 (*(a) == *(b) && (*(a) == '\0' || !strcmp ((a) + 1, (b) + 1))))
301
302/* Test if two strings are equal, but match case-insensitively on systems
303 which have case-insensitive filesystems. Should only be used for
304 filenames! */
305#ifdef HAVE_CASE_INSENSITIVE_FS
306# define patheq(a, b) \
307 ((a) == (b) \
308 || (tolower((unsigned char)*(a)) == tolower((unsigned char)*(b)) \
309 && (*(a) == '\0' || !strcasecmp ((a) + 1, (b) + 1))))
310#else
311# define patheq(a, b) streq(a, b)
312#endif

725/* Some systems (like Solaris, PTX, etc.) do not support the SA_RESTART flag
726 properly according to POSIX. So, we try to wrap common system calls with
727 checks for EINTR. Note that there are still plenty of system calls that
728 can fail with EINTR but this, reportedly, gets the vast majority of
729 failure cases. If you still experience failures you'll need to either get
730 a system where SA_RESTART works, or you need to avoid -j. */
731
732#define EINTRLOOP(_v,_c) while (((_v)=_c)==-1 && errno==EINTR)
733
734/* While system calls that return integers are pretty consistent about
735 returning -1 on failure and setting errno in that case, functions that
736 return pointers are not always so well behaved. Sometimes they return
737 NULL for expected behavior: one good example is readdir() which returns
738 NULL at the end of the directory--and _doesn't_ reset errno. So, we have
739 to do it ourselves here. */
740
741#define ENULLLOOP(_v,_c) do { errno = 0; (_v) = _c; } \
742 while((_v)==0 && errno==EINTR)
