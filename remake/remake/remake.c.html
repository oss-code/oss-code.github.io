45 'struct file'. However, double colon targets have separate 'struct
46 file's; make sure we always use the base of the double colon chain. */
47
48#define start_updating(_f) (((_f)->double_colon ? (_f)->double_colon : (_f))\
49 ->updating = 1)
50#define finish_updating(_f) (((_f)->double_colon ? (_f)->double_colon : (_f))\
51 ->updating = 0)
52#define is_updating(_f) (((_f)->double_colon ? (_f)->double_colon : (_f))\
53 ->updating)

81/* Remake all the goals in the 'struct dep' chain GOALS. Return -1 if nothing
82 was done, 0 if all goals were updated successfully, or 1 if a goal failed.
83
84 If rebuilding_makefiles is nonzero, these goals are makefiles, so -t, -q,
85 and -n should be disabled for them unless they were also command-line
86 targets, and we should only make one goal at a time and return as soon as
87 one goal whose 'changed' member is nonzero is successfully made. */
88
89enum update_status
90update_goal_chain (struct goaldep *goaldeps)
91{
92 int t = touch_flag, q = question_flag, n = just_print_flag;
93 enum update_status status = us_none;
94
95 /* Duplicate the chain so we can remove things from it. */
96
97 struct dep *goals = copy_dep_chain ((struct dep *)goaldeps);
98
99 goal_list = rebuilding_makefiles ? goaldeps : NULL;
100
101#define MTIME(file) (rebuilding_makefiles ? file_mtime_no_search (file) \
102 : file_mtime (file))
103
104 /* Start a fresh batch of consideration. */
105 ++considered;
106
107 /* Update all the goals until they are all finished. */
108
109 while (goals != 0)
110 {
111 register struct dep *g, *lastgoal;
112
113 /* Start jobs that are waiting for the load to go down. */
114
115 start_waiting_jobs (NULL);
116
117 /* Wait for a child to die. */
118
119 reap_children (1, 0, NULL);
120
121 lastgoal = 0;
122 g = goals;
123 while (g != 0)
124 {
125 /* Iterate over all double-colon entries for this file. */
126 struct file *file;
127 int stop = 0, any_not_updated = 0;
128
129 goal_dep = g;
130
131 for (file = g->file->double_colon ? g->file->double_colon : g->file;
132 file != NULL;
133 file = file->prev)
134 {
135 unsigned int ocommands_started;
136 enum update_status fail;
137
138 file->dontcare = ANY_SET (g->flags, RM_DONTCARE);
139
140 check_renamed (file);
141 if (rebuilding_makefiles)
142 {
143 if (file->cmd_target)
144 {
145 touch_flag = t;
146 question_flag = q;
147 just_print_flag = n;
148 }
149 else
150 touch_flag = question_flag = just_print_flag = 0;
151 }
152
153 /* Save the old value of 'commands_started' so we can compare
154 later. It will be incremented when any commands are
155 actually run. */
156 ocommands_started = commands_started;
157
158 fail = update_file (file, rebuilding_makefiles ? 1 : 0, NULL);
159 check_renamed (file);
160
161 /* Set the goal's 'changed' flag if any commands were started
162 by calling update_file above. We check this flag below to
163 decide when to give an "up to date" diagnostic. */
164 if (commands_started > ocommands_started)
165 g->changed = 1;
166
167 stop = 0;
168 if ((fail || file->updated) && status < us_question)
169 {
170 /* We updated this goal. Update STATUS and decide whether
171 to stop. */
172 if (file->update_status)
173 {
174 /* Updating failed, or -q triggered. The STATUS value
175 tells our caller which. */
176 status = file->update_status;
177 /* If -q just triggered, stop immediately. It doesn't
178 matter how much more we run, since we already know
179 the answer to return. */
180 stop = (question_flag && !keep_going_flag
181 && !rebuilding_makefiles);
182 }
183 else
184 {
185 FILE_TIMESTAMP mtime = MTIME (file);
186 check_renamed (file);
187
188 if (file->updated && g->changed &&
189 mtime != file->mtime_before_update)
190 {
191 /* Updating was done. If this is a makefile and
192 just_print_flag or question_flag is set (meaning
193 -n or -q was given and this file was specified
194 as a command-line target), don't change STATUS.
195 If STATUS is changed, we will get re-exec'd, and
196 enter an infinite loop. */
197 if (!rebuilding_makefiles
198 || (!just_print_flag && !question_flag))
199 status = us_success;
200 if (rebuilding_makefiles && file->dontcare)
201 /* This is a default makefile; stop remaking. */
202 stop = 1;
203 }
204 }
205 }
206
207 /* Keep track if any double-colon entry is not finished.
208 When they are all finished, the goal is finished. */
209 any_not_updated |= !file->updated;
210
211 file->dontcare = 0;
212
213 if (stop)
214 break;
215 }
216
217 /* Reset FILE since it is null at the end of the loop. */
218 file = g->file;
219
220 if (stop || !any_not_updated)
221 {
222 /* If we have found nothing whatever to do for the goal,
223 print a message saying nothing needs doing. */
224
225 if (!rebuilding_makefiles
226 /* If the update_status is success, we updated successfully
227 or not at all. G->changed will have been set above if
228 any commands were actually started for this goal. */
229 && file->update_status == us_success && !g->changed
230 /* Never give a message under -s or -q. */
231 && !silent_flag && !question_flag) {
232 OS (message, 1, ((file->phony || file->cmds == 0)
233 ? _("Nothing to be done for '%s'.")
234 : _("'%s' is up to date.")),
235 file->name);
236 if ( debugger_enabled && b_debugger_goal )
237 enter_debugger(NULL, file, -2, DEBUG_GOAL_UPDATED_HIT);
238 }
239
240 /* This goal is finished. Remove it from the chain. */
241 if (lastgoal == 0)
242 goals = g->next;
243 else
244 lastgoal->next = g->next;
245
246 /* Free the storage. */
247 free (g);
248
249 g = lastgoal == 0 ? goals : lastgoal->next;
250
251 if (stop)
252 break;
253 }
254 else
255 {
256 lastgoal = g;
257 g = g->next;
258 }
259 }
260
261 /* If we reached the end of the dependency graph update CONSIDERED
262 for the next pass. */
263 if (g == 0)
264 ++considered;
265 }
266
267 if (rebuilding_makefiles)
268 {
269 touch_flag = t;
270 question_flag = q;
271 just_print_flag = n;
272 }
273
274 return status;
275}

277/* If we're rebuilding an included makefile that failed, and we care
278 about errors, show an error message the first time. */
279
280void
281show_goal_error (void)
282{
283 struct goaldep *goal;
284
285 if ((goal_dep->flags & (RM_INCLUDED|RM_DONTCARE)) != RM_INCLUDED)
286 return;
287
288 for (goal = goal_list; goal; goal = goal->next)
289 if (goal_dep->file == goal->file)
290 {
291 if (goal->error)
292 {
293 OSS (error, &goal->floc, "%s: %s",
294 goal->file->name, strerror ((int)goal->error));
295 goal->error = 0;
296 }
297 return;
298 }
299}

301/* If FILE is not up to date, execute the commands for it.
302 Return 0 if successful, non-0 if unsuccessful;
303 but with some flag settings, just call 'exit' if unsuccessful.
304
305 DEPTH is the depth in recursions of this function.
306 We increment it during the consideration of our dependencies,
307 then decrement it again after finding out whether this file
308 is out of date.
309
310 If there are multiple double-colon entries for FILE,
311 each is considered in turn. */
312
313static enum update_status
314update_file (struct file *file, unsigned int depth,
315 target_stack_node_t *p_call_stack)
316{
317 enum update_status status = us_success;
318 struct file *f;
319
320 f = file->double_colon ? file->double_colon : file;
321
322 /* Prune the dependency graph: if we've already been here on _this_
323 pass through the dependency graph, we don't have to go any further.
324 We won't reap_children until we start the next pass, so no state
325 change is possible below here until then. */
326 if (f->considered == considered)
327 {
328 /* Check for the case where a target has been tried and failed but
329 the diagnostics haven't been issued. If we need the diagnostics
330 then we will have to continue. */
331 if (!(f->updated && f->update_status > us_none
332 && !f->dontcare && f->no_diag))
333 {
334 DBF (DB_VERBOSE, _("Pruning file '%s'.\n"));
335 return f->command_state == cs_finished ? f->update_status : us_success;
336 }
337 }
338
339 /* This loop runs until we start commands for a double colon rule, or until
340 the chain is exhausted. */
341 for (; f != 0; f = f->prev)
342 {
343 enum update_status new;
344
345 f->considered = considered;
346
347 new = update_file_1 (f, depth, p_call_stack);
348 check_renamed (f);
349
350 /* Clean up any alloca() used during the update. */
351 alloca (0);
352
353 /* If we got an error, don't bother with double_colon etc. */
354 if (new && !keep_going_flag)
355 return new;
356
357 if (f->command_state == cs_running
358 || f->command_state == cs_deps_running)
359 /* Don't run other :: rules for this target until
360 this rule is finished. */
361 return us_success;
362
363 if (new > status)
364 status = new;
365 }
366
367 /* Process the remaining rules in the double colon chain so they're marked
368 considered. Start their prerequisites, too. */
369 if (file->double_colon)
370 for (; f != 0 ; f = f->prev)
371 {
372 struct dep *d;
373
374 f->considered = considered;
375
376 for (d = f->deps; d != 0; d = d->next)
377 {
378 enum update_status new = update_file (d->file, depth + 1, p_call_stack);
379 if (new > status)
380 status = new;
381 }
382 }
383
384 return status;
385}

387/* Show a message stating the target failed to build. */
388
389static void
390complain (struct file *file)
391{
392 /* If this file has no_diag set then it means we tried to update it
393 before in the dontcare mode and failed. The target that actually
394 failed is not necessarily this file but could be one of its direct
395 or indirect dependencies. So traverse this file's dependencies and
396 find the one that actually caused the failure. */
397
398 struct dep *d;
399
400 for (d = file->deps; d != 0; d = d->next)
401 {
402 if (d->file->updated && d->file->update_status > us_none && file->no_diag)
403 {
404 complain (d->file);
405 break;
406 }
407 }
408
409 if (d == 0)
410 {
411 show_goal_error ();
412
413 /* Didn't find any dependencies to complain about. */
414 if (file->parent)
415 {
416 size_t l = strlen (file->name) + strlen (file->parent->name) + 4;
417 const char *m = _("%sNo rule to make target '%s', needed by '%s'%s");
418
419 if (!keep_going_flag)
420 fatal (NILF, l, m, "", file->name, file->parent->name, "");
421
422 error (NILF, l, m, "*** ", file->name, file->parent->name, ".");
423 }
424 else
425 {
426 size_t l = strlen (file->name) + 4;
427 const char *m = _("%sNo rule to make target '%s'%s");
428
429 if (!keep_going_flag)
430 fatal (NILF, l, m, "", file->name, "");
431
432 error (NILF, l, m, "*** ", file->name, ".");
433 }
434
435 file->no_diag = 0;
436 }
437}
438
439#define PROFILE_TIME \
440 if (profile_flag && !time_error) { \
441 time_error = time_error || get_time(&finish_time); \
442 file->elapsed_time = time_diff(&start_time, &finish_time); \
443 if (p_call_stack->p_parent) \
444 add_target(file, p_call_stack->p_parent->p_target); \
445 }
446
447#define INCR_PROFILE_TIME \
448 if (profile_flag && !time_error) { \
449 time_error = time_error || get_time(&finish_time); \
450 file->elapsed_time += time_diff(&start_time, &finish_time); \
451 }

453/* Consider a single 'struct file' and update it as appropriate.
454 Return 0 on success, or non-0 on failure. */
455
456static enum update_status
457update_file_1 (struct file *file, unsigned int depth,
458 target_stack_node_t *p_call_stack)
459{
460 enum update_status dep_status = us_success;
461 FILE_TIMESTAMP this_mtime;
462 int noexist, must_make, deps_changed;
463 struct file *ofile;
464 struct dep *d, *ad;
465 struct dep amake;
466 int running = 0;
467 struct timeval finish_time;
468 struct timeval start_time;
469 bool time_error = false;
470
471 if (profile_flag)
472 time_error = time_error || get_time(&start_time);
473
474 DBF (DB_VERBOSE, _("Considering target file '%s'.\n"));
475 p_stack_top = p_call_stack = trace_push_target(p_call_stack, file);
476
477 /* We don't want to step into file dependencies when there are
478 no associated commands. There or too often too many of them.
479 */
480 if ( (i_debugger_stepping && file->cmds) ||
481 (file->tracing & BRK_BEFORE_PREREQ) )
482 enter_debugger(p_call_stack, file, 0, DEBUG_BRKPT_BEFORE_PREREQ);
483
484 if (file->updated)
485 {
486 if (file->update_status > us_none)
487 {
488 DBF (DB_VERBOSE,
489 _("Recently tried and failed to update file '%s'.\n"));
490
491 /* If the file we tried to make is marked no_diag then no message
492 was printed about it when it failed during the makefile rebuild.
493 If we're trying to build it again in the normal rebuild, print a
494 message now. */
495 if (file->no_diag && !file->dontcare)
496 complain (file);
497
498 return file->update_status;
499 }
500
501 DBF (DB_VERBOSE, _("File '%s' was considered already.\n"));
502 INCR_PROFILE_TIME;
503 trace_pop_target(p_call_stack);
504 return 0;
505 }
506
507 switch (file->command_state)
508 {
509 case cs_not_started:
510 case cs_deps_running:
511 break;
512 case cs_running:
513 DBF (DB_VERBOSE, _("Still updating file '%s'.\n"));
514 INCR_PROFILE_TIME;
515 trace_pop_target(p_call_stack);
516 return 0;
517 case cs_finished:
518 DBF (DB_VERBOSE, _("Finished updating file '%s'.\n"));
519 INCR_PROFILE_TIME;
520 trace_pop_target(p_call_stack);
521 return file->update_status;
522 default:
523 abort ();
524 }
525
526 /* Determine whether the diagnostics will be issued should this update
527 fail. */
528 file->no_diag = file->dontcare;
529
530 ++depth;
531
532 /* Notice recursive update of the same file. */
533 start_updating (file);
534
535 /* We might change file if we find a different one via vpath;
536 remember this one to turn off updating. */
537 ofile = file;
538
539 /* Looking at the file's modtime beforehand allows the possibility
540 that its name may be changed by a VPATH search, and thus it may
541 not need an implicit rule. If this were not done, the file
542 might get implicit commands that apply to its initial name, only
543 to have that name replaced with another found by VPATH search. */
544
545 this_mtime = file_mtime (file);
546 check_renamed (file);
547 noexist = this_mtime == NONEXISTENT_MTIME;
548 if (noexist) {
549 DBF (DB_BASIC, _("File '%s' does not exist.\n"));
550 if (i_debugger_nexting && file->cmds) {
551 enter_debugger(p_call_stack, file, 0, DEBUG_STEP_HIT);
552 }
553 } else if (ORDINARY_MTIME_MIN <= this_mtime && this_mtime <= ORDINARY_MTIME_MAX
554 && file->low_resolution_time)
555 {
556 /* Avoid spurious rebuilds due to low resolution time stamps. */
557 int ns = FILE_TIMESTAMP_NS (this_mtime);
558 if (ns != 0)
559 OS (error, NILF,
560 _("*** Warning: .LOW_RESOLUTION_TIME file '%s' has a high resolution time stamp"),
561 file->name);
562 this_mtime += FILE_TIMESTAMPS_PER_S - 1 - ns;
563 }
564
565 must_make = noexist;
566
567 /* If file was specified as a target with no commands,
568 come up with some default commands. */
569
570 if (!file->phony && file->cmds == 0 && !file->tried_implicit)
571 {
572 if (try_implicit_rule (file, depth))
573 DBF (DB_IMPLICIT, _("Found an implicit rule for '%s'.\n"));
574 else
575 DBF (DB_IMPLICIT, _("No implicit rule found for '%s'.\n"));
576 file->tried_implicit = 1;
577 }
578 if (file->cmds == 0 && !file->is_target
579 && default_file != 0 && default_file->cmds != 0)
580 {
581 DBF (DB_IMPLICIT, _("Using default recipe for '%s'.\n"));
582 file->cmds = default_file->cmds;
583 }
584
585 /* Update all non-intermediate files we depend on, if necessary, and see
586 whether any of them is more recent than this file. We need to walk our
587 deps, AND the deps of any also_make targets to ensure everything happens
588 in the correct order. */
589
590 amake.file = file;
591 amake.next = file->also_make;
592 ad = &amake;
593 while (ad)
594 {
595 struct dep *lastd = 0;
596
597 /* Find the deps we're scanning */
598 d = ad->file->deps;
599 ad = ad->next;
600
601 while (d)
602 {
603 enum update_status new;
604 FILE_TIMESTAMP mtime;
605 int maybe_make;
606 int dontcare = 0;
607
608 check_renamed (d->file);
609
610 mtime = file_mtime (d->file);
611 check_renamed (d->file);
612
613 if (is_updating (d->file))
614 {
615 OSS (error, NILF, _("Circular %s <- %s dependency dropped."),
616 file->name, d->file->name);
617 /* We cannot free D here because our the caller will still have
618 a reference to it when we were called recursively via
619 check_dep below. */
620 if (lastd == 0)
621 file->deps = d->next;
622 else
623 lastd->next = d->next;
624 d = d->next;
625 continue;
626 }
627
628 d->file->parent = file;
629 maybe_make = must_make;
630
631 /* Inherit dontcare flag from our parent. */
632 if (rebuilding_makefiles)
633 {
634 dontcare = d->file->dontcare;
635 d->file->dontcare = file->dontcare;
636 }
637
638 new = check_dep (d->file, depth, this_mtime, &maybe_make,
639 p_call_stack);
640 if (new > dep_status)
641 dep_status = new;
642
643 /* Restore original dontcare flag. */
644 if (rebuilding_makefiles)
645 d->file->dontcare = dontcare;
646
647 if (! d->ignore_mtime)
648 must_make = maybe_make;
649
650 check_renamed (d->file);
651
652 {
653 register struct file *f = d->file;
654 if (f->double_colon)
655 f = f->double_colon;
656 do
657 {
658 running |= (f->command_state == cs_running
659 || f->command_state == cs_deps_running);
660 f = f->prev;
661 }
662 while (f != 0);
663 }
664
665 if (dep_status && !keep_going_flag)
666 break;
667
668 if (!running)
669 /* The prereq is considered changed if the timestamp has changed
670 while it was built, OR it doesn't exist. */
671 d->changed = ((file_mtime (d->file) != mtime)
672 || (mtime == NONEXISTENT_MTIME));
673
674 lastd = d;
675 d = d->next;
676 }
677 }
678
679 /* Now we know whether this target needs updating.
680 If it does, update all the intermediate files we depend on. */
681
682 if (must_make || always_make_flag)
683 {
684 for (d = file->deps; d != 0; d = d->next)
685 if (d->file->intermediate)
686 {
687 enum update_status new;
688 int dontcare = 0;
689
690 FILE_TIMESTAMP mtime = file_mtime (d->file);
691 check_renamed (d->file);
692 d->file->parent = file;
693
694 /* Inherit dontcare flag from our parent. */
695 if (rebuilding_makefiles)
696 {
697 dontcare = d->file->dontcare;
698 d->file->dontcare = file->dontcare;
699 }
700
701 /* We may have already considered this file, when we didn't know
702 we'd need to update it. Force update_file() to consider it and
703 not prune it. */
704 d->file->considered = 0;
705
706 new = update_file (d->file, depth, p_call_stack);
707 if (new > dep_status)
708 dep_status = new;
709
710 /* Restore original dontcare flag. */
711 if (rebuilding_makefiles)
712 d->file->dontcare = dontcare;
713
714 check_renamed (d->file);
715
716 {
717 register struct file *f = d->file;
718 if (f->double_colon)
719 f = f->double_colon;
720 do
721 {
722 running |= (f->command_state == cs_running
723 || f->command_state == cs_deps_running);
724 f = f->prev;
725 }
726 while (f != 0);
727 }
728
729 if (dep_status && !keep_going_flag)
730 break;
731
732 if (!running)
733 d->changed = ((file->phony && file->cmds != 0)
734 || file_mtime (d->file) != mtime);
735 }
736 }
737
738 finish_updating (file);
739 finish_updating (ofile);
740
741 DBF (DB_VERBOSE, _("Finished prerequisites of target file '%s'.\n"));
742
743 if ( file->tracing & BRK_AFTER_PREREQ )
744 enter_debugger(p_call_stack, file, 0, DEBUG_BRKPT_AFTER_PREREQ);
745
746 if (running)
747 {
748 set_command_state (file, cs_deps_running);
749 --depth;
750 DBF (DB_VERBOSE, _("The prerequisites of '%s' are being made.\n"));
751 INCR_PROFILE_TIME;
752 trace_pop_target(p_call_stack);
753 return 0;
754 }
755
756 /* If any dependency failed, give up now. */
757
758 if (dep_status)
759 {
760 /* I'm not sure if we can't just assign dep_status... */
761 file->update_status = dep_status == us_none ? us_failed : dep_status;
762 notice_finished_file (file);
763
764 --depth;
765
766 DBF (DB_VERBOSE, _("Giving up on target file '%s'.\n"));
767 PROFILE_TIME;
768
769 if (depth == 0 && keep_going_flag
770 && !just_print_flag && !question_flag)
771 OS (error, NILF,
772 _("Target '%s' not remade because of errors."), file->name);
773
774 trace_pop_target(p_call_stack);
775 return dep_status;
776 }
777
778 if (file->command_state == cs_deps_running)
779 /* The commands for some deps were running on the last iteration, but
780 they have finished now. Reset the command_state to not_started to
781 simplify later bookkeeping. It is important that we do this only
782 when the prior state was cs_deps_running, because that prior state
783 was definitely propagated to FILE's also_make's by set_command_state
784 (called above), but in another state an also_make may have
785 independently changed to finished state, and we would confuse that
786 file's bookkeeping (updated, but not_started is bogus state). */
787 set_command_state (file, cs_not_started);
788
789 /* Now record which prerequisites are more
790 recent than this file, so we can define $?. */
791
792 deps_changed = 0;
793 for (d = file->deps; d != 0; d = d->next)
794 {
795 FILE_TIMESTAMP d_mtime = file_mtime (d->file);
796 check_renamed (d->file);
797
798 if (! d->ignore_mtime)
799 {
800#if 1
801 /* %%% In version 4, remove this code completely to
802 implement not remaking deps if their deps are newer
803 than their parents. */
804 if (d_mtime == NONEXISTENT_MTIME && !d->file->intermediate)
805 /* We must remake if this dep does not
806 exist and is not intermediate. */
807 must_make = 1;
808#endif
809
810 /* Set DEPS_CHANGED if this dep actually changed. */
811 deps_changed |= d->changed;
812 }
813
814 /* Set D->changed if either this dep actually changed,
815 or its dependent, FILE, is older or does not exist. */
816 d->changed |= noexist || d_mtime > this_mtime;
817
818 if (!noexist && ISDB (DB_BASIC|DB_VERBOSE))
819 {
820 const char *fmt = 0;
821
822 if (d->ignore_mtime)
823 {
824 if (ISDB (DB_VERBOSE))
825 fmt = _("Prerequisite '%s' is order-only for target '%s'.\n");
826 }
827 else if (d_mtime == NONEXISTENT_MTIME)
828 {
829 if (ISDB (DB_BASIC))
830 fmt = _("Prerequisite '%s' of target '%s' does not exist.\n");
831 }
832 else if (d->changed)
833 {
834 if (ISDB (DB_BASIC))
835 fmt = _("Prerequisite '%s' is newer than target '%s'.\n");
836 }
837 else if (ISDB (DB_VERBOSE))
838 fmt = _("Prerequisite '%s' is older than target '%s'.\n");
839
840 if (fmt)
841 {
842 print_spaces (depth);
843 printf (fmt, dep_name (d), file->name);
844 fflush (stdout);
845 }
846 }
847 }
848
849 /* Here depth returns to the value it had when we were called. */
850 depth--;
851
852 if (file->double_colon && file->deps == 0)
853 {
854 must_make = 1;
855 DBF (DB_BASIC,
856 _("Target '%s' is double-colon and has no prerequisites.\n"));
857 }
858 else if (!noexist && file->is_target && !deps_changed && file->cmds == 0
859 && !always_make_flag)
860 {
861 must_make = 0;
862 DBF (DB_VERBOSE,
863 _("No recipe for '%s' and no prerequisites actually changed.\n"));
864 }
865 else if (!must_make && file->cmds != 0 && always_make_flag)
866 {
867 must_make = 1;
868 DBF (DB_VERBOSE, _("Making '%s' due to always-make flag.\n"));
869 }
870
871 if (!must_make)
872 {
873 if (ISDB (DB_VERBOSE))
874 {
875 print_spaces (depth);
876 printf (_("No need to remake target '%s'"), file->name);
877 if (!streq (file->name, file->hname))
878 printf (_("; using VPATH name '%s'"), file->hname);
879 puts (".");
880 fflush (stdout);
881 }
882
883 notice_finished_file (file);
884
885 /* Since we don't need to remake the file, convert it to use the
886 VPATH filename if we found one. hfile will be either the
887 local name if no VPATH or the VPATH name if one was found. */
888
889 PROFILE_TIME;
890
891 while (file)
892 {
893 file->name = file->hname;
894 file = file->prev;
895 }
896
897 trace_pop_target(p_call_stack);
898 return 0;
899 }
900
901 DBF (DB_BASIC, _("Must remake target '%s'.\n"));
902
903 /* It needs to be remade. If it's VPATH and not reset via GPATH, toss the
904 VPATH. */
905 if (!streq (file->name, file->hname))
906 {
907 DB (DB_BASIC, (_(" Ignoring VPATH name '%s'.\n"), file->hname));
908 file->ignore_vpath = 1;
909 }
910
911 /* Now, take appropriate actions to remake the file. */
912 remake_file (file, p_call_stack);
913
914 if (file->command_state != cs_finished)
915 {
916 DBF (DB_VERBOSE, _("Recipe of '%s' is being run.\n"));
917 if ( file->tracing & BRK_AFTER_CMD || i_debugger_stepping )
918 enter_debugger(p_call_stack, file, 0, DEBUG_BRKPT_AFTER_CMD);
919 PROFILE_TIME;
920 trace_pop_target(p_call_stack);
921 return 0;
922 }
923
924 switch (file->update_status)
925 {
926 case us_failed:
927 DBF (DB_BASIC, _("Failed to remake target file '%s'.\n"));
928 break;
929 case us_success:
930 DBF (DB_BASIC, _("Successfully remade target file '%s'.\n"));
931 break;
932 case us_question:
933 DBF (DB_BASIC, _("Target file '%s' needs to be remade under -q.\n"));
934 break;
935 case us_none:
936 break;
937 }
938
939 file->updated = 1;
940 if ( file->tracing & BRK_AFTER_CMD || i_debugger_stepping )
941 enter_debugger(p_call_stack, file, 0, DEBUG_BRKPT_AFTER_CMD);
942 PROFILE_TIME;
943
944 trace_pop_target(p_call_stack);
945 return file->update_status;
946}

948/* Set FILE's 'updated' flag and re-check its mtime and the mtime's of all
949 files listed in its 'also_make' member. Under -t, this function also
950 touches FILE.
951
952 On return, FILE->update_status will no longer be us_none if it was. */
953
954void
955notice_finished_file (struct file *file)
956{
957 struct dep *d;
958 int ran = file->command_state == cs_running;
959 int touched = 0;
960
961 file->command_state = cs_finished;
962 file->updated = 1;
963
964 if (touch_flag
965 /* The update status will be:
966 us_success if 0 or more commands (+ or ${MAKE}) were run and won;
967 us_none if this target was not remade;
968 >us_none if some commands were run and lost.
969 We touch the target if it has commands which either were not run
970 or won when they ran (i.e. status is 0). */
971 && file->update_status == us_success)
972 {
973 if (file->cmds != 0 && file->cmds->any_recurse)
974 {
975 /* If all the command lines were recursive,
976 we don't want to do the touching. */
977 unsigned int i;
978 for (i = 0; i < file->cmds->ncommand_lines; ++i)
979 if (!(file->cmds->lines_flags[i] & COMMANDS_RECURSE))
980 goto have_nonrecursing;
981 }
982 else
983 {
984 have_nonrecursing:
985 if (file->phony)
986 file->update_status = us_success;
987 /* According to POSIX, -t doesn't affect targets with no cmds. */
988 else if (file->cmds != 0)
989 {
990 /* Should set file's modification date and do nothing else. */
991 file->update_status = touch_file (file);
992
993 /* Pretend we ran a real touch command, to suppress the
994 "'foo' is up to date" message. */
995 commands_started++;
996
997 /* Request for the timestamp to be updated (and distributed
998 to the double-colon entries). Simply setting ran=1 would
999 almost have done the trick, but messes up with the also_make
1000 updating logic below. */
1001 touched = 1;
1002 }
1003 }
1004 }
1005
1006 if (file->mtime_before_update == UNKNOWN_MTIME)
1007 file->mtime_before_update = file->last_mtime;
1008
1009 if ((ran && !file->phony) || touched)
1010 {
1011 int i = 0;
1012
1013 /* If -n, -t, or -q and all the commands are recursive, we ran them so
1014 really check the target's mtime again. Otherwise, assume the target
1015 would have been updated. */
1016
1017 if ((question_flag || just_print_flag || touch_flag) && file->cmds)
1018 {
1019 for (i = file->cmds->ncommand_lines; i > 0; --i)
1020 if (! (file->cmds->lines_flags[i-1] & COMMANDS_RECURSE))
1021 break;
1022 }
1023
1024 /* If there were no commands at all, it's always new. */
1025
1026 else if (file->is_target && file->cmds == 0)
1027 i = 1;
1028
1029 file->last_mtime = i == 0 ? UNKNOWN_MTIME : NEW_MTIME;
1030 }
1031
1032 if (file->double_colon)
1033 {
1034 /* If this is a double colon rule and it is the last one to be
1035 updated, propagate the change of modification time to all the
1036 double-colon entries for this file.
1037
1038 We do it on the last update because it is important to handle
1039 individual entries as separate rules with separate timestamps
1040 while they are treated as targets and then as one rule with the
1041 unified timestamp when they are considered as a prerequisite
1042 of some target. */
1043
1044 struct file *f;
1045 FILE_TIMESTAMP max_mtime = file->last_mtime;
1046
1047 /* Check that all rules were updated and at the same time find
1048 the max timestamp. We assume UNKNOWN_MTIME is newer then
1049 any other value. */
1050 for (f = file->double_colon; f != 0 && f->updated; f = f->prev)
1051 if (max_mtime != UNKNOWN_MTIME
1052 && (f->last_mtime == UNKNOWN_MTIME || f->last_mtime > max_mtime))
1053 max_mtime = f->last_mtime;
1054
1055 if (f == 0)
1056 for (f = file->double_colon; f != 0; f = f->prev)
1057 f->last_mtime = max_mtime;
1058 }
1059
1060 if (ran && file->update_status != us_none)
1061 /* We actually tried to update FILE, which has
1062 updated its also_make's as well (if it worked).
1063 If it didn't work, it wouldn't work again for them.
1064 So mark them as updated with the same status. */
1065 for (d = file->also_make; d != 0; d = d->next)
1066 {
1067 d->file->command_state = cs_finished;
1068 d->file->updated = 1;
1069 d->file->update_status = file->update_status;
1070
1071 if (ran && !d->file->phony)
1072 /* Fetch the new modification time.
1073 We do this instead of just invalidating the cached time
1074 so that a vpath_search can happen. Otherwise, it would
1075 never be done because the target is already updated. */
1076 f_mtime (d->file, 0);
1077 }
1078 else if (file->update_status == us_none)
1079 /* Nothing was done for FILE, but it needed nothing done.
1080 So mark it now as "succeeded". */
1081 file->update_status = us_success;
1082}

1084/* Check whether another file (whose mtime is THIS_MTIME) needs updating on
1085 account of a dependency which is file FILE. If it does, store 1 in
1086 *MUST_MAKE_PTR. In the process, update any non-intermediate files that
1087 FILE depends on (including FILE itself). Return nonzero if any updating
1088 failed. */
1089
1090static enum update_status
1091check_dep (struct file *file, unsigned int depth,
1092 FILE_TIMESTAMP this_mtime, int *must_make_ptr,
1093 target_stack_node_t *p_call_stack)
1094{
1095 struct file *ofile;
1096 struct dep *d;
1097 enum update_status dep_status = us_success;
1098
1099 ++depth;
1100 start_updating (file);
1101
1102 /* We might change file if we find a different one via vpath;
1103 remember this one to turn off updating. */
1104 ofile = file;
1105
1106 if (file->phony || !file->intermediate)
1107 {
1108 /* If this is a non-intermediate file, update it and record whether it
1109 is newer than THIS_MTIME. */
1110 FILE_TIMESTAMP mtime;
1111 dep_status = update_file (file, depth, p_call_stack);
1112 check_renamed (file);
1113 mtime = file_mtime (file);
1114 check_renamed (file);
1115 if (mtime == NONEXISTENT_MTIME || mtime > this_mtime)
1116 *must_make_ptr = 1;
1117 }
1118 else
1119 {
1120 /* FILE is an intermediate file. */
1121 FILE_TIMESTAMP mtime;
1122
1123 if (!file->phony && file->cmds == 0 && !file->tried_implicit)
1124 {
1125 if (try_implicit_rule (file, depth))
1126 DBF (DB_IMPLICIT, _("Found an implicit rule for '%s'.\n"));
1127 else
1128 DBF (DB_IMPLICIT, _("No implicit rule found for '%s'.\n"));
1129 file->tried_implicit = 1;
1130 }
1131 if (file->cmds == 0 && !file->is_target
1132 && default_file != 0 && default_file->cmds != 0)
1133 {
1134 DBF (DB_IMPLICIT, _("Using default commands for '%s'.\n"));
1135 file->cmds = default_file->cmds;
1136 }
1137
1138 check_renamed (file);
1139 mtime = file_mtime (file);
1140 check_renamed (file);
1141 if (mtime != NONEXISTENT_MTIME && mtime > this_mtime)
1142 /* If the intermediate file actually exists and is newer, then we
1143 should remake from it. */
1144 *must_make_ptr = 1;
1145 else
1146 {
1147 /* Otherwise, update all non-intermediate files we depend on, if
1148 necessary, and see whether any of them is more recent than the
1149 file on whose behalf we are checking. */
1150 struct dep *ld;
1151 int deps_running = 0;
1152
1153 /* If this target is not running, set it's state so that we check it
1154 fresh. It could be it was checked as part of an order-only
1155 prerequisite and so wasn't rebuilt then, but should be now. */
1156 if (file->command_state != cs_running)
1157 {
1158 /* If the target was waiting for a dependency it has to be
1159 reconsidered, as that dependency might have finished. */
1160 if (file->command_state == cs_deps_running)
1161 file->considered = 0;
1162
1163 set_command_state (file, cs_not_started);
1164 }
1165
1166 ld = 0;
1167 d = file->deps;
1168 while (d != 0)
1169 {
1170 enum update_status new;
1171 int maybe_make;
1172
1173 if (is_updating (d->file))
1174 {
1175 OSS (error, NILF, _("Circular %s <- %s dependency dropped."),
1176 file->name, d->file->name);
1177 if (ld == 0)
1178 {
1179 file->deps = d->next;
1180 free_dep (d);
1181 d = file->deps;
1182 }
1183 else
1184 {
1185 ld->next = d->next;
1186 free_dep (d);
1187 d = ld->next;
1188 }
1189 continue;
1190 }
1191
1192 d->file->parent = file;
1193 maybe_make = *must_make_ptr;
1194 new = check_dep (d->file, depth, this_mtime, &maybe_make, p_call_stack);
1195 if (new > dep_status)
1196 dep_status = new;
1197
1198 if (! d->ignore_mtime)
1199 *must_make_ptr = maybe_make;
1200 check_renamed (d->file);
1201 if (dep_status && !keep_going_flag)
1202 break;
1203
1204 if (d->file->command_state == cs_running
1205 || d->file->command_state == cs_deps_running)
1206 deps_running = 1;
1207
1208 ld = d;
1209 d = d->next;
1210 }
1211
1212 if (deps_running)
1213 /* Record that some of FILE's deps are still being made.
1214 This tells the upper levels to wait on processing it until the
1215 commands are finished. */
1216 set_command_state (file, cs_deps_running);
1217 }
1218 }
1219
1220 finish_updating (file);
1221 finish_updating (ofile);
1222
1223 return dep_status;
1224}

1226/* Touch FILE. Return us_success if successful, us_failed if not. */
1227
1228#define TOUCH_ERROR(call) do{ perror_with_name ((call), file->name); \
1229 return us_failed; }while(0)
1230
1231static enum update_status
1232touch_file (struct file *file)
1233{
1234 if (!silent_flag)
1235 OS (message, 0, "touch %s", file->name);
1236
1237 /* Print-only (-n) takes precedence over touch (-t). */
1238 if (just_print_flag)
1239 return us_success;
1240
1241#ifndef NO_ARCHIVES
1242 if (ar_name (file->name))
1243 return ar_touch (file->name) ? us_failed : us_success;
1244 else
1245#endif
1246 {
1247 int fd;
1248
1249 EINTRLOOP (fd, open (file->name, O_RDWR | O_CREAT, 0666));
1250 if (fd < 0)
1251 TOUCH_ERROR ("touch: open: ");
1252 else
1253 {
1254 struct stat statbuf;
1255 char buf = 'x';
1256 int e;
1257
1258 EINTRLOOP (e, fstat (fd, &statbuf));
1259 if (e < 0)
1260 TOUCH_ERROR ("touch: fstat: ");
1261 /* Rewrite character 0 same as it already is. */
1262 EINTRLOOP (e, read (fd, &buf, 1));
1263 if (e < 0)
1264 TOUCH_ERROR ("touch: read: ");
1265 {
1266 off_t o;
1267 EINTRLOOP (o, lseek (fd, 0L, 0));
1268 if (o < 0L)
1269 TOUCH_ERROR ("touch: lseek: ");
1270 }
1271 EINTRLOOP (e, write (fd, &buf, 1));
1272 if (e < 0)
1273 TOUCH_ERROR ("touch: write: ");
1274
1275 /* If file length was 0, we just changed it, so change it back. */
1276 if (statbuf.st_size == 0)
1277 {
1278 (void) close (fd);
1279 EINTRLOOP (fd, open (file->name, O_RDWR | O_TRUNC, 0666));
1280 if (fd < 0)
1281 TOUCH_ERROR ("touch: open: ");
1282 }
1283 (void) close (fd);
1284 }
1285 }
1286
1287 return us_success;
1288}

1290/* Having checked and updated the dependencies of FILE,
1291 do whatever is appropriate to remake FILE itself.
1292 Return the status from executing FILE's commands. */
1293
1294static void
1295remake_file (file_t *file, target_stack_node_t *p_call_stack)
1296{
1297 if (file->cmds == 0)
1298 {
1299 if (file->phony)
1300 /* Phony target. Pretend it succeeded. */
1301 file->update_status = us_success;
1302 else if (file->is_target)
1303 /* This is a nonexistent target file we cannot make.
1304 Pretend it was successfully remade. */
1305 file->update_status = us_success;
1306 else
1307 {
1308 /* This is a dependency file we cannot remake. Fail. */
1309 if (!rebuilding_makefiles || !file->dontcare)
1310 complain (file);
1311 file->update_status = us_failed;
1312 }
1313 }
1314 else
1315 {
1316 chop_commands (file->cmds);
1317
1318 /* The normal case: start some commands. */
1319 if (!touch_flag || file->cmds->any_recurse)
1320 {
1321 execute_file_commands (file, p_call_stack);
1322 return;
1323 }
1324
1325 /* This tells notice_finished_file it is ok to touch the file. */
1326 file->update_status = us_success;
1327 }
1328
1329 /* This does the touching under -t. */
1330 notice_finished_file (file);
1331}

1333/* Return the mtime of a file, given a 'struct file'.
1334 Caches the time in the struct file to avoid excess stat calls.
1335
1336 If the file is not found, and SEARCH is nonzero, VPATH searching and
1337 replacement is done. If that fails, a library (-lLIBNAME) is tried and
1338 the library's actual name (/lib/libLIBNAME.a, etc.) is substituted into
1339 FILE. */
1340
1341FILE_TIMESTAMP
1342f_mtime (struct file *file, int search)
1343{
1344 FILE_TIMESTAMP mtime;
1345 int propagate_timestamp;
1346
1347 /* File's mtime is not known; must get it from the system. */
1348
1349#ifndef NO_ARCHIVES
1350 if (ar_name (file->name))
1351 {
1352 /* This file is an archive-member reference. */
1353
1354 char *arname, *memname;
1355 struct file *arfile;
1356 int found;
1357 time_t member_date;
1358
1359 /* Find the archive's name. */
1360 ar_parse_name (file->name, &arname, &memname);
1361
1362 /* Find the modification time of the archive itself.
1363 Also allow for its name to be changed via VPATH search. */
1364 arfile = lookup_file (arname);
1365 if (arfile == 0)
1366 arfile = enter_file (strcache_add (arname));
1367 mtime = f_mtime (arfile, search);
1368 check_renamed (arfile);
1369 if (search && strcmp (arfile->hname, arname))
1370 {
1371 /* The archive's name has changed.
1372 Change the archive-member reference accordingly. */
1373
1374 char *name;
1375 unsigned int arlen, memlen;
1376
1377 arlen = strlen (arfile->hname);
1378 memlen = strlen (memname);
1379
1380 name = alloca (arlen + 1 + memlen + 2);
1381 memcpy (name, arfile->hname, arlen);
1382 name[arlen] = '(';
1383 memcpy (name + arlen + 1, memname, memlen);
1384 name[arlen + 1 + memlen] = ')';
1385 name[arlen + 1 + memlen + 1] = '\0';
1386
1387 /* If the archive was found with GPATH, make the change permanent;
1388 otherwise defer it until later. */
1389 if (arfile->name == arfile->hname)
1390 rename_file (file, strcache_add (name));
1391 else
1392 rehash_file (file, strcache_add (name));
1393 check_renamed (file);
1394 }
1395
1396 free (arname);
1397
1398 file->low_resolution_time = 1;
1399
1400 if (mtime == NONEXISTENT_MTIME)
1401 /* The archive doesn't exist, so its members don't exist either. */
1402 return NONEXISTENT_MTIME;
1403
1404 found = ar_member_date (file->hname, &member_date);
1405 if (found && member_date == (time_t) 0)
1406 {
1407 OSS (error, NILF,
1408 _("Warning: Archive '%s' seems to have been created in deterministic mode. '%s' will always be updated. Please consider passing the U flag to ar to avoid the problem."),
1409 arfile->name, memname);
1410
1411 }
1412 mtime = found ? file_timestamp_cons (file->hname, member_date, 0) : NONEXISTENT_MTIME;
1413 }
1414 else
1415#endif
1416 {
1417 mtime = name_mtime (file->name);
1418
1419 if (mtime == NONEXISTENT_MTIME && search && !file->ignore_vpath)
1420 {
1421 /* If name_mtime failed, search VPATH. */
1422 const char *name = vpath_search (file->name, &mtime, NULL, NULL);
1423 if (name
1424 /* Last resort, is it a library (-lxxx)? */
1425 || (file->name[0] == '-' && file->name[1] == 'l'
1426 && (name = library_search (file->name, &mtime)) != 0))
1427 {
1428 int name_len;
1429
1430 if (mtime != UNKNOWN_MTIME)
1431 /* vpath_search and library_search store UNKNOWN_MTIME
1432 if they didn't need to do a stat call for their work. */
1433 file->last_mtime = mtime;
1434
1435 /* If we found it in VPATH, see if it's in GPATH too; if so,
1436 change the name right now; if not, defer until after the
1437 dependencies are updated. */
1438#ifndef VMS
1439 name_len = strlen (name) - strlen (file->name) - 1;
1440#else
1441 name_len = strlen (name) - strlen (file->name);
1442 if (name[name_len - 1] == '/')
1443 name_len--;
1444#endif
1445 if (gpath_search (name, name_len))
1446 {
1447 rename_file (file, name);
1448 check_renamed (file);
1449 return file_mtime (file);
1450 }
1451
1452 rehash_file (file, name);
1453 check_renamed (file);
1454 /* If the result of a vpath search is -o or -W, preserve it.
1455 Otherwise, find the mtime of the resulting file. */
1456 if (mtime != OLD_MTIME && mtime != NEW_MTIME)
1457 mtime = name_mtime (name);
1458 }
1459 }
1460 }
1461
1462 /* Files can have bogus timestamps that nothing newly made will be
1463 "newer" than. Updating their dependents could just result in loops.
1464 So notify the user of the anomaly with a warning.
1465
1466 We only need to do this once, for now. */
1467
1468 if (!clock_skew_detected
1469 && mtime != NONEXISTENT_MTIME && mtime != NEW_MTIME
1470 && !file->updated)
1471 {
1472 static FILE_TIMESTAMP adjusted_now;
1473
1474 FILE_TIMESTAMP adjusted_mtime = mtime;
1475
1476#if defined(WINDOWS32) || defined(__MSDOS__)
1477 /* Experimentation has shown that FAT filesystems can set file times
1478 up to 3 seconds into the future! Play it safe. */
1479
1480#define FAT_ADJ_OFFSET (FILE_TIMESTAMP) 3
1481
1482 FILE_TIMESTAMP adjustment = FAT_ADJ_OFFSET << FILE_TIMESTAMP_LO_BITS;
1483 if (ORDINARY_MTIME_MIN + adjustment <= adjusted_mtime)
1484 adjusted_mtime -= adjustment;
1485#elif defined(__EMX__)
1486 /* FAT filesystems round time to the nearest even second!
1487 Allow for any file (NTFS or FAT) to perhaps suffer from this
1488 brain damage. */
1489 FILE_TIMESTAMP adjustment = (((FILE_TIMESTAMP_S (adjusted_mtime) & 1) == 0
1490 && FILE_TIMESTAMP_NS (adjusted_mtime) == 0)
1491 ? (FILE_TIMESTAMP) 1 << FILE_TIMESTAMP_LO_BITS
1492 : 0);
1493#endif
1494
1495 /* If the file's time appears to be in the future, update our
1496 concept of the present and try once more. */
1497 if (adjusted_now < adjusted_mtime)
1498 {
1499 int resolution;
1500 FILE_TIMESTAMP now = file_timestamp_now (&resolution);
1501 adjusted_now = now + (resolution - 1);
1502 if (adjusted_now < adjusted_mtime)
1503 {
1504#ifdef NO_FLOAT
1505 OS (error, NILF,
1506 _("Warning: File '%s' has modification time in the future"),
1507 file->name);
1508#else
1509 double from_now =
1510 (FILE_TIMESTAMP_S (mtime) - FILE_TIMESTAMP_S (now)
1511 + ((FILE_TIMESTAMP_NS (mtime) - FILE_TIMESTAMP_NS (now))
1512 / 1e9));
1513 char from_now_string[100];
1514
1515 if (from_now >= 99 && from_now <= ULONG_MAX)
1516 sprintf (from_now_string, "%lu", (unsigned long) from_now);
1517 else
1518 sprintf (from_now_string, "%.2g", from_now);
1519 OSS (error, NILF,
1520 _("Warning: File '%s' has modification time %s s in the future"),
1521 file->name, from_now_string);
1522#endif
1523 clock_skew_detected = 1;
1524 }
1525 }
1526 }
1527
1528 /* Store the mtime into all the entries for this file for which it is safe
1529 to do so: avoid propagating timestamps to double-colon rules that haven't
1530 been examined so they're run or not based on the pre-update timestamp. */
1531 if (file->double_colon)
1532 file = file->double_colon;
1533
1534 propagate_timestamp = file->updated;
1535 do
1536 {
1537 /* If this file is not implicit but it is intermediate then it was
1538 made so by the .INTERMEDIATE target. If this file has never
1539 been built by us but was found now, it existed before make
1540 started. So, turn off the intermediate bit so make doesn't
1541 delete it, since it didn't create it. */
1542 if (mtime != NONEXISTENT_MTIME && file->command_state == cs_not_started
1543 && !file->tried_implicit && file->intermediate)
1544 file->intermediate = 0;
1545
1546 if (file->updated == propagate_timestamp)
1547 file->last_mtime = mtime;
1548 file = file->prev;
1549 }
1550 while (file != 0);
1551
1552 return mtime;
1553}
1554
1555
1556/* Return the mtime of the file or archive-member reference NAME. */
1557
1558/* First, we check with stat(). If the file does not exist, then we return
1559 NONEXISTENT_MTIME. If it does, and the symlink check flag is set, then
1560 examine each indirection of the symlink and find the newest mtime.
1561 This causes one duplicate stat() when -L is being used, but the code is
1562 much cleaner. */
1563
1564static FILE_TIMESTAMP
1565name_mtime (const char *name)
1566{
1567 FILE_TIMESTAMP mtime;
1568 struct stat st;
1569 int e;
1570
1571 EINTRLOOP (e, stat (name, &st));
1572 if (e == 0)
1573 mtime = FILE_TIMESTAMP_STAT_MODTIME (name, st);
1574 else if (errno == ENOENT || errno == ENOTDIR)
1575 mtime = NONEXISTENT_MTIME;
1576 else
1577 {
1578 perror_with_name ("stat: ", name);
1579 return NONEXISTENT_MTIME;
1580 }
1581
1582 /* If we get here we either found it, or it doesn't exist.
1583 If it doesn't exist see if we can use a symlink mtime instead. */
1584
1585#ifdef MAKE_SYMLINKS
1586#ifndef S_ISLNK
1587# define S_ISLNK(_m) (((_m)&S_IFMT)==S_IFLNK)
1588#endif
1589 if (check_symlink_flag)
1590 {
1591 PATH_VAR (lpath);
1592
1593 /* Check each symbolic link segment (if any). Find the latest mtime
1594 amongst all of them (and the target file of course).
1595 Note that we have already successfully dereferenced all the links
1596 above. So, if we run into any error trying to lstat(), or
1597 readlink(), or whatever, something bizarre-o happened. Just give up
1598 and use whatever mtime we've already computed at that point. */
1599 strcpy (lpath, name);
1600 while (1)
1601 {
1602 FILE_TIMESTAMP ltime;
1603 PATH_VAR (lbuf);
1604 long llen;
1605 char *p;
1606
1607 EINTRLOOP (e, lstat (lpath, &st));
1608 if (e)
1609 {
1610 /* Just take what we have so far. */
1611 if (errno != ENOENT && errno != ENOTDIR)
1612 perror_with_name ("lstat: ", lpath);
1613 break;
1614 }
1615
1616 /* If this is not a symlink, we're done (we started with the real
1617 file's mtime so we don't need to test it again). */
1618 if (!S_ISLNK (st.st_mode))
1619 break;
1620
1621 /* If this mtime is newer than what we had, keep the new one. */
1622 ltime = FILE_TIMESTAMP_STAT_MODTIME (lpath, st);
1623 if (ltime > mtime)
1624 mtime = ltime;
1625
1626 /* Set up to check the file pointed to by this link. */
1627 EINTRLOOP (llen, readlink (lpath, lbuf, GET_PATH_MAX));
1628 if (llen < 0)
1629 {
1630 /* Eh? Just take what we have. */
1631 perror_with_name ("readlink: ", lpath);
1632 break;
1633 }
1634 lbuf[llen] = '\0';
1635
1636 /* If the target is fully-qualified or the source is just a
1637 filename, then the new path is the target. Otherwise it's the
1638 source directory plus the target. */
1639 if (lbuf[0] == '/' || (p = strrchr (lpath, '/')) == NULL)
1640 strcpy (lpath, lbuf);
1641 else if ((p - lpath) + llen + 2 > GET_PATH_MAX)
1642 /* Eh? Path too long! Again, just go with what we have. */
1643 break;
1644 else
1645 /* Create the next step in the symlink chain. */
1646 strcpy (p+1, lbuf);
1647 }
1648 }
1649#endif
1650
1651 return mtime;
1652}

1655/* Search for a library file specified as -lLIBNAME, searching for a
1656 suitable library file in the system library directories and the VPATH
1657 directories. */
1658
1659static const char *
1660library_search (const char *lib, FILE_TIMESTAMP *mtime_ptr)
1661{
1662 static const char *dirs[] =
1663 {
1664 "/lib",
1665 "/usr/lib",
1666#if defined(WINDOWS32) && !defined(LIBDIR)
1667/*
1668 * This is completely up to the user at product install time. Just define
1669 * a placeholder.
1670 */
1671#define LIBDIR "."
1672#endif
1673 LIBDIR, /* Defined by configuration. */
1674 0
1675 };
1676
1677 const char *file = 0;
1678 char *libpatterns;
1679 FILE_TIMESTAMP mtime;
1680
1681 /* Loop variables for the libpatterns value. */
1682 char *p;
1683 const char *p2;
1684 unsigned int len;
1685 unsigned int liblen;
1686
1687 /* Information about the earliest (in the vpath sequence) match. */
1688 unsigned int best_vpath = 0, best_path = 0;
1689
1690 const char **dp;
1691
1692 libpatterns = xstrdup (variable_expand ("$(.LIBPATTERNS)"));
1693
1694 /* Skip the '-l'. */
1695 lib += 2;
1696 liblen = strlen (lib);
1697
1698 /* Loop through all the patterns in .LIBPATTERNS, and search on each one.
1699 To implement the linker-compatible behavior we have to search through
1700 all entries in .LIBPATTERNS and choose the "earliest" one. */
1701 p2 = libpatterns;
1702 while ((p = find_next_token (&p2, &len)) != 0)
1703 {
1704 static char *buf = NULL;
1705 static unsigned int buflen = 0;
1706 static int libdir_maxlen = -1;
1707 static unsigned int std_dirs = 0;
1708 char *libbuf = variable_expand ("");
1709
1710 /* Expand the pattern using LIB as a replacement. */
1711 {
1712 char c = p[len];
1713 char *p3, *p4;
1714
1715 p[len] = '\0';
1716 p3 = find_percent (p);
1717 if (!p3)
1718 {
1719 /* Give a warning if there is no pattern. */
1720 OS (error, NILF,
1721 _(".LIBPATTERNS element '%s' is not a pattern"), p);
1722 p[len] = c;
1723 continue;
1724 }
1725 p4 = variable_buffer_output (libbuf, p, p3-p);
1726 p4 = variable_buffer_output (p4, lib, liblen);
1727 p4 = variable_buffer_output (p4, p3+1, len - (p3-p));
1728 p[len] = c;
1729 }
1730
1731 /* Look first for 'libNAME.a' in the current directory. */
1732 mtime = name_mtime (libbuf);
1733 if (mtime != NONEXISTENT_MTIME)
1734 {
1735 if (mtime_ptr != 0)
1736 *mtime_ptr = mtime;
1737 file = strcache_add (libbuf);
1738 /* This by definition will have the best index, so stop now. */
1739 break;
1740 }
1741
1742 /* Now try VPATH search on that. */
1743
1744 {
1745 unsigned int vpath_index, path_index;
1746 const char* f = vpath_search (libbuf, mtime_ptr ? &mtime : NULL,
1747 &vpath_index, &path_index);
1748 if (f)
1749 {
1750 /* If we have a better match, record it. */
1751 if (file == 0 ||
1752 vpath_index < best_vpath ||
1753 (vpath_index == best_vpath && path_index < best_path))
1754 {
1755 file = f;
1756 best_vpath = vpath_index;
1757 best_path = path_index;
1758
1759 if (mtime_ptr != 0)
1760 *mtime_ptr = mtime;
1761 }
1762 }
1763 }
1764
1765 /* Now try the standard set of directories. */
1766
1767 if (!buflen)
1768 {
1769 for (dp = dirs; *dp != 0; ++dp)
1770 {
1771 int l = strlen (*dp);
1772 if (l > libdir_maxlen)
1773 libdir_maxlen = l;
1774 std_dirs++;
1775 }
1776 buflen = strlen (libbuf);
1777 buf = xmalloc (libdir_maxlen + buflen + 2);
1778 }
1779 else if (buflen < strlen (libbuf))
1780 {
1781 buflen = strlen (libbuf);
1782 buf = xrealloc (buf, libdir_maxlen + buflen + 2);
1783 }
1784
1785 {
1786 /* Use the last std_dirs index for standard directories. This
1787 was it will always be greater than the VPATH index. */
1788 unsigned int vpath_index = ~((unsigned int)0) - std_dirs;
1789
1790 for (dp = dirs; *dp != 0; ++dp)
1791 {
1792 sprintf (buf, "%s/%s", *dp, libbuf);
1793 mtime = name_mtime (buf);
1794 if (mtime != NONEXISTENT_MTIME)
1795 {
1796 if (file == 0 || vpath_index < best_vpath)
1797 {
1798 file = strcache_add (buf);
1799 best_vpath = vpath_index;
1800
1801 if (mtime_ptr != 0)
1802 *mtime_ptr = mtime;
1803 }
1804 }
1805
1806 vpath_index++;
1807 }
1808 }
1809
1810 }
1811
1812 free (libpatterns);
1813 return file;
1814}
