56static struct pspec default_pattern_rules[] =
57 {
58#ifdef VMS
59 { "(%)", "%",
60 "@if f$$search(\"$@\") .eqs. \"\" then $(LIBRARY)/CREATE/"
61 "$(or "
62 "$(patsubst %,TEXT,$(filter %.tlb %.TLB,$@)),"
63 "$(patsubst %,HELP,$(filter %.hlb %.HLB,$@)),"
64 "$(patsubst %,MACRO,$(filter %.mlb %.MLB,$@)),"
65 "$(and "
66 "$(patsubst %,SHARE,$(filter %.olb %.OLB,$@)),"
67 "$(patsubst %,SHARE,$(filter %.exe %.EXE,$<))),"
68 "OBJECT)"
69 " $@\n"
70 "$(AR) $(ARFLAGS) $@ $<" },
71
72#else
73 { "(%)", "%",
74 "$(AR) $(ARFLAGS) $@ $<" },
75#endif
76 /* The X.out rules are only in BSD's default set because
77 BSD Make has no null-suffix rules, so 'foo.out' and
78 'foo' are the same thing. */
79#ifdef VMS
80 { "%.exe", "%",
81 "$(CP) $< $@" },
82
83#endif
84 { "%.out", "%",
85 "@rm -f $@ \n cp $< $@" },
86
87 /* Syntax is "ctangle foo.w foo.ch foo.c". */
88 { "%.c", "%.w %.ch",
89 "$(CTANGLE) $^ $@" },
90 { "%.tex", "%.w %.ch",
91 "$(CWEAVE) $^ $@" },
92
93 { 0, 0, 0 }
94 };

102 "if f$$search(\"$@\") .nes. \"\" then +$(CHECKOUT,v)" },
103 { "%", "[.$$rcs]%$$5lv", /* Multinet style */
104 "if f$$search(\"$@\") .nes. \"\" then +$(CHECKOUT,v)" },
105 { "%", "%_v", /* Normal style */
106 "if f$$search(\"$@\") .nes. \"\" then +$(CHECKOUT,v)" },
107 { "%", "[.rcs]%_v", /* Normal style */
108 "if f$$search(\"$@\") .nes. \"\" then +$(CHECKOUT,v)" },
109
110 /* SCCS. */
111 /* ain't no SCCS on vms */
112
113#else
114 /* RCS. */
115 { "%", "%,v",
116 "$(CHECKOUT,v)" },
117 { "%", "RCS/%,v",
118 "$(CHECKOUT,v)" },
119 { "%", "RCS/%",

124 "$(GET) $(GFLAGS) $(SCCS_OUTPUT_OPTION) $<" },

128 { 0, 0, 0 }

668/* Set up the default .SUFFIXES list. */
669
670void
671set_default_suffixes (void)
672{
673 suffix_file = enter_file (strcache_add (".SUFFIXES"));
674 suffix_file->builtin = 1;
675
676 if (no_builtin_rules_flag)
677 define_variable_cname ("SUFFIXES", "", o_default, 0);
678 else
679 {
680 struct dep *d;
681 const char *p = default_suffixes;
682 suffix_file->deps = enter_prereqs (PARSE_SIMPLE_SEQ ((char **)&p, struct dep),
683 NULL);
684 for (d = suffix_file->deps; d; d = d->next)
685 d->file->builtin = 1;
686
687 define_variable_cname ("SUFFIXES", default_suffixes, o_default, 0);
688 }
689}

691/* Enter the default suffix rules as file rules. This used to be done in
692 install_default_implicit_rules, but that loses because we want the
693 suffix rules installed before reading makefiles, and the pattern rules
694 installed after. */
695
696void
697install_default_suffix_rules (void)
698{
699 const char **s;
700
701 if (no_builtin_rules_flag)
702 return;
703
704 for (s = default_suffix_rules; *s != 0; s += 2)
705 {
706 struct file *f = enter_file (strcache_add (s[0]));
707 /* This function should run before any makefile is parsed. */
708 assert (f->cmds == 0);
709 f->cmds = xmalloc (sizeof (struct commands));
710 f->cmds->fileinfo.filenm = 0;
711 f->cmds->commands = xstrdup (s[1]);
712 f->cmds->command_lines = 0;
713 f->cmds->recipe_prefix = RECIPEPREFIX_DEFAULT;
714 f->builtin = 1;
715 }
716}
