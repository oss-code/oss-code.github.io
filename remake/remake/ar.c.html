26/* Return nonzero if NAME is an archive-member reference, zero if not. An
27 archive-member reference is a name like 'lib(member)' where member is a
28 non-empty string.
29 If a name like 'lib((entry))' is used, a fatal error is signaled at
30 the attempt to use this unsupported feature. */
31
32int
33ar_name (const char *name)
34{
35 const char *p = strchr (name, '(');
36 const char *end;
37
38 if (p == 0 || p == name)
39 return 0;
40
41 end = p + strlen (p) - 1;
42 if (*end != ')' || end == p + 1)
43 return 0;
44
45 if (p[1] == '(' && end[-1] == ')')
46 OS (fatal, NILF, _("attempt to use unsupported feature: '%s'"), name);
47
48 return 1;
49}

52/* Parse the archive-member reference NAME into the archive and member names.
53 Creates one allocated string containing both names, pointed to by ARNAME_P.
54 MEMNAME_P points to the member. */
55
56void
57ar_parse_name (const char *name, char **arname_p, char **memname_p)
58{
59 char *p;
60
61 *arname_p = xstrdup (name);
62 p = strchr (*arname_p, '(');
63 *(p++) = '\0';
64 p[strlen (p) - 1] = '\0';
65 *memname_p = p;
66}

69/* This function is called by 'ar_scan' to find which member to look at. */
70
71struct member_date_lookup
72{
73 const char *name;
74 time_t *member_date;
75};
76
77/* ARGSUSED */
78static long int
79ar_member_date_1 (int desc UNUSED, const char *mem, int truncated,
80 long int hdrpos UNUSED, long int datapos UNUSED,
81 long int size UNUSED, long int date,
82 int uid UNUSED, int gid UNUSED, unsigned int mode UNUSED,
83 const void *data)
84{
85 const struct member_date_lookup *lookup_data = data;
86 if (ar_name_equal (lookup_data->name, mem, truncated))
87 {
88 *lookup_data->member_date = date;
89 return 1;
90 }
91 return 0;
92}

94/* Read the modtime of NAME in MEMBER_DATE.
95 Returns 1 if NAME exists, 0 otherwise. */
96int
97ar_member_date (const char *name, time_t *member_date)
98{
99 char *arname;
100 char *memname;
101 int found;
102 struct member_date_lookup lookup_data;
103
104 ar_parse_name (name, &arname, &memname);
105
106 /* Make sure we know the modtime of the archive itself because we are
107 likely to be called just before commands to remake a member are run,
108 and they will change the archive itself.
109
110 But we must be careful not to enter_file the archive itself if it does
111 not exist, because pattern_search assumes that files found in the data
112 base exist or can be made. */
113 {
114 struct file *arfile;
115 arfile = lookup_file (arname);
116 if (arfile == 0 && file_exists_p (arname))
117 arfile = enter_file (strcache_add (arname));
118
119 if (arfile != 0)
120 (void) f_mtime (arfile, 0);
121 }
122
123 lookup_data.name = memname;
124 lookup_data.member_date = member_date;
125 found = ar_scan (arname, ar_member_date_1, &lookup_data);
126
127 free (arname);
128
129 /* return 0 (not found) if the archive does not exist or has invalid format. */
130 return (found == 1) ? 1 : 0;
131}

135#ifdef VMS
136int
137ar_touch (const char *name)
138{
139 O (error, NILF, _("touch archive member is not available on VMS"));
140 return -1;
141}
142#else
143int
144ar_touch (const char *name)
145{
146 char *arname, *memname;
147 int val;
148
149 ar_parse_name (name, &arname, &memname);
150
151 /* Make sure we know the modtime of the archive itself before we
152 touch the member, since this will change the archive modtime. */
153 {
154 struct file *arfile;
155 arfile = enter_file (strcache_add (arname));
156 f_mtime (arfile, 0);
157 }
158
159 val = 1;
160 switch (ar_member_touch (arname, memname))
161 {
162 case -1:
163 OS (error, NILF, _("touch: Archive '%s' does not exist"), arname);
164 break;
165 case -2:
166 OS (error, NILF, _("touch: '%s' is not a valid archive"), arname);
167 break;
168 case -3:
169 perror_with_name ("touch: ", arname);
170 break;
171 case 1:
172 OSS (error, NILF,
173 _("touch: Member '%s' does not exist in '%s'"), memname, arname);
174 break;
175 case 0:
176 val = 0;
177 break;
178 default:
179 OS (error, NILF,
180 _("touch: Bad return code from ar_member_touch on '%s'"), name);
181 }
182
183 free (arname);
184
185 return val;
186}
187#endif /* !VMS */

189/* State of an 'ar_glob' run, passed to 'ar_glob_match'. */
190
191/* On VMS, (object) modules in libraries do not have suffixes. That is, to
192 find a match for a pattern, the pattern must not have any suffix. So the
193 suffix of the pattern is saved and the pattern is stripped (ar_glob).
194 If there is a match and the match, which is a module name, is added to
195 the chain, the saved suffix is added back to construct a source filename
196 (ar_glob_match). */
197
198struct ar_glob_state
199 {
200 const char *arname;
201 const char *pattern;
202#ifdef VMS
203 char *suffix;
204#endif
205 unsigned int size;
206 struct nameseq *chain;
207 unsigned int n;
208 };
209
210/* This function is called by 'ar_scan' to match one archive
211 element against the pattern in STATE. */
212
213static long int
214ar_glob_match (int desc UNUSED, const char *mem, int truncated UNUSED,
215 long int hdrpos UNUSED, long int datapos UNUSED,
216 long int size UNUSED, long int date UNUSED, int uid UNUSED,
217 int gid UNUSED, unsigned int mode UNUSED, const void *arg)
218{
219 struct ar_glob_state *state = (struct ar_glob_state *)arg;
220
221 if (fnmatch (state->pattern, mem, FNM_PATHNAME|FNM_PERIOD) == 0)
222 {
223 /* We have a match. Add it to the chain. */
224 struct nameseq *new = xcalloc (state->size);
225#ifdef VMS
226 if (state->suffix)
227 new->name = strcache_add(
228 concat(5, state->arname, "(", mem, state->suffix, ")"));
229 else
230#endif
231 new->name = strcache_add(concat(4, state->arname, "(", mem, ")"));
232 new->next = state->chain;
233 state->chain = new;
234 ++state->n;
235 }
236
237 return 0L;
238}

240/* Return nonzero if PATTERN contains any metacharacters.
241 Metacharacters can be quoted with backslashes if QUOTE is nonzero. */
242static int
243ar_glob_pattern_p (const char *pattern, int quote)
244{
245 const char *p;
246 int opened = 0;
247
248 for (p = pattern; *p != '\0'; ++p)
249 switch (*p)
250 {
251 case '?':
252 case '*':
253 return 1;
254
255 case '\\':
256 if (quote)
257 ++p;
258 break;
259
260 case '[':
261 opened = 1;
262 break;
263
264 case ']':
265 if (opened)
266 return 1;
267 break;
268 }
269
270 return 0;
271}

273/* Glob for MEMBER_PATTERN in archive ARNAME.
274 Return a malloc'd chain of matching elements (or nil if none). */
275
276struct nameseq *
277ar_glob (const char *arname, const char *member_pattern, unsigned int size)
278{
279 struct ar_glob_state state;
280 struct nameseq *n;
281 const char **names;
282 unsigned int i;
283 if (! ar_glob_pattern_p (member_pattern, 1))
284 return 0;
285
286 /* Scan the archive for matches.
287 ar_glob_match will accumulate them in STATE.chain. */
288 state.arname = arname;
289 state.pattern = member_pattern;
290 state.size = size;
291 state.chain = 0;
292 state.n = 0;
293 ar_scan (arname, ar_glob_match, &state);
294
295 if (state.chain == 0)
296 return 0;
297
298 /* Now put the names into a vector for sorting. */
299 names = alloca (state.n * sizeof (const char *));
300 i = 0;
301 for (n = state.chain; n != 0; n = n->next)
302 names[i++] = n->name;
303
304 /* Sort them alphabetically. */
305 /* MSVC erroneously warns without a cast here. */
306 qsort ((void *)names, i, sizeof (*names), alpha_compare);
307
308 /* Put them back into the chain in the sorted order. */
309 i = 0;
310 for (n = state.chain; n != 0; n = n->next)
311 n->name = names[i++];
312
313 return state.chain;
314}
