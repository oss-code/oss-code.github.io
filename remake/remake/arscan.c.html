53/* Takes three arguments ARCHIVE, FUNCTION and ARG.
54
55 Open the archive named ARCHIVE, find its members one by one,
56 and for each one call FUNCTION with the following arguments:
57 archive file descriptor for reading the data,
58 member name,
59 member name might be truncated flag,
60 member header position in file,
61 member data position in file,
62 member data size,
63 member date,
64 member uid,
65 member gid,
66 member protection mode,
67 ARG.
68
69 NOTE: on VMS systems, only name, date, and arg are meaningful!
70
71 The descriptor is poised to read the data of the member
72 when FUNCTION is called. It does not matter how much
73 data FUNCTION reads.
74
75 If FUNCTION returns nonzero, we immediately return
76 what FUNCTION returned.
77
78 Returns -1 if archive does not exist,
79 Returns -2 if archive has invalid format.
80 Returns 0 if have scanned successfully. */
81
82long int
83ar_scan (const char *archive, ar_member_func_t function, const void *varg)
84{
85 char *vms_archive;
86
87 static struct dsc$descriptor_s libdesc =
88 { 0, DSC$K_DTYPE_T, DSC$K_CLASS_S, NULL };
89
90 const unsigned long func = LBR$C_READ;
91 const unsigned long type = LBR$C_TYP_UNK;
92 const unsigned long index = 1;
93 unsigned long lib_idx;
94 int status;
95
96 VMS_saved_arg = varg;
97
98 /* Null archive string can show up in test and cause an access violation */
99 if (archive == NULL)
100 {
101 /* Null filenames do not exist */
102 return -1;
103 }
104
105 /* archive path name must be in VMS format */
106 vms_archive = (char *) vmsify(archive, 0);
107
108 status = lbr$ini_control(&VMS_lib_idx, &func, &type, 0);
109
110 if (!$VMS_STATUS_SUCCESS(status))
111 {
112 ON(error, NILF, _("lbr$ini_control() failed with status = %d"), status);
113 return -2;
114 }
115
116 libdesc.dsc$a_pointer = vms_archive;
117 libdesc.dsc$w_length = strlen(vms_archive);
118
119 status = lbr$open(&VMS_lib_idx, &libdesc, 0, NULL, 0, NULL, 0);
120
121 if (!$VMS_STATUS_SUCCESS(status))
122 {
123
124 /* TODO: A library format failure could mean that this is a file
125 generated by the GNU AR utility and in that case, we need to
126 take the UNIX codepath. This will also take a change to the
127 GNV AR wrapper program. */
128
129 switch (status)
130 {
131 case RMS$_FNF:
132 /* Archive does not exist */
133 return -1;
134 default:
135#ifndef TEST
136 OSN(error, NILF,
137 _("unable to open library '%s' to lookup member status %d"),
138 archive, status);
139#endif
140 /* For library format errors, specification says to return -2 */
141 return -2;
142 }
143 }
144
145 VMS_function = function;
146
147 /* Clear the return status, as we are supposed to stop calling the
148 callback function if it becomes non-zero, and this is a static
149 variable. */
150 VMS_function_ret = 0;
151
152 status = lbr$get_index(&VMS_lib_idx, &index, VMS_get_member_info, NULL, 0);
153
154 lbr$close(&VMS_lib_idx);
155
156 /* Unless a failure occurred in the lbr$ routines, return the
157 the status from the 'function' routine. */
158 if ($VMS_STATUS_SUCCESS(status))
159 {
160 return VMS_function_ret;
161 }
162
163 /* This must be something wrong with the library and an error
164 message should already have been printed. */
165 return -2;
166}

255/* Takes three arguments ARCHIVE, FUNCTION and ARG.
256
257 Open the archive named ARCHIVE, find its members one by one,
258 and for each one call FUNCTION with the following arguments:
259 archive file descriptor for reading the data,
260 member name,
261 member name might be truncated flag,
262 member header position in file,
263 member data position in file,
264 member data size,
265 member date,
266 member uid,
267 member gid,
268 member protection mode,
269 ARG.
270
271 The descriptor is poised to read the data of the member
272 when FUNCTION is called. It does not matter how much
273 data FUNCTION reads.
274
275 If FUNCTION returns nonzero, we immediately return
276 what FUNCTION returned.
277
278 Returns -1 if archive does not exist,
279 Returns -2 if archive has invalid format.
280 Returns 0 if have scanned successfully. */
281
282long int
283ar_scan (const char *archive, ar_member_func_t function, const void *arg)
284{
285#ifdef AIAMAG
286 FL_HDR fl_header;
287# ifdef AIAMAGBIG
288 int big_archive = 0;
289 FL_HDR_BIG fl_header_big;
290# endif
291#endif
292 char *namemap = 0;
293 int desc = open (archive, O_RDONLY, 0);
294 if (desc < 0)
295 return -1;
296#ifdef SARMAG
297 {
298 char buf[SARMAG];
299 int nread;
300 EINTRLOOP (nread, read (desc, buf, SARMAG));
301 if (nread != SARMAG || memcmp (buf, ARMAG, SARMAG))
302 {
303 (void) close (desc);
304 return -2;
305 }
306 }
307#else
308#ifdef AIAMAG
309 {
310 int nread;
311 EINTRLOOP (nread, read (desc, &fl_header, FL_HSZ));
312 if (nread != FL_HSZ)
313 {
314 (void) close (desc);
315 return -2;
316 }
317#ifdef AIAMAGBIG
318 /* If this is a "big" archive, then set the flag and
319 re-read the header into the "big" structure. */
320 if (!memcmp (fl_header.fl_magic, AIAMAGBIG, SAIAMAG))
321 {
322 off_t o;
323
324 big_archive = 1;
325
326 /* seek back to beginning of archive */
327 EINTRLOOP (o, lseek (desc, 0, 0));
328 if (o < 0)
329 {
330 (void) close (desc);
331 return -2;
332 }
333
334 /* re-read the header into the "big" structure */
335 EINTRLOOP (nread, read (desc, &fl_header_big, FL_HSZ_BIG));
336 if (nread != FL_HSZ_BIG)
337 {
338 (void) close (desc);
339 return -2;
340 }
341 }
342 else
343#endif
344 /* Check to make sure this is a "normal" archive. */
345 if (memcmp (fl_header.fl_magic, AIAMAG, SAIAMAG))
346 {
347 (void) close (desc);
348 return -2;
349 }
350 }
351#else
352 {
353#ifndef M_XENIX
354 int buf;
355#else
356 unsigned short int buf;
357#endif
358 int nread;
359 EINTRLOOP (nread, read (desc, &buf, sizeof (buf)));
360 if (nread != sizeof (buf) || buf != ARMAG)
361 {
362 (void) close (desc);
363 return -2;
364 }
365 }
366#endif
367#endif
368
369 /* Now find the members one by one. */
370 {
371#ifdef SARMAG
372 register long int member_offset = SARMAG;
373#else
374#ifdef AIAMAG
375 long int member_offset;
376 long int last_member_offset;
377
378#ifdef AIAMAGBIG
379 if ( big_archive )
380 {
381 sscanf (fl_header_big.fl_fstmoff, "%20ld", &member_offset);
382 sscanf (fl_header_big.fl_lstmoff, "%20ld", &last_member_offset);
383 }
384 else
385#endif
386 {
387 sscanf (fl_header.fl_fstmoff, "%12ld", &member_offset);
388 sscanf (fl_header.fl_lstmoff, "%12ld", &last_member_offset);
389 }
390
391 if (member_offset == 0)
392 {
393 /* Empty archive. */
394 close (desc);
395 return 0;
396 }
397#else
398 register long int member_offset = sizeof (int);
399#endif
400#endif
401
402 while (1)
403 {
404 register int nread;
405 struct ar_hdr member_header;
406#ifdef AIAMAGBIG
407 struct ar_hdr_big member_header_big;
408#endif
409#ifdef AIAMAG
410 char name[256];
411 int name_len;
412 long int dateval;
413 int uidval, gidval;
414 long int data_offset;
415#else
416 char namebuf[sizeof member_header.ar_name + 1];
417 char *name;
418 int is_namemap; /* Nonzero if this entry maps long names. */
419 int long_name = 0;
420#endif
421 long int eltsize;
422 unsigned int eltmode;
423 long int fnval;
424 off_t o;
425
426 EINTRLOOP (o, lseek (desc, member_offset, 0));
427 if (o < 0)
428 {
429 (void) close (desc);
430 return -2;
431 }
432
433#ifdef AIAMAG
434#define AR_MEMHDR_SZ(x) (sizeof(x) - sizeof (x._ar_name))
435
436#ifdef AIAMAGBIG
437 if (big_archive)
438 {
439 EINTRLOOP (nread, read (desc, &member_header_big,
440 AR_MEMHDR_SZ(member_header_big)));
441
442 if (nread != AR_MEMHDR_SZ(member_header_big))
443 {
444 (void) close (desc);
445 return -2;
446 }
447
448 sscanf (member_header_big.ar_namlen, "%4d", &name_len);
449 EINTRLOOP (nread, read (desc, name, name_len));
450
451 if (nread != name_len)
452 {
453 (void) close (desc);
454 return -2;
455 }
456
457 name[name_len] = 0;
458
459 sscanf (member_header_big.ar_date, "%12ld", &dateval);
460 sscanf (member_header_big.ar_uid, "%12d", &uidval);
461 sscanf (member_header_big.ar_gid, "%12d", &gidval);
462 sscanf (member_header_big.ar_mode, "%12o", &eltmode);
463 sscanf (member_header_big.ar_size, "%20ld", &eltsize);
464
465 data_offset = (member_offset + AR_MEMHDR_SZ(member_header_big)
466 + name_len + 2);
467 }
468 else
469#endif
470 {
471 EINTRLOOP (nread, read (desc, &member_header,
472 AR_MEMHDR_SZ(member_header)));
473
474 if (nread != AR_MEMHDR_SZ(member_header))
475 {
476 (void) close (desc);
477 return -2;
478 }
479
480 sscanf (member_header.ar_namlen, "%4d", &name_len);
481 EINTRLOOP (nread, read (desc, name, name_len));
482
483 if (nread != name_len)
484 {
485 (void) close (desc);
486 return -2;
487 }
488
489 name[name_len] = 0;
490
491 sscanf (member_header.ar_date, "%12ld", &dateval);
492 sscanf (member_header.ar_uid, "%12d", &uidval);
493 sscanf (member_header.ar_gid, "%12d", &gidval);
494 sscanf (member_header.ar_mode, "%12o", &eltmode);
495 sscanf (member_header.ar_size, "%12ld", &eltsize);
496
497 data_offset = (member_offset + AR_MEMHDR_SZ(member_header)
498 + name_len + 2);
499 }
500 data_offset += data_offset % 2;
501
502 fnval =
503 (*function) (desc, name, 0,
504 member_offset, data_offset, eltsize,
505 dateval, uidval, gidval,
506 eltmode, arg);
507
508#else /* Not AIAMAG. */
509 EINTRLOOP (nread, read (desc, &member_header, AR_HDR_SIZE));
510 if (nread == 0)
511 /* No data left means end of file; that is OK. */
512 break;
513
514 if (nread != AR_HDR_SIZE
515#if defined(ARFMAG) || defined(ARFZMAG)
516 || (
517# ifdef ARFMAG
518 memcmp (member_header.ar_fmag, ARFMAG, 2)
519# else
520 1
521# endif
522 &&
523# ifdef ARFZMAG
524 memcmp (member_header.ar_fmag, ARFZMAG, 2)
525# else
526 1
527# endif
528 )
529#endif
530 )
531 {
532 (void) close (desc);
533 return -2;
534 }
535
536 name = namebuf;
537 memcpy (name, member_header.ar_name, sizeof member_header.ar_name);
538 {
539 register char *p = name + sizeof member_header.ar_name;
540 do
541 *p = '\0';
542 while (p > name && *--p == ' ');
543
544#ifndef AIAMAG
545 /* If the member name is "//" or "ARFILENAMES/" this may be
546 a list of file name mappings. The maximum file name
547 length supported by the standard archive format is 14
548 characters. This member will actually always be the
549 first or second entry in the archive, but we don't check
550 that. */
551 is_namemap = (!strcmp (name, "//")
552 || !strcmp (name, "ARFILENAMES/"));
553#endif /* Not AIAMAG. */
554 /* On some systems, there is a slash after each member name. */
555 if (*p == '/')
556 *p = '\0';
557
558#ifndef AIAMAG
559 /* If the member name starts with a space or a slash, this
560 is an index into the file name mappings (used by GNU ar).
561 Otherwise if the member name looks like #1/NUMBER the
562 real member name appears in the element data (used by
563 4.4BSD). */
564 if (! is_namemap
565 && (name[0] == ' ' || name[0] == '/')
566 && namemap != 0)
567 {
568 name = namemap + atoi (name + 1);
569 long_name = 1;
570 }
571 else if (name[0] == '#'
572 && name[1] == '1'
573 && name[2] == '/')
574 {
575 int namesize = atoi (name + 3);
576
577 name = alloca (namesize + 1);
578 EINTRLOOP (nread, read (desc, name, namesize));
579 if (nread != namesize)
580 {
581 close (desc);
582 return -2;
583 }
584 name[namesize] = '\0';
585
586 long_name = 1;
587 }
588#endif /* Not AIAMAG. */
589 }
590
591 sscanf (TOCHAR (member_header.ar_mode), "%o", &eltmode);
592 eltsize = atol (TOCHAR (member_header.ar_size));
593
594 fnval =
595 (*function) (desc, name, ! long_name, member_offset,
596 member_offset + AR_HDR_SIZE, eltsize,
597 atol (TOCHAR (member_header.ar_date)),
598 atoi (TOCHAR (member_header.ar_uid)),
599 atoi (TOCHAR (member_header.ar_gid)),
600 eltmode, arg);
601
602#endif /* AIAMAG. */
603
604 if (fnval)
605 {
606 (void) close (desc);
607 return fnval;
608 }
609
610#ifdef AIAMAG
611 if (member_offset == last_member_offset)
612 /* End of the chain. */
613 break;
614
615#ifdef AIAMAGBIG
616 if (big_archive)
617 sscanf (member_header_big.ar_nxtmem, "%20ld", &member_offset);
618 else
619#endif
620 sscanf (member_header.ar_nxtmem, "%12ld", &member_offset);
621
622 if (lseek (desc, member_offset, 0) != member_offset)
623 {
624 (void) close (desc);
625 return -2;
626 }
627#else
628
629 /* If this member maps archive names, we must read it in. The
630 name map will always precede any members whose names must
631 be mapped. */
632 if (is_namemap)
633 {
634 char *clear;
635 char *limit;
636
637 namemap = alloca (eltsize);
638 EINTRLOOP (nread, read (desc, namemap, eltsize));
639 if (nread != eltsize)
640 {
641 (void) close (desc);
642 return -2;
643 }
644
645 /* The names are separated by newlines. Some formats have
646 a trailing slash. Null terminate the strings for
647 convenience. */
648 limit = namemap + eltsize;
649 for (clear = namemap; clear < limit; clear++)
650 {
651 if (*clear == '\n')
652 {
653 *clear = '\0';
654 if (clear[-1] == '/')
655 clear[-1] = '\0';
656 }
657 }
658
659 is_namemap = 0;
660 }
661
662 member_offset += AR_HDR_SIZE + eltsize;
663 if (member_offset % 2 != 0)
664 member_offset++;
665#endif
666 }
667 }
668
669 close (desc);
670 return 0;
671}
672#endif /* !VMS */

674/* Return nonzero iff NAME matches MEM.
675 If TRUNCATED is nonzero, MEM may be truncated to
676 sizeof (struct ar_hdr.ar_name) - 1. */
677
678int
679ar_name_equal (const char *name, const char *mem, int truncated)
680{
681 const char *p;
682
683 p = strrchr (name, '/');
684 if (p != 0)
685 name = p + 1;
686
687#ifndef VMS
688 if (truncated)
689 {
690#ifdef AIAMAG
691 /* TRUNCATED should never be set on this system. */
692 abort ();
693#else
694 struct ar_hdr hdr;
695#if !defined (__hpux) && !defined (cray)
696 return strneq (name, mem, sizeof (hdr.ar_name) - 1);
697#else
698 return strneq (name, mem, sizeof (hdr.ar_name) - 2);
699#endif /* !__hpux && !cray */
700#endif /* !AIAMAG */
701 }
702
703 return !strcmp (name, mem);
704#else
705 /* VMS members do not have suffixes, but the filenames usually
706 have.
707 Do we need to strip VMS disk/directory format paths?
708
709 Most VMS compilers etc. by default are case insensitive
710 but produce uppercase external names, incl. module names.
711 However the VMS librarian (ar) and the linker by default
712 are case sensitive: they take what they get, usually
713 uppercase names. So for the non-default settings of the
714 compilers etc. there is a need to have a case sensitive
715 mode. */
716 {
717 int len;
718 len = strlen(mem);
719 int match;
720 char *dot;
721 if ((dot=strrchr(name,'.')))
722 match = (len == dot - name) && !strncasecmp(name, mem, len);
723 else
724 match = !strcasecmp (name, mem);
725 return match;
726 }
727#endif /* !VMS */
728}

730#ifndef VMS
731/* ARGSUSED */
732static long int
733ar_member_pos (int desc UNUSED, const char *mem, int truncated,
734 long int hdrpos, long int datapos UNUSED, long int size UNUSED,
735 long int date UNUSED, int uid UNUSED, int gid UNUSED,
736 unsigned int mode UNUSED, const void *name)
737{
738 if (!ar_name_equal (name, mem, truncated))
739 return 0;
740 return hdrpos;
741}

743/* Set date of member MEMNAME in archive ARNAME to current time.
744 Returns 0 if successful,
745 -1 if file ARNAME does not exist,
746 -2 if not a valid archive,
747 -3 if other random system call error (including file read-only),
748 1 if valid but member MEMNAME does not exist. */
749
750int
751ar_member_touch (const char *arname, const char *memname)
752{
753 long int pos = ar_scan (arname, ar_member_pos, memname);
754 int fd;
755 struct ar_hdr ar_hdr;
756 off_t o;
757 int r;
758 unsigned int ui;
759 struct stat statbuf;
760
761 if (pos < 0)
762 return (int) pos;
763 if (!pos)
764 return 1;
765
766 EINTRLOOP (fd, open (arname, O_RDWR, 0666));
767 if (fd < 0)
768 return -3;
769 /* Read in this member's header */
770 EINTRLOOP (o, lseek (fd, pos, 0));
771 if (o < 0)
772 goto lose;
773 EINTRLOOP (r, read (fd, &ar_hdr, AR_HDR_SIZE));
774 if (r != AR_HDR_SIZE)
775 goto lose;
776 /* Write back the header, thus touching the archive file. */
777 EINTRLOOP (o, lseek (fd, pos, 0));
778 if (o < 0)
779 goto lose;
780 EINTRLOOP (r, write (fd, &ar_hdr, AR_HDR_SIZE));
781 if (r != AR_HDR_SIZE)
782 goto lose;
783 /* The file's mtime is the time we we want. */
784 EINTRLOOP (r, fstat (fd, &statbuf));
785 if (r < 0)
786 goto lose;
787#if defined(ARFMAG) || defined(ARFZMAG) || defined(AIAMAG) || defined(WINDOWS32)
788 /* Advance member's time to that time */
789 for (ui = 0; ui < sizeof ar_hdr.ar_date; ui++)
790 ar_hdr.ar_date[ui] = ' ';
791 sprintf (TOCHAR (ar_hdr.ar_date), "%lu", (long unsigned) statbuf.st_mtime);
792#ifdef AIAMAG
793 ar_hdr.ar_date[strlen (ar_hdr.ar_date)] = ' ';
794#endif
795#else
796 ar_hdr.ar_date = statbuf.st_mtime;
797#endif
798 /* Write back this member's header */
799 EINTRLOOP (o, lseek (fd, pos, 0));
800 if (o < 0)
801 goto lose;
802 EINTRLOOP (r, write (fd, &ar_hdr, AR_HDR_SIZE));
803 if (r != AR_HDR_SIZE)
804 goto lose;
805 close (fd);
806 return 0;
807
808 lose:
809 r = errno;
810 close (fd);
811 errno = r;
812 return -3;
813}
