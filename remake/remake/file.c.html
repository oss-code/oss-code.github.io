47/* Rehash FILE to NAME. This is not as simple as resetting
48 the 'hname' member, since it must be put in a new hash bucket,
49 and possibly merged with an existing file called NAME. */
50
51void
52rehash_file (struct file *from_file, const char *to_hname)
53{
54 struct file file_key;
55 struct file **file_slot;
56 struct file *to_file;
57 struct file *deleted_file;
58 struct file *f;
59
60 /* If it's already that name, we're done. */
61 from_file->builtin = 0;
62 file_key.hname = to_hname;
63 if (! file_hash_cmp (from_file, &file_key))
64 return;
65
66 /* Find the end of the renamed list for the "from" file. */
67 file_key.hname = from_file->hname;
68 while (from_file->renamed != 0)
69 from_file = from_file->renamed;
70 if (file_hash_cmp (from_file, &file_key))
71 /* hname changed unexpectedly!! */
72 abort ();
73
74 /* Remove the "from" file from the hash. */
75 deleted_file = hash_delete (&files, from_file);
76 if (deleted_file != from_file)
77 /* from_file isn't the one stored in files */
78 abort ();
79
80 /* Find where the newly renamed file will go in the hash. */
81 file_key.hname = to_hname;
82 file_slot = (struct file **) hash_find_slot (&files, &file_key);
83 to_file = *file_slot;
84
85 /* Change the hash name for this file. */
86 from_file->hname = to_hname;
87 for (f = from_file->double_colon; f != 0; f = f->prev)
88 f->hname = to_hname;
89
90 /* If the new name doesn't exist yet just set it to the renamed file. */
91 if (HASH_VACANT (to_file))
92 {
93 hash_insert_at (&files, from_file, file_slot);
94 return;
95 }
96
97 /* TO_FILE already exists under TO_HNAME.
98 We must retain TO_FILE and merge FROM_FILE into it. */
99
100 if (from_file->cmds != 0)
101 {
102 if (to_file->cmds == 0)
103 to_file->cmds = from_file->cmds;
104 else if (from_file->cmds != to_file->cmds)
105 {
106 size_t l = strlen (from_file->name);
107 /* We have two sets of commands. We will go with the
108 one given in the rule explicitly mentioning this name,
109 but give a message to let the user know what's going on. */
110 if (to_file->cmds->fileinfo.filenm != 0)
111 error (&from_file->cmds->fileinfo,
112 l + strlen (to_file->cmds->fileinfo.filenm) + INTSTR_LENGTH,
113 _("Recipe was specified for file '%s' at %s:%lu,"),
114 from_file->name, to_file->cmds->fileinfo.filenm,
115 to_file->cmds->fileinfo.lineno);
116 else
117 error (&from_file->cmds->fileinfo, l,
118 _("Recipe for file '%s' was found by implicit rule search,"),
119 from_file->name);
120 l += strlen (to_hname);
121 error (&from_file->cmds->fileinfo, l,
122 _("but '%s' is now considered the same file as '%s'."),
123 from_file->name, to_hname);
124 error (&from_file->cmds->fileinfo, l,
125 _("Recipe for '%s' will be ignored in favor of the one for '%s'."),
126 to_hname, from_file->name);
127 }
128 }
129
130 /* Merge the dependencies of the two files. */
131
132 if (to_file->deps == 0)
133 to_file->deps = from_file->deps;
134 else
135 {
136 struct dep *deps = to_file->deps;
137 while (deps->next != 0)
138 deps = deps->next;
139 deps->next = from_file->deps;
140 }
141
142 merge_variable_set_lists (&to_file->variables, from_file->variables);
143
144 if (to_file->double_colon && from_file->is_target && !from_file->double_colon)
145 OSS (fatal, NILF, _("can't rename single-colon '%s' to double-colon '%s'"),
146 from_file->name, to_hname);
147 if (!to_file->double_colon && from_file->double_colon)
148 {
149 if (to_file->is_target)
150 OSS (fatal, NILF,
151 _("can't rename double-colon '%s' to single-colon '%s'"),
152 from_file->name, to_hname);
153 else
154 to_file->double_colon = from_file->double_colon;
155 }
156
157 if (from_file->last_mtime > to_file->last_mtime)
158 /* %%% Kludge so -W wins on a file that gets vpathized. */
159 to_file->last_mtime = from_file->last_mtime;
160
161 to_file->mtime_before_update = from_file->mtime_before_update;
162
163#define MERGE(field) to_file->field |= from_file->field
164 MERGE (precious);
165 MERGE (tried_implicit);
166 MERGE (updating);
167 MERGE (updated);
168 MERGE (is_target);
169 MERGE (cmd_target);
170 MERGE (phony);
171 MERGE (loaded);
172 MERGE (ignore_vpath);
173#undef MERGE
174
175 to_file->builtin = 0;
176 from_file->renamed = to_file;
177}

179/* Rename FILE to NAME. This is not as simple as resetting
180 the 'name' member, since it must be put in a new hash bucket,
181 and possibly merged with an existing file called NAME. */
182
183void
184rename_file (struct file *from_file, const char *to_hname)
185{
186 rehash_file (from_file, to_hname);
187 while (from_file)
188 {
189 from_file->name = from_file->hname;
190 from_file = from_file->prev;
191 }
192}

194/* Remove all nonprecious intermediate files.
195 If SIG is nonzero, this was caused by a fatal signal,
196 meaning that a different message will be printed, and
197 the message will go to stderr rather than stdout. */
198
199void
200remove_intermediates (int sig)
201{
202 struct file **file_slot;
203 struct file **file_end;
204 int doneany = 0;
205
206 /* If there's no way we will ever remove anything anyway, punt early. */
207 if (question_flag || touch_flag || all_secondary)
208 return;
209
210 if (sig && just_print_flag)
211 return;
212
213 file_slot = (struct file **) files.ht_vec;
214 file_end = file_slot + files.ht_size;
215 for ( ; file_slot < file_end; file_slot++)
216 if (! HASH_VACANT (*file_slot))
217 {
218 struct file *f = *file_slot;
219 /* Is this file eligible for automatic deletion?
220 Yes, IFF: it's marked intermediate, it's not secondary, it wasn't
221 given on the command line, and it's either a -include makefile or
222 it's not precious. */
223 if (f->intermediate && (f->dontcare || !f->precious)
224 && !f->secondary && !f->cmd_target)
225 {
226 int status;
227 if (f->update_status == us_none)
228 /* If nothing would have created this file yet,
229 don't print an "rm" command for it. */
230 continue;
231 if (just_print_flag)
232 status = 0;
233 else
234 {
235 status = unlink (f->name);
236 if (status < 0 && errno == ENOENT)
237 continue;
238 }
239 if (!f->dontcare)
240 {
241 if (sig)
242 OS (error, NILF,
243 _("*** Deleting intermediate file '%s'"), f->name);
244 else
245 {
246 if (! doneany)
247 DB (DB_BASIC, (_("Removing intermediate files...\n")));
248 if (!silent_flag)
249 {
250 if (! doneany)
251 {
252 fputs ("rm ", stdout);
253 doneany = 1;
254 }
255 else
256 putchar (' ');
257 fputs (f->name, stdout);
258 fflush (stdout);
259 }
260 }
261 if (status < 0)
262 perror_with_name ("unlink: ", f->name);
263 }
264 }
265 }
266
267 if (doneany && !sig)
268 {
269 putchar ('\n');
270 fflush (stdout);
271 }
272}

274/* Given a string containing prerequisites (fully expanded), break it up into
275 a struct dep list. Enter each of these prereqs into the file database.
276 */
277struct dep *
278split_prereqs (char *p)
279{
280 struct dep *new = PARSE_FILE_SEQ (&p, struct dep, MAP_PIPE, NULL,
281 PARSEFS_NONE);
282
283 if (*p)
284 {
285 /* Files that follow '|' are "order-only" prerequisites that satisfy the
286 dependency by existing: their modification times are irrelevant. */
287 struct dep *ood;
288
289 ++p;
290 ood = PARSE_SIMPLE_SEQ (&p, struct dep);
291
292 if (! new)
293 new = ood;
294 else
295 {
296 struct dep *dp;
297 for (dp = new; dp->next != NULL; dp = dp->next)
298 ;
299 dp->next = ood;
300 }
301
302 for (; ood != NULL; ood = ood->next)
303 ood->ignore_mtime = 1;
304 }
305
306 return new;
307}

309/* Given a list of prerequisites, enter them into the file database.
310 If STEM is set then first expand patterns using STEM. */
311struct dep *
312enter_prereqs (struct dep *deps, const char *stem)
313{
314 struct dep *d1;
315
316 if (deps == 0)
317 return 0;
318
319 /* If we have a stem, expand the %'s. We use patsubst_expand to translate
320 the prerequisites' patterns into plain prerequisite names. */
321 if (stem)
322 {
323 const char *pattern = "%";
324 char *buffer = variable_expand ("");
325 struct dep *dp = deps, *dl = 0;
326
327 while (dp != 0)
328 {
329 char *percent;
330 int nl = strlen (dp->name) + 1;
331 char *nm = alloca (nl);
332 memcpy (nm, dp->name, nl);
333 percent = find_percent (nm);
334 if (percent)
335 {
336 char *o;
337
338 /* We have to handle empty stems specially, because that
339 would be equivalent to $(patsubst %,dp->name,) which
340 will always be empty. */
341 if (stem[0] == '\0')
342 {
343 memmove (percent, percent+1, strlen (percent));
344 o = variable_buffer_output (buffer, nm, strlen (nm) + 1);
345 }
346 else
347 o = patsubst_expand_pat (buffer, stem, pattern, nm,
348 pattern+1, percent+1);
349
350 /* If the name expanded to the empty string, ignore it. */
351 if (buffer[0] == '\0')
352 {
353 struct dep *df = dp;
354 if (dp == deps)
355 dp = deps = deps->next;
356 else
357 dp = dl->next = dp->next;
358 free_dep (df);
359 continue;
360 }
361
362 /* Save the name. */
363 dp->name = strcache_add_len (buffer, o - buffer);
364 }
365 dp->stem = stem;
366 dp->staticpattern = 1;
367 dl = dp;
368 dp = dp->next;
369 }
370 }
371
372 /* Enter them as files, unless they need a 2nd expansion. */
373 for (d1 = deps; d1 != 0; d1 = d1->next)
374 {
375 if (d1->need_2nd_expansion)
376 continue;
377
378 d1->file = lookup_file (d1->name);
379 if (d1->file == 0)
380 d1->file = enter_file (d1->name);
381 d1->staticpattern = 0;
382 d1->name = 0;
383 }
384
385 return deps;
386}

397/* Expand and parse each dependency line. */
398void
399expand_deps (struct file *f)
400{
401 struct dep *d;
402 struct dep **dp;
403 const char *file_stem = f->stem;
404 int initialized = 0;
405
406 f->updating = 0;
407
408 /* Walk through the dependencies. For any dependency that needs 2nd
409 expansion, expand it then insert the result into the list. */
410 dp = &f->deps;
411 d = f->deps;
412 while (d != 0)
413 {
414 char *p;
415 struct dep *new, *next;
416 char *name = (char *)d->name;
417
418 if (! d->name || ! d->need_2nd_expansion)
419 {
420 /* This one is all set already. */
421 dp = &d->next;
422 d = d->next;
423 continue;
424 }
425
426 /* If it's from a static pattern rule, convert the patterns into
427 "$*" so they'll expand properly. */
428 if (d->staticpattern)
429 {
430 char *o = variable_expand ("");
431 o = subst_expand (o, name, "%", "$*", 1, 2, 0);
432 *o = '\0';
433 free (name);
434 d->name = name = xstrdup (variable_buffer);
435 d->staticpattern = 0;
436 }
437
438 /* We're going to do second expansion so initialize file variables for
439 the file. Since the stem for static pattern rules comes from
440 individual dep lines, we will temporarily set f->stem to d->stem. */
441 if (!initialized)
442 {
443 initialize_file_variables (f, 0);
444 initialized = 1;
445 }
446
447 if (d->stem != 0)
448 f->stem = d->stem;
449
450 set_file_variables (f);
451
452 p = variable_expand_for_file (d->name, f);
453
454 if (d->stem != 0)
455 f->stem = file_stem;
456
457 /* At this point we don't need the name anymore: free it. */
458 free (name);
459
460 /* Parse the prerequisites and enter them into the file database. */
461 new = enter_prereqs (split_prereqs (p), d->stem);
462
463 /* If there were no prereqs here (blank!) then throw this one out. */
464 if (new == 0)
465 {
466 *dp = d->next;
467 free_dep (d);
468 d = *dp;
469 continue;
470 }
471
472 /* Add newly parsed prerequisites. */
473 next = d->next;
474 *dp = new;
475 for (dp = &new->next, d = new->next; d != 0; dp = &d->next, d = d->next)
476 ;
477 *dp = next;
478 d = *dp;
479 }
480}

491/* For each dependency of each file, make the 'struct dep' point
492 at the appropriate 'struct file' (which may have to be created).
493
494 Also mark the files depended on by .PRECIOUS, .PHONY, .SILENT,
495 and various other special targets. */
496
497void
498snap_deps (void)
499{
500 struct file *f;
501 struct file *f2;
502 struct dep *d;
503
504 /* Remember that we've done this. Once we start snapping deps we can no
505 longer define new targets. */
506 snapped_deps = 1;
507
508 /* Perform second expansion and enter each dependency name as a file. We
509 must use hash_dump() here because within these loops we likely add new
510 files to the table, possibly causing an in-situ table expansion.
511
512 We only need to do this if second_expansion has been defined; if it
513 hasn't then all deps were expanded as the makefile was read in. If we
514 ever change make to be able to unset .SECONDARY_EXPANSION this will have
515 to change. */
516
517 if (second_expansion)
518 {
519 struct file **file_slot_0 = (struct file **) hash_dump (&files, 0, 0);
520 struct file **file_end = file_slot_0 + files.ht_fill;
521 struct file **file_slot;
522 const char *suffixes;
523
524 /* Expand .SUFFIXES: its prerequisites are used for $$* calc. */
525 f = lookup_file (".SUFFIXES");
526 suffixes = f ? f->name : 0;
527 for (; f != 0; f = f->prev)
528 expand_deps (f);
529
530 /* For every target that's not .SUFFIXES, expand its prerequisites. */
531
532 for (file_slot = file_slot_0; file_slot < file_end; file_slot++)
533 for (f = *file_slot; f != 0; f = f->prev)
534 if (f->name != suffixes)
535 expand_deps (f);
536 free (file_slot_0);
537 }
538 else
539 /* We're not doing second expansion, so reset updating. */
540 hash_map (&files, reset_updating);
541
542 /* Now manage all the special targets. */
543
544 for (f = lookup_file (".PRECIOUS"); f != 0; f = f->prev)
545 for (d = f->deps; d != 0; d = d->next)
546 for (f2 = d->file; f2 != 0; f2 = f2->prev)
547 f2->precious = 1;
548
549 for (f = lookup_file (".LOW_RESOLUTION_TIME"); f != 0; f = f->prev)
550 for (d = f->deps; d != 0; d = d->next)
551 for (f2 = d->file; f2 != 0; f2 = f2->prev)
552 f2->low_resolution_time = 1;
553
554 for (f = lookup_file (".PHONY"); f != 0; f = f->prev)
555 for (d = f->deps; d != 0; d = d->next)
556 for (f2 = d->file; f2 != 0; f2 = f2->prev)
557 {
558 /* Mark this file as phony nonexistent target. */
559 f2->phony = 1;
560 f2->is_target = 1;
561 f2->last_mtime = NONEXISTENT_MTIME;
562 f2->mtime_before_update = NONEXISTENT_MTIME;
563 }
564
565 for (f = lookup_file (".INTERMEDIATE"); f != 0; f = f->prev)
566 /* Mark .INTERMEDIATE deps as intermediate files. */
567 for (d = f->deps; d != 0; d = d->next)
568 for (f2 = d->file; f2 != 0; f2 = f2->prev)
569 f2->intermediate = 1;
570 /* .INTERMEDIATE with no deps does nothing.
571 Marking all files as intermediates is useless since the goal targets
572 would be deleted after they are built. */
573
574 for (f = lookup_file (".SECONDARY"); f != 0; f = f->prev)
575 /* Mark .SECONDARY deps as both intermediate and secondary. */
576 if (f->deps)
577 for (d = f->deps; d != 0; d = d->next)
578 for (f2 = d->file; f2 != 0; f2 = f2->prev)
579 f2->intermediate = f2->secondary = 1;
580 /* .SECONDARY with no deps listed marks *all* files that way. */
581 else
582 {
583 all_secondary = 1;
584 hash_map (&files, set_intermediate);
585 }
586
587 f = lookup_file (".EXPORT_ALL_VARIABLES");
588 if (f != 0 && f->is_target)
589 export_all_variables = 1;
590
591 f = lookup_file (".IGNORE");
592 if (f != 0 && f->is_target)
593 {
594 if (f->deps == 0)
595 ignore_errors_flag = 1;
596 else
597 for (d = f->deps; d != 0; d = d->next)
598 for (f2 = d->file; f2 != 0; f2 = f2->prev)
599 f2->command_flags |= COMMANDS_NOERROR;
600 }
601
602 f = lookup_file (".SILENT");
603 if (f != 0 && f->is_target)
604 {
605 if (f->deps == 0)
606 silent_flag = 1;
607 else
608 for (d = f->deps; d != 0; d = d->next)
609 for (f2 = d->file; f2 != 0; f2 = f2->prev)
610 f2->command_flags |= COMMANDS_SILENT;
611 }
612
613 f = lookup_file (".NOTPARALLEL");
614 if (f != 0 && f->is_target)
615 not_parallel = 1;
616
617#ifndef NO_MINUS_C_MINUS_O
618 /* If .POSIX was defined, remove OUTPUT_OPTION to comply. */
619 /* This needs more work: what if the user sets this in the makefile?
620 if (posix_pedantic)
621 define_variable_cname ("OUTPUT_OPTION", "", o_default, 1);
622 */
623#endif
624}

639/* Convert an external file timestamp to internal form. */
640
641FILE_TIMESTAMP
642file_timestamp_cons (const char *fname, time_t stamp, long int ns)
643{
644 int offset = ORDINARY_MTIME_MIN + (FILE_TIMESTAMP_HI_RES ? ns : 0);
645 FILE_TIMESTAMP s = stamp;
646 FILE_TIMESTAMP product = (FILE_TIMESTAMP) s << FILE_TIMESTAMP_LO_BITS;
647 FILE_TIMESTAMP ts = product + offset;
648
649 if (! (s <= FILE_TIMESTAMP_S (ORDINARY_MTIME_MAX)
650 && product <= ts && ts <= ORDINARY_MTIME_MAX))
651 {
652 char buf[FILE_TIMESTAMP_PRINT_LEN_BOUND + 1];
653 const char *f = fname ? fname : _("Current time");
654 ts = s <= OLD_MTIME ? ORDINARY_MTIME_MIN : ORDINARY_MTIME_MAX;
655 file_timestamp_sprintf (buf, ts);
656 OSS (error, NILF,
657 _("%s: Timestamp out of range; substituting %s"), f, buf);
658 }
659
660 return ts;
661}

663/* Return the current time as a file timestamp, setting *RESOLUTION to
664 its resolution. */
665FILE_TIMESTAMP
666file_timestamp_now (int *resolution)
667{
668 int r;
669 time_t s;
670 int ns;
671
672 /* Don't bother with high-resolution clocks if file timestamps have
673 only one-second resolution. The code below should work, but it's
674 not worth the hassle of debugging it on hosts where it fails. */
675#if FILE_TIMESTAMP_HI_RES
676# if HAVE_CLOCK_GETTIME && defined CLOCK_REALTIME
677 {
678 struct timespec timespec;
679 if (clock_gettime (CLOCK_REALTIME, &timespec) == 0)
680 {
681 r = 1;
682 s = timespec.tv_sec;
683 ns = timespec.tv_nsec;
684 goto got_time;
685 }
686 }
687# endif
688# if HAVE_GETTIMEOFDAY
689 {
690 struct timeval timeval;
691 if (gettimeofday (&timeval, 0) == 0)
692 {
693 r = 1000;
694 s = timeval.tv_sec;
695 ns = timeval.tv_usec * 1000;
696 goto got_time;
697 }
698 }
699# endif
700#endif
701
702 r = 1000000000;
703 s = time ((time_t *) 0);
704 ns = 0;
705
706#if FILE_TIMESTAMP_HI_RES
707 got_time:
708#endif
709 *resolution = r;
710 return file_timestamp_cons (0, s, ns);
711}

713/* Place into the buffer P a printable representation of the file
714 timestamp TS. */
715void
716file_timestamp_sprintf (char *p, FILE_TIMESTAMP ts)
717{
718 time_t t = FILE_TIMESTAMP_S (ts);
719 struct tm *tm = localtime (&t);
720
721 if (tm)
722 sprintf (p, "%04d-%02d-%02d %02d:%02d:%02d",
723 tm->tm_year + 1900, tm->tm_mon + 1, tm->tm_mday,
724 tm->tm_hour, tm->tm_min, tm->tm_sec);
725 else if (t < 0)
726 sprintf (p, "%ld", (long) t);
727 else
728 sprintf (p, "%lu", (unsigned long) t);
729 p += strlen (p);
730
731 /* Append nanoseconds as a fraction, but remove trailing zeros. We don't
732 know the actual timestamp resolution, since clock_getres applies only to
733 local times, whereas this timestamp might come from a remote filesystem.
734 So removing trailing zeros is the best guess that we can do. */
735 sprintf (p, ".%09d", FILE_TIMESTAMP_NS (ts));
736 p += strlen (p) - 1;
737 while (*p == '0')
738 p--;
739 p += *p != '.';
740
741 *p = '\0';
742}

745/*! Print some or all properties of the data base of files.
746*/
747void
748print_target_props (file_t *p_target, print_target_mask_t i_mask)
749{
750 dep_t *d;
751 dep_t *ood = 0;
752
753 putchar ('\n');
754 if (!p_target->is_target)
755 puts (_("# Not a target:"));
756 printf ("%s:%s", p_target->name, p_target->double_colon ? ":" : "");
757
758 /* Print all normal dependencies; note any order-only deps. */
759 for (d = p_target->deps; d != 0; d = d->next)
760 if (! d->ignore_mtime) {
761 if (i_mask & PRINT_TARGET_NONORDER)
762 printf (" %s", dep_name (d));
763 } else if (! ood)
764 ood = d;
765
766 if (i_mask & PRINT_TARGET_ORDER) {
767 /* Print order-only deps, if we have any. */
768 if (ood)
769 {
770 printf (" | %s", dep_name (ood));
771 for (d = ood->next; d != 0; d = d->next)
772 if (d->ignore_mtime)
773 printf (" %s", dep_name (d));
774 }
775 }
776
777 putchar ('\n');
778
779 if (i_mask & PRINT_TARGET_ATTRS) {
780
781 if (p_target->precious)
782 puts (_("# Precious file (prerequisite of .PRECIOUS)."));
783 if (p_target->phony)
784 puts (_("# Phony target (prerequisite of .PHONY)."));
785 if (p_target->cmd_target)
786 puts (_("# Command-line target."));
787 if (p_target->dontcare)
788 puts (_("# A default, MAKEFILES, or -include/sinclude makefile."));
789 puts (p_target->tried_implicit
790 ? _("# Implicit rule search has been done.")
791 : _("# Implicit rule search has not been done."));
792 if (p_target->stem != 0)
793 printf (_("# Implicit/static pattern stem: `%s'\n"), p_target->stem);
794 if (p_target->intermediate)
795 puts (_("# File is an intermediate prerequisite."));
796 if (p_target->also_make != 0)
797 {
798 fputs (_("# Also makes:"), stdout);
799 for (d = p_target->also_make; d != 0; d = d->next)
800 printf (" %s", dep_name (d));
801 putchar ('\n');
802 }
803 }
804
805 if (i_mask & PRINT_TARGET_TIME) {
806
807 if (p_target->last_mtime == UNKNOWN_MTIME)
808 puts (_("# Modification time never checked."));
809 else if (p_target->last_mtime == NONEXISTENT_MTIME)
810 puts (_("# File does not exist."));
811 else if (p_target->last_mtime == OLD_MTIME)
812 puts (_("# File is very old."));
813 else
814 {
815 char buf[FILE_TIMESTAMP_PRINT_LEN_BOUND + 1];
816 file_timestamp_sprintf (buf, p_target->last_mtime);
817 printf (_("# Last modified %s\n"), buf);
818 }
819 puts (p_target->updated
820 ? _("# File has been updated.")
821 : _("# File has not been updated."));
822 }
823
824 if (i_mask & PRINT_TARGET_STATE) {
825
826 switch (p_target->command_state)
827 {
828 case cs_running:
829 puts (_("# Commands currently running (THIS IS A BUG)."));
830 break;
831 case cs_deps_running:
832 puts (_("# Dependencies commands running (THIS IS A BUG)."));
833 break;
834 case cs_not_started:
835 puts (_("# Commands not yet started."));
836 break;
837 case cs_finished:
838 switch (p_target->update_status)
839 {
840 case us_success:
841 puts (_("# Successfully updated."));
842 break;
843 case us_none:
844 puts (_("# No attempt to update has been made."));
845 break;
846 case us_question:
847 assert (question_flag);
848 puts (_("# Needs to be updated (-q is set)."));
849 break;
850 case us_failed:
851 puts (_("# Failed to be updated."));
852 break;
853 default:
854 puts (_("# Invalid value in `update_status' member!"));
855 fflush (stdout);
856 fflush (stderr);
857 abort ();
858 }
859 break;
860 default:
861 puts (_("# Invalid value in `command_state' member!"));
862 fflush (stdout);
863 fflush (stderr);
864 abort ();
865 }
866 }
867
868 if (p_target->variables != 0 && i_mask & PRINT_TARGET_VARS)
869 print_file_variables (p_target, i_mask & PRINT_TARGET_VARS_HASH);
870
871 if (p_target->cmds != 0 && i_mask & PRINT_TARGET_CMDS)
872 print_commands (p_target, p_target->cmds, false);
873
874 if (p_target->cmds != 0 && i_mask & PRINT_TARGET_CMDS_EXP)
875 print_commands (p_target, p_target->cmds, true);
876
877 if (p_target->prev && i_mask & PRINT_TARGET_PREV)
878 print_target_props (p_target->prev, i_mask);
879}

907static void
908print_file (const void *item)
909{
910 const struct file *f = item;
911
912 /* If we're not using builtin targets, don't show them.
913
914 Ideally we'd be able to delete them altogether but currently there's no
915 facility to ever delete a file once it's been added. */
916 if (no_builtin_rules_flag && f->builtin)
917 return;
918
919 putchar ('\n');
920
921 if (f->cmds && f->cmds->recipe_prefix != cmd_prefix)
922 {
923 fputs (".RECIPEPREFIX = ", stdout);
924 cmd_prefix = f->cmds->recipe_prefix;
925 if (cmd_prefix != RECIPEPREFIX_DEFAULT)
926 putchar (cmd_prefix);
927 putchar ('\n');
928 }
929
930 if (f->variables != 0)
931 print_target_variables (f);
932
933 if (!f->is_target)
934 puts (_("# Not a target:"));
935 printf ("%s:%s", f->name, f->double_colon ? ":" : "");
936 print_prereqs (f->deps);
937
938 if (f->precious)
939 puts (_("# Precious file (prerequisite of .PRECIOUS)."));
940 if (f->phony)
941 puts (_("# Phony target (prerequisite of .PHONY)."));
942 if (f->cmd_target)
943 puts (_("# Command line target."));
944 if (f->dontcare)
945 puts (_("# A default, MAKEFILES, or -include/sinclude makefile."));
946 if (f->builtin)
947 puts (_("# Builtin rule"));
948 puts (f->tried_implicit
949 ? _("# Implicit rule search has been done.")
950 : _("# Implicit rule search has not been done."));
951 if (f->stem != 0)
952 printf (_("# Implicit/static pattern stem: '%s'\n"), f->stem);
953 if (f->intermediate)
954 puts (_("# File is an intermediate prerequisite."));
955 if (f->also_make != 0)
956 {
957 const struct dep *d;
958 fputs (_("# Also makes:"), stdout);
959 for (d = f->also_make; d != 0; d = d->next)
960 printf (" %s", dep_name (d));
961 putchar ('\n');
962 }
963 if (f->last_mtime == UNKNOWN_MTIME)
964 puts (_("# Modification time never checked."));
965 else if (f->last_mtime == NONEXISTENT_MTIME)
966 puts (_("# File does not exist."));
967 else if (f->last_mtime == OLD_MTIME)
968 puts (_("# File is very old."));
969 else
970 {
971 char buf[FILE_TIMESTAMP_PRINT_LEN_BOUND + 1];
972 file_timestamp_sprintf (buf, f->last_mtime);
973 printf (_("# Last modified %s\n"), buf);
974 }
975 puts (f->updated
976 ? _("# File has been updated.") : _("# File has not been updated."));
977 switch (f->command_state)
978 {
979 case cs_running:
980 puts (_("# Recipe currently running (THIS IS A BUG)."));
981 break;
982 case cs_deps_running:
983 puts (_("# Dependencies recipe running (THIS IS A BUG)."));
984 break;
985 case cs_not_started:
986 case cs_finished:
987 switch (f->update_status)
988 {
989 case us_none:
990 break;
991 case us_success:
992 puts (_("# Successfully updated."));
993 break;
994 case us_question:
995 assert (question_flag);
996 puts (_("# Needs to be updated (-q is set)."));
997 break;
998 case us_failed:
999 puts (_("# Failed to be updated."));
1000 break;
1001 }
1002 break;
1003 default:
1004 puts (_("# Invalid value in 'command_state' member!"));
1005 fflush (stdout);
1006 fflush (stderr);
1007 abort ();
1008 }
1009
1010 if (f->variables != 0)
1011 print_file_variables (f, 1);
1012
1013 if (f->cmds != 0) {
1014 print_commands (NULL, f->cmds, false);
1015 // print_commands (p_target, p_target->cmds, false);
1016 }
1017
1018 if (f->prev)
1019 print_file ((const void *) f->prev);
1020}

1045#define VERIFY_CACHED(_p,_n) \
1046 do{ \
1047 if (_p->_n && _p->_n[0] && !strcache_iscached (_p->_n)) \
1048 error (NULL, strlen (_p->name) + CSTRLEN (# _n) + strlen (_p->_n), \
1049 _("%s: Field '%s' not cached: %s"), _p->name, # _n, _p->_n); \
1050 }while(0)

1052static void
1053verify_file (const void *item)
1054{
1055 const struct file *f = item;
1056 const struct dep *d;
1057
1058 VERIFY_CACHED (f, name);
1059 VERIFY_CACHED (f, hname);
1060 VERIFY_CACHED (f, vpath);
1061 VERIFY_CACHED (f, stem);
1062
1063 /* Check the deps. */
1064 for (d = f->deps; d != 0; d = d->next)
1065 {
1066 if (! d->need_2nd_expansion)
1067 VERIFY_CACHED (d, name);
1068 VERIFY_CACHED (d, stem);

1078#define EXPANSION_INCREMENT(_l) ((((_l) / 500) + 1) * 500)
1079
1080char *
1081build_target_list (char *value)
1082{
1083 static unsigned long last_targ_count = 0;
1084
1085 if (files.ht_fill != last_targ_count)
1086 {
1087 unsigned long max = EXPANSION_INCREMENT (strlen (value));
1088 unsigned long len;
1089 char *p;
1090 struct file **fp = (struct file **) files.ht_vec;
1091 struct file **end = &fp[files.ht_size];
1092
1093 /* Make sure we have at least MAX bytes in the allocated buffer. */
1094 value = xrealloc (value, max);
1095
1096 p = value;
1097 len = 0;
1098 for (; fp < end; ++fp)
1099 if (!HASH_VACANT (*fp) && (*fp)->is_target)
1100 {
1101 struct file *f = *fp;
1102 int l = strlen (f->name);
1103
1104 len += l + 1;
1105 if (len > max)
1106 {
1107 unsigned long off = p - value;
1108
1109 max += EXPANSION_INCREMENT (l + 1);
1110 value = xrealloc (value, max);
1111 p = &value[off];
1112 }
1113
1114 memcpy (p, f->name, l);
1115 p += l;
1116 *(p++) = ' ';
1117 }
1118 *(p-1) = '\0';
1119
1120 last_targ_count = files.ht_fill;
1121 }
1122
1123 return value;
1124}
