38static unsigned long

52static int
53dep_hash_cmp (const void *x, const void *y)
54{
55 const struct dep *dx = x;
56 const struct dep *dy = y;
57 return strcmp (dep_name (dx), dep_name (dy));

60/* Set FILE's automatic variables up. */
61
62void
63set_file_variables (struct file *file)
64{
65 struct dep *d;
66 const char *at, *percent, *star, *less;
67
68#ifndef NO_ARCHIVES
69 /* If the target is an archive member 'lib(member)',
70 then $@ is 'lib' and $% is 'member'. */
71
72 if (ar_name (file->name))
73 {
74 unsigned int len;
75 const char *cp;
76 char *p;
77
78 cp = strchr (file->name, '(');
79 p = alloca (cp - file->name + 1);
80 memcpy (p, file->name, cp - file->name);
81 p[cp - file->name] = '\0';
82 at = p;
83 len = strlen (cp + 1);
84 p = alloca (len);
85 memcpy (p, cp + 1, len - 1);
86 p[len - 1] = '\0';
87 percent = p;
88 }
89 else
90#endif /* NO_ARCHIVES. */
91 {
92 at = file->name;
93 percent = "";
94 }
95
96 /* $* is the stem from an implicit or static pattern rule. */
97 if (file->stem == 0)
98 {
99 /* In Unix make, $* is set to the target name with
100 any suffix in the .SUFFIXES list stripped off for
101 explicit rules. We store this in the 'stem' member. */
102 const char *name;
103 unsigned int len;
104
105#ifndef NO_ARCHIVES
106 if (ar_name (file->name))
107 {
108 name = strchr (file->name, '(') + 1;
109 len = strlen (name) - 1;
110 }
111 else
112#endif
113 {
114 name = file->name;
115 len = strlen (name);
116 }
117
118 for (d = enter_file (strcache_add (".SUFFIXES"))->deps; d ; d = d->next)
119 {
120 unsigned int slen = strlen (dep_name (d));
121 if (len > slen && strneq (dep_name (d), name + (len - slen), slen))
122 {
123 file->stem = strcache_add_len (name, len - slen);
124 break;
125 }
126 }
127 if (d == 0)
128 file->stem = "";
129 }
130 star = file->stem;
131
132 /* $< is the first not order-only dependency. */
133 less = "";
134 for (d = file->deps; d != 0; d = d->next)
135 if (!d->ignore_mtime)
136 {
137 if (!d->need_2nd_expansion)
138 less = dep_name (d);
139 break;
140 }
141
142 if (file->cmds == default_file->cmds)
143 /* This file got its commands from .DEFAULT.
144 In this case $< is the same as $@. */
145 less = at;
146
147#define DEFINE_VARIABLE(name, len, value) \
148 (void) define_variable_for_file (name,len,value,o_automatic,0,file)
149
150 /* Define the variables. */
151
152 DEFINE_VARIABLE ("<", 1, less);
153 DEFINE_VARIABLE ("*", 1, star);
154 DEFINE_VARIABLE ("@", 1, at);
155 DEFINE_VARIABLE ("%", 1, percent);
156
157 /* Compute the values for $^, $+, $?, and $|. */
158
159 {
160 static char *plus_value=0, *bar_value=0, *qmark_value=0;
161 static unsigned int plus_max=0, bar_max=0, qmark_max=0;
162
163 unsigned int qmark_len, plus_len, bar_len;
164 char *cp;
165 char *caret_value;
166 char *qp;
167 char *bp;
168 unsigned int len;
169
170 struct hash_table dep_hash;
171 void **slot;
172
173 /* Compute first the value for $+, which is supposed to contain
174 duplicate dependencies as they were listed in the makefile. */
175
176 plus_len = 0;
177 bar_len = 0;
178 for (d = file->deps; d != 0; d = d->next)
179 {
180 if (!d->need_2nd_expansion)
181 {
182 if (d->ignore_mtime)
183 bar_len += strlen (dep_name (d)) + 1;
184 else
185 plus_len += strlen (dep_name (d)) + 1;
186 }
187 }
188
189 if (bar_len == 0)
190 bar_len++;
191
192 if (plus_len == 0)
193 plus_len++;
194
195 if (plus_len > plus_max)
196 plus_value = xrealloc (plus_value, plus_max = plus_len);
197
198 cp = plus_value;
199
200 qmark_len = plus_len + 1; /* Will be this or less. */
201 for (d = file->deps; d != 0; d = d->next)
202 if (! d->ignore_mtime && ! d->need_2nd_expansion)
203 {
204 const char *c = dep_name (d);
205
206#ifndef NO_ARCHIVES
207 if (ar_name (c))
208 {
209 c = strchr (c, '(') + 1;
210 len = strlen (c) - 1;
211 }
212 else
213#endif
214 len = strlen (c);
215
216 memcpy (cp, c, len);
217 cp += len;
218 *cp++ = FILE_LIST_SEPARATOR;
219 if (! (d->changed || always_make_flag))
220 qmark_len -= len + 1; /* Don't space in $? for this one. */
221 }
222
223 /* Kill the last space and define the variable. */
224
225 cp[cp > plus_value ? -1 : 0] = '\0';
226 DEFINE_VARIABLE ("+", 1, plus_value);
227
228 /* Compute the values for $^, $?, and $|. */
229
230 cp = caret_value = plus_value; /* Reuse the buffer; it's big enough. */
231
232 if (qmark_len > qmark_max)
233 qmark_value = xrealloc (qmark_value, qmark_max = qmark_len);
234 qp = qmark_value;
235
236 if (bar_len > bar_max)
237 bar_value = xrealloc (bar_value, bar_max = bar_len);
238 bp = bar_value;
239
240 /* Make sure that no dependencies are repeated in $^, $?, and $|. It
241 would be natural to combine the next two loops but we can't do it
242 because of a situation where we have two dep entries, the first
243 is order-only and the second is normal (see below). */
244
245 hash_init (&dep_hash, 500, dep_hash_1, dep_hash_2, dep_hash_cmp);
246
247 for (d = file->deps; d != 0; d = d->next)
248 {
249 if (d->need_2nd_expansion)
250 continue;
251
252 slot = hash_find_slot (&dep_hash, d);
253 if (HASH_VACANT (*slot))
254 hash_insert_at (&dep_hash, d, slot);
255 else
256 {
257 /* Check if the two prerequisites have different ignore_mtime.
258 If so then we need to "upgrade" one that is order-only. */
259
260 struct dep* hd = (struct dep*) *slot;
261
262 if (d->ignore_mtime != hd->ignore_mtime)
263 d->ignore_mtime = hd->ignore_mtime = 0;
264 }
265 }
266
267 for (d = file->deps; d != 0; d = d->next)
268 {
269 const char *c;
270
271 if (d->need_2nd_expansion || hash_find_item (&dep_hash, d) != d)
272 continue;
273
274 c = dep_name (d);
275#ifndef NO_ARCHIVES
276 if (ar_name (c))
277 {
278 c = strchr (c, '(') + 1;
279 len = strlen (c) - 1;
280 }
281 else
282#endif
283 len = strlen (c);
284
285 if (d->ignore_mtime)
286 {
287 memcpy (bp, c, len);
288 bp += len;
289 *bp++ = FILE_LIST_SEPARATOR;
290 }
291 else
292 {
293 memcpy (cp, c, len);
294 cp += len;
295 *cp++ = FILE_LIST_SEPARATOR;
296 if (d->changed || always_make_flag)
297 {
298 memcpy (qp, c, len);
299 qp += len;
300 *qp++ = FILE_LIST_SEPARATOR;
301 }
302 }
303 }
304
305 hash_free (&dep_hash, 0);
306
307 /* Kill the last spaces and define the variables. */
308
309 cp[cp > caret_value ? -1 : 0] = '\0';
310 DEFINE_VARIABLE ("^", 1, caret_value);
311
312 qp[qp > qmark_value ? -1 : 0] = '\0';
313 DEFINE_VARIABLE ("?", 1, qmark_value);
314
315 bp[bp > bar_value ? -1 : 0] = '\0';
316 DEFINE_VARIABLE ("|", 1, bar_value);
317 }
318
319#undef DEFINE_VARIABLE
320}

322/* Chop CMDS up into individual command lines if necessary.
323 Also set the 'lines_flags' and 'any_recurse' members. */
324
325void
326chop_commands (struct commands *cmds)
327{
328 unsigned int nlines, idx;
329 char **lines;
330
331 /* If we don't have any commands,
332 or we already parsed them, never mind. */
333
334 if (!cmds || cmds->command_lines != 0)
335 return;
336
337 /* Chop CMDS->commands up into lines in CMDS->command_lines. */
338
339 if (one_shell)
340 {
341 int l = strlen (cmds->commands);
342
343 nlines = 1;
344 lines = xmalloc (nlines * sizeof (char *));
345 lines[0] = xstrdup (cmds->commands);
346
347 /* Strip the trailing newline. */
348 if (l > 0 && lines[0][l-1] == '\n')
349 lines[0][l-1] = '\0';
350 }
351 else
352 {
353 const char *p;
354
355 nlines = 5;
356 lines = xmalloc (nlines * sizeof (char *));
357 idx = 0;
358 p = cmds->commands;
359 while (*p != '\0')
360 {
361 const char *end = p;
362 find_end:;
363 end = strchr (end, '\n');
364 if (end == 0)
365 end = p + strlen (p);
366 else if (end > p && end[-1] == '\\')
367 {
368 int backslash = 1;
369 const char *b;
370 for (b = end - 2; b >= p && *b == '\\'; --b)
371 backslash = !backslash;
372 if (backslash)
373 {
374 ++end;
375 goto find_end;
376 }
377 }
378
379 if (idx == nlines)
380 {
381 nlines += 2;
382 lines = xrealloc (lines, nlines * sizeof (char *));
383 }
384 lines[idx++] = xstrndup (p, end - p);
385 p = end;
386 if (*p != '\0')
387 ++p;
388 }
389
390 if (idx != nlines)
391 {
392 nlines = idx;
393 lines = xrealloc (lines, nlines * sizeof (char *));
394 }
395 }
396
397 /* Finally, set the corresponding CMDS->lines_flags elements and the
398 CMDS->any_recurse flag. */
399
400 if (nlines > USHRT_MAX)
401 ON (fatal, &cmds->fileinfo, _("Recipe has too many lines (%ud)"), nlines);
402
403 cmds->ncommand_lines = nlines;
404 cmds->command_lines = lines;
405
406 cmds->any_recurse = 0;
407 cmds->lines_flags = xmalloc (nlines);
408
409 for (idx = 0; idx < nlines; ++idx)
410 {
411 unsigned char flags = 0;
412 const char *p = lines[idx];
413
414 while (ISBLANK (*p) || *p == '-' || *p == '@' || *p == '+')
415 switch (*(p++))
416 {
417 case '+':
418 flags |= COMMANDS_RECURSE;
419 break;
420 case '@':
421 flags |= COMMANDS_SILENT;
422 break;
423 case '-':
424 flags |= COMMANDS_NOERROR;
425 break;
426 }
427
428 /* If no explicit '+' was given, look for MAKE variable references. */
429 if (!(flags & COMMANDS_RECURSE)
430 && (strstr (p, "$(MAKE)") != 0 || strstr (p, "${MAKE}") != 0))
431 flags |= COMMANDS_RECURSE;
432
433 cmds->lines_flags[idx] = flags;
434 cmds->any_recurse |= flags & COMMANDS_RECURSE ? 1 : 0;
435 }
436}

438/* Expand the command lines and store the results in LINES. */
439void expand_command_lines(struct commands *cmds, /*out*/ char **lines,
440 struct file *file)
441{
442 unsigned int i;
443
444 for (i = 0; i < cmds->ncommand_lines; ++i)
445 {
446 /* Collapse backslash-newline combinations that are inside variable
447 or function references. These are left alone by the parser so
448 that they will appear in the echoing of commands (where they look
449 nice); and collapsed by construct_command_argv when it tokenizes.
450 But letting them survive inside function invocations loses because
451 we don't want the functions to see them as part of the text. */
452
453 char *in, *out, *ref;
454
455 /* IN points to where in the line we are scanning.
456 OUT points to where in the line we are writing.
457 When we collapse a backslash-newline combination,
458 IN gets ahead of OUT. */
459
460 in = out = cmds->command_lines[i];
461 while ((ref = strchr (in, '$')) != 0)
462 {
463 ++ref; /* Move past the $. */
464
465 if (out != in)
466 /* Copy the text between the end of the last chunk
467 we processed (where IN points) and the new chunk
468 we are about to process (where REF points). */
469 memmove (out, in, ref - in);
470
471 /* Move both pointers past the boring stuff. */
472 out += ref - in;
473 in = ref;
474
475 if (*ref == '(' || *ref == '{')
476 {
477 char openparen = *ref;
478 char closeparen = openparen == '(' ? ')' : '}';
479 int count;
480 char *p;
481
482 *out++ = *in++; /* Copy OPENPAREN. */
483 /* IN now points past the opening paren or brace.
484 Count parens or braces until it is matched. */
485 count = 0;
486 while (*in != '\0')
487 {
488 if (*in == closeparen && --count < 0)
489 break;
490 else if (*in == '\\' && in[1] == '\n')
491 {
492 /* We have found a backslash-newline inside a
493 variable or function reference. Eat it and
494 any following whitespace. */
495
496 int quoted = 0;
497 for (p = in - 1; p > ref && *p == '\\'; --p)
498 quoted = !quoted;
499
500 if (quoted)
501 /* There were two or more backslashes, so this is
502 not really a continuation line. We don't collapse
503 the quoting backslashes here as is done in
504 collapse_continuations, because the line will
505 be collapsed again after expansion. */
506 *out++ = *in++;
507 else
508 {
509 /* Skip the backslash, newline and
510 any following whitespace. */
511 in = next_token (in + 2);
512
513 /* Discard any preceding whitespace that has
514 already been written to the output. */
515 while (out > ref
516 && isblank ((unsigned char)out[-1]))
517 --out;
518
519 /* Replace it all with a single space. */
520 *out++ = ' ';
521 }
522 }
523 else
524 {
525 if (*in == openparen)
526 ++count;
527
528 *out++ = *in++;
529 }
530 }
531 }
532 }
533
534 /* There are no more references in this line to worry about.
535 Copy the remaining uninteresting text to the output. */
536 if (out != in)
537 memmove (out, in, strlen (in) + 1);
538
539 /* Finally, expand the line. */
540 lines[i] = allocated_variable_expand_for_file (cmds->command_lines[i],
541 file);
542 }
543
544}

548/* Execute the commands to remake FILE. If they are currently executing,
549 return or have already finished executing, just return. Otherwise,
550 fork off a child process to run the first command line in the sequence. */
551
552void
553execute_file_commands (file_t *file, target_stack_node_t *p_call_stack)
554{
555 const char *p;
556
557 /* Don't go through all the preparations if
558 the commands are nothing but whitespace. */
559
560 for (p = file->cmds->commands; *p != '\0'; ++p)
561 if (!ISSPACE (*p) && *p != '-' && *p != '@' && *p != '+')
562 break;
563 if (*p == '\0')
564 {
565 /* If there are no commands, assume everything worked. */
566 set_command_state (file, cs_running);
567 file->update_status = us_success;
568 notice_finished_file (file);
569 return;
570 }
571
572 /* First set the automatic variables according to this file. */
573
574 initialize_file_variables (file, 0);
575
576 set_file_variables (file);
577
578 /* If this is a loaded dynamic object, unload it before remaking.
579 Some systems don't support overwriting a loaded object. */
580 if (file->loaded)
581 unload_file (file->name);
582
583 /* Start the commands running. */
584 new_job (file, p_call_stack);
585}

587/* This is set while we are inside fatal_error_signal,
588 so things can avoid nonreentrant operations. */
589
590int handling_fatal_signal = 0;
591
592/* Handle fatal signals. */
593
594RETSIGTYPE
595fatal_error_signal (int sig)
596{
597#ifdef __MSDOS__
598 extern int dos_status, dos_command_running;
599
600 if (dos_command_running)
601 {
602 /* That was the child who got the signal, not us. */
603 dos_status |= (sig << 8);
604 return;
605 }
606 remove_intermediates (1);
607 exit (EXIT_FAILURE);
608#else /* not __MSDOS__ */
609#ifdef WINDOWS32
610 extern HANDLE main_thread;
611
612 /* Windows creates a sperate thread for handling Ctrl+C, so we need
613 to suspend the main thread, or else we will have race conditions
614 when both threads call reap_children. */
615 if (main_thread)
616 {
617 DWORD susp_count = SuspendThread (main_thread);
618
619 if (susp_count != 0)
620 fprintf (stderr, "SuspendThread: suspend count = %ld\n", susp_count);
621 else if (susp_count == (DWORD)-1)
622 {
623 DWORD ierr = GetLastError ();
624
625 fprintf (stderr, "SuspendThread: error %ld: %s\n",
626 ierr, map_windows32_error_to_string (ierr));
627 }
628 }
629#endif
630 handling_fatal_signal = 1;
631
632 /* Set the handling for this signal to the default.
633 It is blocked now while we run this handler. */
634 signal (sig, SIG_DFL);
635
636 /* A termination signal won't be sent to the entire
637 process group, but it means we want to kill the children. */
638
639 if (sig == SIGTERM)
640 {
641 struct child *c;
642 for (c = children; c != 0; c = c->next)
643 if (!c->remote)
644 (void) kill (c->pid, SIGTERM);
645 }
646
647 /* If we got a signal that means the user
648 wanted to kill make, remove pending targets. */
649
650 if (sig == SIGTERM || sig == SIGINT
651#ifdef SIGHUP
652 || sig == SIGHUP
653#endif
654#ifdef SIGQUIT
655 || sig == SIGQUIT
656#endif
657 )
658 {
659 struct child *c;
660
661 /* Remote children won't automatically get signals sent
662 to the process group, so we must send them. */
663 for (c = children; c != 0; c = c->next)
664 if (c->remote)
665 (void) remote_kill (c->pid, sig);
666
667 for (c = children; c != 0; c = c->next)
668 delete_child_targets (c);
669
670 /* Clean up the children. We don't just use the call below because
671 we don't want to print the "Waiting for children" message. */
672 while (job_slots_used > 0)
673 reap_children (1, 0, NULL);
674 }
675 else
676 /* Wait for our children to die. */
677 while (job_slots_used > 0)
678 reap_children (1, 1, NULL);
679
680 /* Delete any non-precious intermediate files that were made. */
681
682 remove_intermediates (1);
683
684#ifdef SIGQUIT
685 if (sig == SIGQUIT)
686 /* We don't want to send ourselves SIGQUIT, because it will
687 cause a core dump. Just exit instead. */
688 exit (MAKE_TROUBLE);
689#endif
690
691#ifdef WINDOWS32
692 if (main_thread)
693 CloseHandle (main_thread);
694 /* Cannot call W32_kill with a pid (it needs a handle). The exit
695 status of 130 emulates what happens in Bash. */
696 exit (130);
697#else
698 /* Signal the same code; this time it will really be fatal. The signal
699 will be unblocked when we return and arrive then to kill us. */
700 if (kill (getpid (), sig) < 0)
701 pfatal_with_name ("kill");
702#endif /* not WINDOWS32 */
703#endif /* not __MSDOS__ */
704}

706/* Delete FILE unless it's precious or not actually a file (phony),
707 and it has changed on disk since we last stat'd it. */
708
709static void
710delete_target (struct file *file, const char *on_behalf_of)
711{
712 struct stat st;
713 int e;
714
715 if (file->precious || file->phony)
716 return;
717
718#ifndef NO_ARCHIVES
719 if (ar_name (file->name))
720 {
721 time_t file_date = (file->last_mtime == NONEXISTENT_MTIME
722 ? (time_t) -1
723 : (time_t) FILE_TIMESTAMP_S (file->last_mtime));
724 time_t member_date = NONEXISTENT_MTIME;
725 int found;
726 found = ar_member_date (file->name, &member_date);
727 if (found && member_date != file_date)
728 {
729 if (on_behalf_of)
730 OSS (error, NILF,
731 _("*** [%s] Archive member '%s' may be bogus; not deleted"),
732 on_behalf_of, file->name);
733 else
734 OS (error, NILF,
735 _("*** Archive member '%s' may be bogus; not deleted"),
736 file->name);
737 }
738 return;
739 }
740#endif
741
742 EINTRLOOP (e, stat (file->name, &st));
743 if (e == 0
744 && S_ISREG (st.st_mode)
745 && FILE_TIMESTAMP_STAT_MODTIME (file->name, st) != file->last_mtime)
746 {
747 if (on_behalf_of)
748 OSS (error, NILF,
749 _("*** [%s] Deleting file '%s'"), on_behalf_of, file->name);
750 else
751 OS (error, NILF, _("*** Deleting file '%s'"), file->name);
752 if (unlink (file->name) < 0
753 && errno != ENOENT) /* It disappeared; so what. */
754 perror_with_name ("unlink: ", file->name);
755 }
756}

759/* Delete all non-precious targets of CHILD unless they were already deleted.
760 Set the flag in CHILD to say they've been deleted. */
761
762void
763delete_child_targets (struct child *child)
764{
765 struct dep *d;
766
767 if (child->deleted)
768 return;
769
770 /* Delete the target file if it changed. */
771 delete_target (child->file, NULL);
772
773 /* Also remove any non-precious targets listed in the 'also_make' member. */
774 for (d = child->file->also_make; d != 0; d = d->next)
775 delete_target (d->file, child->file->name);
776
777 child->deleted = 1;
778}

780/* Print out the commands in CMDS. */
781
782/*!
783 Print out the commands.
784
785 @param p_cmds location of commands to print out.
786 @param p_target used to set automatic variables if it is non-null.
787 @param b_expand if true, expand the commands to remove MAKE variables.
788*/
789void
790print_commands (file_t *p_target, commands_t *p_cmds, bool b_expand)
791{
792 const char *s;
793
794 fputs (_("# recipe to execute"), stdout);
795
796 if (p_cmds->fileinfo.filenm == 0)
797 puts (_(" (built-in):"));
798 else
799 printf (_(" (from '%s', line %lu):\n"),
800 p_cmds->fileinfo.filenm, p_cmds->fileinfo.lineno);
801
802 if (b_expand && p_target) {
803 variable_set_list_t *p_file_vars = NULL;
804 initialize_file_variables (p_target, 0);
805 set_file_variables (p_target);
806 p_file_vars = p_target->variables;
