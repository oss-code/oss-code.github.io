40/* A 'struct ebuffer' controls the origin of the makefile we are currently
41 eval'ing.
42*/
43
44struct ebuffer
45 {
46 char *buffer; /* Start of the current line in the buffer. */
47 char *bufnext; /* Start of the next line in the buffer. */
48 char *bufstart; /* Start of the entire buffer. */
49 unsigned int size; /* Malloc'd size of buffer. */
50 FILE *fp; /* File, or NULL if this is an internal buffer. */
51 gmk_floc floc; /* Info on the file in fp (if any). */
52 };

74/* A 'struct conditionals' contains the information describing
75 all the active conditionals in a makefile.
76
77 The global variable 'conditionals' contains the conditionals
78 information for the current makefile. It is initialized from
79 the static structure 'toplevel_conditionals' and is later changed
80 to new structures for included makefiles. */
81
82struct conditionals
83 {
84 unsigned int if_cmds; /* Depth of conditional nesting. */
85 unsigned int allocated; /* Elts allocated in following arrays. */
86 char *ignoring; /* Are we ignoring or interpreting?
87 0=interpreting, 1=not yet interpreted,
88 2=already interpreted */
89 char *seen_else; /* Have we already seen an 'else'? */
90 };
91
92static struct conditionals toplevel_conditionals;
93static struct conditionals *conditionals = &toplevel_conditionals;

155/* Read in all the makefiles and return a chain of targets to rebuild. */
156
157struct goaldep *
158read_all_makefiles (const char **makefiles)
159{
160 unsigned int num_makefiles = 0;
161
162 /* Create *_LIST variables, to hold the makefiles, targets, and variables
163 we will be reading. */
164
165 define_variable_cname ("MAKEFILE_LIST", "", o_file, 0);
166
167 if (b_debugger_preread && i_debugger_stepping && !in_debugger)
168 enter_debugger (NULL, NULL, 0, DEBUG_READ_HIT);
169
170 DB (DB_BASIC, (_("Reading makefiles...\n")));
171
172 /* If there's a non-null variable MAKEFILES, its value is a list of
173 files to read first thing. But don't let it prevent reading the
174 default makefiles and don't let the default goal come from there. */
175
176 {
177 char *value;
178 char *name, *p;
179 unsigned int length;
180
181 {
182 /* Turn off --warn-undefined-variables while we expand MAKEFILES. */
183 int save = warn_undefined_variables_flag;
184 warn_undefined_variables_flag = 0;
185
186 value = allocated_variable_expand ("$(MAKEFILES)");
187
188 warn_undefined_variables_flag = save;
189 }
190
191 /* Set NAME to the start of next token and LENGTH to its length.
192 MAKEFILES is updated for finding remaining tokens. */
193 p = value;
194
195 while ((name = find_next_token ((const char **)&p, &length)) != 0)
196 {
197 if (*p != '\0')
198 *p++ = '\0';
199 eval_makefile (name, RM_NO_DEFAULT_GOAL|RM_INCLUDED|RM_DONTCARE);
200 }
201
202 free (value);
203 }
204
205 /* Read makefiles specified with -f switches. */
206
207 if (makefiles != 0)
208 while (*makefiles != 0)
209 {
210 struct goaldep *d = eval_makefile (*makefiles, 0);
211
212 if (errno)
213 perror_with_name ("", *makefiles);
214
215 /* Reuse the storage allocated for the read_file. */
216 *makefiles = dep_name (d);
217 ++num_makefiles;
218 ++makefiles;
219 }
220
221 /* If there were no -f switches, try the default names. */
222
223 if (num_makefiles == 0)
224 {
225 static const char *default_makefiles[] =
226#ifdef WINDOWS32
227 { "GNUmakefile", "makefile", "Makefile", "makefile.mak", 0 };
228#else
229 { "GNUmakefile", "makefile", "Makefile", 0 };
230#endif /* !WINDOWS32 */
231 const char **p = default_makefiles;
232 while (*p != 0 && !file_exists_p (*p))
233 ++p;
234
235 if (*p != 0)
236 {
237 eval_makefile (*p, 0);
238 if (errno)
239 perror_with_name ("", *p);
240 }
241 else
242 {
243 /* No default makefile was found. Add the default makefiles to the
244 'read_files' chain so they will be updated if possible. */
245 struct goaldep *tail = read_makefiles;
246 /* Add them to the tail, after any MAKEFILES variable makefiles. */
247 while (tail != 0 && tail->next != 0)
248 tail = tail->next;
249 for (p = default_makefiles; *p != 0; ++p)
250 {
251 struct goaldep *d = alloc_goaldep ();
252 d->file = enter_file (strcache_add (*p));
253 /* Tell update_goal_chain to bail out as soon as this file is
254 made, and main not to die if we can't make this file. */
255 d->flags = RM_DONTCARE;
256 if (tail == 0)
257 read_makefiles = d;
258 else
259 tail->next = d;
260 tail = d;
261 }
262 if (tail != 0)
263 tail->next = 0;
264 }
265 }
266
267 return read_makefiles;
268}

270/* Install a new conditional and return the previous one. */
271
272static struct conditionals *
273install_conditionals (struct conditionals *new)
274{
275 struct conditionals *save = conditionals;
276
277 memset (new, '\0', sizeof (*new));
278 conditionals = new;
279
280 return save;
281}

283/* Free the current conditionals and reinstate a saved one. */
284
285static void
286restore_conditionals (struct conditionals *saved)
287{
288 /* Free any space allocated by conditional_line. */
289 free (conditionals->ignoring);
290 free (conditionals->seen_else);
291
292 /* Restore state. */
293 conditionals = saved;
294}

296static struct goaldep *
297eval_makefile (const char *filename, int flags)
298{
299 struct goaldep *deps;
300 struct ebuffer ebuf;
301 const gmk_floc *curfile;
302 char *expanded = 0;
303 int makefile_errno;
304
305 ebuf.floc.filenm = filename; /* Use the original file name. */
306 ebuf.floc.lineno = 1;
307 ebuf.floc.offset = 0;
308
309 if (ISDB (DB_VERBOSE|DB_READ_MAKEFILES))
310 {
311 printf (_("Reading makefile '%s'"), filename);
312 if (flags & RM_NO_DEFAULT_GOAL)
313 printf (_(" (no default goal)"));
314 if (flags & RM_INCLUDED)
315 printf (_(" (search path)"));
316 if (flags & RM_DONTCARE)
317 printf (_(" (don't care)"));
318 if (flags & RM_NO_TILDE)
319 printf (_(" (no ~ expansion)"));
320 puts ("...");
321 }
322
323 /* First, get a stream to read. */
324
325 /* Expand ~ in FILENAME unless it came from 'include',
326 in which case it was already done. */
327 if (!(flags & RM_NO_TILDE) && filename[0] == '~')
328 {
329 expanded = remake_tilde_expand (filename);
330 if (expanded != 0)
331 filename = expanded;
332 }
333
334 ENULLLOOP (ebuf.fp, fopen (filename, "r"));
335
336 /* Save the error code so we print the right message later. */
337 makefile_errno = errno;
338
339 /* Check for unrecoverable errors: out of mem or FILE slots. */
340 switch (makefile_errno)
341 {
342#ifdef EMFILE
343 case EMFILE:
344#endif
345#ifdef ENFILE
346 case ENFILE:
347#endif
348 case ENOMEM:
349 {
350 const char *err = strerror (makefile_errno);
351 OS (fatal, reading_file, "%s", err);
352 }
353 }
354
355 /* If the makefile wasn't found and it's either a makefile from
356 the 'MAKEFILES' variable or an included makefile,
357 search the included makefile search path for this makefile. */
358 if (ebuf.fp == 0 && (flags & RM_INCLUDED) && *filename != '/')
359 {
360 unsigned int i;
361 for (i = 0; include_directories[i] != 0; ++i)
362 {
363 const char *included = concat (3, include_directories[i],
364 "/", filename);
365 ebuf.fp = fopen (included, "r");
366 if (ebuf.fp)
367 {
368 filename = included;
369 break;
370 }
371 }
372 }
373
374 /* Now we have the final name for this makefile. Enter it into
375 the cache. */
376 filename = strcache_add (filename);
377
378 /* Add FILENAME to the chain of read makefiles. */
379 deps = alloc_goaldep ();
380 deps->next = read_makefiles;
381 read_makefiles = deps;
382 deps->file = lookup_file (filename);
383 if (deps->file == 0)
384 deps->file = enter_file (filename);
385 filename = deps->file->name;
386 deps->flags = flags;
387
388 free (expanded);
389
390 /* If the makefile can't be found at all, give up entirely. */
391
392 if (ebuf.fp == 0)
393 {
394 /* If we did some searching, errno has the error from the last
395 attempt, rather from FILENAME itself. Store it in case the
396 caller wants to use it in a message. */
397 errno = makefile_errno;
398 return deps;
399 }
400
401 /* Set close-on-exec to avoid leaking the makefile to children, such as
402 $(shell ...). */
403#ifdef HAVE_FILENO
404 CLOSE_ON_EXEC (fileno (ebuf.fp));
405#endif
406
407 /* Add this makefile to the list. */
408 do_variable_definition (&ebuf.floc, "MAKEFILE_LIST", filename, o_file,
409 f_append, 0);
410
411 if (b_debugger_preread && i_debugger_stepping && !in_debugger) {
412 enter_debugger (NULL, NULL, 0, DEBUG_READ_HIT);
413 }
414
415 /* Evaluate the makefile */
416
417 ebuf.size = 200;
418 ebuf.buffer = ebuf.bufnext = ebuf.bufstart = xmalloc (ebuf.size);
419
420 curfile = reading_file;
421 reading_file = &ebuf.floc;
422
423 eval (&ebuf, !(flags & RM_NO_DEFAULT_GOAL));
424
425 reading_file = curfile;
426 deps->file->nlines = ebuf.floc.lineno;
427
428 fclose (ebuf.fp);
429
430 free (ebuf.bufstart);
431 alloca (0);
432
433 errno = 0;
434 return deps;
435}

437void
438eval_buffer (char *buffer, const gmk_floc *flocp)
439{
440 struct ebuffer ebuf;
441 struct conditionals *saved;
442 struct conditionals new;
443 const gmk_floc *curfile;
444
445 /* Evaluate the buffer */
446
447 ebuf.size = strlen (buffer);
448 ebuf.buffer = ebuf.bufnext = ebuf.bufstart = buffer;
449 ebuf.fp = NULL;
450
451 if (flocp)
452 ebuf.floc = *flocp;
453 else if (reading_file)
454 ebuf.floc = *reading_file;
455 else
456 {
457 ebuf.floc.filenm = NULL;
458 ebuf.floc.lineno = 1;
459 ebuf.floc.offset = 0;
460 }
461
462 curfile = reading_file;
463 reading_file = &ebuf.floc;
464
465 saved = install_conditionals (&new);
466
467 eval (&ebuf, 1);
468
469 restore_conditionals (saved);
470
471 reading_file = curfile;
472
473 alloca (0);
474}

476/* Check LINE to see if it's a variable assignment or undefine.
477
478 It might use one of the modifiers "export", "override", "private", or it
479 might be one of the conditional tokens like "ifdef", "include", etc.
480
481 If it's not a variable assignment or undefine, VMOD.V_ASSIGN is 0.
482 Returns LINE.
483
484 Returns a pointer to the first non-modifier character, and sets VMOD
485 based on the modifiers found if any, plus V_ASSIGN is 1.
486 */
487static char *
488parse_var_assignment (const char *line, struct vmodifiers *vmod)
489{
490 const char *p;
491 memset (vmod, '\0', sizeof (*vmod));
492
493 /* Find the start of the next token. If there isn't one we're done. */
494 NEXT_TOKEN (line);
495 if (*line == '\0')
496 return (char *)line;
497
498 p = line;
499 while (1)
500 {
501 int wlen;
502 const char *p2;
503 struct variable v;
504
505 p2 = parse_variable_definition (p, &v);
506
507 /* If this is a variable assignment, we're done. */
508 if (p2)
509 break;
510
511 /* It's not a variable; see if it's a modifier. */
512 p2 = end_of_token (p);
513 wlen = p2 - p;
514
515 if (word1eq ("export"))
516 vmod->export_v = 1;
517 else if (word1eq ("override"))
518 vmod->override_v = 1;
519 else if (word1eq ("private"))
520 vmod->private_v = 1;
521 else if (word1eq ("define"))
522 {
523 /* We can't have modifiers after 'define' */
524 vmod->define_v = 1;
525 p = next_token (p2);
526 break;
527 }
528 else if (word1eq ("undefine"))
529 {
530 /* We can't have modifiers after 'undefine' */
531 vmod->undefine_v = 1;
532 p = next_token (p2);
533 break;
534 }
535 else
536 /* Not a variable or modifier: this is not a variable assignment. */
537 return (char *)line;
538
539 /* It was a modifier. Try the next word. */
540 p = next_token (p2);
541 if (*p == '\0')
542 return (char *)line;
543 }
544
545 /* Found a variable assignment or undefine. */
546 vmod->assign_v = 1;
547 return (char *)p;
548}

554/* Read file FILENAME as a makefile and add its contents to the data base.
555
556 SET_DEFAULT is true if we are allowed to set the default goal. */
557
558static void
559eval (struct ebuffer *ebuf, int set_default)
560{
561 char *collapsed = 0;
562 unsigned int collapsed_length = 0;
563 unsigned int commands_len = 200;
564 char *commands;
565 unsigned int commands_idx = 0;
566 unsigned int cmds_started, tgts_started;
567 int ignoring = 0, in_ignored_define = 0;
568 int no_targets = 0; /* Set when reading a rule without targets. */
569 struct nameseq *filenames = 0;
570 char *depstr = 0;
571 long nlines = 0;
572 int two_colon = 0;
573 char prefix = cmd_prefix;
574 const char *pattern = 0;
575 const char *pattern_percent;
576 gmk_floc *fstart;
577 gmk_floc fi;
578
579 static char *target_description; /* Place to store most recent
580 * target description */
581 static char *prev_target_description; /* Most of the time, we read
582 two targets before
583 processing the first. I
584 think this happens because
585 the second target signals
586 the end of the first
587 target. As a result, we
588 need to save two
589 descriptions to be able to
590 use the previous
591 description for the first
592 target. For the last target
593 of the file though, EOF
594 signals the end of the
595 target so we don't use
596 prev_target_description. */
597#define record_waiting_files() \
598 do \
599 { \
600 if (filenames != 0) \
601 { \
602 /* Have we seen two descriptions since this target or one? */ \
603 char *description = target_description_lineno > tgts_started ? \
604 prev_target_description : target_description; \
605 fi.lineno = tgts_started; \
606 fi.offset = 0; \
607 record_files (filenames, pattern, pattern_percent, depstr, \
608 cmds_started, commands, commands_idx, two_colon, \
609 description, prefix, &fi); \
610 /* Don't use the target_description values more than once. */ \
611 if (target_description_lineno > tgts_started) \
612 prev_target_description = NULL; \
613 else \
614 target_description = NULL; \
615 filenames = 0; \
616 } \
617 commands_idx = 0; \
618 no_targets = 0; \
619 pattern = 0; \
620 } while (0)
621
622 pattern_percent = 0;
623 cmds_started = tgts_started = 1;
624
625 fstart = &ebuf->floc;
626 fi.filenm = ebuf->floc.filenm;
627
628 /* Loop over lines in the file.
629 The strategy is to accumulate target names in FILENAMES, dependencies
630 in DEPS and commands in COMMANDS. These are used to define a rule
631 when the start of the next rule (or eof) is encountered.
632
633 When you see a "continue" in the loop below, that means we are moving on
634 to the next line. If you see record_waiting_files(), then the statement
635 we are parsing also finishes the previous rule. */
636
637 commands = xmalloc (200);
638 prev_target_description = NULL;
639 target_description = NULL;
640
641 while (1)
642 {
643 unsigned int linelen;
644 char *line;
645 unsigned int wlen;
646 char *p;
647 char *p2;
648 struct vmodifiers vmod;
649
650 /* At the top of this loop, we are starting a brand new line. */
651 /* Grab the next line to be evaluated */
652 ebuf->floc.lineno += nlines;
653 nlines = readline (ebuf);
654
655 /* If there is nothing left to eval, we're done. */
656 if (nlines < 0)
657 break;
658
659 line = ebuf->buffer;
660
661 /* If this is the first line, check for a UTF-8 BOM and skip it. */
662 if (ebuf->floc.lineno == 1 && line[0] == (char)0xEF
663 && line[1] == (char)0xBB && line[2] == (char)0xBF)
664 {
665 line += 3;
666 if (ISDB(DB_BASIC))
667 {
668 if (ebuf->floc.filenm)
669 printf (_("Skipping UTF-8 BOM in makefile '%s'\n"),
670 ebuf->floc.filenm);
671 else
672 printf (_("Skipping UTF-8 BOM in makefile buffer\n"));
673 }
674 }
675
676 /* If this line is empty, skip it. */
677 if (line[0] == '\0')
678 continue;
679
680 linelen = strlen (line);
681
682 /* Check for a shell command line first.
683 If it is not one, we can stop treating cmd_prefix specially. */
684 if (line[0] == cmd_prefix)
685 {
686 if (no_targets)
687 /* Ignore the commands in a rule with no targets. */
688 continue;
689
690 /* If there is no preceding rule line, don't treat this line
691 as a command, even though it begins with a recipe prefix.
692 SunOS 4 make appears to behave this way. */
693
694 if (filenames != 0)
695 {
696 if (ignoring)
697 /* Yep, this is a shell command, and we don't care. */
698 continue;
699
700 if (commands_idx == 0)
701 cmds_started = ebuf->floc.lineno;
702
703 /* Append this command line to the line being accumulated.
704 Skip the initial command prefix character. */
705 if (linelen + commands_idx > commands_len)
706 {
707 commands_len = (linelen + commands_idx) * 2;
708 commands = xrealloc (commands, commands_len);
709 }
710 memcpy (&commands[commands_idx], line + 1, linelen - 1);
711 commands_idx += linelen - 1;
712 commands[commands_idx++] = '\n';
713 continue;
714 }
715 }
716
717 /* This line is not a shell command line. Don't worry about whitespace.
718 Get more space if we need it; we don't need to preserve the current
719 contents of the buffer. */
720
721 if (collapsed_length < linelen+1)
722 {
723 collapsed_length = linelen+1;
724 free (collapsed);
725 /* Don't need xrealloc: we don't need to preserve the content. */
726 collapsed = xmalloc (collapsed_length);
727 }
728 strcpy (collapsed, line);
729 /* Collapse continuation lines. */
730 collapse_continuations (collapsed);
731 remove_comments (collapsed, &target_description,
732 &prev_target_description, ebuf->floc.lineno);
733
734 /* Get rid if starting space (including formfeed, vtab, etc.) */
735 p = collapsed;
736 NEXT_TOKEN (p);
737
738 /* See if this is a variable assignment. We need to do this early, to
739 allow variables with names like 'ifdef', 'export', 'private', etc. */
740 p = parse_var_assignment (p, &vmod);
741 if (vmod.assign_v)
742 {
743 struct variable *v;
744 enum variable_origin origin = vmod.override_v ? o_override : o_file;
745
746 /* If we're ignoring then we're done now. */
747 if (ignoring)
748 {
749 if (vmod.define_v)
750 in_ignored_define = 1;
751 continue;
752 }
753
754 /* Variable assignment ends the previous rule. */
755 record_waiting_files ();
756
757 if (vmod.undefine_v)
758 {
759 do_undefine (p, origin, ebuf);
760 continue;
761 }
762 else if (vmod.define_v)
763 v = do_define (p, origin, ebuf);
764 else
765 v = try_variable_definition (fstart, p, origin, 0);
766
767 assert (v != NULL);
768
769 if (vmod.export_v)
770 v->export = v_export;
771 if (vmod.private_v)
772 v->private_var = 1;
773
774 /* This line has been dealt with. */
775 continue;
776 }
777
778 /* If this line is completely empty, ignore it. */
779 if (*p == '\0')
780 continue;
781
782 p2 = end_of_token (p);
783 wlen = p2 - p;
784 NEXT_TOKEN (p2);
785
786 /* If we're in an ignored define, skip this line (but maybe get out). */
787 if (in_ignored_define)
788 {
789 /* See if this is an endef line (plus optional comment). */
790 if (word1eq ("endef") && STOP_SET (*p2, MAP_COMMENT|MAP_NUL))
791 in_ignored_define = 0;
792
793 continue;
794 }
795
796 /* Check for conditional state changes. */
797 {
798 int i = conditional_line (p, wlen, fstart);
799 if (i != -2)
800 {
801 if (i == -1)
802 O (fatal, fstart, _("invalid syntax in conditional"));
803
804 ignoring = i;
805 continue;
806 }
807 }
808
809 /* Nothing to see here... move along. */
810 if (ignoring)
811 continue;
812
813 /* Manage the "export" keyword used outside of variable assignment
814 as well as "unexport". */
815 if (word1eq ("export") || word1eq ("unexport"))
816 {
817 int exporting = *p == 'u' ? 0 : 1;
818
819 /* Export/unexport ends the previous rule. */
820 record_waiting_files ();
821
822 /* (un)export by itself causes everything to be (un)exported. */
823 if (*p2 == '\0')
824 export_all_variables = exporting;
825 else
826 {
827 unsigned int l;
828 const char *cp;
829 char *ap;
830
831 /* Expand the line so we can use indirect and constructed
832 variable names in an (un)export command. */
833 cp = ap = allocated_variable_expand (p2);
834
835 for (p = find_next_token (&cp, &l); p != 0;
836 p = find_next_token (&cp, &l))
837 {
838 struct variable *v = lookup_variable (p, l);
839 if (v == 0)
840 v = define_variable_global (p, l, "", o_file, 0, fstart);
841 v->export = exporting ? v_export : v_noexport;
842 }
843
844 free (ap);
845 }
846 continue;
847 }
848
849 /* Handle the special syntax for vpath. */
850 if (word1eq ("vpath"))
851 {
852 const char *cp;
853 char *vpat;
854 unsigned int l;
855
856 /* vpath ends the previous rule. */
857 record_waiting_files ();
858
859 cp = variable_expand (p2);
860 p = find_next_token (&cp, &l);
861 if (p != 0)
862 {
863 vpat = xstrndup (p, l);
864 p = find_next_token (&cp, &l);
865 /* No searchpath means remove all previous
866 selective VPATH's with the same pattern. */
867 }
868 else
869 /* No pattern means remove all previous selective VPATH's. */
870 vpat = 0;
871 construct_vpath_list (vpat, p);
872 free (vpat);
873
874 continue;
875 }
876
877 /* Handle include and variants. */
878 if (word1eq ("include") || word1eq ("-include") || word1eq ("sinclude"))
879 {
880 /* We have found an 'include' line specifying a nested
881 makefile to be read at this point. */
882 struct conditionals *save;
883 struct conditionals new_conditionals;
884 struct nameseq *files;
885 /* "-include" (vs "include") says no error if the file does not
886 exist. "sinclude" is an alias for this from SGI. */
887 int noerror = (p[0] != 'i');
888
889 /* Include ends the previous rule. */
890 record_waiting_files ();
891
892 p = allocated_variable_expand (p2);
893
894 /* If no filenames, it's a no-op. */
895 if (*p == '\0')
896 {
897 free (p);
898 continue;
899 }
900
901 /* Parse the list of file names. Don't expand archive references! */
902 p2 = p;
903 files = PARSE_FILE_SEQ (&p2, struct nameseq, MAP_NUL, NULL,
904 PARSEFS_NOAR);
905 free (p);
906
907 /* Save the state of conditionals and start
908 the included makefile with a clean slate. */
909 save = install_conditionals (&new_conditionals);
910
911 /* Record the rules that are waiting so they will determine
912 the default goal before those in the included makefile. */
913 record_waiting_files ();
914
915 /* Read each included makefile. */
916 while (files != 0)
917 {
918 struct nameseq *next = files->next;
919 int flags = (RM_INCLUDED | RM_NO_TILDE
920 | (noerror ? RM_DONTCARE : 0)
921 | (set_default ? 0 : RM_NO_DEFAULT_GOAL));
922
923 struct goaldep *d = eval_makefile (files->name, flags);
924
925 if (errno)
926 {
927 d->error = (unsigned short)errno;
928 d->floc = *fstart;
929 }
930
931 free_ns (files);
932 files = next;
933 }
934
935 /* Restore conditional state. */
936 restore_conditionals (save);
937
938 continue;
939 }
940
941 /* Handle the load operations. */
942 if (word1eq ("load") || word1eq ("-load"))
943 {
944 /* A 'load' line specifies a dynamic object to load. */
945 struct nameseq *files;
946 int noerror = (p[0] == '-');
947
948 /* Load ends the previous rule. */
949 record_waiting_files ();
950
951 p = allocated_variable_expand (p2);
952
953 /* If no filenames, it's a no-op. */
954 if (*p == '\0')
955 {
956 free (p);
957 continue;
958 }
959
960 /* Parse the list of file names.
961 Don't expand archive references or strip "./" */
962 p2 = p;
963 files = PARSE_FILE_SEQ (&p2, struct nameseq, MAP_NUL, NULL,
964 PARSEFS_NOAR);
965 free (p);
966
967 /* Load each file. */
968 while (files != 0)
969 {
970 struct nameseq *next = files->next;
971 const char *name = files->name;
972 struct goaldep *deps;
973 int r;
974
975 /* Load the file. 0 means failure. */
976 r = load_file (&ebuf->floc, &name, noerror);
977 if (! r && ! noerror)
978 OS (fatal, &ebuf->floc, _("%s: failed to load"), name);
979
980 free_ns (files);
981 files = next;
982
983 /* Return of -1 means a special load: don't rebuild it. */
984 if (r == -1)
985 continue;
986
987 /* It succeeded, so add it to the list "to be rebuilt". */
988 deps = alloc_goaldep ();
989 deps->next = read_makefiles;
990 read_makefiles = deps;
991 deps->file = lookup_file (name);
992 if (deps->file == 0)
993 deps->file = enter_file (name);
994 deps->file->loaded = 1;
995 }
996
997 continue;
998 }
999
1000 /* This line starts with a tab but was not caught above because there
1001 was no preceding target, and the line might have been usable as a
1002 variable definition. But now we know it is definitely lossage. */
1003 if (line[0] == cmd_prefix)
1004 O (fatal, fstart, _("recipe commences before first target"));
1005
1006 /* This line describes some target files. This is complicated by
1007 the existence of target-specific variables, because we can't
1008 expand the entire line until we know if we have one or not. So
1009 we expand the line word by word until we find the first ':',
1010 then check to see if it's a target-specific variable.
1011
1012 In this algorithm, 'lb_next' will point to the beginning of the
1013 unexpanded parts of the input buffer, while 'p2' points to the
1014 parts of the expanded buffer we haven't searched yet. */
1015
1016 {
1017 enum make_word_type wtype;
1018 char *cmdleft, *semip, *lb_next;
1019 unsigned int plen = 0;
1020 char *colonp;
1021 const char *end, *beg; /* Helpers for whitespace stripping. */
1022
1023 /* Record the previous rule. */
1024
1025 record_waiting_files ();
1026 tgts_started = fstart->lineno;
1027
1028 /* Search the line for an unquoted ; that is not after an
1029 unquoted #. */
1030 cmdleft = find_char_unquote (line, MAP_SEMI|MAP_COMMENT|MAP_VARIABLE);
1031 if (cmdleft != 0 && *cmdleft == '#')
1032 {
1033 /* We found a comment before a semicolon. */
1034 *cmdleft = '\0';
1035 cmdleft = 0;
1036 }
1037 else if (cmdleft != 0)
1038 /* Found one. Cut the line short there before expanding it. */
1039 *(cmdleft++) = '\0';
1040 semip = cmdleft;
1041
1042 collapse_continuations (line);
1043
1044 /* We can't expand the entire line, since if it's a per-target
1045 variable we don't want to expand it. So, walk from the
1046 beginning, expanding as we go, and looking for "interesting"
1047 chars. The first word is always expandable. */
1048 wtype = get_next_mword (line, NULL, &lb_next, &wlen);
1049 switch (wtype)
1050 {
1051 case w_eol:
1052 if (cmdleft != 0)
1053 O (fatal, fstart, _("missing rule before recipe"));
1054 /* This line contained something but turned out to be nothing
1055 but whitespace (a comment?). */
1056 continue;
1057
1058 case w_colon:
1059 case w_dcolon:
1060 /* We accept and ignore rules without targets for
1061 compatibility with SunOS 4 make. */
1062 no_targets = 1;
1063 continue;
1064
1065 default:
1066 break;
1067 }
1068
1069 p2 = variable_expand_string (NULL, lb_next, wlen);
1070
1071 while (1)
1072 {
1073 lb_next += wlen;
1074 if (cmdleft == 0)
1075 {
1076 /* Look for a semicolon in the expanded line. */
1077 cmdleft = find_char_unquote (p2, MAP_SEMI);
1078
1079 if (cmdleft != 0)
1080 {
1081 unsigned long p2_off = p2 - variable_buffer;
1082 unsigned long cmd_off = cmdleft - variable_buffer;
1083 char *pend = p2 + strlen (p2);
1084
1085 /* Append any remnants of lb, then cut the line short
1086 at the semicolon. */
1087 *cmdleft = '\0';
1088
1089 /* One school of thought says that you shouldn't expand
1090 here, but merely copy, since now you're beyond a ";"
1091 and into a command script. However, the old parser
1092 expanded the whole line, so we continue that for
1093 backwards-compatibility. Also, it wouldn't be
1094 entirely consistent, since we do an unconditional
1095 expand below once we know we don't have a
1096 target-specific variable. */
1097 (void)variable_expand_string (pend, lb_next, (long)-1);
1098 lb_next += strlen (lb_next);
1099 p2 = variable_buffer + p2_off;
1100 cmdleft = variable_buffer + cmd_off + 1;
1101 }
1102 }
1103
1104 colonp = find_char_unquote (p2, MAP_COLON);
1105#ifdef HAVE_DOS_PATHS
1106 /* The drive spec brain-damage strikes again... */
1107 /* Note that the only separators of targets in this context
1108 are whitespace and a left paren. If others are possible,
1109 they should be added to the string in the call to index. */
1110 while (colonp && (colonp[1] == '/' || colonp[1] == '\\') &&
1111 colonp > p2 && isalpha ((unsigned char)colonp[-1]) &&
1112 (colonp == p2 + 1 || strchr (" \t(", colonp[-2]) != 0))
1113 colonp = find_char_unquote (colonp + 1, MAP_COLON);
1114#endif
1115 if (colonp != 0)
1116 break;
1117
1118 wtype = get_next_mword (lb_next, NULL, &lb_next, &wlen);
1119 if (wtype == w_eol)
1120 break;
1121
1122 p2 += strlen (p2);
1123 *(p2++) = ' ';
1124 p2 = variable_expand_string (p2, lb_next, wlen);
1125 /* We don't need to worry about cmdleft here, because if it was
1126 found in the variable_buffer the entire buffer has already
1127 been expanded... we'll never get here. */
1128 }
1129
1130 p2 = next_token (variable_buffer);
1131
1132 /* If the word we're looking at is EOL, see if there's _anything_
1133 on the line. If not, a variable expanded to nothing, so ignore
1134 it. If so, we can't parse this line so punt. */
1135 if (wtype == w_eol)
1136 {
1137 if (*p2 == '\0')
1138 continue;
1139
1140 /* There's no need to be ivory-tower about this: check for
1141 one of the most common bugs found in makefiles... */
1142 if (cmd_prefix == '\t' && strneq (line, " ", 8))
1143 O (fatal, fstart, _("missing separator (did you mean TAB instead of 8 spaces?)"));
1144 else
1145 O (fatal, fstart, _("missing separator"));
1146 }
1147
1148 /* Make the colon the end-of-string so we know where to stop
1149 looking for targets. Start there again once we're done. */
1150 *colonp = '\0';
1151 filenames = PARSE_SIMPLE_SEQ (&p2, struct nameseq);
1152 *colonp = ':';
1153 p2 = colonp;
1154
1155 if (!filenames)
1156 {
1157 /* We accept and ignore rules without targets for
1158 compatibility with SunOS 4 make. */
1159 no_targets = 1;
1160 continue;
1161 }
1162 /* This should never be possible; we handled it above. */
1163 assert (*p2 != '\0');
1164 ++p2;
1165
1166 /* Is this a one-colon or two-colon entry? */
1167 two_colon = *p2 == ':';
1168 if (two_colon)
1169 p2++;
1170
1171 /* Test to see if it's a target-specific variable. Copy the rest
1172 of the buffer over, possibly temporarily (we'll expand it later
1173 if it's not a target-specific variable). PLEN saves the length
1174 of the unparsed section of p2, for later. */
1175 if (*lb_next != '\0')
1176 {
1177 unsigned int l = p2 - variable_buffer;
1178 plen = strlen (p2);
1179 variable_buffer_output (p2+plen, lb_next, strlen (lb_next)+1);
1180 p2 = variable_buffer + l;
1181 }
1182
1183 p2 = parse_var_assignment (p2, &vmod);
1184 if (vmod.assign_v)
1185 {
1186 /* If there was a semicolon found, add it back, plus anything
1187 after it. */
1188 if (semip)
1189 {
1190 unsigned int l = p2 - variable_buffer;
1191 *(--semip) = ';';
1192 collapse_continuations (semip);
1193 variable_buffer_output (p2 + strlen (p2),
1194 semip, strlen (semip)+1);
1195 p2 = variable_buffer + l;
1196 }
1197 record_target_var (filenames, p2,
1198 vmod.override_v ? o_override : o_file,
1199 &vmod, fstart);
1200 filenames = 0;
1201 continue;
1202 }
1203
1204 /* This is a normal target, _not_ a target-specific variable.
1205 Unquote any = in the dependency list. */
1206 find_char_unquote (lb_next, MAP_EQUALS);
1207
1208 /* Remember the command prefix for this target. */
1209 prefix = cmd_prefix;
1210
1211 /* We have some targets, so don't ignore the following commands. */
1212 no_targets = 0;
1213
1214 /* Expand the dependencies, etc. */
1215 if (*lb_next != '\0')
1216 {
1217 unsigned int l = p2 - variable_buffer;
1218 (void) variable_expand_string (p2 + plen, lb_next, (long)-1);
1219 p2 = variable_buffer + l;
1220
1221 /* Look for a semicolon in the expanded line. */
1222 if (cmdleft == 0)
1223 {
1224 cmdleft = find_char_unquote (p2, MAP_SEMI);
1225 if (cmdleft != 0)
1226 *(cmdleft++) = '\0';
1227 }
1228 }
1229
1230 /* Is this a static pattern rule: 'target: %targ: %dep; ...'? */
1231 p = strchr (p2, ':');
1232 while (p != 0 && p[-1] == '\\')
1233 {
1234 char *q = &p[-1];
1235 int backslash = 0;
1236 while (*q-- == '\\')
1237 backslash = !backslash;
1238 if (backslash)
1239 p = strchr (p + 1, ':');
1240 else
1241 break;
1242 }
1243#ifdef HAVE_DOS_PATHS
1244 {
1245 int check_again;
1246 do {
1247 check_again = 0;
1248 /* For DOS-style paths, skip a "C:\..." or a "C:/..." */
1249 if (p != 0 && (p[1] == '\\' || p[1] == '/') &&
1250 isalpha ((unsigned char)p[-1]) &&
1251 (p == p2 + 1 || strchr (" \t:(", p[-2]) != 0)) {
1252 p = strchr (p + 1, ':');
1253 check_again = 1;
1254 }
1255 } while (check_again);
1256 }
1257#endif
1258 if (p != 0)
1259 {
1260 struct nameseq *target;
1261 target = PARSE_FILE_SEQ (&p2, struct nameseq, MAP_COLON, NULL,
1262 PARSEFS_NOGLOB);
1263 ++p2;
1264 if (target == 0)
1265 O (fatal, fstart, _("missing target pattern"));
1266 else if (target->next != 0)
1267 O (fatal, fstart, _("multiple target patterns"));
1268 pattern_percent = find_percent_cached (&target->name);
1269 pattern = target->name;
1270 if (pattern_percent == 0)
1271 O (fatal, fstart, _("target pattern contains no '%%'"));
1272 free_ns (target);
1273 }
1274 else
1275 pattern = 0;
1276
1277 /* Strip leading and trailing whitespaces. */
1278 beg = p2;
1279 end = beg + strlen (beg) - 1;
1280 strip_whitespace (&beg, &end);
1281
1282 /* Put all the prerequisites here; they'll be parsed later. */
1283 if (beg <= end && *beg != '\0')
1284 depstr = xstrndup (beg, end - beg + 1);
1285 else
1286 depstr = 0;
1287
1288 commands_idx = 0;
1289 if (cmdleft != 0)
1290 {
1291 /* Semicolon means rest of line is a command. */
1292 unsigned int l = strlen (cmdleft);
1293
1294 cmds_started = fstart->lineno;
1295
1296 /* Add this command line to the buffer. */
1297 if (l + 2 > commands_len)
1298 {
1299 commands_len = (l + 2) * 2;
1300 commands = xrealloc (commands, commands_len);
1301 }
1302 memcpy (commands, cmdleft, l);
1303 commands_idx += l;
1304 commands[commands_idx++] = '\n';
1305 }
1306
1307 /* Determine if this target should be made default. We used to do
1308 this in record_files() but because of the delayed target recording
1309 and because preprocessor directives are legal in target's commands
1310 it is too late. Consider this fragment for example:
1311
1312 foo:
1313
1314 ifeq ($(.DEFAULT_GOAL),foo)
1315 ...
1316 endif
1317
1318 Because the target is not recorded until after ifeq directive is
1319 evaluated the .DEFAULT_GOAL does not contain foo yet as one
1320 would expect. Because of this we have to move the logic here. */
1321
1322 if (set_default && default_goal_var->value[0] == '\0')
1323 {
1324 struct dep *d;
1325 struct nameseq *t = filenames;
1326
1327 for (; t != 0; t = t->next)
1328 {
1329 int reject = 0;
1330 const char *name = t->name;
1331
1332 /* We have nothing to do if this is an implicit rule. */
1333 if (strchr (name, '%') != 0)
1334 break;
1335
1336 /* See if this target's name does not start with a '.',
1337 unless it contains a slash. */
1338 if (*name == '.' && strchr (name, '/') == 0
1339#ifdef HAVE_DOS_PATHS
1340 && strchr (name, '\\') == 0
1341#endif
1342 )
1343 continue;
1344
1345
1346 /* If this file is a suffix, don't let it be
1347 the default goal file. */
1348 for (d = suffix_file->deps; d != 0; d = d->next)
1349 {
1350 register struct dep *d2;
1351 if (*dep_name (d) != '.' && streq (name, dep_name (d)))
1352 {
1353 reject = 1;
1354 break;
1355 }
1356 for (d2 = suffix_file->deps; d2 != 0; d2 = d2->next)
1357 {
1358 unsigned int l = strlen (dep_name (d2));
1359 if (!strneq (name, dep_name (d2), l))
1360 continue;
1361 if (streq (name + l, dep_name (d)))
1362 {
1363 reject = 1;
1364 break;
1365 }
1366 }
1367
1368 if (reject)
1369 break;
1370 }
1371
1372 if (!reject)
1373 {
1374 define_variable_global (".DEFAULT_GOAL", 13, t->name,
1375 o_file, 0, NILF);
1376 break;
1377 }
1378 }
1379 }
1380
1381 continue;
1382 }
1383
1384 /* We get here except in the case that we just read a rule line.
1385 Record now the last rule we read, so following spurious
1386 commands are properly diagnosed. */
1387 record_waiting_files ();
1388 }
1389
1390#undef word1eq
1391
1392 if (conditionals->if_cmds)
1393 O (fatal, fstart, _("missing 'endif'"));
1394
1395 /* At eof, record the last rule. */
1396 record_waiting_files ();
1397
1398 free (collapsed);
1399 free (commands);
1400}

1403/* Remove comments from LINE.
1404 This is done by copying the text at LINE onto itself. */
1405
1406static void
1407remove_comments (char *line, char **target_description,
1408 char **prev_target_description, unsigned long lineno)
1409{
1410 char *comment;
1411
1412 comment = find_char_unquote (line, MAP_COMMENT);
1413
1414 if (comment != 0) {
1415 if (show_tasks_flag || show_targets_flag) {
1416 if (0 == strncmp(comment, "#: ", 3) && target_description) {
1417 *prev_target_description = *target_description;
1418 prev_target_description_lineno = target_description_lineno;
1419 *target_description = xstrdup(&comment[3]);
1420 target_description_lineno = lineno;
1421 }
1422 }
1423 /* Cut off the line at the #. */
1424 *comment = '\0';
1425 }

1428/* Execute a 'undefine' directive.
1429 The undefine line has already been read, and NAME is the name of
1430 the variable to be undefined. */
1431
1432static void
1433do_undefine (char *name, enum variable_origin origin, struct ebuffer *ebuf)
1434{
1435 char *p, *var;
1436
1437 /* Expand the variable name and find the beginning (NAME) and end. */
1438 var = allocated_variable_expand (name);
1439 name = next_token (var);
1440 if (*name == '\0')
1441 O (fatal, &ebuf->floc, _("empty variable name"));
1442 p = name + strlen (name) - 1;
1443 while (p > name && ISBLANK (*p))
1444 --p;
1445 p[1] = '\0';
1446
1447 undefine_variable_global (name, p - name + 1, origin);
1448 free (var);
1449}

1451/* Execute a 'define' directive.
1452 The first line has already been read, and NAME is the name of
1453 the variable to be defined. The following lines remain to be read. */
1454
1455static struct variable *
1456do_define (char *name, enum variable_origin origin, struct ebuffer *ebuf)
1457{
1458 struct variable *v;
1459 struct variable var;
1460 gmk_floc defstart;
1461 int nlevels = 1;
1462 unsigned int length = 100;
1463 char *definition = xmalloc (length);
1464 unsigned int idx = 0;
1465 char *p, *n;
1466
1467 defstart = ebuf->floc;
1468
1469 p = parse_variable_definition (name, &var);
1470 if (p == NULL)
1471 /* No assignment token, so assume recursive. */
1472 var.flavor = f_recursive;
1473 else
1474 {
1475 if (var.value[0] != '\0')
1476 O (error, &defstart, _("extraneous text after 'define' directive"));
1477
1478 /* Chop the string before the assignment token to get the name. */
1479 var.name[var.length] = '\0';
1480 }
1481
1482 /* Expand the variable name and find the beginning (NAME) and end. */
1483 n = allocated_variable_expand (name);
1484 name = next_token (n);
1485 if (name[0] == '\0')
1486 O (fatal, &defstart, _("empty variable name"));
1487 p = name + strlen (name) - 1;
1488 while (p > name && ISBLANK (*p))
1489 --p;
1490 p[1] = '\0';
1491
1492 /* Now read the value of the variable. */
1493 while (1)
1494 {
1495 unsigned int len;
1496 char *line;
1497 long nlines = readline (ebuf);
1498
1499 /* If there is nothing left to be eval'd, there's no 'endef'!! */
1500 if (nlines < 0)
1501 O (fatal, &defstart, _("missing 'endef', unterminated 'define'"));
1502
1503 ebuf->floc.lineno += nlines;
1504 line = ebuf->buffer;
1505
1506 collapse_continuations (line);
1507
1508 /* If the line doesn't begin with a tab, test to see if it introduces
1509 another define, or ends one. Stop if we find an 'endef' */
1510 if (line[0] != cmd_prefix)
1511 {
1512 p = next_token (line);
1513 len = strlen (p);
1514
1515 /* If this is another 'define', increment the level count. */
1516 if ((len == 6 || (len > 6 && ISBLANK (p[6])))
1517 && strneq (p, "define", 6))
1518 ++nlevels;
1519
1520 /* If this is an 'endef', decrement the count. If it's now 0,
1521 we've found the last one. */
1522 else if ((len == 5 || (len > 5 && ISBLANK (p[5])))
1523 && strneq (p, "endef", 5))
1524 {
1525 p += 5;
1526 remove_comments (p, NULL, NULL, 0);
1527 if (*(next_token (p)) != '\0')
1528 O (error, &ebuf->floc,
1529 _("extraneous text after 'endef' directive"));
1530
1531 if (--nlevels == 0)
1532 break;
1533 }
1534 }
1535
1536 /* Add this line to the variable definition. */
1537 len = strlen (line);
1538 if (idx + len + 1 > length)
1539 {
1540 length = (idx + len) * 2;
1541 definition = xrealloc (definition, length + 1);
1542 }
1543
1544 memcpy (&definition[idx], line, len);
1545 idx += len;
1546 /* Separate lines with a newline. */
1547 definition[idx++] = '\n';
1548 }
1549
1550 /* We've got what we need; define the variable. */
1551 if (idx == 0)
1552 definition[0] = '\0';
1553 else
1554 definition[idx - 1] = '\0';
1555
1556 v = do_variable_definition (&defstart, name,
1557 definition, origin, var.flavor, 0);
1558 free (definition);
1559 free (n);
1560 return (v);
1561}

1563/* Interpret conditional commands "ifdef", "ifndef", "ifeq",
1564 "ifneq", "else" and "endif".
1565 LINE is the input line, with the command as its first word.
1566
1567 FILENAME and LINENO are the filename and line number in the
1568 current makefile. They are used for error messages.
1569
1570 Value is -2 if the line is not a conditional at all,
1571 -1 if the line is an invalid conditional,
1572 0 if following text should be interpreted,
1573 1 if following text should be ignored. */
1574
1575static int
1576conditional_line (char *line, int len, const gmk_floc *flocp)
1577{
1578 const char *cmdname;
1579 enum { c_ifdef, c_ifndef, c_ifeq, c_ifneq, c_else, c_endif } cmdtype;
1580 unsigned int i;
1581 unsigned int o;
1582
1583 /* Compare a word, both length and contents. */
1584#define word1eq(s) (len == CSTRLEN (s) && strneq (s, line, CSTRLEN (s)))
1585#define chkword(s, t) if (word1eq (s)) { cmdtype = (t); cmdname = (s); }
1586
1587 /* Make sure this line is a conditional. */
1588 chkword ("ifdef", c_ifdef)
1589 else chkword ("ifndef", c_ifndef)
1590 else chkword ("ifeq", c_ifeq)
1591 else chkword ("ifneq", c_ifneq)
1592 else chkword ("else", c_else)
1593 else chkword ("endif", c_endif)
1594 else
1595 return -2;
1596
1597 /* Found one: skip past it and any whitespace after it. */
1598 line += len;
1599 NEXT_TOKEN (line);
1600
1601#define EXTRATEXT() OS (error, flocp, _("extraneous text after '%s' directive"), cmdname)
1602#define EXTRACMD() OS (fatal, flocp, _("extraneous '%s'"), cmdname)
1603
1604 /* An 'endif' cannot contain extra text, and reduces the if-depth by 1 */
1605 if (cmdtype == c_endif)
1606 {
1607 if (*line != '\0')
1608 EXTRATEXT ();
1609
1610 if (!conditionals->if_cmds)
1611 EXTRACMD ();
1612
1613 --conditionals->if_cmds;
1614
1615 goto DONE;
1616 }
1617
1618 /* An 'else' statement can either be simple, or it can have another
1619 conditional after it. */
1620 if (cmdtype == c_else)
1621 {
1622 const char *p;
1623
1624 if (!conditionals->if_cmds)
1625 EXTRACMD ();
1626
1627 o = conditionals->if_cmds - 1;
1628
1629 if (conditionals->seen_else[o])
1630 O (fatal, flocp, _("only one 'else' per conditional"));
1631
1632 /* Change the state of ignorance. */
1633 switch (conditionals->ignoring[o])
1634 {
1635 case 0:
1636 /* We've just been interpreting. Never do it again. */
1637 conditionals->ignoring[o] = 2;
1638 break;
1639 case 1:
1640 /* We've never interpreted yet. Maybe this time! */
1641 conditionals->ignoring[o] = 0;
1642 break;
1643 }
1644
1645 /* It's a simple 'else'. */
1646 if (*line == '\0')
1647 {
1648 conditionals->seen_else[o] = 1;
1649 goto DONE;
1650 }
1651
1652 /* The 'else' has extra text. That text must be another conditional
1653 and cannot be an 'else' or 'endif'. */
1654
1655 /* Find the length of the next word. */
1656 for (p = line+1; ! STOP_SET (*p, MAP_SPACE|MAP_NUL); ++p)
1657 ;
1658 len = p - line;
1659
1660 /* If it's 'else' or 'endif' or an illegal conditional, fail. */
1661 if (word1eq ("else") || word1eq ("endif")
1662 || conditional_line (line, len, flocp) < 0)
1663 EXTRATEXT ();
1664 else
1665 {
1666 /* conditional_line() created a new level of conditional.
1667 Raise it back to this level. */
1668 if (conditionals->ignoring[o] < 2)
1669 conditionals->ignoring[o] = conditionals->ignoring[o+1];
1670 --conditionals->if_cmds;
1671 }
1672
1673 goto DONE;
1674 }
1675
1676 if (conditionals->allocated == 0)
1677 {
1678 conditionals->allocated = 5;
1679 conditionals->ignoring = xmalloc (conditionals->allocated);
1680 conditionals->seen_else = xmalloc (conditionals->allocated);
1681 }
1682
1683 o = conditionals->if_cmds++;
1684 if (conditionals->if_cmds > conditionals->allocated)
1685 {
1686 conditionals->allocated += 5;
1687 conditionals->ignoring = xrealloc (conditionals->ignoring,
1688 conditionals->allocated);
1689 conditionals->seen_else = xrealloc (conditionals->seen_else,
1690 conditionals->allocated);
1691 }
1692
1693 /* Record that we have seen an 'if...' but no 'else' so far. */
1694 conditionals->seen_else[o] = 0;
1695
1696 /* Search through the stack to see if we're already ignoring. */
1697 for (i = 0; i < o; ++i)
1698 if (conditionals->ignoring[i])
1699 {
1700 /* We are already ignoring, so just push a level to match the next
1701 "else" or "endif", and keep ignoring. We don't want to expand
1702 variables in the condition. */
1703 conditionals->ignoring[o] = 1;
1704 return 1;
1705 }
1706
1707 if (cmdtype == c_ifdef || cmdtype == c_ifndef)
1708 {
1709 char *var;
1710 struct variable *v;
1711 char *p;
1712
1713 /* Expand the thing we're looking up, so we can use indirect and
1714 constructed variable names. */
1715 var = allocated_variable_expand (line);
1716
1717 /* Make sure there's only one variable name to test. */
1718 p = end_of_token (var);
1719 i = p - var;
1720 NEXT_TOKEN (p);
1721 if (*p != '\0')
1722 return -1;
1723
1724 var[i] = '\0';
1725 v = lookup_variable (var, i);
1726
1727 conditionals->ignoring[o] =
1728 ((v != 0 && *v->value != '\0') == (cmdtype == c_ifndef));
1729
1730 free (var);
1731 }
1732 else
1733 {
1734 /* "ifeq" or "ifneq". */
1735 char *s1, *s2;
1736 unsigned int l;
1737 char termin = *line == '(' ? ',' : *line;
1738
1739 if (termin != ',' && termin != '"' && termin != '\'')
1740 return -1;
1741
1742 s1 = ++line;
1743 /* Find the end of the first string. */
1744 if (termin == ',')
1745 {
1746 int count = 0;
1747 for (; *line != '\0'; ++line)
1748 if (*line == '(')
1749 ++count;
1750 else if (*line == ')')
1751 --count;
1752 else if (*line == ',' && count <= 0)
1753 break;
1754 }
1755 else
1756 while (*line != '\0' && *line != termin)
1757 ++line;
1758
1759 if (*line == '\0')
1760 return -1;
1761
1762 if (termin == ',')
1763 {
1764 /* Strip blanks after the first string. */
1765 char *p = line++;
1766 while (ISBLANK (p[-1]))
1767 --p;
1768 *p = '\0';
1769 }
1770 else
1771 *line++ = '\0';
1772
1773 s2 = variable_expand (s1);
1774 /* We must allocate a new copy of the expanded string because
1775 variable_expand re-uses the same buffer. */
1776 l = strlen (s2);
1777 s1 = alloca (l + 1);
1778 memcpy (s1, s2, l + 1);
1779
1780 if (termin != ',')
1781 /* Find the start of the second string. */
1782 NEXT_TOKEN (line);
1783
1784 termin = termin == ',' ? ')' : *line;
1785 if (termin != ')' && termin != '"' && termin != '\'')
1786 return -1;
1787
1788 /* Find the end of the second string. */
1789 if (termin == ')')
1790 {
1791 int count = 0;
1792 s2 = next_token (line);
1793 for (line = s2; *line != '\0'; ++line)
1794 {
1795 if (*line == '(')
1796 ++count;
1797 else if (*line == ')')
1798 {
1799 if (count <= 0)
1800 break;
1801 else
1802 --count;
1803 }
1804 }
1805 }
1806 else
1807 {
1808 ++line;
1809 s2 = line;
1810 while (*line != '\0' && *line != termin)
1811 ++line;
1812 }
1813
1814 if (*line == '\0')
1815 return -1;
1816
1817 *(line++) = '\0';
1818 NEXT_TOKEN (line);
1819 if (*line != '\0')
1820 EXTRATEXT ();
1821
1822 s2 = variable_expand (s2);
1823 conditionals->ignoring[o] = (streq (s1, s2) == (cmdtype == c_ifneq));
1824 }
1825
1826 DONE:
1827 /* Search through the stack to see if we're ignoring. */
1828 for (i = 0; i < conditionals->if_cmds; ++i)
1829 if (conditionals->ignoring[i])
1830 return 1;
1831 return 0;
1832}

1835/* Record target-specific variable values for files FILENAMES.
1836 TWO_COLON is nonzero if a double colon was used.
1837
1838 The links of FILENAMES are freed, and so are any names in it
1839 that are not incorporated into other data structures.
1840
1841 If the target is a pattern, add the variable to the pattern-specific
1842 variable value list. */
1843
1844static void
1845record_target_var (struct nameseq *filenames, char *defn,
1846 enum variable_origin origin, struct vmodifiers *vmod,

1848{

1864 nextf = filenames->next;

1923 && (gv->origin == o_env_override || gv->origin == o_command))

1926 v->value = xstrdup (gv->value);
1927 v->origin = gv->origin;
1928 v->recursive = gv->recursive;
1929 v->append = 0;
1930 }
1931 }
1932 }

1935/* Record a description line for files FILENAMES,
1936 with dependencies DEPS, commands to execute described
1937 by COMMANDS and COMMANDS_IDX, coming from FILENAME:COMMANDS_STARTED.
1938 TWO_COLON is nonzero if a double colon was used.
1939 If not nil, PATTERN is the '%' pattern to make this
1940 a static pattern rule, and PATTERN_PERCENT is a pointer
1941 to the '%' within it.
1942
1943 The links of FILENAMES are freed, and so are any names in it
1944 that are not incorporated into other data structures. */
1945
1946static void
1947record_files (struct nameseq *filenames, const char *pattern,
1948 const char *pattern_percent, char *depstr,
1949 unsigned int cmds_started, char *commands,
1950 unsigned int commands_idx, int two_colon,
1951 char *target_description,
1952 char prefix, const gmk_floc *flocp)
1953{
1954 struct commands *cmds;
1955 struct dep *deps;
1956 const char *implicit_percent;
1957 const char *name;
1958
1959 /* If we've already snapped deps, that means we're in an eval being
1960 resolved after the makefiles have been read in. We can't add more rules
1961 at this time, since they won't get snapped and we'll get core dumps.
1962 See Savannah bug # 12124. */
1963 if (snapped_deps)
1964 O (fatal, flocp, _("prerequisites cannot be defined in recipes"));
1965
1966 /* Determine if this is a pattern rule or not. */
1967 name = filenames->name;
1968 implicit_percent = find_percent_cached (&name);
1969
1970 /* If there's a recipe, set up a struct for it. */
1971 if (commands_idx > 0)
1972 {
1973 cmds = xmalloc (sizeof (struct commands));
1974 cmds->fileinfo.filenm = flocp->filenm;
1975 cmds->fileinfo.lineno = cmds_started;
1976 cmds->fileinfo.offset = 0;
1977 cmds->commands = xstrndup (commands, commands_idx);
1978 cmds->command_lines = 0;
1979 cmds->recipe_prefix = prefix;
1980 }
1981 else
1982 cmds = 0;
1983
1984 /* If there's a prereq string then parse it--unless it's eligible for 2nd
1985 expansion: if so, snap_deps() will do it. */
1986 if (depstr == 0)
1987 deps = 0;
1988 else
1989 {
1990 depstr = unescape_char (depstr, ':');
1991 if (second_expansion && strchr (depstr, '$'))
1992 {
1993 deps = alloc_dep ();
1994 deps->name = depstr;
1995 deps->need_2nd_expansion = 1;
1996 deps->staticpattern = pattern != 0;
1997 }
1998 else
1999 {
2000 deps = split_prereqs (depstr);
2001 free (depstr);
2002
2003 /* We'll enter static pattern prereqs later when we have the stem.
2004 We don't want to enter pattern rules at all so that we don't
2005 think that they ought to exist (make manual "Implicit Rule Search
2006 Algorithm", item 5c). */
2007 if (! pattern && ! implicit_percent)
2008 deps = enter_prereqs (deps, NULL);
2009 }
2010 }
2011

2013 can test the first one to see if we're working with an implicit rule; if
2014 so we handle it specially. */
2015
2016 if (implicit_percent)
2017 {
2018 struct nameseq *nextf;
2019 const char **targets, **target_pats;
2020 unsigned int c;
2021
2022 if (pattern != 0)
2023 O (fatal, flocp, _("mixed implicit and static pattern rules"));
2024
2025 /* Count the targets to create an array of target names.
2026 We already have the first one. */
2027 nextf = filenames->next;
2028 free_ns (filenames);
2029 filenames = nextf;
2030
2031 for (c = 1; nextf; ++c, nextf = nextf->next)
2032 ;
2033 targets = xmalloc (c * sizeof (const char *));
2034 target_pats = xmalloc (c * sizeof (const char *));
2035
2036 targets[0] = name;
2037 target_pats[0] = implicit_percent;
2038
2039 c = 1;
2040 while (filenames)
2041 {
2042 name = filenames->name;
2043 implicit_percent = find_percent_cached (&name);
2044
2045 if (implicit_percent == 0)
2046 O (fatal, flocp, _("mixed implicit and normal rules"));
2047
2048 targets[c] = name;
2049 target_pats[c] = implicit_percent;
2050 ++c;
2051
2052 nextf = filenames->next;
2053 free_ns (filenames);
2054 filenames = nextf;
2055 }
2056
2057 create_pattern_rule (targets, target_pats, c, two_colon, deps, cmds, 1);
2058
2059 return;
2060 }
2061
2062
2063 /* Walk through each target and create it in the database.
2064 We already set up the first target, above. */
2065 while (1)
2066 {
2067 struct nameseq *nextf = filenames->next;
2068 struct file *f;
2069 struct dep *this = 0;
2070
2071 free_ns (filenames);
2072
2073 /* Check for special targets. Do it here instead of, say, snap_deps()
2074 so that we can immediately use the value. */
2075 if (streq (name, ".POSIX"))
2076 {
2077 posix_pedantic = 1;
2078 define_variable_cname (".SHELLFLAGS", "-ec", o_default, 0);
2079 /* These default values are based on IEEE Std 1003.1-2008. */
2080 define_variable_cname ("ARFLAGS", "-rv", o_default, 0);
2081 define_variable_cname ("CC", "c99", o_default, 0);
2082 define_variable_cname ("CFLAGS", "-O", o_default, 0);
2083 define_variable_cname ("FC", "fort77", o_default, 0);
2084 define_variable_cname ("FFLAGS", "-O 1", o_default, 0);
2085 define_variable_cname ("SCCSGETFLAGS", "-s", o_default, 0);
2086 }
2087 else if (streq (name, ".SECONDEXPANSION"))
2088 second_expansion = 1;
2089#if !defined (__MSDOS__) && !defined (__EMX__)
2090 else if (streq (name, ".ONESHELL"))
2091 one_shell = 1;
2092#endif
2093
2094 /* If this is a static pattern rule:
2095 'targets: target%pattern: prereq%pattern; recipe',
2096 make sure the pattern matches this target name. */
2097 if (pattern && !pattern_matches (pattern, pattern_percent, name))
2098 OS (error, flocp,
2099 _("target '%s' doesn't match the target pattern"), name);
2100 else if (deps)
2101 /* If there are multiple targets, copy the chain DEPS for all but the
2102 last one. It is not safe for the same deps to go in more than one
2103 place in the database. */
2104 this = nextf != 0 ? copy_dep_chain (deps) : deps;
2105
2106 /* Find or create an entry in the file database for this target. */
2107 if (!two_colon)
2108 {
2109 /* Single-colon. Combine this rule with the file's existing record,
2110 if any. */
2111 f = enter_file (strcache_add (name));
2112 f->description = target_description;
2113 f->floc.filenm = flocp->filenm;
2114 f->floc.lineno = flocp->lineno;
2115
2116 if (f->double_colon)
2117 OS (fatal, flocp,
2118 _("target file '%s' has both : and :: entries"), f->name);
2119

2121 more than once. Just give a warning since this is harmless. */
2122 if (cmds != 0 && cmds == f->cmds)
2123 OS (error, flocp,
2124 _("target '%s' given more than once in the same rule"),
2125 f->name);

2127 /* Check for two single-colon entries both with commands.
2128 Check is_target so that we don't lose on files such as .c.o
2129 whose commands were preinitialized. */
2130 else if (cmds != 0 && f->cmds != 0 && f->is_target)
2131 {
2132 size_t l = strlen (f->name);
2133 error (&cmds->fileinfo, l,
2134 _("warning: overriding recipe for target '%s'"),
2135 f->name);
2136 error (&f->cmds->fileinfo, l,
2137 _("warning: ignoring old recipe for target '%s'"),
2138 f->name);
2139 }

2142 if (f == default_file && this == 0 && cmds == 0)
2143 f->cmds = 0;
2144 if (cmds != 0)
2145 f->cmds = cmds;
2146
2147 /* Defining .SUFFIXES with no dependencies clears out the list of
2148 suffixes. */
2149 if (f == suffix_file && this == 0)
2150 {
2151 free_dep_chain (f->deps);
2152 f->deps = 0;
2153 }
2154 }
2155 else
2156 {
2157 /* Double-colon. Make a new record even if there already is one. */
2158 f = lookup_file (name);
2159

2161 on default suffix rules or makefiles. */

2163 OS (fatal, flocp,
2164 _("target file '%s' has both : and :: entries"), f->name);
2165
2166 f = enter_file (strcache_add (name));
2167 f->description = target_description;
2168 f->floc.filenm = flocp->filenm;
2169 f->floc.lineno = flocp->lineno;
2170 /* If there was an existing entry and it was a double-colon entry,
2171 enter_file will have returned a new one, making it the prev
2172 pointer of the old one, and setting its double_colon pointer to
2173 the first one. */
2174 if (f->double_colon == 0)
2175 /* This is the first entry for this name, so we must set its
2176 double_colon pointer to itself. */
2177 f->double_colon = f;
2178
2179 f->cmds = cmds;
2180 }
2181
2182 f->is_target = 1;
2183
2184 /* If this is a static pattern rule, set the stem to the part of its
2185 name that matched the '%' in the pattern, so you can use $* in the
2186 commands. If we didn't do it before, enter the prereqs now. */
2187 if (pattern)
2188 {
2189 static const char *percent = "%";
2190 char *buffer = variable_expand ("");
2191 char *o = patsubst_expand_pat (buffer, name, pattern, percent,
2192 pattern_percent+1, percent+1);
2193 f->stem = strcache_add_len (buffer, o - buffer);
2194 if (this)
2195 {
2196 if (! this->need_2nd_expansion)
2197 this = enter_prereqs (this, f->stem);
2198 else
2199 this->stem = f->stem;
2200 }
2201 }
2202
2203 /* Add the dependencies to this file entry. */
2204 if (this != 0)
2205 {
2206 /* Add the file's old deps and the new ones in THIS together. */
2207 if (f->deps == 0)
2208 f->deps = this;
2209 else if (cmds != 0)
2210 {
2211 struct dep *d = this;
2212
2213 /* If this rule has commands, put these deps first. */
2214 while (d->next != 0)
2215 d = d->next;
2216
2217 d->next = f->deps;
2218 f->deps = this;
2219 }
2220 else
2221 {
2222 struct dep *d = f->deps;
2223
2224 /* A rule without commands: put its prereqs at the end. */
2225 while (d->next != 0)
2226 d = d->next;
2227
2228 d->next = this;
2229 }
2230 }
2231
2232 name = f->name;
2233
2234 /* All done! Set up for the next one. */
2235 if (nextf == 0)
2236 break;
2237
2238 filenames = nextf;
2239
2240 /* Reduce escaped percents. If there are any unescaped it's an error */

2248/* Search STRING for an unquoted STOPCHAR or blank (if BLANK is nonzero).
2249 Backslashes quote STOPCHAR, blanks if BLANK is nonzero, and backslash.
2250 Quoting backslashes are removed from STRING by compacting it into
2251 itself. Returns a pointer to the first unquoted STOPCHAR if there is
2252 one, or nil if there are none. STOPCHARs inside variable references are
2253 ignored if IGNOREVARS is true.
2254
2255 STOPCHAR _cannot_ be '$' if IGNOREVARS is true. */
2256
2257static char *
2258find_char_unquote (char *string, int map)
2259{
2260 unsigned int string_len = 0;
2261 char *p = string;
2262
2263 /* Always stop on NUL. */
2264 map |= MAP_NUL;
2265
2266 while (1)
2267 {
2268 while (! STOP_SET (*p, map))
2269 ++p;
2270
2271 if (*p == '\0')
2272 break;
2273
2274 /* If we stopped due to a variable reference, skip over its contents. */
2275 if (STOP_SET (*p, MAP_VARIABLE))
2276 {
2277 char openparen = p[1];
2278
2279 /* Check if '$' is the last character in the string. */
2280 if (openparen == '\0')
2281 break;
2282
2283 p += 2;
2284
2285 /* Skip the contents of a non-quoted, multi-char variable ref. */
2286 if (openparen == '(' || openparen == '{')
2287 {
2288 unsigned int pcount = 1;
2289 char closeparen = (openparen == '(' ? ')' : '}');
2290
2291 while (*p)
2292 {
2293 if (*p == openparen)
2294 ++pcount;
2295 else if (*p == closeparen)
2296 if (--pcount == 0)
2297 {
2298 ++p;
2299 break;
2300 }
2301 ++p;
2302 }
2303 }
2304
2305 /* Skipped the variable reference: look for STOPCHARS again. */
2306 continue;
2307 }
2308
2309 if (p > string && p[-1] == '\\')
2310 {
2311 /* Search for more backslashes. */
2312 int i = -2;
2313 while (&p[i] >= string && p[i] == '\\')
2314 --i;
2315 ++i;
2316 /* Only compute the length if really needed. */
2317 if (string_len == 0)
2318 string_len = strlen (string);
2319 /* The number of backslashes is now -I.
2320 Copy P over itself to swallow half of them. */
2321 memmove (&p[i], &p[i/2], (string_len - (p - string)) - (i/2) + 1);
2322 p += i/2;
2323 if (i % 2 == 0)
2324 /* All the backslashes quoted each other; the STOPCHAR was
2325 unquoted. */
2326 return p;
2327
2328 /* The STOPCHAR was quoted by a backslash. Look for another. */
2329 }
2330 else
2331 /* No backslash in sight. */
2332 return p;
2333 }
2334
2335 /* Never hit a STOPCHAR or blank (with BLANK nonzero). */
2336 return 0;
2337}

2339/* Unescape a character in a string. The string is compressed onto itself. */
2340
2341static char *
2342unescape_char (char *string, int c)
2343{
2344 char *p = string;
2345 char *s = string;
2346
2347 while (*s != '\0')
2348 {
2349 if (*s == '\\')
2350 {
2351 char *e = s;
2352 int l;
2353
2354 /* We found a backslash. See if it's escaping our character. */
2355 while (*e == '\\')
2356 ++e;
2357 l = e - s;
2358
2359 if (*e != c || l%2 == 0)
2360 {
2361 /* It's not; just take it all without unescaping. */
2362 memmove (p, s, l);
2363 p += l;
2364

2366 if (*e == '\0')
2367 break;
2368 }

2373 memmove (p, s, l);
2374 p += l;
2375 }
2376
2377 s = e;
2378 }
2379
2380 *(p++) = *(s++);
2381 }
2382
2383 *p = '\0';

2395/* Search STRING for an unquoted % and handle quoting. Returns a pointer to
2396 the % or NULL if no % was found.
2397 This version is used with strings in the string cache: if there's a need to
2398 modify the string a new version will be added to the string cache and
2399 *STRING will be set to that. */
2400
2401const char *
2402find_percent_cached (const char **string)
2403{
2404 const char *p = *string;
2405 char *new = 0;
2406 int slen = 0;
2407
2408 /* If the first char is a % return now. This lets us avoid extra tests
2409 inside the loop. */
2410 if (*p == '%')
2411 return p;
2412
2413 while (1)
2414 {
2415 while (! STOP_SET (*p, MAP_PERCENT|MAP_NUL))
2416 ++p;
2417
2418 if (*p == '\0')
2419 break;
2420
2421 /* See if this % is escaped with a backslash; if not we're done. */
2422 if (p[-1] != '\\')
2423 break;
2424
2425 {
2426 /* Search for more backslashes. */
2427 char *pv;
2428 int i = -2;
2429
2430 while (&p[i] >= *string && p[i] == '\\')
2431 --i;
2432 ++i;
2433
2434 /* At this point we know we'll need to allocate a new string.
2435 Make a copy if we haven't yet done so. */
2436 if (! new)
2437 {
2438 slen = strlen (*string);
2439 new = alloca (slen + 1);
2440 memcpy (new, *string, slen + 1);
2441 p = new + (p - *string);
2442 *string = new;
2443 }
2444
2445 /* At this point *string, p, and new all point into the same string.
2446 Get a non-const version of p so we can modify new. */
2447 pv = new + (p - *string);
2448
2449 /* The number of backslashes is now -I.
2450 Copy P over itself to swallow half of them. */
2451 memmove (&pv[i], &pv[i/2], (slen - (pv - new)) - (i/2) + 1);
2452 p += i/2;
2453
2454 /* If the backslashes quoted each other; the % was unquoted. */
2455 if (i % 2 == 0)
2456 break;
2457 }
2458 }
2459
2460 /* If we had to change STRING, add it to the strcache. */
2461 if (new)
2462 {
2463 *string = strcache_add (*string);
2464 p = *string + (p - new);
2465 }
2466
2467 /* If we didn't find a %, return NULL. Otherwise return a ptr to it. */
2468 return (*p == '\0') ? NULL : p;
2469}

2471/* Find the next line of text in an eval buffer, combining continuation lines
2472 into one line.
2473 Return the number of actual lines read (> 1 if continuation lines).
2474 Returns -1 if there's nothing left in the buffer.
2475
2476 After this function, ebuf->buffer points to the first character of the
2477 line we just found.
2478 */
2479
2480/* Read a line of text from a STRING.
2481 Since we aren't really reading from a file, don't bother with linenumbers.
2482 */
2483
2484static long
2485readstring (struct ebuffer *ebuf)
2486{
2487 char *eol;
2488
2489 /* If there is nothing left in this buffer, return 0. */
2490 if (ebuf->bufnext >= ebuf->bufstart + ebuf->size)
2491 return -1;
2492
2493 /* Set up a new starting point for the buffer, and find the end of the
2494 next logical line (taking into account backslash/newline pairs). */
2495
2496 eol = ebuf->buffer = ebuf->bufnext;
2497
2498 while (1)
2499 {
2500 int backslash = 0;
2501 const char *bol = eol;
2502 const char *p;
2503
2504 /* Find the next newline. At EOS, stop. */
2505 p = eol = strchr (eol , '\n');
2506 if (!eol)
2507 {
2508 ebuf->bufnext = ebuf->bufstart + ebuf->size + 1;
2509 return 0;
2510 }
2511
2512 /* Found a newline; if it's escaped continue; else we're done. */
2513 while (p > bol && *(--p) == '\\')
2514 backslash = !backslash;
2515 if (!backslash)
2516 break;
2517 ++eol;
2518 }
2519
2520 /* Overwrite the newline char. */
2521 *eol = '\0';
2522 ebuf->bufnext = eol+1;

2524 return 0;

2527static long
2528readline (struct ebuffer *ebuf)
2529{
2530 char *p;
2531 char *end;
2532 char *start;
2533 long nlines = 0;
2534
2535 /* The behaviors between string and stream buffers are different enough to
2536 warrant different functions. Do the Right Thing. */
2537
2538 if (!ebuf->fp)
2539 return readstring (ebuf);
2540
2541 /* When reading from a file, we always start over at the beginning of the
2542 buffer for each new line. */
2543
2544 p = start = ebuf->bufstart;
2545 end = p + ebuf->size;
2546 *p = '\0';
2547
2548 while (fgets (p, end - p, ebuf->fp) != 0)
2549 {
2550 char *p2;
2551 unsigned long len;
2552 int backslash;
2553
2554 len = strlen (p);
2555 if (len == 0)
2556 {
2557 /* This only happens when the first thing on the line is a '\0'.
2558 It is a pretty hopeless case, but (wonder of wonders) Athena
2559 lossage strikes again! (xmkmf puts NULs in its makefiles.)
2560 There is nothing really to be done; we synthesize a newline so
2561 the following line doesn't appear to be part of this line. */
2562 O (error, &ebuf->floc,
2563 _("warning: NUL character seen; rest of line ignored"));
2564 p[0] = '\n';
2565 len = 1;
2566 }
2567
2568 /* Jump past the text we just read. */
2569 p += len;
2570
2571 /* If the last char isn't a newline, the whole line didn't fit into the
2572 buffer. Get some more buffer and try again. */
2573 if (p[-1] != '\n')
2574 goto more_buffer;
2575
2576 /* We got a newline, so add one to the count of lines. */
2577 ++nlines;
2578
2579#if !defined(WINDOWS32) && !defined(__MSDOS__) && !defined(__EMX__)
2580 /* Check to see if the line was really ended with CRLF; if so ignore
2581 the CR. */
2582 if ((p - start) > 1 && p[-2] == '\r')
2583 {
2584 --p;
2585 memmove (p-1, p, strlen (p) + 1);
2586 }
2587#endif
2588
2589 backslash = 0;
2590 for (p2 = p - 2; p2 >= start; --p2)
2591 {
2592 if (*p2 != '\\')
2593 break;
2594 backslash = !backslash;
2595 }
2596
2597 if (!backslash)
2598 {
2599 p[-1] = '\0';
2600 break;
2601 }
2602
2603 /* It was a backslash/newline combo. If we have more space, read
2604 another line. */
2605 if (end - p >= 80)
2606 continue;
2607
2608 /* We need more space at the end of our buffer, so realloc it.
2609 Make sure to preserve the current offset of p. */
2610 more_buffer:
2611 {
2612 unsigned long off = p - start;
2613 ebuf->size *= 2;
2614 start = ebuf->buffer = ebuf->bufstart = xrealloc (start, ebuf->size);
2615 p = start + off;
2616 end = start + ebuf->size;
2617 *p = '\0';
2618 }
2619 }
2620
2621 if (ferror (ebuf->fp))
2622 pfatal_with_name (ebuf->floc.filenm);
2623
2624 /* If we found some lines, return how many.
2625 If we didn't, but we did find _something_, that indicates we read the last
2626 line of a file with no final newline; return 1.
2627 If we read nothing, we're at EOF; return -1. */
2628
2629 return nlines ? nlines : p == ebuf->bufstart ? -1 : 1;
2630}

2632/* Parse the next "makefile word" from the input buffer, and return info
2633 about it.
2634
2635 A "makefile word" is one of:
2636
2637 w_bogus Should never happen
2638 w_eol End of input
2639 w_static A static word; cannot be expanded
2640 w_variable A word containing one or more variables/functions
2641 w_colon A colon
2642 w_dcolon A double-colon
2643 w_semicolon A semicolon
2644 w_varassign A variable assignment operator (=, :=, ::=, +=, ?=, or !=)
2645
2646 Note that this function is only used when reading certain parts of the
2647 makefile. Don't use it where special rules hold sway (RHS of a variable,
2648 in a command list, etc.) */
2649
2650static enum make_word_type
2651get_next_mword (char *buffer, char *delim, char **startp, unsigned int *length)
2652{
2653 enum make_word_type wtype = w_bogus;
2654 char *p = buffer, *beg;
2655 char c;
2656
2657 /* Skip any leading whitespace. */
2658 while (ISBLANK (*p))
2659 ++p;
2660
2661 beg = p;
2662 c = *(p++);
2663 switch (c)
2664 {
2665 case '\0':
2666 wtype = w_eol;
2667 break;
2668
2669 case ';':
2670 wtype = w_semicolon;
2671 break;
2672
2673 case '=':
2674 wtype = w_varassign;
2675 break;
2676
2677 case ':':
2678 wtype = w_colon;
2679 switch (*p)
2680 {
2681 case ':':
2682 ++p;
2683 if (p[1] != '=')
2684 wtype = w_dcolon;
2685 else
2686 {
2687 wtype = w_varassign;
2688 ++p;
2689 }
2690 break;
2691
2692 case '=':
2693 ++p;
2694 wtype = w_varassign;
2695 break;
2696 }
2697 break;
2698
2699 case '+':
2700 case '?':
2701 case '!':
2702 if (*p == '=')
2703 {
2704 ++p;
2705 wtype = w_varassign;
2706 break;
2707 }
2708
2709 default:
2710 if (delim && strchr (delim, c))
2711 wtype = w_static;
2712 break;
2713 }
2714
2715 /* Did we find something? If so, return now. */
2716 if (wtype != w_bogus)
2717 goto done;
2718
2719 /* This is some non-operator word. A word consists of the longest
2720 string of characters that doesn't contain whitespace, one of [:=#],
2721 or [?+!]=, or one of the chars in the DELIM string. */
2722
2723 /* We start out assuming a static word; if we see a variable we'll
2724 adjust our assumptions then. */
2725 wtype = w_static;
2726
2727 /* We already found the first value of "c", above. */
2728 while (1)
2729 {
2730 char closeparen;
2731 int count;
2732
2733 switch (c)
2734 {
2735 case '\0':
2736 case ' ':
2737 case '\t':
2738 case '=':
2739 goto done_word;
2740
2741 case ':':
2742#ifdef HAVE_DOS_PATHS
2743 /* A word CAN include a colon in its drive spec. The drive
2744 spec is allowed either at the beginning of a word, or as part
2745 of the archive member name, like in "libfoo.a(d:/foo/bar.o)". */
2746 if (!(p - beg >= 2
2747 && (*p == '/' || *p == '\\') && isalpha ((unsigned char)p[-2])
2748 && (p - beg == 2 || p[-3] == '(')))
2749#endif
2750 goto done_word;
2751
2752 case '$':
2753 c = *(p++);
2754 if (c == '$')
2755 break;
2756 if (c == '\0')
2757 goto done_word;
2758
2759 /* This is a variable reference, so note that it's expandable.
2760 Then read it to the matching close paren. */
2761 wtype = w_variable;
2762
2763 if (c == '(')
2764 closeparen = ')';
2765 else if (c == '{')
2766 closeparen = '}';
2767 else
2768 /* This is a single-letter variable reference. */
2769 break;
2770
2771 for (count=0; *p != '\0'; ++p)
2772 {
2773 if (*p == c)
2774 ++count;
2775 else if (*p == closeparen && --count < 0)
2776 {
2777 ++p;
2778 break;
2779 }
2780 }
2781 break;
2782
2783 case '?':
2784 case '+':
2785 if (*p == '=')
2786 goto done_word;
2787 break;
2788
2789 case '\\':
2790 switch (*p)
2791 {
2792 case ':':
2793 case ';':
2794 case '=':
2795 case '\\':
2796 ++p;
2797 break;
2798 }
2799 break;
2800
2801 default:
2802 if (delim && strchr (delim, c))
2803 goto done_word;
2804 break;
2805 }
2806
2807 c = *(p++);
2808 }
2809 done_word:
2810 --p;
2811
2812 done:
2813 if (startp)
2814 *startp = beg;
2815 if (length)
2816 *length = p - beg;
2817 return wtype;
2818}
