22#define NAMESEQ(_t) \
23 _t *next; \
24 const char *name

31/* Flag bits for the second argument to 'read_makefile'.
32 These flags are saved in the 'flags' field of each
33 'struct goaldep' in the chain returned by 'read_all_makefiles'. */
34
35#define RM_NO_DEFAULT_GOAL (1 << 0) /* Do not set default goal. */
36#define RM_INCLUDED (1 << 1) /* Search makefile search path. */
37#define RM_DONTCARE (1 << 2) /* No error if it doesn't exist. */
38#define RM_NO_TILDE (1 << 3) /* Don't expand ~ in file name. */
39#define RM_NOFLAG 0

41/* Structure representing one dependency of a file.
42 Each struct file's 'deps' points to a chain of these,
43 chained through the 'next'. 'stem' is the stem for this
44 dep line of static pattern rule or NULL.
45
46 Note that the first two words of this match a struct nameseq. */
47
48#define DEP(_t) \
49 NAMESEQ (_t); \
50 struct file *file; \
51 const char *stem; \
52 unsigned short flags : 8; \
53 unsigned short changed : 1; \
54 unsigned short ignore_mtime : 1; \
55 unsigned short staticpattern : 1; \
56 unsigned short need_2nd_expansion : 1
57
58struct dep
59 {
60 DEP (struct dep);
61 };
62
63/* Structure representing one goal.
64 The goals to be built constitute a chain of these, chained through 'next'.
65 'stem' is not used, but it's simpler to include and ignore it. */
66
67struct goaldep
68 {
69 DEP (struct goaldep);
70 unsigned short error;
71 gmk_floc floc;
72 };
