19HANDLE_DI_FLAG(0, Zero) // Use it as zero value.
20 // For example: void foo(DIFlags Flags = FlagZero).
21HANDLE_DI_FLAG(1, Private)
22HANDLE_DI_FLAG(2, Protected)
23HANDLE_DI_FLAG(3, Public)

38HANDLE_DI_FLAG((1 << 15), Reserved)
39HANDLE_DI_FLAG((1 << 16), SingleInheritance)
40HANDLE_DI_FLAG((2 << 16), MultipleInheritance)
41HANDLE_DI_FLAG((3 << 16), VirtualInheritance)
42HANDLE_DI_FLAG((1 << 18), IntroducedVirtual)

44HANDLE_DI_FLAG((1 << 20), NoReturn)

47// To avoid needing a dedicated value for IndirectVirtualBase, we use
48// the bitwise or of Virtual and FwdDecl, which does not otherwise
49// make sense for inheritance.
50HANDLE_DI_FLAG((1 << 2) | (1 << 5), IndirectVirtualBase)
