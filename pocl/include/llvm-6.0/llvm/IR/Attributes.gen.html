124struct EnumAttr {
125 static bool isSet(const Function &Fn,
126 Attribute::AttrKind Kind) {
127 return Fn.hasFnAttribute(Kind);
128 }
129
130 static void set(Function &Fn,
131 Attribute::AttrKind Kind, bool Val) {
132 if (Val)
133 Fn.addFnAttr(Kind);
134 else
135 Fn.removeFnAttr(Kind);
136 }
137};

139struct StrBoolAttr {
140 static bool isSet(const Function &Fn,
141 StringRef Kind) {
142 auto A = Fn.getFnAttribute(Kind);
143 return A.getValueAsString().equals("true");
144 }
145
146 static void set(Function &Fn,
147 StringRef Kind, bool Val) {
148 Fn.addFnAttr(Kind, Val ? "true" : "false");
149 }
150};

461static inline bool hasCompatibleFnAttrs(const Function &Caller,
462 const Function &Callee) {
463 bool Ret = true;
464
465 Ret &= isEqual<SanitizeAddressAttr>(Caller, Callee);
466 Ret &= isEqual<SanitizeThreadAttr>(Caller, Callee);
467 Ret &= isEqual<SanitizeMemoryAttr>(Caller, Callee);
468 Ret &= isEqual<SanitizeHWAddressAttr>(Caller, Callee);
469 Ret &= isEqual<SafeStackAttr>(Caller, Callee);
470
471 return Ret;
472}
473
474static inline void mergeFnAttrs(Function &Caller,
475 const Function &Callee) {
476 setOR<NoJumpTablesAttr>(Caller, Callee);
477 setOR<ProfileSampleAccurateAttr>(Caller, Callee);
478 adjustCallerSSPLevel(Caller, Callee);
479 adjustCallerStackProbes(Caller, Callee);
480 adjustCallerStackProbeSize(Caller, Callee);
481 setAND<LessPreciseFPMADAttr>(Caller, Callee);
482 setAND<NoInfsFPMathAttr>(Caller, Callee);
483 setAND<NoNansFPMathAttr>(Caller, Callee);
484 setAND<UnsafeFPMathAttr>(Caller, Callee);
485 setOR<NoImplicitFloatAttr>(Caller, Callee);
486}
