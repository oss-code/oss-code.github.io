10// This file defines the MDBuilder class, which is used as a convenient way to
11// create LLVM metadata with a consistent and simplified interface.
12//
13//===----------------------------------------------------------------------===//
14
15#ifndef LLVM_IR_MDBUILDER_H
16#define LLVM_IR_MDBUILDER_H
17
18#include "llvm/ADT/DenseSet.h"
19#include "llvm/ADT/StringRef.h"
20#include "llvm/IR/GlobalValue.h"
21#include "llvm/Support/DataTypes.h"
22#include <utility>
23
24namespace llvm {
25
26class APInt;
27template <typename T> class ArrayRef;
28class LLVMContext;
29class Constant;
30class ConstantAsMetadata;
31class MDNode;
32class MDString;
33class Metadata;
34
35class MDBuilder {
36 LLVMContext &Context;
37
38public:
39 MDBuilder(LLVMContext &context) : Context(context) {}
40
41 /// \brief Return the given string as metadata.
42 MDString *createString(StringRef Str);
43
44 /// \brief Return the given constant as metadata.
45 ConstantAsMetadata *createConstant(Constant *C);
46
47 //===------------------------------------------------------------------===//
48 // FPMath metadata.
49 //===------------------------------------------------------------------===//
50
51 /// \brief Return metadata with the given settings. The special value 0.0
52 /// for the Accuracy parameter indicates the default (maximal precision)
53 /// setting.
54 MDNode *createFPMath(float Accuracy);
55
56 //===------------------------------------------------------------------===//
57 // Prof metadata.
58 //===------------------------------------------------------------------===//
59
60 /// \brief Return metadata containing two branch weights.
61 MDNode *createBranchWeights(uint32_t TrueWeight, uint32_t FalseWeight);
62
63 /// \brief Return metadata containing a number of branch weights.
64 MDNode *createBranchWeights(ArrayRef<uint32_t> Weights);
65
66 /// Return metadata specifying that a branch or switch is unpredictable.
67 MDNode *createUnpredictable();
68
69 /// Return metadata containing the entry \p Count for a function, and the
70 /// GUIDs stored in \p Imports that need to be imported for sample PGO, to
71 /// enable the same inlines as the profiled optimized binary
72 MDNode *createFunctionEntryCount(uint64_t Count,
73 const DenseSet<GlobalValue::GUID> *Imports);
74
75 /// Return metadata containing the section prefix for a function.
76 MDNode *createFunctionSectionPrefix(StringRef Prefix);
77
78 //===------------------------------------------------------------------===//
79 // Range metadata.
80 //===------------------------------------------------------------------===//
81
82 /// \brief Return metadata describing the range [Lo, Hi).
83 MDNode *createRange(const APInt &Lo, const APInt &Hi);
84
85 /// \brief Return metadata describing the range [Lo, Hi).
86 MDNode *createRange(Constant *Lo, Constant *Hi);
87
88 //===------------------------------------------------------------------===//
89 // Callees metadata.
90 //===------------------------------------------------------------------===//
91
92 /// \brief Return metadata indicating the possible callees of indirect
93 /// calls.
94 MDNode *createCallees(ArrayRef<Function *> Callees);
95
96 //===------------------------------------------------------------------===//
97 // AA metadata.
98 //===------------------------------------------------------------------===//
99
100protected:
101 /// \brief Return metadata appropriate for a AA root node (scope or TBAA).
102 /// Each returned node is distinct from all other metadata and will never
103 /// be identified (uniqued) with anything else.
104 MDNode *createAnonymousAARoot(StringRef Name = StringRef(),
105 MDNode *Extra = nullptr);
106
107public:
108 /// \brief Return metadata appropriate for a TBAA root node. Each returned
109 /// node is distinct from all other metadata and will never be identified
110 /// (uniqued) with anything else.
111 MDNode *createAnonymousTBAARoot() {
112 return createAnonymousAARoot();
113 }
114
115 /// \brief Return metadata appropriate for an alias scope domain node.
116 /// Each returned node is distinct from all other metadata and will never
117 /// be identified (uniqued) with anything else.
118 MDNode *createAnonymousAliasScopeDomain(StringRef Name = StringRef()) {
119 return createAnonymousAARoot(Name);
120 }
121
122 /// \brief Return metadata appropriate for an alias scope root node.
123 /// Each returned node is distinct from all other metadata and will never
124 /// be identified (uniqued) with anything else.
125 MDNode *createAnonymousAliasScope(MDNode *Domain,
126 StringRef Name = StringRef()) {
127 return createAnonymousAARoot(Name, Domain);
128 }
129
130 /// \brief Return metadata appropriate for a TBAA root node with the given
131 /// name. This may be identified (uniqued) with other roots with the same
132 /// name.
133 MDNode *createTBAARoot(StringRef Name);
134
135 /// \brief Return metadata appropriate for an alias scope domain node with
136 /// the given name. This may be identified (uniqued) with other roots with
137 /// the same name.
138 MDNode *createAliasScopeDomain(StringRef Name);
139
140 /// \brief Return metadata appropriate for an alias scope node with
141 /// the given name. This may be identified (uniqued) with other scopes with
142 /// the same name and domain.
143 MDNode *createAliasScope(StringRef Name, MDNode *Domain);
144
145 /// \brief Return metadata for a non-root TBAA node with the given name,
146 /// parent in the TBAA tree, and value for 'pointsToConstantMemory'.
147 MDNode *createTBAANode(StringRef Name, MDNode *Parent,
148 bool isConstant = false);
149
150 struct TBAAStructField {
151 uint64_t Offset;
152 uint64_t Size;
153 MDNode *Type;
154 TBAAStructField(uint64_t Offset, uint64_t Size, MDNode *Type) :
155 Offset(Offset), Size(Size), Type(Type) {}
156 };
157
158 /// \brief Return metadata for a tbaa.struct node with the given
159 /// struct field descriptions.
160 MDNode *createTBAAStructNode(ArrayRef<TBAAStructField> Fields);
161
162 /// \brief Return metadata for a TBAA struct node in the type DAG
163 /// with the given name, a list of pairs (offset, field type in the type DAG).
164 MDNode *
165 createTBAAStructTypeNode(StringRef Name,
166 ArrayRef<std::pair<MDNode *, uint64_t>> Fields);
167
168 /// \brief Return metadata for a TBAA scalar type node with the
169 /// given name, an offset and a parent in the TBAA type DAG.
170 MDNode *createTBAAScalarTypeNode(StringRef Name, MDNode *Parent,
171 uint64_t Offset = 0);
172
173 /// \brief Return metadata for a TBAA tag node with the given
174 /// base type, access type and offset relative to the base type.
175 MDNode *createTBAAStructTagNode(MDNode *BaseType, MDNode *AccessType,
176 uint64_t Offset, bool IsConstant = false);
177
178 /// \brief Return metadata for a TBAA type node in the TBAA type DAG with the
179 /// given parent type, size in bytes, type identifier and a list of fields.
180 MDNode *createTBAATypeNode(MDNode *Parent, uint64_t Size, Metadata *Id,
181 ArrayRef<TBAAStructField> Fields =
182 ArrayRef<TBAAStructField>());
183
184 /// \brief Return metadata for a TBAA access tag with the given base type,
185 /// final access type, offset of the access relative to the base type, size of
186 /// the access and flag indicating whether the accessed object can be
187 /// considered immutable for the purposes of the TBAA analysis.
188 MDNode *createTBAAAccessTag(MDNode *BaseType, MDNode *AccessType,
189 uint64_t Offset, uint64_t Size,
190 bool IsImmutable = false);
191
192 /// \brief Return metadata containing an irreducible loop header weight.
193 MDNode *createIrrLoopHeaderWeight(uint64_t Weight);
194};
195
196} // end namespace llvm
197
198#endif
