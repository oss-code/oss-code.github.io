78/// Option class for critical edge splitting.
79///
80/// This provides a builder interface for overriding the default options used
81/// during critical edge splitting.
82struct CriticalEdgeSplittingOptions {

84 LoopInfo *LI;
85 bool MergeIdenticalEdges = false;
86 bool DontDeleteUselessPHIs = false;
87 bool PreserveLCSSA = false;
88
89 CriticalEdgeSplittingOptions(DominatorTree *DT = nullptr,
90 LoopInfo *LI = nullptr)
91 : DT(DT), LI(LI) {}

109/// If this edge is a critical edge, insert a new node to split the critical
110/// edge. This will update the analyses passed in through the option struct.
111/// This returns the new block if the edge was split, null otherwise.
112///
113/// If MergeIdenticalEdges in the options struct is true (not the default),
114/// *all* edges from TI to the specified successor will be merged into the same
115/// critical edge block. This is most commonly interesting with switch
116/// instructions, which may have many edges to any one destination. This
117/// ensures that all edges to that dest go to one block instead of each going
118/// to a different block, but isn't the standard definition of a "critical
119/// edge".
120///
121/// It is invalid to call this function on a critical edge that starts at an
122/// IndirectBrInst. Splitting these edges will almost always create an invalid
123/// program because the address of the new block won't be the one that is jumped
124/// to.
125BasicBlock *SplitCriticalEdge(TerminatorInst *TI, unsigned SuccNum,
126 const CriticalEdgeSplittingOptions &Options =
127 CriticalEdgeSplittingOptions());
128
129inline BasicBlock *
130SplitCriticalEdge(BasicBlock *BB, succ_iterator SI,
131 const CriticalEdgeSplittingOptions &Options =
132 CriticalEdgeSplittingOptions()) {
133 return SplitCriticalEdge(BB->getTerminator(), SI.getSuccessorIndex(),
134 Options);
135}
136
137/// If the edge from *PI to BB is not critical, return false. Otherwise, split
138/// all edges between the two blocks and return true. This updates all of the
139/// same analyses as the other SplitCriticalEdge function. If P is specified, it
140/// updates the analyses described above.
141inline bool SplitCriticalEdge(BasicBlock *Succ, pred_iterator PI,
142 const CriticalEdgeSplittingOptions &Options =
143 CriticalEdgeSplittingOptions()) {
144 bool MadeChange = false;
145 TerminatorInst *TI = (*PI)->getTerminator();
146 for (unsigned i = 0, e = TI->getNumSuccessors(); i != e; ++i)
147 if (TI->getSuccessor(i) == Succ)
148 MadeChange |= !!SplitCriticalEdge(TI, i, Options);
149 return MadeChange;
150}
151
152/// If an edge from Src to Dst is critical, split the edge and return true,
153/// otherwise return false. This method requires that there be an edge between
154/// the two blocks. It updates the analyses passed in the options struct
155inline BasicBlock *
156SplitCriticalEdge(BasicBlock *Src, BasicBlock *Dst,
157 const CriticalEdgeSplittingOptions &Options =
158 CriticalEdgeSplittingOptions()) {
159 TerminatorInst *TI = Src->getTerminator();
160 unsigned i = 0;
161 while (true) {
162 assert(i != TI->getNumSuccessors() && "Edge doesn't exist!");
163 if (TI->getSuccessor(i) == Dst)
164 return SplitCriticalEdge(TI, i, Options);
165 ++i;
166 }
167}

288// Split critical edges where the source of the edge is an indirectbr
289// instruction. This isn't always possible, but we can handle some easy cases.
290// This is useful because MI is unable to split such critical edges,
291// which means it will not be able to sink instructions along those edges.
292// This is especially painful for indirect branches with many successors, where
293// we end up having to prepare all outgoing values in the origin block.
294//
295// Our normal algorithm for splitting critical edges requires us to update
296// the outgoing edges of the edge origin block, but for an indirectbr this
297// is hard, since it would require finding and updating the block addresses
298// the indirect branch uses. But if a block only has a single indirectbr
299// predecessor, with the others being regular branches, we can do it in a
300// different way.
301// Say we have A -> D, B -> D, I -> D where only I -> D is an indirectbr.
302// We can split D into D0 and D1, where D0 contains only the PHIs from D,
303// and D1 is the D block body. We can then duplicate D0 as D0A and D0B, and
304// create the following structure:
305// A -> D0A, B -> D0A, I -> D0B, D0A -> D1, D0B -> D1
306// If BPI and BFI aren't non-null, BPI/BFI will be updated accordingly.
307bool SplitIndirectBrCriticalEdges(Function &F,
308 BranchProbabilityInfo *BPI = nullptr,
309 BlockFrequencyInfo *BFI = nullptr);
