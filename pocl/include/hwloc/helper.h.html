33/** \brief Get the first largest object included in the given cpuset \p set.
34 *
35 * \return the first object that is included in \p set and whose parent is not.
36 *
37 * This is convenient for iterating over all largest objects within a CPU set
38 * by doing a loop getting the first largest object and clearing its CPU set
39 * from the remaining CPU set.
40 *
41 * \note This function cannot work if the root object does not have a CPU set,
42 * e.g. if the topology is made of different machines.
43 */
44static __hwloc_inline hwloc_obj_t
45hwloc_get_first_largest_obj_inside_cpuset(hwloc_topology_t topology, hwloc_const_cpuset_t set)
46{
47 hwloc_obj_t obj = hwloc_get_root_obj(topology);
48 if (!obj->cpuset || !hwloc_bitmap_intersects(obj->cpuset, set))
49 return NULL;
50 while (!hwloc_bitmap_isincluded(obj->cpuset, set)) {
51 /* while the object intersects without being included, look at its children */
52 hwloc_obj_t child = obj->first_child;
53 while (child) {
54 if (child->cpuset && hwloc_bitmap_intersects(child->cpuset, set))
55 break;
56 child = child->next_sibling;
57 }
58 if (!child)
59 /* no child intersects, return their father */
60 return obj;
61 /* found one intersecting child, look at its children */
62 obj = child;
63 }
64 /* obj is included, return it */
65 return obj;
66}

78/** \brief Return the next object at depth \p depth included in CPU set \p set.
79 *
80 * If \p prev is \c NULL, return the first object at depth \p depth
81 * included in \p set. The next invokation should pass the previous
82 * return value in \p prev so as to obtain the next object in \p set.
83 *
84 * \note This function cannot work if objects at the given depth do
85 * not have CPU sets or if the topology is made of different machines.
86 */
87static __hwloc_inline hwloc_obj_t
88hwloc_get_next_obj_inside_cpuset_by_depth (hwloc_topology_t topology, hwloc_const_cpuset_t set,
89 unsigned depth, hwloc_obj_t prev)
90{
91 hwloc_obj_t next = hwloc_get_next_obj_by_depth(topology, depth, prev);
92 if (!next || !next->cpuset)
93 return NULL;
94 while (next && !hwloc_bitmap_isincluded(next->cpuset, set))
95 next = next->next_cousin;
96 return next;
97}

118/** \brief Return the (logically) \p idx -th object at depth \p depth included in CPU set \p set.
119 *
120 * \note This function cannot work if objects at the given depth do
121 * not have CPU sets or if the topology is made of different machines.
122 */
123static __hwloc_inline hwloc_obj_t
124hwloc_get_obj_inside_cpuset_by_depth (hwloc_topology_t topology, hwloc_const_cpuset_t set,
125 unsigned depth, unsigned idx) __hwloc_attribute_pure;
126static __hwloc_inline hwloc_obj_t
127hwloc_get_obj_inside_cpuset_by_depth (hwloc_topology_t topology, hwloc_const_cpuset_t set,
128 unsigned depth, unsigned idx)
129{
130 hwloc_obj_t obj = hwloc_get_obj_by_depth (topology, depth, 0);
131 unsigned count = 0;
132 if (!obj || !obj->cpuset)
133 return NULL;
134 while (obj) {
135 if (hwloc_bitmap_isincluded(obj->cpuset, set)) {
136 if (count == idx)
137 return obj;
138 count++;
139 }
140 obj = obj->next_cousin;
141 }
142 return NULL;
143}

215/** \brief Return the logical index among the objects included in CPU set \p set.
216 *
217 * Consult all objects in the same level as \p obj and inside CPU set \p set
218 * in the logical order, and return the index of \p obj within them.
219 * If \p set covers the entire topology, this is the logical index of \p obj.
220 * Otherwise, this is similar to a logical index within the part of the topology
221 * defined by CPU set \p set.
222 */
223static __hwloc_inline int
224hwloc_get_obj_index_inside_cpuset (hwloc_topology_t topology __hwloc_attribute_unused, hwloc_const_cpuset_t set,
225 hwloc_obj_t obj) __hwloc_attribute_pure;
226static __hwloc_inline int
227hwloc_get_obj_index_inside_cpuset (hwloc_topology_t topology __hwloc_attribute_unused, hwloc_const_cpuset_t set,
228 hwloc_obj_t obj)
229{
230 int idx = 0;
231 if (!hwloc_bitmap_isincluded(obj->cpuset, set))
232 return -1;
233 /* count how many objects are inside the cpuset on the way from us to the beginning of the level */
234 while ((obj = obj->prev_cousin) != NULL)
235 if (hwloc_bitmap_isincluded(obj->cpuset, set))
236 idx++;
237 return idx;
238}

248/** \brief Get the child covering at least CPU set \p set.
249 *
250 * \return \c NULL if no child matches or if \p set is empty.
251 *
252 * \note This function cannot work if parent does not have a CPU set.
253 */
254static __hwloc_inline hwloc_obj_t
255hwloc_get_child_covering_cpuset (hwloc_topology_t topology __hwloc_attribute_unused, hwloc_const_cpuset_t set,
256 hwloc_obj_t parent) __hwloc_attribute_pure;
257static __hwloc_inline hwloc_obj_t
258hwloc_get_child_covering_cpuset (hwloc_topology_t topology __hwloc_attribute_unused, hwloc_const_cpuset_t set,
259 hwloc_obj_t parent)
260{
261 hwloc_obj_t child;
262 if (!parent->cpuset || hwloc_bitmap_iszero(set))
263 return NULL;
264 child = parent->first_child;
265 while (child) {
266 if (child->cpuset && hwloc_bitmap_isincluded(set, child->cpuset))
267 return child;
268 child = child->next_sibling;
269 }
270 return NULL;
271}

273/** \brief Get the lowest object covering at least CPU set \p set
274 *
275 * \return \c NULL if no object matches or if \p set is empty.
276 *
277 * \note This function cannot work if the root object does not have a CPU set,
278 * e.g. if the topology is made of different machines.
279 */
280static __hwloc_inline hwloc_obj_t
281hwloc_get_obj_covering_cpuset (hwloc_topology_t topology, hwloc_const_cpuset_t set) __hwloc_attribute_pure;
282static __hwloc_inline hwloc_obj_t
283hwloc_get_obj_covering_cpuset (hwloc_topology_t topology, hwloc_const_cpuset_t set)
284{
285 struct hwloc_obj *current = hwloc_get_root_obj(topology);
286 if (hwloc_bitmap_iszero(set) || !current->cpuset || !hwloc_bitmap_isincluded(set, current->cpuset))
287 return NULL;
288 while (1) {
289 hwloc_obj_t child = hwloc_get_child_covering_cpuset(topology, set, current);
290 if (!child)
291 return current;
292 current = child;
293 }
294}

297 *
298 * If object \p prev is \c NULL, return the first object at depth \p
299 * depth covering at least part of CPU set \p set. The next
300 * invokation should pass the previous return value in \p prev so as
301 * to obtain the next object covering at least another part of \p set.
302 *
303 * \note This function cannot work if objects at the given depth do
304 * not have CPU sets or if the topology is made of different machines.
305 */
306static __hwloc_inline hwloc_obj_t
307hwloc_get_next_obj_covering_cpuset_by_depth(hwloc_topology_t topology, hwloc_const_cpuset_t set,
308 unsigned depth, hwloc_obj_t prev)
309{
310 hwloc_obj_t next = hwloc_get_next_obj_by_depth(topology, depth, prev);
311 if (!next || !next->cpuset)
312 return NULL;
313 while (next && !hwloc_bitmap_intersects(set, next->cpuset))

315 return next;

347/** \defgroup hwlocality_helper_ancestors Looking at Ancestor and Child Objects
348 * @{
349 *
350 * Be sure to see the figure in \ref termsanddefs that shows a
351 * complete topology tree, including depths, child/sibling/cousin
352 * relationships, and an example of an asymmetric topology where one
353 * package has fewer caches than its peers.
354 */
355
356/** \brief Returns the ancestor object of \p obj at depth \p depth. */
357static __hwloc_inline hwloc_obj_t
358hwloc_get_ancestor_obj_by_depth (hwloc_topology_t topology __hwloc_attribute_unused, unsigned depth, hwloc_obj_t obj) __hwloc_attribute_pure;
359static __hwloc_inline hwloc_obj_t
360hwloc_get_ancestor_obj_by_depth (hwloc_topology_t topology __hwloc_attribute_unused, unsigned depth, hwloc_obj_t obj)
361{
362 hwloc_obj_t ancestor = obj;
363 if (obj->depth < depth)
364 return NULL;
365 while (ancestor && ancestor->depth > depth)
366 ancestor = ancestor->parent;
367 return ancestor;
368}
369
370/** \brief Returns the ancestor object of \p obj with type \p type. */
371static __hwloc_inline hwloc_obj_t
372hwloc_get_ancestor_obj_by_type (hwloc_topology_t topology __hwloc_attribute_unused, hwloc_obj_type_t type, hwloc_obj_t obj) __hwloc_attribute_pure;
373static __hwloc_inline hwloc_obj_t
374hwloc_get_ancestor_obj_by_type (hwloc_topology_t topology __hwloc_attribute_unused, hwloc_obj_type_t type, hwloc_obj_t obj)
375{
376 hwloc_obj_t ancestor = obj->parent;
377 while (ancestor && ancestor->type != type)
378 ancestor = ancestor->parent;
379 return ancestor;
380}

382/** \brief Returns the common parent object to objects \p obj1 and \p obj2 */
383static __hwloc_inline hwloc_obj_t
384hwloc_get_common_ancestor_obj (hwloc_topology_t topology __hwloc_attribute_unused, hwloc_obj_t obj1, hwloc_obj_t obj2) __hwloc_attribute_pure;
385static __hwloc_inline hwloc_obj_t
386hwloc_get_common_ancestor_obj (hwloc_topology_t topology __hwloc_attribute_unused, hwloc_obj_t obj1, hwloc_obj_t obj2)
387{
388 /* the loop isn't so easy since intermediate ancestors may have
389 * different depth, causing us to alternate between using obj1->parent
390 * and obj2->parent. Also, even if at some point we find ancestors of
391 * of the same depth, their ancestors may have different depth again.
392 */
393 while (obj1 != obj2) {
394 while (obj1->depth > obj2->depth)
395 obj1 = obj1->parent;
396 while (obj2->depth > obj1->depth)
397 obj2 = obj2->parent;
398 if (obj1 != obj2 && obj1->depth == obj2->depth) {
399 obj1 = obj1->parent;
400 obj2 = obj2->parent;
401 }
402 }
403 return obj1;
404}

406/** \brief Returns true if \p obj is inside the subtree beginning with ancestor object \p subtree_root.
407 *
408 * \note This function assumes that both \p obj and \p subtree_root have a \p cpuset.
409 */
410static __hwloc_inline int
411hwloc_obj_is_in_subtree (hwloc_topology_t topology __hwloc_attribute_unused, hwloc_obj_t obj, hwloc_obj_t subtree_root) __hwloc_attribute_pure;
412static __hwloc_inline int
413hwloc_obj_is_in_subtree (hwloc_topology_t topology __hwloc_attribute_unused, hwloc_obj_t obj, hwloc_obj_t subtree_root)

437 * @{

440/** \brief Find the depth of cache objects matching cache depth and type.
441 *
442 * Return the depth of the topology level that contains cache objects
443 * whose attributes match \p cachedepth and \p cachetype. This function
444 * intends to disambiguate the case where hwloc_get_type_depth() returns
445 * ::HWLOC_TYPE_DEPTH_MULTIPLE.
446 *
447 * If no cache level matches, ::HWLOC_TYPE_DEPTH_UNKNOWN is returned.
448 *
449 * If \p cachetype is ::HWLOC_OBJ_CACHE_UNIFIED, the depth of the
450 * unique matching unified cache level is returned.
451 *
452 * If \p cachetype is ::HWLOC_OBJ_CACHE_DATA or ::HWLOC_OBJ_CACHE_INSTRUCTION,
453 * either a matching cache, or a unified cache is returned.
454 *
455 * If \p cachetype is \c -1, it is ignored and multiple levels may
456 * match. The function returns either the depth of a uniquely matching
457 * level or ::HWLOC_TYPE_DEPTH_MULTIPLE.
458 */
459static __hwloc_inline int
460hwloc_get_cache_type_depth (hwloc_topology_t topology,
461 unsigned cachelevel, hwloc_obj_cache_type_t cachetype)
462{
463 int depth;
464 int found = HWLOC_TYPE_DEPTH_UNKNOWN;
465 for (depth=0; ; depth++) {
466 hwloc_obj_t obj = hwloc_get_obj_by_depth(topology, depth, 0);
467 if (!obj)
468 break;
469 if (obj->type != HWLOC_OBJ_CACHE || obj->attr->cache.depth != cachelevel)
470 /* doesn't match, try next depth */
471 continue;
472 if (cachetype == (hwloc_obj_cache_type_t) -1) {
473 if (found != HWLOC_TYPE_DEPTH_UNKNOWN) {
474 /* second match, return MULTIPLE */
475 return HWLOC_TYPE_DEPTH_MULTIPLE;
476 }
477 /* first match, mark it as found */
478 found = depth;
479 continue;
480 }
481 if (obj->attr->cache.type == cachetype || obj->attr->cache.type == HWLOC_OBJ_CACHE_UNIFIED)
482 /* exact match (either unified is alone, or we match instruction or data), return immediately */
483 return depth;
484 }
485 /* went to the bottom, return what we found */
486 return found;
487}

489/** \brief Get the first cache covering a cpuset \p set
490 *
491 * \return \c NULL if no cache matches.
492 *
493 * \note This function cannot work if the root object does not have a CPU set,
494 * e.g. if the topology is made of different machines.
495 */
496static __hwloc_inline hwloc_obj_t
497hwloc_get_cache_covering_cpuset (hwloc_topology_t topology, hwloc_const_cpuset_t set) __hwloc_attribute_pure;
498static __hwloc_inline hwloc_obj_t
499hwloc_get_cache_covering_cpuset (hwloc_topology_t topology, hwloc_const_cpuset_t set)
500{
501 hwloc_obj_t current = hwloc_get_obj_covering_cpuset(topology, set);
502 while (current) {
503 if (current->type == HWLOC_OBJ_CACHE)
504 return current;
505 current = current->parent;
506 }
507 return NULL;
508}

510/** \brief Get the first cache shared between an object and somebody else.
511 *
512 * \return \c NULL if no cache matches or if an invalid object is given.
513 */
514static __hwloc_inline hwloc_obj_t
515hwloc_get_shared_cache_covering_obj (hwloc_topology_t topology __hwloc_attribute_unused, hwloc_obj_t obj) __hwloc_attribute_pure;
516static __hwloc_inline hwloc_obj_t
517hwloc_get_shared_cache_covering_obj (hwloc_topology_t topology __hwloc_attribute_unused, hwloc_obj_t obj)
518{
519 hwloc_obj_t current = obj->parent;
520 if (!obj->cpuset)
521 return NULL;
522 while (current && current->cpuset) {
523 if (!hwloc_bitmap_isequal(current->cpuset, obj->cpuset)
524 && current->type == HWLOC_OBJ_CACHE)
525 return current;
526 current = current->parent;
527 }
528 return NULL;
529}

544/** \brief Returns the object of type ::HWLOC_OBJ_PU with \p os_index.
545 *
546 * This function is useful for converting a CPU set into the PU
547 * objects it contains.
548 * When retrieving the current binding (e.g. with hwloc_get_cpubind()),
549 * one may iterate over the bits of the resulting CPU set with
550 * hwloc_bitmap_foreach_begin(), and find the corresponding PUs
551 * with this function.
552 */
553static __hwloc_inline hwloc_obj_t
554hwloc_get_pu_obj_by_os_index(hwloc_topology_t topology, unsigned os_index) __hwloc_attribute_pure;
555static __hwloc_inline hwloc_obj_t
556hwloc_get_pu_obj_by_os_index(hwloc_topology_t topology, unsigned os_index)
557{
558 hwloc_obj_t obj = NULL;
559 while ((obj = hwloc_get_next_obj_by_type(topology, HWLOC_OBJ_PU, obj)) != NULL)
560 if (obj->os_index == os_index)
561 return obj;
562 return NULL;
563}

565/** \brief Returns the object of type ::HWLOC_OBJ_NUMANODE with \p os_index.
566 *
567 * This function is useful for converting a nodeset into the NUMA node
568 * objects it contains.
569 * When retrieving the current binding (e.g. with hwloc_get_membind_nodeset()),
570 * one may iterate over the bits of the resulting nodeset with
571 * hwloc_bitmap_foreach_begin(), and find the corresponding NUMA nodes
572 * with this function.
573 */
574static __hwloc_inline hwloc_obj_t
575hwloc_get_numanode_obj_by_os_index(hwloc_topology_t topology, unsigned os_index) __hwloc_attribute_pure;
576static __hwloc_inline hwloc_obj_t
577hwloc_get_numanode_obj_by_os_index(hwloc_topology_t topology, unsigned os_index)
578{
579 hwloc_obj_t obj = NULL;
580 while ((obj = hwloc_get_next_obj_by_type(topology, HWLOC_OBJ_NUMANODE, obj)) != NULL)
581 if (obj->os_index == os_index)

665/** \defgroup hwlocality_helper_distribute Distributing items over a topology
666 * @{
667 */
668
669/** \brief Flags to be given to hwloc_distrib().
670 */
671enum hwloc_distrib_flags_e {
672 /** \brief Distrib in reverse order, starting from the last objects.
673 * \hideinitializer
674 */
675 HWLOC_DISTRIB_FLAG_REVERSE = (1UL<<0)
676};
677
678/** \brief Distribute \p n items over the topology under \p roots
679 *
680 * Array \p set will be filled with \p n cpusets recursively distributed
681 * linearly over the topology under objects \p roots, down to depth \p until
682 * (which can be INT_MAX to distribute down to the finest level).
683 *
684 * \p n_roots is usually 1 and \p roots only contains the topology root object
685 * so as to distribute over the entire topology.
686 *
687 * This is typically useful when an application wants to distribute \p n
688 * threads over a machine, giving each of them as much private cache as
689 * possible and keeping them locally in number order.
690 *
691 * The caller may typically want to also call hwloc_bitmap_singlify()
692 * before binding a thread so that it does not move at all.
693 *
694 * \p flags should be 0 or a OR'ed set of ::hwloc_distrib_flags_e.
695 *
696 * \note This function requires the \p roots objects to have a CPU set.
697 *
698 * \note This function replaces the now deprecated hwloc_distribute()
699 * and hwloc_distributev() functions.
700 */
701static __hwloc_inline int
702hwloc_distrib(hwloc_topology_t topology,
703 hwloc_obj_t *roots, unsigned n_roots,
704 hwloc_cpuset_t *set,
705 unsigned n,
706 unsigned until, unsigned long flags)
707{
708 unsigned i;
709 unsigned tot_weight;
710 unsigned given, givenweight;
711 hwloc_cpuset_t *cpusetp = set;
712
713 if (flags & ~HWLOC_DISTRIB_FLAG_REVERSE) {
714 errno = EINVAL;
715 return -1;
716 }
717
718 tot_weight = 0;
719 for (i = 0; i < n_roots; i++)
720 if (roots[i]->cpuset)
721 tot_weight += hwloc_bitmap_weight(roots[i]->cpuset);
722
723 for (i = 0, given = 0, givenweight = 0; i < n_roots; i++) {
724 unsigned chunk, weight;
725 hwloc_obj_t root = roots[flags & HWLOC_DISTRIB_FLAG_REVERSE ? n_roots-1-i : i];
726 hwloc_cpuset_t cpuset = root->cpuset;
727 if (!cpuset)
728 continue;
729 weight = hwloc_bitmap_weight(cpuset);
730 if (!weight)
731 continue;
732 /* Give to root a chunk proportional to its weight.
733 * If previous chunks got rounded-up, we may get a bit less. */
734 chunk = (( (givenweight+weight) * n + tot_weight-1) / tot_weight)
735 - (( givenweight * n + tot_weight-1) / tot_weight);
736 if (!root->arity || chunk <= 1 || root->depth >= until) {
737 /* We can't split any more, put everything there. */
738 if (chunk) {
739 /* Fill cpusets with ours */
740 unsigned j;
741 for (j=0; j < chunk; j++)
742 cpusetp[j] = hwloc_bitmap_dup(cpuset);
743 } else {
744 /* We got no chunk, just merge our cpuset to a previous one
745 * (the first chunk cannot be empty)
746 * so that this root doesn't get ignored.
747 */
748 assert(given);
749 hwloc_bitmap_or(cpusetp[-1], cpusetp[-1], cpuset);
750 }
751 } else {
752 /* Still more to distribute, recurse into children */
753 hwloc_distrib(topology, root->children, root->arity, cpusetp, chunk, until, flags);
754 }
755 cpusetp += chunk;
756 given += chunk;
757 givenweight += weight;
758 }
759
760 return 0;
761}

895/** \defgroup hwlocality_helper_nodeset_convert Converting between CPU sets and node sets
896 *
897 * There are two semantics for converting cpusets to nodesets depending on how
898 * non-NUMA machines are handled.
899 *
900 * When manipulating nodesets for memory binding, non-NUMA machines should be
901 * considered as having a single NUMA node. The standard conversion routines
902 * below should be used so that marking the first bit of the nodeset means
903 * that memory should be bound to a non-NUMA whole machine.
904 *
905 * When manipulating nodesets as an actual list of NUMA nodes without any
906 * need to handle memory binding on non-NUMA machines, the strict conversion
907 * routines may be used instead.
908 * @{
909 */
910
911/** \brief Convert a CPU set into a NUMA node set and handle non-NUMA cases
912 *
913 * If some NUMA nodes have no CPUs at all, this function never sets their
914 * indexes in the output node set, even if a full CPU set is given in input.
915 *
916 * If the topology contains no NUMA nodes, the machine is considered
917 * as a single memory node, and the following behavior is used:
918 * If \p cpuset is empty, \p nodeset will be emptied as well.
919 * Otherwise \p nodeset will be entirely filled.
920 */
921static __hwloc_inline void
922hwloc_cpuset_to_nodeset(hwloc_topology_t topology, hwloc_const_cpuset_t _cpuset, hwloc_nodeset_t nodeset)
923{
924 int depth = hwloc_get_type_depth(topology, HWLOC_OBJ_NUMANODE);
925 hwloc_obj_t obj;
926
927 if (depth == HWLOC_TYPE_DEPTH_UNKNOWN) {
928 if (hwloc_bitmap_iszero(_cpuset))
929 hwloc_bitmap_zero(nodeset);
930 else
931 /* Assume the whole system */
932 hwloc_bitmap_fill(nodeset);
933 return;
934 }
935
936 hwloc_bitmap_zero(nodeset);
937 obj = NULL;
938 while ((obj = hwloc_get_next_obj_covering_cpuset_by_depth(topology, _cpuset, depth, obj)) != NULL)
939 hwloc_bitmap_set(nodeset, obj->os_index);
940}

943 *
944 * This is the strict variant of hwloc_cpuset_to_nodeset(). It does not fix
945 * non-NUMA cases. If the topology contains some NUMA nodes, behave exactly
946 * the same. However, if the topology contains no NUMA nodes, return an empty
947 * nodeset.
948 */

950hwloc_cpuset_to_nodeset_strict(struct hwloc_topology *topology, hwloc_const_cpuset_t _cpuset, hwloc_nodeset_t nodeset)

953 hwloc_obj_t obj;

955 return;

957 obj = NULL;
958 while ((obj = hwloc_get_next_obj_covering_cpuset_by_depth(topology, _cpuset, depth, obj)) != NULL)
959 hwloc_bitmap_set(nodeset, obj->os_index);
960}

963 *

965 * as a single memory node, and the following behavior is used:
966 * If \p nodeset is empty, \p cpuset will be emptied as well.
967 * Otherwise \p cpuset will be entirely filled.
968 * This is useful for manipulating memory binding sets.

970static __hwloc_inline void
971hwloc_cpuset_from_nodeset(hwloc_topology_t topology, hwloc_cpuset_t _cpuset, hwloc_const_nodeset_t nodeset)

1082/** \brief Get distances for the given depth and covering some objects
1083 *
1084 * Return a distance matrix that describes depth \p depth and covers at
1085 * least object \p obj and all its children.
1086 *
1087 * When looking for the distance between some objects, a common ancestor should
1088 * be passed in \p obj.
1089 *
1090 * \p firstp is set to logical index of the first object described by the matrix.
1091 *
1092 * The returned structure belongs to the hwloc library. The caller should
1093 * not modify or free it.
1094 */
1095static __hwloc_inline const struct hwloc_distances_s *
1096hwloc_get_distance_matrix_covering_obj_by_depth(hwloc_topology_t topology,
1097 hwloc_obj_t obj, unsigned depth,
1098 unsigned *firstp)
1099{
1100 while (obj && obj->cpuset) {
1101 unsigned i;
1102 for(i=0; i<obj->distances_count; i++)
1103 if (obj->distances[i]->relative_depth == depth - obj->depth) {
1104 if (!obj->distances[i]->nbobjs)
1105 continue;
1106 *firstp = hwloc_get_next_obj_inside_cpuset_by_depth(topology, obj->cpuset, depth, NULL)->logical_index;
1107 return obj->distances[i];
1108 }
1109 obj = obj->parent;
1110 }
1111 return NULL;
1112}

1114/** \brief Get the latency in both directions between two objects.
1115 *
1116 * Look at ancestor objects from the bottom to the top until one of them
1117 * contains a distance matrix that matches the objects exactly.
1118 *
1119 * \p latency gets the value from object \p obj1 to \p obj2, while
1120 * \p reverse_latency gets the reverse-direction value, which
1121 * may be different on some architectures.
1122 *
1123 * \return -1 if no ancestor contains a matching latency matrix.
1124 */
1125static __hwloc_inline int
1126hwloc_get_latency(hwloc_topology_t topology,
1127 hwloc_obj_t obj1, hwloc_obj_t obj2,
1128 float *latency, float *reverse_latency)
1129{
1130 hwloc_obj_t ancestor;
1131 const struct hwloc_distances_s * distances;
1132 unsigned first_logical ;
1133
1134 if (obj1->depth != obj2->depth) {
1135 errno = EINVAL;
1136 return -1;
1137 }
1138
1139 ancestor = hwloc_get_common_ancestor_obj(topology, obj1, obj2);
1140 distances = hwloc_get_distance_matrix_covering_obj_by_depth(topology, ancestor, obj1->depth, &first_logical);
1141 if (distances && distances->latency) {
1142 const float * latency_matrix = distances->latency;
1143 unsigned nbobjs = distances->nbobjs;
1144 unsigned l1 = obj1->logical_index - first_logical;
1145 unsigned l2 = obj2->logical_index - first_logical;
1146 *latency = latency_matrix[l1*nbobjs+l2];
1147 *reverse_latency = latency_matrix[l2*nbobjs+l1];
1148 return 0;
1149 }
1150
1151 errno = ENOSYS;
1152 return -1;
1153}

1246/* \brief Checks whether a given bridge covers a given PCI bus.
1247 */
1248static __hwloc_inline int
1249hwloc_bridge_covers_pcibus(hwloc_obj_t bridge,
1250 unsigned domain, unsigned bus)
1251{
1252 return bridge->type == HWLOC_OBJ_BRIDGE
1253 && bridge->attr->bridge.downstream_type == HWLOC_OBJ_BRIDGE_PCI
1254 && bridge->attr->bridge.downstream.pci.domain == domain
1255 && bridge->attr->bridge.downstream.pci.secondary_bus <= bus
1256 && bridge->attr->bridge.downstream.pci.subordinate_bus >= bus;
1257}

1259/** \brief Find the hostbridge that covers the given PCI bus.
1260 *
1261 * This is useful for finding the locality of a bus because
1262 * it is the hostbridge parent cpuset.
1263 */
1264static __hwloc_inline hwloc_obj_t
1265hwloc_get_hostbridge_by_pcibus(hwloc_topology_t topology,
1266 unsigned domain, unsigned bus)
1267{
1268 hwloc_obj_t obj = NULL;
1269 while ((obj = hwloc_get_next_bridge(topology, obj)) != NULL) {
1270 if (hwloc_bridge_covers_pcibus(obj, domain, bus)) {
1271 /* found bridge covering this pcibus, make sure it's a hostbridge */
1272 assert(obj->attr->bridge.upstream_type == HWLOC_OBJ_BRIDGE_HOST);
1273 assert(obj->parent->type != HWLOC_OBJ_BRIDGE);
1274 assert(obj->parent->cpuset);
1275 return obj;
1276 }
1277 }
1278 return NULL;
1279}
