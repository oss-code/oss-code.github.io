25/** \defgroup hwlocality_bitmap The bitmap API
26 *
27 * The ::hwloc_bitmap_t type represents a set of objects, typically OS
28 * processors -- which may actually be hardware threads (represented
29 * by ::hwloc_cpuset_t, which is a typedef for ::hwloc_bitmap_t) -- or
30 * memory nodes (represented by ::hwloc_nodeset_t, which is also a
31 * typedef for ::hwloc_bitmap_t).
32 *
33 * <em>Both CPU and node sets are always indexed by OS physical number.</em>
34 *
35 * \note CPU sets and nodesets are described in \ref hwlocality_object_sets.
36 *
37 * A bitmap may be of infinite size.
38 *
39 * \note Several examples of using the bitmap API are available under the
40 * doc/examples/ directory in the source tree.
41 * Regression tests such as tests/hwloc/hwloc_bitmap*.c also make intensive use
42 * of this API.
43 * @{
44 */
45
46
47/** \brief
48 * Set of bits represented as an opaque pointer to an internal bitmap.
49 */
50typedef struct hwloc_bitmap_s * hwloc_bitmap_t;
51/** \brief a non-modifiable ::hwloc_bitmap_t */
52typedef const struct hwloc_bitmap_s * hwloc_const_bitmap_t;

91/** \brief Stringify a bitmap.
92 *

95 * If \p buflen is 0, \p buf may safely be \c NULL.

98 * or that would have been written (not including the ending \\0).

121 * or that would have been written (not including the ending \\0).

252/** \brief Compute the "weight" of bitmap \p bitmap (i.e., number of
253 * indexes that are in the bitmap).
254 *
255 * \return the number of indexes that are in the bitmap.
256 */
257HWLOC_DECLSPEC int hwloc_bitmap_weight(hwloc_const_bitmap_t bitmap) __hwloc_attribute_pure;
258
259/** \brief Loop macro iterating on bitmap \p bitmap
260 *
261 * The loop must start with hwloc_bitmap_foreach_begin() and end
262 * with hwloc_bitmap_foreach_end() followed by a terminating ';'.
263 *
264 * \p index is the loop variable; it should be an unsigned int. The
265 * first iteration will set \p index to the lowest index in the bitmap.
266 * Successive iterations will iterate through, in order, all remaining
267 * indexes set in the bitmap. To be specific: each iteration will return a
268 * value for \p index such that hwloc_bitmap_isset(bitmap, index) is true.
269 *
270 * The assert prevents the loop from being infinite if the bitmap is infinite.
271 *
272 * \hideinitializer
273 */
274#define hwloc_bitmap_foreach_begin(id, bitmap) \
275do { \
276 assert(hwloc_bitmap_weight(bitmap) != -1); \
277 for (id = hwloc_bitmap_first(bitmap); \
278 (unsigned) id != (unsigned) -1; \
279 id = hwloc_bitmap_next(bitmap, id)) {
280
281/** \brief End of loop macro iterating on a bitmap.
282 *
283 * Needs a terminating ';'.
284 *
285 * \sa hwloc_bitmap_foreach_begin()
286 * \hideinitializer
287 */
288#define hwloc_bitmap_foreach_end() \
289 } \
290} while (0)

341/** \brief Compare bitmaps \p bitmap1 and \p bitmap2 using their lowest index.
342 *
343 * Smaller least significant bit is smaller.
344 * The empty bitmap is considered higher than anything.
345 */
346HWLOC_DECLSPEC int hwloc_bitmap_compare_first(hwloc_const_bitmap_t bitmap1, hwloc_const_bitmap_t bitmap2) __hwloc_attribute_pure;
347
348/** \brief Compare bitmaps \p bitmap1 and \p bitmap2 in lexicographic order.
349 *
350 * Lexicographic comparison of bitmaps, starting for their highest indexes.
351 * Compare last indexes first, then second, etc.
352 * The empty bitmap is considered lower than anything.
353 *
354 * \note This is different from the non-existing hwloc_bitmap_compare_last()
355 * which would only compare the highest index of each bitmap.
356 */
357HWLOC_DECLSPEC int hwloc_bitmap_compare(hwloc_const_bitmap_t bitmap1, hwloc_const_bitmap_t bitmap2) __hwloc_attribute_pure;
