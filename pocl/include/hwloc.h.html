97/** \defgroup hwlocality_object_sets Object Sets (hwloc_cpuset_t and hwloc_nodeset_t)
98 *
99 * Hwloc uses bitmaps to represent two distinct kinds of object sets:
100 * CPU sets (::hwloc_cpuset_t) and NUMA node sets (::hwloc_nodeset_t).
101 * These types are both typedefs to a common back end type
102 * (::hwloc_bitmap_t), and therefore all the hwloc bitmap functions
103 * are applicable to both ::hwloc_cpuset_t and ::hwloc_nodeset_t (see
104 * \ref hwlocality_bitmap).
105 *
106 * The rationale for having two different types is that even though
107 * the actions one wants to perform on these types are the same (e.g.,
108 * enable and disable individual items in the set/mask), they're used
109 * in very different contexts: one for specifying which processors to
110 * use and one for specifying which NUMA nodes to use. Hence, the
111 * name difference is really just to reflect the intent of where the
112 * type is used.
113 *
114 * @{
115 */
116
117/** \brief A CPU set is a bitmap whose bits are set according to CPU
118 * physical OS indexes.
119 *
120 * It may be consulted and modified with the bitmap API as any
121 * ::hwloc_bitmap_t (see hwloc/bitmap.h).
122 *
123 * Each bit may be converted into a PU object using
124 * hwloc_get_pu_obj_by_os_index().
125 */
126typedef hwloc_bitmap_t hwloc_cpuset_t;
127/** \brief A non-modifiable ::hwloc_cpuset_t. */
128typedef hwloc_const_bitmap_t hwloc_const_cpuset_t;
129
130/** \brief A node set is a bitmap whose bits are set according to NUMA
131 * memory node physical OS indexes.
132 *
133 * It may be consulted and modified with the bitmap API as any
134 * ::hwloc_bitmap_t (see hwloc/bitmap.h).
135 * Each bit may be converted into a NUMA node object using
136 * hwloc_get_numanode_obj_by_os_index().
137 *
138 * When binding memory on a system without any NUMA node
139 * (when the whole memory is considered as a single memory bank),
140 * the nodeset may be either empty (no memory selected)
141 * or full (whole system memory selected).
142 *
143 * See also \ref hwlocality_helper_nodeset_convert.
144 */
145typedef hwloc_bitmap_t hwloc_nodeset_t;
146/** \brief A non-modifiable ::hwloc_nodeset_t.
147 */
148typedef hwloc_const_bitmap_t hwloc_const_nodeset_t;

290/** \brief Compare the depth of two object types
291 *
292 * Types shouldn't be compared as they are, since newer ones may be added in
293 * the future. This function returns less than, equal to, or greater than zero
294 * respectively if \p type1 objects usually include \p type2 objects, are the
295 * same as \p type2 objects, or are included in \p type2 objects. If the types
296 * can not be compared (because neither is usually contained in the other),
297 * ::HWLOC_TYPE_UNORDERED is returned. Object types containing CPUs can always
298 * be compared (usually, a system contains machines which contain nodes which
299 * contain packages which contain caches, which contain cores, which contain
300 * processors).
301 *
302 * \note ::HWLOC_OBJ_PU will always be the deepest.
303 * \note This does not mean that the actual topology will respect that order:
304 * e.g. as of today cores may also contain caches, and packages may also contain
305 * nodes. This is thus just to be seen as a fallback comparison method.
306 */
307HWLOC_DECLSPEC int hwloc_compare_types (hwloc_obj_type_t type1, hwloc_obj_type_t type2) __hwloc_attribute_const;

310 HWLOC_TYPE_UNORDERED = INT_MAX /**< \brief Value returned by hwloc_compare_types() when types can not be compared. \hideinitializer */

341/** \brief Structure of a topology object
342 *
343 * Applications must not modify any field except hwloc_obj.userdata.
344 */
345struct hwloc_obj {
346 /* physical information */
347 hwloc_obj_type_t type; /**< \brief Type of object */
348 unsigned os_index; /**< \brief OS-provided physical index number.
349 * It is not guaranteed unique across the entire machine,
350 * except for PUs and NUMA nodes.
351 */
352 char *name; /**< \brief Object description if any */
353
354 struct hwloc_obj_memory_s memory; /**< \brief Memory attributes */
355
356 union hwloc_obj_attr_u *attr; /**< \brief Object type-specific Attributes,
357 * may be \c NULL if no attribute value was found */
358
359 /* global position */
360 unsigned depth; /**< \brief Vertical index in the hierarchy.
361 * If the topology is symmetric, this is equal to the
362 * parent depth plus one, and also equal to the number
363 * of parent/child links from the root object to here.
364 */
365 unsigned logical_index; /**< \brief Horizontal index in the whole list of similar objects,
366 * hence guaranteed unique across the entire machine.
367 * Could be a "cousin_rank" since it's the rank within the "cousin" list below
368 */
369 signed os_level; /**< \brief OS-provided physical level, -1 if unknown or meaningless */
370
371 /* cousins are all objects of the same type (and depth) across the entire topology */
372 struct hwloc_obj *next_cousin; /**< \brief Next object of same type and depth */
373 struct hwloc_obj *prev_cousin; /**< \brief Previous object of same type and depth */
374
375 /* children of the same parent are siblings, even if they may have different type and depth */
376 struct hwloc_obj *parent; /**< \brief Parent, \c NULL if root (system object) */
377 unsigned sibling_rank; /**< \brief Index in parent's \c children[] array */
378 struct hwloc_obj *next_sibling; /**< \brief Next object below the same parent */
379 struct hwloc_obj *prev_sibling; /**< \brief Previous object below the same parent */
380
381 /* children array below this object */
382 unsigned arity; /**< \brief Number of children */
383 struct hwloc_obj **children; /**< \brief Children, \c children[0 .. arity -1] */
384 struct hwloc_obj *first_child; /**< \brief First child */
385 struct hwloc_obj *last_child; /**< \brief Last child */
386
387 /* misc */
388 void *userdata; /**< \brief Application-given private data pointer,
389 * initialized to \c NULL, use it as you wish.
390 * See hwloc_topology_set_userdata_export_callback()
391 * if you wish to export this field to XML. */
392
393 /* cpusets and nodesets */
394 hwloc_cpuset_t cpuset; /**< \brief CPUs covered by this object
395 *
396 * This is the set of CPUs for which there are PU objects in the topology
397 * under this object, i.e. which are known to be physically contained in this
398 * object and known how (the children path between this object and the PU
399 * objects).
400 *
401 * If the ::HWLOC_TOPOLOGY_FLAG_WHOLE_SYSTEM configuration flag is set, some of
402 * these CPUs may be offline, or not allowed for binding, see online_cpuset
403 * and allowed_cpuset.
404 *
405 * \note Its value must not be changed, hwloc_bitmap_dup() must be used instead.
406 */
407 hwloc_cpuset_t complete_cpuset; /**< \brief The complete CPU set of logical processors of this object,
408 *
409 * This includes not only the same as the cpuset field, but also the CPUs for
410 * which topology information is unknown or incomplete, and the CPUs that are
411 * ignored when the ::HWLOC_TOPOLOGY_FLAG_WHOLE_SYSTEM flag is not set.
412 * Thus no corresponding PU object may be found in the topology, because the
413 * precise position is undefined. It is however known that it would be somewhere
414 * under this object.
415 *
416 * \note Its value must not be changed, hwloc_bitmap_dup() must be used instead.
417 */
418 hwloc_cpuset_t online_cpuset; /**< \brief The CPU set of online logical processors
419 *
420 * This includes the CPUs contained in this object that are online, i.e. draw
421 * power and can execute threads. It may however not be allowed to bind to
422 * them due to administration rules, see allowed_cpuset.
423 *
424 * \note Its value must not be changed, hwloc_bitmap_dup() must be used instead.
425 */
426 hwloc_cpuset_t allowed_cpuset; /**< \brief The CPU set of allowed logical processors
427 *
428 * This includes the CPUs contained in this object which are allowed for
429 * binding, i.e. passing them to the hwloc binding functions should not return
430 * permission errors. This is usually restricted by administration rules.
431 * Some of them may however be offline so binding to them may still not be
432 * possible, see online_cpuset.
433 *
434 * \note Its value must not be changed, hwloc_bitmap_dup() must be used instead.
435 */
436
437 hwloc_nodeset_t nodeset; /**< \brief NUMA nodes covered by this object or containing this object
438 *
439 * This is the set of NUMA nodes for which there are NODE objects in the
440 * topology under or above this object, i.e. which are known to be physically
441 * contained in this object or containing it and known how (the children path
442 * between this object and the NODE objects).
443 *
444 * In the end, these nodes are those that are close to the current object.
445 *
446 * If the ::HWLOC_TOPOLOGY_FLAG_WHOLE_SYSTEM configuration flag is set, some of
447 * these nodes may not be allowed for allocation, see allowed_nodeset.
448 *
449 * If there are no NUMA nodes in the machine, all the memory is close to this
450 * object, so \p nodeset is full.
451 *
452 * \note Its value must not be changed, hwloc_bitmap_dup() must be used instead.
453 */
454 hwloc_nodeset_t complete_nodeset; /**< \brief The complete NUMA node set of this object,
455 *
456 * This includes not only the same as the nodeset field, but also the NUMA
457 * nodes for which topology information is unknown or incomplete, and the nodes
458 * that are ignored when the ::HWLOC_TOPOLOGY_FLAG_WHOLE_SYSTEM flag is not set.
459 * Thus no corresponding NODE object may be found in the topology, because the
460 * precise position is undefined. It is however known that it would be
461 * somewhere under this object.
462 *
463 * If there are no NUMA nodes in the machine, all the memory is close to this
464 * object, so \p complete_nodeset is full.
465 *
466 * \note Its value must not be changed, hwloc_bitmap_dup() must be used instead.
467 */
468 hwloc_nodeset_t allowed_nodeset; /**< \brief The set of allowed NUMA memory nodes
469 *
470 * This includes the NUMA memory nodes contained in this object which are
471 * allowed for memory allocation, i.e. passing them to NUMA node-directed
472 * memory allocation should not return permission errors. This is usually
473 * restricted by administration rules.
474 *
475 * If there are no NUMA nodes in the machine, all the memory is close to this
476 * object, so \p allowed_nodeset is full.
477 *
478 * \note Its value must not be changed, hwloc_bitmap_dup() must be used instead.
479 */
480
481 struct hwloc_distances_s **distances; /**< \brief Distances between all objects at same depth below this object */
482 unsigned distances_count;
483
484 struct hwloc_obj_info_s *infos; /**< \brief Array of stringified info type=name. */
485 unsigned infos_count; /**< \brief Size of infos array. */
486
487 int symmetric_subtree; /**< \brief Set if the subtree of objects below this object is symmetric,
488 * which means all children and their children have identical subtrees.
489 * If set in the topology root object, lstopo may export the topology
490 * as a synthetic string.
491 */
492};

543/** \brief Distances between objects
544 *
545 * One object may contain a distance structure describing distances
546 * between all its descendants at a given relative depth. If the
547 * containing object is the root object of the topology, then the
548 * distances are available for all objects in the machine.
549 *
550 * If the \p latency pointer is not \c NULL, the pointed array contains
551 * memory latencies (non-zero values), see below.
552 *
553 * In the future, some other types of distances may be considered.
554 * In these cases, \p latency may be \c NULL.
555 */
556struct hwloc_distances_s {
557 unsigned relative_depth; /**< \brief Relative depth of the considered objects
558 * below the object containing this distance information. */
559 unsigned nbobjs; /**< \brief Number of objects considered in the matrix.
560 * It is the number of descendant objects at \p relative_depth
561 * below the containing object.
562 * It corresponds to the result of hwloc_get_nbobjs_inside_cpuset_by_depth(). */
563
564 float *latency; /**< \brief Matrix of latencies between objects, stored as a one-dimension array.
565 * May be \c NULL if the distances considered here are not latencies.
566 *
567 * Unless defined by the user, this currently contains latencies
568 * between NUMA nodes (as reported in the System Locality Distance Information Table
569 * (SLIT) in the ACPI specification), which may or may not be accurate.
570 * It corresponds to the latency for accessing the memory of one node
571 * from a core in another node.
572 *
573 * Values are normalized to get 1.0 as the minimal value in the matrix.
574 * Latency from i-th to j-th object is stored in slot i*nbobjs+j.
575 */
576 float latency_max; /**< \brief The maximal value in the latency matrix. */
577 float latency_base; /**< \brief The multiplier that should be applied to latency matrix
578 * to retrieve the original OS-provided latencies.
579 * Usually 10 on Linux since ACPI SLIT uses 10 for local latency.
580 */
581};

1334/** \defgroup hwlocality_cpubinding CPU binding
1335 *
1336 * It is often useful to call hwloc_bitmap_singlify() first so that a single CPU
1337 * remains in the set. This way, the process will not even migrate between
1338 * different CPUs inside the given set.
1339 * Some operating systems also only support that kind of binding.
1340 *
1341 * Some operating systems do not provide all hwloc-supported
1342 * mechanisms to bind processes, threads, etc.
1343 * hwloc_topology_get_support() may be used to query about the actual CPU
1344 * binding support in the currently used operating system.
1345 *
1346 * When the requested binding operation is not available and the
1347 * ::HWLOC_CPUBIND_STRICT flag was passed, the function returns -1.
1348 * \p errno is set to \c ENOSYS when it is not possible to bind the requested kind of object
1349 * processes/threads. errno is set to \c EXDEV when the requested cpuset
1350 * can not be enforced (e.g. some systems only allow one CPU, and some
1351 * other systems only allow one NUMA node).
1352 *
1353 * If ::HWLOC_CPUBIND_STRICT was not passed, the function may fail as well,
1354 * or the operating system may use a slightly different operation
1355 * (with side-effects, smaller binding set, etc.)
1356 * when the requested operation is not exactly supported.
1357 *
1358 * The most portable version that should be preferred over the others,
1359 * whenever possible, is the following one which just binds the current program,
1360 * assuming it is single-threaded:
1361 *
1362 * \code
1363 * hwloc_set_cpubind(topology, set, 0),
1364 * \endcode
1365 *
1366 * If the program may be multithreaded, the following one should be preferred
1367 * to only bind the current thread:
1368 *
1369 * \code
1370 * hwloc_set_cpubind(topology, set, HWLOC_CPUBIND_THREAD),
1371 * \endcode
1372 *
1373 * \sa Some example codes are available under doc/examples/ in the source tree.
1374 *
1375 * \note To unbind, just call the binding function with either a full cpuset or
1376 * a cpuset equal to the system cpuset.
1377 *
1378 * \note On some operating systems, CPU binding may have effects on memory binding, see
1379 * ::HWLOC_CPUBIND_NOMEMBIND
1380 *
1381 * \note Running lstopo --top or hwloc-ps can be a very convenient tool to check
1382 * how binding actually happened.
1383 * @{
1384 */

1386/** \brief Process/Thread binding flags.
1387 *
1388 * These bit flags can be used to refine the binding policy.
1389 *
1390 * The default (0) is to bind the current process, assumed to be
1391 * single-threaded, in a non-strict way. This is the most portable
1392 * way to bind as all operating systems usually provide it.
1393 *
1394 * \note Not all systems support all kinds of binding. See the
1395 * "Detailed Description" section of \ref hwlocality_cpubinding for a
1396 * description of errors that can occur.
1397 */

1399 /** \brief Bind all threads of the current (possibly) multithreaded process.
1400 * \hideinitializer */

1402
1403 /** \brief Bind current thread of current process.
1404 * \hideinitializer */
1405 HWLOC_CPUBIND_THREAD = (1<<1),
1406
1407 /** \brief Request for strict binding from the OS.
1408 *
1409 * By default, when the designated CPUs are all busy while other
1410 * CPUs are idle, operating systems may execute the thread/process
1411 * on those other CPUs instead of the designated CPUs, to let them
1412 * progress anyway. Strict binding means that the thread/process
1413 * will _never_ execute on other cpus than the designated CPUs, even
1414 * when those are busy with other tasks and other CPUs are idle.
1415 *
1416 * \note Depending on the operating system, strict binding may not
1417 * be possible (e.g., the OS does not implement it) or not allowed
1418 * (e.g., for an administrative reasons), and the function will fail

1425 *

1427 * thread.
1428 * \hideinitializer

1438 * memory. This will however reduce the support of CPU bindings,

1549/** \defgroup hwlocality_membinding Memory binding
1550 *
1551 * Memory binding can be done three ways:
1552 *
1553 * - explicit memory allocation thanks to hwloc_alloc_membind() and friends:
1554 * the binding will have effect on the memory allocated by these functions.
1555 * - implicit memory binding through binding policy: hwloc_set_membind() and
1556 * friends only define the current policy of the process, which will be
1557 * applied to the subsequent calls to malloc() and friends.
1558 * - migration of existing memory ranges, thanks to hwloc_set_area_membind()
1559 * and friends, which move already-allocated data.
1560 *
1561 * Not all operating systems support all three ways.
1562 * hwloc_topology_get_support() may be used to query about the actual memory
1563 * binding support in the currently used operating system.
1564 *
1565 * When the requested binding operation is not available and the
1566 * ::HWLOC_MEMBIND_STRICT flag was passed, the function returns -1.
1567 * \p errno will be set to \c ENOSYS when the system does support
1568 * the specified action or policy
1569 * (e.g., some systems only allow binding memory on a per-thread
1570 * basis, whereas other systems only allow binding memory for all
1571 * threads in a process).
1572 * \p errno will be set to EXDEV when the requested cpuset can not be enforced
1573 * (e.g., some systems only allow binding memory to a single NUMA node).
1574 *
1575 * If ::HWLOC_MEMBIND_STRICT was not passed, the function may fail as well,
1576 * or the operating system may use a slightly different operation
1577 * (with side-effects, smaller binding set, etc.)
1578 * when the requested operation is not exactly supported.
1579 *
1580 * The most portable form that should be preferred over the others
1581 * whenever possible is as follows.
1582 * It allocates some memory hopefully bound to the specified set.
1583 * To do so, hwloc will possibly have to change the current memory
1584 * binding policy in order to actually get the memory bound, if the OS
1585 * does not provide any other way to simply allocate bound memory
1586 * without changing the policy for all allocations. That is the
1587 * difference with hwloc_alloc_membind(), which will never change the
1588 * current memory binding policy.
1589 *
1590 * \code
1591 * hwloc_alloc_membind_policy(topology, size, set,
1592 * HWLOC_MEMBIND_BIND, 0);
1593 * \endcode
1594 *
1595 * Each hwloc memory binding function is available in two forms: one
1596 * that takes a CPU set argument and another that takes a NUMA memory
1597 * node set argument (see \ref hwlocality_object_sets and \ref
1598 * hwlocality_bitmap for a discussion of CPU sets and NUMA memory node
1599 * sets). The names of the latter form end with _nodeset. It is also
1600 * possible to convert between CPU set and node set using
1601 * hwloc_cpuset_to_nodeset() or hwloc_cpuset_from_nodeset().
1602 *
1603 * \sa Some example codes are available under doc/examples/ in the source tree.
1604 *
1605 * \note On some operating systems, memory binding affects the CPU
1606 * binding; see ::HWLOC_MEMBIND_NOCPUBIND
1607 * @{
1608 */

1617 * binding policy support in the currently used operating system.
1618 * See the "Detailed Description" section of \ref hwlocality_membinding

1626 * \hideinitializer */
