31extern "C" {
32#endif

34#define TARGETED_WGX_ALIGN 128
35#define TARGETED_WGY_ALIGN 64
36#define TARGETED_WGZ_ALIGN 32
37
38 int
39 exec_matadd_kernel (cl_context context, cl_device_id device,
40 cl_command_queue cmd_queue, cl_program program,
41 int sizeX, int sizeY, int sizeZ, cl_float *srcA,
42 cl_float *srcB, cl_float *dst)
43 {
44 cl_kernel kernel = NULL;
45 cl_mem memobjs[3] = { 0, 0, 0 };
46
47 int dataSize = sizeX * sizeY * sizeZ;
48 /* Align the grid size so we get nicely aligned work groups.
49 The kernel will check for data boundaries. */
50#define ALIGN_TO(VAL, ALIGN) \
51 VAL % ALIGN == 0 ? VAL : (((VAL / ALIGN) + 1) * ALIGN)
52 size_t global_work_size[] = { ALIGN_TO (sizeX, TARGETED_WGX_ALIGN),
53 ALIGN_TO (sizeY, TARGETED_WGY_ALIGN),
54 ALIGN_TO (sizeZ, TARGETED_WGZ_ALIGN) };
55 cl_int err = CL_SUCCESS;
56 int i;
57
58 poclu_bswap_cl_float_array (device, (cl_float *)srcA,
59 sizeof (cl_float) * dataSize);
60 poclu_bswap_cl_float_array (device, (cl_float *)srcB,
61 sizeof (cl_float) * dataSize);
62
63 memobjs[0]
64 = clCreateBuffer (context, CL_MEM_READ_ONLY | CL_MEM_COPY_HOST_PTR,
65 sizeof (cl_float) * dataSize, srcA, &err);
66 CHECK_CL_ERROR2 (err);
67
68 memobjs[1]
69 = clCreateBuffer (context, CL_MEM_READ_ONLY | CL_MEM_COPY_HOST_PTR,
70 sizeof (cl_float) * dataSize, srcB, &err);
71 CHECK_CL_ERROR2 (err);
72
73 memobjs[2] = clCreateBuffer (context, CL_MEM_READ_WRITE,
74 sizeof (cl_float) * dataSize, NULL, &err);
75 CHECK_CL_ERROR2 (err);
76
77 kernel = clCreateKernel (program, "boxadd", NULL);
78 CHECK_CL_ERROR2 (err);
79
80 err = clSetKernelArg (kernel, 0, sizeof (cl_mem), (void *)&memobjs[0]);
81 CHECK_CL_ERROR2 (err);
82
83 err = clSetKernelArg (kernel, 1, sizeof (cl_mem), (void *)&memobjs[1]);
84 CHECK_CL_ERROR2 (err);
85
86 err = clSetKernelArg (kernel, 2, sizeof (cl_mem), (void *)&memobjs[2]);
87 CHECK_CL_ERROR2 (err);
88
89 err = clSetKernelArg (kernel, 3, sizeof (cl_int), &sizeX);
90 CHECK_CL_ERROR2 (err);
91
92 err = clSetKernelArg (kernel, 4, sizeof (cl_int), &sizeY);
93 CHECK_CL_ERROR2 (err);
94
95 err = clSetKernelArg (kernel, 5, sizeof (cl_int), &sizeZ);
96 CHECK_CL_ERROR2 (err);
97
98 err = clEnqueueNDRangeKernel (cmd_queue, kernel, 3, NULL, global_work_size,
99 NULL, 0, NULL, NULL);
100 CHECK_CL_ERROR2 (err);
101
102 err = clEnqueueReadBuffer (cmd_queue, memobjs[2], CL_TRUE, 0,
103 dataSize * sizeof (cl_float), dst, 0, NULL,
104 NULL);
105 CHECK_CL_ERROR2 (err);
106
107 poclu_bswap_cl_float_array (device, (cl_float *)dst,
108 sizeof (cl_float) * dataSize);
109 poclu_bswap_cl_float_array (device, (cl_float *)srcA,
110 sizeof (cl_float) * dataSize);
111 poclu_bswap_cl_float_array (device, (cl_float *)srcB,
112 sizeof (cl_float) * dataSize);
113
114 ERROR:
115 clReleaseMemObject (memobjs[0]);
116 clReleaseMemObject (memobjs[1]);
117 clReleaseMemObject (memobjs[2]);
118 clReleaseKernel (kernel);
119 return err;
120 }
