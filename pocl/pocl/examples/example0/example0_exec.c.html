12 int
13 exec_integer_mad_kernel (cl_context context, cl_device_id device,
14 cl_command_queue cmd_queue, cl_program program,
15 unsigned n, cl_uint *srcA, cl_uint *srcB,
16 cl_uint *dst)
17 {
18 cl_kernel kernel = NULL;
19 cl_mem memobjs[3] = { 0, 0, 0 };
20 size_t global_work_size[1];
21 size_t local_work_size[1];
22 cl_int err = CL_SUCCESS;
23 int i;
24
25 poclu_bswap_cl_int_array (device, (cl_uint *)srcA, n);
26 poclu_bswap_cl_int_array (device, (cl_uint *)srcB, n);
27
28 memobjs[0] = clCreateBuffer (context, CL_MEM_READ_ONLY,
29 sizeof (cl_uint) * n, NULL, &err);
30 CHECK_CL_ERROR2 (err);
31
32 memobjs[1] = clCreateBuffer (context, CL_MEM_READ_ONLY,
33 sizeof (cl_uint) * n, NULL, &err);
34 CHECK_CL_ERROR2 (err);
35
36 memobjs[2] = clCreateBuffer (context, CL_MEM_READ_WRITE,
37 sizeof (cl_uint) * n, NULL, &err);
38 CHECK_CL_ERROR2 (err);
39
40 err = clEnqueueWriteBuffer (cmd_queue, memobjs[0], CL_TRUE, 0,
41 n * sizeof (cl_uint), srcA, 0, NULL, NULL);
42 CHECK_CL_ERROR2 (err);
43
44 err = clEnqueueWriteBuffer (cmd_queue, memobjs[1], CL_TRUE, 0,
45 n * sizeof (cl_uint), srcB, 0, NULL, NULL);
46 CHECK_CL_ERROR2 (err);
47
48 kernel = clCreateKernel (program, "integer_mad", NULL);
49 CHECK_CL_ERROR2 (err);
50
51 err = clSetKernelArg (kernel, 0, sizeof (cl_mem), (void *)&memobjs[0]);
52 CHECK_CL_ERROR2 (err);
53
54 err = clSetKernelArg (kernel, 1, sizeof (cl_mem), (void *)&memobjs[1]);
55 CHECK_CL_ERROR2 (err);
56
57 err = clSetKernelArg (kernel, 2, sizeof (cl_mem), (void *)&memobjs[2]);
58 CHECK_CL_ERROR2 (err);
59
60 global_work_size[0] = n;
61 local_work_size[0] = 8;
62
63 err = clEnqueueNDRangeKernel (cmd_queue, kernel, 1, NULL, global_work_size,
64 local_work_size, 0, NULL, NULL);
65 CHECK_CL_ERROR2 (err);
66
67 err = clEnqueueReadBuffer (cmd_queue, memobjs[2], CL_TRUE, 0,
68 n * sizeof (cl_uint), dst, 0, NULL, NULL);
69 CHECK_CL_ERROR2 (err);
70
71 poclu_bswap_cl_int_array (device, (cl_uint *)dst, n);
72 poclu_bswap_cl_int_array (device, (cl_uint *)srcA, n);
73 poclu_bswap_cl_int_array (device, (cl_uint *)srcB, n);
74
75 ERROR:
76 clReleaseMemObject (memobjs[0]);
77 clReleaseMemObject (memobjs[1]);
78 clReleaseMemObject (memobjs[2]);
79 clReleaseKernel (kernel);
80 return err;
81 }
