22#define GRID_GRANULARITY 2

24typedef struct grid_t {
25 cl_double dt; // time step
26 cl_double dx, dy, dz; // resolution
27 cl_int ai, aj, ak; // allocated size
28 cl_int ni, nj, nk; // used size
29} grid_t;
30
31static int initialised = 0;
32static cl_context context;
33static cl_command_queue cmd_queue;
34static cl_program program;
35static cl_kernel kernel;
36
37
38#ifdef _CL_DISABLE_DOUBLE
39#error Scalarwave test requires cl_khr_fp64 support.
40#endif
41
42int
43exec_scalarwave_kernel(char const *const program_source,
44 cl_double *const phi,
45 cl_double const *const phi_p,
46 cl_double const *const phi_p_p,
47 grid_t const *const grid)
48{
49
50 if (!initialised) {
51 initialised = 1;
52
53 context = poclu_create_any_context();
54 if (!context) return -1;
55
56 size_t ndevices;
57 clGetContextInfo(context, CL_CONTEXT_DEVICES, 0, NULL, &ndevices);
58 ndevices /= sizeof(cl_device_id);
59 cl_device_id *devices = (cl_device_id*)malloc(ndevices * sizeof(cl_device_id));
60 clGetContextInfo(context, CL_CONTEXT_DEVICES,
61 ndevices*sizeof(cl_device_id), devices, NULL);
62
63 cmd_queue =
64 clCreateCommandQueue(context, devices[0], 0, NULL);
65 if (!cmd_queue) return -1;
66
67 program =
68 clCreateProgramWithSource(context, 1, (const char**)&program_source,
69 NULL, NULL);
70 if (!program) return -1;
71
72 int ierr;
73 ierr = clBuildProgram(program, 0, NULL, NULL, NULL, NULL);
74 if (ierr) return -1;
75
76 kernel = clCreateKernel(program, "scalarwave", NULL);
77 if (!kernel) return -1;
78
79 free (devices);
80 }
81
82 size_t const npoints = grid->ai * grid->aj * grid->ak;
83 cl_mem const mem_phi =
84 clCreateBuffer(context, 0,
85 npoints*sizeof(*phi), NULL, NULL);
86 if (!mem_phi) return -1;
87 cl_mem const mem_phi_p =
88 clCreateBuffer(context, CL_MEM_COPY_HOST_PTR,
89 npoints*sizeof(*phi_p), (cl_double*)phi_p, NULL);
90 if (!mem_phi_p) return -1;
91 cl_mem const mem_phi_p_p =
92 clCreateBuffer(context, CL_MEM_COPY_HOST_PTR,
93 npoints*sizeof(*phi_p_p), (cl_double*)phi_p_p, NULL);
94 if (!mem_phi_p_p) return -1;
95 cl_mem const mem_grid =
96 clCreateBuffer(context, CL_MEM_READ_ONLY | CL_MEM_COPY_HOST_PTR,
97 sizeof(*grid), (grid_t*)grid, NULL);
98 if (!mem_grid) return -1;
99
100 int ierr;
101 ierr = clSetKernelArg(kernel, 0, sizeof(cl_mem), &mem_phi);
102 if (ierr) return -1;
103 ierr = clSetKernelArg(kernel, 1, sizeof(cl_mem), &mem_phi_p);
104 if (ierr) return -1;
105 ierr = clSetKernelArg(kernel, 2, sizeof(cl_mem), &mem_phi_p_p);
106 if (ierr) return -1;
107 ierr = clSetKernelArg(kernel, 3, sizeof(cl_mem), &mem_grid);
108 if (ierr) return -1;
109
110 size_t const global_work_size[3] =
111 {grid->ai, grid->aj, grid->ak};
112 size_t const local_work_size[3] =
113 {GRID_GRANULARITY, GRID_GRANULARITY, GRID_GRANULARITY};
114
115 ierr = clEnqueueNDRangeKernel(cmd_queue, kernel, 3, NULL,
116 global_work_size, local_work_size,
117 0, NULL, NULL);
118 if (ierr) return -1;
119
120 ierr = clFinish(cmd_queue);
121 if (ierr) return -1;
122
123 ierr = clEnqueueReadBuffer(cmd_queue, mem_phi, CL_TRUE,
124 0, npoints*sizeof(*phi), phi,
125 0, NULL, NULL);
126 if (ierr) return -1;
127
128 clReleaseMemObject(mem_phi);
129 clReleaseMemObject(mem_phi_p);
130 clReleaseMemObject(mem_phi_p_p);
131 clReleaseMemObject(mem_grid);
132
133 return 0;
134}

138#define ALPHA 0.5 // CFL factor
139#define NT 4 // time steps
140#define NX 17 // grid size
141
142
143
144// Round up to next multiple of GRID_GRANULARITY
145static int roundup(int const nx)
146{
147 return (nx + GRID_GRANULARITY-1) / GRID_GRANULARITY * GRID_GRANULARITY;
148}

150
151int
152main(void)
153{
154 FILE *const source_file = fopen(SRCDIR "/examples/scalarwave/scalarwave.cl", "r");
155 assert(source_file != NULL && "scalarwave.cl not found!");
156
157 fseek(source_file, 0, SEEK_END);
158 size_t const source_size = ftell(source_file);
159 fseek(source_file, 0, SEEK_SET);
160
161 char *source = (char*)malloc(source_size + 1);
162 fread(source, source_size, 1, source_file);
163 source[source_size] = '\0';
164
165 fclose(source_file);
166
167 grid_t grid;
168 grid.dt = ALPHA/(NX-1);
169 grid.dx = grid.dy = grid.dz = 1.0/(NX-1);
170 grid.ai = grid.aj = grid.ak = roundup(NX);
171 grid.ni = grid.nj = grid.nk = NX;
172
173 cl_double *restrict phi = (cl_double*)
174 malloc (grid.ai*grid.aj*grid.ak * sizeof *phi );
175 cl_double *restrict phi_p = (cl_double*)
176 malloc (grid.ai*grid.aj*grid.ak * sizeof *phi_p );
177 cl_double *restrict phi_p_p = (cl_double*)
178 malloc (grid.ai*grid.aj*grid.ak * sizeof *phi_p_p);
179
180 // Set up initial data (TODO: do this on the device instead)
181 printf ("Initial condition: t=%g\n", 0.0);
182 double const kx = M_PI;
183 double const ky = M_PI;
184 double const kz = M_PI;
185 double const omega = sqrt(pow(kx,2)+pow(ky,2)+pow(kz,2));
186 for (int k = 0; k < NX; ++k) {
187 for (int j = 0; j < NX; ++j) {
188 for (int i = 0; i < NX; ++i) {
189 double const t0 = 0.0;
190 double const t1 = -grid.dt;
191 double const x = i*grid.dx;
192 double const y = j*grid.dy;
193 double const z = k*grid.dz;
194 int const ind3d = i+grid.ai*(j+grid.aj*k);
195 phi [ind3d] = sin(kx*x) * sin(ky*y) * sin(kz*z) * cos(omega*t0);
196 phi_p[ind3d] = sin(kx*x) * sin(ky*y) * sin(kz*z) * cos(omega*t1);
197 }
198 }
199 }
200
201 // Take some time steps
202 for (int n=0; n<NT; ++n) {
203 printf ("Time step %d: t=%g\n", n+1, (n+1)*grid.dt);
204
205 // Cycle time levels
206 {
207 cl_double *tmp = phi_p_p;
208 phi_p_p = phi_p;
209 phi_p = phi;
210 phi = tmp;
211 }
212
213 // TODO: We allocate the buffers each time, which is slow. But
214 // then, we only want to test correctness, not performance. (Yet?)
215 int const ierr =
216 exec_scalarwave_kernel (source, phi, phi_p, phi_p_p, &grid);
217 assert(!ierr);
218
219 } // for n
220
221 clReleaseKernel (kernel);
222 clReleaseProgram (program);
223 clReleaseCommandQueue (cmd_queue);
224 cl_platform_id pocl;
225 clGetPlatformIDs (1, &pocl, NULL);
226 clUnloadPlatformCompiler (pocl);
227 clReleaseContext (context);
228
229 for (int i=0; i<NX; ++i) {
230 int const j = i;
231 int const k = i;
232 double const x = grid.dx*i;
233 double const y = grid.dy*j;
234 double const z = grid.dz*k;
235 int const ind3d = i+grid.ai*(j+grid.aj*k);
236
237 printf ("phi(%-8g,%-8g,%-8g) = %g\n", x,y,z, phi[ind3d]);
238 }
239
240 printf ("Done.\n");
241
242 free (phi);
243 free (phi_p);
244 free (phi_p_p);
245 free(source);
246
247 return 0;
248}
