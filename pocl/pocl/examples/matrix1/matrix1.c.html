27#include <CL/opencl.h>
28#include <assert.h>
29#include <errno.h>
30#include <math.h>

37#define min(X, Y) X < Y ? X : Y

39int
40exec_matrix_kernel (cl_context context, cl_device_id device,
41 cl_command_queue cmd_queue, cl_program program, cl_uint n,
42 cl_float *srcA, cl_float *srcB, cl_float *dst,
43 const char *kernel_name, const size_t *global_work_size,
44 const size_t *local_work_size, int transpose)
45{
46 cl_kernel kernel = NULL;
47 cl_mem memobjs[3] = { 0, 0, 0 };
48 cl_int err = CL_SUCCESS;
49 cl_mem temp = NULL;
50
51 poclu_bswap_cl_float_array (device, (cl_float *)srcA, n * n);
52 poclu_bswap_cl_float_array (device, (cl_float *)srcB, n * n);
53
54 memobjs[0]
55 = clCreateBuffer (context, CL_MEM_READ_ONLY | CL_MEM_COPY_HOST_PTR,
56 sizeof (cl_float) * n * n, srcA, &err);
57 CHECK_CL_ERROR2 (err);
58
59 if (transpose)
60 {
61 temp = clCreateBuffer (context, CL_MEM_READ_ONLY | CL_MEM_COPY_HOST_PTR,
62 sizeof (cl_float) * n * n, srcB, &err);
63 CHECK_CL_ERROR2 (err);
64
65 memobjs[1] = clCreateBuffer (context, CL_MEM_READ_WRITE,
66 sizeof (cl_float) * n * n, NULL, &err);
67 CHECK_CL_ERROR2 (err);
68
69 kernel = clCreateKernel (program, "transpose", NULL);
70 CHECK_CL_ERROR2 (err);
71
72 err = clSetKernelArg (kernel, 0, sizeof (cl_uint), (void *)&n);
73 CHECK_CL_ERROR2 (err);
74
75 err = clSetKernelArg (kernel, 1, sizeof (cl_uint), (void *)&n);
76 CHECK_CL_ERROR2 (err);
77
78 err = clSetKernelArg (kernel, 2, sizeof (cl_mem), (void *)&temp);
79 CHECK_CL_ERROR2 (err);
80
81 err = clSetKernelArg (kernel, 3, sizeof (cl_mem), (void *)&memobjs[1]);
82 CHECK_CL_ERROR2 (err);
83
84 cl_event ev;
85 size_t global[2] = { n, n };
86 size_t local[2] = { 8, 8 };
87 err = clEnqueueNDRangeKernel (cmd_queue, kernel, 2, NULL, global, local,
88 0, NULL, &ev);
89 CHECK_CL_ERROR2 (err);
90 }
91 else
92 {
93 memobjs[1]
94 = clCreateBuffer (context, CL_MEM_READ_ONLY | CL_MEM_COPY_HOST_PTR,
95 sizeof (cl_float) * n * n, srcB, &err);
96 CHECK_CL_ERROR2 (err);
97 }
98
99 memobjs[2] = clCreateBuffer (context, CL_MEM_WRITE_ONLY,
100 sizeof (cl_float) * n * n, NULL, &err);
101 CHECK_CL_ERROR2 (err);
102
103 kernel = clCreateKernel (program, kernel_name, NULL);
104 CHECK_CL_ERROR2 (err);
105
106 err = clSetKernelArg (kernel, 0, sizeof (cl_mem), (void *)&memobjs[0]);
107 CHECK_CL_ERROR2 (err);
108
109 err = clSetKernelArg (kernel, 1, sizeof (cl_mem), (void *)&memobjs[1]);
110 CHECK_CL_ERROR2 (err);
111
112 err = clSetKernelArg (kernel, 2, sizeof (cl_mem), (void *)&memobjs[2]);
113 CHECK_CL_ERROR2 (err);
114
115 err = clSetKernelArg (kernel, 3, sizeof (cl_uint), (void *)&n);
116 CHECK_CL_ERROR2 (err);
117
118 err = clSetKernelArg (kernel, 4, sizeof (cl_uint), (void *)&n);
119 CHECK_CL_ERROR2 (err);
120
121 err = clSetKernelArg (kernel, 5, sizeof (cl_uint), (void *)&n);
122 CHECK_CL_ERROR2 (err);
123
124 cl_event ev;
125
126 printf ("gws: %lu %lu lws: %lu %lu\n", global_work_size[0],
127 global_work_size[1], local_work_size[0], local_work_size[1]);
128 err = clEnqueueNDRangeKernel (cmd_queue, kernel, 2, NULL, global_work_size,
129 local_work_size, 0, NULL, &ev);
130 CHECK_CL_ERROR2 (err);
131
132 err = clEnqueueReadBuffer (cmd_queue, memobjs[2], CL_TRUE, 0,
133 n * n * sizeof (cl_float), dst, 0, NULL, NULL);
134 CHECK_CL_ERROR2 (err);
135
136 cl_ulong startTime;
137 cl_ulong endTime;
138
139 // Get kernel profiling info
140 err = clGetEventProfilingInfo (ev, CL_PROFILING_COMMAND_START,
141 sizeof (cl_ulong), &startTime, 0);
142 CHECK_CL_ERROR2 (err);
143 err = clGetEventProfilingInfo (ev, CL_PROFILING_COMMAND_END,
144 sizeof (cl_ulong), &endTime, 0);
145 CHECK_CL_ERROR2 (err);
146
147 cl_ulong ev_nsec = endTime - startTime;
148 double nsec = (double)ev_nsec;
149 double msec = ev_nsec / 1000000.0;
150 size_t f = 2UL * n * n * n;
151 double flops = (double)f;
152 double gflops = f / 1000000000.0;
153 double perf = flops / nsec;
154
155 printf ("Performance: %lf GFLOPS/s | Time: %lf msec | Total Ops to "
156 "execute: %lf G \n",
157 perf, msec, gflops);
158 clReleaseEvent (ev);
159
160 poclu_bswap_cl_float_array (device, (cl_float *)dst, n * n);
161
162ERROR:
163 if (temp)
164 clReleaseMemObject (temp);
165 clReleaseMemObject (memobjs[0]);
166 clReleaseMemObject (memobjs[1]);
167 clReleaseMemObject (memobjs[2]);
168 clReleaseKernel (kernel);
169 return err;
170}

172int
173main (int argc, char **argv)
174{
175 cl_float *srcA = NULL;
176 cl_float *srcB = NULL;
177 cl_float *dst = NULL;
178 long *sums = NULL;
179 long i, j, sum;
180 int is_binary = 0;
181 int spir = 0;
182 int spirv = 0;
183 int poclbin = 0;
184 int use_locals = 0;
185 int use_2d_reg_block = 0;
186 int use_fma = 0;
187 cl_int err;
188
189 cl_context context = NULL;
190 cl_device_id device = NULL;
191 cl_platform_id platform = NULL;
192 cl_command_queue queue = NULL;
193 cl_program program = NULL;
194 size_t max_wg_size = 0;
195 cl_ulong local_mem_size = 0;
196 unsigned local_wg = 0;
197 long matrix_size = 0;
198 const char *explicit_binary_path = NULL;
199
200 printf ("argc: %i \n", argc);
201
202 err = poclu_get_any_device2 (&context, &device, &queue, &platform);
203 CHECK_OPENCL_ERROR_IN ("clCreateContext");
204
205 if (argc > 1)
206 {
207 errno = 0;
208 matrix_size = strtol (argv[1], NULL, 10);
209 }
210
211 if (matrix_size < 64 || matrix_size > 65536 || errno)
212 {
213 printf ("USAGE: matrix1 MATRIX_SIZE [options] [path-to-binary]\n"
214 "Matrix width must be power-of-4, in [64, 65536] range\n");
215 return EXIT_FAILURE;
216 }
217
218 /**************************************************************************/
219
220 long matrix_2d_size = matrix_size * matrix_size;
221
222 int arg_i = 2;
223 while ((argc > arg_i) && (argv[arg_i][0] == '-'))
224 {
225 if (argv[arg_i][1] == 's')
226 spir = 1;
227 if (argv[arg_i][1] == 'v')
228 spirv = 1;
229 if (argv[arg_i][1] == 'b')
230 poclbin = 1;
231 if (argv[arg_i][1] == 'l')
232 use_locals = 1;
233 if (argv[arg_i][1] == 'r')
234 use_2d_reg_block = 1;
235 if (argv[arg_i][1] == 'f')
236 use_fma = 1;
237 ++arg_i;
238 }
239
240 if ((spir + spirv + poclbin) > 1)
241 {
242 printf ("only one type of binary can be specified \n");
243 return EXIT_FAILURE;
244 }
245
246 is_binary = (spir + spirv + poclbin);
247
248 if ((argc > arg_i) && (argv[arg_i][0] != '-'))
249 explicit_binary_path = argv[arg_i];
250
251 printf ("OPTIONS: SPIR %i SPIR-V %i POCLBIN %i USE_LOCALS %i USE_REGS %i "
252 "USE_FMA %i \n"
253 "EXPLICIT BINARY: %s \n",
254 spir, spirv, poclbin, use_locals, use_2d_reg_block, use_fma,
255 explicit_binary_path);
256
257 if (explicit_binary_path && ((spir + spirv + poclbin) == 0))
258 {
259 printf ("explicit binary given, but no binary type specified!\n");
260 return EXIT_FAILURE;
261 }
262
263 /**************************************************************************/
264
265 err = clGetDeviceInfo (device, CL_DEVICE_MAX_WORK_GROUP_SIZE,
266 sizeof (max_wg_size), &max_wg_size, NULL);
267 CHECK_OPENCL_ERROR_IN ("clGetDeviceInfo");
268
269 err = clGetDeviceInfo (device, CL_DEVICE_LOCAL_MEM_SIZE,
270 sizeof (local_mem_size), &local_mem_size, NULL);
271 CHECK_OPENCL_ERROR_IN ("clGetDeviceInfo");
272
273 /**************************************************************************/
274
275 local_wg = 1;
276 while (local_wg * local_wg < max_wg_size)
277 local_wg <<= 1;
278
279 if (local_wg * local_wg > max_wg_size)
280 local_wg >>= 1;
281
282 assert (local_wg > 0);
283
284 if (is_binary)
285 {
286 use_fma = 0;
287
288 local_wg = 32;
289 printf ("Binary local_wg: %u \n", local_wg);
290 if (local_wg * local_wg > max_wg_size)
291 {
292 printf ("local_wg exceeds this device's capabilities.\n");
293 return 0;
294 }
295 }
296 else
297 {
298 printf ("Autodetected local_wg: %u max wg size: %lu\n", local_wg,
299 max_wg_size);
300 }
301
302 /**************************************************************************/
303
304 size_t global_work_size[2];
305 size_t local_work_size[2];
306 const char *kernel_name = NULL;
307 const char *fma = use_fma ? "-DFMA " : "";
308 char extra_opts[1024];
309 extra_opts[0] = 0;
310
311 /* simple */
312 if (!use_locals && !use_2d_reg_block)
313 {
314 printf ("Using simplest kernel (myGEMM2)\n");
315 global_work_size[0] = global_work_size[1] = matrix_size;
316 local_work_size[0] = local_work_size[1] = local_wg; // TODO
317 kernel_name = "myGEMM2";
318 if (use_fma)
319 strcpy (extra_opts, fma);
320 }
321
322 /* myGEMM4 */
323 if (use_locals)
324 {
325 kernel_name = "myGEMM4";
326
327 printf ("Using locals (myGEMM4)\n");
328
329 if (is_binary)
330 {
331 if (local_wg * local_wg * 8 > local_mem_size)
332 {
333 printf ("required local memory exceeds this device's "
334 "capabilities.\n");
335 return 0;
336 }
337 }
338 else
339 {
340 /* 8 = 2x array: float[local_wg][local_wg] */
341 while (local_wg * local_wg * 8 > local_mem_size)
342 local_wg >>= 1;
343 }
344
345#define WPT (local_wg / 4)
346 global_work_size[0] = matrix_size;
347 global_work_size[1] = matrix_size / WPT;
348 local_work_size[0] = local_wg;
349 local_work_size[1] = 4;
350#undef WPT
351
352 snprintf (extra_opts, 1024, "%s-DMYGEMM4 -DLOCAL_SIZE=%u", fma,
353 local_wg);
354 printf ("Using local group size: [%u, %u]\n", local_wg, 4);
355 }
356
357 /* myGEMM6 */
358 if (use_2d_reg_block)
359 {
360 kernel_name = "myGEMM6";
361
362 printf ("using 2d reg block (myGEMM6)\n");
363
364#define WPT 8
365 /* const size_t global[2] = { M/WPTM, N/WPTN }; */
366 /* const size_t local[2] = { TSM/WPTM, TSN/WPTN }; */
367 global_work_size[0] = global_work_size[1] = matrix_size / WPT;
368 local_wg >>= 2;
369 local_work_size[0] = local_work_size[1] = local_wg;
370 size_t TSM = local_wg * WPT;
371 size_t TSK = local_wg;
372
373 if (is_binary)
374 {
375 if ((TSM * TSK * 2 * 4) > local_mem_size)
376 {
377 printf ("required local memory exceeds this device's "
378 "capabilities.\n");
379 return 0;
380 }
381 }
382 else
383 {
384 /* 2 buffers x sizeof(float) * [TSK][TSM] */
385 while ((TSK < 8) && (TSM * TSK * 2 * 4 < local_mem_size))
386 TSK <<= 1;
387 }
388
389 snprintf (extra_opts, 1024, "%s-DMYGEMM6 -DTSM=%zu -DTSN=%zu -DTSK=%zu",
390 fma, TSM, TSM, TSK);
391 printf ("GLOBAL: [%zu, %zu] LOCAL: [%zu, %zu] TSM/TSN: %zu TSK: %zu\n",
392 global_work_size[0], global_work_size[1], local_work_size[0],
393 local_work_size[1], TSM, TSK);
394 }
395
396 /*****************************************************************************/
397
398 err = poclu_load_program (context, device, "matrix1", spir, spirv, poclbin,
399 explicit_binary_path, extra_opts, &program);
400 if (err != CL_SUCCESS)
401 goto FINISH;
402
403 srcA = (cl_float *)malloc (matrix_2d_size * sizeof (cl_float));
404 srcB = (cl_float *)malloc (matrix_2d_size * sizeof (cl_float));
405 dst = (cl_float *)malloc (matrix_2d_size * sizeof (cl_float));
406 sums = (long *)calloc (matrix_size, sizeof (long));
407
408 srandom (time (NULL));
409
410 for (i = 0; i < matrix_size; ++i)
411 {
412 for (j = 0; j < matrix_size; ++j)
413 {
414 long r = random ();
415 int r1 = (r >> 8) % 64;
416 int r2 = (r >> 16) % 64;
417 sums[j] += r2;
418 long x = i * matrix_size + j;
419 srcA[x] = (cl_float) (r1);
420 srcB[x] = (cl_float) (r2);
421 dst[x] = 0.0f;
422 }
423 }
424
425 sum = 0;
426 for (i = 0; i < matrix_size; ++i)
427 {
428 for (j = 0; j < matrix_size; ++j)
429 {
430 long x = i * matrix_size + j;
431 long tmp = srcA[x] * sums[i];
432 sum += tmp;
433 }
434 }
435 printf ("\nExpected sum of all elements: %lu \n", sum);
436
437 err = 0;
438
439 local_work_size[0] = min (global_work_size[0], local_work_size[0]);
440 local_work_size[1] = min (global_work_size[1], local_work_size[1]);
441
442 if (exec_matrix_kernel (context, device, queue, program, matrix_size, srcA,
443 srcB, dst, kernel_name, global_work_size,
444 local_work_size, use_2d_reg_block))
445 {
446 printf ("Error running the tests\n");
447 err = 1;
448 goto FINISH;
449 }
450
451 long total = 0;
452 for (i = 0; i < (matrix_2d_size); ++i)
453 {
454 if (isnormal (dst[i]))
455 total += (long)dst[i];
456 }
457
458 printf ("Sum of all elements: %li \n", total);
459
460 if (total == sum)
461 printf ("OK\n");
462 else
463 printf ("FAIL\n");
464
465FINISH:
466 CHECK_CL_ERROR (clReleaseProgram (program));
467 CHECK_CL_ERROR (clReleaseCommandQueue (queue));
468 CHECK_CL_ERROR (clUnloadPlatformCompiler (platform));
469 CHECK_CL_ERROR (clReleaseContext (context));
470 free (dst);
471 free (srcA);
472 free (srcB);
473 free (sums);
474 return err;
475}
