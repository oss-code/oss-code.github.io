34int
35exec_matadd_kernel (cl_context context, cl_device_id device,
36 cl_command_queue cmd_queue, cl_program program,
37 int n, int m, cl_float *srcA, cl_float *srcB,
38 cl_float *dst)
39{
40 cl_kernel kernel = NULL;
41 cl_mem memobjs[3] = { 0, 0, 0 };
42 size_t global_work_size[] = {n, m};
43 cl_int err = CL_SUCCESS;
44 int i;
45
46 poclu_bswap_cl_float_array (device, (cl_float *)srcA, sizeof (cl_float) * n * m);
47 poclu_bswap_cl_float_array (device, (cl_float *)srcB, sizeof (cl_float) * n * m);
48
49 memobjs[0]
50 = clCreateBuffer (context, CL_MEM_READ_ONLY | CL_MEM_COPY_HOST_PTR,
51 sizeof (cl_float) * n * m, srcA, &err);
52 CHECK_CL_ERROR2 (err);
53
54 memobjs[1]
55 = clCreateBuffer (context, CL_MEM_READ_ONLY | CL_MEM_COPY_HOST_PTR,
56 sizeof (cl_float) * n * m, srcB, &err);
57 CHECK_CL_ERROR2 (err);
58
59 memobjs[2] = clCreateBuffer (context, CL_MEM_READ_WRITE,
60 sizeof (cl_float) * n * m, NULL, &err);
61 CHECK_CL_ERROR2 (err);
62
63 kernel = clCreateKernel (program, "matadd", NULL);
64 CHECK_CL_ERROR2 (err);
65
66 err = clSetKernelArg (kernel, 0, sizeof (cl_mem), (void *)&memobjs[0]);
67 CHECK_CL_ERROR2 (err);
68
69 err = clSetKernelArg (kernel, 1, sizeof (cl_mem), (void *)&memobjs[1]);
70 CHECK_CL_ERROR2 (err);
71
72 err = clSetKernelArg (kernel, 2, sizeof (cl_mem), (void *)&memobjs[2]);
73 CHECK_CL_ERROR2 (err);
74
75 err = clEnqueueNDRangeKernel (cmd_queue, kernel, 2, NULL, global_work_size,
76 NULL, 0, NULL, NULL);
77 CHECK_CL_ERROR2 (err);
78
79 err = clEnqueueReadBuffer (cmd_queue, memobjs[2], CL_TRUE, 0,
80 n * m * sizeof (cl_float), dst, 0, NULL, NULL);
81 CHECK_CL_ERROR2 (err);
82
83 poclu_bswap_cl_float_array (device, (cl_float *)dst,
84 sizeof (cl_float) * n * m);
85 poclu_bswap_cl_float_array (device, (cl_float *)srcA,
86 sizeof (cl_float) * n * m);
87 poclu_bswap_cl_float_array (device, (cl_float *)srcB,
88 sizeof (cl_float) * n * m);
89
90 ERROR:
91 clReleaseMemObject (memobjs[0]);
92 clReleaseMemObject (memobjs[1]);
93 clReleaseMemObject (memobjs[2]);
94 clReleaseKernel (kernel);
95 return err;
96}
