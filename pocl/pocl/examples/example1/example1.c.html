26#include <assert.h>
27#include <stdio.h>
28#include <stdlib.h>
29#include <CL/opencl.h>
30#include <poclu.h>
31
32#define N 4
33
34#ifdef __cplusplus
35# define CALLAPI "C"
36#else
37# define CALLAPI
38#endif
39
40extern CALLAPI int
41exec_dot_product_kernel (cl_context context, cl_device_id device,
42 cl_command_queue cmd_queue, cl_program program, int n,
43 cl_float4 *srcA, cl_float4 *srcB, cl_float *dst);
44
45int
46main (int argc, char **argv)
47{
48 cl_float4 *srcA = NULL;
49 cl_float4 *srcB = NULL;
50 cl_float *dst = NULL;
51 int i, err, spir, spirv, poclbin;
52
53 cl_context context = NULL;
54 cl_device_id device = NULL;
55 cl_platform_id platform = NULL;
56 cl_command_queue queue = NULL;
57 cl_program program = NULL;
58
59 err = poclu_get_any_device2 (&context, &device, &queue, &platform);
60 CHECK_OPENCL_ERROR_IN ("clCreateContext");
61
62 spir = (argc > 1 && argv[1][0] == 's');
63 spirv = (argc > 1 && argv[1][0] == 'v');
64 poclbin = (argc > 1 && argv[1][0] == 'b');
65 const char *explicit_binary_path = (poclbin && (argc > 2)) ? argv[2] : NULL;
66
67 const char *basename = "example1";
68 err = poclu_load_program (context, device, basename, spir, spirv, poclbin,
69 explicit_binary_path, NULL, &program);
70 if (err != CL_SUCCESS)
71 goto FINISH;
72
73 srcA = (cl_float4 *) malloc (N * sizeof (cl_float4));
74 srcB = (cl_float4 *) malloc (N * sizeof (cl_float4));
75 dst = (cl_float *) malloc (N * sizeof (cl_float));
76
77 for (i = 0; i < N; ++i)
78 {
79 srcA[i].s[0] = (cl_float)i;
80 srcA[i].s[1] = (cl_float)i;
81 srcA[i].s[2] = (cl_float)i;
82 srcA[i].s[3] = (cl_float)i;
83 srcB[i].s[0] = (cl_float)i;
84 srcB[i].s[1] = (cl_float)i;
85 srcB[i].s[2] = (cl_float)i;
86 srcB[i].s[3] = (cl_float)i;
87 dst[i] = (cl_float)i;
88 }
89
90 err = 0;
91
92 if (exec_dot_product_kernel (context, device, queue, program, N, srcA, srcB,
93 dst))
94 {
95 printf ("Error running the tests\n");
96 err = 1;
97 goto FINISH;
98 }
99
100 for (i = 0; i < 4; ++i)
101 {
102 printf ("(%f, %f, %f, %f) . (%f, %f, %f, %f) = %f\n",
103 srcA[i].s[0], srcA[i].s[1], srcA[i].s[2], srcA[i].s[3],
104 srcB[i].s[0], srcB[i].s[1], srcB[i].s[2], srcB[i].s[3],
105 dst[i]);
106 if (srcA[i].s[0] * srcB[i].s[0] +
107 srcA[i].s[1] * srcB[i].s[1] +
108 srcA[i].s[2] * srcB[i].s[2] +
109 srcA[i].s[3] * srcB[i].s[3] != dst[i])
110 {
111 printf ("FAIL\n");
112 err = 1;
113 goto FINISH;
114 }
115 }
116
117 printf ("OK\n");
118
119FINISH:
120 CHECK_CL_ERROR (clReleaseProgram (program));
121 CHECK_CL_ERROR (clReleaseCommandQueue (queue));
122 CHECK_CL_ERROR (clUnloadPlatformCompiler (platform));
123 CHECK_CL_ERROR (clReleaseContext (context));
124 free (srcA);
125 free (srcB);
126 free (dst);
127
128 return err;
129}
