5void
6delete_memobjs(cl_mem *memobjs, int n)
7{
8 int i;
9 for (i=0; i<n; i++)
10 clReleaseMemObject(memobjs[i]);

13int
14exec_trig_kernel(const char *program_source,
15 int n, void *srcA, void *dst)
16{
17 cl_context context;
18 cl_command_queue cmd_queue;
19 cl_device_id *devices;
20 cl_program program;
21 cl_kernel kernel;
22 cl_mem memobjs[2];
23 size_t global_work_size[1];
24 size_t local_work_size[1];
25 size_t cb;
26 cl_int err;
27
28 float c = 7.3f; // a scalar number to test non-pointer args
29
30 // create the OpenCL context on a GPU device
31 context = poclu_create_any_context();
32 if (context == (cl_context)0)
33 return -1;
34
35 // get the list of GPU devices associated with context
36 clGetContextInfo(context, CL_CONTEXT_DEVICES, 0, NULL, &cb);
37 devices = (cl_device_id *) malloc(cb);
38 clGetContextInfo(context, CL_CONTEXT_DEVICES, cb, devices, NULL);
39
40 // create a command-queue
41 cmd_queue = clCreateCommandQueue(context, devices[0], 0, NULL);
42 if (cmd_queue == (cl_command_queue)0)
43 {
44 clReleaseContext(context);
45 free(devices);
46 return -1;
47 }
48 free(devices);
49
50 // allocate the buffer memory objects
51 memobjs[0] = clCreateBuffer(context,
52 CL_MEM_READ_ONLY | CL_MEM_COPY_HOST_PTR,
53 sizeof(cl_float4) * n, srcA, NULL);
54 if (memobjs[0] == (cl_mem)0)
55 {
56 clReleaseCommandQueue(cmd_queue);
57 clReleaseContext(context);
58 return -1;
59 }
60
61 memobjs[1] = clCreateBuffer(context,
62 CL_MEM_READ_WRITE,
63 sizeof(cl_float4) * n, NULL, NULL);
64 if (memobjs[1] == (cl_mem)0)
65 {
66 delete_memobjs(memobjs, 1);
67 clReleaseCommandQueue(cmd_queue);
68 clReleaseContext(context);
69 return -1;
70 }
71
72 // create the program
73 program = clCreateProgramWithSource(context,
74 1, (const char**)&program_source, NULL, NULL);
75 if (program == (cl_program)0)
76 {
77 delete_memobjs(memobjs, 2);
78 clReleaseCommandQueue(cmd_queue);
79 clReleaseContext(context);
80 return -1;
81 }
82
83 // build the program
84 err = clBuildProgram(program, 0, NULL, NULL, NULL, NULL);
85 if (err != CL_SUCCESS)
86 {
87 delete_memobjs(memobjs, 2);
88 clReleaseProgram(program);
89 clReleaseCommandQueue(cmd_queue);
90 clReleaseContext(context);
91 return -1;
92 }
93
94 // create the kernel
95 kernel = clCreateKernel(program, "trig", NULL);
96 if (kernel == (cl_kernel)0)
97 {
98 delete_memobjs(memobjs, 2);
99 clReleaseProgram(program);
100 clReleaseCommandQueue(cmd_queue);
101 clReleaseContext(context);
102 return -1;
103 }
104
105 // set the args values
106 err = clSetKernelArg(kernel, 0,
107 sizeof(cl_mem), (void *) &memobjs[0]);
108 err |= clSetKernelArg(kernel, 1,
109 sizeof(cl_mem), (void *) &memobjs[1]);
110 err |= clSetKernelArg(kernel, 2,
111 sizeof(float), (void *) &c);
112
113 if (err != CL_SUCCESS)
114 {
115 delete_memobjs(memobjs, 2);
116 clReleaseKernel(kernel);
117 clReleaseProgram(program);
118 clReleaseCommandQueue(cmd_queue);
119 clReleaseContext(context);
120 return -1;
121 }
122
123 // set work-item dimensions
124 global_work_size[0] = n;
125 local_work_size[0]= 2;
126
127 // execute kernel
128 err = clEnqueueNDRangeKernel(cmd_queue, kernel, 1, NULL,
129 global_work_size, local_work_size,
130 0, NULL, NULL);
131 if (err != CL_SUCCESS)
132 {
133 delete_memobjs(memobjs, 2);
134 clReleaseKernel(kernel);
135 clReleaseProgram(program);
136 clReleaseCommandQueue(cmd_queue);
137 clReleaseContext(context);
138 return -1;
139 }
140
141 // read output image
142 err = clEnqueueReadBuffer(cmd_queue, memobjs[1], CL_TRUE,
143 0, n * sizeof(cl_float4), dst,
144 0, NULL, NULL);
145 if (err != CL_SUCCESS)
146 {
147 delete_memobjs(memobjs, 2);
148 clReleaseKernel(kernel);
149 clReleaseProgram(program);
150 clReleaseCommandQueue(cmd_queue);
151 clReleaseContext(context);
152 return -1;
153 }
154
155 // release kernel, program, and memory objects
156 delete_memobjs(memobjs, 2);
157 clReleaseKernel(kernel);
158 clReleaseProgram(program);
159 clReleaseCommandQueue(cmd_queue);
160 clReleaseContext(context);
161 return 0; // success...
162}
