60// Divide while rounding down
61static inline size_t divdown(size_t const a, size_t const b)
62{
63 return a/b;
64}
65
66// Divide while rounding up
67static inline size_t divup(size_t const a, size_t const b)
68{
69 return (a+b-1)/b;
70}
71
72// Round down
73static inline size_t rounddown(size_t const a, size_t const b)
74{
75 return divdown(a, b) * b;
76}
77
78// Round up
79static inline size_t roundup(size_t const a, size_t const b)
80{
81 return divup(a, b) * b;

424static void allocate(cGH const* const cctkGH,
425 ptr_t* const ptr,
426 CCTK_REAL const val)
427{
428 int const nsize =
429 cctkGH->cctk_ash[0] * cctkGH->cctk_ash[1] * cctkGH->cctk_ash[2];
430 ptr->ptr = malloc(nsize * sizeof(CCTK_REAL));
431 assert(ptr->ptr);
432 for (int k=0; k<cctkGH->cctk_lsh[2]; ++k) {
433 for (int j=0; j<cctkGH->cctk_lsh[1]; ++j) {
434 for (int i=0; i<cctkGH->cctk_lsh[0]; ++i) {
435 int const ind3d =
436 i + cctkGH->cctk_ash[0] * (j + cctkGH->cctk_ash[1] * k);
437 ptr->ptr[ind3d] = val;
438 }
439 }
440 }
441 ptr->mem = clCreateBuffer(context, CL_MEM_COPY_HOST_PTR,
442 nsize * sizeof(CCTK_REAL), ptr->ptr, NULL);
443 assert(ptr->mem);
444}

464void setup(const char* program_source1, const char* program_source2)
465{
466 cl_int cerr;
467
468 // Choose a platform and a context (basically a device)
469 cl_uint num_platforms;
470 clGetPlatformIDs(0, NULL, &num_platforms);
471 cl_platform_id platform_ids[num_platforms];
472 clGetPlatformIDs(num_platforms, &platform_ids[0], &num_platforms);
473 if (num_platforms <= 0) {
474 fprintf(stderr, "No OpenCL platforms found\n");
475 assert(0);
476 }
477 assert(num_platforms > 0);
478
479 cl_device_type const want_device_types = CL_DEVICE_TYPE_CPU;
480 // CL_DEVICE_TYPE_GPU
481 // CL_DEVICE_TYPE_ACCELERATOR
482 // CL_DEVICE_TYPE_CPU | CL_DEVICE_TYPE_GPU | CL_DEVICE_TYPE_ACCELERATOR
483
484 // Loop over all platforms
485 platform_id = 0;
486 for (cl_uint platform = 0; platform < num_platforms; ++platform) {
487
488 cl_platform_id const tmp_platform_id = platform_ids[platform];
489 printf("OpenCL platform #%d:\n", platform);
490
491 size_t platform_name_length;
492 clGetPlatformInfo(tmp_platform_id, CL_PLATFORM_NAME,
493 0, NULL, &platform_name_length);
494 char platform_name[platform_name_length];
495 clGetPlatformInfo(tmp_platform_id, CL_PLATFORM_NAME,
496 platform_name_length, platform_name, NULL);
497 printf(" OpenCL platform name: %s\n", platform_name);
498 size_t platform_vendor_length;
499 clGetPlatformInfo(tmp_platform_id, CL_PLATFORM_VENDOR,
500 0, NULL, &platform_vendor_length);
501 char platform_vendor[platform_vendor_length];
502 clGetPlatformInfo(tmp_platform_id, CL_PLATFORM_VENDOR,
503 platform_vendor_length, platform_vendor, NULL);
504 printf(" OpenCL platform vendor: %s\n", platform_vendor);
505
506 cl_context_properties const cprops[] =
507 {CL_CONTEXT_PLATFORM, (cl_context_properties)tmp_platform_id, 0};
508 context =
509 clCreateContextFromType(cprops, want_device_types, NULL, NULL, &cerr);
510 if (cerr == CL_SUCCESS) {
511 platform_id = tmp_platform_id;
512 }
513 }
514 if (platform_id == 0) {
515 // Could not find a context on any platform, abort
516 fprintf(stderr, "Could not create OpenCL context for selected device type\n");
517 assert(0);
518 }
519
520 size_t ndevice_ids;
521 clGetContextInfo(context, CL_CONTEXT_DEVICES, 0, NULL, &ndevice_ids);
522 ndevice_ids /= sizeof(cl_device_id);
523 cl_device_id device_ids[ndevice_ids];
524 clGetContextInfo(context, CL_CONTEXT_DEVICES,
525 ndevice_ids*sizeof(cl_device_id), device_ids, NULL);
526 assert(ndevice_ids >= 1);
527 main_device_id = device_ids[0];
528
529 if (use_subdev)
530 {
531 const cl_device_partition_property props[]
532 = { CL_DEVICE_PARTITION_EQUALLY, 2, 0 };
533 cl_device_id subdevs[128];
534 cl_uint retval;
535 int err
536 = clCreateSubDevices (main_device_id, props, 128, subdevs, &retval);
537 assert (err == CL_SUCCESS);
538 device_id = subdevs[0];
539 }
540 else
541 device_id = main_device_id;
542
543 size_t device_name_length;
544 clGetDeviceInfo(device_id, CL_DEVICE_NAME, 0, NULL, &device_name_length);
545 char device_name[device_name_length];
546 clGetDeviceInfo(device_id, CL_DEVICE_NAME,
547 device_name_length, device_name, NULL);
548 printf("OpenCL device name: %s\n", device_name);
549
550 clGetDeviceInfo(device_id, CL_DEVICE_PLATFORM,
551 sizeof platform_id, &platform_id, NULL);
552 size_t platform_name_length;
553 clGetPlatformInfo(platform_id, CL_PLATFORM_NAME,
554 0, NULL, &platform_name_length);
555 char platform_name[platform_name_length];
556 clGetPlatformInfo(platform_id, CL_PLATFORM_NAME,
557 platform_name_length, platform_name, NULL);
558 printf("OpenCL platform name: %s\n", platform_name);
559 size_t platform_vendor_length;
560 clGetPlatformInfo(platform_id, CL_PLATFORM_VENDOR,
561 0, NULL, &platform_vendor_length);
562 char platform_vendor[platform_vendor_length];
563 clGetPlatformInfo(platform_id, CL_PLATFORM_VENDOR,
564 platform_vendor_length, platform_vendor, NULL);
565 printf("OpenCL platform vendor: %s\n", platform_vendor);
566
567 cmd_queue = clCreateCommandQueue(context, device_id, 0, NULL);
568 assert(cmd_queue);

571 char const* const options =
572 "-DVECTORISE_ALIGNED_ARRAYS=" STR(VECTORISE_ALIGNED_ARRAYS) " "
573 "-DVECTOR_SIZE_I=" STR(VECTOR_SIZE_I) " "
574 "-DVECTOR_SIZE_J=" STR(VECTOR_SIZE_J) " "
575 "-DVECTOR_SIZE_K=" STR(VECTOR_SIZE_K) " "
576 "-DUNROLL_SIZE_I=" STR(UNROLL_SIZE_I) " "
577 "-DUNROLL_SIZE_J=" STR(UNROLL_SIZE_J) " "
578 "-DUNROLL_SIZE_K=" STR(UNROLL_SIZE_K) " "
579 "-DGROUP_SIZE_I=" STR(GROUP_SIZE_I) " "
580 "-DGROUP_SIZE_J=" STR(GROUP_SIZE_J) " "
581 "-DGROUP_SIZE_K=" STR(GROUP_SIZE_K) " "

605 "********************************************************************************\n"
606 "%s\n"
607 "********************************************************************************\n", build_log);
608 assert(0);

1342 char const* const name,
1343 ptr_t* const ptr,

1354 for (int i=0; i<cctkGH->cctk_lsh[0]; ++i) {
1355 int const is_int = (i>=cctkGH->imin[0] && i<cctkGH->imax[0] &&
1356 j>=cctkGH->imin[1] && j<cctkGH->imax[1] &&
1357 k>=cctkGH->imin[2] && k<cctkGH->imax[2]);
1358 double const val = is_int ? val_int : val_bnd;
1359 int const ind3d =
1360 i + cctkGH->cctk_ash[0] * (j + cctkGH->cctk_ash[1] * k);
1361 if (! (fabs(ptr->ptr[ind3d] - val) <= 1.0e-15)) {
1362 printf("%s[%d,%d,%d] is:%.17g should:%.17g\n",
1363 name, i,j,k, ptr->ptr[ind3d], val);
1364 }
1365 assert(fabs(ptr->ptr[ind3d] - val) <= 1.0e-15);
1366 }
1367 }
1368 }
