33#define DEVICE_INFO_MAX_LENGTH 2048
34#define NUM_OF_DEVICE_ID 32
35#define NUM_OPTIONS 6

37#define ERRNO_EXIT(filename) do { \
38 printf("IO error on file %s: %s\n", filename, strerror(errno)); \
39 exit(2); \
40 } while(0)

42char *kernel_source = NULL;
43char *output_file = NULL;
44cl_uint opencl_device = CL_DEVICE_TYPE_DEFAULT;
45unsigned opencl_device_id = 0;
46int list_devices = 0;
47int list_devices_only = 0;
48char *build_options = NULL;

50/**********************************************************/
51
52typedef int(*poclcc_process)(int, char **, int);
53
54typedef struct _poclcc_option {
55 poclcc_process fct;
56 char *id;
57 char *helper;
58 int num_args_read;
59} poclcc_option;

61/**********************************************************/

63poclcc_option *options_help;

65static int
66print_help()
67{
68 printf("USAGE: poclcc [OPTION]... [FILE]\n");
69 printf("\n");
70 printf("OPTIONS:\n");
71 int i;
72 for (i=0; i<NUM_OPTIONS; i++)
73 printf("%s", options_help[i].helper);
74
75 return -1;
76}
77
78static int
79poclcc_error(char *msg)
80{
81 printf("ERROR: %s", msg);
82 return print_help();
83}

85/**********************************************************
86 * MANAGE INPUT KERNEL FILE */
87
88static int
89process_kernel_file(int arg, char **argv, int argc)
90{
91 if (arg >= argc)
92 return poclcc_error("Incomplete argument for input file!\n");
93
94 char *filename = argv[arg];
95 char *ext = ".pocl";
96 kernel_source = poclu_read_file(filename);
97 if (!kernel_source)
98 ERRNO_EXIT(filename);
99 if (output_file == NULL)
100 {
101 output_file = malloc(strlen(filename)+strlen(ext));
102 strcpy(output_file, filename);
103 strcat(output_file, ext);
104 }
105 return 0;
106}

109 * OPTIONS PROCESS FUNCTIONS*/

111static int
112process_help(int arg, char **argv, int argc)
113{
114 print_help();
115 return 0;

118static int
119process_output(int arg, char **argv, int argc)
120{
121 if (arg >= argc)
122 return poclcc_error("Incomplete argument for output file!\n");
123
124 output_file = argv[arg];
125 return 0;

128static int
129process_opencl_device(int arg, char **argv, int argc)
130{
131 if (arg >= argc)
132 return poclcc_error("Incomplete argument for device_type!\n");
133
134 char *opencl_string = argv[arg];
135 if (!strcmp(opencl_string, "CL_DEVICE_TYPE_CPU"))
136 opencl_device = CL_DEVICE_TYPE_CPU;
137 else if (!strcmp(opencl_string, "CL_DEVICE_TYPE_GPU"))
138 opencl_device = CL_DEVICE_TYPE_GPU;
139 else if (!strcmp(opencl_string, "CL_DEVICE_TYPE_ACCELERATOR"))
140 opencl_device = CL_DEVICE_TYPE_ACCELERATOR;
141 else if (!strcmp(opencl_string, "CL_DEVICE_TYPE_DEFAULT"))
142 opencl_device = CL_DEVICE_TYPE_DEFAULT;
143 else if (!strcmp(opencl_string, "CL_DEVICE_TYPE_ALL"))
144 opencl_device = CL_DEVICE_TYPE_ALL;
145 else
146 {
147 printf("Invalid argument for device_type!\n");
148 return print_help();
149 }
150 return 0;
151}

154process_build_options(int arg, char **argv, int argc)

164process_device_id(int arg, char **argv, int argc)
165{
166 if (arg >= argc)
167 return poclcc_error("Incomplete argument for build_options!\n");
168
169 opencl_device_id = atoi(argv[arg]);

173static int
174process_list_devices(int arg, char **argv, int argc)

183static poclcc_option options[NUM_OPTIONS] =
184{
185 {process_help, "-h",
186 "\t-h\n"
187 "\t\tDisplay the help\n",
188 1},
189 {process_build_options, "-b",
190 "\t-b <options>\n"
191 "\t\tBuild the program with <options> options\n",
192 2},
193 {process_opencl_device, "-d",
194 "\t-d <device_type>\n"
195 "\t\tSelect <device_type> as the device_type for clGetDeviceIDs.\n"
196 "\t\tDefault: CL_DEVICE_TYPE_DEFAULT\n",
197 2},
198 {process_list_devices, "-l",
199 "\t-l\n"
200 "\t\tList the opencl device found (that match the <device_type>\n",
201 1},
202 {process_device_id, "-i",
203 "\t-i <device_id>\n"
204 "\t\tSelect the <device_id> opencl device to generate the pocl binary file\n"
205 "\t\tDefault: 0\n",
206 2},
207 {process_output, "-o",
208 "\t-o <file>\n"
209 "\t\tWrite output to <file>\n",
210 2}
211};

213/**********************************************************/
214
215static int
216search_process(char *arg)
217{
218 int i;
219 for (i=0; i<NUM_OPTIONS; i++)
220 {
221 if (!strcmp(options[i].id, arg))
222 return i;
223 }
224 return -1;
225}
226
227static int
228process_arg(int *arg, char **argv, int argc)
229{
230 int prev_arg = *arg;
231 char *current_arg = argv[*arg];
232 int current_process = search_process(current_arg);
233 if (current_process == -1)
234 return poclcc_error("Unknown argument!\n");
235 else
236 {
237 poclcc_option * current_option = &options[current_process];
238 int num_args_read = current_option->num_args_read;
239 *arg = prev_arg + num_args_read;
240 return current_option->fct(prev_arg + 1, argv, argc);
241 }
242}

244/**********************************************************/
245
246int
247main(int argc, char **argv)
248{
249//MANAGEMENT OF ARGUMENTS
250 options_help = options;
251 int arg_num=1;
252 if (argc < 2)
253 return poclcc_error("Invalid argument!\n");
254
255 while (arg_num < argc-1)
256 if (process_arg(&arg_num, argv, argc))
257 return -1;
258
259 if (arg_num >= argc && list_devices)
260 list_devices_only = 1;
261 else if (arg_num >= argc)
262 poclcc_error("Invalid arguments!\n");
263 else
264 {
265 int current_process = search_process(argv[arg_num]);
266 if (current_process == -1 && process_kernel_file(arg_num, argv, argc))
267 return -1;
268 else if (current_process != -1)
269 {
270 process_arg(&arg_num, argv, argc);
271 list_devices_only = 1;
272 }
273 }
274
275//OPENCL STUFF
276 cl_platform_id cpPlatform;
277 cl_device_id device_ids[NUM_OF_DEVICE_ID];
278 cl_context context;
279 cl_program program;
280 cl_int err;
281 cl_uint num_devices, i;
282
283 CHECK_CL_ERROR(clGetPlatformIDs(1, &cpPlatform, NULL));
284
285 CHECK_CL_ERROR(clGetDeviceIDs(cpPlatform, opencl_device,
286 NUM_OF_DEVICE_ID, device_ids, &num_devices));
287
288 if (opencl_device_id >= num_devices)
289 return poclcc_error("Invalid opencl device_id!\n");
290
291 if (list_devices)
292 {
293 context = clCreateContext(0, num_devices, device_ids, NULL, NULL, &err);
294 CHECK_OPENCL_ERROR_IN("clCreateContext");
295
296 printf("LIST OF DEVICES:\n");
297 for (i=0; i<num_devices; i++)
298 {
299 char str[DEVICE_INFO_MAX_LENGTH];
300 CHECK_CL_ERROR(clGetDeviceInfo(device_ids[i], CL_DEVICE_VENDOR,
301 DEVICE_INFO_MAX_LENGTH, str, NULL));
302
303 printf("%i:\n Vendor: %s\n", i, str);
304
305 CHECK_CL_ERROR(clGetDeviceInfo(device_ids[i], CL_DEVICE_NAME,
306 DEVICE_INFO_MAX_LENGTH, str, NULL));
307 printf(" Name: %s\n", str);
308
309 // to print device->poclbin_hash_string
310 CHECK_CL_ERROR(clGetDeviceInfo(device_ids[i], CL_DEVICE_VERSION,
311 DEVICE_INFO_MAX_LENGTH, str, NULL));
312 printf(" Version: %s\n", str);
313 }
314
315 clReleaseContext(context);
316 }
317 if (list_devices_only)
318 return 0;
319
320 context = clCreateContext(0, 1, &device_ids[opencl_device_id], NULL, NULL, &err);
321 CHECK_OPENCL_ERROR_IN("clCreateContext");
322
323 program = clCreateProgramWithSource(context, 1, (const char **)&kernel_source, NULL, &err);
324 CHECK_OPENCL_ERROR_IN("clCreateProgramWithSource");
325
326 CHECK_CL_ERROR(clBuildProgram(program, 0, NULL, build_options, NULL, NULL));
327
328 size_t binary_sizes;
329 char *binary;
330
331 CHECK_CL_ERROR(clGetProgramInfo(program, CL_PROGRAM_BINARY_SIZES,
332 sizeof(size_t), &binary_sizes, NULL));
333
334 binary = malloc(sizeof(char)*binary_sizes);
335 if (!binary)
336 {
337 printf("malloc(binary) failed\n");
338 exit(1);
339 }
340
341 CHECK_CL_ERROR(clGetProgramInfo(program, CL_PROGRAM_BINARIES,
342 sizeof(unsigned char*), &binary, NULL));
343
344 CHECK_CL_ERROR(clReleaseProgram(program));
345 CHECK_CL_ERROR(clReleaseContext(context));
346
347 if (poclu_write_file(output_file, binary, binary_sizes))
348 ERRNO_EXIT(output_file);
349
350 free(binary);
351
352 return 0;
353}
