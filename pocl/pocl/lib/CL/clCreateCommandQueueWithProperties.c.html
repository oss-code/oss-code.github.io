26CL_API_ENTRY cl_command_queue CL_API_CALL
27POname(clCreateCommandQueueWithProperties)(cl_context context,
28 cl_device_id device,
29 const cl_queue_properties *properties,
30 cl_int *errcode_ret) CL_API_SUFFIX__VERSION_1_0
31{
32 unsigned i = 0;
33 int errcode;
34 cl_bool found = CL_FALSE;
35 cl_command_queue_properties queue_props = 0;
36 int queue_props_set = 0;
37 cl_uint queue_size = 0;
38 const cl_command_queue_properties valid_prop_flags =
39 (CL_QUEUE_OUT_OF_ORDER_EXEC_MODE_ENABLE
40 | CL_QUEUE_PROFILING_ENABLE
41 | CL_QUEUE_ON_DEVICE
42 | CL_QUEUE_ON_DEVICE_DEFAULT);
43
44 POCL_GOTO_ERROR_COND((context == NULL), CL_INVALID_CONTEXT);
45
46 POCL_GOTO_ERROR_COND ((device == NULL), CL_INVALID_DEVICE);
47
48 for (i=0; i<context->num_devices; i++)
49 {
50 if (context->devices[i] == pocl_real_dev (device))
51 found = CL_TRUE;
52 }
53
54 POCL_GOTO_ERROR_ON((found == CL_FALSE), CL_INVALID_DEVICE,
55 "Could not find device in the context\n");
56
57 i = 0;
58 if (properties)
59 while(properties[i])
60 switch(properties[i])
61 {
62 case CL_QUEUE_PROPERTIES:
63 queue_props = (cl_command_queue_properties)properties[i+1];
64 queue_props_set = 1;
65 i+=2;
66 break;
67 case CL_QUEUE_SIZE:
68 queue_size = (cl_uint)properties[i+1];
69 i+=2;
70 break;
71 default:
72 POCL_GOTO_ERROR_ON(1, CL_INVALID_VALUE, "Invalid values it properties\n");
73 }
74
75 if (queue_props_set)
76 {
77 if (queue_props & CL_QUEUE_ON_DEVICE)
78 {
79 if (queue_size == 0)
80 queue_size = device->dev_queue_pref_size;
81
82 POCL_GOTO_ERROR_COND((queue_size > device->dev_queue_max_size),
83 CL_INVALID_QUEUE_PROPERTIES);
84
85 // create a device side queue
86 POCL_ABORT_UNIMPLEMENTED("Device side queue");
87 }
88 else
89 POCL_GOTO_ERROR_ON((queue_size > 0), CL_INVALID_VALUE,
90 "To specify queue size, you must use CL_QUEUE_ON_DEVICE in flags\n");
91
92 /* validate flags */
93 POCL_GOTO_ERROR_ON((queue_props & (!valid_prop_flags)), CL_INVALID_VALUE,
94 "CL_QUEUE_PROPERTIES contain invalid entries");
95 }
96
97 // currently thhere's only support for host side queues.
98 return POname(clCreateCommandQueue)(context, device, queue_props, errcode_ret);
99
100ERROR:
101 if(errcode_ret)
102 {
103 *errcode_ret = errcode;
104 }
105 return NULL;
106}
