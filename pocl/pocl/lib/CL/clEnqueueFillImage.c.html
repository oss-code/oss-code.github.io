24#include <CL/cl.h>
25#include <string.h>
26
27#include "pocl_util.h"
28#include "pocl_image_util.h"
29
30extern CL_API_ENTRY cl_int CL_API_CALL
31POname(clEnqueueFillImage)(cl_command_queue command_queue,
32 cl_mem image,
33 const void * fill_color,
34 const size_t* origin,
35 const size_t * region,
36 cl_uint num_events_in_wait_list,
37 const cl_event* event_wait_list,
38 cl_event* event)
39CL_API_SUFFIX__VERSION_1_2
40{
41 int errcode = CL_SUCCESS;
42 _cl_command_node *cmd = NULL;
43 void *fill_pixel = NULL;
44
45 POCL_RETURN_ERROR_COND((command_queue == NULL), CL_INVALID_COMMAND_QUEUE);
46
47 POCL_RETURN_ERROR_COND((image == NULL), CL_INVALID_MEM_OBJECT);
48 POCL_RETURN_ERROR_COND((origin == NULL), CL_INVALID_VALUE);
49 POCL_RETURN_ERROR_COND((region == NULL), CL_INVALID_VALUE);
50 POCL_RETURN_ERROR_COND((fill_color == NULL), CL_INVALID_VALUE);
51
52 POCL_RETURN_ERROR_ON((command_queue->context != image->context), CL_INVALID_CONTEXT,
53 "image and command_queue are not from the same context\n");
54
55 POCL_RETURN_ERROR_ON ((!image->is_image), CL_INVALID_MEM_OBJECT,
56 "image argument is not an image\n");
57 POCL_RETURN_ON_UNSUPPORTED_IMAGE (image, command_queue->device);
58
59 errcode = pocl_check_event_wait_list (command_queue, num_events_in_wait_list,
60 event_wait_list);
61 if (errcode != CL_SUCCESS)
62 return errcode;
63
64 errcode = pocl_check_image_origin_region (image, origin, region);
65 if (errcode != CL_SUCCESS)
66 return errcode;
67
68 fill_pixel = malloc (4 * sizeof(int));
69 if (fill_pixel == NULL)
70 {
71 errcode = CL_OUT_OF_HOST_MEMORY;
72 goto ERROR_CLEAN;
73 }
74
75 /* The fill color is:
76 *
77 * a four component RGBA floating-point color value if the image channel
78 * data type is NOT an unnormalized signed and unsigned integer type,
79 *
80 * a four component signed integer value if the image channel data type
81 * is an unnormalized signed integer type and
82 *
83 * a four component unsigned integer value if the image channel data type
84 * is an unormalized unsigned integer type.
85 *
86 * The fill color will be converted to the appropriate
87 * image channel format and order associated with image.
88 */
89 pocl_write_pixel_zero (fill_pixel, fill_color, image->image_channel_order,
90 image->image_elem_size,
91 image->image_channel_data_type);
92
93 size_t px = image->image_elem_size * image->image_channels;
94
95 if (IS_IMAGE1D_BUFFER (image))
96 {
97 return POname (clEnqueueFillBuffer) (
98 command_queue,
99 image->buffer,
100 fill_pixel, 16,
101 origin[0] * px,
102 region[0] * px,
103 num_events_in_wait_list, event_wait_list, event);
104 }
105
106 errcode = pocl_create_command (&cmd, command_queue, CL_COMMAND_FILL_IMAGE,
107 event, num_events_in_wait_list,
108 event_wait_list, 1, &image);
109 if (errcode != CL_SUCCESS)
110 goto ERROR_CLEAN;
111
112 cmd->command.fill_image.fill_pixel = fill_pixel;
113 cmd->command.fill_image.pixel_size = px;
114
115 cmd->command.fill_image.mem_id
116 = &image->device_ptrs[command_queue->device->dev_id];
117
118 cmd->command.fill_image.origin[0] = origin[0];
119 cmd->command.fill_image.origin[1] = origin[1];
120 cmd->command.fill_image.origin[2] = origin[2];
121 cmd->command.fill_image.region[0] = region[0];
122 cmd->command.fill_image.region[1] = region[1];
123 cmd->command.fill_image.region[2] = region[2];
124
125 POname(clRetainMemObject) (image);
126 image->owning_device = command_queue->device;
127 pocl_command_enqueue(command_queue, cmd);
128
129 return errcode;
130
131 ERROR_CLEAN:
132 POCL_MEM_FREE(fill_pixel);
133 return errcode;
134}
