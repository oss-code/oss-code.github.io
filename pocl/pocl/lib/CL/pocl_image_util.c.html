29cl_int opencl_image_type_to_index (cl_mem_object_type image_type)
30{
31 switch (image_type)
32 {
33 case CL_MEM_OBJECT_IMAGE2D : return 0;
34 case CL_MEM_OBJECT_IMAGE3D : return 1;
35 case CL_MEM_OBJECT_IMAGE2D_ARRAY : return 2;
36 case CL_MEM_OBJECT_IMAGE1D : return 3;
37 case CL_MEM_OBJECT_IMAGE1D_ARRAY : return 4;
38 case CL_MEM_OBJECT_IMAGE1D_BUFFER : return 5;
39 default: return -1;
40 }
41}

43void
44origin_to_bytes (cl_mem mem, const size_t *origin, size_t *byte_offset)
45{
46 *byte_offset = origin[0] + origin[1] * mem->image_row_pitch
47 + origin[2] * mem->image_slice_pitch;
48 *byte_offset *= (mem->image_elem_size * mem->image_channels);
49}

51static unsigned
52pocl_get_image_dim (const cl_mem image)
53{
54 if ((image->type == CL_MEM_OBJECT_IMAGE1D)
55 || (image->type == CL_MEM_OBJECT_IMAGE1D_BUFFER))
56 return 1;
57 if ((image->type == CL_MEM_OBJECT_IMAGE2D)
58 || (image->type == CL_MEM_OBJECT_IMAGE1D_ARRAY))
59 return 2;
60 if ((image->type == CL_MEM_OBJECT_IMAGE3D)
61 || (image->type == CL_MEM_OBJECT_IMAGE2D_ARRAY))
62 return 3;
63
64 return (unsigned)-1;
65}

67extern cl_int
68pocl_check_image_origin_region (const cl_mem image,
69 const size_t *origin,
70 const size_t *region)
71{
72 POCL_RETURN_ERROR_COND((image == NULL), CL_INVALID_MEM_OBJECT);
73
74 POCL_RETURN_ERROR_COND((origin == NULL), CL_INVALID_VALUE);
75 POCL_RETURN_ERROR_COND((region == NULL), CL_INVALID_VALUE);
76
77 unsigned dim = pocl_get_image_dim (image);
78
79 if (dim < 3)
80 {
81 /* If image is a 2D image object, origin[2] must be 0.
82 * If image is a 1D image or 1D image buffer object,
83 * origin[1] and origin[2] must be 0.
84 * If image is a 2D image object, region[2] must be 1.
85 * If image is a 1D image or 1D image buffer object,
86 * region[1] and region[2] must be 1.
87 * If image is a 1D image array object, region[2] must be 1.
88 */
89 unsigned i;
90 for (i = dim; i < 3; i++)
91 {
92 POCL_RETURN_ERROR_ON (
93 (origin[i] != 0), CL_INVALID_VALUE,
94 "Image origin[x](=%zu) must be 0 for x(=%u) >= image_dim\n",
95 origin[i], i);
96 POCL_RETURN_ERROR_ON (
97 (region[i] != 1), CL_INVALID_VALUE,
98 "Image region[x](=%zu) must be 1 for x(=%u) >= image_dim\n",
99 region[i], i);
100 }
101 }
102
103 /* check if origin + region in each dimension is with in image bounds */
104 POCL_RETURN_ERROR_ON (
105 ((origin[0] + region[0]) > image->image_width), CL_INVALID_VALUE,
106 "(origin[0](=%zu) + region[0](=%zu)) > image->image_width(=%zu)",
107 origin[0], region[0], image->image_width);
108 POCL_RETURN_ERROR_ON (
109 (image->image_height > 0
110 && ((origin[1] + region[1]) > image->image_height)),
111 CL_INVALID_VALUE,
112 "(origin[1](=%zu) + region[1](=%zu)) > image->image_height(=%zu)",
113 origin[1], region[2], image->image_height);
114 POCL_RETURN_ERROR_ON (
115 (image->image_depth > 0 && (origin[2] + region[2]) > image->image_depth),
116 CL_INVALID_VALUE,
117 "(origin[2](=%zu) + region[2](=%zu)) > image->image_depth(=%zu)",
118 origin[1], region[2], image->image_depth);
119 return CL_SUCCESS;
120}

122extern cl_int
123pocl_check_device_supports_image (cl_device_id device,
124 const cl_image_format *image_format,
125 const cl_image_desc *image_desc,
126 cl_uint image_type_idx,
127 int *device_support)
128{
129 cl_uint i;
130 size_t m;
131
132 if (device_support)
133 *device_support = 0;
134
135 POCL_RETURN_ERROR_ON((!device->image_support), CL_INVALID_OPERATION,
136 "Device does not support images");
137
138 if (image_desc->image_type == CL_MEM_OBJECT_IMAGE1D
139 || image_desc->image_type == CL_MEM_OBJECT_IMAGE1D_ARRAY)
140 {
141 POCL_RETURN_ERROR_ON (
142 (image_desc->image_width > device->image2d_max_width),
143 CL_INVALID_IMAGE_SIZE, "Image width > device.image2d_max_width\n");
144 }
145
146 if (image_desc->image_type == CL_MEM_OBJECT_IMAGE2D
147 || image_desc->image_type == CL_MEM_OBJECT_IMAGE2D_ARRAY)
148 {
149 POCL_RETURN_ERROR_ON (
150 (image_desc->image_width > device->image2d_max_width),
151 CL_INVALID_IMAGE_SIZE, "Image width > device.image2d_max_width\n");
152 POCL_RETURN_ERROR_ON (
153 (image_desc->image_height > device->image2d_max_height),
154 CL_INVALID_IMAGE_SIZE, "Image height > device.image2d_max_height\n");
155 }
156
157 if (image_desc->image_type == CL_MEM_OBJECT_IMAGE3D)
158 {
159 POCL_RETURN_ERROR_ON (
160 (image_desc->image_width > device->image3d_max_width),
161 CL_INVALID_IMAGE_SIZE, "Image width > device.image3d_max_width\n");
162 POCL_RETURN_ERROR_ON (
163 (image_desc->image_height > device->image3d_max_height),
164 CL_INVALID_IMAGE_SIZE, "Image height > device.image3d_max_height\n");
165 POCL_RETURN_ERROR_ON (
166 (image_desc->image_depth > device->image3d_max_depth),
167 CL_INVALID_IMAGE_SIZE, "Image depth > device.image3d_max_depth\n");
168 }
169
170 if ((image_desc->image_type == CL_MEM_OBJECT_IMAGE1D_ARRAY)
171 || (image_desc->image_type == CL_MEM_OBJECT_IMAGE2D_ARRAY))
172 {
173 POname (clGetDeviceInfo (device, CL_DEVICE_IMAGE_MAX_ARRAY_SIZE,
174 sizeof (m), &m, NULL));
175 POCL_RETURN_ERROR_ON ((m < image_desc->image_array_size),
176 CL_INVALID_IMAGE_SIZE,
177 "Image array size > device.max_array_size\n");
178 }
179
180 if (image_desc->image_type == CL_MEM_OBJECT_IMAGE1D_BUFFER)
181 {
182 POname (clGetDeviceInfo (device, CL_DEVICE_IMAGE_MAX_BUFFER_SIZE,
183 sizeof (m), &m, NULL));
184 POCL_RETURN_ERROR_ON (
185 (m < image_desc->image_width), CL_INVALID_IMAGE_SIZE,
186 "Image buffer size (width) > device.max_buffer_size\n");
187 }
188
189 if (device_support)
190 *device_support |= DEVICE_IMAGE_SIZE_SUPPORT;
191
192 for (i = 0; i < device->num_image_formats[image_type_idx]; i++)
193 {
194 const cl_image_format *p = device->image_formats[image_type_idx];
195 assert (p != NULL);
196 if (p[i].image_channel_order
197 == image_format->image_channel_order
198 && p[i].image_channel_data_type
199 == image_format->image_channel_data_type)
200 {
201 if (device_support)
202 *device_support |= DEVICE_IMAGE_FORMAT_SUPPORT;
203
204 return CL_SUCCESS;
205 }
206 }
207
208 POCL_MSG_ERR2 ("CL_INVALID_IMAGE_FORMAT_DESCRIPTOR",
209 "The image format is not supported by the device\n");
210 return CL_INVALID_IMAGE_FORMAT_DESCRIPTOR;
211}

213extern void
214pocl_get_image_information (cl_channel_order ch_order,
215 cl_channel_type ch_type,
216 int* channels_out,
217 int* elem_size_out)
218{
219 if (ch_type == CL_SNORM_INT8 || ch_type == CL_UNORM_INT8 ||
220 ch_type == CL_SIGNED_INT8 || ch_type == CL_UNSIGNED_INT8)
221 {
222 *elem_size_out = 1; /* 1 byte */
223 }
224 else if (ch_type == CL_UNSIGNED_INT32 || ch_type == CL_SIGNED_INT32 ||
225 ch_type == CL_FLOAT || ch_type == CL_UNORM_INT_101010)
226 {
227 *elem_size_out = 4; /* 32bit -> 4 bytes */
228 }
229 else if (ch_type == CL_SNORM_INT16 || ch_type == CL_UNORM_INT16 ||
230 ch_type == CL_SIGNED_INT16 || ch_type == CL_UNSIGNED_INT16 ||
231 ch_type == CL_UNORM_SHORT_555 || ch_type == CL_UNORM_SHORT_565 ||
232 ch_type == CL_HALF_FLOAT)
233 {
234 *elem_size_out = 2; /* 16bit -> 2 bytes */
235 }
236 else
237 *elem_size_out = 0;
238
239 /* channels TODO: verify num of channels*/
240 if (ch_order == CL_RGB || ch_order == CL_RGBx || ch_order == CL_R ||
241 ch_order == CL_Rx || ch_order == CL_A)
242 {
243 *channels_out = 1;
244 }
245 else if (ch_order == CL_RG || ch_order == CL_RGx || ch_order == CL_RA)
246 {
247 *channels_out = 2;
248 }
249 else
250 {
251 *channels_out = 4;
252 }
253}

391static cl_uint4
392map_channels (cl_uint4 color, int order)
393{
394 switch (order)
395 {
396 case CL_ARGB:
397 {
398 // return color.wxyz;
399 cl_uint4 ret;
400 ret.s[0] = color.s[3];
401 ret.s[1] = color.s[0];
402 ret.s[2] = color.s[1];
403 ret.s[3] = color.s[2];
404 return ret;
405 }
406 case CL_BGRA:
407 {
408 // return color.zyxw;
409 cl_uint4 ret;
410 ret.s[0] = color.s[2];
411 ret.s[1] = color.s[1];
412 ret.s[2] = color.s[0];
413 ret.s[3] = color.s[3];
414 return ret;
415 }
416 case CL_RGBA:
417 default:
418 return color;
419 }
420}

422/* only for CL_FLOAT, CL_SNORM_INT8, CL_UNORM_INT8,
423 * CL_SNORM_INT16, CL_UNORM_INT16 channel types */
424static void
425write_float4_pixel (cl_float4 color, void *data, int type)
426{
427 if (type == CL_FLOAT)
428 {
429 cl_float4 *p = (cl_float4 *)data;
430 FOR4
431 p->s[i] = color.s[i];
432 return;
433 }
434 if (type == CL_HALF_FLOAT)
435 {
436 /* TODO: convert to builtins */
437 ((uint16_t *)data)[0] = float_to_half (color.s0);
438 ((uint16_t *)data)[1] = float_to_half (color.s1);
439 ((uint16_t *)data)[2] = float_to_half (color.s2);
440 ((uint16_t *)data)[3] = float_to_half (color.s3);
441 return;
442 }
443 const cl_float f127 = ((cl_float) (CL_CHAR_MAX));
444 const cl_float f32767 = ((cl_float) (CL_SHRT_MAX));
445 const cl_float f255 = ((cl_float) (CL_UCHAR_MAX));
446 const cl_float f65535 = ((cl_float) (CL_USHRT_MAX));
447 if (type == CL_SNORM_INT8)
448 {
449 /* <-1.0, 1.0> to <I*_MIN, I*_MAX> */
450 cl_float4 colorf;
451 FOR4
452 colorf.s[i] = color.s[i] * f127;
453 cl_char4 final_color = convert_char4_sat (colorf);
454 *((cl_char4 *)data) = final_color;
455 return;
456 }
457 if (type == CL_SNORM_INT16)
458 {
459 cl_float4 colorf;
460 FOR4
461 colorf.s[i] = color.s[i] * f32767;
462 cl_short4 final_color = convert_short4_sat (colorf);
463 *((cl_short4 *)data) = final_color;
464 return;
465 }
466 if (type == CL_UNORM_INT8)
467 {
468 /* <0, I*_MAX> to <0.0, 1.0> */
469 /* <-1.0, 1.0> to <I*_MIN, I*_MAX> */
470 cl_float4 colorf;
471 FOR4
472 colorf.s[i] = color.s[i] * f255;
473 cl_uchar4 final_color = convert_uchar4_sat (colorf);
474 *((cl_uchar4 *)data) = final_color;
475 return;
476 }
477 if (type == CL_UNORM_INT16)
478 {
479 cl_float4 colorf;
480 FOR4
481 colorf.s[i] = color.s[i] * f65535;
482 cl_ushort4 final_color = convert_ushort4_sat (colorf);
483 *((cl_ushort4 *)data) = final_color;
484 return;
485 }
486
487 return;
488}

490/* only for CL_FLOAT, CL_SNORM_INT8, CL_UNORM_INT8,
491 * CL_SNORM_INT16, CL_UNORM_INT16 channel types */
492static void
493write_float_pixel (cl_float color, void *data, int type)
494{
495 if (type == CL_FLOAT)
496 {
497 *((float *)data) = color;
498 return;
499 }
500 if (type == CL_HALF_FLOAT)
501 {
502 /* TODO: convert to builtins */
503 *((uint16_t *)data) = float_to_half (color);
504 return;
505 }
506 const cl_float f127 = ((cl_float)CL_CHAR_MAX);
507 const cl_float f32767 = ((cl_float)CL_SHRT_MAX);
508 const cl_float f255 = ((cl_float)CL_UCHAR_MAX);
509 const cl_float f65535 = ((cl_float)CL_USHRT_MAX);
510 if (type == CL_SNORM_INT8)
511 {
512 /* <-1.0, 1.0> to <I*_MIN, I*_MAX> */
513 cl_float colorf = color * f127;
514 cl_char final_color = convert_char_sat (colorf);
515 *((cl_char *)data) = final_color;
516 return;
517 }
518 if (type == CL_SNORM_INT16)
519 {
520 cl_float colorf = color * f32767;
521 cl_short final_color = convert_short_sat (colorf);
522 *((cl_short *)data) = final_color;
523 return;
524 }
525 if (type == CL_UNORM_INT8)
526 {
527 /* <0, I*_MAX> to <0.0, 1.0> */
528 /* <-1.0, 1.0> to <I*_MIN, I*_MAX> */
529 cl_float colorf = color * f255;
530 cl_uchar final_color = convert_uchar_sat (colorf);
531 *((cl_uchar *)data) = final_color;
532 return;
533 }
534 if (type == CL_UNORM_INT16)
535 {
536 cl_float colorf = color * f65535;
537 cl_ushort final_color = convert_ushort_sat (colorf);
538 *((cl_ushort *)data) = final_color;
539 return;
540 }
541
542 return;
543}

545/* for use inside filter functions
546 * no channel mapping
547 * no pointers to img metadata */
548static void
549pocl_write_pixel_fast_ui (cl_uint4 color, int order, int elem_size, void *data)
550{
551 if (order == CL_A)
552 {
553 if (elem_size == 1)
554 *((cl_uchar *)data) = convert_uchar_sat_int (color.s[3]);
555 else if (elem_size == 2)
556 *((cl_ushort *)data) = convert_ushort_sat_int (color.s[3]);
557 else if (elem_size == 4)
558 *((cl_uint *)data) = color.s[3];
559 return;
560 }
561
562 if (elem_size == 1)
563 {
564 *((cl_uchar4 *)data) = convert_uchar4_sat_int (color);
565 }
566 else if (elem_size == 2)
567 {
568 *((cl_ushort4 *)data) = convert_ushort4_sat_int (color);
569 }
570 else if (elem_size == 4)
571 {
572 *((cl_uint4 *)data) = color;
573 }
574
575 return;
576}

619 {
620 *((cl_short4 *)data) = convert_short4_sat_int (color);
621 }
622 else if (elem_size == 4)
623 {
624 *((cl_int4 *)data) = color;
625 }
626 return;

629/* full write with channel map conversion etc
630 * Writes a four element pixel to an image pixel pointed by integer coords.
631 */
632void
633pocl_write_pixel_zero (void *data, const void *color_ptr, int order,
634 int elem_size, int channel_type)
635{
636 cl_uint4 color;
637 FOR4
638 color.s[i] = ((cl_uint4 *)color_ptr)->s[i];
639
640 color = map_channels (color, order);
641
642 typedef union
643 {
644 cl_uint4 ui;
645 cl_int4 i;
646 cl_float4 f;
647 } u;
648
649 u ucolor;
650 ucolor.ui = color;
651
652 if ((channel_type == CL_SIGNED_INT8) || (channel_type == CL_SIGNED_INT16)
653 || (channel_type == CL_SIGNED_INT32))
654 pocl_write_pixel_fast_i (ucolor.i, order, elem_size, data);
655 else if ((channel_type == CL_UNSIGNED_INT8)
656 || (channel_type == CL_UNSIGNED_INT16)
657 || (channel_type == CL_UNSIGNED_INT32))
658 pocl_write_pixel_fast_ui (ucolor.ui, order, elem_size, data);
659 else // TODO unsupported channel types
660 pocl_write_pixel_fast_f (ucolor.f, channel_type, order, data);
661}
