25#include "config.h"
26#include "pocl_cl.h"
27#include "pocl_debug.h"
28#include "pocl_util.h"
29#include <assert.h>
30#include <stdbool.h>
31#include <string.h>

33CL_API_ENTRY cl_int CL_API_CALL
34POname(clSetKernelArg)(cl_kernel kernel,
35 cl_uint arg_index,
36 size_t arg_size,
37 const void *arg_value) CL_API_SUFFIX__VERSION_1_0
38{
39 size_t arg_alignment, arg_alloc_size;
40 struct pocl_argument *p;
41 struct pocl_argument_info *pi;
42 void *value;
43
44 POCL_RETURN_ERROR_COND((kernel == NULL), CL_INVALID_KERNEL);
45
46 POCL_RETURN_ERROR_ON ((arg_index >= kernel->meta->num_args),
47 CL_INVALID_ARG_INDEX,
48 "This kernel has %u args, cannot set arg %u\n",
49 (unsigned)kernel->meta->num_args, (unsigned)arg_index);
50
51 POCL_RETURN_ERROR_ON((kernel->dyn_arguments == NULL), CL_INVALID_KERNEL,
52 "This kernel has no arguments that could be set\n");
53
54 pi = &(kernel->meta->arg_info[arg_index]);
55 int is_local = ARGP_IS_LOCAL (pi);
56
57 if (POCL_DEBUGGING_ON)
58 {
59 const void *ptr_value = NULL;
60 if (((pi->type == POCL_ARG_TYPE_POINTER)
61 || (pi->type == POCL_ARG_TYPE_IMAGE))
62 && arg_value)
63 ptr_value = *(const void **)arg_value;
64
65 uint32_t uint32_value = 0;
66 if (arg_value && (arg_size == 4))
67 uint32_value = *(uint32_t *)arg_value;
68
69 POCL_MSG_PRINT_GENERAL ("Kernel %15s || SetArg idx %3u || %8s || "
70 "Local %1i || Size %6zu || Value %p || "
71 "*Value %p || *(uint32*)Value: %8u \n",
72 kernel->name, arg_index, pi->type_name, is_local,
73 arg_size, arg_value, ptr_value, uint32_value);
74 }
75
76 POCL_RETURN_ERROR_ON (
77 ((arg_value != NULL) && is_local), CL_INVALID_ARG_VALUE,
78 "arg_value != NULl and arg %u is in local address space\n", arg_index);
79
80 /* Trigger CL_INVALID_ARG_VALUE if arg_value specified is NULL
81 * for an argument that is not declared with the __local qualifier. */
82 POCL_RETURN_ERROR_ON (
83 ((arg_value == NULL) && (!is_local)
84 && (pi->type != POCL_ARG_TYPE_POINTER)),
85 CL_INVALID_ARG_VALUE,
86 "arg_value == NULL and arg %u is not in local address space\n",
87 arg_index);
88
89 /* Trigger CL_INVALID_ARG_SIZE if arg_size is zero
90 * and the argument is declared with the __local qualifier. */
91 POCL_RETURN_ERROR_ON (((arg_size == 0) && is_local), CL_INVALID_ARG_SIZE,
92 "arg_size == 0 and arg %u is in local address space\n",
93 arg_index);
94
95 POCL_RETURN_ERROR_ON (
96 ((pi->type == POCL_ARG_TYPE_SAMPLER) && (arg_value == NULL)),
97 CL_INVALID_SAMPLER, "arg_value == NULL and arg is a cl_sampler\n");
98
99 if (pi->type == POCL_ARG_TYPE_POINTER || pi->type == POCL_ARG_TYPE_IMAGE
100 || pi->type == POCL_ARG_TYPE_SAMPLER)
101 POCL_RETURN_ERROR_ON (((!is_local) && (arg_size != sizeof (cl_mem))),
102 CL_INVALID_ARG_SIZE,
103 "Arg %u is pointer/buffer/image, but arg_size is "
104 "not sizeof(cl_mem)\n",
105 arg_index);
106 else if (pi->type_size)
107 {
108 size_t as = arg_size;
109 /* handle <type>3 vectors, we accept both <type>3 and <type>4 sizes */
110 if (as % 3 == 0)
111 as = (as / 3) * 4;
112 POCL_RETURN_ERROR_ON (
113 (pi->type_size != as), CL_INVALID_ARG_SIZE,
114 "Arg %u is %s, but arg_size is not sizeof(%s) == %u\n", arg_index,
115 pi->type_name, pi->type_name, pi->type_size);
116 }
117
118 p = &(kernel->dyn_arguments[arg_index]);
119 POCL_LOCK_OBJ (kernel);
120 p->is_set = 0;
121
122 if (arg_value != NULL
123 && !(pi->type == POCL_ARG_TYPE_POINTER && *(const int *)arg_value == 0))
124 {
125 pocl_aligned_free (p->value);
126 p->value = NULL;
127
128 /* FIXME: this is a cludge to determine an acceptable alignment,
129 * we should probably extract the argument alignment from the
130 * LLVM bytecode during kernel header generation. */
131 arg_alignment = pocl_size_ceil2(arg_size);
132 if (arg_alignment >= MAX_EXTENDED_ALIGNMENT)
133 arg_alignment = MAX_EXTENDED_ALIGNMENT;
134
135 arg_alloc_size = arg_size;
136 if (arg_alloc_size < arg_alignment)
137 arg_alloc_size = arg_alignment;
138
139 value = pocl_aligned_malloc (arg_alignment, arg_alloc_size);
140 if (value == NULL)
141 {
142 POCL_UNLOCK_OBJ (kernel);
143 return CL_OUT_OF_HOST_MEMORY;
144 }
145
146 memcpy (value, arg_value, arg_size);
147 p->value = value;
148 }
149 else
150 {
151 pocl_aligned_free (p->value);
152 p->value = NULL;
153 }
154
155#if 0
156 printf(
157 "### clSetKernelArg for %s arg %d (size %u) set to %x points to %x\n",
158 kernel->name, arg_index, arg_size, p->value, *(int*)arg_value);
159#endif
160
161 p->size = arg_size;
162 p->is_set = 1;
163
164 POCL_UNLOCK_OBJ (kernel);
165 return CL_SUCCESS;
166}
