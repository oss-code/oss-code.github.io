24#include "pocl_cl.h"
25#include "pocl_image_util.h"
26
27extern CL_API_ENTRY cl_int CL_API_CALL
28POname(clGetSupportedImageFormats) (cl_context context,
29 cl_mem_flags flags,
30 cl_mem_object_type image_type,
31 cl_uint num_entries,
32 cl_image_format * image_formats,
33 cl_uint * num_image_formats)
34CL_API_SUFFIX__VERSION_1_0
35{
36 POCL_RETURN_ERROR_COND((context == NULL), CL_INVALID_CONTEXT);
37
38 POCL_RETURN_ERROR_COND((context->num_devices == 0), CL_INVALID_CONTEXT);
39
40 POCL_RETURN_ERROR_COND((num_entries == 0 && image_formats != NULL), CL_INVALID_VALUE);
41
42 cl_int idx = opencl_image_type_to_index (image_type);
43
44 POCL_RETURN_ERROR_ON ((idx < 0), CL_INVALID_VALUE,
45 "invalid image type\n");
46
47 if (image_formats != NULL)
48 {
49 if (num_entries > context->num_image_formats[idx])
50 num_entries = context->num_image_formats[idx];
51 memcpy (image_formats, context->image_formats[idx],
52 sizeof (cl_image_format) * num_entries);
53 }
54
55 if (num_image_formats != NULL)
56 {
57 *num_image_formats = context->num_image_formats[idx];
58 }
59
60 return CL_SUCCESS;
61}
