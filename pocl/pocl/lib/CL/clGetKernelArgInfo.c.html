27CL_API_ENTRY cl_int CL_API_CALL
28POname(clGetKernelArgInfo)(cl_kernel kernel ,
29 cl_uint arg_indx ,
30 cl_kernel_arg_info param_name ,
31 size_t param_value_size ,
32 void * param_value ,
33 size_t * param_value_size_ret) CL_API_SUFFIX__VERSION_1_2
34{
35 POCL_RETURN_ERROR_COND((kernel == NULL), CL_INVALID_KERNEL);
36
37 POCL_RETURN_ERROR_ON ((arg_indx >= kernel->meta->num_args),
38 CL_INVALID_ARG_INDEX,
39 "This kernel has %u args, cannot getInfo on arg %u\n",
40 (unsigned)kernel->meta->num_args, (unsigned)arg_indx);
41
42 /* pocl always uses -cl-kernel-arg-info because it needs the arg metadata, but
43 * to the user programs we should report missing arg info in case they don't
44 * request it. Piglit tests this. */
45 if (kernel->program->compiler_options)
46 POCL_RETURN_ERROR_ON (
47 (!strstr (kernel->program->compiler_options, "cl-kernel-arg-info")),
48 CL_KERNEL_ARG_INFO_NOT_AVAILABLE,
49 "argument information is not available!\n");
50
51 struct pocl_argument_info *arg = &kernel->meta->arg_info[arg_indx];
52 cl_bitfield am = kernel->meta->has_arg_metadata;
53 switch (param_name) {
54 case CL_KERNEL_ARG_ADDRESS_QUALIFIER:
55 if (!(am & POCL_HAS_KERNEL_ARG_ADDRESS_QUALIFIER))
56 return CL_KERNEL_ARG_INFO_NOT_AVAILABLE;
57 POCL_RETURN_GETINFO(cl_kernel_arg_address_qualifier, arg->address_qualifier);
58 case CL_KERNEL_ARG_ACCESS_QUALIFIER:
59 if (!(am & POCL_HAS_KERNEL_ARG_ACCESS_QUALIFIER))
60 return CL_KERNEL_ARG_INFO_NOT_AVAILABLE;
61 POCL_RETURN_GETINFO(cl_kernel_arg_access_qualifier, arg->access_qualifier);
62 case CL_KERNEL_ARG_TYPE_NAME:
63 if (!(am & POCL_HAS_KERNEL_ARG_TYPE_NAME))
64 return CL_KERNEL_ARG_INFO_NOT_AVAILABLE;
65 POCL_RETURN_GETINFO_STR(arg->type_name);
66 case CL_KERNEL_ARG_TYPE_QUALIFIER:
67 if (!(am & POCL_HAS_KERNEL_ARG_TYPE_QUALIFIER))
68 return CL_KERNEL_ARG_INFO_NOT_AVAILABLE;
69 POCL_RETURN_GETINFO(cl_kernel_arg_type_qualifier, arg->type_qualifier);
70 case CL_KERNEL_ARG_NAME:
71 if (!(am & POCL_HAS_KERNEL_ARG_NAME))
72 return CL_KERNEL_ARG_INFO_NOT_AVAILABLE;
73 POCL_RETURN_GETINFO_STR(arg->name);
74 }
75 return CL_INVALID_VALUE;
76}
