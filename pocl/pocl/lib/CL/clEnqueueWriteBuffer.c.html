29CL_API_ENTRY cl_int CL_API_CALL
30POname(clEnqueueWriteBuffer)(cl_command_queue command_queue,
31 cl_mem buffer,
32 cl_bool blocking_write,
33 size_t offset,
34 size_t size,
35 const void *ptr,
36 cl_uint num_events_in_wait_list,
37 const cl_event *event_wait_list,
38 cl_event *event) CL_API_SUFFIX__VERSION_1_0
39{
40 cl_device_id device;
41 unsigned i;
42 _cl_command_node *cmd = NULL;
43 int errcode;
44
45 POCL_RETURN_ERROR_COND((command_queue == NULL), CL_INVALID_COMMAND_QUEUE);
46
47 POCL_RETURN_ERROR_COND((buffer == NULL), CL_INVALID_MEM_OBJECT);
48
49 POCL_RETURN_ON_SUB_MISALIGN (buffer, command_queue);
50
51 POCL_RETURN_ERROR_ON((command_queue->context != buffer->context),
52 CL_INVALID_CONTEXT, "buffer and command_queue are not from the same context\n");
53
54 POCL_RETURN_ERROR_ON (
55 (buffer->flags & (CL_MEM_HOST_READ_ONLY | CL_MEM_HOST_NO_ACCESS)),
56 CL_INVALID_OPERATION,
57 "buffer has been created with CL_MEM_HOST_READ_ONLY "
58 "or CL_MEM_HOST_NO_ACCESS\n");
59
60 POCL_RETURN_ERROR_COND((ptr == NULL), CL_INVALID_VALUE);
61
62 if (pocl_buffer_boundcheck (buffer, offset, size) != CL_SUCCESS)
63 return CL_INVALID_VALUE;
64
65 POCL_CONVERT_SUBBUFFER_OFFSET (buffer, offset);
66
67 POCL_RETURN_ERROR_ON((buffer->size > command_queue->device->max_mem_alloc_size),
68 CL_OUT_OF_RESOURCES,
69 "buffer is larger than device's MAX_MEM_ALLOC_SIZE\n");
70
71 errcode = pocl_check_event_wait_list (command_queue, num_events_in_wait_list,
72 event_wait_list);
73 if (errcode != CL_SUCCESS)
74 return errcode;
75
76 POCL_CHECK_DEV_IN_CMDQ;
77
78 errcode = pocl_create_command (&cmd, command_queue,
79 CL_COMMAND_WRITE_BUFFER,
80 event, num_events_in_wait_list,
81 event_wait_list, 1, &buffer);
82 if (errcode != CL_SUCCESS)
83 return errcode;
84
85 cmd->command.write.src_host_ptr = ptr;
86 cmd->command.write.dst_mem_id = &buffer->device_ptrs[device->dev_id];
87 cmd->command.write.offset = offset;
88 cmd->command.write.size = size;
89
90 POname(clRetainMemObject) (buffer);
91 buffer->owning_device = command_queue->device;
92
93 pocl_command_enqueue (command_queue, cmd);
94
95 if (blocking_write)
96 POname(clFinish) (command_queue);
97
98 return CL_SUCCESS;
99}
