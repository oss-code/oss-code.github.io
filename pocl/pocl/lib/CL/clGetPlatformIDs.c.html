238/*
239 * Get the number of supported platforms on this system.
240 * On POCL, this trivially reduces to 1 - POCL itself.
241 */
242CL_API_ENTRY cl_int CL_API_CALL
243POname(clGetPlatformIDs)(cl_uint num_entries,
244 cl_platform_id * platforms,
245 cl_uint * num_platforms) CL_API_SUFFIX__VERSION_1_0
246{
247 POCL_RETURN_ERROR_COND ((platforms == NULL && num_entries > 0),
248 CL_INVALID_VALUE);
249
250 POCL_RETURN_ERROR_COND ((platforms != NULL && num_entries == 0),
251 CL_INVALID_VALUE);
252
253 POCL_RETURN_ERROR_COND ((num_platforms == NULL && num_entries == 0),
254 CL_INVALID_VALUE);
255
256 if (platforms != NULL) {
257 platforms[0] = &_platforms[0];
258 }
259
260 if (num_platforms != NULL)
261 *num_platforms = 1;
262
263 return CL_SUCCESS;
264}
