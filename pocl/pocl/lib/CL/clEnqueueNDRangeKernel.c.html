47/* Euclid's algorithm for the Greatest Common Divisor */
48static inline size_t
49gcd (size_t a, size_t b)
50{
51 int c;
52 while (a) {
53 c = a; a = b % a; b = c;
54 }
55 return b;
56}
57
58/* Find the largest divisor of dividend which is less than limit */
59static inline size_t
60upper_divisor (size_t dividend, size_t limit)
61{
62 /* The algorithm is currently not very smart, we
63 * start from limit and subtract until we find something
64 * that divides dividend. In optimal conditions this is found
65 * quickly, but it takes limit steps if dividend is prime.
66 * TODO FIXME improve algorithm
67 */
68 if (dividend < limit) return dividend; // small optimization
69 assert (limit > 0); // should never be called with limit == 0
70 while (dividend % limit != 0) --limit;
71 return limit;
72}
73
74/* Check that a divides b and b divides c */
75static inline int
76divide_chain (size_t a, size_t b, size_t c)
77{
78 return (b % a == 0 && c % b == 0);
79}

82CL_API_ENTRY cl_int CL_API_CALL
83POname(clEnqueueNDRangeKernel)(cl_command_queue command_queue,
84 cl_kernel kernel,
85 cl_uint work_dim,
86 const size_t *global_work_offset,
87 const size_t *global_work_size,
88 const size_t *local_work_size,
89 cl_uint num_events_in_wait_list,
90 const cl_event *event_wait_list,
91 cl_event *event) CL_API_SUFFIX__VERSION_1_0
92{
93 size_t offset_x, offset_y, offset_z;
94 size_t global_x, global_y, global_z;
95 size_t local_x, local_y, local_z;
96 offset_x = offset_y = offset_z = 0;
97 global_x = global_y = global_z = 0;
98 local_x = local_y = local_z = 0;
99 /* cached values for max_work_item_sizes,
100 * since we are going to access them repeatedly */
101 size_t max_local_x, max_local_y, max_local_z;
102 /* cached values for max_work_group_size,
103 * since we are going to access them repeatedly */
104 size_t max_group_size;
105
106 int b_migrate_count, buffer_count;
107 unsigned i;
108 int errcode = 0;
109 cl_device_id realdev = NULL;
110 _cl_command_node *command_node;
111 /* alloc from stack to avoid malloc. num_args is the absolute max needed */
112 cl_mem mem_list[kernel->meta->num_args + 1];
113 /* reserve space for potential buffer migrate events */
114 cl_event new_event_wait_list[num_events_in_wait_list + kernel->meta->num_args + 1];
115
116 POCL_RETURN_ERROR_COND((command_queue == NULL), CL_INVALID_COMMAND_QUEUE);
117
118 POCL_RETURN_ERROR_COND((kernel == NULL), CL_INVALID_KERNEL);
119
120 POCL_RETURN_ERROR_ON((command_queue->context != kernel->context),
121 CL_INVALID_CONTEXT,
122 "kernel and command_queue are not from the same context\n");
123
124 errcode = pocl_check_event_wait_list (command_queue, num_events_in_wait_list,
125 event_wait_list);
126 if (errcode != CL_SUCCESS)
127 return errcode;
128
129 POCL_RETURN_ERROR_COND((work_dim < 1), CL_INVALID_WORK_DIMENSION);
130 POCL_RETURN_ERROR_ON(
131 (work_dim > command_queue->device->max_work_item_dimensions),
132 CL_INVALID_WORK_DIMENSION,
133 "work_dim exceeds devices' max workitem dimensions\n");
134
135 assert (command_queue->device->max_work_item_dimensions <= 3);
136
137 realdev = pocl_real_dev (command_queue->device);
138
139 if (global_work_offset != NULL)
140 {
141 offset_x = global_work_offset[0];
142 offset_y = work_dim > 1 ? global_work_offset[1] : 0;
143 offset_z = work_dim > 2 ? global_work_offset[2] : 0;
144 }
145 else
146 {
147 offset_x = 0;
148 offset_y = 0;
149 offset_z = 0;
150 }
151
152 global_x = global_work_size[0];
153 global_y = work_dim > 1 ? global_work_size[1] : 1;
154 global_z = work_dim > 2 ? global_work_size[2] : 1;
155
156 POCL_RETURN_ERROR_COND((global_x == 0 || global_y == 0 || global_z == 0),
157 CL_INVALID_GLOBAL_WORK_SIZE);
158
159 max_local_x = command_queue->device->max_work_item_sizes[0];
160 max_local_y
161 = work_dim > 1 ? command_queue->device->max_work_item_sizes[1] : 1;
162 max_local_z
163 = work_dim > 2 ? command_queue->device->max_work_item_sizes[2] : 1;
164 max_group_size = command_queue->device->max_work_group_size;
165
166 if (local_work_size != NULL)
167 {
168 local_x = local_work_size[0];
169 local_y = work_dim > 1 ? local_work_size[1] : 1;
170 local_z = work_dim > 2 ? local_work_size[2] : 1;
171
172 POCL_RETURN_ERROR_ON((local_x * local_y * local_z > max_group_size),
173 CL_INVALID_WORK_GROUP_SIZE,
174 "Local worksize dimensions exceed device's max workgroup size\n");
175
176 POCL_RETURN_ERROR_ON((local_x > max_local_x),
177 CL_INVALID_WORK_ITEM_SIZE,
178 "local_work_size.x > device's max_workitem_sizes[0]\n");
179
180 if (work_dim > 1)
181 POCL_RETURN_ERROR_ON((local_y > max_local_y),
182 CL_INVALID_WORK_ITEM_SIZE,
183 "local_work_size.y > device's max_workitem_sizes[1]\n");
184
185 if (work_dim > 2)
186 POCL_RETURN_ERROR_ON((local_z > max_local_z),
187 CL_INVALID_WORK_ITEM_SIZE,
188 "local_work_size.z > device's max_workitem_sizes[2]\n");
189
190 /* TODO For full 2.x conformance the 'local must divide global'
191 * requirement will have to be limited to the cases of kernels compiled
192 * with the -cl-uniform-work-group-size option
193 */
194 POCL_RETURN_ERROR_COND((global_x % local_x != 0),
195 CL_INVALID_WORK_GROUP_SIZE);
196 POCL_RETURN_ERROR_COND((global_y % local_y != 0),
197 CL_INVALID_WORK_GROUP_SIZE);
198 POCL_RETURN_ERROR_COND((global_z % local_z != 0),
199 CL_INVALID_WORK_GROUP_SIZE);
200
201 }
202
203 /* If the kernel has the reqd_work_group_size attribute, then the local
204 * work size _must_ be specified, and it _must_ match the attribute
205 * specification
206 */
207 if (kernel->meta->reqd_wg_size[0] > 0 &&
208 kernel->meta->reqd_wg_size[1] > 0 &&
209 kernel->meta->reqd_wg_size[2] > 0)
210 {
211 POCL_RETURN_ERROR_COND ((local_work_size == NULL
212 || local_x != kernel->meta->reqd_wg_size[0]
213 || local_y != kernel->meta->reqd_wg_size[1]
214 || local_z != kernel->meta->reqd_wg_size[2]),
215 CL_INVALID_WORK_GROUP_SIZE);
216 }
217 /* otherwise, if the local work size was not specified find the optimal one.
218 * Note that at some point we also checked for local > global. This doesn't
219 * make sense while we only have 1.2 support for kernel enqueue (and
220 * when only uniform group sizes are allowed), but it might turn useful
221 * when picking the hardware sub-group size in more sophisticated
222 * 2.0 support scenarios.
223 */
224 else if (local_work_size == NULL)
225 {
226 /* Embarrassingly parallel kernel with a free work-group size. Try to
227 * figure out one which utilizes all the resources efficiently. Assume
228 * work-groups are scheduled to compute units, so try to split it to a
229 * number of work groups at the equal to the number of CUs, while still
230 * trying to respect the preferred WG size multiple (for better SIMD
231 * instruction utilization).
232 */
233 size_t preferred_wg_multiple = realdev->preferred_wg_size_multiple;
234
235 if (!preferred_wg_multiple) /* unlikely */
236 preferred_wg_multiple = 1;
237
238 POCL_MSG_PRINT_INFO("Preferred WG size multiple %zu\n",
239 preferred_wg_multiple);
240
241 /* However, we have some constraints about the local size:
242 * 1. local_{x,y,z} must divide global_{x,y,z} exactly, at least
243 * as long as we only support uniform group sizes (i.e. OpenCL 1.x);
244 * 2. each of local_{x,y,z} must be less than the corresponding max size
245 * for the device;
246 * 3. the product of local_{x,y,z} must be less than the maximum local
247 * work-group size.
248 *
249 * Due to constraint 1., we may not have the possibility to proceed by
250 * multiples of the preferred_wg_multiple (e.g. if preferred = 16 and
251 * global size = 24). Our stepping granularity in each direction will
252 * therefore be the GCD of the global size in that direction and the
253 * preferred wg size.
254 *
255 * Note that the grain might actually be as low as 1, if the two values
256 * are coprimes (e.g. preferred = 8, global size = 17). There is no good
257 * solution in this case, and there's nothing we can do about it. On the
258 * opposite side of the spectrum, we might be lucky and grain_* =
259 * preferred_wg_multiple (this is the case e.g. if the programmer already
260 * checked for the preferred wg multiple and rounded the global size up
261 * to the multiple of it).
262 */
263
264 const size_t grain_x = gcd (preferred_wg_multiple, global_x);
265 const size_t grain_y = gcd (preferred_wg_multiple, global_y);
266 const size_t grain_z = gcd (preferred_wg_multiple, global_z);
267
268 /* We now want to get the largest multiple of the grain size that still
269 * divides global_* _and_ is less than the maximum local size in each
270 * direction.
271 *
272 * So we have G = K*g and we want to find k such that k*g < M and
273 * k*g still divides G, i.e. k must divide K.
274 * The largest multiple of g that is less than M can be found as
275 * (M/g)*g (integer division), so our upper bound for k is k' = M/g.
276 */
277
278 /* /------- K ------\ /-------- k' -------\ */
279 local_x = upper_divisor (global_x / grain_x, max_local_x / grain_x);
280 local_x *= grain_x;
281
282 if (work_dim > 1)
283 {
284 local_y = upper_divisor (global_y / grain_y, max_local_y / grain_y);
285 local_y *= grain_y;
286 }
287 else
288 local_y = 1;
289
290 if (work_dim > 2)
291 {
292 local_z = upper_divisor (global_z / grain_z, max_local_z / grain_z);
293 local_z *= grain_z;
294 }
295 else
296 local_z = 1;
297
298 /* So we now have the largest possible local sizes that divide the global
299 * sizes while being multiples of the grain size.
300 * We still have to ensure that the work-group size overall is not larger
301 * than the maximum allowed, and we have to do this while preserving the
302 * 'local divides global' condition, and we would like to preserve the
303 * 'multiple of grain' too, if possible.
304 * We always reduce z first, then y, then x, on the assumption that
305 * kernels will work with x varying faster, and thus being a better
306 * vectorization candidate, followed by y and then by z. (This assumption
307 * is in some sense sanctioned by the standard itself, see e.g. the
308 * get_{global,local}_linear_id functions in OpenCL 2.x)
309 * TODO this might not be optimal in all cases. For example, devices with
310 * a hardware sampler might benefit from more evenly sized work-groups
311 * for kernels that use images. Some kind of kernel + device analysis
312 * would be needed here.
313 */
314
315 while (local_x * local_y * local_z > max_group_size)
316 {
317 /* We are going to try three strategies, in order:
318 *
319 * Halving a coordinate, if the halved coordinate is still a multiple
320 * of the grain size and a divisor of the global size.
321 *
322 * Setting the coordinates with the smallest grain to 1,
323 * since they aren't good candidates for vectorizations anyway.
324 *
325 * Setting to 1 any coordinate, as a desperate measure.
326 */
327
328#define TRY_HALVE(coord) \
329if ((local_##coord & 1) == 0 && \
330 divide_chain (grain_##coord, local_##coord/2, global_##coord)) \
331 { \
332 local_##coord /= 2; \
333 continue; \
334 }
335
336#define TRY_LEAST_GRAIN(c1, c2, c3) \
337if (local_##c1 > 1 && grain_##c1 <= grain_##c2 && grain_##c1 <= grain_##c3) \
338 { \
339 local_##c1 = 1; \
340 continue; \
341 }
342
343#define DESPERATE_CASE(coord) \
344if (local_##coord > 1) \
345 { \
346 local_##coord = 1; \
347 continue; \
348 }
349 /* Halving attempt first */
350 TRY_HALVE(z) else TRY_HALVE(y) else TRY_HALVE(x)
351
352 /* Ok no luck. Find the coordinate with the smallest grain and
353 * kill that */
354 TRY_LEAST_GRAIN(z, x, y) else
355 TRY_LEAST_GRAIN(y, z, x) else
356 TRY_LEAST_GRAIN(x, y, z)
357
358 /* No luck either? Give up, kill everything */
359 DESPERATE_CASE(z) else DESPERATE_CASE(y) else DESPERATE_CASE(x)
360#undef DESPERATE_CASE
361#undef TRY_LEAST_GRAIN
362#undef TRY_HALVE
363 }
364
365 /* We now have the largest possible local work-group size that satisfies
366 * all the hard constraints (divide global, per-dimension bound, overall
367 * bound) and our soft constraint of being as close as possible a
368 * multiple of the preferred work-group size multiple. Such a greedy
369 * algorithm minimizes the total number of work-groups. In moderate-sized
370 * launch grid, this may result in less work-groups than the number of
371 * Compute Units, with a resulting imbalance in the workload
372 * distribution. At the same time, we want to avoid too many work-groups,
373 * since some devices are penalized by such fragmentation. Finding a good
374 * balance between the two is a hard problem, and generally depends on
375 * the device as well as the kernel utilization of its resources.
376 * Lacking that, as a first step we will simply try to guarantee that we
377 * have at least one work-group per CU, as long as the local work size
378 * does not drop below a given threshold.
379 */
380
381 /* Pick a minimum work-group size of 4 times the preferred work-group
382 * size multiple, under the assumption that this would be a good
383 * candidate below which a Compute Unit will not do enough work.
384 */
385 const size_t min_group_size = 4 * preferred_wg_multiple;
386
387 /* We need the number of Compute Units in the device, since we want
388 * at least that many work-groups, if possible */
389
390 cl_uint ncus = command_queue->device->max_compute_units;
391
392 /* number of workgroups */
393 size_t nwg_x = global_x / local_x;
394 size_t nwg_y = global_y / local_y;
395 size_t nwg_z = global_z / local_z;
396
397 size_t splits; /* number of splits to bring ngws to reach ncu */
398 /* Only proceed if splitting wouldn't bring us below the minimum
399 * group size */
400 while (((splits = ncus / (nwg_x * nwg_y * nwg_z)) > 1) &&
401 (local_x * local_y * local_z > splits * min_group_size))
402 {
403 /* Very simple splitting approach: find a dimension divisible by
404 * split, and lacking that divide by something less, if possible.
405 * If we fail at splitting at all, we will try killing the smaller of
406 * the dimensions.
407 * We will set splits to 0 if we succeed in the TRY_SPLIT, so that
408 * we know that we can skip the rest.
409 * If we get to the end of the while without splitting and without
410 * killing a dimension, we bail out early because it means we
411 * couldn't do anything useful without dropping below min_group_size.
412 */
413
414#define TRY_SPLIT(coord) \
415if ((local_##coord % splits) == 0 && \
416 divide_chain (grain_##coord, local_##coord/splits, global_##coord)) \
417 { \
418 local_##coord /= splits; nwg_##coord *= splits; splits = 0; \
419 continue; \
420 }
421
422#define TRY_LEAST_DIM(c1, c2, c3) \
423if (local_##c1 > 1 && local_##c1 <= local_##c2 && local_##c1 <= local_##c3 && \
424 local_##c2*local_##c3 >= min_group_size) \
425 { \
426 local_##c1 = 1; nwg_##c1 = global_##c1; \
427 continue; \
428 }
429
430 while (splits > 1)
431 {
432 TRY_SPLIT(z) else TRY_SPLIT(y) else TRY_SPLIT(x)
433 else splits--;
434 }
435 /* When we get here, splits will be 0 if we split, 1 if we failed:
436 * in which case we will just kill one of the dimensions instead,
437 * using the same TRY_LEAST_GRAIN and DESPERATE_CASE seen before
438 */
439 if (splits == 0)
440 continue;
441
442 TRY_LEAST_DIM(z, x, y) else TRY_LEAST_DIM(y, z, x) else
443 TRY_LEAST_DIM(x, y, z) else break;
444#undef TRY_LEAST_DIM
445#undef TRY_SPLIT
446 }
447 }
448
449 POCL_MSG_PRINT_INFO("Queueing kernel %s with local size %u x %u x %u group "
450 "sizes %u x %u x %u...\n",
451 kernel->name,
452 (unsigned)local_x, (unsigned)local_y, (unsigned)local_z,
453 (unsigned)(global_x / local_x),
454 (unsigned)(global_y / local_y),
455 (unsigned)(global_z / local_z));
456
457 assert (local_x * local_y * local_z <= max_group_size);
458 assert (local_x <= max_local_x);
459 assert (local_y <= max_local_y);
460 assert (local_z <= max_local_z);
461
462 /* See TODO above for 'local must divide global' */
463 assert (global_x % local_x == 0);
464 assert (global_y % local_y == 0);
465 assert (global_z % local_z == 0);
466
467 b_migrate_count = 0;
468 buffer_count = 0;
469
470 /* count mem objects and enqueue needed mem migrations */
471 for (i = 0; i < kernel->meta->num_args; ++i)
472 {
473 struct pocl_argument_info *a = &kernel->meta->arg_info[i];
474 struct pocl_argument *al = &(kernel->dyn_arguments[i]);
475
476 POCL_GOTO_ERROR_ON ((!al->is_set),
477 CL_INVALID_KERNEL_ARGS,
478 "The %i-th kernel argument is not set!\n", i);
479
480
481 if (a->type == POCL_ARG_TYPE_IMAGE
482 || (!ARGP_IS_LOCAL (a) && a->type == POCL_ARG_TYPE_POINTER
483 && al->value != NULL))
484 {
485 cl_mem buf = *(cl_mem *) (al->value);
486
487 if (a->type == POCL_ARG_TYPE_IMAGE)
488 {
489 POCL_GOTO_ON_UNSUPPORTED_IMAGE (buf, command_queue->device);
490 }
491 else
492 {
493 POCL_GOTO_ON_SUB_MISALIGN (buf, command_queue);
494
495 if (buf->parent != NULL)
496 {
497 *(cl_mem *)(al->value) = buf->parent;
498 al->offset = buf->origin;
499 buf = buf->parent;
500 }
501 else
502 al->offset = 0;
503
504 POCL_GOTO_ERROR_ON ((buf->size > command_queue->device->max_mem_alloc_size),
505 CL_OUT_OF_RESOURCES,
506 "ARG %u: buffer is larger than "
507 "device's MAX_MEM_ALLOC_SIZE\n", i);
508 }
509
510 mem_list[buffer_count++] = buf;
511 POname(clRetainMemObject) (buf);
512 /* if buffer has no owner,
513 it has not been used yet -> just claim it */
514 if (buf->owning_device == NULL)
515 buf->owning_device = realdev;
516 /* If buffer is located located in another global memory
517 (other device), it needs to be migrated before this kernel
518 may be executed */
519 if (buf->owning_device != NULL &&
520 buf->owning_device->global_mem_id !=
521 command_queue->device->global_mem_id)
522 {
523#if DEBUG_NDRANGE
524 printf("mem migrate needed: owning dev = %d, target dev = %d\n",
525 buf->owning_device->global_mem_id,
526 command_queue->device->global_mem_id);
527#endif
528 POname(clEnqueueMigrateMemObjects)
529 (command_queue, 1, &buf, 0, 0, NULL,
530 &new_event_wait_list[b_migrate_count++]);
531 }
532 buf->owning_device = realdev;
533 }
534 }
535
536 if (num_events_in_wait_list)
537 {
538 memcpy (&new_event_wait_list[b_migrate_count], event_wait_list,
539 sizeof(cl_event) * num_events_in_wait_list);
540 }
541
542 errcode = pocl_create_command (&command_node, command_queue,
543 CL_COMMAND_NDRANGE_KERNEL, event,
544 num_events_in_wait_list + b_migrate_count,
545 (num_events_in_wait_list + b_migrate_count)?
546 new_event_wait_list : NULL,
547 buffer_count, mem_list);
548 if (errcode != CL_SUCCESS)
549 goto ERROR;
550
551
552 command_node->type = CL_COMMAND_NDRANGE_KERNEL;
553 command_node->command.run.kernel = kernel;
554 command_node->command.run.pc.local_size[0] = local_x;
555 command_node->command.run.pc.local_size[1] = local_y;
556 command_node->command.run.pc.local_size[2] = local_z;
557 command_node->command.run.pc.work_dim = work_dim;
558 command_node->command.run.pc.num_groups[0] = global_x / local_x;
559 command_node->command.run.pc.num_groups[1] = global_y / local_y;
560 command_node->command.run.pc.num_groups[2] = global_z / local_z;
561 command_node->command.run.pc.global_offset[0] = offset_x;
562 command_node->command.run.pc.global_offset[1] = offset_y;
563 command_node->command.run.pc.global_offset[2] = offset_z;
564
565 int realdev_i = pocl_cl_device_to_index (kernel->program, realdev);
566 assert (realdev_i >= 0);
567 command_node->device_i = (unsigned)realdev_i;
568 command_node->command.run.hash = kernel->meta->build_hash[realdev_i];
569
570 /* Copy the currently set kernel arguments because the same kernel
571 object can be reused for new launches with different arguments. */
572 command_node->command.run.arguments =
573 (struct pocl_argument *) malloc ((kernel->meta->num_args) *
574 sizeof (struct pocl_argument));
575
576 for (i = 0; i < kernel->meta->num_args; ++i)
577 {
578 struct pocl_argument *arg = &command_node->command.run.arguments[i];
579 size_t arg_alloc_size = kernel->dyn_arguments[i].size;
580 arg->size = arg_alloc_size;
581 arg->offset = kernel->dyn_arguments[i].offset;
582
583 if (kernel->dyn_arguments[i].value == NULL)
584 {
585 arg->value = NULL;
586 }
587 else
588 {
589 /* FIXME: this is a cludge to determine an acceptable alignment,
590 * we should probably extract the argument alignment from the
591 * LLVM bytecode during kernel header generation. */
592 size_t arg_alignment = pocl_size_ceil2 (arg_alloc_size);
593 if (arg_alignment >= MAX_EXTENDED_ALIGNMENT)
594 arg_alignment = MAX_EXTENDED_ALIGNMENT;
595 if (arg_alloc_size < arg_alignment)
596 arg_alloc_size = arg_alignment;
597
598 arg->value = pocl_aligned_malloc (arg_alignment, arg_alloc_size);
599 memcpy (arg->value, kernel->dyn_arguments[i].value, arg->size);
600 }
601 }
602
603 command_node->next = NULL;
604
605 POname(clRetainKernel) (kernel);
606
607 if (pocl_cq_profiling_enabled)
608 {
609 pocl_cq_profiling_register_event (command_node->event);
610 clRetainKernel (kernel);
611 command_node->event->meta_data->kernel = kernel;
612 }
613
