169uint32_t
170byteswap_uint32_t (uint32_t word, char should_swap)
171{
172 union word_union
173 {
174 uint32_t full_word;
175 unsigned char bytes[4];
176 } old, neww;
177 if (!should_swap) return word;
178
179 old.full_word = word;
180 neww.bytes[0] = old.bytes[3];
181 neww.bytes[1] = old.bytes[2];
182 neww.bytes[2] = old.bytes[1];
183 neww.bytes[3] = old.bytes[0];
184 return neww.full_word;
185}

205size_t
206pocl_size_ceil2(size_t x) {
207 /* Rounds up to the next highest power of two without branching and
208 * is as fast as a BSR instruction on x86, see:
209 *
210 * http://graphics.stanford.edu/~seander/bithacks.html#RoundUpPowerOf2
211 */
212 --x;
213 x |= x >> 1;
214 x |= x >> 2;
215 x |= x >> 4;
216 x |= x >> 8;
217 x |= x >> 16;
218#if SIZE_MAX > 0xFFFFFFFF
219 x |= x >> 32;
220#endif
221 return ++x;
222}

242static void*
243pocl_memalign_alloc(size_t align_width, size_t size)
244{
245 void *ptr;
246 int status;
247
248#ifdef __ANDROID__
249 ptr = memalign (align_width, size);
250 return ptr;
251#elif defined(HAVE_POSIX_MEMALIGN)
252 status = posix_memalign (&ptr, align_width, size);
253 return ((status == 0) ? ptr : NULL);
254#elif (defined(__STDC_VERSION__) && (__STDC_VERSION__ >= 201112L))
255 return aligned_alloc (align_width, size);
256#else
257#error Cannot find aligned malloc
258#endif
259}

261void *
262pocl_aligned_malloc (size_t alignment, size_t size)
263{
264#ifdef HAVE_ALIGNED_ALLOC
265 assert (alignment > 0);
266 /* make sure that size is a multiple of alignment, as posix_memalign
267 * does not perform this test, whereas aligned_alloc does */
268 if ((size & (alignment - 1)) != 0)
269 {
270 size = size | (alignment - 1);
271 size += 1;
272 }
273
274 /* posix_memalign requires alignment to be at least sizeof(void *) */
275 if (alignment < sizeof(void *))
276 alignment = sizeof(void* );
277
278 void* result;
279
280 result = pocl_memalign_alloc(alignment, size);
281 if (result == NULL)
282 {
283 errno = -1;
284 return NULL;
285 }
286
287 return result;
288
289#else
290#error Cannot find aligned malloc
291#endif
292
293#if 0
294 /* this code works in theory, but there many places in pocl
295 * where aligned memory is used in the same pointers
296 * as memory allocated by other means */
297 /* allow zero-sized allocations, force alignment to 1 */
298 if (!size)
299 alignment = 1;
300
301 /* make sure alignment is a non-zero power of two and that
302 * size is a multiple of alignment */
303 size_t mask = alignment - 1;
304 if (!alignment || ((alignment & mask) != 0) || ((size & mask) != 0))
305 {
306 errno = EINVAL;
307 return NULL;
308 }
309
310 /* allocate memory plus space for alignment header */
311 uintptr_t address = (uintptr_t)malloc(size + mask + sizeof(void *));
312 if (!address)
313 return NULL;
314
315 /* align the address, and store original pointer for future use
316 * with free in the preceding bytes */
317 uintptr_t aligned_address = (address + mask + sizeof(void *)) & ~mask;
318 void** address_ptr = (void **)(aligned_address - sizeof(void *));
319 *address_ptr = (void *)address;
320 return (void *)aligned_address;
321
322#endif
323}

340void
341pocl_lock_events_inorder (cl_event ev1, cl_event ev2)
342{
343 assert (ev1 != ev2);
344 assert (ev1->id != ev2->id);
345 if (ev1->id < ev2->id)
346 {
347 POCL_LOCK_OBJ (ev1);
348 POCL_LOCK_OBJ (ev2);
349 }
350 else
351 {
352 POCL_LOCK_OBJ (ev2);
353 POCL_LOCK_OBJ (ev1);
354 }
355}

357void
358pocl_unlock_events_inorder (cl_event ev1, cl_event ev2)
359{
360 assert (ev1 != ev2);
361 assert (ev1->id != ev2->id);
362 if (ev1->id < ev2->id)
363 {
364 POCL_UNLOCK_OBJ (ev1);
365 POCL_UNLOCK_OBJ (ev2);
366 }
367 else
368 {
369 POCL_UNLOCK_OBJ (ev2);
370 POCL_UNLOCK_OBJ (ev1);
371 }
372}

374cl_int pocl_create_event (cl_event *event, cl_command_queue command_queue,
375 cl_command_type command_type, size_t num_buffers,
376 const cl_mem *buffers, cl_context context)
377{
378 static unsigned int event_id_counter = 0;
379
380 if (context == NULL || !(context->valid))
381 return CL_INVALID_CONTEXT;
382
383 if (event != NULL)
384 {
385 *event = pocl_mem_manager_new_event ();
386 if (*event == NULL)
387 return CL_OUT_OF_HOST_MEMORY;
388
389 (*event)->context = context;
390 POname(clRetainContext) (context);
391 (*event)->queue = command_queue;
392
393 /* user events have a NULL command queue, don't retain it */
394 if (command_queue)
395 POname(clRetainCommandQueue) (command_queue);
396
397 (*event)->command_type = command_type;
398 (*event)->id = POCL_ATOMIC_INC (event_id_counter);
399 (*event)->num_buffers = num_buffers;
400
401 POCL_MSG_PRINT_EVENTS ("created event with id %d\n", (*event)->id);
402
403 if (num_buffers > 0)
404 {
405 (*event)->mem_objs = malloc (num_buffers * sizeof(cl_mem));
406 memcpy ((*event)->mem_objs, buffers, num_buffers * sizeof(cl_mem));
407 }
408 (*event)->status = CL_QUEUED;
409 }
410
411 return CL_SUCCESS;
412}

414static int
415pocl_create_event_sync (cl_event waiting_event, cl_event notifier_event)
416{
417 event_node * volatile notify_target = NULL;
418 event_node * volatile wait_list_item = NULL;
419
420 if (notifier_event == NULL)
421 return CL_SUCCESS;
422
423 POCL_MSG_PRINT_EVENTS ("create event sync: waiting %d, notifier %d\n",
424 waiting_event->id, notifier_event->id);
425
426 pocl_lock_events_inorder (waiting_event, notifier_event);
427
428 assert (notifier_event->pocl_refcount != 0);
429 assert (waiting_event != notifier_event);
430
431 LL_FOREACH (waiting_event->wait_list, wait_list_item)
432 {
433 if (wait_list_item->event == notifier_event)
434 goto FINISH;
435 }
436
437 if (notifier_event->status == CL_COMPLETE)
438 goto FINISH;
439 notify_target = pocl_mem_manager_new_event_node();
440 wait_list_item = pocl_mem_manager_new_event_node();
441 if (!notify_target || !wait_list_item)
442 return CL_OUT_OF_HOST_MEMORY;
443
444 notify_target->event = waiting_event;
445 wait_list_item->event = notifier_event;
446 LL_PREPEND (notifier_event->notify_list, notify_target);
447 LL_PREPEND (waiting_event->wait_list, wait_list_item);
448
449FINISH:
450 pocl_unlock_events_inorder (waiting_event, notifier_event);
451 return CL_SUCCESS;
452}

454cl_int pocl_create_command (_cl_command_node **cmd,
455 cl_command_queue command_queue,
456 cl_command_type command_type, cl_event *event_p,
457 cl_int num_events, const cl_event *wait_list,
458 size_t num_buffers, const cl_mem *buffers)
459{
460 int i;
461 int err;
462 cl_event *event = NULL;
463
464 if ((wait_list == NULL && num_events != 0) ||
465 (wait_list != NULL && num_events == 0))
466 {
467 assert(0);
468 return CL_INVALID_EVENT_WAIT_LIST;
469 }
470
471 for (i = 0; i < num_events; ++i)
472 {
473 if (wait_list[i] == NULL)
474 return CL_INVALID_EVENT_WAIT_LIST;
475 }
476
477 *cmd = pocl_mem_manager_new_command ();
478 if (*cmd == NULL)
479 return CL_OUT_OF_HOST_MEMORY;
480
481 (*cmd)->type = command_type;
482
483 /* If user does not provide an event pointer, create an implicit one. */
484 event = &((*cmd)->event);
485 err = pocl_create_event (event, command_queue, 0, num_buffers, buffers,
486 command_queue->context);
487
488 if (err != CL_SUCCESS)
489 {
490 POCL_MEM_FREE(*cmd);
491 return err;
492 }
493 (*event)->command_type = command_type;
494
495 /* if host application wants this commands event
496 one reference for the host and one for the runtime/driver */
497 if (event_p)
498 {
499 POCL_MSG_PRINT_EVENTS ("event pointer provided\n");
500 *event_p = *event;
501 (*event)->implicit_event = 0;
502 (*event)->pocl_refcount = 2;
503 }
504 else
505 {
506 (*event)->implicit_event = 1;
507 (*event)->pocl_refcount = 1;
508 }
509
510 (*cmd)->device = command_queue->device;
511 (*cmd)->event->command = (*cmd);
512
513 /* in case of in-order queue, synchronize to previously enqueued command
514 if available */
515 if (!(command_queue->properties & CL_QUEUE_OUT_OF_ORDER_EXEC_MODE_ENABLE))
516 {
517 POCL_LOCK_OBJ (command_queue);
518 if (command_queue->last_event.event)
519 {
520 pocl_create_event_sync ((*cmd)->event,
521 command_queue->last_event.event);
522 }
523 POCL_UNLOCK_OBJ (command_queue);
524 }
525 /* Form event synchronizations based on the given wait list */
526 for (i = 0; i < num_events; ++i)
527 {
528 cl_event wle = wait_list[i];
529 pocl_create_event_sync ((*cmd)->event, wle);
530 }
531 POCL_MSG_PRINT_EVENTS ("Created command struct (event %d, type %X)\n",
532 (*cmd)->event->id, command_type);
533 return CL_SUCCESS;
534}

536/* call with node->event UNLOCKED */
537void pocl_command_enqueue (cl_command_queue command_queue,
538 _cl_command_node *node)
539{
540 cl_event event;
541
542 POCL_LOCK_OBJ (node->event);
543 assert (node->event->status == CL_QUEUED);
544 assert (command_queue == node->event->queue);
545 POCL_UNLOCK_OBJ (node->event);
546
547 POCL_LOCK_OBJ (command_queue);
548 ++command_queue->command_count;
549 if ((node->type == CL_COMMAND_BARRIER || node->type == CL_COMMAND_MARKER) &&
550 node->command.barrier.has_wait_list == 0)
551 {
552 DL_FOREACH (command_queue->events, event)
553 {
554 pocl_create_event_sync (node->event, event);
555 }
556 }
557 if (node->type == CL_COMMAND_BARRIER)
558 command_queue->barrier = node->event;
559 else
560 {
561 if (command_queue->barrier)
562 {
563 pocl_create_event_sync (node->event, command_queue->barrier);
564 }
565 }
566 DL_APPEND (command_queue->events, node->event);
567
568 POCL_MSG_PRINT_EVENTS ("Last event id %d to CQ.\n", node->event->id);
569 command_queue->last_event.event = node->event;
570 POCL_UNLOCK_OBJ (command_queue);
571
572 POCL_LOCK_OBJ (node->event);
573 pocl_update_event_queued (node->event);
574 command_queue->device->ops->submit(node, command_queue);
575 /* node->event is unlocked by device_ops->submit */
576
577}

579/* call (and return) with node->event locked */
580void
581pocl_command_push (_cl_command_node *node,
582 _cl_command_node *volatile *ready_list,
583 _cl_command_node *volatile *pending_list)
584{
585 assert (node != NULL);
586
587 /* If the last command inserted is a barrier,
588 command is necessary not ready */
589 if ((*ready_list) != NULL && (*ready_list)->prev
590 && (*ready_list)->prev->type == CL_COMMAND_BARRIER)
591 {
592 CDL_PREPEND ((*pending_list), node);
593 return;
594 }
595 if (pocl_command_is_ready(node->event))
596 {
597 pocl_update_event_submitted (node->event);
598 CDL_PREPEND ((*ready_list), node);
599 }
600 else
601 {
602 CDL_PREPEND ((*pending_list), node);
603 }
604}

606void
607pocl_cl_mem_inherit_flags (cl_mem mem, cl_mem from_buffer, cl_mem_flags flags)
608{
609 if ((flags & CL_MEM_READ_WRITE) | (flags & CL_MEM_READ_ONLY)
610 | (flags & CL_MEM_WRITE_ONLY))
611 {
612 mem->flags = (flags & CL_MEM_READ_WRITE) | (flags & CL_MEM_READ_ONLY)
613 | (flags & CL_MEM_WRITE_ONLY);
614 }
615 else
616 {
617 mem->flags = (from_buffer->flags & CL_MEM_READ_WRITE)
618 | (from_buffer->flags & CL_MEM_READ_ONLY)
619 | (from_buffer->flags & CL_MEM_WRITE_ONLY);
620 }
621
622 if ((flags & CL_MEM_HOST_NO_ACCESS) | (flags & CL_MEM_HOST_READ_ONLY)
623 | (flags & CL_MEM_HOST_WRITE_ONLY))
624 {
625 mem->flags = mem->flags | ((flags & CL_MEM_HOST_NO_ACCESS)
626 | (flags & CL_MEM_HOST_READ_ONLY)
627 | (flags & CL_MEM_HOST_WRITE_ONLY));
628 }
629 else
630 {
631 mem->flags
632 = mem->flags | ((from_buffer->flags & CL_MEM_HOST_NO_ACCESS)
633 | (from_buffer->flags & CL_MEM_HOST_READ_ONLY)
634 | (from_buffer->flags & CL_MEM_HOST_WRITE_ONLY));
635 }
636
637 mem->flags = mem->flags | (from_buffer->flags & CL_MEM_USE_HOST_PTR)
638 | (from_buffer->flags & CL_MEM_ALLOC_HOST_PTR)
639 | (from_buffer->flags & CL_MEM_COPY_HOST_PTR);
640}

654int pocl_buffer_boundcheck_3d(const size_t buffer_size,
655 const size_t *origin,
656 const size_t *region,
657 size_t *row_pitch,
658 size_t *slice_pitch,
659 const char* prefix)
660{
661 size_t rp = *row_pitch;
662 size_t sp = *slice_pitch;
663
664 /* CL_INVALID_VALUE if row_pitch is not 0 and is less than region[0]. */
665 POCL_RETURN_ERROR_ON((rp != 0 && rp<region[0]),
666 CL_INVALID_VALUE, "%srow_pitch is not 0 and is less than region[0]\n", prefix);
667
668 if (rp == 0) rp = region[0];
669
670 /* CL_INVALID_VALUE if slice_pitch is not 0 and is less than region[1] * row_pitch
671 * or if slice_pitch is not 0 and is not a multiple of row_pitch.
672 */
673 POCL_RETURN_ERROR_ON((sp != 0 && sp < (region[1] * rp)),
674 CL_INVALID_VALUE, "%sslice_pitch is not 0 and is less than "
675 "region[1] * %srow_pitch\n", prefix, prefix);
676 POCL_RETURN_ERROR_ON((sp != 0 && (sp % rp != 0)),
677 CL_INVALID_VALUE, "%sslice_pitch is not 0 and is not a multiple "
678 "of %srow_pitch\n", prefix, prefix);
679
680 if (sp == 0) sp = region[1] * rp;
681
682 *row_pitch = rp;
683 *slice_pitch = sp;
684
685 size_t byte_offset_begin = origin[2] * sp +
686 origin[1] * rp +
687 origin[0];
688
689 size_t byte_offset_end = origin[0] + region[0]-1 +
690 rp * (origin[1] + region[1]-1) +
691 sp * (origin[2] + region[2]-1);
692
693
694 POCL_RETURN_ERROR_ON((byte_offset_begin > buffer_size), CL_INVALID_VALUE,
695 "%sorigin is outside the %sbuffer", prefix, prefix);
696 POCL_RETURN_ERROR_ON((byte_offset_end >= buffer_size), CL_INVALID_VALUE,
697 "%sorigin+region is outside the %sbuffer", prefix, prefix);
698 return CL_SUCCESS;
699}

724int pocl_buffers_overlap(cl_mem src_buffer,
725 cl_mem dst_buffer,
726 size_t src_offset,
727 size_t dst_offset,
728 size_t size) {
729 /* The regions overlap if src_offset ≤ to dst_offset ≤ to src_offset + size - 1,
730 * or if dst_offset ≤ to src_offset ≤ to dst_offset + size - 1.
731 */
732 if (src_buffer == dst_buffer) {
733 POCL_RETURN_ERROR_ON(((src_offset <= dst_offset) && (dst_offset <=
734 (src_offset + size - 1))), CL_MEM_COPY_OVERLAP, "dst_offset lies inside \
735 the src region and the src_buffer == dst_buffer");
736 POCL_RETURN_ERROR_ON(((dst_offset <= src_offset) && (src_offset <=
737 (dst_offset + size - 1))), CL_MEM_COPY_OVERLAP, "src_offset lies inside \
738 the dst region and the src_buffer == dst_buffer");
739 }
740
741 /* sub buffers overlap check */
742 if (src_buffer->parent && dst_buffer->parent &&
743 (src_buffer->parent == dst_buffer->parent)) {
744 src_offset = src_buffer->origin + src_offset;
745 dst_offset = dst_buffer->origin + dst_offset;
746
747 POCL_RETURN_ERROR_ON (((src_offset <= dst_offset)
748 && (dst_offset <= (src_offset + size - 1))),
749 CL_MEM_COPY_OVERLAP, "dst_offset lies inside \
750 the src region and src_buffer + dst_buffer are subbuffers of the same buffer");
751 POCL_RETURN_ERROR_ON (((dst_offset <= src_offset)
752 && (src_offset <= (dst_offset + size - 1))),
753 CL_MEM_COPY_OVERLAP, "src_offset lies inside \
754 the dst region and src_buffer + dst_buffer are subbuffers of the same buffer");
755
756 }
757
758 return CL_SUCCESS;
759}

783int
784check_copy_overlap(const size_t src_offset[3],
785 const size_t dst_offset[3],
786 const size_t region[3],
787 const size_t row_pitch, const size_t slice_pitch)
788{
789 const size_t src_min[] = {src_offset[0], src_offset[1], src_offset[2]};
790 const size_t src_max[] = {src_offset[0] + region[0],
791 src_offset[1] + region[1],
792 src_offset[2] + region[2]};
793 const size_t dst_min[] = {dst_offset[0], dst_offset[1], dst_offset[2]};
794 const size_t dst_max[] = {dst_offset[0] + region[0],
795 dst_offset[1] + region[1],
796 dst_offset[2] + region[2]};
797 int overlap = 1;
798 unsigned i;
799 for (i=0; i != 3; ++i)
800 {
801 overlap = overlap && (src_min[i] < dst_max[i])
802 && (src_max[i] > dst_min[i]);
803 }
804
805 size_t dst_start = dst_offset[2] * slice_pitch +
806 dst_offset[1] * row_pitch + dst_offset[0];
807 size_t dst_end = dst_start + (region[2] * slice_pitch +
808 region[1] * row_pitch + region[0]);
809 size_t src_start = src_offset[2] * slice_pitch +
810 src_offset[1] * row_pitch + src_offset[0];
811 size_t src_end = src_start + (region[2] * slice_pitch +
812 region[1] * row_pitch + region[0]);
813
814 if (!overlap)
815 {
816 size_t delta_src_x = (src_offset[0] + region[0] > row_pitch) ?
817 src_offset[0] + region[0] - row_pitch : 0;
818 size_t delta_dst_x = (dst_offset[0] + region[0] > row_pitch) ?
819 dst_offset[0] + region[0] - row_pitch : 0;
820 if ( (delta_src_x > 0 && delta_src_x > dst_offset[0]) ||
821 (delta_dst_x > 0 && delta_dst_x > src_offset[0]) )
822 {
823 if ( (src_start <= dst_start && dst_start < src_end) ||
824 (dst_start <= src_start && src_start < dst_end) )
825 overlap = 1;
826 }
827
828 if (region[2] > 1)
829 {
830 size_t src_height = slice_pitch / row_pitch;
831 size_t dst_height = slice_pitch / row_pitch;
832
833 size_t delta_src_y = (src_offset[1] + region[1] > src_height) ?
834 src_offset[1] + region[1] - src_height : 0;
835 size_t delta_dst_y = (dst_offset[1] + region[1] > dst_height) ?
836 dst_offset[1] + region[1] - dst_height : 0;
837
838 if ( (delta_src_y > 0 && delta_src_y > dst_offset[1]) ||
839 (delta_dst_y > 0 && delta_dst_y > src_offset[1]) )
840 {
841 if ( (src_start <= dst_start && dst_start < src_end) ||
842 (dst_start <= src_start && src_start < dst_end) )
843 overlap = 1;
844 }
845 }
846 }
847
848 return overlap;
849}

861/* Make a list of unique devices. If any device is a subdevice,
862 * replace with parent, then remove duplicate parents. */
863cl_device_id * pocl_unique_device_list(const cl_device_id * in, cl_uint num, cl_uint *real)
864{
865 cl_uint real_num = num;
866 cl_device_id * out = calloc(num, sizeof(cl_device_id));
867 if (!out)
868 return NULL;
869
870 unsigned i;
871 for (i=0; i < num; ++i)
872 out[i] = (in[i] ? pocl_real_dev (in[i]) : NULL);
873
874 i=1;
875 unsigned device_i=0;
876 while (i < real_num)
877 {
878 device_i=0;
879 while (device_i < i)
880 {
881 if (out[device_i] == out[i])
882 {
883 out[device_i] = out[--real_num];
884 out[real_num] = NULL;
885 }
886 else
887 device_i++;
888 }
889 i++;
890 }
891
892 *real = real_num;
893 return out;
894}

896static void
897image_format_union (const cl_image_format *dev_formats,
898 cl_uint num_dev_formats,
899 cl_image_format **context_formats,
900 cl_uint *num_context_formats)
901{
902 if ((dev_formats == NULL) || (num_dev_formats == 0))
903 return;
904
905 if ((*num_context_formats == 0) || (*context_formats == NULL))
906 {
907 // alloc & copy
908 *context_formats = malloc (sizeof (cl_image_format) * num_dev_formats);
909 memcpy (*context_formats, dev_formats,
910 sizeof (cl_image_format) * num_dev_formats);
911 *num_context_formats = num_dev_formats;
912 }
913 else
914 {
915 // realloc & merge
916 cl_uint i, j;
917 cl_uint ncf = *num_context_formats;
918 size_t size = sizeof (cl_image_format) * (num_dev_formats + ncf);
919 cl_image_format *ctf
920 = realloc (*context_formats, size );
921 assert (ctf);
922 for (i = 0; i < num_dev_formats; ++i)
923 {
924 for (j = 0; j < ncf; ++j)
925 if (memcmp (ctf + j, dev_formats + i, sizeof (cl_image_format))
926 == 0)
927 break;
928 if (j < ncf)
929 {
930 // format already in context, skip
931 continue;
932 }
933 else
934 {
935 memcpy (ctf + ncf, dev_formats + i, sizeof (cl_image_format));
936 ++ncf;
937 }
938 }
939 *context_formats = ctf;
940 *num_context_formats = ncf;
941 }
942}

944/* Setup certain info about context that comes up later in API calls */
945void
946pocl_setup_context (cl_context context)
947{
948 unsigned i, j;
949 size_t alignment = context->devices[0]->mem_base_addr_align;
950 context->max_mem_alloc_size = 0;
951 context->svm_allocdev = NULL;
952
953 memset (context->image_formats, 0, sizeof (void *) * NUM_OPENCL_IMAGE_TYPES);
954 memset (context->num_image_formats, 0,
955 sizeof (cl_uint) * NUM_OPENCL_IMAGE_TYPES);
956
957 for(i=0; i<context->num_devices; i++)
958 {
959 if (context->devices[i]->should_allocate_svm)
960 context->svm_allocdev = context->devices[i];
961
962 if (context->devices[i]->mem_base_addr_align < alignment)
963 alignment = context->devices[i]->mem_base_addr_align;
964
965 if (context->devices[i]->max_mem_alloc_size
966 > context->max_mem_alloc_size)
967 context->max_mem_alloc_size =
968 context->devices[i]->max_mem_alloc_size;
969
970 if (context->devices[i]->image_support == CL_TRUE)
971 {
972 for (j = 0; j < NUM_OPENCL_IMAGE_TYPES; ++j)
973 image_format_union (
974 context->devices[i]->image_formats[j],
975 context->devices[i]->num_image_formats[j],
976 &context->image_formats[j], &context->num_image_formats[j]);
977 }
978 }
979
980 if (context->svm_allocdev == NULL)
981 for(i=0; i<context->num_devices; i++)
982 if (DEVICE_IS_SVM_CAPABLE(context->devices[i]))
983 {
984 context->svm_allocdev = context->devices[i];
985 break;
986 }
987
988 context->min_buffer_alignment = alignment;
989}

1113/*
1114 * This replaces a simple system(), because:
1115 *
1116 * 1) system() was causing issues (gpu lockups) with HSA when
1117 * compiling code (via compile_parallel_bc_to_brig)
1118 * with OpenCL 2.0 atomics (like CalcPie from AMD SDK).
1119 * The reason of lockups is unknown (yet).
1120 *
1121 * 2) system() uses fork() which copies page table maps, and runs
1122 * out of AS when pocl has already allocated huge buffers in memory.
1123 * this happened in llvm_codegen()
1124 *
1125 * vfork() does not copy pagetables.
1126 */
1127int
1128pocl_run_command (char *const *args)
1129{
1130 POCL_MSG_PRINT_INFO ("Launching: %s\n", args[0]);
1131#ifdef HAVE_VFORK
1132 pid_t p = vfork ();
1133#elif defined(HAVE_FORK)
1134 pid_t p = fork ();
1135#else
1136#error Must have fork() or vfork() system calls for HSA
1137#endif
1138 if (p == 0)
1139 {
1140 return execv (args[0], args);
1141 }
1142 else
1143 {
1144 if (p < 0)
1145 return EXIT_FAILURE;
1146 int status;
1147 if (waitpid (p, &status, 0) < 0)
1148 POCL_ABORT ("pocl: waitpid() failed.\n");
1149 if (WIFEXITED (status))
1150 return WEXITSTATUS (status);
1151 else if (WIFSIGNALED (status))
1152 return WTERMSIG (status);
1153 else
1154 return EXIT_FAILURE;
1155 }
1156}

1278 POname (clReleaseEvent) (event);

1296/*
1297 * float 2 half / half 2 float
1298 */
1299
1300static int const shift = 13;
1301static int const shiftSign = 16;
1302
1303static int32_t const infN = 0x7F800000; /* flt32 infinity */
1304static int32_t const maxN = 0x477FE000; /* max flt16 normal as a flt32 */
1305static int32_t const minN = 0x38800000; /* min flt16 normal as a flt32 */
1306static int32_t const signN = 0x80000000; /* flt32 sign bit */
1307
1308/* static int32_t const infC = infN >> shift;
1309 * static int32_t const infC = 0x3FC00;
1310 * static int32_t const nanN = (infC + 1) << shift; // minimum flt16 nan as a flt32
1311 */
1312static int32_t const nanN = 0x7f802000;
1313/* static int32_t const maxC = maxN >> shift; */
1314static int32_t const maxC = 0x23bff;
1315/* static int32_t const minC = minN >> shift;
1316 * static int32_t const minC = 0x1c400;
1317 * static int32_t const signC = signN >> shiftSign; // flt16 sign bit
1318 */
1319static int32_t const signC = 0x40000; /* flt16 sign bit */
1320
1321static int32_t const mulN = 0x52000000; /* (1 << 23) / minN */
1322static int32_t const mulC = 0x33800000; /* minN / (1 << (23 - shift)) */
1323
1324static int32_t const subC = 0x003FF; /* max flt32 subnormal down shifted */
1325static int32_t const norC = 0x00400; /* min flt32 normal down shifted */
1326
1327/* static int32_t const maxD = infC - maxC - 1; */
1328static int32_t const maxD = 0x1c000;
1329/* static int32_t const minD = minC - subC - 1; */
1330static int32_t const minD = 0x1c000;
1331
1332typedef union
1333{
1334 float f;
1335 int32_t si;
1336 uint32_t ui;
1337} H2F_Bits;
1338
1339float
1340half_to_float (uint16_t value)
1341{
1342 H2F_Bits v;
1343 v.ui = value;
1344 int32_t sign = v.si & signC;
1345 v.si ^= sign;
1346 sign <<= shiftSign;
1347 v.si ^= ((v.si + minD) ^ v.si) & -(v.si > subC);
1348 v.si ^= ((v.si + maxD) ^ v.si) & -(v.si > maxC);
1349 H2F_Bits s;
1350 s.si = mulC;
1351 s.f *= v.si;
1352 int32_t mask = -(norC > v.si);
1353 v.si <<= shift;
1354 v.si ^= (s.si ^ v.si) & mask;
1355 v.si |= sign;
1356 return v.f;
1357}
1358
1359uint16_t
1360float_to_half (float value)
1361{
1362 H2F_Bits v, s;
1363 v.f = value;
1364 uint32_t sign = v.si & signN;
1365 v.si ^= sign;
1366 sign >>= shiftSign;
1367 s.si = mulN;
1368 s.si = s.f * v.f;
1369 v.si ^= (s.si ^ v.si) & -(minN > v.si);
