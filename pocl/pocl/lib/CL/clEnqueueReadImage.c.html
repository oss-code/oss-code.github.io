29extern CL_API_ENTRY cl_int CL_API_CALL
30POname(clEnqueueReadImage)(cl_command_queue command_queue,
31 cl_mem image,
32 cl_bool blocking_read,
33 const size_t * origin, /* [3] */
34 const size_t * region, /* [3] */
35 size_t row_pitch,
36 size_t slice_pitch,
37 void * ptr,
38 cl_uint num_events_in_wait_list,
39 const cl_event * event_wait_list,
40 cl_event * event)
41CL_API_SUFFIX__VERSION_1_0
42{
43 cl_int errcode;
44 _cl_command_node *cmd = NULL;
45
46 POCL_RETURN_ERROR_COND((command_queue == NULL), CL_INVALID_COMMAND_QUEUE);
47
48 POCL_RETURN_ERROR_COND((image == NULL), CL_INVALID_MEM_OBJECT);
49
50 POCL_RETURN_ERROR_COND((ptr == NULL), CL_INVALID_VALUE);
51
52 if (IS_IMAGE1D_BUFFER (image))
53 {
54 IMAGE1D_ORIG_REG_TO_BYTES (image, origin, region);
55 return POname (clEnqueueReadBuffer) (
56 command_queue, image,
57 blocking_read,
58 i1d_origin[0], i1d_region[0],
59 ptr,
60 num_events_in_wait_list, event_wait_list, event);
61 }
62
63 POCL_RETURN_ERROR_ON((command_queue->context != image->context),
64 CL_INVALID_CONTEXT, "image and command_queue are not from the same context\n");
65
66 POCL_RETURN_ERROR_ON ((!image->is_image), CL_INVALID_MEM_OBJECT,
67 "image argument is not an image\n");
68 POCL_RETURN_ON_UNSUPPORTED_IMAGE (image, command_queue->device);
69
70 errcode = pocl_check_event_wait_list (command_queue, num_events_in_wait_list,
71 event_wait_list);
72 if (errcode != CL_SUCCESS)
73 return errcode;
74
75 POCL_RETURN_ERROR_ON (
76 (image->flags & (CL_MEM_HOST_WRITE_ONLY | CL_MEM_HOST_NO_ACCESS)),
77 CL_INVALID_OPERATION,
78 "image has been created with CL_MEM_HOST_WRITE_ONLY "
79 "or CL_MEM_HOST_NO_ACCESS\n");
80
81 if (image->buffer)
82 POCL_RETURN_ERROR_ON (
83 (image->buffer->flags
84 & (CL_MEM_HOST_WRITE_ONLY | CL_MEM_HOST_NO_ACCESS)),
85 CL_INVALID_OPERATION,
86 "1D Image buffer has been created with CL_MEM_HOST_WRITE_ONLY "
87 "or CL_MEM_HOST_NO_ACCESS\n");
88
89 if (errcode != CL_SUCCESS)
90 return errcode;
91
92 errcode = pocl_check_image_origin_region (image, origin, region);
93 if (errcode != CL_SUCCESS)
94 return errcode;
95
96 errcode = pocl_create_command (&cmd, command_queue, CL_COMMAND_READ_IMAGE,
97 event, num_events_in_wait_list,
98 event_wait_list, 1, &image);
99 if (errcode != CL_SUCCESS)
100 {
101 POCL_MEM_FREE(cmd);
102 return errcode;
103 }
104
105 cl_device_id dev = command_queue->device;
106 cmd->command.read_image.src_mem_id = &image->device_ptrs[dev->dev_id];
107 cmd->command.read_image.dst_host_ptr = ptr;
108 cmd->command.read_image.dst_mem_id = NULL;
109
110 cmd->command.read_image.origin[0] = origin[0];
111 cmd->command.read_image.origin[1] = origin[1];
112 cmd->command.read_image.origin[2] = origin[2];
113 cmd->command.read_image.region[0] = region[0];
114 cmd->command.read_image.region[1] = region[1];
115 cmd->command.read_image.region[2] = region[2];
116
117 cmd->command.read_image.dst_row_pitch = row_pitch;
118 cmd->command.read_image.dst_slice_pitch = slice_pitch;
119 cmd->command.read_image.dst_offset = 0;
120
121 POname(clRetainMemObject) (image);
122 image->owning_device = command_queue->device;
123 pocl_command_enqueue(command_queue, cmd);
124
125 if (blocking_read)
126 POname(clFinish) (command_queue);
127
128 return errcode;
129}
