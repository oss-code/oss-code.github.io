28CL_API_ENTRY cl_int CL_API_CALL
29POname(clGetMemObjectInfo)(cl_mem memobj ,
30 cl_mem_info param_name ,
31 size_t param_value_size ,
32 void * param_value ,
33 size_t * param_value_size_ret) CL_API_SUFFIX__VERSION_1_0
34{
35 POCL_RETURN_ERROR_COND((memobj == NULL), CL_INVALID_MEM_OBJECT);
36
37 switch (param_name) {
38 case CL_MEM_TYPE:
39 POCL_RETURN_GETINFO (cl_mem_object_type, memobj->type);
40 case CL_MEM_FLAGS:
41 POCL_RETURN_GETINFO (cl_mem_flags, memobj->flags);
42 case CL_MEM_SIZE:
43 POCL_RETURN_GETINFO (size_t, memobj->size);
44 case CL_MEM_HOST_PTR:
45 if (memobj->flags & CL_MEM_USE_HOST_PTR)
46 POCL_RETURN_GETINFO (
47 void *,
48 (memobj->parent ? (memobj->parent->mem_host_ptr + memobj->origin)
49 : memobj->mem_host_ptr));
50 else
51 POCL_RETURN_GETINFO (void *, NULL);
52 case CL_MEM_MAP_COUNT:
53 POCL_RETURN_GETINFO (cl_uint, memobj->map_count);
54 case CL_MEM_REFERENCE_COUNT:
55 POCL_RETURN_GETINFO (cl_uint, memobj->pocl_refcount);
56 case CL_MEM_CONTEXT:
57 POCL_RETURN_GETINFO (cl_context, memobj->context);
58 case CL_MEM_ASSOCIATED_MEMOBJECT:
59 POCL_RETURN_GETINFO (cl_mem, memobj->parent);
60 case CL_MEM_OFFSET:
61 if (memobj->parent == NULL)
62 POCL_RETURN_GETINFO (size_t, 0);
63 else
64 POCL_RETURN_GETINFO (size_t, memobj->origin);
65 }
66 return CL_INVALID_VALUE;
67}
