29CL_API_ENTRY cl_int CL_API_CALL
30POname(clEnqueueUnmapMemObject)(cl_command_queue command_queue,
31 cl_mem memobj,
32 void * mapped_ptr,
33 cl_uint num_events_in_wait_list,
34 const cl_event * event_wait_list,
35 cl_event * event) CL_API_SUFFIX__VERSION_1_0
36{
37 int errcode;
38 cl_device_id device;
39 unsigned i;
40 mem_mapping_t *mapping = NULL;
41 _cl_command_node *cmd = NULL;
42
43 POCL_RETURN_ERROR_COND((memobj == NULL), CL_INVALID_MEM_OBJECT);
44
45 POCL_RETURN_ERROR_COND((command_queue == NULL), CL_INVALID_COMMAND_QUEUE);
46
47 POCL_RETURN_ERROR_ON((command_queue->context != memobj->context),
48 CL_INVALID_CONTEXT, "memobj and command_queue are not from the same context\n");
49
50 errcode = pocl_check_event_wait_list (command_queue, num_events_in_wait_list,
51 event_wait_list);
52 if (errcode != CL_SUCCESS)
53 return errcode;
54
55 POCL_CHECK_DEV_IN_CMDQ;
56
57 POCL_RETURN_ERROR_ON ((memobj->flags & CL_MEM_HOST_NO_ACCESS),
58 CL_INVALID_OPERATION,
59 "buffer has been created with "
60 "CL_MEM_HOST_WRITE_ONLY or CL_MEM_HOST_NO_ACCESS and "
61 "CL_MAP_READ is set in map_flags\n");
62
63 if (memobj->parent)
64 memobj = memobj->parent;
65
66 POCL_LOCK_OBJ (memobj);
67 DL_FOREACH (memobj->mappings, mapping)
68 {
69 POCL_MSG_PRINT_MEMORY (
70 "UnMap %p search Mapping: host_ptr %p offset %zu requested: %lu\n", mapped_ptr,
71 mapping->host_ptr, mapping->offset, mapping->unmap_requested);
72
73 if (mapping->host_ptr == mapped_ptr && mapping->unmap_requested == 0)
74 break;
75 }
76 if (mapping)
77 mapping->unmap_requested = 1;
78 POCL_UNLOCK_OBJ (memobj);
79 POCL_RETURN_ERROR_ON((mapping == NULL), CL_INVALID_VALUE,
80 "Could not find mapping of this memobj\n");
81
82 errcode = pocl_create_command (&cmd, command_queue,
83 CL_COMMAND_UNMAP_MEM_OBJECT,
84 event, num_events_in_wait_list,
85 event_wait_list, 1, &memobj);
86
87 if (errcode != CL_SUCCESS)
88 goto ERROR;
89
90 cmd->command.unmap.mapping = mapping;
91 cmd->command.unmap.mem_id = &memobj->device_ptrs[device->dev_id];
92
93 POname(clRetainMemObject) (memobj);
94 memobj->owning_device = command_queue->device;
95 pocl_command_enqueue(command_queue, cmd);
96
97 return CL_SUCCESS;
98
99ERROR:
100 POCL_MEM_FREE(cmd);
101 return errcode;
102}
