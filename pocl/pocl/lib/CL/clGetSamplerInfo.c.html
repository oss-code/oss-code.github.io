27CL_API_ENTRY cl_int CL_API_CALL
28POname(clGetSamplerInfo)(cl_sampler sampler ,
29 cl_sampler_info param_name ,
30 size_t param_value_size ,
31 void * param_value ,
32 size_t * param_value_size_ret ) CL_API_SUFFIX__VERSION_1_0
33{
34 POCL_RETURN_ERROR_COND ((sampler == NULL), CL_INVALID_SAMPLER);
35
36 switch (param_name)
37 {
38 case CL_SAMPLER_REFERENCE_COUNT:
39 POCL_RETURN_GETINFO (cl_uint, sampler->pocl_refcount);
40 case CL_SAMPLER_CONTEXT:
41 POCL_RETURN_GETINFO (cl_context, sampler->context);
42 case CL_SAMPLER_NORMALIZED_COORDS:
43 POCL_RETURN_GETINFO (cl_bool, sampler->normalized_coords);
44 case CL_SAMPLER_ADDRESSING_MODE:
45 POCL_RETURN_GETINFO (cl_addressing_mode, sampler->addressing_mode);
46 case CL_SAMPLER_FILTER_MODE:
47 POCL_RETURN_GETINFO (cl_filter_mode, sampler->filter_mode);
48 }
49
50 return CL_INVALID_VALUE;
51}
