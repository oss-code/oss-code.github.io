52// The old way of getting kernel metadata from "opencl.kernels" module meta.
53// LLVM < 3.9 and SPIR
54static int pocl_get_kernel_arg_module_metadata(llvm::Function *Kernel,
55 llvm::Module *input,
56 pocl_kernel_metadata_t *kernel_meta) {
57 // find the right kernel in "opencl.kernels" metadata
58 llvm::NamedMDNode *opencl_kernels = input->getNamedMetadata("opencl.kernels");
59 llvm::MDNode *kernel_metadata = nullptr;
60
61 if (!(opencl_kernels && opencl_kernels->getNumOperands()))
62 // Perhaps it is a SPIR kernel without the "opencl.kernels" metadata
63 return 1;
64
65 for (unsigned i = 0, e = opencl_kernels->getNumOperands(); i != e; ++i) {
66 llvm::MDNode *kernel_iter = opencl_kernels->getOperand(i);
67
68 llvm::Value *meta =
69 dyn_cast<llvm::ValueAsMetadata>(kernel_iter->getOperand(0))->getValue();
70 llvm::Function *temp = llvm::cast<llvm::Function>(meta);
71 if (temp->getName().compare(Kernel->getName()) == 0) {
72 kernel_metadata = kernel_iter;
73 break;
74 }
75 }
76
77 kernel_meta->arg_info = (struct pocl_argument_info *)calloc(
78 kernel_meta->num_args, sizeof(struct pocl_argument_info));
79 memset(kernel_meta->arg_info, 0,
80 sizeof(struct pocl_argument_info) * kernel_meta->num_args);
81
82 kernel_meta->has_arg_metadata = 0;
83
84 assert(kernel_metadata && "kernel NOT found in opencl.kernels metadata");
85
86 unsigned e = kernel_metadata->getNumOperands();
87 for (unsigned i = 1; i != e; ++i) {
88 llvm::MDNode *meta_node =
89 llvm::cast<MDNode>(kernel_metadata->getOperand(i));
90
91 // argument num
92 unsigned arg_num = meta_node->getNumOperands();
93#ifndef NDEBUG
94 int has_meta_for_every_arg = ((arg_num - 1) == kernel_meta->num_args);
95#endif
96
97 llvm::MDString *meta_name_node =
98 llvm::cast<MDString>(meta_node->getOperand(0));
99 std::string meta_name = meta_name_node->getString().str();
100
101 for (unsigned j = 1; j != arg_num; ++j) {
102 llvm::Value *meta_arg_value = nullptr;
103 if (isa<ValueAsMetadata>(meta_node->getOperand(j)))
104 meta_arg_value =
105 dyn_cast<ValueAsMetadata>(meta_node->getOperand(j))->getValue();
106 else if (isa<ConstantAsMetadata>(meta_node->getOperand(j)))
107 meta_arg_value =
108 dyn_cast<ConstantAsMetadata>(meta_node->getOperand(j))->getValue();
109 struct pocl_argument_info *current_arg = &kernel_meta->arg_info[j - 1];
110
111 if (meta_arg_value != nullptr && isa<ConstantInt>(meta_arg_value) &&
112 meta_name == "kernel_arg_addr_space") {
113 assert(has_meta_for_every_arg &&
114 "kernel_arg_addr_space meta incomplete");
115 kernel_meta->has_arg_metadata |= POCL_HAS_KERNEL_ARG_ADDRESS_QUALIFIER;
116 // std::cout << "is ConstantInt / kernel_arg_addr_space" << std::endl;
117 llvm::ConstantInt *m = llvm::cast<ConstantInt>(meta_arg_value);
118 unsigned long val = (unsigned long)m->getLimitedValue(UINT_MAX);
119 // We have an LLVM fixed to produce always SPIR AS ids for the argument
120 // info metadata.
121 switch (val) {
122 case 0:
123 current_arg->address_qualifier = CL_KERNEL_ARG_ADDRESS_PRIVATE;
124 break;
125 case 1:
126 current_arg->address_qualifier = CL_KERNEL_ARG_ADDRESS_GLOBAL;
127 break;
128 case 3:
129 current_arg->address_qualifier = CL_KERNEL_ARG_ADDRESS_LOCAL;
130 break;
131 case 2:
132 current_arg->address_qualifier = CL_KERNEL_ARG_ADDRESS_CONSTANT;
133 break;
134 }
135
136 } else if (isa<MDString>(meta_node->getOperand(j))) {
137 // std::cout << "is MDString" << std::endl;
138 llvm::MDString *m = llvm::cast<MDString>(meta_node->getOperand(j));
139 std::string val = m->getString().str();
140
141 if (meta_name == "kernel_arg_access_qual") {
142 assert(has_meta_for_every_arg &&
143 "kernel_arg_access_qual meta incomplete");
144 kernel_meta->has_arg_metadata |= POCL_HAS_KERNEL_ARG_ACCESS_QUALIFIER;
145 if (val == "read_write")
146 current_arg->access_qualifier = CL_KERNEL_ARG_ACCESS_READ_WRITE;
147 else if (val == "read_only")
148 current_arg->access_qualifier = CL_KERNEL_ARG_ACCESS_READ_ONLY;
149 else if (val == "write_only")
150 current_arg->access_qualifier = CL_KERNEL_ARG_ACCESS_WRITE_ONLY;
151 else if (val == "none")
152 current_arg->access_qualifier = CL_KERNEL_ARG_ACCESS_NONE;
153 else
154 std::cout << "UNKNOWN kernel_arg_access_qual value: " << val
155 << std::endl;
156 } else if (meta_name == "kernel_arg_type") {
157 assert(has_meta_for_every_arg && "kernel_arg_type meta incomplete");
158 kernel_meta->has_arg_metadata |= POCL_HAS_KERNEL_ARG_TYPE_NAME;
159 current_arg->type_name = (char *)malloc(val.size() + 1);
160 std::strcpy(current_arg->type_name, val.c_str());
161 } else if (meta_name == "kernel_arg_base_type") {
162 // may or may not be present even in SPIR
163 } else if (meta_name == "kernel_arg_type_qual") {
164 assert(has_meta_for_every_arg &&
165 "kernel_arg_type_qual meta incomplete");
166 kernel_meta->has_arg_metadata |= POCL_HAS_KERNEL_ARG_TYPE_QUALIFIER;
167 current_arg->type_qualifier = 0;
168 if (val.find("const") != std::string::npos)
169 current_arg->type_qualifier |= CL_KERNEL_ARG_TYPE_CONST;
170 if (val.find("restrict") != std::string::npos)
171 current_arg->type_qualifier |= CL_KERNEL_ARG_TYPE_RESTRICT;
172 if (val.find("volatile") != std::string::npos)
173 current_arg->type_qualifier |= CL_KERNEL_ARG_TYPE_VOLATILE;
174 } else if (meta_name == "kernel_arg_name") {
175 assert(has_meta_for_every_arg && "kernel_arg_name meta incomplete");
176 kernel_meta->has_arg_metadata |= POCL_HAS_KERNEL_ARG_NAME;
177 current_arg->name = (char *)malloc(val.size() + 1);
178 std::strcpy(current_arg->name, val.c_str());
179 } else
180 std::cout << "UNKNOWN opencl metadata name: " << meta_name
181 << std::endl;
182 } else if (meta_name != "reqd_work_group_size")
183 std::cout << "UNKNOWN opencl metadata class for: " << meta_name
184 << std::endl;
185 }
186 }
187 return 0;
188}

274// Clang 3.9 uses function metadata instead of module metadata for presenting
275// OpenCL kernel information.
276static int pocl_get_kernel_arg_function_metadata(llvm::Function *Kernel,
277 llvm::Module *input,
278 pocl_kernel_metadata_t *kernel_meta) {
279 assert(Kernel);
280
281 // SPIR still uses the "opencl.kernels" MD.
282 llvm::NamedMDNode *opencl_kernels = input->getNamedMetadata("opencl.kernels");
283 int bitcode_is_old_spir = (opencl_kernels != nullptr);
284
285 if (bitcode_is_old_spir) {
286 return pocl_get_kernel_arg_module_metadata(Kernel, input, kernel_meta);
287 }
288 // Else go on, because it might be SPIR encoded with modern LLVM
289
290 kernel_meta->has_arg_metadata = 0;
291
292 llvm::MDNode *meta_node;
293 llvm::Value *meta_arg_value = nullptr;
294 struct pocl_argument_info *current_arg = nullptr;
295
296 kernel_meta->arg_info = (struct pocl_argument_info *)calloc(
297 kernel_meta->num_args, sizeof(struct pocl_argument_info));
298 memset(kernel_meta->arg_info, 0,
299 sizeof(struct pocl_argument_info) * kernel_meta->num_args);
300
301 // kernel_arg_addr_space
302 meta_node = Kernel->getMetadata("kernel_arg_addr_space");
303 assert(meta_node != nullptr);
304 unsigned arg_num = meta_node->getNumOperands();
305#ifndef NDEBUG
306 int has_meta_for_every_arg = (arg_num == kernel_meta->num_args);
307#endif
308 for (unsigned j = 0; j < arg_num; ++j) {
309 assert(has_meta_for_every_arg && "kernel_arg_addr_space meta incomplete");
310
311 current_arg = &kernel_meta->arg_info[j];
312 kernel_meta->has_arg_metadata |= POCL_HAS_KERNEL_ARG_ADDRESS_QUALIFIER;
313 // std::cout << "is ConstantInt / kernel_arg_addr_space" << std::endl;
314 meta_arg_value =
315 dyn_cast<ConstantAsMetadata>(meta_node->getOperand(j))->getValue();
316 llvm::ConstantInt *m = llvm::cast<ConstantInt>(meta_arg_value);
317 unsigned long val = (unsigned long)m->getLimitedValue(UINT_MAX);
318
319 // We have an LLVM fixed to produce always SPIR AS ids for the argument
320 // info metadata.
321 switch (val) {
322 case 0:
323 current_arg->address_qualifier = CL_KERNEL_ARG_ADDRESS_PRIVATE;
324 break;
325 case 1:
326 current_arg->address_qualifier = CL_KERNEL_ARG_ADDRESS_GLOBAL;
327 break;
328 case 3:
329 current_arg->address_qualifier = CL_KERNEL_ARG_ADDRESS_LOCAL;
330 break;
331 case 2:
332 current_arg->address_qualifier = CL_KERNEL_ARG_ADDRESS_CONSTANT;
333 break;
334 default:
335 POCL_MSG_ERR("Unknown address space ID %lu\n", val);
336 break;
337 }
338 }
339
340 // kernel_arg_access_qual
341 meta_node = Kernel->getMetadata("kernel_arg_access_qual");
342 arg_num = meta_node->getNumOperands();
343#ifndef NDEBUG
344 has_meta_for_every_arg = (arg_num == kernel_meta->num_args);
345#endif
346 assert(has_meta_for_every_arg && "kernel_arg_access_qual meta incomplete");
347
348 for (unsigned j = 0; j < meta_node->getNumOperands(); ++j) {
349 current_arg = &kernel_meta->arg_info[j];
350 // std::cout << "is MDString" << std::endl;
351 llvm::MDString *m = llvm::cast<MDString>(meta_node->getOperand(j));
352 std::string val = m->getString().str();
353
354 assert(has_meta_for_every_arg && "kernel_arg_access_qual meta incomplete");
355 kernel_meta->has_arg_metadata |= POCL_HAS_KERNEL_ARG_ACCESS_QUALIFIER;
356 if (val == "read_write")
357 current_arg->access_qualifier = CL_KERNEL_ARG_ACCESS_READ_WRITE;
358 else if (val == "read_only")
359 current_arg->access_qualifier = CL_KERNEL_ARG_ACCESS_READ_ONLY;
360 else if (val == "write_only")
361 current_arg->access_qualifier = CL_KERNEL_ARG_ACCESS_WRITE_ONLY;
362 else if (val == "none")
363 current_arg->access_qualifier = CL_KERNEL_ARG_ACCESS_NONE;
364 else
365 std::cout << "UNKNOWN kernel_arg_access_qual value: " << val << std::endl;
366 }
367
368 // kernel_arg_type
369 meta_node = Kernel->getMetadata("kernel_arg_type");
370 assert(meta_node != nullptr);
371 arg_num = meta_node->getNumOperands();
372#ifndef NDEBUG
373 has_meta_for_every_arg = (arg_num == kernel_meta->num_args);
374#endif
375 assert(has_meta_for_every_arg && "kernel_arg_type meta incomplete");
376
377 for (unsigned j = 0; j < meta_node->getNumOperands(); ++j) {
378 llvm::MDString *m = llvm::cast<MDString>(meta_node->getOperand(j));
379 std::string val = m->getString().str();
380
381 current_arg = &kernel_meta->arg_info[j];
382 kernel_meta->has_arg_metadata |= POCL_HAS_KERNEL_ARG_TYPE_NAME;
383 current_arg->type_name = (char *)malloc(val.size() + 1);
384 if (type_size_map.find(val) != type_size_map.end())
385 current_arg->type_size = type_size_map[val];
386 else
387 current_arg->type_size = 0;
388 std::strcpy(current_arg->type_name, val.c_str());
389 }
390
391 // kernel_arg_type_qual
392 meta_node = Kernel->getMetadata("kernel_arg_type_qual");
393 arg_num = meta_node->getNumOperands();
394#ifndef NDEBUG
395 has_meta_for_every_arg = (arg_num == kernel_meta->num_args);
396#endif
397 assert(has_meta_for_every_arg && "kernel_arg_type_qual meta incomplete");
398 for (unsigned j = 0; j < meta_node->getNumOperands(); ++j) {
399 llvm::MDString *m = llvm::cast<MDString>(meta_node->getOperand(j));
400 std::string val = m->getString().str();
401
402 current_arg = &kernel_meta->arg_info[j];
403 assert(has_meta_for_every_arg && "kernel_arg_type_qual meta incomplete");
404 kernel_meta->has_arg_metadata |= POCL_HAS_KERNEL_ARG_TYPE_QUALIFIER;
405 current_arg->type_qualifier = 0;
406 if (val.find("const") != std::string::npos)
407 current_arg->type_qualifier |= CL_KERNEL_ARG_TYPE_CONST;
408 if (val.find("restrict") != std::string::npos)
409 current_arg->type_qualifier |= CL_KERNEL_ARG_TYPE_RESTRICT;
410 if (val.find("volatile") != std::string::npos)
411 current_arg->type_qualifier |= CL_KERNEL_ARG_TYPE_VOLATILE;
412 }
413
414 // kernel_arg_name
415 meta_node = Kernel->getMetadata("kernel_arg_name");
416 if (meta_node) {
417 arg_num = meta_node->getNumOperands();
418#ifndef NDEBUG
419 has_meta_for_every_arg = (arg_num == kernel_meta->num_args);
420#endif
421 assert(has_meta_for_every_arg && "kernel_arg_name meta incomplete");
422 for (unsigned j = 0; j < meta_node->getNumOperands(); ++j) {
423 llvm::MDString *m = llvm::cast<MDString>(meta_node->getOperand(j));
424 std::string val = m->getString().str();
425
426 current_arg = &kernel_meta->arg_info[j];
427 kernel_meta->has_arg_metadata |= POCL_HAS_KERNEL_ARG_NAME;
428 current_arg->name = (char *)malloc(val.size() + 1);
429 std::strcpy(current_arg->name, val.c_str());
430 }
431 }
432 else {
433 POCL_MSG_WARN("no name metadata for kernel args"
434 "(this is normal when compiling from SPIR) \n");
435 // With SPIR 2.0 generated by more modern (3.9) Clang there is no more
436 // "kernel_arg_name" metadata, so retrieve the name in another way
437 // \todo Implement walking on the arguments themselves to get the names
438 }
439
440 return 0;
441}

445int pocl_llvm_get_kernels_metadata(cl_program program, unsigned device_i) {
446
447 PoclCompilerMutexGuard lockHolder(nullptr);
448 InitializeLLVM();
449
450 unsigned i,j;
451 llvm::Module *input = nullptr;
452
453 cl_device_id Device = program->devices[device_i];
454 assert(Device->llvm_target_triplet && "Device has no target triple set");
455
456 if (program->llvm_irs != nullptr && program->llvm_irs[device_i] != nullptr)
457 input = static_cast<llvm::Module *>(program->llvm_irs[device_i]);
458 else {
459 return CL_INVALID_PROGRAM_EXECUTABLE;
460 }
461
462 DataLayout *TD = nullptr;
463 const std::string &ModuleDataLayout =
464 input->getDataLayout().getStringRepresentation();
465 assert(!ModuleDataLayout.empty());
466 TD = new DataLayout(ModuleDataLayout);
467
468 std::vector<llvm::Function *> kernels;
469
470 llvm::NamedMDNode *opencl_kernels = input->getNamedMetadata("opencl.kernels");
471 if (opencl_kernels) {
472 for (unsigned i = 0, e = opencl_kernels->getNumOperands(); i != e; ++i) {
473 llvm::MDNode *kernel_iter = opencl_kernels->getOperand(i);
474
475 llvm::Value *meta =
476 dyn_cast<llvm::ValueAsMetadata>(kernel_iter->getOperand(0))->getValue();
477 llvm::Function *kernel = llvm::cast<llvm::Function>(meta);
478 //kernel_names.push_back(kernel_prototype->getName().str());
479 kernels.push_back(kernel);
480 }
481 }
482 // LLVM 3.9 does not use opencl.kernels meta, but kernel_arg_* function meta
483 else {
484 for (llvm::Module::iterator i = input->begin(), e = input->end(); i != e; ++i) {
485 if (i->getMetadata("kernel_arg_access_qual")) {
486 kernels.push_back(&*i);
487 }
488 }
489 }
490
491 assert ((kernels.size() > 0) && "empty program - no kernels");
492 assert (kernels.size() == program->num_kernels);
493
494/**************************************************************************/
495
496 SmallVector<GlobalVariable *, 8> locals;
497
498 for (j = 0; j < program->num_kernels; ++j) {
499
500 pocl_kernel_metadata_t *meta = &program->kernel_meta[j];
501 meta->data = (void**)calloc(program->num_devices, sizeof(void*));
502
503 llvm::Function *KernelFunction = kernels[j];
504
505 meta->num_args = KernelFunction->arg_size();
506 meta->name = strdup(KernelFunction->getName().str().c_str());
507
508 if (pocl_get_kernel_arg_function_metadata(KernelFunction, input, meta)) {
509 return CL_INVALID_KERNEL;
510 }
511
512#ifdef DEBUG_POCL_LLVM_API
513 printf("### fetching kernel metadata for kernel %s program %p "
514 "input llvm::Module %p\n",
515 kernel_name, program, input);
516#endif
517
518 locals.clear();
519
520 std::string funcName = KernelFunction->getName().str();
521
522 for (llvm::Module::global_iterator i = input->global_begin(),
523 e = input->global_end();
524 i != e; ++i) {
525 if (pocl::isAutomaticLocal(funcName, *i)) {
526 POCL_MSG_PRINT_LLVM("Automatic local detected in kernel %s: %s\n",
527 meta->name, (*i).getName().data());
528 locals.push_back(&*i);
529 }
530 }
531
532 meta->num_locals = locals.size();
533 meta->local_sizes = (size_t*)calloc(locals.size(), sizeof(size_t));
534
535 /* Fill up automatic local arguments. */
536 for (unsigned i = 0; i < meta->num_locals; ++i) {
537 unsigned auto_local_size =
538 TD->getTypeAllocSize(locals[i]->getInitializer()->getType());
539 meta->local_sizes[i] = auto_local_size;
540
541 #ifdef DEBUG_POCL_LLVM_API
542 printf("### automatic local %d size %u\n", i, auto_local_size);
543 #endif
544 }
545
546 i = 0;
547 for (llvm::Function::const_arg_iterator ii = KernelFunction->arg_begin(),
548 ee = KernelFunction->arg_end();
549 ii != ee; ii++) {
550 llvm::Type *t = ii->getType();
551 struct pocl_argument_info &ArgInfo = meta->arg_info[i];
552 ArgInfo.type = POCL_ARG_TYPE_NONE;
553 const llvm::PointerType *p = dyn_cast<llvm::PointerType>(t);
554 if (p && !ii->hasByValAttr()) {
555 ArgInfo.type = POCL_ARG_TYPE_POINTER;
556 // index 0 is for function attributes, parameters start at 1.
557 // TODO: detect the address space from MD.
558 }
559 if (pocl::is_image_type(*t)) {
560 ArgInfo.type = POCL_ARG_TYPE_IMAGE;
561 } else if (pocl::is_sampler_type(*t)) {
562 ArgInfo.type = POCL_ARG_TYPE_SAMPLER;
563 }
564 i++;
565 }
566
567 std::stringstream attrstr;
568 std::string vectypehint;
569
570 size_t reqdx = 0, reqdy = 0, reqdz = 0;
571
572 llvm::MDNode *ReqdWGSize =
573 KernelFunction->getMetadata("reqd_work_group_size");
574 if (ReqdWGSize != nullptr) {
575 reqdx = (llvm::cast<ConstantInt>(
576 llvm::dyn_cast<ConstantAsMetadata>(
577 ReqdWGSize->getOperand(0))->getValue()))->getLimitedValue();
578 reqdy = (llvm::cast<ConstantInt>(
579 llvm::dyn_cast<ConstantAsMetadata>(
580 ReqdWGSize->getOperand(1))->getValue()))->getLimitedValue();
581 reqdz = (llvm::cast<ConstantInt>(
582 llvm::dyn_cast<ConstantAsMetadata>(
583 ReqdWGSize->getOperand(2))->getValue()))->getLimitedValue();
584 }
585
586 // TODO: implement vec_type_hint / work_group_size_hint attributes
587 meta->reqd_wg_size[0] = reqdx;
588 meta->reqd_wg_size[1] = reqdy;
589 meta->reqd_wg_size[2] = reqdz;
590 if (reqdx || reqdy || reqdz)
591 attrstr << "__attribute__((reqd_work_group_size("
592 << reqdx << ", " << reqdy
593 << ", " << reqdz << " )))";
594 if (vectypehint.size() > 0) {
595 if (reqdx || reqdy || reqdz)
596 attrstr << " ";
597 attrstr << "__attribute__ ((vec_type_hint (" << vectypehint << ")))";
598 }
599
600 std::string r = attrstr.str();
601 if (r.size() > 0) {
602 meta->attributes = (char *)malloc(r.size() + 1);
603 std::memcpy(meta->attributes, r.c_str(), r.size());
604 meta->attributes[r.size()] = 0;
605 } else
606 meta->attributes = nullptr;
607
608 } // for each kernel
609
610 delete TD;
611
612 return CL_SUCCESS;
613}

616unsigned pocl_llvm_get_kernel_count(cl_program program, unsigned device_i) {
617
618 PoclCompilerMutexGuard lockHolder(nullptr);
619 InitializeLLVM();
620
621 /* any device's module will do for metadata, just use first non-nullptr */
622 llvm::Module *mod = (llvm::Module *)program->llvm_irs[device_i];
623 if (mod == nullptr)
624 return 0;
625 llvm::NamedMDNode *md = mod->getNamedMetadata("opencl.kernels");
626 if (md) {
627 return md->getNumOperands();
628 }
629 // LLVM 3.9 does not use opencl.kernels meta, but kernel_arg_* function meta
630 else {
631 unsigned kernel_count = 0;
632 for (llvm::Module::iterator i = mod->begin(), e = mod->end(); i != e; ++i) {
633 if (i->getMetadata("kernel_arg_access_qual")) {
634 ++kernel_count;
635 }
636 }
637 return kernel_count;
638 }
