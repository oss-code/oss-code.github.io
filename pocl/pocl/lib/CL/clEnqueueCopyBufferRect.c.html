29CL_API_ENTRY cl_int CL_API_CALL
30POname(clEnqueueCopyBufferRect)(cl_command_queue command_queue,
31 cl_mem src_buffer,
32 cl_mem dst_buffer,
33 const size_t *src_origin,
34 const size_t *dst_origin,
35 const size_t *region,
36 size_t src_row_pitch,
37 size_t src_slice_pitch,
38 size_t dst_row_pitch,
39 size_t dst_slice_pitch,
40 cl_uint num_events_in_wait_list,
41 const cl_event *event_wait_list,
42 cl_event *event) CL_API_SUFFIX__VERSION_1_1
43{
44 _cl_command_node *cmd = NULL;
45
46 POCL_RETURN_ERROR_COND ((command_queue == NULL), CL_INVALID_COMMAND_QUEUE);
47
48 cl_int err = pocl_rect_copy (
49 command_queue,
50 CL_COMMAND_COPY_BUFFER_RECT,
51 src_buffer, CL_FALSE,
52 dst_buffer, CL_FALSE,
53 src_origin, dst_origin, region,
54 src_row_pitch, src_slice_pitch,
55 dst_row_pitch, dst_slice_pitch,
56 num_events_in_wait_list, event_wait_list,
57 event,
58 &cmd);
59
60 if (err != CL_SUCCESS)
61 return err;
62
63 size_t src_offset = 0;
64 POCL_CONVERT_SUBBUFFER_OFFSET (src_buffer, src_offset);
65
66 size_t dst_offset = 0;
67 POCL_CONVERT_SUBBUFFER_OFFSET (dst_buffer, dst_offset);
68
69 POCL_RETURN_ERROR_ON((src_buffer->size > command_queue->device->max_mem_alloc_size),
70 CL_OUT_OF_RESOURCES,
71 "src is larger than device's MAX_MEM_ALLOC_SIZE\n");
72
73 POCL_RETURN_ERROR_ON((dst_buffer->size > command_queue->device->max_mem_alloc_size),
74 CL_OUT_OF_RESOURCES,
75 "src is larger than device's MAX_MEM_ALLOC_SIZE\n");
76
77 cl_device_id dev = command_queue->device;
78 cmd->command.copy_rect.src_mem_id = &src_buffer->device_ptrs[dev->dev_id];
79 cmd->command.copy_rect.dst_mem_id = &dst_buffer->device_ptrs[dev->dev_id];
80
81 cmd->command.copy_rect.src_origin[0] = src_offset + src_origin[0];
82 cmd->command.copy_rect.src_origin[1] = src_origin[1];
83 cmd->command.copy_rect.src_origin[2] = src_origin[2];
84 cmd->command.copy_rect.dst_origin[0] = dst_offset + dst_origin[0];
85 cmd->command.copy_rect.dst_origin[1] = dst_origin[1];
86 cmd->command.copy_rect.dst_origin[2] = dst_origin[2];
87 cmd->command.copy_rect.region[0] = region[0];
88 cmd->command.copy_rect.region[1] = region[1];
89 cmd->command.copy_rect.region[2] = region[2];
90
91 POname (clRetainMemObject) (src_buffer);
92 src_buffer->owning_device = dev;
93 POname (clRetainMemObject) (dst_buffer);
94 dst_buffer->owning_device = dev;
95
96 pocl_command_enqueue (command_queue, cmd);
97
98 return CL_SUCCESS;
99}
