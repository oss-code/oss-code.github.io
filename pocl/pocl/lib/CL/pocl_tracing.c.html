54static const struct pocl_event_tracer *event_tracer = NULL;

56/* Called with event locked, and must also return with a locked event. */
57void
58pocl_event_updated (cl_event event, int status)
59{
60 event_callback_item *cb_ptr;
61
62 /* Event callback handling calls functions in the same order
63 they were added if the status matches the specified one. */
64 for (cb_ptr = event->callback_list; cb_ptr; cb_ptr = cb_ptr->next)
65 {
66 if (cb_ptr->trigger_status == status)
67 {
68 POCL_UNLOCK_OBJ (event);
69 cb_ptr->callback_function (event, cb_ptr->trigger_status,
70 cb_ptr->user_data);
71 POCL_LOCK_OBJ (event);
72 }
73 }
74
75 if (event_tracer && event_tracer->event_updated &&
76 ((1 << status) & event_trace_filter))
77 event_tracer->event_updated (event, status);
78}

80static void
81pocl_parse_event_filter ()
82{
83 const char *trace_filter;
84 char *tmp_str, *save_ptr, *token;
85
86 trace_filter = pocl_get_string_option ("POCL_TRACING_FILTER", NULL);
87 if (trace_filter == NULL)
88 return;
89
90 tmp_str = strdup (trace_filter);
91 if (tmp_str == NULL)
92 return;
93
94 event_trace_filter = 0;
95 while (1)
96 {
97 token = strtok_r (tmp_str, ",", &save_ptr);
98 if (token == NULL)
99 goto PARSE_OUT;
100 if (strcmp (token, "queued") == 0)
101 event_trace_filter |= (1 << CL_QUEUED);
102 else if (strcmp (token, "submitted") == 0)
103 event_trace_filter |= (1 << CL_SUBMITTED);
104 else if (strcmp (token, "running") == 0)
105 event_trace_filter |= (1 << CL_RUNNING);
106 else if (strcmp (token, "complete") == 0)
107 event_trace_filter |= (1 << CL_COMPLETE);
108
109 tmp_str = NULL;
110 }
111
112PARSE_OUT:
113 free (tmp_str);
114}

116void
117pocl_tracing_init ()
118{
119 const char *trace_env;
120 unsigned i;
121
122 if (tracing_initialized)
123 return;
124
125 trace_env = pocl_get_string_option ("POCL_TRACING", NULL);
126 if (trace_env == NULL)
127 goto EVENT_INIT_OUT;
128
129 /* Check if a tracer has a name matching the supplied one */
130 for (i = 0; i < POCL_TRACER_COUNT; i++)
131 {
132 if (strcmp (trace_env, pocl_event_tracers[i]->name) == 0)
133 {
134 event_tracer = pocl_event_tracers[i];
135 break;
136 }
137 }
138 if (event_tracer == NULL)
139 goto EVENT_INIT_OUT;
140
141 pocl_parse_event_filter ();
142
143 event_tracer->init ();
144
145EVENT_INIT_OUT:
146 tracing_initialized = 1;
147}

150/* Basic text logger, useful for grep/cut/sed operations
151 */
152static FILE *text_tracer_file = NULL;
153static pocl_lock_t text_tracer_lock = POCL_LOCK_INITIALIZER;
154
155static void
156text_tracer_init ()
157{
158 const char *text_tracer_output;
159
160 text_tracer_output = pocl_get_string_option ("POCL_TRACING_OPT",
161 "pocl_trace_events.log");
162 text_tracer_file = fopen (text_tracer_output, "w");
163 if (!text_tracer_file)
164 POCL_ABORT ("Failed to open text tracer output\n");
165}

167static void
168text_tracer_event_updated (cl_event event, int status)
169{
170 cl_ulong ts;
171 switch (status)
172 {
173 case CL_QUEUED:
174 ts = event->time_queue;
175 break;
176 case CL_SUBMITTED:
177 ts = event->time_submit;
178 break;
179 case CL_RUNNING:
180 ts = event->time_start;
181 break;
182 case CL_COMPLETE:
183 default:
184 ts = event->time_end;
185 }
186 _cl_command_node *node = event->command;
187 char tmp_buffer[512];
188 char *cur_buf = tmp_buffer;
189 int text_size = 0;
190
191 /* Some less integrated commands (clEnqueueReadBufferRect) do not use
192 * standard mecanism, so check node to be non null */
193 if (node == NULL)
194 return;
195
196 text_size = sprintf (cur_buf, "%"PRIu64" %s %s ", ts,
197 pocl_command_to_str (event->command_type),
198 pocl_status_to_str(event->status));
199 cur_buf += text_size;
200 /* Print more informations for some commonly used commands */
201 switch (event->command_type)
202 {
203 case CL_COMMAND_READ_BUFFER:
204 text_size += sprintf (cur_buf, "size=%" PRIuS ", host_ptr=%p\n",
205 node->command.read.size,
206 node->command.read.dst_host_ptr);
207 break;
208 case CL_COMMAND_WRITE_BUFFER:
209 text_size += sprintf (cur_buf, "size=%" PRIuS ", host_ptr=%p\n\n",
210 node->command.write.size,
211 node->command.write.src_host_ptr);
212 break;
213 case CL_COMMAND_COPY_BUFFER:
214 text_size
215 += sprintf (cur_buf, "size=%" PRIuS "\n", node->command.copy.size);
216 break;
217 case CL_COMMAND_NDRANGE_KERNEL:
218 text_size += sprintf (cur_buf, "name=%s\n",
219 node->command.run.kernel->name);
220 break;
221 case CL_COMMAND_FILL_BUFFER:
222 text_size += sprintf (cur_buf, "size=%"PRIuS"\n",
223 node->command.memfill.size);
224 break;
225 default:
226 cur_buf[0] = '\n';
227 text_size++;
228 }
229
230 /* TODO: Make text_logger less intrusive by merging it with cq_profiler.
231 Now it actually sprintfs after every event change which has a significant
232 footprint. It could just use the collected events and sprintf the log
233 after N events or atexit() to avoid the printing affecting the profile. */
234
235 POCL_LOCK (text_tracer_lock);
236 fwrite (tmp_buffer, text_size, 1, text_tracer_file);
237 POCL_UNLOCK (text_tracer_lock);
238}

266static void
267lttng_tracer_event_updated (cl_event event, int status)
268{
269 _cl_command_node *node = event->command;
270
271 if (node == NULL)
272 return;
273
274 switch (event->command_type)
275 {
276 case CL_COMMAND_NDRANGE_KERNEL:
277 tracepoint (pocl_trace, ndrange_kernel, event->id, status,
278 node->command.run.kernel->name);
279 break;
280 case CL_COMMAND_READ_BUFFER:
281 tracepoint (pocl_trace, read_buffer, event->id, status,
282 node->command.read.dst_host_ptr, node->command.read.size);
283 break;
284 case CL_COMMAND_WRITE_BUFFER:
285 tracepoint (pocl_trace, write_buffer, event->id, status,
286 node->command.write.src_host_ptr, node->command.write.size);
287 break;
288 case CL_COMMAND_COPY_BUFFER:
289 tracepoint (pocl_trace, copy_buffer, event->id, status, node->command.copy.size);
290 break;
291 case CL_COMMAND_FILL_BUFFER:
292 tracepoint (pocl_trace, fill_buffer, event->id, status, node->command.copy.size);
293 break;
294 case CL_COMMAND_MAP_BUFFER:
295 case CL_COMMAND_MAP_IMAGE:
296 tracepoint (pocl_trace, map, event->id, status, node->command.map.mapping->size);
297 break;
298 default:
299 tracepoint (pocl_trace, command, event->id, status,
300 pocl_command_to_str (event->command_type));
301 break;
302 }
303}
