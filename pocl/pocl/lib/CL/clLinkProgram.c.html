28CL_API_ENTRY cl_program CL_API_CALL
29POname (clLinkProgram) (cl_context context,
30 cl_uint num_devices,
31 const cl_device_id *device_list,
32 const char *options,
33 cl_uint num_input_programs,
34 const cl_program *input_programs,
35 void (CL_CALLBACK *pfn_notify) (cl_program program, void *user_data),
36 void *user_data,
37 cl_int *errcode_ret)
38CL_API_SUFFIX__VERSION_1_2
39{
40 int errcode; unsigned i;
41 cl_program program = NULL;
42 cl_device_id *unique_devlist = NULL;
43
44 POCL_GOTO_LABEL_COND (PFN_NOTIFY, (context == NULL), CL_INVALID_CONTEXT);
45
46 POCL_GOTO_LABEL_COND (PFN_NOTIFY, (num_input_programs == 0),
47 CL_INVALID_VALUE);
48
49 POCL_GOTO_LABEL_COND (PFN_NOTIFY, (input_programs == NULL),
50 CL_INVALID_VALUE);
51
52 POCL_GOTO_LABEL_COND (PFN_NOTIFY, (num_devices > 0 && device_list == NULL),
53 CL_INVALID_VALUE);
54 POCL_GOTO_LABEL_COND (PFN_NOTIFY, (num_devices == 0 && device_list != NULL),
55 CL_INVALID_VALUE);
56
57 for (i = 0; i < num_input_programs; i++)
58 {
59 cl_program p = input_programs[i];
60 POCL_GOTO_LABEL_ON (
61 PFN_NOTIFY,
62 ((p->binary_type != CL_PROGRAM_BINARY_TYPE_LIBRARY)
63 && (p->binary_type != CL_PROGRAM_BINARY_TYPE_COMPILED_OBJECT)),
64 CL_INVALID_OPERATION,
65 "clLinkProgram called for !library && !compiled_obj\n");
66 }
67
68 if (num_devices == 0)
69 {
70 num_devices = context->num_devices;
71 device_list = context->devices;
72 }
73 else
74 {
75 /* convert subdevices to devices and remove duplicates */
76 cl_uint real_num_devices = 0;
77 unique_devlist = pocl_unique_device_list (device_list,
78 num_devices,
79 &real_num_devices);
80 num_devices = real_num_devices;
81 device_list = unique_devlist;
82 }
83
84 program = create_program_skeleton (context, num_devices, device_list,
85 NULL, NULL, NULL, &errcode, 1);
86 if (errcode != CL_SUCCESS)
87 goto PFN_NOTIFY;
88
89 assert (num_devices == program->num_devices);
90
91 /* link the program */
92 errcode = compile_and_link_program (0, 1, program,
93 num_devices, device_list, options,
94 0, NULL, NULL,
95 num_input_programs, input_programs,
96 pfn_notify, user_data);
97
98 /* compile_and_link_program already called the callback */
99 goto ERROR;
100
101PFN_NOTIFY:
102 if (pfn_notify)
103 pfn_notify (program, user_data);
104
105ERROR:
106 POCL_MEM_FREE (unique_devlist);
107
108 if (errcode_ret)
109 *errcode_ret = errcode;
110
111 if (errcode == CL_SUCCESS)
112 {
113 return program;
114 }
115 else
116 {
117 POname (clReleaseProgram) (program);
118 return NULL;
119 }
120}
