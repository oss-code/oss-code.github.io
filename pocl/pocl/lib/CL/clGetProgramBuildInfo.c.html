31CL_API_ENTRY cl_int CL_API_CALL
32POname(clGetProgramBuildInfo)(cl_program program,
33 cl_device_id device,
34 cl_program_build_info param_name,
35 size_t param_value_size,
36 void * param_value,
37 size_t * param_value_size_ret) CL_API_SUFFIX__VERSION_1_0
38{
39 const char *str;
40
41 POCL_RETURN_ERROR_COND((program == NULL), CL_INVALID_PROGRAM);
42
43 POCL_RETURN_ERROR_COND ((device == NULL), CL_INVALID_DEVICE);
44
45 int device_i = pocl_cl_device_to_index(program, device);
46 POCL_RETURN_ERROR_ON((device_i < 0), CL_INVALID_DEVICE, "Program does not have "
47 "this device in it's device list\n");
48
49 switch (param_name) {
50 case CL_PROGRAM_BUILD_STATUS:
51 {
52 POCL_RETURN_GETINFO(cl_build_status, program->build_status);
53 }
54
55 case CL_PROGRAM_BUILD_OPTIONS:
56 {
57 str = (program->compiler_options)? program->compiler_options: "";
58 POCL_RETURN_GETINFO_STR(str);
59 }
60
61 case CL_PROGRAM_BUILD_LOG:
62 {
63 POCL_RETURN_ERROR_ON((program->build_status == CL_BUILD_NONE),
64 CL_INVALID_PROGRAM,
65 "Program was not built");
66 if (program->builtin_kernel_names != NULL)
67 {
68 POCL_RETURN_GETINFO_STR ("");
69 }
70 if (program->main_build_log[0])
71 {
72 POCL_RETURN_GETINFO_STR (program->main_build_log);
73 }
74 else if (program->build_log[device_i])
75 {
76 POCL_RETURN_GETINFO_STR (program->build_log[device_i]);
77 }
78 else
79 {
80 char *build_log = pocl_cache_read_buildlog (program, device_i);
81 if (build_log)
82 POCL_RETURN_GETINFO_STR_FREE (build_log);
83 }
84
85 POCL_RETURN_GETINFO_STR ("");
86 }
87 case CL_PROGRAM_BINARY_TYPE:
88 {
89 POCL_RETURN_GETINFO(cl_program_binary_type, program->binary_type);
90 }
91 }
92
93 return CL_INVALID_VALUE;
94}
95POsym(clGetProgramBuildInfo)
