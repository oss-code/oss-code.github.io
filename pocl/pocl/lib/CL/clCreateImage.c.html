24#include "pocl_image_util.h"
25#include "pocl_shared.h"
26#include "pocl_util.h"

28extern CL_API_ENTRY cl_mem CL_API_CALL
29POname(clCreateImage) (cl_context context,
30 cl_mem_flags flags,
31 const cl_image_format * image_format,
32 const cl_image_desc * image_desc,
33 void * host_ptr,
34 cl_int * errcode_ret)
35CL_API_SUFFIX__VERSION_1_2
36{
37 cl_mem mem = NULL;
38 unsigned i, num_devices_supporting_image = 0;
39 size_t size = 0;
40 int errcode = CL_SUCCESS;
41 int *device_image_support = NULL;
42 size_t row_pitch;
43 size_t slice_pitch;
44 int elem_size;
45 int channels;
46 size_t elem_bytes;
47 cl_int image_type_idx;
48 cl_mem_object_type image_type;
49
50 POCL_GOTO_ERROR_COND((context == NULL), CL_INVALID_CONTEXT);
51
52 POCL_GOTO_ERROR_COND((image_format == NULL), CL_INVALID_IMAGE_FORMAT_DESCRIPTOR);
53
54 POCL_GOTO_ERROR_COND((image_desc == NULL), CL_INVALID_IMAGE_DESCRIPTOR);
55
56 if (image_desc->num_mip_levels != 0 || image_desc->num_samples != 0) {
57 POCL_ABORT_UNIMPLEMENTED("clCreateImage with image_desc->num_mip_levels != 0"
58 " || image_desc->num_samples != 0 ");
59 }
60
61 image_type = image_desc->image_type;
62 image_type_idx = opencl_image_type_to_index (image_type);
63 POCL_GOTO_ERROR_ON ((image_type_idx < 0),
64 CL_INVALID_VALUE, "unknown image type\n");
65
66 /* CL_INVALID_IMAGE_SIZE if image dimensions specified in image_desc exceed
67 * the minimum maximum image dimensions described in the table of allowed
68 * values for param_name for clGetDeviceInfo FOR ALL DEVICES IN CONTEXT.
69 */
70 device_image_support = calloc (context->num_devices, sizeof (int));
71 for (i = 0; i < context->num_devices; i++)
72 {
73 cl_device_id dev = context->devices[i];
74 if (!dev->image_support)
75 continue;
76 else
77 {
78 if (pocl_check_device_supports_image (dev, image_format,
79 image_desc, image_type_idx,
80 &device_image_support[i])
81 == CL_SUCCESS)
82 {
83 /* can't break here as we need device_image_support[]
84 * for all devices */
85 ++num_devices_supporting_image;
86 }
87 }
88 }
89 POCL_GOTO_ERROR_ON ((num_devices_supporting_image == 0),
90 CL_INVALID_OPERATION,
91 "There are no devices in context that support this "
92 "image type + size \n");
93
94 pocl_get_image_information (image_format->image_channel_order,
95 image_format->image_channel_data_type,
96 &channels, &elem_size);
97 elem_bytes = (size_t)elem_size * (size_t)channels;
98
99 row_pitch = image_desc->image_row_pitch;
100 slice_pitch = image_desc->image_slice_pitch;
101
102 /* This must be 0 if host_ptr is NULL and can be either 0 or ≥
103 * image_width * size of element in bytes if host_ptr is not NULL.
104 * If host_ptr is not NULL and image_row_pitch = 0, image_row_pitch
105 * is calculated as image_width * size of element in bytes. If
106 * image_row_pitch is not 0, it must be a multiple of the
107 * image element size in bytes.
108 */
109 if (row_pitch == 0)
110 {
111 row_pitch = image_desc->image_width * elem_bytes;
112 }
113 else
114 {
115 POCL_GOTO_ERROR_COND ((row_pitch % elem_bytes), CL_INVALID_VALUE);
116 }
117
118 /* The size in bytes of each 2D slice in the 3D image or the size in bytes
119 * of each image in a 1D or 2D image array. This must be 0 if host_ptr is
120 * NULL. If host_ptr is not NULL, image_slice_pitch can be either 0 or ≥
121 * image_row_pitch * image_height for a 2D image array or 3D image and can
122 * be either 0 or ≥ image_row_pitch for a 1D image array. If host_ptr is
123 * not NULL and image_slice_pitch = 0, image_slice_pitch is calculated as
124 * image_row_pitch * image_height for a 2D image array or 3D image and
125 * image_row_pitch for a 1D image array. If image_slice_pitch is not 0,
126 * it must be a multiple of the image_row_pitch.
127 */
128
129 if (slice_pitch == 0)
130 {
131 if (image_desc->image_type == CL_MEM_OBJECT_IMAGE3D ||
132 image_desc->image_type == CL_MEM_OBJECT_IMAGE2D_ARRAY)
133 {
134 slice_pitch = row_pitch * image_desc->image_height;
135 }
136 if (image_desc->image_type == CL_MEM_OBJECT_IMAGE1D_ARRAY)
137 {
138 slice_pitch = row_pitch;
139 }
140 }
141 else
142 {
143 POCL_GOTO_ERROR_COND ((slice_pitch % row_pitch), CL_INVALID_VALUE);
144 }
145
146 if (image_desc->image_type == CL_MEM_OBJECT_IMAGE3D)
147 size = slice_pitch * image_desc->image_depth;
148
149 if (image_desc->image_type == CL_MEM_OBJECT_IMAGE2D)
150 size = row_pitch * image_desc->image_height;
151
152 if (image_desc->image_type == CL_MEM_OBJECT_IMAGE1D
153 || image_desc->image_type == CL_MEM_OBJECT_IMAGE1D_BUFFER)
154 size = row_pitch;
155
156 if (image_desc->image_type == CL_MEM_OBJECT_IMAGE2D_ARRAY
157 || image_desc->image_type == CL_MEM_OBJECT_IMAGE1D_ARRAY)
158 {
159 size = slice_pitch * image_desc->image_array_size;
160 }
161
162 /* Create buffer and fill in missing parts */
163 if (image_desc->image_type == CL_MEM_OBJECT_IMAGE1D_BUFFER)
164 {
165 POCL_GOTO_ERROR_COND ((image_desc->buffer == NULL),
166 CL_INVALID_MEM_OBJECT);
167
168 POCL_GOTO_ERROR_ON ((image_desc->buffer->parent != NULL),
169 CL_INVALID_MEM_OBJECT,
170 "pocl does not support Image 1D"
171 " Buffer over SubBuffers\n");
172
173 POCL_GOTO_ERROR_COND ((image_desc->buffer->size < size),
174 CL_INVALID_MEM_OBJECT);
175
176 mem = (cl_mem) calloc (1, sizeof (struct _cl_mem));
177 POCL_GOTO_ERROR_COND ((mem == NULL), CL_OUT_OF_HOST_MEMORY);
178 POCL_INIT_OBJECT (mem);
179
180 cl_mem b = image_desc->buffer;
181 mem->buffer = b;
182
183 mem->size = size;
184 mem->origin = 0;
185
186 mem->context = context;
187 assert (mem->context == b->context);
188
189 pocl_cl_mem_inherit_flags (mem, b, flags);
190
191 /* Retain the buffer we're referencing */
192 POname (clRetainMemObject) (b);
193
194 POCL_MSG_PRINT_MEMORY ("CREATED IMAGE: %p REF BUFFER: %p \n\n", mem,
195 b);
196 }
197 else
198 {
199 mem = POname (clCreateBuffer) (context, flags, size, host_ptr,
200 &errcode);
201 POCL_GOTO_ERROR_ON ((mem == NULL), CL_OUT_OF_HOST_MEMORY,
202 "clCreateBuffer (for backing the image) failed\n");
203
204 for (i = 0; i < context->num_devices; i++)
205 {
206 cl_device_id dev = context->devices[i];
207 if (!dev->image_support)
208 {
209 continue;
210 } // image_data[i] = NULL
211 if (dev->ops->create_image)
212 mem->device_ptrs[dev->dev_id].image_data
213 = dev->ops->create_image (dev, image_format, image_desc, mem,
214 &errcode);
215 if (errcode)
216 goto ERROR;
217 }
218
219 mem->buffer = NULL;
220 }
221
222 mem->type = image_desc->image_type;
223 mem->device_supports_this_image = device_image_support;
224 mem->is_image = CL_TRUE;
225 mem->image_width = image_desc->image_width;
226 if (image_desc->image_type == CL_MEM_OBJECT_IMAGE2D
227 || image_desc->image_type == CL_MEM_OBJECT_IMAGE2D_ARRAY
228 || image_desc->image_type == CL_MEM_OBJECT_IMAGE3D)
229 mem->image_height = image_desc->image_height;
230 else
231 mem->image_height = 0;
232 if (image_desc->image_type == CL_MEM_OBJECT_IMAGE3D)
233 mem->image_depth = image_desc->image_depth;
234 else
235 mem->image_depth = 0;
236 if (image_desc->image_type == CL_MEM_OBJECT_IMAGE2D_ARRAY
237 || image_desc->image_type == CL_MEM_OBJECT_IMAGE1D_ARRAY)
238 mem->image_array_size = image_desc->image_array_size;
239 else
240 mem->image_array_size = 0;
241 mem->image_row_pitch = row_pitch;
242 mem->image_slice_pitch = slice_pitch;
243 mem->image_channel_data_type = image_format->image_channel_data_type;
244 mem->image_channel_order = image_format->image_channel_order;
245 mem->num_mip_levels = image_desc->num_mip_levels;
246 mem->num_samples = image_desc->num_samples;
247 mem->image_channels = channels;
248 mem->image_elem_size = elem_size;
249
250#if 0
251 printf("flags = %X\n",mem->flags);
252 printf("mem_image_width %d\n", mem->image_width);
253 printf("mem_image_height %d\n", mem->image_height);
254 printf("mem_image_depth %d\n", mem->image_depth);
255 printf("mem_image_array_size %d\n", mem->image_array_size);
256 printf("mem_image_row_pitch %d\n", mem->image_row_pitch);
257 printf("mem_image_slice_pitch %d\n", mem->image_slice_pitch);
258 printf("mem_host_ptr %u\n", mem->mem_host_ptr);
259 printf("mem_image_channel_data_type %x \n",mem->image_channel_data_type);
260 printf("device_ptrs[0] %x \n \n", mem->device_ptrs[0]);
261#endif
262
263 if (errcode_ret != NULL)
264 *errcode_ret = CL_SUCCESS;
265
266 return mem;
267
268 ERROR:
269 POCL_MEM_FREE (device_image_support);
270 if (errcode_ret)
271 {
272 *errcode_ret = errcode;
273 }
274 return NULL;
275}
