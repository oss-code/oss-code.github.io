10CL_API_ENTRY cl_int CL_API_CALL
11POname(clEnqueueNativeKernel)(cl_command_queue command_queue ,
12 void (CL_CALLBACK * user_func)(void *),
13 void * args ,
14 size_t cb_args ,
15 cl_uint num_mem_objects ,
16 const cl_mem * mem_list ,
17 const void ** args_mem_loc ,
18 cl_uint num_events_in_wait_list ,
19 const cl_event * event_wait_list ,
20 cl_event * event ) CL_API_SUFFIX__VERSION_1_0
21{
22 cl_uint i = 0;
23 _cl_command_node *command_node = NULL;
24 void *args_copy = NULL;
25 cl_int errcode;
26
27 POCL_RETURN_ERROR_COND((command_queue == NULL), CL_INVALID_COMMAND_QUEUE);
28
29 POCL_RETURN_ERROR_COND((user_func == NULL), CL_INVALID_VALUE);
30
31 POCL_RETURN_ERROR_COND(((args == NULL) && (cb_args > 0 )), CL_INVALID_VALUE);
32 POCL_RETURN_ERROR_COND(((args == NULL) && (num_mem_objects > 0)), CL_INVALID_VALUE);
33
34 POCL_RETURN_ERROR_COND(((args != NULL) && (cb_args == 0)), CL_INVALID_VALUE);
35
36 POCL_RETURN_ERROR_COND(((num_mem_objects > 0) && (mem_list == NULL)), CL_INVALID_VALUE);
37 POCL_RETURN_ERROR_COND(((num_mem_objects > 0) && (args_mem_loc == NULL)), CL_INVALID_VALUE);
38
39 POCL_RETURN_ERROR_COND(((num_mem_objects == 0) && (mem_list != NULL)), CL_INVALID_VALUE);
40 POCL_RETURN_ERROR_COND(((num_mem_objects == 0) && (args_mem_loc != NULL)), CL_INVALID_VALUE);
41
42 POCL_RETURN_ERROR_ON(!(command_queue->device->execution_capabilities &
43 CL_EXEC_NATIVE_KERNEL), CL_INVALID_OPERATION, "device associated with "
44 "command_queue cannot execute the native kernel\n");
45
46 errcode = pocl_check_event_wait_list (command_queue, num_events_in_wait_list,
47 event_wait_list);
48 if (errcode != CL_SUCCESS)
49 return errcode;
50
51
52 errcode = pocl_create_command (&command_node, command_queue,
53 CL_COMMAND_NATIVE_KERNEL,
54 event, num_events_in_wait_list,
55 event_wait_list, num_mem_objects, mem_list);
56 if (errcode != CL_SUCCESS)
57 return errcode;
58
59 command_node->command.native.user_func = user_func;
60
61 /* Specification specifies that args passed to user_func is a copy of the
62 * one passed to this function */
63 if (cb_args)
64 {
65 args_copy = malloc (cb_args);
66 if (args_copy == NULL)
67 {
68 POCL_MEM_FREE(command_node);
69 return CL_OUT_OF_HOST_MEMORY;
70 }
71 memcpy (args_copy, args, cb_args);
72 }
73
74 for (i = 0; i < num_mem_objects; i++)
75 {
76 void *buf;
77 const char *loc = (const char *) args_mem_loc[i];
78 void *arg_loc;
79
80 if (mem_list[i] == NULL)
81 {
82 POCL_MEM_FREE(args_copy);
83 POCL_MEM_FREE(command_node);
84 return CL_INVALID_MEM_OBJECT;
85 }
86
87 cl_mem m = mem_list[i];
88 if (m->parent != NULL)
89 m = m->parent;
90
91 /* put the device ptr of the clmem in the argument */
92 buf = m->device_ptrs[command_queue->device->dev_id].mem_ptr;
93 m->owning_device = command_queue->device;
94 POname (clRetainMemObject) (m);
95 /* args_mem_loc is a pointer relative to the original args, since we
96 * recopy them, we must do some relocation */
97 ptrdiff_t offset = (uintptr_t) loc - (uintptr_t) args;
98
99 arg_loc = (void *) ((uintptr_t) args_copy + (uintptr_t)offset);
100
101 if (command_queue->device->address_bits == 32)
102 *((uint32_t *) arg_loc) = (uint32_t) (((uintptr_t) buf) & 0xFFFFFFFF);
103 else
104 *((uint64_t *) arg_loc) = (uint64_t) (uintptr_t) buf;
105 }
106 command_node->command.native.args = args_copy;
107 command_node->command.native.cb_args = cb_args;
108
109 pocl_command_enqueue (command_queue, command_node);
110
111 return CL_SUCCESS;
112}
