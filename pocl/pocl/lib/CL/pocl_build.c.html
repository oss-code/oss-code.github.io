95// append token, growing modded_options, if necessary, by max(strlen(token)+1, 256)
96#define APPEND_TOKEN() \
97 do \
98 { \
99 needed = strlen (token) + 1; \
100 assert (size > (i + needed)); \
101 i += needed; \
102 strcat (modded_options, token); \
103 strcat (modded_options, " "); \
104 } \
105 while (0)

107#define APPEND_TO_MAIN_BUILD_LOG(...) \
108 POCL_MSG_ERR(__VA_ARGS__); \
109 { \

111 snprintf(program->main_build_log + l, (640 - l), __VA_ARGS__); \

114#ifdef OCS_AVAILABLE
115cl_int
116program_compile_dynamic_wg_binaries (cl_program program)
117{
118 unsigned i, device_i;
119 _cl_command_node cmd;
120
121 assert(program->build_status == CL_BUILD_SUCCESS);
122 if (program->num_kernels == 0)
123 return CL_SUCCESS;
124
125 /* For binaries of other than Executable type (libraries, compiled but
126 * not linked programs, etc), do not attempt to compile the kernels. */
127 if (program->binary_type != CL_PROGRAM_BINARY_TYPE_EXECUTABLE)
128 return CL_SUCCESS;
129
130 memset(&cmd, 0, sizeof(_cl_command_node));
131 cmd.type = CL_COMMAND_NDRANGE_KERNEL;
132
133 POCL_LOCK_OBJ (program);
134
135 /* Build the dynamic WG sized parallel.bc and device specific code,
136 for each kernel & device combo. */
137 for (device_i = 0; device_i < program->num_devices; ++device_i)
138 {
139 cl_device_id device = program->devices[device_i];
140
141 /* program may not be built for some of its devices */
142 if (program->pocl_binaries[device_i] || (!program->binaries[device_i]))
143 continue;
144
145 cmd.device = device;
146 cmd.device_i = device_i;
147
148 struct _cl_kernel fake_k;
149 memset (&fake_k, 0, sizeof (fake_k));
150 fake_k.context = program->context;
151 fake_k.program = program;
152 fake_k.next = NULL;
153 cl_kernel kernel = &fake_k;
154
155 for (i=0; i < program->num_kernels; i++)
156 {
157 fake_k.meta = &program->kernel_meta[i];
158 fake_k.name = fake_k.meta->name;
159 cmd.command.run.hash = fake_k.meta->build_hash[device_i];
160
161 size_t local_x = 0, local_y = 0, local_z = 0;
162
163 if (kernel->meta->reqd_wg_size[0] > 0
164 && kernel->meta->reqd_wg_size[1] > 0
165 && kernel->meta->reqd_wg_size[2] > 0)
166 {
167 local_x = kernel->meta->reqd_wg_size[0];
168 local_y = kernel->meta->reqd_wg_size[1];
169 local_z = kernel->meta->reqd_wg_size[2];
170 }
171
172 cmd.command.run.pc.local_size[0] = local_x;
173 cmd.command.run.pc.local_size[1] = local_y;
174 cmd.command.run.pc.local_size[2] = local_z;
175
176 cmd.command.run.kernel = kernel;
177
178 /* Force generate a generic WG function to ensure generality. */
179 device->ops->compile_kernel (&cmd, kernel, device, 0);
180
181 /* Then generate a specialized one with goffset 0 since it's a very
182 common case. */
183
184 cmd.command.run.pc.global_offset[0]
185 = cmd.command.run.pc.global_offset[1]
186 = cmd.command.run.pc.global_offset[2] = 0;
187 cmd.command.run.force_large_grid_wg_func = 1;
188 device->ops->compile_kernel (&cmd, kernel, device, 1);
189 }
190 }
191
192 POCL_UNLOCK_OBJ (program);
193
194 return CL_SUCCESS;
195}

199/* options must be non-NULL.
200 * modded_options[size] + link_options are preallocated outputs
201 */
202static cl_int
203process_options (const char *options, char *modded_options, char *link_options,
204 cl_program program, int compiling, int linking,
205 int *create_library, unsigned *flush_denorms,
206 int *requires_correctly_rounded_sqrt_div,
207 int *spir_build, size_t size)
208{
209 cl_int error;
210 char *token = NULL;
211 char *saveptr = NULL;
212
213 *create_library = 0;
214 *flush_denorms = 0;
215 *requires_correctly_rounded_sqrt_div = 0;
216 *spir_build = 0;
217 int enable_link_options = 0;
218 link_options[0] = 0;
219 modded_options[0] = 0;
220 int ret_error = (linking ? (compiling ? CL_INVALID_BUILD_OPTIONS
221 : CL_INVALID_LINKER_OPTIONS)
222 : CL_INVALID_COMPILER_OPTIONS);
223
224 assert (options);
225 assert (modded_options);
226 assert (compiling || linking);
227
228 size_t i = 1; /* terminating char */
229 size_t needed = 0;
230 char *temp_options = (char*) malloc (strlen (options) + 1);
231 strcpy (temp_options, options);
232
233 token = strtok_r (temp_options, " ", &saveptr);
234 while (token != NULL)
235 {
236 /* check if parameter is supported compiler parameter */
237 if (strncmp (token, "-cl", 3) == 0 || strncmp (token, "-w", 2) == 0
238 || strncmp (token, "-Werror", 7) == 0)
239 {
240 if (strstr (cl_program_link_options, token))
241 {
242 /* when linking, only a subset of -cl* options are valid,
243 * and only with -enable-link-options */
244 if (linking && (!compiling))
245 {
246 if (!enable_link_options)
247 {
248 APPEND_TO_MAIN_BUILD_LOG (
249 "Not compiling but link options were not enabled, "
250 "therefore %s is an invalid option\n",
251 token);
252 error = ret_error;
253 goto ERROR;
254 }
255 strcat (link_options, token);
256 }
257 if (strstr (token, "-cl-denorms-are-zero"))
258 {
259 *flush_denorms = 1;
260 }
261 if (strstr (token, "-cl-fp32-correctly-rounded-divide-sqrt"))
262 {
263 *requires_correctly_rounded_sqrt_div = 1;
264 }
265 }
266 if (strstr (cl_parameters, token))
267 {
268 /* the LLVM API call pushes the parameters directly to the
269 frontend without using -Xclang */
270 }
271 else if (strstr (cl_parameters_not_yet_supported_by_clang, token))
272 {
273 APPEND_TO_MAIN_BUILD_LOG (
274 "This build option is not yet supported by clang: %s\n",
275 token);
276 token = strtok_r (NULL, " ", &saveptr);
277 continue;
278 }
279 else
280 {
281 APPEND_TO_MAIN_BUILD_LOG("Invalid build option: %s\n", token);
282 error = ret_error;
283 goto ERROR;
284 }
285 }
286 else if (strncmp (token, "-g", 2) == 0)
287 {
288 token = "-dwarf-column-info -debug-info-kind=limited " \
289 "-dwarf-version=4 -debugger-tuning=gdb";
290 }
291 else if (strncmp (token, "-D", 2) == 0 || strncmp (token, "-I", 2) == 0)
292 {
293 APPEND_TOKEN();
294 /* if there is a space in between, then next token is part
295 of the option */
296 if (strlen (token) == 2)
297 token = strtok_r (NULL, " ", &saveptr);
298 else
299 {
300 token = strtok_r (NULL, " ", &saveptr);
301 continue;
302 }
303 }
304 else if (strncmp (token, "-x", 2) == 0 && strlen (token) == 2)
305 {
306 /* only "-x spir" is valid for the "-x" option */
307 token = strtok_r (NULL, " ", &saveptr);
308 if (!token || strncmp (token, "spir", 4) != 0)
309 {
310 APPEND_TO_MAIN_BUILD_LOG (
311 "Invalid parameter to -x build option\n");
312 error = ret_error;
313 goto ERROR;
314 }
315 /* "-x spir" is not valid if we are building from source */
316 else if (program->source)
317 {
318 APPEND_TO_MAIN_BUILD_LOG (
319 "\"-x spir\" is not valid when building from source\n");
320 error = ret_error;
321 goto ERROR;
322 }
323 else
324 *spir_build = 1;
325 token = strtok_r (NULL, " ", &saveptr);
326 continue;
327 }
328 else if (strncmp (token, "-spir-std=1.2", 13) == 0)
329 {
330 /* "-spir-std=" flags are not valid when building from source */
331 if (program->source)
332 {
333 APPEND_TO_MAIN_BUILD_LOG ("\"-spir-std=\" flag is not valid "
334 "when building from source\n");
335 error = ret_error;
336 goto ERROR;
337 }
338 else
339 *spir_build = 1;
340 token = strtok_r (NULL, " ", &saveptr);
341 continue;
342 }
343 else if (strncmp (token, "-create-library", 15) == 0)
344 {
345 if (!linking)
346 {
347 APPEND_TO_MAIN_BUILD_LOG (
348 "\"-create-library\" flag is only valid when linking\n");
349 error = ret_error;
350 goto ERROR;
351 }
352 *create_library = 1;
353 token = strtok_r (NULL, " ", &saveptr);
354 continue;
355 }
356 else if (strncmp (token, "-enable-link-options", 20) == 0)
357 {
358 if (!linking)
359 {
360 APPEND_TO_MAIN_BUILD_LOG ("\"-enable-link-options\" flag is "
361 "only valid when linking\n");
362 error = ret_error;
363 goto ERROR;
364 }
365 if (!(*create_library))
366 {
367 APPEND_TO_MAIN_BUILD_LOG ("\"-enable-link-options\" flag is "
368 "only valid when -create-library "
369 "option was given\n");
370 error = ret_error;
371 goto ERROR;
372 }
373 enable_link_options = 1;
374 token = strtok_r (NULL, " ", &saveptr);
375 continue;
376 }
377 else
378 {
379 APPEND_TO_MAIN_BUILD_LOG ("Invalid build option: %s\n", token);
380 error = ret_error;
381 goto ERROR;
382 }
383 APPEND_TOKEN ();
384 token = strtok_r (NULL, " ", &saveptr);
385 }
386
387 error = CL_SUCCESS;
388
389 /* remove trailing whitespace */
390 i = strlen (modded_options);
391 if ((i > 0) && (modded_options[i - 1] == ' '))
392 modded_options[i - 1] = 0;
393ERROR:
394 POCL_MEM_FREE (temp_options);
395 return error;
396}

400static void
401free_meta (cl_program program)
402{
403 size_t i;
404 unsigned j;
405
406 if (program->num_kernels)
407 {
408 for (i = 0; i < program->num_kernels; i++)
409 {
410 pocl_kernel_metadata_t *meta = &program->kernel_meta[i];
411 POCL_MEM_FREE (meta->build_hash);
412 if (meta->builtin_kernel)
413 continue;
414 POCL_MEM_FREE (meta->attributes);
415 POCL_MEM_FREE (meta->name);
416 for (j = 0; j < meta->num_args; ++j)
417 {
418 POCL_MEM_FREE (meta->arg_info[j].name);
419 POCL_MEM_FREE (meta->arg_info[j].type_name);
420 }
421 POCL_MEM_FREE (meta->arg_info);
422 for (j = 0; j < program->num_devices; ++j)
423 if (meta->data[j] != NULL)
424 meta->data[j] = NULL; // TODO free data in driver callback
425 POCL_MEM_FREE (meta->data);
426 POCL_MEM_FREE (meta->local_sizes);
427 }
428 POCL_MEM_FREE (program->kernel_meta);
429 }
430}

432static void
433clean_program_on_rebuild (cl_program program)
434{
435 /* if we're rebuilding the program, release the kernels and reset log/status
436 */
437 size_t i;
438 if ((program->build_status != CL_BUILD_NONE) || program->num_kernels > 0)
439 {
440 /* Spec says:
441 CL_INVALID_OPERATION if there are kernel objects attached to program.
442 ...and we check for that earlier.
443 */
444 assert (program->kernels == NULL);
445
446 free_meta (program);
447
448 program->num_kernels = 0;
449 program->build_status = CL_BUILD_NONE;
450
451 for (i = 0; i < program->num_devices; ++i)
452 {
453 POCL_MEM_FREE (program->build_log[i]);
454 memset (program->build_hash[i], 0, sizeof (SHA1_digest_t));
455 if (program->source)
456 {
457 POCL_MEM_FREE (program->binaries[i]);
458 program->binary_sizes[i] = 0;
459#ifdef OCS_AVAILABLE
460 if (program->llvm_irs[i])
461 pocl_free_llvm_irs (program, i);
462#endif
463 POCL_MEM_FREE (program->pocl_binaries[i]);
464 program->pocl_binary_sizes[i] = 0;
465 }
466 }
467 program->main_build_log[0] = 0;
468 }
469}

472cl_int
473compile_and_link_program(int compile_program,
474 int link_program,
475 cl_program program,
476 cl_uint num_devices,
477 const cl_device_id *device_list,
478 const char *options,
479 cl_uint num_input_headers,
480 const cl_program *input_headers,
481 const char **header_include_names,
482 cl_uint num_input_programs,
483 const cl_program *input_programs,
484 void (CL_CALLBACK *pfn_notify) (cl_program program,
485 void *user_data),
486 void *user_data)
487{
488 char program_bc_path[POCL_FILENAME_LENGTH];
489 char link_options[512];
490 int errcode, error;
491 int create_library = 0;
492 int requires_cr_sqrt_div = 0;
493 int spir_build = 0;
494 unsigned flush_denorms = 0;
495 uint64_t fsize;
496 cl_device_id *unique_devlist = NULL;
497 char *binary = NULL;
498 unsigned device_i = 0, actually_built = 0;
499 size_t i, j;
500 char *temp_options = NULL;
501 const char *extra_build_options =
502 pocl_get_string_option ("POCL_EXTRA_BUILD_FLAGS", NULL);
503 int build_error_code
504 = (link_program ? CL_BUILD_PROGRAM_FAILURE : CL_COMPILE_PROGRAM_FAILURE);
505
506 POCL_GOTO_LABEL_COND (PFN_NOTIFY, (program == NULL), CL_INVALID_PROGRAM);
507
508 POCL_GOTO_LABEL_COND (PFN_NOTIFY, (num_devices > 0 && device_list == NULL),
509 CL_INVALID_VALUE);
510 POCL_GOTO_LABEL_COND (PFN_NOTIFY, (num_devices == 0 && device_list != NULL),
511 CL_INVALID_VALUE);
512
513 POCL_GOTO_LABEL_COND (PFN_NOTIFY, (pfn_notify == NULL && user_data != NULL),
514 CL_INVALID_VALUE);
515
516 POCL_GOTO_LABEL_ON (PFN_NOTIFY, program->kernels, CL_INVALID_OPERATION,
517 "Program already has kernels\n");
518
519 POCL_GOTO_LABEL_ON (
520 PFN_NOTIFY,
521 (program->source == NULL && program->binaries == NULL
522 && program->builtin_kernel_names == NULL),
523 CL_INVALID_PROGRAM,
524 "Program doesn't have sources, binaries nor builtin-kernel names. You "
525 "need "
526 "to call clCreateProgramWith{Binary|Source|BuiltinKernels} first\n");
527
528 POCL_GOTO_LABEL_ON (PFN_NOTIFY,
529 ((program->source == NULL) && (link_program == 0)),
530 CL_INVALID_OPERATION,
531 "Cannot clCompileProgram when program has no source\n");
532
533 POCL_LOCK_OBJ (program);
534
535 program->main_build_log[0] = 0;
536
537 /* TODO this should be somehow utilized at linking */
538 POCL_MEM_FREE (program->compiler_options);
539
540 if (extra_build_options)
541 {
542 size_t len = (options != NULL) ? strlen (options) : 0;
543 len += strlen (extra_build_options) + 2;
544 temp_options = (char *)malloc (len);
545
546 temp_options[0] = 0;
547 if (options != NULL)
548 {
549 strcpy (temp_options, options);
550 strcat (temp_options, " ");
551 }
552 strcat (temp_options, extra_build_options);
553 }
554 else
555 temp_options = (char*) options;
556
557 if (temp_options)
558 {
559 i = strlen (temp_options);
560 size_t size = i + 512; /* add some space for pocl-added options */
561 program->compiler_options = (char *)malloc (size);
562 errcode = process_options (temp_options, program->compiler_options,
563 link_options, program, compile_program,
564 link_program, &create_library, &flush_denorms,
565 &requires_cr_sqrt_div, &spir_build, size);
566 if (errcode != CL_SUCCESS)
567 goto ERROR_CLEAN_OPTIONS;
568 }
569
570 POCL_MSG_PRINT_LLVM ("building program with options %s\n",
571 program->compiler_options);
572
573
574 program->flush_denorms = flush_denorms;
575#if !(defined(__x86_64__) && defined(__GNUC__))
576 if (flush_denorms)
577 {
578 POCL_MSG_WARN ("flush to zero is currently only implemented for "
579 "x86-64 & gcc/clang, ignoring flag\n");
580 }
581#endif
582
583 /* DEVICE LIST */
584 if (num_devices == 0)
585 {
586 num_devices = program->num_devices;
587 device_list = program->devices;
588 }
589 else
590 {
591 // convert subdevices to devices and remove duplicates
592 cl_uint real_num_devices = 0;
593 unique_devlist = pocl_unique_device_list (device_list, num_devices,
594 &real_num_devices);
595 num_devices = real_num_devices;
596 device_list = unique_devlist;
597 }
598
599 clean_program_on_rebuild (program);
600
601 /* Build the fully linked non-parallel bitcode for all
602 devices. */
603 for (device_i = 0; device_i < program->num_devices; ++device_i)
604 {
605 cl_device_id device = program->devices[device_i];
606
607 /* find the device in the supplied devices-to-build-for list */
608 int found = 0;
609 for (i = 0; i < num_devices; ++i)
610 if (device_list[i] == device) found = 1;
611 if (!found) continue;
612
613 if (requires_cr_sqrt_div
614 && !(device->single_fp_config & CL_FP_CORRECTLY_ROUNDED_DIVIDE_SQRT))
615 {
616 APPEND_TO_MAIN_BUILD_LOG (REQUIRES_CR_SQRT_DIV_ERR);
617 POCL_GOTO_ERROR_ON (1, build_error_code,
618 REQUIRES_CR_SQRT_DIV_ERR " %s\n",
619 device->short_name);
620 }
621 actually_built++;
622
623 /* clCreateProgramWithBuiltinKernels */
624 /* No build step supported at the moment for built-in kernels. */
625 if (program->builtin_kernel_names)
626 continue;
627
628 /* clCreateProgramWithSource */
629 else if (program->source)
630 {
631#ifdef OCS_AVAILABLE
632 if (device->compiler_available == CL_TRUE)
633 {
634 POCL_MSG_PRINT_INFO ("building from sources for device %d\n",
635 device_i);
636 error = pocl_llvm_build_program (
637 program, device_i, program->compiler_options,
638 program_bc_path, num_input_headers, input_headers,
639 header_include_names, (create_library ? 0 : link_program));
640 POCL_GOTO_ERROR_ON ((error != 0), build_error_code,
641 "pocl_llvm_build_program() failed\n");
642 }
643 else
644#endif
645 {
646 APPEND_TO_MAIN_BUILD_LOG (
647 "Cannot build a program from sources with pocl "
648 "that does not have online compiler support\n");
649 POCL_GOTO_ERROR_ON (1, CL_COMPILER_NOT_AVAILABLE, "%s",
650 program->main_build_log);
651 }
652 }
653 /* clCreateProgramWithBinaries */
654 else if (program->binaries[device_i]
655 && (program->pocl_binaries[device_i] == NULL))
656 {
657#ifdef OCS_AVAILABLE
658 /* bitcode is now either plain LLVM IR or SPIR IR */
659 int spir_binary = bitcode_is_spir ((char*)program->binaries[device_i],
660 program->binary_sizes[device_i]);
661 if (spir_binary)
662 POCL_MSG_PRINT_LLVM ("LLVM-SPIR binary detected\n");
663 else
664 POCL_MSG_PRINT_LLVM ("building from a BC binary for device %d\n",
665 device_i);
666
667 if (spir_binary)
668 {
669#ifdef ENABLE_SPIR
670 if (!strstr (device->extensions, "cl_khr_spir"))
671 {
672 APPEND_TO_MAIN_BUILD_LOG (REQUIRES_SPIR_SUPPORT);
673 POCL_GOTO_ERROR_ON (1, build_error_code,
674 REQUIRES_SPIR_SUPPORT " %s\n",
675 device->short_name);
676 }
677 if (!spir_build)
678 POCL_MSG_WARN (
679 "SPIR binary provided, but no spir in build options\n");
680 /* SPIR binaries need to be explicitly linked to the kernel
681 * library. for non-SPIR binaries this happens as part of build
682 * process when program.bc is generated. */
683 error
684 = pocl_llvm_link_program (program, device_i, program_bc_path,
685 0, NULL, NULL, NULL, 0, 1);
686
687 POCL_GOTO_ERROR_ON (error, CL_LINK_PROGRAM_FAILURE,
688 "Failed to link SPIR program.bc\n");
689#else
690 APPEND_TO_MAIN_BUILD_LOG (REQUIRES_SPIR_SUPPORT);
691 POCL_GOTO_ERROR_ON (1, build_error_code,
692 REQUIRES_SPIR_SUPPORT " %s\n",
693 device->short_name);
694#endif
695 }
696
697#else
698 APPEND_TO_MAIN_BUILD_LOG (
699 "Cannot build program from LLVM IR binaries with "
700 "pocl that does not have online compiler support\n");
701 POCL_GOTO_ERROR_ON (1, CL_COMPILER_NOT_AVAILABLE, "%s",
702 program->main_build_log);
703#endif
704 }
705 else if (program->pocl_binaries[device_i])
706 {
707 POCL_MSG_PRINT_INFO("having a poclbinary for device %d\n", device_i);
708#ifdef OCS_AVAILABLE
709 if (program->binaries[device_i] == NULL)
710 {
711 POCL_MSG_WARN (
712 "pocl-binary for this device doesn't contain "
713 "program.bc - you won't be able to rebuild/link it\n");
714 /* do not try to read program.bc or LLVM IRs
715 * TODO maybe read LLVM IRs ?*/
716 continue;
717 }
718#else
719 continue;
720#endif
721 }
722 else if (link_program && (num_input_programs > 0))
723 {
724#ifdef OCS_AVAILABLE
725 /* just link binaries. */
726 unsigned char *cur_device_binaries[num_input_programs];
727 size_t cur_device_binary_sizes[num_input_programs];
728 void *cur_llvm_irs[num_input_programs];
729 for (j = 0; j < num_input_programs; j++)
730 {
731 assert (device == input_programs[j]->devices[device_i]);
732 cur_device_binaries[j] = input_programs[j]->binaries[device_i];
733
734 assert (cur_device_binaries[j]);
735 cur_device_binary_sizes[j]
736 = input_programs[j]->binary_sizes[device_i];
737
738 if (input_programs[j]->llvm_irs[device_i] == NULL)
739 pocl_update_program_llvm_irs (input_programs[j], device_i);
740
741 cur_llvm_irs[j] = input_programs[j]->llvm_irs[device_i];
742 assert (cur_llvm_irs[j]);
743 }
744 error = pocl_llvm_link_program (
745 program, device_i, program_bc_path, num_input_programs,
746 cur_device_binaries, cur_device_binary_sizes, cur_llvm_irs,
747 create_library, 0);
748 POCL_GOTO_ERROR_ON ((error != CL_SUCCESS), CL_LINK_PROGRAM_FAILURE,
749 "pocl_llvm_link_program() failed\n");
750#else
751 POCL_GOTO_ERROR_ON ((1), CL_LINK_PROGRAM_FAILURE,
752 "clCompileProgram/clLinkProgram/clBuildProgram"
753 " require a pocl built with LLVM\n");
754
755#endif
756 }
757 else
758 {
759 POCL_GOTO_ERROR_ON (1, CL_INVALID_BINARY,
760 "No sources nor binaries for device %s - can't "
761 "build the program\n", device->short_name);
762 }
763
764#ifdef OCS_AVAILABLE
765 /* Read binaries from program.bc to memory */
766 if (program->binaries[device_i] == NULL)
767 {
768 errcode = pocl_read_file (program_bc_path, &binary, &fsize);
769 POCL_GOTO_ERROR_ON (errcode, CL_BUILD_ERROR,
770 "Failed to read binaries from program.bc to "
771 "memory: %s\n",
772 program_bc_path);
773
774 program->binary_sizes[device_i] = (size_t)fsize;
775 program->binaries[device_i] = (unsigned char *)binary;
776 }
777
778 if (program->llvm_irs[device_i] == NULL)
779 {
780 pocl_update_program_llvm_irs (program, device_i);
781 }
782#endif
783
784 /* Maintain a 'last_accessed' file in every program's
785 * cache directory. Will be useful for cache pruning script
786 * that flushes old directories based on LRU */
787 pocl_cache_update_program_last_access (program, device_i);
788 }
789
790 POCL_GOTO_ERROR_ON ((actually_built < num_devices), build_error_code,
791 "Some of the devices on the argument-supplied list are"
792 "not available for the program, or do not exist\n");
793
794 program->build_status = CL_BUILD_SUCCESS;
795 program->binary_type = CL_PROGRAM_BINARY_TYPE_EXECUTABLE;
796 /* if program will be compiled using clCompileProgram its binary_type
797 * will be set to CL_PROGRAM_BINARY_TYPE_COMPILED_OBJECT.
798 *
799 * if program was created by clLinkProgram which is called
800 * with the –createlibrary link option its binary_type will be set to
801 * CL_PROGRAM_BINARY_TYPE_LIBRARY.
802 */
803 if (create_library)
804 program->binary_type = CL_PROGRAM_BINARY_TYPE_LIBRARY;
805 if (compile_program && !link_program)
806 program->binary_type = CL_PROGRAM_BINARY_TYPE_COMPILED_OBJECT;
807
808 assert(program->num_kernels == 0);
809
810 if (program->binary_type != CL_PROGRAM_BINARY_TYPE_EXECUTABLE)
811 {
812 errcode = CL_SUCCESS;
813 goto FINISH;
814 }
815
816 /* get non-device-specific kernel metadata. We can stop after finding
817 * the first method that works.*/
818 for (device_i = 0; device_i < program->num_devices; device_i++)
819 {
820 if (program->builtin_kernel_names)
821 {
822 program->num_kernels = program->num_builtin_kernels;
823 if (program->num_kernels)
824 {
825 cl_device_id device = program->devices[device_i];
826 program->kernel_meta = calloc (program->num_kernels,
827 sizeof (pocl_kernel_metadata_t));
828
829 size_t i;
830 for (i = 0; i < program->num_kernels; ++i)
831 {
832 device->ops->get_builtin_kernel_metadata (
833 device->data, program->builtin_kernel_names[i],
834 &program->kernel_meta[i]);
835 }
836 }
837 break;
838 }
839#ifdef OCS_AVAILABLE
840 if (program->binaries[device_i])
841 {
842 program->num_kernels
843 = pocl_llvm_get_kernel_count (program, device_i);
844 if (program->num_kernels)
845 {
846 program->kernel_meta = calloc (program->num_kernels,
847 sizeof (pocl_kernel_metadata_t));
848 pocl_llvm_get_kernels_metadata (program, device_i);
849 }
850 break;
851 }
852#endif
853 if (program->pocl_binaries[device_i])
854 {
855 program->num_kernels
856 = pocl_binary_get_kernel_count (program, device_i);
857 if (program->num_kernels)
858 {
859 program->kernel_meta = calloc (program->num_kernels,
860 sizeof (pocl_kernel_metadata_t));
861 pocl_binary_get_kernels_metadata (program, device_i);
862 }
863 break;
864 }
865 }
866
867 POCL_GOTO_ERROR_ON ((device_i >= program->num_devices), CL_INVALID_BINARY,
868 "Couldn't find kernel metadata in the built program\n");
869
870 /* calculate device-specific kernel hashes. */
871 for (j = 0; j < program->num_kernels; ++j)
872 {
873 program->kernel_meta[j].build_hash
874 = calloc (program->num_devices, sizeof (pocl_kernel_hash_t));
875
876 if (program->builtin_kernel_names == NULL)
877 for (device_i = 0; device_i < program->num_devices; device_i++)
878 {
879 pocl_calculate_kernel_hash (program, j, device_i);
880 }
881 }
882
883 errcode = CL_SUCCESS;
884 goto FINISH;
885
886ERROR:
887 free_meta (program);
888
889 program->kernels = NULL;
890
891 for (device_i = 0; device_i < program->num_devices; device_i++)
892 {
893 if (program->source)
894 {
895 POCL_MEM_FREE (program->binaries[device_i]);
896 program->binary_sizes[device_i] = 0;
897 }
898 }
899
900ERROR_CLEAN_OPTIONS:
901 if (temp_options != options)
902 free (temp_options);
903
904 program->build_status = CL_BUILD_ERROR;
905
906FINISH:
907 POCL_UNLOCK_OBJ (program);
908 POCL_MEM_FREE (unique_devlist);
909
910PFN_NOTIFY:
911 if (pfn_notify)
912 pfn_notify (program, user_data);
913
914 return errcode;
915}
