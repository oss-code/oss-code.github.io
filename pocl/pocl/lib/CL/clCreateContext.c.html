32int context_set_properties(cl_context context,
33 const cl_context_properties * properties,
34 cl_int * errcode)
35{
36 unsigned i;
37 int num_properties = 0;
38
39 context->properties = NULL;
40
41 /* verify if data in properties is valid
42 * and set them */
43 if (properties)
44 {
45 const cl_context_properties *p = properties;
46 const cl_context_properties *q;
47
48 cl_platform_id platforms[1];
49 cl_uint num_platforms;
50 cl_bool platform_found;
51
52 POname(clGetPlatformIDs)(1, platforms, &num_platforms);
53
54 num_properties = 0;
55 while (p[0] != 0)
56 {
57 /* redefinition of the same property */
58 for(q=properties; q<p; q+=2)
59 if (q[0] == p[0])
60 {
61 POCL_MSG_ERR("Duplicate properties: %lu\n", (unsigned long)q[0]);
62 *errcode = CL_INVALID_PROPERTY;
63 return 0;
64 }
65
66 switch (p[0])
67 {
68 case CL_CONTEXT_PLATFORM:
69
70 /* pocl just have one platform */
71 platform_found = CL_FALSE;
72 for (i=0; i<num_platforms; i++)
73 if ((cl_platform_id)p[1] == platforms[i])
74 platform_found = CL_TRUE;
75
76 if (platform_found == CL_FALSE)
77 {
78 POCL_MSG_ERR("Could not find platform %p\n", (void*)p[1]);
79 *errcode = CL_INVALID_PLATFORM;
80 return 0;
81 }
82
83 p += 2;
84 break;
85
86 default:
87 POCL_MSG_ERR("Unknown context property: %lu\n", (unsigned long)p[0]);
88 *errcode = CL_INVALID_PROPERTY;
89 return 0;
90 }
91 num_properties++;
92 }
93
94 context->properties = (cl_context_properties *) malloc
95 ((num_properties * 2 + 1) * sizeof(cl_context_properties));
96 if (context->properties == NULL)
97 {
98 *errcode = CL_OUT_OF_HOST_MEMORY;
99 return 0;
100 }
101
102 memcpy(context->properties, properties,
103 (num_properties * 2 + 1) * sizeof(cl_context_properties));
104 context->num_properties = num_properties;
105
106 *errcode = 0;
107 return num_properties;
108 }
109 else
110 {
111 context->properties = NULL;
112 context->num_properties = 0;
113
114 *errcode = 0;
115 return 0;
116 }
117}

119unsigned cl_context_count = 0;
120pocl_lock_t pocl_context_handling_lock = POCL_LOCK_INITIALIZER;
121
122CL_API_ENTRY cl_context CL_API_CALL
123POname(clCreateContext)(const cl_context_properties * properties,
124 cl_uint num_devices,
125 const cl_device_id * devices,
126 void (CL_CALLBACK * pfn_notify)(const char *, const void *, size_t, void *),
127 void * user_data,
128 cl_int * errcode_ret) CL_API_SUFFIX__VERSION_1_0
129{
130 unsigned i;
131 cl_device_id device_ptr;
132 cl_int errcode = 0;
133 cl_context context = NULL;
134
135 POCL_LOCK (pocl_context_handling_lock);
136
137 POCL_GOTO_ERROR_COND((devices == NULL || num_devices == 0), CL_INVALID_VALUE);
138
139 POCL_GOTO_ERROR_COND((pfn_notify == NULL && user_data != NULL), CL_INVALID_VALUE);
140
141 int offline_compile = pocl_get_bool_option("POCL_OFFLINE_COMPILE", 0);
142
143 errcode = pocl_init_devices();
144 /* clCreateContext cannot return CL_DEVICE_NOT_FOUND, which is what
145 * pocl_init_devices() returns if no devices could be probed. Hence,
146 * remap this error to CL_INVALID_DEVICE. Note that this particular
147 * situation should never arise, since an application should issue
148 * clGetDeviceIDs before clCreateContext, and we would have returned
149 * CL_DEVICE_NOT_FOUND already from clGetDeviceIDs. Still, no reason
150 * not to handle it.
151 */
152 if (errcode == CL_DEVICE_NOT_FOUND)
153 errcode = CL_INVALID_DEVICE;
154 POCL_GOTO_ERROR_ON ((errcode != CL_SUCCESS), errcode,
155 "Could not initialize devices\n");
156
157 context = (cl_context)calloc (1, sizeof (struct _cl_context));
158 if (context == NULL)
159 {
160 errcode = CL_OUT_OF_HOST_MEMORY;
161 goto ERROR;
162 }
163
164 POCL_INIT_OBJECT(context);
165
166 context_set_properties(context, properties, &errcode);
167 if (errcode)
168 {
169 goto ERROR;
170 }
171
172 context->devices = pocl_unique_device_list(devices, num_devices,
173 &context->num_devices);
174 if (context->devices == NULL)
175 {
176 errcode = CL_OUT_OF_HOST_MEMORY;
177 goto ERROR;
178 }
179
180 i = 0;
181 while (i < context->num_devices)
182 {
183 device_ptr = context->devices[i++];
184 if (device_ptr == NULL)
185 {
186 POCL_MSG_ERR("one of the devices in device list is NULL\n");
187 errcode = CL_INVALID_DEVICE;
188 goto ERROR_CLEAN_CONTEXT_AND_DEVICES;
189 }
190
191 if (!offline_compile && (device_ptr->available != CL_TRUE))
192 {
193 POCL_MSG_WARN("Offline compilation disabled - dropping unavailable device: %s\n", device_ptr->long_name);
194 context->devices[i] = context->devices[--context->num_devices];
195 }
196 else if((device_ptr->compiler_available != CL_TRUE) && (device_ptr->available != CL_TRUE))
197 {
198 POCL_MSG_WARN("Device unavailable and device compiler unavailable - dropping device: %s\n", device_ptr->long_name);
199 context->devices[i] = context->devices[--context->num_devices];
200 }
201 else
202 POname(clRetainDevice)(device_ptr);
203 }
204
205 if (context->num_devices == 0)
206 {
207 POCL_MSG_ERR ("Zero devices after dropping the unavailable ones\n");
208 errcode = CL_INVALID_DEVICE;
209 goto ERROR_CLEAN_CONTEXT_AND_DEVICES;
210 }
211
212 pocl_setup_context(context);
213
214 pocl_init_mem_manager ();
215
216 if (errcode_ret)
217 *errcode_ret = CL_SUCCESS;
218 context->valid = 1;
219
220 cl_context_count += 1;
221 POCL_UNLOCK (pocl_context_handling_lock);
222
223 return context;
224
225ERROR_CLEAN_CONTEXT_AND_DEVICES:
226 POCL_MEM_FREE(context->devices);
227 /*ERROR_CLEAN_CONTEXT_AND_PROPERTIES:*/
228 POCL_MEM_FREE(context->properties);
229ERROR:
230 POCL_MEM_FREE(context);
231 if(errcode_ret != NULL)
232 {
233 *errcode_ret = errcode;
234 }
235
236 POCL_UNLOCK (pocl_context_handling_lock);
237 return NULL;
238}
