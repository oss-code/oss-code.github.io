75 do { \

149 #define POCL_MSG_PRINT_F(FILTER, TYPE, ERRCODE, ...) \
150 do { \
151 if (pocl_debug_messages_filter & POCL_DEBUG_FLAG_ ## FILTER) { \
152 pocl_debug_output_lock (); \
153 POCL_DEBUG_HEADER(FILTER, POCL_FILTER_TYPE_ ## TYPE) \
154 if (stderr_is_a_tty) \
155 fprintf (stderr, "%s", POCL_COLOR_BOLDRED \
156 ERRCODE " " POCL_COLOR_RESET); \
157 else \
158 fprintf (stderr, "%s", ERRCODE " "); \
159 fprintf (stderr, __VA_ARGS__); \
160 pocl_debug_output_unlock (); \
161 } \
162 } while (0)
163
164 #define POCL_MSG_PRINT2(FILTER, func, line, ...) \
165 do { \
166 if (pocl_debug_messages_filter & POCL_DEBUG_FLAG_ ## FILTER) { \
167 pocl_debug_output_lock (); \
168 pocl_debug_print_header (func, line, \
169 #FILTER, POCL_FILTER_TYPE_INFO); \
170 fprintf (stderr, __VA_ARGS__); \
171 pocl_debug_output_unlock (); \
172 } \
173 } while (0)
