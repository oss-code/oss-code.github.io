38/* pocl binary identifier */
39#define POCLCC_STRING_ID "poclbin"
40#define POCLCC_STRING_ID_LENGTH 8
41/* changes for version 2: added program.bc right after header */
42/* changes for version 3: added flush_denorms flag into header */
43/* changes for version 4: kernel library is now linked into
44 program.bc, so older binaries may fail
45 to run with "undefined symbol" errors. */
46/* changes for version 5: added program binary_type into header */
47/* changes for version 6: added reqd_wg_size informations into
48 pocl_binary_kernel structure */
49/* changes for version 7: removed dyn_arguments from storage, instead added
50 arg_info[i]->type_size; removed is_local and is_set
51 from storage, no need to store these (is_set makes
52 no sense in binary, and whether argument is local
53 is already in cl_kernel_arg_address_qualifier);
54 add has_arg_metadata & kernel attributes */
55

59/* pocl binary structures */
60
61/* Note that structs are not 1:1 to what's serialized on-disk. In particular
62 * 1) for integer values, endianness is forced to LITTLE_ENDIAN
63 * 2) pointers in general are not written at all, rather reconstructed from data
64 * 3) char* strings are written as: | uint32_t strlen | strlen bytes of content |
65 * 4) files are written as two strings: | uint32_t | relative filename | uint32_t | content |
66 */
67
68typedef struct pocl_binary_kernel_s
69{
70 /* the first 3 fields are sizes in bytes of the data pieces that follow
71 * (to allow quickly jumping between records in the serialized binary;
72 * this is required to e.g. extract kernel metadata without having to
73 * completely deserialize everything in the binary) */
74
75 /* size of this entire struct serialized, including all data (binaries & arginfo sizes)
76 * current offset in binary + struct_size = offset of the next pocl_binary_kernel */
77 uint64_t struct_size;
78 /* size of kernel cachedir content serialized */
79 uint64_t binaries_size;
80 /* size of "arginfo" array of structs serialized */
81 uint32_t arginfo_size;
82
83 /* size of the kernel_name string */
84 uint32_t sizeof_kernel_name;
85 /* kernel_name string */
86 char *kernel_name;
87
88 /* number of kernel arguments */
89 uint32_t num_args;
90 /* number of kernel local variables */
91 uint32_t num_locals;
92
93 /* required work-group size */
94 uint64_t reqd_wg_size[OPENCL_MAX_DIMENSION];
95
96 uint64_t has_arg_metadata;
97
98 uint32_t sizeof_attributes;
99 char* attributes;
100
101 /* arguments and argument metadata. Note that not everything is stored
102 * in the serialized binary */
103 size_t *local_sizes;
104} pocl_binary_kernel;
105
106typedef struct pocl_binary_s
107{
108 /* file format "magic" marker */
109 char pocl_id[POCLCC_STRING_ID_LENGTH];
110 /* llvm triple + target hash */
111 uint64_t device_id;
112 /* binary format version */
113 uint32_t version;
114 /* number of kernels in the serialized pocl binary */
115 uint32_t num_kernels;
116 /* various flags */
117 uint64_t flags;
118 /* program->build_hash[device_i], required to restore files into pocl cache */
119 SHA1_digest_t program_build_hash;
120} pocl_binary;
121
122#define POCL_BINARY_FLAG_FLUSH_DENORMS (1 << 0)
123
124#define TO_LE(x) \
125 ((sizeof(x) == 8) ? htole64((uint64_t)x) : \
126 ((sizeof(x) == 4) ? htole32((uint32_t)x) : \
127 ((sizeof(x) == 2) ? htole16((uint16_t)x) : \
128 ((sizeof(x) == 1) ? (uint8_t)(x) : 0 ))))
129
130#define FROM_LE(x) \
131 ((sizeof(x) == 8) ? le64toh((uint64_t)x) : \
132 ((sizeof(x) == 4) ? le32toh((uint32_t)x) : \
133 ((sizeof(x) == 2) ? le16toh((uint16_t)x) : \
134 ((sizeof(x) == 1) ? (uint8_t)(x) : 0 ))))
135
136/***********************************************************/
137
138#define BUFFER_STORE(elem, type) \
139 do \
140 { \
141 type b_s_tmp = (type) TO_LE ( (type) elem); \
142 memcpy (buffer, &b_s_tmp, sizeof (type)); \
143 buffer += sizeof (type); \
144 } \
145 while(0)
146
147#define BUFFER_READ(elem, type) \
148 memcpy (&elem, buffer, sizeof (type)); \
149 elem = (type) FROM_LE ( (type) elem); \
150 buffer += sizeof (type)
151
152#define BUFFER_STORE_STR2(elem, len) \
153 do { \
154 BUFFER_STORE(len, uint32_t); \
155 if (len) \
156 { \
157 memcpy(buffer, elem, len); \
158 buffer += len; \
159 } \
160 } while (0)
161
162#define BUFFER_READ_STR2(elem, len) \
163 do \
164 { \
165 BUFFER_READ(len, uint32_t); \
166 if (len) \
167 { \
168 elem = malloc (len + 1); \
169 memcpy (elem, buffer, len); \
170 elem[len] = 0; \
171 buffer += len; \
172 } \
173 } while (0)
174
175#define BUFFER_STORE_STR(elem) \
176 do { uint32_t len = strlen(elem); \
177 BUFFER_STORE_STR2(elem, len); } while (0)
178
179#define BUFFER_READ_STR(elem) \
180 do { uint32_t len = 0; \
181 BUFFER_READ_STR2(elem, len); } while (0)
182
183#define ADD_STRLEN(else_b) \
184 if (serialized) \
185 { \
186 unsigned char* t = buffer+res; \
187 res += *(uint32_t*)t; \
188 res += sizeof(uint32_t); \
189 } \
190 else \
191 { \
192 res += sizeof(uint32_t); \
193 res += else_b; \
194 }
195

198static unsigned char*
199read_header(pocl_binary *b, const unsigned char *buffer)
200{
201 memset(b, 0, sizeof(pocl_binary));
202 memcpy(b->pocl_id, buffer, POCLCC_STRING_ID_LENGTH);
203 buffer += POCLCC_STRING_ID_LENGTH;
204 BUFFER_READ(b->device_id, uint64_t);
205 BUFFER_READ(b->version, uint32_t);
206 BUFFER_READ(b->num_kernels, uint32_t);
207 BUFFER_READ (b->flags, uint64_t);
208 memcpy(b->program_build_hash, buffer, sizeof(SHA1_digest_t));
209 buffer += sizeof(SHA1_digest_t);
210 return (unsigned char*)buffer;
211}

213#define FNV_OFFSET UINT64_C(0xcbf29ce484222325)
214#define FNV_PRIME UINT64_C(0x100000001b3)
215static uint64_t
216pocl_binary_get_device_id(cl_device_id device)
217{
218 /* FNV-1A with whatever device returns
219 * as its build hash string */
220 uint64_t result = FNV_OFFSET;
221 char *dev_hash = device->ops->build_hash(device);
222
223 int i, length = strlen(dev_hash);
224 for (i=0; i<length; i++)
225 {
226 result *= FNV_PRIME;
227 result ^= dev_hash[i];
228 }
229 free(dev_hash);
230
231 return result;
232}

234static unsigned char*
235check_binary(cl_device_id device, const unsigned char *binary)
236{
237 pocl_binary b;
238 unsigned char *p = read_header(&b, binary);
239 if (strncmp (b.pocl_id, POCLCC_STRING_ID, POCLCC_STRING_ID_LENGTH))
240 {
241 POCL_MSG_WARN ("File is not a pocl binary\n");
242 return NULL;
243 }
244 if (b.version < FIRST_SUPPORTED_POCLCC_VERSION)
245 {
246 POCL_MSG_WARN ("PoclBinary version %i is not supported by "
247 "this pocl (the minimal is: %i)\n",
248 b.version, FIRST_SUPPORTED_POCLCC_VERSION);
249 return NULL;
250 }
251 if (pocl_binary_get_device_id(device) != b.device_id)
252 {
253 POCL_MSG_WARN ("PoclBinary device id mismatch\n");
254 return NULL;
255 }
256 return p;
257}
258
259int
260pocl_binary_check_binary(cl_device_id device, const unsigned char *binary)
261{
262 return (check_binary(device, binary) != NULL);
263}

288/* serializes a single file. */
289static unsigned char*
290serialize_file(char* path, size_t basedir_offset, unsigned char* buffer)
291{
292 char* content;
293 uint64_t fsize;
294 char* p = path + basedir_offset;
295 BUFFER_STORE_STR(p);
296 pocl_read_file(path, &content, &fsize);
297 BUFFER_STORE_STR2(content, fsize);
298 free(content);
299 return buffer;
300}
301
302/* recursively serializes files/directories by calling
303 * either itself (on directory), or serialize_file (on files) */
304static unsigned char*
305recursively_serialize_path (char* path,
306 size_t basedir_offset,
307 unsigned char* buffer)
308{
309 struct stat st;
310 stat (path, &st);
311
312 if (S_ISREG (st.st_mode))
313 buffer = serialize_file (path, basedir_offset, buffer);
314
315 if (S_ISDIR (st.st_mode))
316 {
317 DIR *d;
318 struct dirent *entry;
319 char subpath[POCL_FILENAME_LENGTH];
320
321 strncpy (subpath, path, POCL_FILENAME_LENGTH-1);
322 char* p = subpath + strlen(subpath);
323 *p++ = '/';
324 d = opendir (path);
325 while ((entry = readdir (d)))
326 {
327 if (strcmp (entry->d_name, ".") == 0) continue;
328 if (strcmp (entry->d_name, "..") == 0) continue;
329 strcpy (p, entry->d_name);
330 buffer =
331 recursively_serialize_path (subpath, basedir_offset, buffer);
332 }
333 closedir (d);
334 }
335
336 return buffer;
337}

339/* serializes an entire pocl kernel cachedir. */
340static unsigned char*
341serialize_kernel_cachedir (cl_program program,
342 const char* kernel_name,
343 unsigned device_i,
344 unsigned char* buffer)
345{
346 char path[POCL_FILENAME_LENGTH];
347 char basedir[POCL_FILENAME_LENGTH];
348
349 pocl_cache_program_path (basedir, program, device_i);
350 size_t basedir_len = strlen (basedir);
351
352 pocl_cache_kernel_cachedir (path, program, device_i, kernel_name);
353 POCL_MSG_PRINT_INFO ("Kernel %s: recur serializing cachedir %s\n",
354 kernel_name, path);
355 buffer = recursively_serialize_path (path, basedir_len, buffer);
356
357 return buffer;
358}

360/* serializes a single kernel */
361static unsigned char*
362pocl_binary_serialize_kernel_to_buffer(cl_program program,
363 pocl_kernel_metadata_t *meta,
364 unsigned device_i,
365 unsigned char *buf)
366{
367 unsigned char *buffer = buf;
368 unsigned i;
369
370 BUFFER_STORE(0, uint64_t); // struct_size
371 BUFFER_STORE(0, uint64_t); // binaries_size
372 BUFFER_STORE(0, uint32_t); // arginfo size
373 uint32_t namelen = strlen (meta->name);
374 BUFFER_STORE_STR2 (meta->name, namelen);
375 BUFFER_STORE (meta->num_args, uint32_t);
376 BUFFER_STORE (meta->num_locals, uint32_t);
377
378 for (i = 0; i < OPENCL_MAX_DIMENSION; i++)
379 {
380 BUFFER_STORE (meta->reqd_wg_size[i], uint64_t);
381 }
382
383 for (i = 0; i < meta->num_locals; i++)
384 {
385 uint64_t temp = meta->local_sizes[i];
386 BUFFER_STORE (temp, uint64_t);
387 }
388
389 uint32_t attrlen = meta->attributes ? strlen (meta->attributes) : 0;
390 BUFFER_STORE_STR2(meta->attributes, attrlen);
391 BUFFER_STORE(meta->has_arg_metadata, uint64_t);
392
393 /***********************************************************************/
394 unsigned char *start = buffer;
395 for (i = 0; i < meta->num_args; i++)
396 {
397 pocl_argument_info *ai = &meta->arg_info[i];
398 BUFFER_STORE(ai->access_qualifier, cl_kernel_arg_access_qualifier);
399 BUFFER_STORE(ai->address_qualifier, cl_kernel_arg_address_qualifier);
400 BUFFER_STORE(ai->type_qualifier, cl_kernel_arg_type_qualifier);
401 BUFFER_STORE(ai->type, uint32_t);
402 BUFFER_STORE (ai->type_size, uint32_t);
403 BUFFER_STORE_STR(ai->name);
404 BUFFER_STORE_STR(ai->type_name);
405 }
406 /***********************************************************************/
407
408 uint32_t arginfo_size = buffer - start;
409
410 unsigned char *end
411 = serialize_kernel_cachedir (program, meta->name, device_i, buffer);
412 uint64_t binaries_size = end - buffer;
413
414 /* write struct size properly */
415 buffer = buf;
416 uint64_t struct_size = end - buf;
417 BUFFER_STORE(struct_size, uint64_t);
418 BUFFER_STORE(binaries_size, uint64_t);
419 BUFFER_STORE(arginfo_size, uint32_t);
420
421 return end;
422}

424/**
425 * Deserializes a single file from the binary to disk.
426 *
427 * Returns the number of bytes read.
428 */
429static size_t
430deserialize_file (unsigned char* buffer,
431 char* basedir,
432 size_t offset)
433{
434 unsigned char* orig_buffer = buffer;
435 size_t len;
436
437 char* relpath = NULL;
438 BUFFER_READ_STR2 (relpath, len);
439 assert (len > 0);
440
441 char* content = NULL;
442 BUFFER_READ_STR2 (content, len);
443 assert (len > 0);
444
445 char *p = basedir + offset;
446 strcpy (p, relpath);
447 free (relpath);
448
449 char *fullpath = basedir;
450 if (pocl_exists (fullpath))
451 goto RET;
452
453 char* dir = strdup (basedir);
454 char* dirpath = dirname (dir);
455 if (!pocl_exists (dirpath))
456 pocl_mkdir_p (dirpath);
457 free (dir);
458
459 pocl_write_file (fullpath, content, len, 0, 0);
460
461RET:
462 free (content);
463 return (buffer - orig_buffer);
464}

466/* Deserializes all files of a single pocl kernel cachedir. */
467static unsigned char*
468deserialize_kernel_cachedir (char* basedir, unsigned char* buffer, size_t bytes)
469{
470 size_t done = 0;
471 size_t offset = strlen (basedir);
472
473 while (done < bytes)
474 {
475 done += deserialize_file (buffer + done, basedir, offset);
476 }
477 assert(done == bytes);
478 return (buffer + done);
479}

483/* Deserializes a single kernel.
484
485 This has two modes of operation:
486
487 1) if kernel_index is >= 0 it only fills in pocl_binary_kernel
488 with metadata (doesn't unpack files) and only if the name matches - used by
489 pocl_binary_get_kernel_metadata()
490
491 2) if name_len and name_match are NULL, unpacks kernel cachedir on disk, but
492 does not set up kernel metadata of pocl_binary_kernel argument - used by
493 pocl_binary_deserialize()
494 */
495
496static int
497pocl_binary_deserialize_kernel_from_buffer (pocl_binary *b,
498 unsigned char **buf,
499 pocl_binary_kernel *kernel,
500 pocl_kernel_metadata_t *meta,
501 char *basedir)
502{
503 unsigned i;
504 unsigned char *buffer = *buf;
505 uint64_t *dynarg_sizes;
506
507 memset(kernel, 0, sizeof(pocl_binary_kernel));
508 BUFFER_READ(kernel->struct_size, uint64_t);
509 BUFFER_READ(kernel->binaries_size, uint64_t);
510 BUFFER_READ(kernel->arginfo_size, uint32_t);
511 BUFFER_READ_STR2(kernel->kernel_name, kernel->sizeof_kernel_name);
512 BUFFER_READ(kernel->num_args, uint32_t);
513 BUFFER_READ(kernel->num_locals, uint32_t);
514
515 dynarg_sizes = alloca (sizeof(uint64_t) * kernel->num_args);
516
517 for (i = 0; i < OPENCL_MAX_DIMENSION; i++)
518 {
519 BUFFER_READ(kernel->reqd_wg_size[i], uint64_t);
520 }
521
522 if (meta)
523 {
524 if (b->version < 7)
525 {
526 for (i = 0; i < kernel->num_args; i++)
527 {
528 BUFFER_READ (dynarg_sizes[i], uint64_t);
529 }
530 }
531
532 kernel->local_sizes = calloc (kernel->num_locals, sizeof (size_t));
533 for (i = 0; i < kernel->num_locals; i++)
534 {
535 uint64_t temp;
536 BUFFER_READ (temp, uint64_t);
537 kernel->local_sizes[i] = temp;
538 }
539
540 if (b->version >= 7)
541 {
542 BUFFER_READ_STR2(kernel->attributes, kernel->sizeof_attributes);
543 BUFFER_READ(kernel->has_arg_metadata, uint64_t);
544 }
545 else
546 {
547 kernel->attributes = NULL;
548 kernel->has_arg_metadata = (-1);
549 }
550
551 meta->arg_info = calloc (kernel->num_args, sizeof (struct pocl_argument_info));
552 POCL_RETURN_ERROR_COND ((!meta->arg_info), CL_OUT_OF_HOST_MEMORY);
553
554 for (i = 0; i < kernel->num_args; i++)
555 {
556 pocl_argument_info *ai = &meta->arg_info[i];
557 BUFFER_READ (ai->access_qualifier, cl_kernel_arg_access_qualifier);
558 BUFFER_READ (ai->address_qualifier, cl_kernel_arg_address_qualifier);
559 BUFFER_READ (ai->type_qualifier, cl_kernel_arg_type_qualifier);
560 if (b->version < 7)
561 {
562 char t1, t2;
563 BUFFER_READ (t1, char);
564 BUFFER_READ (t2, char);
565 }
566
567 BUFFER_READ (ai->type, uint32_t);
568 if (b->version >= 7)
569 {
570 BUFFER_READ (ai->type_size, uint32_t);
571 }
572 else
573 {
574 ai->type_size = dynarg_sizes[i];
575 }
576 BUFFER_READ_STR (ai->name);
577 BUFFER_READ_STR (ai->type_name);
578 }
579
580 }
581 else
582 {
583 /* skip the arg_info and all kernel metadata */
584 buffer = *buf + (kernel->struct_size - kernel->binaries_size);
585 deserialize_kernel_cachedir (basedir, buffer, kernel->binaries_size);
586 POCL_MEM_FREE (kernel->kernel_name);
587 }
588
589 /* always skip to the next kernel */
590 *buf = *buf + kernel->struct_size;
591 return CL_SUCCESS;
592
593}

597cl_int
598pocl_binary_serialize(cl_program program, unsigned device_i, size_t *size)
599{
600 unsigned char *buffer = program->pocl_binaries[device_i];
601 size_t sizeof_buffer = program->pocl_binary_sizes[device_i];
602 unsigned char *end_of_buffer = buffer + sizeof_buffer;
603 unsigned char *start = buffer;
604
605 unsigned num_kernels = program->num_kernels;
606
607 memcpy(buffer, POCLCC_STRING_ID, POCLCC_STRING_ID_LENGTH);
608 buffer += POCLCC_STRING_ID_LENGTH;
609 BUFFER_STORE(pocl_binary_get_device_id(program->devices[device_i]), uint64_t);
610 BUFFER_STORE(POCLCC_VERSION, uint32_t);
611 BUFFER_STORE(num_kernels, uint32_t);
612 uint64_t flags = 0;
613 if (program->flush_denorms)
614 flags |= POCL_BINARY_FLAG_FLUSH_DENORMS;
615 flags |= (program->binary_type << 1);
616 BUFFER_STORE (flags, uint64_t);
617 memcpy(buffer, program->build_hash[device_i], sizeof(SHA1_digest_t));
618 buffer += sizeof(SHA1_digest_t);
619
620 assert(buffer < end_of_buffer);
621
622 char basedir[POCL_FILENAME_LENGTH];
623 pocl_cache_program_path (basedir, program, device_i);
624 size_t basedir_len = strlen (basedir);
625 char program_bc_path[POCL_FILENAME_LENGTH];
626 pocl_cache_program_bc_path (program_bc_path, program, device_i);
627 POCL_MSG_PRINT_INFO ("serializing program.bc: %s\n", program_bc_path);
628 buffer = serialize_file (program_bc_path, basedir_len, buffer);
629
630 unsigned i;
631 for (i=0; i < num_kernels; i++)
632 {
633 buffer = pocl_binary_serialize_kernel_to_buffer
634 (program, &program->kernel_meta[i], device_i, buffer);
635 assert(buffer <= end_of_buffer);
636 }
637
638 if (size)
639 *size = (buffer - start);
640 return CL_SUCCESS;
641}

643cl_int
644pocl_binary_deserialize(cl_program program, unsigned device_i)
645{
646 unsigned char *buffer = program->pocl_binaries[device_i];
647 size_t sizeof_buffer = program->pocl_binary_sizes[device_i];
648 unsigned char *end_of_buffer = buffer + sizeof_buffer;
649
650 pocl_binary b;
651 buffer = read_header(&b, buffer);
652 program->flush_denorms = (b.flags & POCL_BINARY_FLAG_FLUSH_DENORMS);
653 program->binary_type = (b.flags >> 1);
654
655 //assert(pocl_binary_check_binary_header(&b));
656 assert (buffer < end_of_buffer);
657
658 char basedir[POCL_FILENAME_LENGTH];
659 pocl_cache_program_path (basedir, program, device_i);
660 size_t basedir_len = strlen (basedir);
661 buffer += deserialize_file (buffer, basedir, basedir_len);
662
663 pocl_binary_kernel k;
664 unsigned i;
665 for (i = 0; i < b.num_kernels; i++)
666 {
667 pocl_cache_program_path (basedir, program, device_i);
668 if (pocl_binary_deserialize_kernel_from_buffer (&b, &buffer, &k, NULL, basedir)
669 != CL_SUCCESS)
670 goto ERROR;
671 assert (buffer <= end_of_buffer);
672 }
673 return CL_SUCCESS;
674
675ERROR:
676 return CL_OUT_OF_HOST_MEMORY;
677}

679#define MAX_BINARY_SIZE (256 << 20)
680
681size_t
682pocl_binary_sizeof_binary(cl_program program, unsigned device_i)
683{
684 if (program->pocl_binary_sizes[device_i])
685 return program->pocl_binary_sizes[device_i];
686
687 assert(program->pocl_binaries[device_i] == NULL);
688 /* dumb solution, but
689 * 1) it's simple,
690 * 2) we'll likely need the binary itself soon anyway,
691 * 3) memory is COW these days.. */
692 size_t res = 0;
693 unsigned char *temp_buf = malloc(MAX_BINARY_SIZE);
694 program->pocl_binaries[device_i] = temp_buf;
695 program->pocl_binary_sizes[device_i] = MAX_BINARY_SIZE;
696
697 if (pocl_binary_serialize(program, device_i, &res) != CL_SUCCESS)
698 {
699 POCL_MEM_FREE(program->pocl_binaries[device_i]);
700 program->pocl_binary_sizes[device_i] = 0;
701 return 0;
702 }
703
704 program->pocl_binaries[device_i] = malloc(res);
705 program->pocl_binary_sizes[device_i] = res;
706 memcpy(program->pocl_binaries[device_i], temp_buf, res);
707 free(temp_buf);
708 return res;
709
710}

713
714cl_int
715pocl_binary_get_kernels_metadata (cl_program program, unsigned device_i)
716{
717 unsigned char *binary = program->pocl_binaries[device_i];
718 cl_device_id device = program->devices[device_i];
719
720 pocl_binary b;
721 memset(&b, 0, sizeof (pocl_binary));
722 pocl_binary_kernel k;
723 memset(&k, 0, sizeof (pocl_binary_kernel));
724
725 unsigned char* buffer = read_header (&b, binary);
726 POCL_RETURN_ERROR_ON ((!pocl_binary_check_binary (device, binary)),
727 CL_INVALID_PROGRAM,
728 "Deserialized a binary, but it doesn't seem to be "
729 "for this device.\n");
730 size_t len;
731 /* skip real path of program.bc */
732 BUFFER_READ(len, uint32_t);
733 assert (len > 0);
734 buffer += len;
735
736 /* skip content of program.bc */
737 BUFFER_READ(len, uint32_t);
738 assert (len > 0);
739 buffer += len;
740
741 unsigned j;
742 assert (b.num_kernels > 0);
743 assert (b.num_kernels == program->num_kernels);
744
745 /* for each kernel, setup its metadata */
746 for (j = 0; j < b.num_kernels; j++)
747 {
748 pocl_kernel_metadata_t *km = &program->kernel_meta[j];
749
750 POCL_RETURN_ERROR_ON (pocl_binary_deserialize_kernel_from_buffer (
751 &b, &buffer, &k, km, NULL),
752 CL_INVALID_PROGRAM,
753 "Can't deserialize kernel %u \n", j);
754
755 km->num_args = k.num_args;
756 km->num_locals = k.num_locals;
757 km->local_sizes = k.local_sizes;
758 km->attributes = k.attributes;
759 km->has_arg_metadata = k.has_arg_metadata;
760 km->name = k.kernel_name;
761 km->data = (void **)calloc (program->num_devices, sizeof (void *));
762 assert (km->name);
763
764 unsigned l;
765 for (l = 0; l < OPENCL_MAX_DIMENSION; l++)
766 {
767 km->reqd_wg_size[l] = k.reqd_wg_size[l];
768 }
769 }
770
771 return CL_SUCCESS;
772}
