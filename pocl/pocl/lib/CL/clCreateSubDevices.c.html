30/* Creates an array of sub-devices that each reference a non-intersecting
31 set of compute units within in_device, according to a partition scheme
32 given by properties.
33 TODO we should actually pin the subdevices to specific cores,
34 and possibly correct the cache information accordingly.
35 */
36
37CL_API_ENTRY cl_int CL_API_CALL
38POname(clCreateSubDevices)(cl_device_id in_device,
39 const cl_device_partition_property *properties,
40 cl_uint num_devices,
41 cl_device_id *out_devices,
42 cl_uint *num_devices_ret) CL_API_SUFFIX__VERSION_1_2
43{
44 cl_int errcode = CL_SUCCESS;
45 cl_uint count_devices = 0;
46 cl_device_id *new_devs = NULL;
47 // number of elements in (copies of) properties, including terminating null
48 cl_uint num_props = 0;
49 cl_uint i;
50
51 POCL_GOTO_ERROR_COND((in_device == NULL), CL_INVALID_DEVICE);
52 POCL_GOTO_ERROR_COND((properties == NULL), CL_INVALID_VALUE);
53 POCL_GOTO_ERROR_COND((num_devices && !out_devices), CL_INVALID_VALUE);
54 POCL_GOTO_ERROR_COND((!num_devices && out_devices), CL_INVALID_VALUE);
55
56 POCL_GOTO_ERROR_ON (
57 (in_device->max_sub_devices == 0), CL_DEVICE_PARTITION_FAILED,
58 "Device %s cannot be further partitioned\n", in_device->short_name);
59
60 /* check that the partition property is supported by the device */
61 POCL_GOTO_ERROR_ON ((in_device->num_partition_properties == 0),
62 CL_INVALID_VALUE,
63 "Device %s does not support any partition property\n",
64 in_device->short_name);
65
66 for (i = 0; i < in_device->num_partition_properties; ++i) {
67 if (properties[0] == in_device->partition_properties[i]) {
68 break;
69 }
70 }
71
72 POCL_GOTO_ERROR_ON (
73 (i == in_device->num_partition_properties), CL_INVALID_VALUE,
74 "Device %s does not support the requested partition property\n",
75 in_device->short_name);
76
77 /* Ok, it's a supported partition property, count the number of devices; currently,
78 * we only support EQUALLY and BY_COUNTS, which enumerate the number of devices
79 * differently */
80 if (properties[0] == CL_DEVICE_PARTITION_EQUALLY)
81 {
82 /* error out if the number of CUs per device is 0 or bigger than the
83 * number of CUs of in_device */
84 POCL_GOTO_ERROR_COND (
85 (properties[1] == 0
86 || (cl_uint)properties[1] > in_device->max_compute_units),
87 CL_INVALID_VALUE);
88 // error out if properties isn't zero-terminated
89 POCL_GOTO_ERROR_COND (properties[2] != 0, CL_INVALID_VALUE);
90
91 count_devices = in_device->max_compute_units / properties[1];
92 num_props = 3; // partition type, CUs per device, terminating 0
93 }
94 else if (properties[0] == CL_DEVICE_PARTITION_BY_COUNTS)
95 {
96 cl_uint total_cus = 0;
97 i = 1;
98 while (properties[i] != 0)
99 {
100 count_devices++;
101 total_cus += properties[i];
102 ++i;
103 }
104 /* error out if the total number of CUs surpasses the number of device
105 * CUs, or if we have to many subdevices */
106 POCL_GOTO_ERROR_COND ((total_cus == 0),
107 CL_INVALID_DEVICE_PARTITION_COUNT);
108 POCL_GOTO_ERROR_COND ((total_cus > in_device->max_compute_units),
109 CL_INVALID_DEVICE_PARTITION_COUNT);
110 POCL_GOTO_ERROR_COND ((count_devices > in_device->max_sub_devices),
111 CL_INVALID_DEVICE_PARTITION_COUNT);
112 num_props = count_devices
113 + 2; /* partition type, one spec per device, terminating 0 */
114 }
115 else
116 {
117 /* we end here if some of our devices claim to support a different
118 * partition type, but this function was not updated accordingly */
119 POCL_GOTO_ERROR_ON (1, CL_INVALID_VALUE,
120 "Device reported partition type 0x%x "
121 "is not supported by Pocl\n",
122 (unsigned int)properties[0]);
123 }
124
125 // num_devices must be greater than or equal to count_devices if non-zero
126 POCL_GOTO_ERROR_COND((num_devices && num_devices < count_devices), CL_INVALID_VALUE);
127
128 if (out_devices) {
129 // we allocate our own array of devices to simplify management
130 new_devs = calloc(count_devices, sizeof(cl_device_id));
131 POCL_GOTO_ERROR_COND((!new_devs), CL_OUT_OF_HOST_MEMORY);
132 unsigned sum = 0;
133
134 for (i = 0; i < count_devices; ++i) {
135 new_devs[i] = calloc(1, sizeof(struct _cl_device_id));
136 POCL_GOTO_ERROR_COND((new_devs[i] == NULL), CL_OUT_OF_HOST_MEMORY);
137
138 // clone in_device
139 memcpy(new_devs[i], in_device, sizeof(struct _cl_device_id));
140 /* this must be done AFTER the clone, otherwise we end up with
141 * lock states and refcounts copied from parent device */
142 POCL_INIT_OBJECT (new_devs[i]);
143
144 new_devs[i]->parent_device = in_device;
145
146 new_devs[i]->max_sub_devices = new_devs[i]->max_compute_units
147 = (properties[0] == CL_DEVICE_PARTITION_EQUALLY
148 ? properties[1]
149 : properties[i + 1]);
150
151 /* for devices with 1 CU, report zero subdevices and
152 * no partitioning support. */
153 if (new_devs[i]->max_compute_units == 1)
154 {
155 new_devs[i]->max_sub_devices = 0;
156 new_devs[i]->num_partition_properties = 0;
157 new_devs[i]->partition_properties = NULL;
158 }
159
160 /* copy the partition type argument, for clGetDeviceInfo() */
161 new_devs[i]->partition_type = calloc(num_props, sizeof(*properties));
162 POCL_GOTO_ERROR_COND((new_devs[i]->partition_type == NULL),
163 CL_OUT_OF_HOST_MEMORY);
164 memcpy(new_devs[i]->partition_type, properties, num_props*sizeof(*properties));
165 new_devs[i]->num_partition_types = num_props;
166
167 new_devs[i]->core_count = new_devs[i]->max_compute_units;
168 if (in_device->parent_device)
169 new_devs[i]->core_start = in_device->core_start + sum;
170 else
171 new_devs[i]->core_start = sum;
172 sum += new_devs[i]->core_count;
173 }
174
175 memcpy(out_devices, new_devs, count_devices*sizeof(cl_device_id));
176 POCL_MEM_FREE(new_devs);
177 }
178
179 if (num_devices_ret)
180 *num_devices_ret = count_devices;
181
182 return errcode;
183
184ERROR:
185 if (new_devs) {
186 // release all objects
187 for (i = 0; i < count_devices; ++i) {
188 int new_refcount = 0;
189 // the first NULL new_devs indicates we're done
190 // with the ones we actually managed to allocate
191 if (new_devs[i] == NULL)
192 break;
193 POCL_RELEASE_OBJECT (new_devs[i], new_refcount);
194 if (new_refcount == 0)
195 {
196 POCL_MEM_FREE (new_devs[i]);
197 POCL_MEM_FREE (new_devs[i]->partition_type);
198 }
199 }
200
201 POCL_MEM_FREE (new_devs);
202 }
203 return errcode;
204
205}
