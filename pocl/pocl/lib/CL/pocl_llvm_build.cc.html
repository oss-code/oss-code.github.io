94// Read input source to clang::FrontendOptions.
95// The source is contained in the program->source array,
96// but if debugging option is enabled in the kernel compiler
97// we need to dump the file to disk first for the debugger
98// to find it.
99static inline int
100load_source(FrontendOptions &fe,
101 cl_program program)
102{
103 char source_file[POCL_FILENAME_LENGTH];
104 POCL_RETURN_ERROR_ON(pocl_cache_write_program_source(source_file, program),
105 CL_OUT_OF_HOST_MEMORY, "Could not write program source");
106
107 fe.Inputs.push_back
108 (FrontendInputFile(source_file, clang::InputKind::OpenCL));
109
110 return 0;
111}
112
113// Unlink input sources
114static inline int
115unlink_source(FrontendOptions &fe)
116{
117 if (pocl_get_bool_option("POCL_LEAVE_KERNEL_COMPILER_TEMP_FILES", 0) != 0)
118 return 0;
119
120 FrontendInputFile const& file = fe.Inputs.front();
121 if (file.isFile() && !file.isSystem()) {
122 return pocl_remove(file.getFile().str().c_str());
123 } else {
124 return 0; // nothing to do
125 }

129static void appendToProgramBuildLog(cl_program program, unsigned device_i,
130 std::string &s) {
131 if (!s.empty()) {
132 POCL_MSG_ERR("%s", s.c_str());
133 /* this may not actually write anything if the buildhash is invalid,
134 * but program->build_log still gets written. */
135 pocl_cache_append_to_buildlog(program, device_i, s.c_str(), s.size());
136 if (program->build_log[device_i]) {
137 size_t len = strlen(program->build_log[device_i]);
138 size_t len2 = strlen(s.c_str());
139 char *newlog = (char *)malloc(len + len2 + 1);
140 memcpy(newlog, program->build_log[device_i], len);
141 memcpy(newlog + len, s.c_str(), len2);
142 newlog[len + len2] = 0;
143 POCL_MEM_FREE(program->build_log[device_i]);
144 program->build_log[device_i] = newlog;
145 } else
146 program->build_log[device_i] = strdup(s.c_str());
147 }
148}

150static void get_build_log(cl_program program,
151 unsigned device_i,
152 std::stringstream &ss_build_log,
153 clang::TextDiagnosticBuffer *diagsBuffer,
154 const SourceManager *SM)
155{
156 for (TextDiagnosticBuffer::const_iterator i = diagsBuffer->err_begin(),
157 e = diagsBuffer->err_end(); i != e; ++i)
158 {
159 ss_build_log << "error: "
160 << (SM == nullptr ? "" : (i->first.printToString(*SM) + ": "))
161 << i->second << std::endl;
162 }
163 for (TextDiagnosticBuffer::const_iterator i = diagsBuffer->warn_begin(),
164 e = diagsBuffer->warn_end(); i != e; ++i)
165 {
166 ss_build_log << "warning: "
167 << (SM == nullptr ? "" : (i->first.printToString(*SM) + ": "))
168 << i->second << std::endl;
169 }
170
171 std::string log = ss_build_log.str();
172 appendToProgramBuildLog(program, device_i, log);
173}

182 std::string result = std::string(soname);

184 result = result.substr(0, last_slash+1);

194int pocl_llvm_build_program(cl_program program,
195 unsigned device_i,
196 const char *user_options_cstr,
197 char *program_bc_path,
198 cl_uint num_input_headers,
199 const cl_program *input_headers,
200 const char **header_include_names,
201 int linking_program)
202
203{
204 char tempfile[POCL_FILENAME_LENGTH];
205 tempfile[0] = 0;
206 llvm::Module **mod = NULL;
207 char temp_include_dir[POCL_FILENAME_LENGTH];
208 std::string user_options(user_options_cstr ? user_options_cstr : "");
209 size_t n = 0;
210 int error;
211
212 PoclCompilerMutexGuard lockHolder(NULL);
213 InitializeLLVM();
214
215 if (num_input_headers > 0) {
216 error = pocl_cache_create_tempdir(temp_include_dir);
217 if(error)
218 {
219 POCL_MSG_ERR ("pocl_cache_create_tempdir (%s)"
220 " failed with %i\n", temp_include_dir, error);
221 return error;
222 }
223 std::string tempdir(temp_include_dir);
224
225 for (n = 0; n < num_input_headers; n++) {
226 char *input_header = input_headers[n]->source;
227 size_t input_header_size = strlen(input_header);
228 const char *header_name = header_include_names[n];
229 std::string header(header_name);
230 /* TODO this path stuff should be in utils */
231 std::string path(tempdir);
232 path.append("/");
233 path.append(header_name);
234 size_t last_slash = header.rfind('/');
235 if (last_slash != std::string::npos) {
236 std::string dir(path, 0, (tempdir.size() + 1 + last_slash));
237 pocl_mkdir_p(dir.c_str());
238 }
239 pocl_write_file(path.c_str(), input_header, input_header_size, 0, 1);
240 }
241 }
242 // Use CompilerInvocation::CreateFromArgs to initialize
243 // CompilerInvocation. This way we can reuse the Clang's
244 // command line parsing.
245 llvm::IntrusiveRefCntPtr<clang::DiagnosticIDs> diagID =
246 new clang::DiagnosticIDs();
247 llvm::IntrusiveRefCntPtr<clang::DiagnosticOptions> diagOpts =
248 new clang::DiagnosticOptions();
249 clang::TextDiagnosticBuffer *diagsBuffer =
250 new clang::TextDiagnosticBuffer();
251
252 clang::DiagnosticsEngine diags(diagID, &*diagOpts, diagsBuffer);
253
254 CompilerInstance CI;
255 CompilerInvocation &pocl_build = CI.getInvocation();
256
257 std::stringstream ss;
258 std::stringstream ss_build_log;
259
260 // add device specific switches, if any
261 // TODO this currently passes NULL as device tmpdir
262 cl_device_id device = program->devices[device_i];
263 if (device->ops->init_build != NULL)
264 {
265 char *device_switches =
266 device->ops->init_build (device->data);
267 if (device_switches != NULL)
268 {
269 ss << device_switches << " ";
270 }
271 POCL_MEM_FREE(device_switches);
272 }
273
274 llvm::StringRef extensions(device->extensions);
275
276 std::string cl_ext;
277 if (extensions.size() > 0) {
278 size_t e_start = 0, e_end = 0;
279 while (e_end < std::string::npos) {
280 e_end = extensions.find(' ', e_start);
281 llvm::StringRef tok = extensions.slice(e_start, e_end);
282 e_start = e_end + 1;
283 ss << "-D" << tok.str() << " ";
284 cl_ext += "+";
285 cl_ext += tok.str();
286 cl_ext += ",";
287 }
288 }
289 if (!cl_ext.empty()) {
290 cl_ext.back() = ' '; // replace last "," with space
291 ss << "-cl-ext=-all," << cl_ext;
292 }
293 /* temp dir takes preference */
294 if (num_input_headers > 0)
295 ss << "-I" << temp_include_dir << " ";
296
297 if (device->has_64bit_long)
298 ss << "-Dcl_khr_int64 ";
299
300 ss << "-DPOCL_DEVICE_ADDRESS_BITS=" << device->address_bits << " ";
301 ss << "-D__USE_CLANG_OPENCL_C_H ";
302
303 ss << "-xcl ";
304 // Remove the inline keywords to force the user functions
305 // to be included in the program. Otherwise they will
306 // be removed and not inlined due to -O0.
307 ss << "-Dinline= ";
308 // The current directory is a standard search path.
309 ss << "-I. ";
310 // required for clGetKernelArgInfo()
311 ss << "-cl-kernel-arg-info ";
312
313 ss << user_options << " ";
314
315 if (device->endian_little)
316 ss << "-D__ENDIAN_LITTLE__=1 ";
317
318 if (device->image_support)
319 ss << "-D__IMAGE_SUPPORT__=1 ";
320
321 ss << "-DCL_DEVICE_MAX_GLOBAL_VARIABLE_SIZE=" << device->global_var_max_size << " ";
322
323 if (user_options.find("cl-fast-relaxed-math") != std::string::npos)
324 ss << "-D__FAST_RELAXED_MATH__=1 ";
325
326 ss << "-D__OPENCL_VERSION__=" << device->cl_version_int << " ";
327
328 if (user_options.find("-cl-std=") == std::string::npos)
329 ss << "-cl-std=" << device->cl_version_std << " ";
330
331 std::string temp(ss.str());
332 size_t pos = temp.find("-cl-std=CL");
333 pos += 10;
334 int cl_std_major = temp.c_str()[pos] - '0';
335 int cl_std_minor = temp.c_str()[pos+2] - '0';
336 int cl_std_i = cl_std_major * 100 + cl_std_minor * 10;
337 ss << "-D__OPENCL_C_VERSION__=" << cl_std_i << " ";
338
339 ss << "-fno-builtin ";
340 /* with fp-contract=on we get calls to fma with processors which do not
341 * have fma instructions. These ruin the performance.
342 *
343 * TODO find out which processors. Seems to be at least TCE
344 *
345 * default fp-contract is "on" which means "enable if enabled by a pragma".
346 */
347 llvm::Triple triple (device->llvm_target_triplet);
348 if (triple.getArch () == Triple::tce)
349 ss << "-ffp-contract=off ";
350
351 // This is required otherwise the initialization fails with
352 // unknown triple ''
353 ss << "-triple=" << device->llvm_target_triplet << " ";
354 if (device->llvm_cpu != NULL)
355 ss << "-target-cpu " << device->llvm_cpu << " ";
356
357 POCL_MSG_PRINT_LLVM("all build options: %s\n", ss.str().c_str());
358
359 std::istream_iterator<std::string> begin(ss);
360 std::istream_iterator<std::string> end;
361 std::istream_iterator<std::string> i = begin;
362 std::vector<const char*> itemcstrs;
363 std::vector<std::string> itemstrs;
364 while (i != end) {
365 itemstrs.push_back(*i);
366 ++i;
367 }
368
369 for (unsigned idx = 0; idx < itemstrs.size(); idx++) {
370 // note: if itemstrs is modified after this, itemcstrs will be full
371 // of invalid pointers! Could make copies, but would have to clean up then...
372 itemcstrs.push_back(itemstrs[idx].c_str());
373 }
374
375#ifdef DEBUG_POCL_LLVM_API
376 // TODO: for some reason the user_options are replicated,
377 // they appear twice in a row in the output
378 std::cerr << "### options: " << ss.str()
379 << "user_options: " << user_options << std::endl;
380#endif
381
382 if (program->build_log[device_i])
383 POCL_MEM_FREE(program->build_log[device_i]);
384
385 if (!CompilerInvocation::CreateFromArgs
386 (pocl_build, itemcstrs.data(), itemcstrs.data() + itemcstrs.size(),
387 diags)) {
388 pocl_cache_create_program_cachedir(program, device_i, NULL, 0,
389 program_bc_path);
390 get_build_log(program, device_i, ss_build_log, diagsBuffer,
391 CI.hasSourceManager() ? &CI.getSourceManager() : nullptr);
392 return CL_INVALID_BUILD_OPTIONS;
393 }
394
395 LangOptions *la = pocl_build.getLangOpts();
396 PreprocessorOptions &po = pocl_build.getPreprocessorOpts();
397
398 pocl_build.setLangDefaults
399 (*la, clang::InputKind::OpenCL, triple, po,
400 clang::LangStandard::lang_opencl12);
401
402 // LLVM 3.3 and older do not set that char is signed which is
403 // defined by the OpenCL C specs (but not by C specs).
404 la->CharIsSigned = true;
405
406 // the per-file types don't seem to override this
407 la->OpenCLVersion = cl_std_i;
408 la->FakeAddressSpaceMap = false;
409 la->Blocks = true; //-fblocks
410 la->MathErrno = false; // -fno-math-errno
411 la->NoBuiltin = true; // -fno-builtin
412 la->AsmBlocks = true; // -fasm (?)
413
414 la->setStackProtector(LangOptions::StackProtectorMode::SSPOff);
415
416 la->PICLevel = PICLevel::BigPIC;
417 la->PIE = 1;
418
419 std::string IncludeRoot;
420 std::string KernelH;
421 std::string BuiltinRenamesH;
422 std::string PoclTypesH;
423 std::string ClangResourceDir;
424
425#ifdef ENABLE_POCL_BUILDING
426 if (pocl_get_bool_option("POCL_BUILDING", 0)) {
427 IncludeRoot = SRCDIR;
428#else
429 if (0) {
430#endif
431 } else {
432 IncludeRoot = getPoclPrivateDataDir();
433#ifdef ENABLE_RELOCATION
434 ClangResourceDir = IncludeRoot;
435#endif
436 }
437 if (ClangResourceDir.empty()) {
438#ifndef LLVM_OLDER_THAN_9_0
439 ClangResourceDir = driver::Driver::GetResourcesPath(CLANG);
440#else
441 DiagnosticsEngine Diags{new DiagnosticIDs, new DiagnosticOptions};
442 driver::Driver TheDriver(CLANG, "", Diags);
443 ClangResourceDir = TheDriver.ResourceDir;
444#endif
445 }
446 KernelH = IncludeRoot + "/include/_kernel.h";
447 BuiltinRenamesH = IncludeRoot + "/include/_builtin_renames.h";
448 PoclTypesH = IncludeRoot + "/include/pocl_types.h";
449
450 po.Includes.push_back(PoclTypesH);
451 po.Includes.push_back(BuiltinRenamesH);
452 // Use Clang's opencl-c.h header.
453#ifndef LLVM_OLDER_THAN_9_0
454 po.Includes.push_back(ClangResourceDir + "/include/opencl-c-base.h");
455#endif
456 po.Includes.push_back(ClangResourceDir + "/include/opencl-c.h");
457 po.Includes.push_back(KernelH);
458 clang::TargetOptions &ta = pocl_build.getTargetOpts();
459 ta.Triple = device->llvm_target_triplet;
460 if (device->llvm_cpu != NULL)
461 ta.CPU = device->llvm_cpu;
462
463#ifdef DEBUG_POCL_LLVM_API
464 std::cout << "### Triple: " << ta.Triple.c_str() << ", CPU: " << ta.CPU.c_str();
465#endif
466 CI.createDiagnostics(diagsBuffer, false);
467
468 FrontendOptions &fe = pocl_build.getFrontendOpts();
469 // The CreateFromArgs created an stdin input which we should remove first.
470 fe.Inputs.clear();
471 if (load_source(fe, program) != 0)
472 return CL_OUT_OF_HOST_MEMORY;
473
474 CodeGenOptions &cg = pocl_build.getCodeGenOpts();
475 cg.EmitOpenCLArgMetadata = true;
476 cg.StackRealignment = true;
477 // Let the vectorizer or another optimization pass unroll the loops,
478 // in case it sees beneficial.
479 cg.UnrollLoops = false;
480 // Lets leave vectorization to later compilation phase
481 cg.VectorizeLoop = false;
482 cg.VectorizeSLP = false;
483 cg.VerifyModule = true;
484
485 PreprocessorOutputOptions &poo = pocl_build.getPreprocessorOutputOpts();
486 poo.ShowCPP = 1;
487 poo.ShowComments = 0;
488 poo.ShowLineMarkers = 0;
489 poo.ShowMacroComments = 0;
490 poo.ShowMacros = 1;
491 poo.RewriteIncludes = 0;
492
493 pocl_cache_tempname(tempfile, ".preproc.cl", NULL);
494 fe.OutputFile.assign(tempfile);
495
496 bool success = true;
497 clang::PrintPreprocessedAction Preprocess;
498 success = CI.ExecuteAction(Preprocess);
499 char *PreprocessedOut = nullptr;
500 uint64_t PreprocessedSize = 0;
501
502 if (success) {
503 pocl_read_file(tempfile, &PreprocessedOut, &PreprocessedSize);
504 }
505 /* always remove preprocessed output - the sources are in different files */
506 pocl_remove(tempfile);
507
508 if (pocl_get_bool_option("POCL_LEAVE_KERNEL_COMPILER_TEMP_FILES", 0) == 0) {
509 if (num_input_headers > 0)
510 pocl_rm_rf(temp_include_dir);
511 }
512
513 if (PreprocessedOut == nullptr) {
514 pocl_cache_create_program_cachedir(program, device_i, NULL, 0,
515 program_bc_path);
516 get_build_log(program, device_i, ss_build_log, diagsBuffer,
517 CI.hasSourceManager() ? &CI.getSourceManager() : nullptr);
518 return CL_BUILD_PROGRAM_FAILURE;
519 }
520
521 pocl_cache_create_program_cachedir(program, device_i, PreprocessedOut,
522 static_cast<size_t>(PreprocessedSize), program_bc_path);
523
524 POCL_MEM_FREE(PreprocessedOut);
525
526 unlink_source(fe);
527
528 if (pocl_exists(program_bc_path))
529 return CL_SUCCESS;
530
531 // TODO: use pch: it is possible to disable the strict checking for
532 // the compilation flags used to compile it and the current translation
533 // unit via the preprocessor options directly.
534 llvm::LLVMContext &c = GlobalContext();
535 clang::EmitLLVMOnlyAction EmitLLVM(&c);
536 success = CI.ExecuteAction(EmitLLVM);
537
538 get_build_log(program, device_i, ss_build_log, diagsBuffer, &CI.getSourceManager());
539
540 if (!success)
541 return CL_BUILD_PROGRAM_FAILURE;
542
543 mod = (llvm::Module **)&program->llvm_irs[device_i];
544 if (*mod != NULL) {
545 delete *mod;
546 --numberOfIRs;
547 }
548
549 *mod = EmitLLVM.takeModule().release();
550
551 if (*mod == NULL)
552 return CL_BUILD_PROGRAM_FAILURE;
553
554 ++numberOfIRs;
555
556 if ((*mod)->getModuleFlag("PIC Level") == nullptr)
557 (*mod)->setPICLevel(PICLevel::BigPIC);
558 if ((*mod)->getModuleFlag("PIE Level") == nullptr)
559 (*mod)->setPIELevel(PIELevel::Large);
560
561 // link w kernel lib, but not if we're called from clCompileProgram()
562 // Later this should be replaced with indexed linking of source code
563 // and/or bitcode for each kernel.
564 if (linking_program) {
565 currentPoclDevice = device;
566 llvm::Module *libmodule = getKernelLibrary(device);
567 assert(libmodule != NULL);
568 std::string log("Error(s) while linking: \n");
569 if (link(*mod, libmodule, log)) {
570 appendToProgramBuildLog(program, device_i, log);
571 std::string msg = getDiagString();
572 appendToProgramBuildLog(program, device_i, msg);
573 delete *mod;
574 *mod = nullptr;
575 --numberOfIRs;
576 return CL_BUILD_PROGRAM_FAILURE;
577 }
578 }
579
580 POCL_MSG_PRINT_LLVM("Writing program.bc to %s.\n", program_bc_path);
581
582 /* Always retain program.bc. Its required in clBuildProgram */
583 error = pocl_write_module(*mod, program_bc_path, 0);
584 if(error)
585 return error;
586
587 /* To avoid writing & reading the same back,
588 * save program->binaries[i]
589 */
590 std::string content;
591 writeModuleIR(*mod, content);
592
593 if (program->binaries[device_i])
594 POCL_MEM_FREE(program->binaries[device_i]);
595
596 n = content.size();
597 program->binary_sizes[device_i] = n;
598 program->binaries[device_i] = (unsigned char *) malloc(n);
599 std::memcpy(program->binaries[device_i], content.c_str(), n);
600
601 return CL_SUCCESS;
602}

604int pocl_llvm_link_program(cl_program program, unsigned device_i,
605 char *program_bc_path, cl_uint num_input_programs,
606 unsigned char **cur_device_binaries,
607 size_t *cur_device_binary_sizes, void **cur_llvm_irs,
608 int create_library, int spir) {
609
610 std::string concated_binaries;
611 llvm::Module *linked_module = nullptr;
612 size_t n = 0, i;
613 cl_device_id device = program->devices[device_i];
614 llvm::Module **modptr = (llvm::Module **)&program->llvm_irs[device_i];
615 int error;
616
617 currentPoclDevice = device;
618 llvm::Module *libmodule = getKernelLibrary(device);
619 assert(libmodule != NULL);
620
621 PoclCompilerMutexGuard lockHolder(NULL);
622 InitializeLLVM();
623
624 if (spir) {
625#ifdef ENABLE_SPIR
626
627 POCL_RETURN_ERROR_ON ((device->endian_little == CL_FALSE),
628 CL_LINK_PROGRAM_FAILURE,
629 "SPIR is only supported on little-endian devices\n");
630
631 linked_module = parseModuleIRMem((char *)program->binaries[device_i],
632 program->binary_sizes[device_i]);
633
634 const std::string &spir_triple = linked_module->getTargetTriple();
635 size_t spir_addrbits = Triple(spir_triple).isArch64Bit() ? 64 : 32;
636
637 if (device->address_bits != spir_addrbits) {
638 delete linked_module;
639 POCL_RETURN_ERROR_ON (1, CL_LINK_PROGRAM_FAILURE,
640 "Device address bits != SPIR binary triple address "
641 "bits, device: %s / module: %s\n",
642 device->llvm_target_triplet, spir_triple.c_str());
643 }
644
645 /* Note this is a hack to get SPIR working. We'll be linking the
646 * host kernel library (plain LLVM IR) to the SPIR program.bc,
647 * so LLVM complains about incompatible DataLayouts.
648 */
649 linked_module->setTargetTriple(libmodule->getTargetTriple());
650 linked_module->setDataLayout(libmodule->getDataLayout());
651 if (linked_module->getModuleFlag("PIC Level") == nullptr)
652 linked_module->setPICLevel(PICLevel::BigPIC);
653 if (linked_module->getModuleFlag("PIE Level") == nullptr)
654 linked_module->setPIELevel(PIELevel::Large);
655
656#else
657 POCL_MSG_ERR("SPIR not supported\n");
658 return CL_LINK_PROGRAM_FAILURE;
659#endif
660 } else {
661
662 std::unique_ptr<llvm::Module> mod(
663 new llvm::Module(StringRef("linked_program"), GlobalContext()));
664
665 for (i = 0; i < num_input_programs; i++) {
666 assert(cur_device_binaries[i]);
667 assert(cur_device_binary_sizes[i]);
668 concated_binaries.append(std::string((char *)cur_device_binaries[i],
669 cur_device_binary_sizes[i]));
670
671 llvm::Module *p = (llvm::Module *)cur_llvm_irs[i];
672 assert(p);
673
674#ifdef LLVM_OLDER_THAN_7_0
675 if (Linker::linkModules(*mod, llvm::CloneModule(p))) {
676#else
677 if (Linker::linkModules(*mod, llvm::CloneModule(*p))) {
678#endif
679 std::string msg = getDiagString();
680 appendToProgramBuildLog(program, device_i, msg);
681 return CL_LINK_PROGRAM_FAILURE;
682 }
683 }
684
685 linked_module = mod.release();
686 }
687
688 if (linked_module == nullptr)
689 return CL_LINK_PROGRAM_FAILURE;
690
691 if (*modptr != nullptr) {
692 delete *modptr;
693 --numberOfIRs;
694 *modptr = nullptr;
695 }
696
697 if (!create_library) {
698 // linked all the programs together, now link in the kernel library
699 std::string log("Error(s) while linking: \n");
700 if (link(linked_module, libmodule, log)) {
701 appendToProgramBuildLog(program, device_i, log);
702 std::string msg = getDiagString();
703 appendToProgramBuildLog(program, device_i, msg);
704 delete linked_module;
705 return CL_BUILD_PROGRAM_FAILURE;
706 }
707 }
708
709 *modptr = linked_module;
710 ++numberOfIRs;
711
712 /* TODO currently cached on concated binary contents (in undefined order),
713 this is not terribly useful (but we have to store it somewhere..) */
714 error = pocl_cache_create_program_cachedir(program, device_i,
715 concated_binaries.c_str(),
716 concated_binaries.size(),
717 program_bc_path);
718 if (error)
719 {
720 POCL_MSG_ERR ("pocl_cache_create_program_cachedir(%s)"
721 " failed with %i\n", program_bc_path, error);
722 return error;
723 }
724
725 POCL_MSG_PRINT_LLVM("Writing program.bc to %s.\n", program_bc_path);
726
727 /* Always retain program.bc. Its required in clBuildProgram */
728 error = pocl_write_module(linked_module, program_bc_path, 0);
729 if (error)
730 return error;
731
732 /* To avoid writing & reading the same back,
733 * save program->binaries[i]
734 */
735 std::string content;
736 writeModuleIR(linked_module, content);
737
738 if (program->binaries[device_i])
739 POCL_MEM_FREE(program->binaries[device_i]);
740
741 n = content.size();
742 program->binary_sizes[device_i] = n;
743 program->binaries[device_i] = (unsigned char *)malloc(n);
744 std::memcpy(program->binaries[device_i], content.c_str(), n);
745
746 return CL_SUCCESS;
747}

750#ifdef KERNELLIB_HOST_DISTRO_VARIANTS
751const char *getX86KernelLibName() {
752 StringMap<bool> Features;
753 llvm::sys::getHostCPUFeatures(Features);
754 const char *res = NULL;
755
756 if (Features["sse2"])
757 res = "sse2";
758 else
759 POCL_ABORT("Pocl on x86_64 requires at least SSE2\n");
760 if (Features["ssse3"] && Features["cx16"])
761 res = "ssse3";
762 if (Features["sse4.1"] && Features["cx16"])
763 res = "sse41";
764 if (Features["avx"] && Features["cx16"] && Features["popcnt"])
765 res = "avx";
766 if (Features["avx"] && Features["cx16"] && Features["popcnt"] && Features["f16c"])
767 res = "avx_f16c";
768 if (Features["avx"] && Features["cx16"] && Features["popcnt"]
769 && Features["xop"] && Features["fma4"])
770 res = "avx_fma4";
771 if (Features["avx"] && Features["avx2"] && Features["cx16"]
772 && Features["popcnt"] && Features["lzcnt"] && Features["f16c"]
773 && Features["fma"] && Features["bmi"] && Features["bmi2"])
774 res = "avx2";
775 if (Features["avx512f"] )
776 res = "avx512";
777
778 return res;
779}

783static std::map<cl_device_id, llvm::Module *> kernelLibraryMap;
784
785/**
786 * Return the OpenCL C built-in function library bitcode
787 * for the given device.
788 */
789static llvm::Module* getKernelLibrary(cl_device_id device)
790{
791 Triple triple(device->llvm_target_triplet);
792
793 if (kernelLibraryMap.find(device) != kernelLibraryMap.end())
794 return kernelLibraryMap[device];
795
796 const char *subdir = "host";
797 bool is_host = true;
798 // TODO: move this to the device layer, a property to ask for
799 // the kernel builtin bitcode library name, including its subdir
800#ifdef TCE_AVAILABLE
801 if (triple.getArch() == Triple::tce || triple.getArch() == Triple::tcele) {
802 subdir = "tce";
803 is_host = false;
804 }
805#endif
806#ifdef BUILD_HSA
807 if (triple.getArch() == Triple::hsail64) {
808 subdir = "hsail64";
809 is_host = false;
810 }
811#endif
812#ifdef AMDGCN_ENABLED
813 if (triple.getArch == Triple::amdgcn) {
814 subdir = "amdgcn";
815 is_host = false;
816 }
817#endif
818#ifdef BUILD_CUDA
819 if (triple.getArch() == Triple::nvptx ||
820 triple.getArch() == Triple::nvptx64) {
821 subdir = "cuda";
822 is_host = false;
823 }
824#endif
825
826 std::string kernellib;
827 std::string kernellib_fallback;
828#ifdef ENABLE_POCL_BUILDING
829 if (pocl_get_bool_option("POCL_BUILDING", 0)) {
830 kernellib = BUILDDIR;
831 kernellib += "/lib/kernel/";
832 kernellib += subdir;
833 } else // POCL_BUILDING == 0, use install dir
834#endif
835 kernellib = getPoclPrivateDataDir();
836 kernellib += "/kernel-";
837 kernellib += device->llvm_target_triplet;
838 if (is_host) {
839 kernellib += '-';
840#ifdef KERNELLIB_HOST_DISTRO_VARIANTS
841 kernellib += getX86KernelLibName();
842#else
843 kernellib_fallback = kernellib;
844 kernellib_fallback += OCL_KERNEL_TARGET_CPU;
845 kernellib_fallback += ".bc";
846 kernellib += device->llvm_cpu;
847#endif
848 }
849 kernellib += ".bc";
850
851 llvm::Module *lib;
852
853 if (pocl_exists(kernellib.c_str()))
854 {
855 POCL_MSG_PRINT_LLVM("Using %s as the built-in lib.\n", kernellib.c_str());
856 lib = parseModuleIR(kernellib.c_str());
857 }
858 else
859 {
860#ifndef KERNELLIB_HOST_DISTRO_VARIANTS
861 if (is_host && pocl_exists(kernellib_fallback.c_str()))
862 {
863 POCL_MSG_WARN("Using fallback %s as the built-in lib.\n",
864 kernellib_fallback.c_str());
865 lib = parseModuleIR(kernellib_fallback.c_str());
866 }
867 else
868#endif
869 POCL_ABORT("Kernel library file %s doesn't exist.\n", kernellib.c_str());
870 }
871 assert (lib != NULL);
872 kernelLibraryMap[device] = lib;
873
874 return lib;
875}

877void cleanKernelLibrary() {
878 for (auto i = kernelLibraryMap.begin(), e = kernelLibraryMap.end();
879 i != e; ++i) {
880 delete (llvm::Module *)i->second;
881 }
882 kernelLibraryMap.clear();
883}

885/**
886 * Invoke the Clang compiler through its Driver API.
887 *
888 * @param Device the device of which toolchain to use.
889 * @param Args the command line arguments that would be passed to Clang
890 * (a NULL terminated list). Args[0] should be the path to
891 * the Clang binary.
892 * @return 0 on success, error code otherwise.
893 */
894int pocl_invoke_clang(cl_device_id Device, const char** Args) {
895
896 // Borrowed from driver.cpp (clang driver). We do not really care about
897 // diagnostics, but just want to get the compilation command invoked with
898 // the target's toolchain as defined in Clang.
899 IntrusiveRefCntPtr<DiagnosticOptions> DiagOpts = new DiagnosticOptions;
900
901 TextDiagnosticPrinter *DiagClient =
902 new TextDiagnosticPrinter(llvm::errs(), &*DiagOpts);
903
904 IntrusiveRefCntPtr<DiagnosticIDs> DiagID(new DiagnosticIDs());
905
906 DiagnosticsEngine Diags(DiagID, &*DiagOpts, DiagClient);
907
908 driver::Driver TheDriver(CLANG, Device->llvm_target_triplet, Diags);
909
910 const char **ArgsEnd = Args;
911 while (*ArgsEnd++ != nullptr) {}
912
913 llvm::ArrayRef<const char*> ArgsArray(Args, ArgsEnd);
914
915 std::unique_ptr<driver::Compilation> C(
916 TheDriver.BuildCompilation(ArgsArray));
917
918 if (C && !C->containsError()) {
919 SmallVector<std::pair<int, const driver::Command *>, 4> FailingCommands;
920 return TheDriver.ExecuteCompilation(*C, FailingCommands);
921 } else {
922 return -1;
923 }
