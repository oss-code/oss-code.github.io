6POname(clEnqueueCopyBufferToImage)(cl_command_queue command_queue,
7 cl_mem src_buffer,
8 cl_mem dst_image,
9 size_t src_offset,
10 const size_t * dst_origin, /*[3]*/
11 const size_t * region, /*[3]*/
12 cl_uint num_events_in_wait_list,
13 const cl_event * event_wait_list,
14 cl_event * event )
15CL_API_SUFFIX__VERSION_1_0
16{
17 _cl_command_node *cmd = NULL;
18 /* pass src_origin through in a format pocl_rect_copy understands */
19 const size_t src_origin[3] = { src_offset, 0, 0};
20
21 POCL_RETURN_ERROR_COND ((command_queue == NULL), CL_INVALID_COMMAND_QUEUE);
22
23 POCL_RETURN_ERROR_COND ((dst_image == NULL), CL_INVALID_MEM_OBJECT);
24
25 if (IS_IMAGE1D_BUFFER (dst_image))
26 {
27 /* If src_image is a 1D image or 1D image buffer object, src_origin[1]
28 * and src_origin[2] must be 0 If src_image is a 1D image or 1D image
29 * buffer object, region[1] and region[2] must be 1. */
30 IMAGE1D_ORIG_REG_TO_BYTES (dst_image, dst_origin, region);
31 return POname (clEnqueueCopyBufferRect (
32 command_queue, src_buffer, dst_image->buffer,
33 src_origin, i1d_origin, i1d_region,
34 dst_image->image_row_pitch, 0,
35 dst_image->image_row_pitch, 0,
36 num_events_in_wait_list, event_wait_list, event));
37 }
38
39 POCL_RETURN_ON_SUB_MISALIGN (src_buffer, command_queue);
40
41 cl_int err = pocl_rect_copy(
42 command_queue,
43 CL_COMMAND_COPY_BUFFER_TO_IMAGE,
44 src_buffer, CL_FALSE,
45 dst_image, CL_TRUE,
46 src_origin, dst_origin, region,
47 0, 0,
48 0, 0,
49 num_events_in_wait_list, event_wait_list,
50 event,
51 &cmd);
52
53 if (err != CL_SUCCESS)
54 return err;
55
56 POCL_CONVERT_SUBBUFFER_OFFSET (src_buffer, src_offset);
57
58 POCL_RETURN_ERROR_ON((src_buffer->size > command_queue->device->max_mem_alloc_size),
59 CL_OUT_OF_RESOURCES,
60 "src is larger than device's MAX_MEM_ALLOC_SIZE\n");
61
62 cl_device_id dev = command_queue->device;
63
64 cmd->command.write_image.dst_mem_id = &dst_image->device_ptrs[dev->dev_id];
65 cmd->command.write_image.src_host_ptr = NULL;
66 cmd->command.write_image.src_mem_id = &src_buffer->device_ptrs[dev->dev_id];
67
68 // TODO
69 cmd->command.write_image.src_row_pitch = 0; // dst_image->image_row_pitch;
70 cmd->command.write_image.src_slice_pitch = 0; // dst_image->image_slice_pitch;
71 cmd->command.write_image.src_offset = src_offset;
72
73 cmd->command.write_image.origin[0] = dst_origin[0];
74 cmd->command.write_image.origin[1] = dst_origin[1];
75 cmd->command.write_image.origin[2] = dst_origin[2];
76 cmd->command.write_image.region[0] = region[0];
77 cmd->command.write_image.region[1] = region[1];
78 cmd->command.write_image.region[2] = region[2];
79
80 POname (clRetainMemObject) (dst_image);
81 dst_image->owning_device = dev;
82 POname (clRetainMemObject) (src_buffer);
83 src_buffer->owning_device = dev;
84
85 pocl_command_enqueue (command_queue, cmd);
86
87 return CL_SUCCESS;
