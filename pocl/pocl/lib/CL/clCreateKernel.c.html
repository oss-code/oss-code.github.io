39CL_API_ENTRY cl_kernel CL_API_CALL
40POname(clCreateKernel)(cl_program program,
41 const char *kernel_name,
42 cl_int *errcode_ret) CL_API_SUFFIX__VERSION_1_0
43{
44 cl_kernel kernel = NULL;
45 int errcode = CL_SUCCESS;
46 size_t i;
47
48 POCL_GOTO_ERROR_COND((kernel_name == NULL), CL_INVALID_VALUE);
49
50 POCL_GOTO_ERROR_COND((program == NULL), CL_INVALID_PROGRAM);
51
52 POCL_GOTO_ERROR_ON((program->num_devices == 0),
53 CL_INVALID_PROGRAM, "Invalid program (has no devices assigned)\n");
54
55 POCL_GOTO_ERROR_ON((program->build_status == CL_BUILD_NONE),
56 CL_INVALID_PROGRAM_EXECUTABLE, "You must call clBuildProgram first!"
57 " (even for programs created with binaries)\n");
58
59 POCL_GOTO_ERROR_ON((program->build_status != CL_BUILD_SUCCESS),
60 CL_INVALID_PROGRAM_EXECUTABLE, "Last BuildProgram() was not successful\n");
61
62 POCL_GOTO_ERROR_ON (
63 (program->builtin_kernel_names == NULL && program->llvm_irs == NULL),
64 CL_INVALID_PROGRAM_EXECUTABLE,
65 "No built binaries nor built-in kernels in program "
66 "(this shouldn't happen...)\n");
67
68 kernel = (cl_kernel) calloc(1, sizeof(struct _cl_kernel));
69 POCL_GOTO_ERROR_ON((kernel == NULL), CL_OUT_OF_HOST_MEMORY,
70 "clCreateKernel couldn't allocate memory");
71
72 POCL_INIT_OBJECT (kernel);
73
74 for (i = 0; i < program->num_kernels; ++i)
75 if (strcmp (program->kernel_meta[i].name, kernel_name) == 0)
76 break;
77
78 POCL_GOTO_ERROR_ON ((i >= program->num_kernels), CL_INVALID_KERNEL_NAME,
79 "Can't find a kernel with name %s in this program\n",
80 kernel_name);
81
82 kernel->meta = &program->kernel_meta[i];
83 kernel->name = kernel->meta->name;
84 kernel->context = program->context;
85 kernel->program = program;
86
87 kernel->dyn_arguments
88 = calloc ((kernel->meta->num_args), sizeof (struct pocl_argument));
89 POCL_GOTO_ERROR_COND ((kernel->dyn_arguments == NULL),
90 CL_OUT_OF_HOST_MEMORY);
91
92 POCL_LOCK_OBJ (program);
93 LL_PREPEND (program->kernels, kernel);
94 POCL_RETAIN_OBJECT_UNLOCKED (program);
95 POCL_UNLOCK_OBJ (program);
96
97 errcode = CL_SUCCESS;
98 goto SUCCESS;
99
100ERROR:
101 if (kernel)
102 POCL_MEM_FREE (kernel->dyn_arguments);
103 POCL_MEM_FREE (kernel);
104 kernel = NULL;
105
106SUCCESS:
107 if(errcode_ret != NULL)
108 {
109 *errcode_ret = errcode;
110 }
111 return kernel;
112}
