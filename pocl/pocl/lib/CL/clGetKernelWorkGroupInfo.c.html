25#include "devices/devices.h"
26#include "pocl_util.h"
27
28extern CL_API_ENTRY cl_int CL_API_CALL
29POname(clGetKernelWorkGroupInfo)
30(cl_kernel kernel,
31 cl_device_id device,
32 cl_kernel_work_group_info param_name,
33 size_t param_value_size,
34 void *param_value,
35 size_t * param_value_size_ret)
36 CL_API_SUFFIX__VERSION_1_0
37{
38
39 POCL_RETURN_ERROR_COND ((kernel == NULL), CL_INVALID_KERNEL);
40
41 /* Check that kernel is associated with device, or that there is no
42 risk of confusion. */
43 if (device != NULL)
44 {
45 unsigned i;
46 int found_it = 0;
47 for (i = 0; i < kernel->context->num_devices; i++)
48 if (pocl_real_dev (device) == kernel->context->devices[i])
49 {
50 found_it = 1;
51 break;
52 }
53 POCL_RETURN_ERROR_ON((!found_it), CL_INVALID_DEVICE, "could not find the "
54 "device supplied in argument\n");
55 }
56 else
57 {
58 POCL_RETURN_ERROR_ON((kernel->context->num_devices > 1), CL_INVALID_DEVICE,
59 "No device given and context has > 1 device\n");
60 device = kernel->context->devices[0];
61 }
62
63 switch (param_name)
64 {
65 case CL_KERNEL_WORK_GROUP_SIZE:
66 return POname(clGetDeviceInfo)
67 (device, CL_DEVICE_MAX_WORK_GROUP_SIZE, param_value_size,
68 param_value, param_value_size_ret);
69 case CL_KERNEL_COMPILE_WORK_GROUP_SIZE:
70 {
71 typedef struct { size_t size[3]; } size_t_3;
72 POCL_MSG_PRINT_GENERAL (
73 "### reqd wg sizes %zu %zu %zu\n", kernel->meta->reqd_wg_size[0],
74 kernel->meta->reqd_wg_size[1], kernel->meta->reqd_wg_size[2]);
75 POCL_RETURN_GETINFO (size_t_3,
76 *(size_t_3 *)kernel->meta->reqd_wg_size);
77 }
78 case CL_KERNEL_PREFERRED_WORK_GROUP_SIZE_MULTIPLE:
79 POCL_RETURN_GETINFO(size_t, device->preferred_wg_size_multiple);
80 case CL_KERNEL_LOCAL_MEM_SIZE:
81 {
82 size_t local_size = 0, i;
83
84 /* Count the host-allocated locals. */
85 for (i = 0; i < kernel->meta->num_args; ++i)
86 {
87 if (!ARG_IS_LOCAL (kernel->meta->arg_info[i]))
88 continue;
89 local_size += kernel->dyn_arguments[i].size;
90 }
91 /* Count the automatic locals. */
92 for (i = 0; i < kernel->meta->num_locals; ++i)
93 {
94 local_size += kernel->meta->local_sizes[i];
95 }
96 POCL_RETURN_GETINFO(cl_ulong, local_size);
97 }
98 case CL_KERNEL_PRIVATE_MEM_SIZE:
99 POCL_MSG_WARN ("clGetKernelWorkGroupInfo: CL_KERNEL_PRIVATE_MEM_SIZE "
100 "implementation is incomplete\n");
101 POCL_RETURN_GETINFO (cl_ulong, 1024);
102
103 default:
104 return CL_INVALID_VALUE;
105 }
106 return CL_SUCCESS;
107}
