39CL_API_ENTRY cl_int CL_API_CALL
40POname(clReleaseProgram)(cl_program program) CL_API_SUFFIX__VERSION_1_0
41{
42 int new_refcount;
43 unsigned i, j;
44
45 POCL_RETURN_ERROR_COND((program == NULL), CL_INVALID_PROGRAM);
46
47 POCL_RELEASE_OBJECT (program, new_refcount);
48 POCL_MSG_PRINT_REFCOUNTS ("Release program %p, new refcount: %d, kernel #: %zu \n", program, new_refcount, program->num_kernels);
49
50 if (new_refcount == 0)
51 {
52 cl_context context = program->context;
53 POCL_MSG_PRINT_REFCOUNTS ("Free program %p\n", program);
54
55 /* there should be no kernels left when we're releasing the program */
56 assert (program->kernels == NULL);
57
58 if (program->devices != program->context->devices)
59 POCL_MEM_FREE(program->devices);
60
61 POCL_MEM_FREE(program->source);
62
63 POCL_MEM_FREE (program->program_il);
64
65 POCL_MEM_FREE(program->binary_sizes);
66 if (program->binaries)
67 for (i = 0; i < program->num_devices; ++i)
68 POCL_MEM_FREE(program->binaries[i]);
69 POCL_MEM_FREE(program->binaries);
70
71 POCL_MEM_FREE(program->pocl_binary_sizes);
72 if (program->pocl_binaries)
73 for (i = 0; i < program->num_devices; ++i)
74 POCL_MEM_FREE(program->pocl_binaries[i]);
75 POCL_MEM_FREE(program->pocl_binaries);
76
77 pocl_cache_cleanup_cachedir(program);
78
79 if (program->build_log)
80 for (i = 0; i < program->num_devices; ++i)
81 POCL_MEM_FREE(program->build_log[i]);
82 POCL_MEM_FREE(program->build_log);
83
84 if (program->num_kernels)
85 {
86 for (i = 0; i < program->num_kernels; i++)
87 {
88 pocl_kernel_metadata_t *meta = &program->kernel_meta[i];
89 POCL_MEM_FREE (meta->attributes);
90 POCL_MEM_FREE (meta->name);
91 for (j = 0; j < meta->num_args; ++j)
92 {
93 POCL_MEM_FREE (meta->arg_info[j].name);
94 POCL_MEM_FREE (meta->arg_info[j].type_name);
95 }
96 POCL_MEM_FREE (meta->arg_info);
97 if (meta->data != NULL)
98 for (j = 0; j < program->num_devices; ++j)
99 if (meta->data[j] != NULL)
100 meta->data[j] = NULL; // TODO free data in driver callback
101 POCL_MEM_FREE (meta->data);
102 POCL_MEM_FREE (meta->local_sizes);
103 POCL_MEM_FREE (meta->build_hash);
104 }
105 POCL_MEM_FREE (program->kernel_meta);
106 }
107
108 POCL_MEM_FREE(program->build_hash);
109 POCL_MEM_FREE(program->compiler_options);
110
111#ifdef OCS_AVAILABLE
112 if (program->llvm_irs)
113 for (i = 0; i < program->num_devices; ++i)
114 pocl_free_llvm_irs (program, i);
115#endif
116
117 POCL_MEM_FREE(program->llvm_irs);
118 POCL_DESTROY_OBJECT (program);
119 POCL_MEM_FREE(program);
120
121 POname(clReleaseContext)(context);
122 }
123
124 return CL_SUCCESS;
125}
