26CL_API_ENTRY cl_int CL_API_CALL
27POname(clReleaseDevice)(cl_device_id device) CL_API_SUFFIX__VERSION_1_2
28{
29 POCL_RETURN_ERROR_COND ((device == NULL), CL_INVALID_DEVICE);
30
31 if (device->parent_device == NULL)
32 return CL_SUCCESS;
33
34 int new_refcount;
35 POCL_RELEASE_OBJECT (device, new_refcount);
36
37 if (new_refcount == 0)
38 {
39 POCL_DESTROY_OBJECT (device);
40 POCL_MEM_FREE (device->partition_type);
41 POCL_MSG_PRINT_REFCOUNTS ("Free Device %p\n", device);
42 POCL_MEM_FREE (device);
43 }
44 else
45 POCL_MSG_PRINT_REFCOUNTS ("Release Device %p : %u\n", device,
46 device->pocl_refcount);
47
48 return CL_SUCCESS;
49}
