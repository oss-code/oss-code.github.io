24#include "pocl_util.h"
25#include <string.h>
26
27extern CL_API_ENTRY cl_int CL_API_CALL
28POname(clEnqueueFillBuffer)(cl_command_queue command_queue,
29 cl_mem buffer,
30 const void * pattern,
31 size_t pattern_size,
32 size_t offset,
33 size_t size,
34 cl_uint num_events_in_wait_list,
35 const cl_event* event_wait_list,
36 cl_event* event)
37CL_API_SUFFIX__VERSION_1_2
38{
39 int errcode = CL_SUCCESS;
40 _cl_command_node *cmd = NULL;
41
42 POCL_RETURN_ERROR_COND((command_queue == NULL), CL_INVALID_COMMAND_QUEUE);
43
44 POCL_RETURN_ERROR_COND((buffer == NULL), CL_INVALID_MEM_OBJECT);
45
46 POCL_RETURN_ERROR_ON((buffer->type != CL_MEM_OBJECT_BUFFER), CL_INVALID_MEM_OBJECT,
47 "buffer is not a CL_MEM_OBJECT_BUFFER\n");
48
49 POCL_RETURN_ERROR_ON((command_queue->context != buffer->context), CL_INVALID_CONTEXT,
50 "buffer and command_queue are not from the same context\n");
51
52 errcode = pocl_check_event_wait_list (command_queue, num_events_in_wait_list,
53 event_wait_list);
54 if (errcode != CL_SUCCESS)
55 return errcode;
56
57 errcode = pocl_buffer_boundcheck(buffer, offset, size);
58 if (errcode != CL_SUCCESS)
59 return errcode;
60
61 /* CL_INVALID_VALUE if pattern is NULL or if pattern_size is 0
62 * or if pattern_size is not one of {1, 2, 4, 8, 16, 32, 64, 128}. */
63 POCL_RETURN_ERROR_COND((pattern == NULL), CL_INVALID_VALUE);
64 POCL_RETURN_ERROR_COND((pattern_size == 0), CL_INVALID_VALUE);
65 POCL_RETURN_ERROR_COND((pattern_size > 128), CL_INVALID_VALUE);
66
67 POCL_RETURN_ERROR_ON((__builtin_popcount(pattern_size) > 1), CL_INVALID_VALUE,
68 "pattern_size(%zu) must be a power-of-two value", pattern_size);
69
70 /* CL_INVALID_VALUE if offset and size are not a multiple of pattern_size. */
71 POCL_RETURN_ERROR_ON((offset % pattern_size), CL_INVALID_VALUE,
72 "offset(%zu) must be a multiple of pattern_size(%zu)\n",
73 offset, pattern_size);
74 POCL_RETURN_ERROR_ON((size % pattern_size), CL_INVALID_VALUE,
75 "size(%zu) must be a multiple of pattern_size(%zu)\n",
76 size, pattern_size);
77
78 POCL_RETURN_ON_SUB_MISALIGN (buffer, command_queue);
79
80 POCL_CONVERT_SUBBUFFER_OFFSET (buffer, offset);
81
82 POCL_RETURN_ERROR_ON((buffer->size > command_queue->device->max_mem_alloc_size),
83 CL_OUT_OF_RESOURCES,
84 "buffer is larger than device's MAX_MEM_ALLOC_SIZE\n");
85
86 errcode = pocl_create_command (&cmd, command_queue, CL_COMMAND_FILL_BUFFER,
87 event, num_events_in_wait_list,
88 event_wait_list, 1, &buffer);
89 if (errcode != CL_SUCCESS)
90 return errcode;
91
92 cmd->command.memfill.dst_mem_id
93 = &buffer->device_ptrs[command_queue->device->dev_id];
94 cmd->command.memfill.size = size;
95 cmd->command.memfill.offset = offset;
96 void *p = pocl_aligned_malloc(pattern_size, pattern_size);
97 memcpy(p, pattern, pattern_size);
98 cmd->command.memfill.pattern = p;
99 cmd->command.memfill.pattern_size = pattern_size;
100
101 POname(clRetainMemObject) (buffer);
102 buffer->owning_device = command_queue->device;
103
104 pocl_command_enqueue(command_queue, cmd);
105
106 return CL_SUCCESS;
107
108}
