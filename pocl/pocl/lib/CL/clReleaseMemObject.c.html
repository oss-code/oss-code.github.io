24#include "utlist.h"
25#include "pocl_cl.h"
26
27CL_API_ENTRY cl_int CL_API_CALL
28POname(clReleaseMemObject)(cl_mem memobj) CL_API_SUFFIX__VERSION_1_0
29{
30 int new_refcount;
31 cl_device_id dev;
32 cl_mem parent = NULL;
33 unsigned i;
34 mem_mapping_t *mapping, *temp;
35 mem_destructor_callback_t *callback, *next_callback;
36
37 POCL_RETURN_ERROR_COND((memobj == NULL), CL_INVALID_MEM_OBJECT);
38
39 cl_context context = memobj->context;
40
41 POCL_RELEASE_OBJECT(memobj, new_refcount);
42
43 POCL_MSG_PRINT_REFCOUNTS ("Release mem obj %p %d\n", memobj, new_refcount);
44
45 /* OpenCL 1.2 Page 118:
46
47 After the memobj reference count becomes zero and commands queued for execution on
48 a command-queue(s) that use memobj have finished, the memory object is deleted. If
49 memobj is a buffer object, memobj cannot be deleted until all sub-buffer objects associated
50 with memobj are deleted.
51 */
52
53 if (new_refcount == 0)
54 {
55 if (memobj->is_image && (memobj->type == CL_MEM_OBJECT_IMAGE1D_BUFFER))
56 {
57 cl_mem b = memobj->buffer;
58 assert (b);
59 cl_int err = POname (clReleaseMemObject) (b);
60 POCL_MEM_FREE (memobj);
61 return err;
62 }
63
64 POCL_MSG_PRINT_REFCOUNTS ("Free mem obj %p\n", memobj);
65 if (memobj->parent == NULL)
66 {
67 cl_device_id shared_mem_owner_dev =
68 memobj->shared_mem_allocation_owner;
69
70 for (i = 0; i < context->num_devices; ++i)
71 {
72 /* owner is called last */
73 if (shared_mem_owner_dev == context->devices[i])
74 continue;
75 dev = context->devices[i];
76 if (memobj->is_image && dev->image_support
77 && dev->ops->free_image)
78 dev->ops->free_image (
79 dev, memobj, memobj->device_ptrs[dev->dev_id].image_data);
80 dev->ops->free (dev, memobj);
81 memobj->device_ptrs[dev->dev_id].mem_ptr = NULL;
82 }
83 if (shared_mem_owner_dev)
84 shared_mem_owner_dev->ops->free (shared_mem_owner_dev, memobj);
85
86 }
87 DL_FOREACH_SAFE(memobj->mappings, mapping, temp)
88 {
89 POCL_MEM_FREE(mapping);
90 }
91 memobj->mappings = NULL;
92
93 parent = memobj->parent;
94
95 /* Free host mem allocated by the runtime (not for sub buffers) */
96 if (memobj->parent == NULL && (memobj->flags & CL_MEM_ALLOC_HOST_PTR)
97 && memobj->mem_host_ptr != NULL)
98 {
99 POCL_MEM_FREE(memobj->mem_host_ptr);
100 }
101 POCL_MEM_FREE(memobj->device_ptrs);
102
103 /* Fire any registered destructor callbacks */
104 callback = memobj->destructor_callbacks;
105 while (callback)
106 {
107 callback->pfn_notify (memobj, callback->user_data);
108 next_callback = callback->next;
109 free (callback);
110 callback = next_callback;
111 }
112
113 if (memobj->is_image)
114 POCL_MEM_FREE (memobj->device_supports_this_image);
115
116 POCL_DESTROY_OBJECT (memobj);
117 POCL_MEM_FREE(memobj);
118
119 if (parent)
120 POname(clReleaseMemObject)(parent);
121 POname(clReleaseContext)(context);
122 }
123 return CL_SUCCESS;
124}
