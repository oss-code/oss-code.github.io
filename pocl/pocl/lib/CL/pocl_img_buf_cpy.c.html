30/* Copies between images and rectangular buffer copies share most of the code,
31 with specializations only needed for specific checks. The actual API calls
32 thus defer to this function, with the additional information of which of src
33 and/or dst is an image and which is a buffer.
34 */
35
36cl_int pocl_rect_copy(cl_command_queue command_queue,
37 cl_command_type command_type,
38 cl_mem src,
39 cl_int src_is_image,
40 cl_mem dst,
41 cl_int dst_is_image,
42 const size_t *src_origin,
43 const size_t *dst_origin,
44 const size_t *region,
45 size_t src_row_pitch,
46 size_t src_slice_pitch,
47 size_t dst_row_pitch,
48 size_t dst_slice_pitch,
49 cl_uint num_events_in_wait_list,
50 const cl_event *event_wait_list,
51 cl_event *event,
52 _cl_command_node **cmd)
53{
54 cl_int errcode;
55 cl_device_id device;
56 unsigned i;
57 cl_mem buffers[2] = {src, dst};
58 _cl_command_node *c;
59
60
61 POCL_RETURN_ERROR_ON (
62 ((command_queue->context != src->context)
63 || (command_queue->context != dst->context)),
64 CL_INVALID_CONTEXT,
65 "src, dst and command_queue are not from the same context\n");
66
67 POCL_RETURN_ERROR_COND (
68 (event_wait_list == NULL && num_events_in_wait_list > 0),
69 CL_INVALID_EVENT_WAIT_LIST);
70
71 POCL_RETURN_ERROR_COND (
72 (event_wait_list != NULL && num_events_in_wait_list == 0),
73 CL_INVALID_EVENT_WAIT_LIST);
74
75 POCL_RETURN_ERROR_COND((src == NULL), CL_INVALID_MEM_OBJECT);
76 POCL_RETURN_ERROR_COND((dst == NULL), CL_INVALID_MEM_OBJECT);
77 POCL_RETURN_ERROR_COND((src_origin == NULL), CL_INVALID_VALUE);
78 POCL_RETURN_ERROR_COND((dst_origin == NULL), CL_INVALID_VALUE);
79 POCL_RETURN_ERROR_COND((region == NULL), CL_INVALID_VALUE);
80
81 if (src_is_image)
82 {
83 POCL_RETURN_ERROR_ON ((!src->is_image), CL_INVALID_MEM_OBJECT,
84 "src is not an image\n");
85 POCL_RETURN_ON_UNSUPPORTED_IMAGE (src, command_queue->device);
86 POCL_RETURN_ERROR_ON((src->type == CL_MEM_OBJECT_IMAGE2D && src_origin[2] != 0),
87 CL_INVALID_VALUE, "src_origin[2] must be 0 for 2D src_image\n");
88 }
89 else
90 {
91 POCL_RETURN_ERROR_ON((src->type != CL_MEM_OBJECT_BUFFER),
92 CL_INVALID_MEM_OBJECT, "src is not a CL_MEM_OBJECT_BUFFER\n");
93 POCL_RETURN_ON_SUB_MISALIGN (src, command_queue);
94 POCL_RETURN_ERROR_ON((src->size > command_queue->device->max_mem_alloc_size),
95 CL_OUT_OF_RESOURCES,
96 "src is larger than device's MAX_MEM_ALLOC_SIZE\n");
97 }
98
99 if (dst_is_image)
100 {
101 POCL_RETURN_ERROR_ON ((!dst->is_image), CL_INVALID_MEM_OBJECT,
102 "dst is not an image\n");
103 POCL_RETURN_ON_UNSUPPORTED_IMAGE (dst, command_queue->device);
104 POCL_RETURN_ERROR_ON((dst->type == CL_MEM_OBJECT_IMAGE2D && dst_origin[2] != 0),
105 CL_INVALID_VALUE, "dst_origin[2] must be 0 for 2D dst_image\n");
106 }
107 else
108 {
109 POCL_RETURN_ERROR_ON((dst->type != CL_MEM_OBJECT_BUFFER),
110 CL_INVALID_MEM_OBJECT, "dst is not a CL_MEM_OBJECT_BUFFER\n");
111 POCL_RETURN_ON_SUB_MISALIGN (dst, command_queue);
112 POCL_RETURN_ERROR_ON((dst->size > command_queue->device->max_mem_alloc_size),
113 CL_OUT_OF_RESOURCES,
114 "dst is larger than device's MAX_MEM_ALLOC_SIZE\n");
115 }
116
117 if (src_is_image && dst_is_image)
118 {
119 POCL_RETURN_ERROR_ON((src->image_channel_order != dst->image_channel_order),
120 CL_IMAGE_FORMAT_MISMATCH, "src and dst have different image channel order\n");
121 POCL_RETURN_ERROR_ON((src->image_channel_data_type != dst->image_channel_data_type),
122 CL_IMAGE_FORMAT_MISMATCH, "src and dst have different image channel data type\n");
123 POCL_RETURN_ERROR_ON((
124 (dst->type == CL_MEM_OBJECT_IMAGE2D || src->type == CL_MEM_OBJECT_IMAGE2D) &&
125 region[2] != 1),
126 CL_INVALID_VALUE, "for any 2D image copy, region[2] must be 1\n");
127 }
128
129 /* Images need to recompute the regions in bytes for checking */
130 size_t mod_region[3], mod_src_origin[3], mod_dst_origin[3];
131 memcpy(mod_region, region, 3*sizeof(size_t));
132 memcpy(mod_src_origin, src_origin, 3*sizeof(size_t));
133 memcpy(mod_dst_origin, dst_origin, 3*sizeof(size_t));
134
135 /* NOTE: 1D image array has row_pitch == slice_pitch;
136 * need to zero it for bufferbound checks.
137 */
138 if (src_is_image)
139 {
140 mod_region[0] *= src->image_elem_size * src->image_channels;
141 mod_src_origin[0] *= src->image_elem_size * src->image_channels;
142 src_row_pitch = src->image_row_pitch;
143 if (src->type == CL_MEM_OBJECT_IMAGE1D_ARRAY)
144 src_slice_pitch = 0;
145 else
146 src_slice_pitch = src->image_slice_pitch;
147 }
148 if (dst_is_image)
149 {
150 if (!src_is_image)
151 mod_region[0] *= dst->image_elem_size * dst->image_channels;
152 mod_dst_origin[0] *= dst->image_elem_size * dst->image_channels;
153 dst_row_pitch = dst->image_row_pitch;
154 if (dst->type == CL_MEM_OBJECT_IMAGE1D_ARRAY)
155 dst_slice_pitch = 0;
156 else
157 dst_slice_pitch = dst->image_slice_pitch;
158 }
159
160
161 if (pocl_buffer_boundcheck_3d(src->size, mod_src_origin, mod_region,
162 &src_row_pitch, &src_slice_pitch, "src_") != CL_SUCCESS)
163 return CL_INVALID_VALUE;
164
165 if (pocl_buffer_boundcheck_3d(dst->size, mod_dst_origin, mod_region,
166 &dst_row_pitch, &dst_slice_pitch, "dst_") != CL_SUCCESS)
167 return CL_INVALID_VALUE;
168
169 if (src == dst)
170 {
171 POCL_RETURN_ERROR_ON((src_slice_pitch != dst_slice_pitch),
172 CL_INVALID_VALUE, "src and dst are the same object,"
173 " but the given dst & src slice pitch differ\n");
174
175 POCL_RETURN_ERROR_ON((src_row_pitch != dst_row_pitch),
176 CL_INVALID_VALUE, "src and dst are the same object,"
177 " but the given dst & src row pitch differ\n");
178 // TODO
179 POCL_RETURN_ERROR_ON(
180 (check_copy_overlap(mod_src_origin, mod_dst_origin, mod_region,
181 src_row_pitch, src_slice_pitch)),
182 CL_MEM_COPY_OVERLAP, "src and dst are the same object,"
183 "and source and destination regions overlap\n");
184
185 }
186
187 POCL_CHECK_DEV_IN_CMDQ;
188
189 errcode = pocl_create_command (cmd, command_queue, command_type,
190 event, num_events_in_wait_list,
191 event_wait_list, 2, buffers);
192
193 /* for CL_COMMAND_COPY_BUFFER_RECT, we also need to setup
194 * the output row/slice pitch parameters. */
195 if (errcode == CL_SUCCESS && command_type == CL_COMMAND_COPY_BUFFER_RECT)
196 {
197 c = *cmd;
198 c->command.copy_rect.src_row_pitch = src_row_pitch;
199 c->command.copy_rect.src_slice_pitch = src_slice_pitch;
200 c->command.copy_rect.dst_row_pitch = dst_row_pitch;
201 c->command.copy_rect.dst_slice_pitch = dst_slice_pitch;
202 }
203
204 return errcode;
205}
