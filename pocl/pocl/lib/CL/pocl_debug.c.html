32 void
33 pocl_debug_messages_setup (const char* debug)
34 {
35 pocl_debug_messages_filter = 0;
36 if (strlen (debug) == 1)
37 {
38 if (debug[0] == '1')
39 pocl_debug_messages_filter = POCL_DEBUG_FLAG_GENERAL
40 | POCL_DEBUG_FLAG_WARNING
41 | POCL_DEBUG_FLAG_ERROR;
42 return;
43 }
44 /* else parse */
45 char* tokenize = strdup (debug);
46 char* ptr = NULL;
47 ptr = strtok (tokenize, ",");
48
49 while (ptr != NULL)
50 {
51 if (strncmp (ptr, "general", 7) == 0)
52 pocl_debug_messages_filter |= POCL_DEBUG_FLAG_GENERAL;
53 else if (strncmp (ptr, "event", 5) == 0)
54 pocl_debug_messages_filter |= POCL_DEBUG_FLAG_EVENTS;
55 else if (strncmp (ptr, "cache", 5) == 0)
56 pocl_debug_messages_filter |= POCL_DEBUG_FLAG_CACHE;
57 else if (strncmp (ptr, "llvm", 4) == 0)
58 pocl_debug_messages_filter |= POCL_DEBUG_FLAG_LLVM;
59 else if (strncmp (ptr, "refc", 4) == 0)
60 pocl_debug_messages_filter |= POCL_DEBUG_FLAG_REFCOUNTS;
61 else if (strncmp (ptr, "lock", 4) == 0)
62 pocl_debug_messages_filter |= POCL_DEBUG_FLAG_LOCKING;
63 else if (strncmp (ptr, "cuda", 4) == 0)
64 pocl_debug_messages_filter |= POCL_DEBUG_FLAG_CUDA;
65 else if (strncmp (ptr, "warn", 4) == 0)
66 pocl_debug_messages_filter |= (POCL_DEBUG_FLAG_WARNING | POCL_DEBUG_FLAG_ERROR);
67 else if (strncmp (ptr, "hsa", 3) == 0)
68 pocl_debug_messages_filter |= POCL_DEBUG_FLAG_HSA;
69 else if (strncmp (ptr, "tce", 3) == 0)
70 pocl_debug_messages_filter |= POCL_DEBUG_FLAG_TCE;
71 else if (strncmp (ptr, "mem", 3) == 0)
72 pocl_debug_messages_filter |= POCL_DEBUG_FLAG_MEMORY;
73 else if (strncmp (ptr, "tim", 3) == 0)
74 pocl_debug_messages_filter |= POCL_DEBUG_FLAG_TIMING;
75 else if (strncmp (ptr, "all", 3) == 0)
76 pocl_debug_messages_filter |= POCL_DEBUG_FLAG_ALL;
77 else if (strncmp (ptr, "err", 3) == 0)
78 pocl_debug_messages_filter |= POCL_DEBUG_FLAG_ERROR;
79 else
80 POCL_MSG_WARN ("Unknown token in POCL_DEBUG env var: %s", ptr);
81
82 ptr = strtok (NULL,",");
83 }
84
85 free (tokenize);
86 if (pocl_debug_messages_filter)
87 fprintf (stderr, "** Final POCL_DEBUG flags: %" PRIX64 " \n",
88 pocl_debug_messages_filter);
89 }

91 void
92 pocl_debug_print_header (const char* func, unsigned line,
93 const char *filter, int filter_type)
94 {
95
96 int year, mon, day, hour, min, sec, nanosec;
97 pocl_gettimereal(&year, &mon, &day, &hour, &min, &sec, &nanosec);
98
99 const char *filter_type_str;
100 const char *formatstring;
101
102 if (filter_type == POCL_FILTER_TYPE_ERR)
103 filter_type_str = (stderr_is_a_tty ? POCL_COLOR_RED : " *** ERROR *** ");
104 else if (filter_type == POCL_FILTER_TYPE_WARN)
105 filter_type_str = (stderr_is_a_tty ? POCL_COLOR_YELLOW : " *** WARNING *** ");
106 else if (filter_type == POCL_FILTER_TYPE_INFO)
107 filter_type_str = (stderr_is_a_tty ? POCL_COLOR_GREEN : " *** INFO *** ");
108 else
109 filter_type_str = (stderr_is_a_tty ? POCL_COLOR_GREEN : " *** UNKNOWN *** ");
110
111 if (stderr_is_a_tty)
112 formatstring = POCL_COLOR_BLUE
113 "[%04i-%02i-%02i %02i:%02i:%02i.%09li]"
114 POCL_COLOR_RESET "POCL: in fn %s "
115 POCL_COLOR_RESET "at line %u:\n %s | %9s | ";
116 else
117 formatstring = "[%04i-%02i-%02i %02i:%02i:%02i.%09i] "
118 "POCL: in fn %s at line %u:\n %s | %9s | ";
119
120 fprintf (stderr, formatstring, year, mon, day, hour, min, sec,
121 nanosec, func, line, filter_type_str, filter);
122 }

140 void pocl_debug_print_duration(const char* func, unsigned line,
141 const char* msg, uint64_t nanosecs)
142 {
143 if (!(pocl_debug_messages_filter & POCL_DEBUG_FLAG_TIMING))
144 return;
145 const char* formatstring;
146 if (stderr_is_a_tty)
147 formatstring = " >>> " POCL_COLOR_MAGENTA " %3" PRIu64
148 ".%03" PRIu64 " " POCL_COLOR_RESET " %s %s\n";
149 else
150 formatstring = " >>> %3" PRIu64 ".%03"
151 PRIu64 " %s %s\n";
152
153 uint64_t nsec = nanosecs % 1000000000;
154 uint64_t sec = nanosecs / 1000000000;
155 uint64_t a, b;
156
157 if ((sec == 0) && (nsec < 1000))
158 {
159 b = nsec % 1000;
160 if (stderr_is_a_tty)
161 formatstring = " >>> " POCL_COLOR_MAGENTA
162 " %3" PRIu64 " " POCL_COLOR_RESET " ns %s\n";
163 else
164 formatstring = " >>> %3" PRIu64 " ns %s\n";
165 PRINT_DURATION (func, line, formatstring, b, msg);
166 }
167 else if ((sec == 0) && (nsec < 1000000))
168 {
169 a = nsec / 1000;
170 b = nsec % 1000;
171 PRINT_DURATION (func, line, formatstring, a, b, "us", msg);
172 }
173 else if (sec == 0)
174 {
175 a = nsec / 1000000;
176 b = (nsec % 1000000) / 1000;
177 PRINT_DURATION (func, line, formatstring, a, b, "ms", msg);
178 }
179 else
180 {
181 if (stderr_is_a_tty)
182 formatstring = " >>> " POCL_COLOR_MAGENTA " %3" PRIu64
183 ".%09" PRIu64 " " POCL_COLOR_RESET " %s %s\n";
184 else
185 formatstring = " >>> %3" PRIu64 ".%09"
186 PRIu64 " %s %s\n";
187
188 PRINT_DURATION (func, line, formatstring, sec, nsec, "s", msg);
189 }
190
191 }
