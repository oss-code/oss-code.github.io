24#include "pocl_util.h"
25#include "devices.h"
26
27CL_API_ENTRY void* CL_API_CALL
28POname(clSVMAlloc)(cl_context context,
29 cl_svm_mem_flags flags,
30 size_t size,
31 unsigned int alignment) CL_API_SUFFIX__VERSION_2_0
32{
33 unsigned i;
34 int p;
35
36 POCL_MSG_PRINT_INFO("clSVMAlloc\n");
37
38 POCL_RETURN_ERROR_COND((context == NULL), NULL);
39
40 POCL_RETURN_ERROR_ON((!context->svm_allocdev), NULL,
41 "None of the devices in this context is SVM-capable\n");
42
43 POCL_RETURN_ERROR_COND((size == 0), NULL);
44
45 POCL_RETURN_ERROR_ON ((size > context->max_mem_alloc_size), NULL,
46 "size(%zu) > CL_DEVICE_MAX_MEM_ALLOC_SIZE value "
47 "for some device in context\n",
48 size);
49
50 /* flags does not contain CL_MEM_SVM_FINE_GRAIN_BUFFER
51 * but does contain CL_MEM_SVM_ATOMICS. */
52 POCL_RETURN_ERROR_COND((flags & CL_MEM_SVM_ATOMICS) &&
53 ((flags & CL_MEM_SVM_FINE_GRAIN_BUFFER) == 0), NULL);
54
55 /* Flags */
56 p = __builtin_popcount(flags & (CL_MEM_READ_WRITE
57 | CL_MEM_WRITE_ONLY | CL_MEM_READ_ONLY));
58 POCL_RETURN_ERROR_ON((p > 1), NULL, "flags may contain only one of "
59 "CL_MEM_READ_WRITE | CL_MEM_WRITE_ONLY | CL_MEM_READ_ONLY\n");
60
61 const cl_svm_mem_flags valid_flags = (CL_MEM_SVM_ATOMICS | CL_MEM_SVM_FINE_GRAIN_BUFFER
62 | CL_MEM_READ_WRITE | CL_MEM_WRITE_ONLY
63 | CL_MEM_READ_ONLY);
64 POCL_RETURN_ERROR_ON((flags & (!valid_flags)), NULL, "flags argument "
65 "contains invalid bits (nonexistent flags)\n");
66
67 /* CL_MEM_SVM_FINE_GRAIN_BUFFER or CL_MEM_SVM_ATOMICS is specified in flags
68 * and these are not supported by at least one device in context. */
69 if (flags & CL_MEM_SVM_FINE_GRAIN_BUFFER)
70 for (i=0; i < context->num_devices; i++)
71 POCL_RETURN_ERROR_ON((DEVICE_SVM_FINEGR(context->devices[i]) == 0), NULL,
72 "One of the devices in the context doesn't support "
73 "fine-grained buffers, and it's in flags\n");
74
75 if (flags & CL_MEM_SVM_ATOMICS)
76 for (i=0; i < context->num_devices; i++)
77 POCL_RETURN_ERROR_ON((DEVICE_SVM_ATOM(context->devices[i]) == 0), NULL,
78 "One of the devices in the context doesn't support "
79 "SVM atomics buffers, and it's in flags\n");
80
81#define dev context->svm_allocdev
82
83 if (alignment == 0)
84 alignment = dev->min_data_type_align_size;
85
86 /* alignment is not a power of two or the OpenCL implementation cannot support
87 * the specified alignment for at least one device in context. */
88 p = __builtin_popcount(alignment);
89 POCL_RETURN_ERROR_ON((p > 1), NULL, "aligment argument must be a power of 2\n");
90
91 for (i=0; i < context->num_devices; i++)
92 POCL_RETURN_ERROR_ON((context->devices[i]->min_data_type_align_size < alignment),
93 NULL, "All devices must support the requested memory "
94 "aligment (%u) \n", alignment);
95
96 void *ptr = dev->ops->svm_alloc (dev, flags, size);
97 return ptr;
98}
