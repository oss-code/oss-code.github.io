24#include "pocl_cl.h"
25
26CL_API_ENTRY cl_int CL_API_CALL
27POname(clWaitForEvents)(cl_uint num_events ,
28 const cl_event * event_list ) CL_API_SUFFIX__VERSION_1_0
29{
30 unsigned event_i;
31 cl_device_id dev;
32 cl_int ret = CL_SUCCESS;
33 POCL_RETURN_ERROR_COND((num_events == 0 || event_list == NULL), CL_INVALID_VALUE);
34
35 for (event_i = 0; event_i < num_events; ++event_i)
36 {
37 POCL_RETURN_ERROR_COND((event_list[event_i] == NULL), CL_INVALID_EVENT);
38 if (event_i > 0)
39 {
40 POCL_RETURN_ERROR_COND((event_list[event_i]->context != event_list[event_i - 1]->context), CL_INVALID_CONTEXT);
41 }
42 }
43
44 // dummy implementation, waits until *all* events have completed.
45 for (event_i = 0; event_i < num_events; ++event_i)
46 {
47 /* lets handle user events later */
48 if (event_list[event_i]->command_type == CL_COMMAND_USER)
49 continue;
50 dev = event_list[event_i]->queue->device;
51 if (dev->ops->wait_event)
52 dev->ops->wait_event(dev, event_list[event_i]);
53 else
54 POname(clFinish)(event_list[event_i]->queue);
55 if (event_list[event_i]->status < 0)
56 ret = CL_EXEC_STATUS_ERROR_FOR_EVENTS_IN_WAIT_LIST;
57 }
58
59 if (ret != CL_SUCCESS)
60 return ret;
61
62 /* wait for user events */
63 struct timespec time_to_wait = { 0, 0 };
64
65 for (event_i = 0; event_i < num_events; ++event_i)
66 {
67 cl_event e = event_list[event_i];
68 POCL_LOCK_OBJ (e);
69 pocl_user_event_data *p = e->data;
70 if (e->command_type == CL_COMMAND_USER)
71 {
72 while (e->status > CL_COMPLETE)
73 {
74 time_to_wait.tv_sec = time (NULL) + 1;
75 pthread_cond_timedwait (&p->wakeup_cond, &e->pocl_lock,
76 &time_to_wait);
77 }
78 if (e->status < 0)
79 ret = CL_EXEC_STATUS_ERROR_FOR_EVENTS_IN_WAIT_LIST;
80 }
81 POCL_UNLOCK_OBJ (e);
82 }
83
84 return ret;
85}
