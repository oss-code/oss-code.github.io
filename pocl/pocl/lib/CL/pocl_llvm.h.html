64/* This function links the input kernel LLVM bitcode and the OpenCL kernel
65 * runtime library into one LLVM module, then runs pocl's kernel compiler
66 * passes on that module to produce a function that executes all work-items in
67 * a work-group.
68 *
69 * Output is a LLVM bitcode file that contains a work-group function and its
70 * associated launchers. If @param Specialize is set to true, generates a
71 * WG function that might be specialized according to the properties of
72 * the given Command.
73 */
74int pocl_llvm_generate_workgroup_function (unsigned DeviceI,
75 cl_device_id Device,
76 cl_kernel Kernel,
77 _cl_command_node *Command,
78 int Specialize);
79
80int pocl_llvm_generate_workgroup_function_nowrite (
81 unsigned DeviceI, cl_device_id Device, cl_kernel Kernel,
82 _cl_command_node *Command, void **output, int Specialize);
83/**
84 * Free the LLVM IR of a program for a given device
85 */
86void pocl_free_llvm_irs(cl_program program, unsigned device_i);
87
88/* calls delete on the module. */
89void pocl_destroy_llvm_module(void *modp);
90
91int pocl_llvm_remove_file_on_signal (const char *file);
92
93void pocl_llvm_release();
94/**
95 * Update the program->binaries[] representation of the kernels
96 * from the program->llvm_irs[] representation.
97 * Also updates the 'program.bc' file in the POCL_TEMP_DIR cache.
98 */
99void pocl_llvm_update_binaries (cl_program program);

101/**
102 * Count the number of "__kernel" functions in 'program'.
