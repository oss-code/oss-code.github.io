25#include "pocl_cl.h"
26#include "pocl_util.h"
27#include <string.h>
28
29CL_API_ENTRY cl_program CL_API_CALL
30POname(clCreateProgramWithSource)(cl_context context,
31 cl_uint count,
32 const char **strings,
33 const size_t *lengths,
34 cl_int *errcode_ret) CL_API_SUFFIX__VERSION_1_0
35{
36 cl_program program = NULL;
37 size_t size = 0;
38 char *source = NULL;
39 unsigned i;
40 int errcode;
41
42 POCL_GOTO_ERROR_COND ((context == NULL), CL_INVALID_CONTEXT);
43
44 POCL_GOTO_ERROR_COND((count == 0), CL_INVALID_VALUE);
45
46 program = (cl_program) calloc(1, sizeof(struct _cl_program));
47 if (program == NULL)
48 {
49 errcode = CL_OUT_OF_HOST_MEMORY;
50 goto ERROR;
51 }
52
53 POCL_INIT_OBJECT(program);
54
55 for (i = 0; i < count; ++i)
56 {
57 POCL_GOTO_ERROR_ON((strings[i] == NULL), CL_INVALID_VALUE,
58 "strings[%i] is NULL\n", i);
59
60 if (lengths == NULL)
61 size += strlen(strings[i]);
62 else if (lengths[i] == 0)
63 size += strlen(strings[i]);
64 else
65 size += lengths[i];
66 }
67
68 source = (char *) malloc(size + 1);
69 if (source == NULL)
70 {
71 errcode = CL_OUT_OF_HOST_MEMORY;
72 goto ERROR;
73 }
74
75 program->source = source;
76
77 for (i = 0; i < count; ++i)
78 {
79 if (lengths == NULL)
80 {
81 memcpy(source, strings[i], strlen(strings[i]));
82 source += strlen(strings[i]);
83 }
84 else if (lengths[i] == 0)
85 {
86 memcpy(source, strings[i], strlen(strings[i]));
87 source += strlen(strings[i]);
88 }
89 else
90 {
91 memcpy(source, strings[i], lengths[i]);
92 source += lengths[i];
93 }
94 }
95
96 *source = '\0';
97
98 program->context = context;
99 program->num_devices = context->num_devices;
100 program->devices = context->devices;
101 program->build_status = CL_BUILD_NONE;
102 program->binary_type = CL_PROGRAM_BINARY_TYPE_NONE;
103
104 if ((program->binary_sizes =
105 (size_t*) calloc (program->num_devices, sizeof(size_t))) == NULL ||
106 (program->binaries = (unsigned char**)
107 calloc (program->num_devices, sizeof(unsigned char*))) == NULL ||
108 (program->pocl_binaries = (unsigned char**)
109 calloc (program->num_devices, sizeof(unsigned char*))) == NULL ||
110 (program->pocl_binary_sizes =
111 (size_t*) calloc (program->num_devices, sizeof(size_t))) == NULL ||
112 (program->build_log = (char**)
113 calloc (program->num_devices, sizeof(char*))) == NULL ||
114 ((program->llvm_irs =
115 (void**) calloc (program->num_devices, sizeof(void*))) == NULL) ||
116 ((program->build_hash = (SHA1_digest_t*)
117 calloc (program->num_devices, sizeof(SHA1_digest_t))) == NULL))
118 {
119 errcode = CL_OUT_OF_HOST_MEMORY;
120 goto ERROR;
121 }
122
123 POCL_RETAIN_OBJECT(context);
124
125 if (errcode_ret != NULL)
126 *errcode_ret = CL_SUCCESS;
127 return program;
128
129ERROR:
130 if (program) {
131 POCL_MEM_FREE(program->build_hash);
132 POCL_MEM_FREE(program->llvm_irs);
133 POCL_MEM_FREE(program->build_log);
134 POCL_MEM_FREE(program->binaries);
135 POCL_MEM_FREE(program->binary_sizes);
136 POCL_MEM_FREE(program->pocl_binaries);
137 POCL_MEM_FREE(program->pocl_binary_sizes);
138 POCL_MEM_FREE(program->source);
139 }
140 POCL_MEM_FREE(program);
141 if(errcode_ret)
142 {
143 *errcode_ret = errcode;
144 }
145 return NULL;
146}
