25#include "pocl_util.h"
26#include <string.h>

28CL_API_ENTRY cl_int CL_API_CALL
29POname(clEnqueueMigrateMemObjects) (cl_command_queue command_queue,
30 cl_uint num_mem_objects,
31 const cl_mem *mem_objects,
32 cl_mem_migration_flags flags,
33 cl_uint num_events_in_wait_list,
34 const cl_event *event_wait_list,
35 cl_event *event)CL_API_SUFFIX__VERSION_1_0
36{
37 unsigned i;
38 int errcode;
39 _cl_command_node *cmd = NULL;
40 cl_mem *new_mem_objects = NULL;
41
42 POCL_RETURN_ERROR_COND((command_queue == NULL), CL_INVALID_COMMAND_QUEUE);
43 POCL_RETURN_ERROR_COND((num_mem_objects == 0), CL_INVALID_VALUE);
44 POCL_RETURN_ERROR_COND((mem_objects == NULL), CL_INVALID_VALUE);
45
46 cl_mem_migration_flags invalid_flags =
47 ~(CL_MIGRATE_MEM_OBJECT_CONTENT_UNDEFINED | CL_MIGRATE_MEM_OBJECT_HOST);
48 POCL_RETURN_ERROR_COND (((flags != 0) && (flags & invalid_flags)),
49 CL_INVALID_VALUE);
50 /* TODO check if it's OK to ignore flags. */
51
52 errcode = pocl_check_event_wait_list (command_queue, num_events_in_wait_list,
53 event_wait_list);
54 if (errcode != CL_SUCCESS)
55 return errcode;
56
57 new_mem_objects = calloc (sizeof (cl_mem), num_mem_objects);
58
59 for (i = 0; i < num_mem_objects; ++i)
60 {
61 POCL_GOTO_ERROR_COND ((mem_objects[i] == NULL), CL_INVALID_MEM_OBJECT);
62
63 POCL_GOTO_ERROR_COND (
64 (mem_objects[i]->context != command_queue->context),
65 CL_INVALID_CONTEXT);
66
67 if (mem_objects[i]->parent == NULL)
68 new_mem_objects[i] = mem_objects[i];
69 else
70 new_mem_objects[i] = mem_objects[i]->parent;
71
72 IMAGE1D_TO_BUFFER (new_mem_objects[i]);
73 }
74
75 errcode = pocl_create_command (&cmd, command_queue,
76 CL_COMMAND_MIGRATE_MEM_OBJECTS, event,
77 num_events_in_wait_list, event_wait_list,
78 num_mem_objects, new_mem_objects);
79
80 if (errcode != CL_SUCCESS)
81 goto ERROR;
82
83 cmd->command.migrate.data = command_queue->device->data;
84 cmd->command.migrate.num_mem_objects = num_mem_objects;
85 cmd->command.migrate.mem_objects = new_mem_objects;
86 cmd->command.migrate.source_devices = malloc
87 (num_mem_objects * sizeof (cl_device_id));
88
89 for (i = 0; i < num_mem_objects; ++i)
90 {
91 POname (clRetainMemObject) (new_mem_objects[i]);
92
93 cmd->command.migrate.source_devices[i]
94 = new_mem_objects[i]->owning_device;
95
96 new_mem_objects[i]->owning_device = command_queue->device;
97 }
98
99 pocl_command_enqueue (command_queue, cmd);
100
101 return CL_SUCCESS;
102
103ERROR:
104 POCL_MEM_FREE (new_mem_objects);
105 free (cmd);
106 free (event);
107 return errcode;
108}
