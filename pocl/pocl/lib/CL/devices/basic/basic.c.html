180void
181pocl_init_cpu_device_infos (cl_device_id dev)
182{
183 size_t i;
184 dev->type = CL_DEVICE_TYPE_CPU;
185 dev->max_work_item_dimensions = 3;
186 dev->final_linkage_flags = final_ld_flags;
187
188 SETUP_DEVICE_CL_VERSION(HOST_DEVICE_CL_VERSION_MAJOR, HOST_DEVICE_CL_VERSION_MINOR)
189 /*
190 The hard restriction will be the context data which is
191 stored in stack that can be as small as 8K in Linux.
192 Thus, there should be enough work-items alive to fill up
193 the SIMD lanes times the vector units, but not more than
194 that to avoid stack overflow and cache trashing.
195 */
196 int max_wg
197 = pocl_get_int_option ("POCL_MAX_WORK_GROUP_SIZE", DEFAULT_WG_SIZE);
198 assert (max_wg > 0);
199 max_wg = min (max_wg, DEFAULT_WG_SIZE);
200 if (max_wg < 0)
201 max_wg = DEFAULT_WG_SIZE;
202
203 dev->max_work_item_sizes[0] = dev->max_work_item_sizes[1]
204 = dev->max_work_item_sizes[2] = dev->max_work_group_size = max_wg;
205
206 dev->preferred_wg_size_multiple = 8;
207#ifdef OCS_AVAILABLE
208 cpu_setup_vector_widths (dev);
209#else
210 dev->preferred_vector_width_char = POCL_DEVICES_PREFERRED_VECTOR_WIDTH_CHAR;
211 dev->preferred_vector_width_short = POCL_DEVICES_PREFERRED_VECTOR_WIDTH_SHORT;
212 dev->preferred_vector_width_int = POCL_DEVICES_PREFERRED_VECTOR_WIDTH_INT;
213 dev->preferred_vector_width_long = POCL_DEVICES_PREFERRED_VECTOR_WIDTH_LONG;
214 dev->preferred_vector_width_float = POCL_DEVICES_PREFERRED_VECTOR_WIDTH_FLOAT;
215 /* TODO: figure out what the difference between preferred and native widths are */
216 dev->native_vector_width_char = POCL_DEVICES_NATIVE_VECTOR_WIDTH_CHAR;
217 dev->native_vector_width_short = POCL_DEVICES_NATIVE_VECTOR_WIDTH_SHORT;
218 dev->native_vector_width_int = POCL_DEVICES_NATIVE_VECTOR_WIDTH_INT;
219 dev->native_vector_width_long = POCL_DEVICES_NATIVE_VECTOR_WIDTH_LONG;
220 dev->native_vector_width_float = POCL_DEVICES_NATIVE_VECTOR_WIDTH_FLOAT;
221
222#ifdef _CL_DISABLE_DOUBLE
223 dev->native_vector_width_double = 0;
224 dev->preferred_vector_width_double = 0;
225#else
226 dev->native_vector_width_double = POCL_DEVICES_NATIVE_VECTOR_WIDTH_DOUBLE;
227 dev->preferred_vector_width_double = POCL_DEVICES_PREFERRED_VECTOR_WIDTH_DOUBLE;
228#endif
229#ifdef _CL_DISABLE_HALF
230 dev->preferred_vector_width_half = 0;
231 dev->native_vector_width_half = 0;
232#else
233 dev->preferred_vector_width_half = POCL_DEVICES_PREFERRED_VECTOR_WIDTH_HALF;
234 dev->native_vector_width_half = POCL_DEVICES_NATIVE_VECTOR_WIDTH_HALF;
235#endif
236
237#endif
238
239 dev->grid_width_specialization_limit = USHRT_MAX;
240 dev->address_bits = HOST_DEVICE_ADDRESS_BITS;
241 dev->image_support = CL_TRUE;
242 /* Use the minimum values until we get a more sensible upper limit from
243 somewhere. */
244 dev->max_read_image_args = dev->max_write_image_args
245 = dev->max_read_write_image_args = 128;
246 dev->image2d_max_width = dev->image2d_max_height = 8192;
247 dev->image3d_max_width = dev->image3d_max_height = dev->image3d_max_depth = 2048;
248 dev->max_samplers = 16;
249
250 for (i = 0; i < NUM_OPENCL_IMAGE_TYPES; ++i)
251 {
252 dev->num_image_formats[i]
253 = sizeof (supported_image_formats) / sizeof (cl_image_format);
254 dev->image_formats[i] = supported_image_formats;
255 }
256
257 dev->image_max_buffer_size = 65536;
258 dev->image_max_array_size = 2048;
259 dev->max_constant_args = 8;
260 dev->max_mem_alloc_size = 0;
261 dev->max_parameter_size = 1024;
262 dev->min_data_type_align_size = MAX_EXTENDED_ALIGNMENT;
263 dev->mem_base_addr_align = MAX_EXTENDED_ALIGNMENT;
264 dev->half_fp_config = 0;
265 dev->single_fp_config = CL_FP_ROUND_TO_NEAREST | CL_FP_INF_NAN;
266#ifdef __x86_64__
267 dev->single_fp_config |= (CL_FP_DENORM | CL_FP_ROUND_TO_INF
268 | CL_FP_ROUND_TO_ZERO
269 | CL_FP_CORRECTLY_ROUNDED_DIVIDE_SQRT);
270#ifdef OCS_AVAILABLE
271 if (cpu_has_fma())
272 dev->single_fp_config |= CL_FP_FMA;
273#endif
274#endif
275
276#ifdef _CL_DISABLE_DOUBLE
277 dev->double_fp_config = 0;
278#else
279 /* TODO: all of these are the minimum mandated, but not all CPUs may actually
280 * support all of them. */
281 dev->double_fp_config = CL_FP_FMA | CL_FP_ROUND_TO_NEAREST
282 | CL_FP_ROUND_TO_ZERO | CL_FP_ROUND_TO_INF
283 | CL_FP_INF_NAN | CL_FP_DENORM;
284 /* this is a workaround for issue 28 in https://github.com/Oblomov/clinfo
285 * https://github.com/Oblomov/clinfo/issues/28 */
286 dev->double_fp_config |= CL_FP_CORRECTLY_ROUNDED_DIVIDE_SQRT;
287#endif
288
289 dev->global_mem_cache_type = CL_NONE;
290 dev->global_mem_cacheline_size = 0;
291 dev->global_mem_cache_size = 0;
292 dev->global_mem_size = 0;
293 dev->max_constant_buffer_size = 0;
294 dev->max_constant_args = 8;
295 dev->local_mem_type = CL_GLOBAL;
296 dev->local_mem_size = 0;
297 dev->error_correction_support = CL_FALSE;
298 dev->host_unified_memory = CL_TRUE;
299
300 dev->profiling_timer_resolution = pocl_timer_resolution;
301
302 dev->endian_little = !(WORDS_BIGENDIAN);
303 dev->available = CL_TRUE;
304 dev->compiler_available = CL_TRUE;
305 dev->spmd = CL_FALSE;
306 dev->arg_buffer_launcher = CL_FALSE;
307 dev->workgroup_pass = CL_TRUE;
308 dev->execution_capabilities = CL_EXEC_KERNEL | CL_EXEC_NATIVE_KERNEL;
309 dev->platform = 0;
310
311 dev->parent_device = NULL;
312 /* These two are only used for subdevices.
313 * Each subdevice has these two setup when created.
314 * The subdevice will then use these CUs:
315 * [start, start+1, ..., start+count-1]
316 * this may not work with more complicated partitioning schemes,
317 * but is good enough for now. */
318 dev->core_start = 0;
319 dev->core_count = 0;
320 /* basic does not support partitioning */
321 dev->max_sub_devices = 1;
322 dev->num_partition_properties = 1;
323 dev->partition_properties = basic_partition_properties;
324 dev->num_partition_types = 0;
325 dev->partition_type = NULL;
326
327 dev->device_side_printf = 1;
328 dev->printf_buffer_size = 16 * 1024 * 1024;
329
330 dev->vendor = "pocl";
331 dev->profile = "FULL_PROFILE";
332 /* Note: The specification describes identifiers being delimited by
333 only a single space character. Some programs that check the device's
334 extension string assume this rule. Future extension additions should
335 ensure that there is no more than a single space between
336 identifiers. */
337
338 dev->global_as_id = dev->local_as_id = dev->constant_as_id = 0;
339
340 dev->should_allocate_svm = 0;
341 /* OpenCL 2.0 properties */
342 dev->svm_caps = CL_DEVICE_SVM_COARSE_GRAIN_BUFFER
343 | CL_DEVICE_SVM_FINE_GRAIN_BUFFER
344 | CL_DEVICE_SVM_ATOMICS;
345 /* TODO these are minimums, figure out whats a reasonable value */
346 dev->max_events = 1024;
347 dev->max_queues = 1;
348 dev->max_pipe_args = 16;
349 dev->max_pipe_active_res = 1;
350 dev->max_pipe_packet_size = 1024;
351 dev->dev_queue_pref_size = 16 * 1024;
352 dev->dev_queue_max_size = 256 * 1024;
353 dev->on_dev_queue_props = CL_QUEUE_OUT_OF_ORDER_EXEC_MODE_ENABLE
354 | CL_QUEUE_PROFILING_ENABLE;
355 dev->on_host_queue_props = CL_QUEUE_PROFILING_ENABLE;
356 dev->extensions = HOST_DEVICE_EXTENSIONS;
357 dev->has_64bit_long = 1;
358 dev->autolocals_to_args = 1;
359 dev->device_alloca_locals = 0;
360
361#ifdef OCS_AVAILABLE
362
363 dev->llvm_target_triplet = OCL_KERNEL_TARGET;
364#ifdef HOST_CPU_FORCED
365 dev->llvm_cpu = OCL_KERNEL_TARGET_CPU;
366#else
367 dev->llvm_cpu = get_llvm_cpu_name ();
368#endif
369
370 dev->spirv_version = "SPIR-V_1.2";
371#else /* No compiler, no CPU info */
372 dev->llvm_cpu = NULL;
373 dev->llvm_target_triplet = "";
374#endif
375
376}

378unsigned int
379pocl_basic_probe(struct pocl_device_ops *ops)
380{
381 int env_count = pocl_device_get_env_count(ops->device_name);
382
383 /* No env specified, so pthread will be used instead of basic */
384 if(env_count < 0)
385 return 0;
386
387 return env_count;
388}

390cl_int
391pocl_basic_init (unsigned j, cl_device_id device, const char* parameters)
392{
393 struct data *d;
394 cl_int ret = CL_SUCCESS;
395 int err;
396 static int first_basic_init = 1;
397
398 if (first_basic_init)
399 {
400 POCL_MSG_WARN ("INIT dlcache DOTO delete\n");
401 pocl_init_dlhandle_cache();
402 first_basic_init = 0;
403 }
404 device->global_mem_id = 0;
405
406 d = (struct data *) calloc (1, sizeof (struct data));
407 if (d == NULL)
408 return CL_OUT_OF_HOST_MEMORY;
409
410 d->current_kernel = NULL;
411 device->data = d;
412
413 pocl_init_cpu_device_infos (device);
414
415 /* hwloc probes OpenCL device info at its initialization in case
416 the OpenCL extension is enabled. This causes to printout
417 an unimplemented property error because hwloc is used to
418 initialize global_mem_size which it is not yet. Just put
419 a nonzero there for now. */
420 device->global_mem_size = 1;
421 err = pocl_topology_detect_device_info(device);
422 if (err)
423 ret = CL_INVALID_DEVICE;
424
425 POCL_INIT_LOCK (d->cq_lock);
426
427 assert (device->printf_buffer_size > 0);
428 d->printf_buffer = pocl_aligned_malloc (MAX_EXTENDED_ALIGNMENT,
429 device->printf_buffer_size);
430 assert (d->printf_buffer != NULL);
431
432 pocl_cpuinfo_detect_device_info(device);
433 pocl_set_buffer_image_limits(device);
434
435 /* in case hwloc doesn't provide a PCI ID, let's generate
436 a vendor id that hopefully is unique across vendors. */
437 const char *magic = "pocl";
438 if (device->vendor_id == 0)
439 device->vendor_id =
440 magic[0] | magic[1] << 8 | magic[2] << 16 | magic[3] << 24;
441
442 device->vendor_id += j;
443
444 /* The basic driver represents only one "compute unit" as
445 it doesn't exploit multiple hardware threads. Multiple
446 basic devices can be still used for task level parallelism
447 using multiple OpenCL devices. */
448 device->max_compute_units = 1;
449
450 return ret;
451}

453cl_int
454pocl_basic_alloc_mem_obj (cl_device_id device, cl_mem mem_obj, void* host_ptr)
455{
456 void *b = NULL;
457 cl_mem_flags flags = mem_obj->flags;
458 unsigned i;
459 POCL_MSG_PRINT_MEMORY (" mem %p, dev %d\n", mem_obj, device->dev_id);
460 /* check if some driver has already allocated memory for this mem_obj
461 in our global address space, and use that*/
462 for (i = 0; i < mem_obj->context->num_devices; ++i)
463 {
464 if (!mem_obj->device_ptrs[i].available)
465 continue;
466
467 if (mem_obj->device_ptrs[i].global_mem_id == device->global_mem_id
468 && mem_obj->device_ptrs[i].mem_ptr != NULL)
469 {
470 mem_obj->device_ptrs[device->dev_id].mem_ptr =
471 mem_obj->device_ptrs[i].mem_ptr;
472 POCL_MSG_PRINT_MEMORY (
473 "mem %p dev %d, using already allocated mem\n", mem_obj,
474 device->dev_id);
475 return CL_SUCCESS;
476 }
477 }
478
479 /* memory for this global memory is not yet allocated -> do it */
480 if (flags & CL_MEM_USE_HOST_PTR)
481 {
482 // mem_host_ptr must be non-NULL
483 assert(host_ptr != NULL);
484 b = host_ptr;
485 }
486 else
487 {
488 POCL_MSG_PRINT_MEMORY ("!USE_HOST_PTR\n");
489 b = pocl_aligned_malloc_global_mem (device, MAX_EXTENDED_ALIGNMENT,
490 mem_obj->size);
491 if (b==NULL)
492 return CL_MEM_OBJECT_ALLOCATION_FAILURE;
493
494 mem_obj->shared_mem_allocation_owner = device;
495 }
496
497 /* use this dev mem allocation as host ptr */
498 if ((flags & CL_MEM_ALLOC_HOST_PTR) && (mem_obj->mem_host_ptr == NULL))
499 mem_obj->mem_host_ptr = b;
500
501 if (flags & CL_MEM_COPY_HOST_PTR)
502 {
503 POCL_MSG_PRINT_MEMORY ("COPY_HOST_PTR\n");
504 // mem_host_ptr must be non-NULL
505 assert(host_ptr != NULL);
506 memcpy (b, host_ptr, mem_obj->size);
507 }
508
509 mem_obj->device_ptrs[device->dev_id].mem_ptr = b;
510
511 return CL_SUCCESS;
512}

515void
516pocl_basic_free (cl_device_id device, cl_mem memobj)
517{
518 cl_mem_flags flags = memobj->flags;
519
520 /* aloocation owner executes freeing */
521 if (flags & CL_MEM_USE_HOST_PTR ||
522 memobj->shared_mem_allocation_owner != device)
523 return;
524
525 void* ptr = memobj->device_ptrs[device->dev_id].mem_ptr;
526 size_t size = memobj->size;
527
528 if (memobj->flags | CL_MEM_ALLOC_HOST_PTR)
529 memobj->mem_host_ptr = NULL;
530 pocl_free_global_mem(device, ptr, size);
531}

561void
562pocl_basic_run (void *data, _cl_command_node *cmd)
563{
564 struct data *d;
565 struct pocl_argument *al;
566 size_t x, y, z;
567 unsigned i;
568 cl_kernel kernel = cmd->command.run.kernel;
569 pocl_kernel_metadata_t *meta = kernel->meta;
570 struct pocl_context *pc = &cmd->command.run.pc;
571
572 assert (data != NULL);
573 d = (struct data *) data;
574
575 d->current_kernel = kernel;
576
577 void **arguments = (void **)malloc (sizeof (void *)
578 * (meta->num_args + meta->num_locals));
579
580 /* Process the kernel arguments. Convert the opaque buffer
581 pointers to real device pointers, allocate dynamic local
582 memory buffers, etc. */
583 for (i = 0; i < meta->num_args; ++i)
584 {
585 al = &(cmd->command.run.arguments[i]);
586 if (ARG_IS_LOCAL (meta->arg_info[i]))
587 {
588 if (cmd->device->device_alloca_locals)
589 {
590 /* Local buffers are allocated in the device side work-group
591 launcher. Let's pass only the sizes of the local args in
592 the arg buffer. */
593 assert (sizeof (size_t) == sizeof (void *));
594 arguments[i] = (void *)al->size;
595 }
596 else
597 {
598 arguments[i] = malloc (sizeof (void *));
599 *(void **)(arguments[i]) =
600 pocl_aligned_malloc(MAX_EXTENDED_ALIGNMENT, al->size);
601 }
602 }
603 else if (meta->arg_info[i].type == POCL_ARG_TYPE_POINTER)
604 {
605 /* It's legal to pass a NULL pointer to clSetKernelArguments. In
606 that case we must pass the same NULL forward to the kernel.
607 Otherwise, the user must have created a buffer with per device
608 pointers stored in the cl_mem. */
609 arguments[i] = malloc (sizeof (void *));
610 if (al->value == NULL)
611 {
612 *(void **)arguments[i] = NULL;
613 }
614 else
615 {
616 cl_mem m = (*(cl_mem *)(al->value));
617 void *ptr = m->device_ptrs[cmd->device->dev_id].mem_ptr;
618 *(void **)arguments[i] = (char *)ptr + al->offset;
619 }
620 }
621 else if (meta->arg_info[i].type == POCL_ARG_TYPE_IMAGE)
622 {
623 dev_image_t di;
624 fill_dev_image_t (&di, al, cmd->device);
625
626 void *devptr = pocl_aligned_malloc (MAX_EXTENDED_ALIGNMENT,
627 sizeof (dev_image_t));
628 arguments[i] = malloc (sizeof (void *));
629 *(void **)(arguments[i]) = devptr;
630 memcpy (devptr, &di, sizeof (dev_image_t));
631 }
632 else if (meta->arg_info[i].type == POCL_ARG_TYPE_SAMPLER)
633 {
634 dev_sampler_t ds;
635 fill_dev_sampler_t(&ds, al);
636 arguments[i] = malloc (sizeof (void *));
637 *(void **)(arguments[i]) = (void *)ds;
638 }
639 else
640 {
641 arguments[i] = al->value;
642 }
643 }
644
645 if (!cmd->device->device_alloca_locals)
646 for (i = 0; i < meta->num_locals; ++i)
647 {
648 size_t s = meta->local_sizes[i];
649 size_t j = meta->num_args + i;
650 arguments[j] = malloc (sizeof (void *));
651 void *pp = pocl_aligned_malloc (MAX_EXTENDED_ALIGNMENT, s);
652 *(void **)(arguments[j]) = pp;
653 }
654
655 pc->printf_buffer = d->printf_buffer;
656 assert (pc->printf_buffer != NULL);
657 pc->printf_buffer_capacity = cmd->device->printf_buffer_size;
658 assert (pc->printf_buffer_capacity > 0);
659 uint32_t position = 0;
660 pc->printf_buffer_position = &position;
661
662 unsigned rm = pocl_save_rm ();
663 pocl_set_default_rm ();
664 unsigned ftz = pocl_save_ftz ();
665 pocl_set_ftz (kernel->program->flush_denorms);
666
667 for (z = 0; z < pc->num_groups[2]; ++z)
668 for (y = 0; y < pc->num_groups[1]; ++y)
669 for (x = 0; x < pc->num_groups[0]; ++x)
670 ((pocl_workgroup_func) cmd->command.run.wg)
671 ((uint8_t *)arguments, (uint8_t *)pc, x, y, z);
672
673 pocl_restore_rm (rm);
674 pocl_restore_ftz (ftz);
675
676 if (position > 0)
677 {
678 write (STDOUT_FILENO, pc->printf_buffer, position);
679 position = 0;
680 }
681
682 for (i = 0; i < meta->num_args; ++i)
683 {
684 if (ARG_IS_LOCAL (meta->arg_info[i]))
685 {
686 if (!cmd->device->device_alloca_locals)
687 {
688 POCL_MEM_FREE(*(void **)(arguments[i]));
689 POCL_MEM_FREE(arguments[i]);
690 }
691 else
692 {
693 /* Device side local space allocation has deallocation via stack
694 unwind. */
695 }
696 }
697 else if (meta->arg_info[i].type == POCL_ARG_TYPE_IMAGE
698 || meta->arg_info[i].type == POCL_ARG_TYPE_SAMPLER)
699 {
700 if (meta->arg_info[i].type != POCL_ARG_TYPE_SAMPLER)
701 POCL_MEM_FREE (*(void **)(arguments[i]));
702 POCL_MEM_FREE(arguments[i]);
703 }
704 else if (meta->arg_info[i].type == POCL_ARG_TYPE_POINTER)
705 {
706 POCL_MEM_FREE(arguments[i]);
707 }
708 }
709
710 if (!cmd->device->device_alloca_locals)
711 for (i = 0; i < meta->num_locals; ++i)
712 {
713 POCL_MEM_FREE (*(void **)(arguments[meta->num_args + i]));
714 POCL_MEM_FREE (arguments[meta->num_args + i]);
715 }
716 free(arguments);
717
718 pocl_release_dlhandle_cache (cmd);
719}

932void
933pocl_basic_memfill (void *data, pocl_mem_identifier *dst_mem_id,
934 cl_mem dst_buf, size_t size, size_t offset,
935 const void *__restrict__ pattern, size_t pattern_size)
936{
937 void *__restrict__ ptr = dst_mem_id->mem_ptr;
938 size_t i;
939 unsigned j;
940
941 /* memfill size is in bytes, we wanto make it into elements */
942 size /= pattern_size;
943 offset /= pattern_size;
944
945 switch (pattern_size)
946 {
947 case 1:
948 {
949 uint8_t * p = (uint8_t*)ptr + offset;
950 for (i = 0; i < size; i++)
951 p[i] = *(uint8_t*)pattern;
952 }
953 break;
954 case 2:
955 {
956 uint16_t * p = (uint16_t*)ptr + offset;
957 for (i = 0; i < size; i++)
958 p[i] = *(uint16_t*)pattern;
959 }
960 break;
961 case 4:
962 {
963 uint32_t * p = (uint32_t*)ptr + offset;
964 for (i = 0; i < size; i++)
965 p[i] = *(uint32_t*)pattern;
966 }
967 break;
968 case 8:
969 {
970 uint64_t * p = (uint64_t*)ptr + offset;
971 for (i = 0; i < size; i++)
972 p[i] = *(uint64_t*)pattern;
973 }
974 break;
975 case 16:
976 {
977 uint64_t * p = (uint64_t*)ptr + (offset << 1);
978 for (i = 0; i < size; i++)
979 for (j = 0; j < 2; j++)
980 p[(i<<1) + j] = *((uint64_t*)pattern + j);
981 }
982 break;
983 case 32:
984 {
985 uint64_t * p = (uint64_t*)ptr + (offset << 2);
986 for (i = 0; i < size; i++)
987 for (j = 0; j < 4; j++)
988 p[(i<<2) + j] = *((uint64_t*)pattern + j);
989 }
990 break;
991 case 64:
992 {
993 uint64_t * p = (uint64_t*)ptr + (offset << 3);
994 for (i = 0; i < size; i++)
995 for (j = 0; j < 8; j++)
996 p[(i<<3) + j] = *((uint64_t*)pattern + j);
997 }
998 break;
999 case 128:
1000 {
1001 uint64_t * p = (uint64_t*)ptr + (offset << 4);
1002 for (i = 0; i < size; i++)
1003 for (j = 0; j < 16; j++)
1004 p[(i<<4) + j] = *((uint64_t*)pattern + j);
1005 }
1006 break;
1007 default:
1008 assert (0 && "Invalid pattern size");
1009 break;
1010 }
1011}

1013cl_int
1014pocl_basic_map_mem (void *data,
1015 pocl_mem_identifier * src_mem_id,
1016 cl_mem src_buf,
1017 mem_mapping_t *map)
1018{
1019 void *__restrict__ src_device_ptr = src_mem_id->mem_ptr;
1020 void *host_ptr = map->host_ptr;
1021 size_t offset = map->offset;
1022 size_t size = map->size;
1023
1024 if (host_ptr == NULL)
1025 {
1026 map->host_ptr = ((char *)src_device_ptr + offset);
1027 return CL_SUCCESS;
1028 }
1029
1030 if (map->map_flags & CL_MAP_WRITE_INVALIDATE_REGION)
1031 return CL_SUCCESS;
1032
1033 /* If the buffer was allocated with CL_MEM_ALLOC_HOST_PTR |
1034 * CL_MEM_COPY_HOST_PTR,
1035 * the dst_host_ptr is not the same memory as pocl's device_ptrs[], and we
1036 * need to copy pocl's buffer content back to dst_host_ptr. */
1037 if (host_ptr != (src_device_ptr + offset))
1038 {
1039 POCL_MSG_PRINT_MEMORY ("device: MAP memcpy() "
1040 "src_device_ptr %p + offset %zu"
1041 "to dst_host_ptr %p\n",
1042 src_device_ptr, offset, host_ptr);
1043 memcpy ((char *)host_ptr, (char *)src_device_ptr + offset, size);
1044 }
1045 return CL_SUCCESS;
1046}

1048cl_int
1049pocl_basic_unmap_mem(void *data,
1050 pocl_mem_identifier * dst_mem_id,
1051 cl_mem dst_buf,
1052 mem_mapping_t *map)
1053{
1054 void *__restrict__ dst_device_ptr = dst_mem_id->mem_ptr;
1055 /* it could be CL_MAP_READ | CL_MAP_WRITE(..invalidate) which has to be
1056 * handled like a write */
1057 if (map->map_flags == CL_MAP_READ)
1058 return CL_SUCCESS;
1059
1060 void *host_ptr = map->host_ptr;
1061 assert (host_ptr != NULL);
1062 size_t offset = map->offset;
1063 size_t size = map->size;
1064
1065 /* If the buffer was allocated with CL_MEM_ALLOC_src_host_ptr |
1066 * CL_MEM_COPY_src_host_ptr,
1067 * the src_host_ptr is not the same memory as pocl's device_ptrs[], and we
1068 * need to copy src_host_ptr content back to pocl's device_ptrs[]. */
1069 if (host_ptr != (dst_device_ptr + offset))
1070 {
1071 POCL_MSG_PRINT_MEMORY ("device: UNMAP memcpy() "
1072 "host_ptr %p to buf_ptr %p + offset %zu\n",
1073 host_ptr, dst_device_ptr, offset);
1074 memcpy ((char *)dst_device_ptr + offset, (char *)host_ptr, size);
1075 }
1076 return CL_SUCCESS;
1077}

1110static void basic_command_scheduler (struct data *d)
1111{
1112 _cl_command_node *node;
1113
1114 /* execute commands from ready list */
1115 while ((node = d->ready_list))
1116 {
1117 assert (pocl_command_is_ready(node->event));
1118 assert (node->event->status == CL_SUBMITTED);
1119 CDL_DELETE (d->ready_list, node);
1120 POCL_UNLOCK (d->cq_lock);
1121 pocl_exec_command (node);
1122 POCL_LOCK (d->cq_lock);
1123 }
1124
1125 return;
1126}

1128void
1129pocl_basic_submit (_cl_command_node *node, cl_command_queue cq)
1130{
1131 struct data *d = node->device->data;
1132
1133 if (node != NULL && node->type == CL_COMMAND_NDRANGE_KERNEL)
1134 pocl_check_kernel_dlhandle_cache (node, 1, 1);
1135
1136 node->ready = 1;
1137 POCL_LOCK (d->cq_lock);
1138 pocl_command_push(node, &d->ready_list, &d->command_list);
1139
1140 POCL_UNLOCK_OBJ (node->event);
1141 basic_command_scheduler (d);
1142 POCL_UNLOCK (d->cq_lock);
1143
1144 return;
1145}

1168void
1169pocl_basic_notify (cl_device_id device, cl_event event, cl_event finished)
1170{
1171 struct data *d = (struct data*)device->data;
1172 _cl_command_node * volatile node = event->command;
1173
1174 if (finished->status < CL_COMPLETE)
1175 {
1176 pocl_update_event_failed (event);
1177 return;
1178 }
1179
1180 if (!node->ready)
1181 return;
1182
1183 if (pocl_command_is_ready (event))
1184 {
1185 if (event->status == CL_QUEUED)
1186 {
1187 pocl_update_event_submitted (event);
1188 POCL_LOCK (d->cq_lock);
1189 CDL_DELETE (d->command_list, node);
1190 CDL_PREPEND (d->ready_list, node);
1191 basic_command_scheduler (d);
1192 POCL_UNLOCK (d->cq_lock);
1193 }
1194 return;
1195 }

1400pocl_basic_fill_image (void *data, cl_mem image,
1401 pocl_mem_identifier *image_data, const size_t *origin,
1402 const size_t *region,
1403 const void *__restrict__ fill_pixel, size_t pixel_size)
1404{
1405 POCL_MSG_PRINT_MEMORY ("BASIC / FILL IMAGE \n"
1406 "image %p data %p \n"
1407 "origin [0,1,2] %zu %zu %zu \n"
1408 "region [0,1,2] %zu %zu %zu \n"
1409 "pixel %p size %zu \n",
1410 image, image_data,
1411 origin[0], origin[1], origin[2],
1412 region[0], region[1], region[2],
1413 fill_pixel, pixel_size);
1414
1415 size_t row_pitch = image->image_row_pitch;
1416 size_t slice_pitch = image->image_slice_pitch;
1417 char *__restrict const adjusted_device_ptr
1418 = (char *)image_data->mem_ptr
1419 + origin[0] * pixel_size
1420 + row_pitch * origin[1]
1421 + slice_pitch * origin[2];
1422
1423 size_t i, j, k;
1424
1425 for (k = 0; k < region[2]; ++k)
1426 for (j = 0; j < region[1]; ++j)
1427 for (i = 0; i < region[0]; ++i)
1428 memcpy (adjusted_device_ptr
1429 + pixel_size * i
1430 + row_pitch * j
1431 + slice_pitch * k,
1432 fill_pixel,
1433 pixel_size);
1434 return CL_SUCCESS;
