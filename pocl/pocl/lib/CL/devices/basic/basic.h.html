24/**
25 * @file basic.h
26 *
27 * The purpose of the 'basic' device driver is to serve as an example of
28 * a minimalistic (but still working) device driver for pocl.
29 *
30 * It is a "native device" without multithreading and uses the malloc
31 * directly for buffer allocation etc. It also executes work groups in
32 * their sequential (increasing) order, thus makes it useful as a test
33 * device.
34 */
35

46#endif /* POCL_BASIC_H */
