46 * Sets up:
47 * max_compute_units
48 * global_mem_size
49 * global_mem_cache_type
50 * global_mem_cacheline_size
51 * global_mem_cache_size

56int
57pocl_topology_detect_device_info(cl_device_id device)
58{
59 hwloc_topology_t pocl_topology;
60 int ret = 0;
61
62#ifdef HWLOC_API_2
63 if (hwloc_get_api_version () < 0x20000)
64 POCL_MSG_ERR ("pocl was compiled against libhwloc 2.x but is"
65 "actually running against libhwloc 1.x \n");
66#else
67 if (hwloc_get_api_version () >= 0x20000)
68 POCL_MSG_ERR ("pocl was compiled against libhwloc 1.x but is"
69 "actually running against libhwloc 2.x \n");
70#endif
71
72 /*
73
74 * hwloc's OpenCL backend causes problems at the initialization stage
75 * because it reloads libpocl.so via the ICD loader.
76 *
77 * See: https://github.com/pocl/pocl/issues/261
78 *
79 * The only trick to stop hwloc from initializing the OpenCL plugin
80 * I could find is to point the plugin search path to a place where there
81 * are no plugins to be found.
82 */
83 setenv ("HWLOC_PLUGINS_PATH", "/dev/null", 1);
84
85 ret = hwloc_topology_init (&pocl_topology);
86 if (ret == -1)
87 {
88 POCL_MSG_ERR ("Cannot initialize the topology.\n");
89 return ret;
90 }
91
92#ifdef HWLOC_API_2
93 hwloc_topology_set_io_types_filter(pocl_topology, HWLOC_TYPE_FILTER_KEEP_NONE);
94 hwloc_topology_set_type_filter (pocl_topology, HWLOC_OBJ_SYSTEM, HWLOC_TYPE_FILTER_KEEP_NONE);
95 hwloc_topology_set_type_filter (pocl_topology, HWLOC_OBJ_GROUP, HWLOC_TYPE_FILTER_KEEP_NONE);
96 hwloc_topology_set_type_filter (pocl_topology, HWLOC_OBJ_BRIDGE, HWLOC_TYPE_FILTER_KEEP_NONE);
97 hwloc_topology_set_type_filter (pocl_topology, HWLOC_OBJ_MISC, HWLOC_TYPE_FILTER_KEEP_NONE);
98 hwloc_topology_set_type_filter (pocl_topology, HWLOC_OBJ_PCI_DEVICE, HWLOC_TYPE_FILTER_KEEP_NONE);
99 hwloc_topology_set_type_filter (pocl_topology, HWLOC_OBJ_OS_DEVICE, HWLOC_TYPE_FILTER_KEEP_NONE);
100#else
101 hwloc_topology_ignore_type (pocl_topology, HWLOC_TOPOLOGY_FLAG_WHOLE_IO);
102 hwloc_topology_ignore_type (pocl_topology, HWLOC_OBJ_SYSTEM);
103 hwloc_topology_ignore_type (pocl_topology, HWLOC_OBJ_GROUP);
104 hwloc_topology_ignore_type (pocl_topology, HWLOC_OBJ_BRIDGE);
105 hwloc_topology_ignore_type (pocl_topology, HWLOC_OBJ_MISC);
106 hwloc_topology_ignore_type (pocl_topology, HWLOC_OBJ_PCI_DEVICE);
107 hwloc_topology_ignore_type (pocl_topology, HWLOC_OBJ_OS_DEVICE);
108#endif
109
110 ret = hwloc_topology_load (pocl_topology);
111 if (ret == -1)
112 {
113 POCL_MSG_ERR ("Cannot load the topology.\n");
114 goto exit_destroy;
115 }
116
117#ifdef HWLOC_API_2
118 device->global_mem_size =
119 hwloc_get_root_obj(pocl_topology)->total_memory;
120#else
121 device->global_mem_size =
122 hwloc_get_root_obj(pocl_topology)->memory.total_memory;
123#endif
124
125 // Try to get the number of CPU cores from topology
126 int depth = hwloc_get_type_depth(pocl_topology, HWLOC_OBJ_PU);
127 if(depth != HWLOC_TYPE_DEPTH_UNKNOWN)
128 device->max_compute_units = hwloc_get_nbobjs_by_depth(pocl_topology, depth);
129
130 /* Find information about global memory cache by looking at the first
131 * cache covering the first PU */
132 do {
133 size_t cache_size = 0, cacheline_size = 0;
134
135 hwloc_obj_t core
136 = hwloc_get_next_obj_by_type (pocl_topology, HWLOC_OBJ_CORE, NULL);
137 if (core)
138 {
139 hwloc_obj_t cache
140 = hwloc_get_shared_cache_covering_obj (pocl_topology, core);
141 if ((cache) && (cache->attr))
142 {
143 cacheline_size = cache->attr->cache.linesize;
144 cache_size = cache->attr->cache.size;
145 }
146 else
147 core = NULL; /* fallback to L1 cache size */
148 }
149
150 hwloc_obj_t pu
151 = hwloc_get_next_obj_by_type (pocl_topology, HWLOC_OBJ_PU, NULL);
152 if (!core && pu)
153 {
154 hwloc_obj_t cache
155 = hwloc_get_shared_cache_covering_obj (pocl_topology, pu);
156 if ((cache) && (cache->attr))
157 {
158 cacheline_size = cache->attr->cache.linesize;
159 cache_size = cache->attr->cache.size;
160 }
161 }
162
163 if (!cache_size || !cacheline_size)
164 break;
165
166 device->global_mem_cache_type
167 = 0x2; // CL_READ_WRITE_CACHE, without including all of CL/cl.h
168 device->global_mem_cacheline_size = cacheline_size;
169 device->global_mem_cache_size = cache_size;
170 } while (0);
171
172 // Destroy topology object and return
173exit_destroy:
174 hwloc_topology_destroy (pocl_topology);
175 return ret;
176
177}

182#define L3_CACHE_SIZE "/sys/devices/system/cpu/cpu0/cache/index3/size"
183#define L2_CACHE_SIZE "/sys/devices/system/cpu/cpu0/cache/index2/size"
184#define CPUS "/sys/devices/system/cpu/possible"
185#define MEMINFO "/proc/meminfo"
186
187int
188pocl_topology_detect_device_info (cl_device_id device)
189{
190 device->global_mem_cacheline_size = HOST_CPU_CACHELINE_SIZE;
191 device->global_mem_cache_type
192 = 0x2; // CL_READ_WRITE_CACHE, without including all of CL/cl.h
193
194 /* global mem cache size */
195
196 char *content;
197 uint64_t filesize;
198
199 if (pocl_read_file (L3_CACHE_SIZE, &content, &filesize) == 0)
200 {
201 long val = atol (content);
202 device->global_mem_cache_size = val * 1024;
203 POCL_MEM_FREE (content);
204 }
205 else
206 {
207 if (pocl_read_file (L2_CACHE_SIZE, &content, &filesize) == 0)
208 {
209 long val = atol (content);
210 device->global_mem_cache_size = val * 1024;
211 POCL_MEM_FREE (content);
212 }
213 else
214 {
215 POCL_MSG_WARN (
216 "Could not figure out CPU cache size, using bogus value\n");
217 device->global_mem_cache_size = 1 << 20;
218 }
219 }
220
221 /* global_mem_size */
222 if (pocl_read_file (MEMINFO, &content, &filesize) == 0)
223 {
224 char *tmp = content;
225 unsigned long memsize_kb;
226 size_t i;
227
228 while (*tmp && (*tmp != '\n'))
229 ++tmp;
230 *tmp = 0;
231 tmp = content;
232 while (*tmp && (*tmp != 0x20))
233 ++tmp;
234 while (*tmp && (*tmp == 0x20))
235 ++tmp;
236 int items = sscanf (tmp, "%lu kB", &memsize_kb);
237
238 assert (items == 1);
239
240 device->global_mem_size = memsize_kb * 1024;
241 POCL_MEM_FREE (content);
242 }
243 else
244 {
245 POCL_MSG_WARN ("Cannot get memory size\n");
246 device->global_mem_size = 256 << 20;
247 }
248
249 /* max_compute_units */
250 if (pocl_read_file (CPUS, &content, &filesize) == 0)
251 {
252 assert (content);
253 assert (filesize > 0);
254 unsigned long start, end;
255 int items = sscanf (content, "%lu-%lu", &start, &end);
256 assert (items == 2);
257 device->max_compute_units = (unsigned)end + 1;
258 POCL_MEM_FREE (content);
259 }
260 else
261 {
262 POCL_MSG_WARN ("Cannot get logical CPU number\n");
263 device->max_compute_units = 1;
264 }
265
266 return 0;
267}
