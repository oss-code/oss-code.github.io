75void
76pocl_pthread_init_device_ops(struct pocl_device_ops *ops)
77{
78 pocl_basic_init_device_ops(ops);
79
80 ops->device_name = "pthread";
81
82 /* implementation that differs from basic */
83 ops->probe = pocl_pthread_probe;
84 ops->uninit = pocl_pthread_uninit;
85 ops->reinit = pocl_pthread_reinit;
86 ops->init = pocl_pthread_init;
87 ops->run = pocl_pthread_run;
88 ops->join = pocl_pthread_join;
89 ops->submit = pocl_pthread_submit;
90 ops->notify = pocl_pthread_notify;
91 ops->broadcast = pocl_broadcast;
92 ops->flush = pocl_pthread_flush;
93 ops->wait_event = pocl_pthread_wait_event;
94 ops->notify_event_finished = pocl_pthread_notify_event_finished;
95 ops->notify_cmdq_finished = pocl_pthread_notify_cmdq_finished;
96 ops->update_event = pocl_pthread_update_event;
97 ops->free_event_data = pocl_pthread_free_event_data;
98 ops->build_hash = pocl_pthread_build_hash;
99
100 ops->init_queue = pocl_pthread_init_queue;
101 ops->free_queue = pocl_pthread_free_queue;
102}

104char *
105pocl_pthread_build_hash (cl_device_id device)
106{
107 char* res = calloc(1000, sizeof(char));
108#ifdef KERNELLIB_HOST_DISTRO_VARIANTS
109 char *name = get_llvm_cpu_name ();
110 snprintf (res, 1000, "pthread-%s-%s", HOST_DEVICE_BUILD_HASH, name);
111 POCL_MEM_FREE (name);
112#else
113 snprintf (res, 1000, "pthread-%s", HOST_DEVICE_BUILD_HASH);
114#endif
115 return res;
116}

118unsigned int
119pocl_pthread_probe (struct pocl_device_ops *ops)
120{
121 int env_count = pocl_device_get_env_count(ops->device_name);
122 /* Env was not specified, default behavior was to use 1 pthread device */
123 if (env_count < 0)
124 return 1;
125
126 return env_count;
127}

129static cl_device_partition_property pthread_partition_properties[2]
130 = { CL_DEVICE_PARTITION_EQUALLY, CL_DEVICE_PARTITION_BY_COUNTS };
131
132#define FALLBACK_MAX_THREAD_COUNT 8
133
134char scheduler_initialized = 0;
135
136cl_int
137pocl_pthread_init (unsigned j, cl_device_id device, const char* parameters)
138{
139 struct data *d;
140 cl_int ret = CL_SUCCESS;
141 int err;
142
143 d = (struct data *) calloc (1, sizeof (struct data));
144 if (d == NULL)
145 return CL_OUT_OF_HOST_MEMORY;
146
147 d->current_kernel = NULL;
148 device->data = d;
149
150 pocl_init_cpu_device_infos (device);
151
152 device->on_host_queue_props
153 = CL_QUEUE_OUT_OF_ORDER_EXEC_MODE_ENABLE | CL_QUEUE_PROFILING_ENABLE;
154
155 /* hwloc probes OpenCL device info at its initialization in case
156 the OpenCL extension is enabled. This causes to printout
157 an unimplemented property error because hwloc is used to
158 initialize global_mem_size which it is not yet. Just put
159 a nonzero there for now. */
160 device->global_mem_size = 1;
161 err = pocl_topology_detect_device_info (device);
162 if (err)
163 ret = CL_INVALID_DEVICE;
164
165 /* device->max_compute_units was set up by topology_detect,
166 * but if the user requests, lower it */
167 int fallback = (device->max_compute_units == 0) ? FALLBACK_MAX_THREAD_COUNT
168 : device->max_compute_units;
169 int max_thr = pocl_get_int_option ("POCL_MAX_PTHREAD_COUNT", fallback);
170
171 device->max_compute_units
172 = max ((unsigned)max_thr,
173 (unsigned)pocl_get_int_option ("POCL_PTHREAD_MIN_THREADS", 1));
174
175 pocl_cpuinfo_detect_device_info(device);
176 pocl_set_buffer_image_limits(device);
177
178 /* in case hwloc doesn't provide a PCI ID, let's generate
179 a vendor id that hopefully is unique across vendors. */
180 const char *magic = "pocl";
181 if (device->vendor_id == 0)
182 device->vendor_id =
183 magic[0] | magic[1] << 8 | magic[2] << 16 | magic[3] << 24;
184
185 device->vendor_id += j;
186
187 // pthread has elementary partitioning support
188 device->max_sub_devices = device->max_compute_units;
189 device->num_partition_properties = 2;
190 device->partition_properties = pthread_partition_properties;
191 device->num_partition_types = 0;
192 device->partition_type = NULL;
193
194 if (!scheduler_initialized)
195 {
196 scheduler_initialized = 1;
197 pocl_init_dlhandle_cache();
198 pocl_init_kernel_run_command_manager();
199 pthread_scheduler_init (device);
200 }
201 /* system mem as global memory */
202 device->global_mem_id = 0;
203 return ret;
204}

206cl_int
207pocl_pthread_uninit (unsigned j, cl_device_id device)
208{
209 struct data *d = (struct data*)device->data;
210
211 if (scheduler_initialized)
212 {
213 pthread_scheduler_uninit ();
214 scheduler_initialized = 0;
215 }
216
217 POCL_MEM_FREE(d);
218 device->data = NULL;
219 return CL_SUCCESS;
220}
221
222cl_int
223pocl_pthread_reinit (unsigned j, cl_device_id device)
224{
225 struct data *d;
226
227 d = (struct data *)calloc (1, sizeof (struct data));
228 if (d == NULL)
229 return CL_OUT_OF_HOST_MEMORY;
230
231 d->current_kernel = NULL;
232 device->data = d;
233
234 if (!scheduler_initialized)
235 {
236 pthread_scheduler_init (device);
237 scheduler_initialized = 1;
238 }
239
240 return CL_SUCCESS;
241}

249void
250pocl_pthread_submit (_cl_command_node *node, cl_command_queue cq)
251{
252 node->ready = 1;
253 if (pocl_command_is_ready (node->event))
254 {
255 pocl_update_event_submitted (node->event);
256 pthread_scheduler_push_command (node);
257 }
258 POCL_UNLOCK_OBJ (node->event);
259 return;
260}

268void
269pocl_pthread_join(cl_device_id device, cl_command_queue cq)
270{
271 POCL_LOCK_OBJ (cq);
272 pthread_cond_t *cq_cond = (pthread_cond_t *)cq->data;
273 while (1)
274 {
275 if (cq->command_count == 0)
276 {
277 POCL_UNLOCK_OBJ (cq);
278 return;
279 }
280 else
281 {
282 int r = pthread_cond_wait (cq_cond, &cq->pocl_lock);
283 assert (r == 0);
284 }
285 }
286 return;
287}

289void
290pocl_pthread_notify (cl_device_id device, cl_event event, cl_event finished)
291{
292 int wake_thread = 0;
293 _cl_command_node * volatile node = event->command;
294
295 if (finished->status < CL_COMPLETE)
296 {
297 pocl_update_event_failed (event);
298 return;
299 }
300
301 if (!node->ready)
302 return;
303
304 if (pocl_command_is_ready (node->event))
305 {
306 if (event->status == CL_QUEUED)
307 {
308 pocl_update_event_submitted (event);
309 wake_thread = 1;
310 }
311 }
312 if (wake_thread)
313 {
314 pthread_scheduler_push_command (node);
315 }
316 return;
317}

319void
320pocl_pthread_notify_cmdq_finished (cl_command_queue cq)
321{
322 /* must be called with CQ already locked.
323 * this must be a broadcast since there could be multiple
324 * user threads waiting on the same command queue
325 * in pthread_scheduler_wait_cq(). */
326 pthread_cond_t *cq_cond = (pthread_cond_t *)cq->data;
327 int r = pthread_cond_broadcast (cq_cond);
328 assert (r == 0);
329}

331void
332pocl_pthread_notify_event_finished (cl_event event)
333{
334 struct event_data *e_d = event->data;
335 pthread_cond_broadcast (&e_d->event_cond);
336}

338void
339pocl_pthread_update_event (cl_device_id device, cl_event event)
340{
341 struct event_data *e_d = NULL;
342 if (event->data == NULL && event->status == CL_QUEUED)
343 {
344 e_d = malloc(sizeof(struct event_data));
345 assert(e_d);
346
347 pthread_cond_init(&e_d->event_cond, NULL);
348 event->data = (void *) e_d;
349 }
350}
351
352void pocl_pthread_wait_event (cl_device_id device, cl_event event)
353{
354 struct event_data *e_d = event->data;
355
356 POCL_LOCK_OBJ (event);
357 while (event->status > CL_COMPLETE)
358 {
359 pthread_cond_wait(&e_d->event_cond, &event->pocl_lock);
360 }
361 POCL_UNLOCK_OBJ (event);
362}

372cl_int
373pocl_pthread_init_queue (cl_command_queue queue)
374{
375 queue->data
376 = pocl_aligned_malloc (HOST_CPU_CACHELINE_SIZE, sizeof (pthread_cond_t));
377 pthread_cond_t *cond = (pthread_cond_t *)queue->data;
378 int r = pthread_cond_init (cond, NULL);
379 assert (r == 0);
380 return CL_BUILD_SUCCESS;
381}
382
383void
384pocl_pthread_free_queue (cl_command_queue queue)
385{
386 pthread_cond_t *cond = (pthread_cond_t *)queue->data;
387 int r = pthread_cond_destroy (cond);
388 assert (r == 0);
389 POCL_MEM_FREE (queue->data);
390}
