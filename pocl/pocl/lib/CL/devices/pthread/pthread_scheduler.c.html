46struct pool_thread_data
47{
48 pthread_t thread __attribute__ ((aligned (HOST_CPU_CACHELINE_SIZE)));
49
50 unsigned long executed_commands;
51 /* per-CU (= per-thread) local memory */
52 void *local_mem;
53 unsigned current_ftz;
54 unsigned num_threads;
55 /* index of this particular thread
56 * [0, num_threads-1]
57 * used for deciding whether a particular thread should run
58 * commands scheduled on a subdevice. */
59 unsigned index;
60 /* printf buffer*/
61 void *printf_buffer;
62} __attribute__ ((aligned (HOST_CPU_CACHELINE_SIZE)));
63
64typedef struct scheduler_data_
65{
66 unsigned num_threads;
67 unsigned printf_buf_size;
68
69 struct pool_thread_data *thread_pool;
70 size_t local_mem_size;
71
72 _cl_command_node *work_queue
73 __attribute__ ((aligned (HOST_CPU_CACHELINE_SIZE)));
74 kernel_run_command *kernel_queue;
75
76 pthread_cond_t wake_pool __attribute__ ((aligned (HOST_CPU_CACHELINE_SIZE)));
77 POCL_FAST_LOCK_T wq_lock_fast __attribute__ ((aligned (HOST_CPU_CACHELINE_SIZE)));
78
79 int thread_pool_shutdown_requested;
80} scheduler_data __attribute__ ((aligned (HOST_CPU_CACHELINE_SIZE)));

82static scheduler_data scheduler;

84void
85pthread_scheduler_init (cl_device_id device)
86{
87 unsigned i;
88 size_t num_worker_threads = device->max_compute_units;
89 POCL_FAST_INIT (scheduler.wq_lock_fast);
90
91 pthread_cond_init (&(scheduler.wake_pool), NULL);
92
93 scheduler.thread_pool = pocl_aligned_malloc (
94 HOST_CPU_CACHELINE_SIZE,
95 num_worker_threads * sizeof (struct pool_thread_data));
96 memset (scheduler.thread_pool, 0,
97 num_worker_threads * sizeof (struct pool_thread_data));
98
99 scheduler.num_threads = num_worker_threads;
100 assert (num_worker_threads > 0);
101 scheduler.printf_buf_size = device->printf_buffer_size;
102 assert (device->printf_buffer_size > 0);
103
104 /* safety margin - aligning pointers later (in kernel arg setup)
105 * may require more local memory than actual local mem size.
106 * TODO fix this */
107 scheduler.local_mem_size = device->local_mem_size << 4;
108
109 for (i = 0; i < num_worker_threads; ++i)
110 {
111 scheduler.thread_pool[i].index = i;
112 pthread_create (&scheduler.thread_pool[i].thread, NULL,
113 pocl_pthread_driver_thread,
114 (void*)&scheduler.thread_pool[i]);
115 }
116
117}

119void
120pthread_scheduler_uninit ()
121{
122 unsigned i;
123
124 POCL_FAST_LOCK (scheduler.wq_lock_fast);
125 scheduler.thread_pool_shutdown_requested = 1;
126 pthread_cond_broadcast (&scheduler.wake_pool);
127 POCL_FAST_UNLOCK (scheduler.wq_lock_fast);
128
129 for (i = 0; i < scheduler.num_threads; ++i)
130 {
131 pthread_join (scheduler.thread_pool[i].thread, NULL);
132 }
133
134 pocl_aligned_free (scheduler.thread_pool);
135 POCL_FAST_DESTROY (scheduler.wq_lock_fast);
136 pthread_cond_destroy (&scheduler.wake_pool);
137
138 scheduler.thread_pool_shutdown_requested = 0;
139}

141/* push_command and push_kernel MUST use broadcast and wake up all threads,
142 because commands can be for subdevices (= not all threads) */
143void pthread_scheduler_push_command (_cl_command_node *cmd)
144{
145 POCL_FAST_LOCK (scheduler.wq_lock_fast);
146 DL_APPEND (scheduler.work_queue, cmd);
147 pthread_cond_broadcast (&scheduler.wake_pool);
148 POCL_FAST_UNLOCK (scheduler.wq_lock_fast);
149}

151static void
152pthread_scheduler_push_kernel (kernel_run_command *run_cmd)
153{
154 POCL_FAST_LOCK (scheduler.wq_lock_fast);
155 DL_APPEND (scheduler.kernel_queue, run_cmd);
156 pthread_cond_broadcast (&scheduler.wake_pool);
157 POCL_FAST_UNLOCK (scheduler.wq_lock_fast);
158}

160/* if subd is not a subdevice, returns 1
161 * if subd is subdevice, takes a look at the subdevice CUs
162 * and if they match the current driver thread, returns 1
163 * otherwise returns 0 */
164static int
165shall_we_run_this (thread_data *td, cl_device_id subd)
166{
167
168 if (subd && subd->parent_device)
169 {
170 if (!((td->index >= subd->core_start)
171 && (td->index < (subd->core_start + subd->core_count))))
172 {
173 return 0;
174 }
175 }
176 return 1;
177}

179/* Maximum and minimum chunk sizes for get_wg_index_range().
180 * Each pthread driver's thread fetches work from a kernel's WG pool in
181 * chunks, this determines the limits (scaled up by # of threads). */
182#define POCL_PTHREAD_MAX_WGS 256
183#define POCL_PTHREAD_MIN_WGS 32
184
185static int
186get_wg_index_range (kernel_run_command *k, unsigned *start_index,
187 unsigned *end_index, int *last_wgs, unsigned num_threads)
188{
189 const unsigned scaled_max_wgs = POCL_PTHREAD_MAX_WGS * num_threads;
190 const unsigned scaled_min_wgs = POCL_PTHREAD_MIN_WGS * num_threads;
191
192 unsigned max_wgs;
193 POCL_FAST_LOCK (k->lock);
194 if (k->remaining_wgs == 0)
195 {
196 POCL_FAST_UNLOCK (k->lock);
197 return 0;
198 }
199
200 /* If the work is comprised of huge number of WGs of small WIs,
201 * then get_wg_index_range() becomes a problem on manycore CPUs
202 * because lock contention on k->lock.
203 *
204 * If we have enough workgroups, scale up the requests linearly by
205 * num_threads, otherwise fallback to smaller workgroups.
206 */
207 if (k->remaining_wgs <= (scaled_max_wgs * num_threads))
208 max_wgs = min (scaled_min_wgs, (1 + k->remaining_wgs / num_threads));
209 else
210 max_wgs = min (scaled_max_wgs, (1 + k->remaining_wgs / num_threads));
211
212 max_wgs = min (max_wgs, k->remaining_wgs);
213 assert (max_wgs > 0);
214
215 *start_index = k->wgs_dealt;
216 *end_index = k->wgs_dealt + max_wgs-1;
217 k->remaining_wgs -= max_wgs;
218 k->wgs_dealt += max_wgs;
219 if (k->remaining_wgs == 0)
220 *last_wgs = 1;
221 POCL_FAST_UNLOCK (k->lock);
222
223 return 1;
224}

226inline static void translate_wg_index_to_3d_index (kernel_run_command *k,
227 unsigned index,
228 size_t *index_3d,
229 unsigned xy_slice,
230 unsigned row_size)
231{
232 index_3d[2] = index / xy_slice;
233 index_3d[1] = (index % xy_slice) / row_size;
234 index_3d[0] = (index % xy_slice) % row_size;
235}

237static int
238work_group_scheduler (kernel_run_command *k,
239 struct pool_thread_data *thread_data)
240{
241 pocl_kernel_metadata_t *meta = k->kernel->meta;
242
243 void *arguments[meta->num_args + meta->num_locals + 1];
244 void *arguments2[meta->num_args + meta->num_locals + 1];
245 struct pocl_context pc;
246 unsigned i;
247 unsigned start_index;
248 unsigned end_index;
249 int last_wgs = 0;
250
251 if (!get_wg_index_range (k, &start_index, &end_index, &last_wgs,
252 thread_data->num_threads))
253 return 0;
254
255 assert (end_index >= start_index);
256
257 setup_kernel_arg_array_with_locals (
258 (void **)&arguments, (void **)&arguments2, k, thread_data->local_mem,
259 scheduler.local_mem_size);
260 memcpy (&pc, &k->pc, sizeof (struct pocl_context));
261
262 // capacity and position already set up
263 pc.printf_buffer = thread_data->printf_buffer;
264 uint32_t position = 0;
265 pc.printf_buffer_position = &position;
266 assert (pc.printf_buffer != NULL);
267 assert (pc.printf_buffer_capacity > 0);
268 assert (pc.printf_buffer_position != NULL);
269
270 /* Flush to zero is only set once at start of kernel (because FTZ is
271 * a compilation option), but we need to reset rounding mode after every
272 * iteration (since it can be changed during kernel execution). */
273 unsigned flush = k->kernel->program->flush_denorms;
274 if (thread_data->current_ftz != flush)
275 {
276 pocl_set_ftz (flush);
277 thread_data->current_ftz = flush;
278 }
279
280 unsigned slice_size = k->pc.num_groups[0] * k->pc.num_groups[1];
281 unsigned row_size = k->pc.num_groups[0];
282
283 do
284 {
285 if (last_wgs)
286 {
287 POCL_FAST_LOCK (scheduler.wq_lock_fast);
288 DL_DELETE (scheduler.kernel_queue, k);
289 POCL_FAST_UNLOCK (scheduler.wq_lock_fast);
290 }
291
292 for (i = start_index; i <= end_index; ++i)
293 {
294 size_t gids[3];
295 translate_wg_index_to_3d_index (k, i, gids,
296 slice_size, row_size);
297
298#ifdef DEBUG_MT
299 printf("### exec_wg: gid_x %zu, gid_y %zu, gid_z %zu\n",
300 gids[0], gids[1], gids[2]);
301#endif
302 pocl_set_default_rm ();
303 k->workgroup ((uint8_t*)arguments, (uint8_t*)&pc,
304 gids[0], gids[1], gids[2]);
305 }
306 }
307 while (get_wg_index_range (k, &start_index, &end_index, &last_wgs,
308 thread_data->num_threads));
309
310 if (position > 0)
311 {
312 write (STDOUT_FILENO, pc.printf_buffer, position);
313 }
314
315 free_kernel_arg_array_with_locals ((void **)&arguments, (void **)&arguments2,
316 k);
317
318 return 1;
319}

321static void
322finalize_kernel_command (struct pool_thread_data *thread_data,
323 kernel_run_command *k)
324{
325#ifdef DEBUG_MT
326 printf("### kernel %s finished\n", k->cmd->command.run.kernel->name);
327#endif
328
329 free_kernel_arg_array (k);
330
331 pocl_release_dlhandle_cache (k->cmd);
332
333 pocl_ndrange_node_cleanup (k->cmd);
334
335 POCL_UPDATE_EVENT_COMPLETE_MSG (k->cmd->event, "NDRange Kernel ");
336
337 pocl_mem_manager_free_command (k->cmd);
338 POCL_FAST_DESTROY (k->lock);
339 free_kernel_run_command (k);
340}

342static void
343pocl_pthread_prepare_kernel (void *data, _cl_command_node *cmd)
344{
345 kernel_run_command *run_cmd;
346 cl_kernel kernel = cmd->command.run.kernel;
347 struct pocl_context *pc = &cmd->command.run.pc;
348
349 pocl_check_kernel_dlhandle_cache (cmd, 1, 1);
350
351 size_t num_groups = pc->num_groups[0] * pc->num_groups[1] * pc->num_groups[2];
352
353 run_cmd = new_kernel_run_command ();
354 run_cmd->data = data;
355 run_cmd->kernel = kernel;
356 run_cmd->device = cmd->device;
357 run_cmd->pc = *pc;
358 run_cmd->cmd = cmd;
359 run_cmd->pc.printf_buffer = NULL;
360 run_cmd->pc.printf_buffer_capacity = scheduler.printf_buf_size;
361 run_cmd->pc.printf_buffer_position = NULL;
362 run_cmd->remaining_wgs = num_groups;
363 run_cmd->wgs_dealt = 0;
364 run_cmd->workgroup = cmd->command.run.wg;
365 run_cmd->kernel_args = cmd->command.run.arguments;
366 run_cmd->next = NULL;
367 run_cmd->ref_count = 0;
368 POCL_FAST_INIT (run_cmd->lock);
369
370 setup_kernel_arg_array (run_cmd);
371
372 pocl_update_event_running (cmd->event);
373
374 pthread_scheduler_push_kernel (run_cmd);
375}

377/*
378 These two check the entire kernel/cmd queue. This is necessary
379 because of commands for subdevices. The old code only checked
380 the head of each queue; this can lead to a deadlock:
381
382 two driver threads, each assigned two subdevices A, B, one
383 driver queue C
384
385 cmd A1 for A arrives in C, A starts processing
386 cmd B1 for B arrives in C, B starts processing
387 cmds A2, A3, B2 are pushed to C
388 B finishes processing B1, checks queue head, A2 isn't for it, goes to sleep
389 A finishes processing A1, processes A2 + A3 but ignores B2, it's not for it
390 application calls clFinish to wait for queue
391
392 ...now B is sleeping and not possible to wake up
393 (since no new commands can arrive) and there's a B2 command
394 which will never be processed.
395
396 it's possible to workaround but it's cleaner to just check the whole queue.
397 */
398
399static _cl_command_node *
400check_cmd_queue_for_device (thread_data *td)
401{
402 _cl_command_node *cmd;
403 DL_FOREACH (scheduler.work_queue, cmd)
404 {
405 cl_device_id subd = cmd->device;
406 if (shall_we_run_this (td, subd))
407 {
408 DL_DELETE (scheduler.work_queue, cmd)
409 return cmd;
410 }
411 }
412
413 return NULL;
414}
415
416static kernel_run_command *
417check_kernel_queue_for_device (thread_data *td)
418{
419 kernel_run_command *cmd;
420 DL_FOREACH (scheduler.kernel_queue, cmd)
421 {
422 cl_device_id subd = cmd->device;
423 if (shall_we_run_this (td, subd))
424 return cmd;
425 }
426
427 return NULL;
428}

430static int
431pthread_scheduler_get_work (thread_data *td)
432{
433 _cl_command_node *cmd;
434 kernel_run_command *run_cmd;
435
436 /* execute kernel if available */
437 POCL_FAST_LOCK (scheduler.wq_lock_fast);
438 int do_exit = 0;
439
440RETRY:
441 do_exit = scheduler.thread_pool_shutdown_requested;
442
443 run_cmd = check_kernel_queue_for_device (td);
444 /* execute kernel if available */
445 if (run_cmd)
446 {
447 ++run_cmd->ref_count;
448 POCL_FAST_UNLOCK (scheduler.wq_lock_fast);
449
450 work_group_scheduler (run_cmd, td);
451
452 POCL_FAST_LOCK (scheduler.wq_lock_fast);
453 if ((--run_cmd->ref_count) == 0)
454 {
455 POCL_FAST_UNLOCK (scheduler.wq_lock_fast);
456 finalize_kernel_command (td, run_cmd);
457 POCL_FAST_LOCK (scheduler.wq_lock_fast);
458 }
459 }
460
461 /* execute a command if available */
462 cmd = check_cmd_queue_for_device (td);
463 if (cmd)
464 {
465 POCL_FAST_UNLOCK (scheduler.wq_lock_fast);
466
467 assert (pocl_command_is_ready (cmd->event));
468
469 if (cmd->type == CL_COMMAND_NDRANGE_KERNEL)
470 {
471 pocl_pthread_prepare_kernel (cmd->device->data, cmd);
472 }
473 else
474 {
475 pocl_exec_command (cmd);
476 }
477
478 POCL_FAST_LOCK (scheduler.wq_lock_fast);
479 ++td->executed_commands;
480 }
481
482 /* if neither a command nor a kernel was available, sleep */
483 if ((cmd == NULL) && (run_cmd == NULL) && (do_exit == 0))
484 {
485 pthread_cond_wait (&scheduler.wake_pool, &scheduler.wq_lock_fast);
486 goto RETRY;
487 }
488
489 POCL_FAST_UNLOCK (scheduler.wq_lock_fast);
490
491 return do_exit;
492}

495static
496void*
497pocl_pthread_driver_thread (void *p)
498{
499 struct pool_thread_data *td = (struct pool_thread_data*)p;
500 int do_exit = 0;
501 assert (td);
502 /* some random value, doesn't matter as long as it's not a valid bool - to
503 * force a first FTZ setup */
504 td->current_ftz = 213;
505 td->num_threads = scheduler.num_threads;
506 td->printf_buffer = pocl_aligned_malloc (MAX_EXTENDED_ALIGNMENT,
507 scheduler.printf_buf_size);
508 assert (td->printf_buffer != NULL);
509
510 assert (scheduler.local_mem_size > 0);
511 td->local_mem = pocl_aligned_malloc (MAX_EXTENDED_ALIGNMENT,
512 scheduler.local_mem_size);
513 assert (td->local_mem);
514#ifdef __linux__
515 if (pocl_get_bool_option ("POCL_AFFINITY", 0))
516 {
517 cpu_set_t set;
518 CPU_ZERO (&set);
519 CPU_SET (td->index, &set);
520 pthread_setaffinity_np (td->thread, sizeof (cpu_set_t), &set);
521 }
522#endif
523
524 while (1)
525 {
526 do_exit = pthread_scheduler_get_work (td);
527 if (do_exit)
528 {
529 pocl_aligned_free (td->printf_buffer);
530 pocl_aligned_free (td->local_mem);
531 pthread_exit (NULL);
532 }
533 }
