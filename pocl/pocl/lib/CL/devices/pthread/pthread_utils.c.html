34static kernel_run_command *volatile kernel_pool = 0;
35static int kernel_pool_initialized = 0;
36static pocl_lock_t kernel_pool_lock;
37
38
39void pocl_init_kernel_run_command_manager (void)
40{
41 if (!kernel_pool_initialized)
42 {
43 kernel_pool_initialized = 1;
44 POCL_INIT_LOCK (kernel_pool_lock);
45 }
46}
47
48void pocl_init_thread_argument_manager (void)
49{
50 if (!kernel_pool_initialized)
51 {
52 kernel_pool_initialized = 1;
53 POCL_INIT_LOCK (kernel_pool_lock);
54 }
55}

57kernel_run_command* new_kernel_run_command ()
58{
59 kernel_run_command *volatile k = NULL;
60 POCL_LOCK (kernel_pool_lock);
61 if ((k = kernel_pool))
62 {
63 LL_DELETE (kernel_pool, k);
64 memset (k, 0, sizeof(kernel_run_command));
65 pthread_mutex_init(&k->lock, NULL);
66 POCL_UNLOCK (kernel_pool_lock);
67 return k;
68 }
69
70 POCL_UNLOCK (kernel_pool_lock);
71 k = (kernel_run_command*)calloc (1, sizeof (kernel_run_command));
72 pthread_mutex_init (&k->lock, NULL);
73 return k;
74}
75
76void free_kernel_run_command (kernel_run_command *k)
77{
78 POCL_LOCK (kernel_pool_lock);
79 pthread_mutex_destroy (&k->lock);
80 LL_PREPEND (kernel_pool, k);
81 POCL_UNLOCK (kernel_pool_lock);
82}

86#define ARGS_SIZE (sizeof (void *) * (meta->num_args + meta->num_locals + 1))
87
88static char *
89align_ptr (char *p)
90{
91 uintptr_t r = (uintptr_t)p;
92 if (r & (MAX_EXTENDED_ALIGNMENT - 1))
93 {
94 r = r & (~(MAX_EXTENDED_ALIGNMENT - 1));
95 r += MAX_EXTENDED_ALIGNMENT;
96 }
97 return (char *)r;
98}

100/* called from kernel setup code.
101 * Sets up the actual arguments, except the local ones. */
102void
103setup_kernel_arg_array (kernel_run_command *k)
104{
105 struct pocl_argument *al;
106
107 pocl_kernel_metadata_t *meta = k->kernel->meta;
108 cl_uint i;
109 void **arguments;
110 void **arguments2;
111 k->arguments = arguments
112 = pocl_aligned_malloc (MAX_EXTENDED_ALIGNMENT, ARGS_SIZE);
113 k->arguments2 = arguments2
114 = pocl_aligned_malloc (MAX_EXTENDED_ALIGNMENT, ARGS_SIZE);
115
116 for (i = 0; i < meta->num_args; ++i)
117 {
118 al = &(k->kernel_args[i]);
119 if (ARG_IS_LOCAL (meta->arg_info[i]))
120 {
121 arguments[i] = NULL;
122 arguments2[i] = NULL;
123 }
124 else if (meta->arg_info[i].type == POCL_ARG_TYPE_POINTER)
125 {
126 /* It's legal to pass a NULL pointer to clSetKernelArguments. In
127 that case we must pass the same NULL forward to the kernel.
128 Otherwise, the user must have created a buffer with per device
129 pointers stored in the cl_mem. */
130 arguments[i] = &arguments2[i];
131 if (al->value == NULL)
132 {
133 arguments2[i] = NULL;
134 }
135 else
136 {
137 cl_mem m = *(cl_mem *)al->value;
138 if (m->device_ptrs)
139 arguments2[i] = m->device_ptrs[k->device->dev_id].mem_ptr;
140 else
141 arguments2[i] = m->mem_host_ptr;
142
143 arguments2[i] += al->offset;
144 }
145 }
146 else if (meta->arg_info[i].type == POCL_ARG_TYPE_IMAGE)
147 {
148 dev_image_t di;
149 fill_dev_image_t(&di, al, k->device);
150 void *devptr = pocl_aligned_malloc (MAX_EXTENDED_ALIGNMENT,
151 sizeof (dev_image_t));
152 arguments[i] = &arguments2[i];
153 arguments2[i] = devptr;
154 memcpy (devptr, &di, sizeof (dev_image_t));
155 }
156 else if (meta->arg_info[i].type == POCL_ARG_TYPE_SAMPLER)
157 {
158 dev_sampler_t ds;
159 fill_dev_sampler_t(&ds, al);
160
161 arguments[i] = &arguments2[i];
162 arguments2[i] = (void *)ds;
163 }
164 else
165 arguments[i] = al->value;
166 }
167}

169/* called from each driver thread.
170 * "arguments" and "arguments2" are the output:
171 * driver-thread-local copies of kern args.
172 *
173 * they're set up by 1) memcpy from kernel_run_command, 2) all
174 * local args are set to thread-local "local memory" storage. */
175void
176setup_kernel_arg_array_with_locals (void **arguments, void **arguments2,
177 kernel_run_command *k, char *local_mem,
178 size_t local_mem_size)
179{
180 pocl_kernel_metadata_t *meta = k->kernel->meta;
181 cl_uint i;
182
183 memcpy (arguments2, k->arguments2, ARGS_SIZE);
184 memcpy (arguments, k->arguments, ARGS_SIZE);
185
186 char *start = local_mem;
187
188 for (i = 0; i < meta->num_args; ++i)
189 {
190 if (ARG_IS_LOCAL (meta->arg_info[i]))
191 {
192 size_t size = k->kernel_args[i].size;
193 if (!k->device->device_alloca_locals)
194 {
195 arguments[i] = &arguments2[i];
196 arguments2[i] = start;
197 start += size;
198 start = align_ptr (start);
199 assert ((size_t) (start - local_mem) <= local_mem_size);
200 }
201 else
202 {
203 /* Local buffers are allocated in the device side work-group
204 launcher. Let's pass only the sizes of the local args in
205 the arg buffer. */
206 assert (sizeof (size_t) == sizeof (void *));
207 arguments[i] = (void *)size;
208 }
209 }
210 }
211 if (!k->device->device_alloca_locals)
212 /* Allocate the automatic local buffers which are implemented as implicit
213 extra arguments at the end of the kernel argument list. */
214 for (i = 0; i < meta->num_locals; ++i)
215 {
216 cl_uint j = meta->num_args + i;
217 size_t size = meta->local_sizes[i];
218 arguments[j] = &arguments2[j];
219 arguments2[j] = start;
220 start += size;
221 start = align_ptr (start);
222 assert ((size_t) (start - local_mem) <= local_mem_size);
223 }
224}

226/* called from kernel teardown code.
227 * frees the actual arguments, except the local ones. */
228void
229free_kernel_arg_array (kernel_run_command *k)
230{
231 cl_uint i;
232 pocl_kernel_metadata_t *meta = k->kernel->meta;
233 void **arguments = k->arguments;
234 void **arguments2 = k->arguments2;
235
236 for (i = 0; i < meta->num_args; ++i)
237 {
238 if (ARG_IS_LOCAL (meta->arg_info[i]))
239 {
240 if (!k->device->device_alloca_locals)
241 {
242 assert (arguments[i] == NULL);
243 assert (arguments2[i] == NULL);
244 }
245 else
246 {
247 /* Device side local space allocation has deallocation via stack
248 unwind. */
249 }
250 }
251 else if (meta->arg_info[i].type == POCL_ARG_TYPE_IMAGE)
252 {
253 POCL_MEM_FREE (arguments2[i]);
254 }
255 }
256
257 POCL_MEM_FREE (k->arguments);
258 POCL_MEM_FREE (k->arguments2);
259}

261/* called from each driver thread.
262 * frees the local arguments. */
263void
264free_kernel_arg_array_with_locals (void **arguments, void **arguments2,
265 kernel_run_command *k)
266{
267 pocl_kernel_metadata_t *meta = k->kernel->meta;
268 cl_uint i;
269
270 for (i = 0; i < meta->num_args; ++i)
271 {
272 if (ARG_IS_LOCAL (meta->arg_info[i]))
273 {
274 arguments[i] = NULL;
275 arguments2[i] = NULL;
276 }
277 }
278
279 for (i = 0; i < meta->num_locals; ++i)
280 {
281 arguments[meta->num_args + i] = NULL;
282 arguments2[meta->num_args + i] = NULL;
283 }
