23/**
24 * This file implements a customized memory allocator for OpenCL buffers.
25 *
26 * The allocator is designed to be fast and simple by tuning it to the
27 * common OpenCL allocation patterns. It can be used from host to allocate
28 * external device's memory or it can be used from within the device to
29 * dynamically allocate local memory chunks. The interface is address space
30 * agnostic; it treats memory addresses and regions of memory as integers.
31 *
32 * Certain assumptions of OpenCL allocation patterns are made to optimize and
33 * simplify the implementation:
34 *
35 * 1) The allocations are often quite large and the "lifetimes" of the
36 * buffers are of similar length.
37 *
38 * This affects memory fragmentation patterns. When two or more buffers
39 * with similar life times are allocated (typically all the buffer
40 * arguments to an OpenCL kernel invocation) and freed we should try to
41 * allocate sequential chunks for the buffers and in the free merge
42 * the freed buffer with a succeeding and preceeding ones. This should
43 * lead to large physical chunks allocated for each kernel's buffers
44 * which are also deallocated back to large region chunks.
45 *
46 * 2) The total number of allocations is not large.
47 *
48 * Traversing through the list of chunk infos when searching for an
49 * available freed chunk can be considered to be not very costly.
50 *
51 * 3a) There is no lack of (global) memory.
52 *
53 * A wasteful but fast strategy can be used. Here the chunk is always
54 * tried to be allocated to the end of the region to enforce "sequential
55 * large chunks" for multiple buffers of a kernel.
56 *
57 * 3b) The memory is more tight (e.g. when allocating from local memory).
58 *
59 * A slower but less wasteful strategy should be used. In this version
60 * the list of old chunks should be traversed first and reused in case
61 * a large enough unallocated one is found. This version can be used
62 * also for the case where there's a single region (basically heap) that
63 * grows towards the stack or the global data area of the memory.
64 *
65 * @author Pekka Jääskeläinen 2011-2012
66 *
67 * @file bufalloc.c
68 */

79void
80print_chunk (chunk_info_t *chunk)
81{
82 printf ("### chunk %p: allocated: %d start: %zx size: %zu prev: %p next: %p\n",
83 chunk, chunk->is_allocated, chunk->start_address,
84 chunk->size, chunk->prev, chunk->next);
85}

87void
88print_chunks (chunk_info_t *first)
89{
90 chunk_info_t *chunk;
91 DL_FOREACH (first, chunk)
92 {
93 print_chunk (chunk);
94 }
95}

97static int
98chunk_slack (chunk_info_t* chunk, size_t size, size_t* last_chunk_size)
99{
100 memory_address_t aligned_start_addr =
101 (chunk->start_address + chunk->parent_region->alignment - 1) &
102 ~(chunk->parent_region->alignment - 1);
103 size_t end_chunk = chunk->start_address + chunk->size;
104 size_t end_addr = aligned_start_addr + size;
105 if(last_chunk_size)
106 *last_chunk_size = end_chunk - end_addr;
107 return end_chunk >= end_addr;
108}

110/**
111 * Tries to create a new chunk to the end of the given region.
112 *
113 * Assumes the last_chunk is always the last chunk in the region and
114 * it is unallocated in case there is free space at the end.
115 *
116 * @return The address of the chunk if it fits, 0 otherwise.
117 */
118static chunk_info_t *
119append_new_chunk (memory_region_t *region,
120 size_t size)
121{
122
123 chunk_info_t* new_chunk = NULL;
124 BA_LOCK (region->lock);
125
126#ifdef ENABLE_ASSERTS
127 assert (!region->last_chunk->is_allocated);
128#endif
129 /* if the last_chunk is too small we cannot append
130 a new chunk before it */
131 if (!chunk_slack (region->last_chunk, size, NULL))
132 {
133 BA_UNLOCK (region->lock);
134 return NULL;
135 }
136
137 /* ok, there should be space at the end, create a new chunk
138 before the last_chunk */
139 new_chunk = region->free_chunks;
140
141 if (new_chunk == NULL)
142 {
143 BA_UNLOCK (region->lock);
144 return NULL;
145 }
146 else
147 {
148 DL_DELETE (region->free_chunks, new_chunk);
149 }
150
151 /* Round the start address up towards the closest aligned
152 address. */
153 new_chunk->start_address =
154 (region->last_chunk->start_address + region->alignment - 1) &
155 ~(region->alignment - 1);
156 new_chunk->parent_region = region;
157 new_chunk->size = size;
158 new_chunk->is_allocated = 1;
159 new_chunk->children = NULL;
160
161 chunk_slack (region->last_chunk, size, (size_t*)&region->last_chunk->size);
162 region->last_chunk->start_address =
163 new_chunk->start_address + new_chunk->size;
164
165 DL_DELETE (region->chunks, region->last_chunk);
166 DL_APPEND (region->chunks, new_chunk);
167 DL_APPEND (region->chunks, region->last_chunk);
168
169#ifdef DEBUG_BUFALLOC
170 printf ("#### after append_new_chunk (%x, %u)\n", region, size);
171 print_chunks (region->chunks);
172 printf ("\n");
173#endif
174
175 BA_UNLOCK (region->lock);
176
177 return new_chunk;
178}

180/**
181 * Allocates a chunk of memory from the given memory region.
182 *
183 * @return The chunk, or NULL if no space available in the region.
184 */
185chunk_info_t*
186alloc_buffer_from_region (memory_region_t *region, size_t size)
187{
188#ifdef ENABLE_ASSERTS
189 assert (region != NULL);
190#endif
191 /* The memory-wasteful but fast strategy:
192
193 Assume there's plenty of memory so just try to append the
194 buffer to the end of the region without trying to reuse
195 unallocated ones first. */
196 chunk_info_t* chunk = NULL, *cursor;
197 if (region->strategy == BALLOCS_WASTEFUL)
198 {
199 chunk = append_new_chunk(region, size);
200 if (chunk != NULL) return chunk;
201 }
202
203 BA_LOCK (region->lock);
204
205 DL_FOREACH (region->chunks, cursor)
206 {
207 if (cursor == region->last_chunk ||
208 cursor->is_allocated ||
209 !chunk_slack (cursor, size, NULL))
210 {
211#ifdef DEBUG_BUFALLOC
212 printf ("### CHUNK REJECTED: SIZE: %zu IS_LAST: %u CHUNK %p: "
213 "allocated: %d start: %zx size: %zu prev: %p next: %p\n",
214 size, (unsigned)(cursor == region->last_chunk),
215 cursor, cursor->is_allocated, cursor->start_address,
216 cursor->size, cursor->prev, cursor->next);
217#endif
218 continue; /* doesn't fit */
219 }
220 /* found one */
221 chunk = cursor;
222 chunk->is_allocated = 1;
223
224#ifdef DEBUG_BUFALLOC
225 printf ("#### after reusing a chunk in region %x\n", region);
226 print_chunks (region->chunks);
227 printf ("\n");
228#endif
229 break;
230 }
231
232 BA_UNLOCK (region->lock);
233
234 if (chunk == NULL && region->strategy != BALLOCS_WASTEFUL)
235 {
236 return append_new_chunk (region, size);
237 }
238 return chunk;
239}

241/**
242 * Allocates a chunk of memory from one of the given memory regions.
243 *
244 * The address ranges of the different regions must not overlap. Searches
245 * through the regions in the order of the region pointer array.
246 *
247 * @param regions A linked list of region pointers.
248 * @param size The size of the chunk to allocate.
249 * @return The start address of the chunk, or 0 if no space available
250 * in the buffer.
251 */
252
253#ifndef BUFALLOC_NO_MULTIPLE_REGIONS
254chunk_info_t *
255alloc_buffer (memory_region_t *regions, size_t size)
256{
257 chunk_info_t *chunk = NULL;
258 memory_region_t *region = NULL;
259 LL_FOREACH(regions, region)
260 {
261 chunk = alloc_buffer_from_region (region, size);
262 if (chunk != NULL)
263 return chunk;
264 }
265 return NULL;
266}
267#endif

269/**
270 * Creates a reference to a part of a chunk.
271 *
272 * @todo Register to the parent also so it can free the
273 * child references.

275#ifndef BUFALLOC_NO_SUB_CHUNKS
276chunk_info_t *
277create_sub_chunk (chunk_info_t *parent, size_t offset, size_t size)
278{
279 chunk_info_t *subchunk = (chunk_info_t*)malloc(sizeof(struct chunk_info));
280 subchunk->start_address = parent->start_address + offset;
281 subchunk->size = size;
282 subchunk->parent = parent;
283 DL_APPEND(parent->children, subchunk);
284 return subchunk;
285}
286#endif

288/**
289 * Merges two unallocated chunks together to a larger chunk, if both
290 * are unallocated.
291 *
292 * The first chunk must before the second chunk in the memory region.
293 * Must be called inside a locked region.
294 *
295 * @return A pointer to the coalesced chunk, or the second chunk in case
296 * coalsecing could not be done.
297 */
298
299#ifndef BUFALLOC_NO_CHUNK_COALESCING
300static chunk_info_t *
301coalesce_chunks (chunk_info_t* first,
302 chunk_info_t* second)
303{
304 if (first == NULL) return second;
305 if (second == NULL) return first;
306 if (first->is_allocated || second->is_allocated) return second;
307
308 /* The linked list head has a prev pointing to the last (sentinel),
309 detect that here and do not merge first with the second. */
310 if (first->start_address > second->start_address) return second;
311
312#ifdef DEBUG_BUFALLOC
313 printf ("### coalescing chunks:\n");
314 print_chunk (first);
315 print_chunk (second);
316 puts ("\n");
317#endif
318
319 /* Should not just add the size of the second chunk as we might have
320 done alignment adjustment to the start address */
321 first->size = second->start_address + second->size - first->start_address;
322 DL_DELETE (first->parent_region->chunks, second);
323 DL_APPEND (first->parent_region->free_chunks, second);
324
325 /* Did we coalesce away the sentinel chunk? Need to set it to
326 a new valid one. */
327 if (second->parent_region->last_chunk == second)
328 second->parent_region->last_chunk = first;
329
330 return first;
331}
332#endif

334memory_region_t *
335free_buffer (memory_region_t *regions, memory_address_t addr)
336{
337 memory_region_t *region = NULL;
338
339#ifdef DEBUG_BUFALLOC
340 printf ("#### free_buffer(%p, %x)\n", regions, addr);
341#endif
342
343 LL_FOREACH (regions, region)
344 {
345 chunk_info_t *chunk = NULL;
346 BA_LOCK (region->lock);
347 DL_FOREACH (region->chunks, chunk)
348 {
349 if (chunk->start_address == addr)
350 {
351 chunk->is_allocated = 0;
352#ifndef BUFALLOC_NO_CHUNK_COALESCING
353 coalesce_chunks (coalesce_chunks (chunk->prev, chunk), chunk->next);
354#endif
355 BA_UNLOCK (region->lock);
356#ifdef DEBUG_BUFALLOC
357 printf ("#### region %x after free_buffer at addr %x\n",
358 region, addr);
359 print_chunks (region->chunks);
360 printf ("\n");
361#endif
362 return region;
363 }
364 }
365 BA_UNLOCK (region->lock);
366 }
367 return NULL;
368}

370/**
371 * Frees the given chunk.
372 *
373 * Successive unallocated chunks in the region, if found, are merged to
374 * form larger unallocated chunks.
375 */
376void
377free_chunk (chunk_info_t* chunk)
378{
379 memory_region_t *region = chunk->parent_region;
380 BA_LOCK (region->lock);
381 chunk->is_allocated = 0;
382#ifndef BUFALLOC_NO_CHUNK_COALESCING
383 coalesce_chunks (coalesce_chunks (chunk->prev, chunk), chunk->next);
384#endif
385 BA_UNLOCK (region->lock);
386
387#ifdef DEBUG_BUFALLOC
388 printf ("#### after free_chunk (%x)\n", chunk);
389 print_chunks (region->chunks);
390 printf ("\n");
391#endif
392
393}

395/** Initialize a memory_region_t.
396 * @param region is a pointer to a existing memory_region_t data structure.
397 * @param start the base address of the memory region to be managed.
398 * @Param size the size of the region (in bytes?)
399 */
400void
401init_mem_region (memory_region_t *region, memory_address_t start, size_t size)
402{
403 int i;
404 BA_INIT_LOCK (region->lock);
405
406 region->strategy = BALLOCS_WASTEFUL;
407 region->chunks = NULL;
408 region->free_chunks = NULL;
409 region->alignment = 64;
410 region->next = NULL;
411 region->prev = NULL;
412 /* Create the "sentinel chunk" */
413 region->last_chunk = &region->all_chunks[0];
414 region->last_chunk->start_address = start;
415 region->last_chunk->size = size;
416 region->last_chunk->is_allocated = 0;
417 region->last_chunk->parent_region = region;
418
419 DL_APPEND(region->chunks, region->last_chunk);
420
421 /* Setup the linked list of free chunk data structures */
422 for (i = 1; i < MAX_CHUNKS_IN_REGION; ++i)
423 DL_APPEND (region->free_chunks, &region->all_chunks[i]);
424
425#ifdef DEBUG_BUFALLOC
426 printf ("#### memory region %x created. start: %x size: %u\n",
427 region, start, size);
428#endif
429}
