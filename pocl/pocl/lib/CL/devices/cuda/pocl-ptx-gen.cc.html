65int pocl_ptx_gen(const char *BitcodeFilename, const char *PTXFilename,
66 const char *KernelName, const char *Arch,
67 const char *LibDevicePath, int HasOffsets) {
68 llvm::ErrorOr<std::unique_ptr<llvm::MemoryBuffer>> Buffer =
69 llvm::MemoryBuffer::getFile(BitcodeFilename);
70 if (!Buffer) {
71 POCL_MSG_ERR("[CUDA] ptx-gen: failed to open bitcode file\n");
72 return 1;
73 }
74
75 // Load the LLVM bitcode module.
76 llvm::LLVMContext Context;
77 llvm::Expected<std::unique_ptr<llvm::Module>> Module =
78 parseBitcodeFile(Buffer->get()->getMemBufferRef(), Context);
79 if (!Module) {
80 POCL_MSG_ERR("[CUDA] ptx-gen: failed to load bitcode\n");
81 return 1;
82 }
83
84 // Apply transforms to prepare for lowering to PTX.
85 fixPrintF(Module->get());
86 fixConstantMemArgs(Module->get(), KernelName);
87 fixLocalMemArgs(Module->get(), KernelName);
88 handleGetWorkDim(Module->get(), KernelName);
89 addKernelAnnotations(Module->get(), KernelName);
90 mapLibDeviceCalls(Module->get());
91 linkLibDevice(Module->get(), KernelName, LibDevicePath);
92 if (pocl_get_bool_option("POCL_CUDA_DUMP_NVVM", 0)) {
93 std::string ModuleString;
94 llvm::raw_string_ostream ModuleStringStream(ModuleString);
95 (*Module)->print(ModuleStringStream, NULL);
96 POCL_MSG_PRINT_INFO("NVVM module:\n%s\n", ModuleString.c_str());
97 }
98
99 std::string Error;
100
101 // Verify module.
102 if (pocl_get_bool_option("POCL_CUDA_VERIFY_MODULE", 1)) {
103 llvm::raw_string_ostream Errs(Error);
104 if (llvm::verifyModule(*Module->get(), &Errs)) {
105 POCL_MSG_ERR("\n%s\n", Error.c_str());
106 POCL_ABORT("[CUDA] ptx-gen: module verification failed\n");
107 }
108 }
109
110 llvm::StringRef Triple =
111 (sizeof(void *) == 8) ? "nvptx64-nvidia-cuda" : "nvptx-nvidia-cuda";
112
113 // Get NVPTX target.
114 const llvm::Target *Target =
115 llvm::TargetRegistry::lookupTarget(Triple, Error);
116 if (!Target) {
117 POCL_MSG_ERR("[CUDA] ptx-gen: failed to get target\n");
118 POCL_MSG_ERR("%s\n", Error.c_str());
119 return 1;
120 }
121
122 // TODO: Set options?
123 llvm::TargetOptions Options;
124
125 // TODO: CPU and features?
126 std::unique_ptr<llvm::TargetMachine> Machine(
127 Target->createTargetMachine(Triple, Arch, "+ptx40", Options, llvm::None));
128
129 llvm::legacy::PassManager Passes;
130
131 // Add pass to emit PTX.
132 llvm::SmallVector<char, 4096> Data;
133 llvm::raw_svector_ostream PTXStream(Data);
134 if (Machine->addPassesToEmitFile(Passes, PTXStream,
135#ifndef LLVM_OLDER_THAN_7_0
136 nullptr,
137#endif
138 llvm::TargetMachine::CGFT_AssemblyFile)) {
139 POCL_MSG_ERR("[CUDA] ptx-gen: failed to add passes\n");
140 return 1;
141 }
142
143 // Run passes.
144 Passes.run(**Module);
145
146 std::string PTX = PTXStream.str();
147 return pocl_write_file(PTXFilename, PTX.c_str(), PTX.size(), 0, 0);
148}

150// Add the metadata needed to mark a function as a kernel in PTX.
151void addKernelAnnotations(llvm::Module *Module, const char *KernelName) {
152 llvm::LLVMContext &Context = Module->getContext();
153
154 // Remove existing nvvm.annotations metadata since it is sometimes corrupt.
155 auto *Annotations = Module->getNamedMetadata("nvvm.annotations");
156 if (Annotations)
157 Annotations->eraseFromParent();
158
159 // Add nvvm.annotations metadata to mark kernel entry point.
160 Annotations = Module->getOrInsertNamedMetadata("nvvm.annotations");
161
162 // Get handle to function.
163 auto *Function = Module->getFunction(KernelName);
164 if (!Function)
165 POCL_ABORT("[CUDA] ptx-gen: kernel function not found in module\n");
166
167 // Create metadata.
168 llvm::Constant *One =
169 llvm::ConstantInt::getSigned(llvm::Type::getInt32Ty(Context), 1);
170 llvm::Metadata *FuncMD = llvm::ValueAsMetadata::get(Function);
171 llvm::Metadata *NameMD = llvm::MDString::get(Context, "kernel");
172 llvm::Metadata *OneMD = llvm::ConstantAsMetadata::get(One);
173
174 llvm::MDNode *Node = llvm::MDNode::get(Context, {FuncMD, NameMD, OneMD});
175 Annotations->addOperand(Node);
176}

178// PTX doesn't support variadic functions, so we need to modify the IR to
179// support printf. The vprintf system call that is provided is described here:
180// http://docs.nvidia.com/cuda/ptx-writers-guide-to-interoperability/index.html#system-calls
181// Essentially, the variadic list of arguments is replaced with a single array
182// instead.
183//
184// This function changes the prototype of printf to take an array instead
185// of a variadic argument list. It updates the function body to read from
186// this array to retrieve each argument instead of using the dummy __cl_va_arg
187// function. We then visit each printf callsite and generate the argument
188// array to pass instead of the variadic list.
189void fixPrintF(llvm::Module *Module) {
190 llvm::Function *OldPrintF = Module->getFunction("printf");
191 if (!OldPrintF)
192 return;
193
194 llvm::LLVMContext &Context = Module->getContext();
195 llvm::Type *I32 = llvm::Type::getInt32Ty(Context);
196 llvm::Type *I64 = llvm::Type::getInt64Ty(Context);
197 llvm::Type *I64Ptr = llvm::PointerType::get(I64, 0);
198 llvm::Type *FormatType = OldPrintF->getFunctionType()->getParamType(0);
199
200 // Remove calls to va_start and va_end.
201 pocl::eraseFunctionAndCallers(Module->getFunction("llvm.va_start"));
202 pocl::eraseFunctionAndCallers(Module->getFunction("llvm.va_end"));
203
204 // Create new non-variadic printf function.
205 llvm::Type *ReturnType = OldPrintF->getReturnType();
206 llvm::FunctionType *NewPrintfType =
207 llvm::FunctionType::get(ReturnType, {FormatType, I64Ptr}, false);
208 llvm::Function *NewPrintF = llvm::Function::Create(
209 NewPrintfType, OldPrintF->getLinkage(), "", Module);
210 NewPrintF->takeName(OldPrintF);
211
212 // Take function body from old function.
213 NewPrintF->getBasicBlockList().splice(NewPrintF->begin(),
214 OldPrintF->getBasicBlockList());
215
216 // Create i32 to hold current argument index.
217 llvm::AllocaInst *ArgIndexPtr =
218 new llvm::AllocaInst(I32, 0, llvm::ConstantInt::get(I32, 1));
219 ArgIndexPtr->insertBefore(&*NewPrintF->begin()->begin());
220 llvm::StoreInst *ArgIndexInit =
221 new llvm::StoreInst(llvm::ConstantInt::get(I32, 0), ArgIndexPtr);
222 ArgIndexInit->insertAfter(ArgIndexPtr);
223
224 // Replace calls to _cl_va_arg with reads from new i64 array argument.
225 llvm::Function *VaArgFunc = Module->getFunction("__cl_va_arg");
226 if (VaArgFunc) {
227 auto args = NewPrintF->arg_begin();
228 args++;
229 llvm::Argument *ArgsIn = args;
230 std::vector<llvm::Value *> VaArgCalls(VaArgFunc->user_begin(),
231 VaArgFunc->user_end());
232 for (auto &U : VaArgCalls) {
233 llvm::CallInst *Call = llvm::dyn_cast<llvm::CallInst>(U);
234 if (!Call)
235 continue;
236
237 // Get current argument index.
238 llvm::LoadInst *ArgIndex = new llvm::LoadInst(ArgIndexPtr);
239 ArgIndex->insertBefore(Call);
240
241 // Get pointer to argument data.
242 llvm::Value *ArgOut = Call->getArgOperand(1);
243 llvm::GetElementPtrInst *ArgIn =
244 llvm::GetElementPtrInst::Create(I64, ArgsIn, {ArgIndex});
245 ArgIn->insertAfter(ArgIndex);
246
247 // Cast ArgOut pointer to i64*.
248 llvm::BitCastInst *ArgOutBC = new llvm::BitCastInst(ArgOut, I64Ptr);
249 ArgOutBC->insertAfter(ArgIn);
250 ArgOut = ArgOutBC;
251
252 // Load argument.
253 llvm::LoadInst *ArgValue = new llvm::LoadInst(ArgIn);
254 ArgValue->insertAfter(ArgIn);
255 llvm::StoreInst *ArgStore = new llvm::StoreInst(ArgValue, ArgOut);
256 ArgStore->insertAfter(ArgOutBC);
257
258 // Increment argument index.
259 llvm::BinaryOperator *Inc = llvm::BinaryOperator::Create(
260 llvm::BinaryOperator::Add, ArgIndex, llvm::ConstantInt::get(I32, 1));
261 Inc->insertAfter(ArgIndex);
262 llvm::StoreInst *StoreInc = new llvm::StoreInst(Inc, ArgIndexPtr);
263 StoreInc->insertAfter(Inc);
264
265 // Remove call to _cl_va_arg.
266 Call->eraseFromParent();
267 }
268
269 // Remove function from module.
270 VaArgFunc->eraseFromParent();
271 }
272
273 // Loop over function callers.
274 // Generate array of i64 arguments to replace variadic arguments/
275 std::vector<llvm::Value *> Callers(OldPrintF->user_begin(),
276 OldPrintF->user_end());
277 for (auto &U : Callers) {
278 llvm::CallInst *Call = llvm::dyn_cast<llvm::CallInst>(U);
279 if (!Call)
280 continue;
281
282 unsigned NumArgs = Call->getNumArgOperands() - 1;
283 llvm::Value *Format = Call->getArgOperand(0);
284
285 // Allocate array for arguments.
286 // TODO: Deal with vector arguments.
287 llvm::AllocaInst *Args =
288 new llvm::AllocaInst(I64, 0, llvm::ConstantInt::get(I32, NumArgs));
289 Args->insertBefore(Call);
290
291 // Loop over arguments (skipping format).
292 for (unsigned A = 0; A < NumArgs; A++) {
293 llvm::Value *Arg = Call->getArgOperand(A + 1);
294 llvm::Type *ArgType = Arg->getType();
295
296 // Cast pointers to the generic address space.
297 if (ArgType->isPointerTy() && ArgType->getPointerAddressSpace() != 0) {
298 llvm::CastInst *AddrSpaceCast =llvm::CastInst::CreatePointerCast(
299 Arg, ArgType->getPointerElementType()->getPointerTo());
300 AddrSpaceCast->insertBefore(Call);
301 Arg = AddrSpaceCast;
302 ArgType = Arg->getType();
303 }
304
305 // Get pointer to argument in i64 array.
306 // TODO: promote arguments that are shorter than 32 bits.
307 llvm::Constant *ArgIndex = llvm::ConstantInt::get(I32, A);
308 llvm::Instruction *ArgPtr =
309 llvm::GetElementPtrInst::Create(I64, Args, {ArgIndex});
310 ArgPtr->insertBefore(Call);
311
312 // Cast pointer to correct type if necessary.
313 if (ArgPtr->getType()->getPointerElementType() != ArgType) {
314 llvm::BitCastInst *ArgPtrBC =
315 new llvm::BitCastInst(ArgPtr, ArgType->getPointerTo(0));
316 ArgPtrBC->insertAfter(ArgPtr);
317 ArgPtr = ArgPtrBC;
318 }
319
320 // Store argument to i64 array.
321 llvm::StoreInst *Store = new llvm::StoreInst(Arg, ArgPtr);
322 Store->insertBefore(Call);
323 }
324
325 // Fix address space of undef format values.
326 if (Format->getValueID() == llvm::Value::UndefValueVal) {
327 Format = llvm::UndefValue::get(FormatType);
328 }
329
330 // Replace call with new non-variadic function.
331 llvm::CallInst *NewCall = llvm::CallInst::Create(NewPrintF, {Format, Args});
332 NewCall->insertBefore(Call);
333 Call->replaceAllUsesWith(NewCall);
334 Call->eraseFromParent();
335 }
336
337 // Update arguments.
338 llvm::Function::arg_iterator OldArg = OldPrintF->arg_begin();
339 llvm::Function::arg_iterator NewArg = NewPrintF->arg_begin();
340 NewArg->takeName(&*OldArg);
341 OldArg->replaceAllUsesWith(&*NewArg);
342
343 // Remove old function.
344 OldPrintF->eraseFromParent();
345
346 // Get handle to vprintf function.
347 llvm::Function *VPrintF = Module->getFunction("vprintf");
348 if (!VPrintF)
349 return;
350
351 // If vprintf format address space is already generic, then we're done.
352 auto *VPrintFFormatType = VPrintF->getFunctionType()->getParamType(0);
353 if (VPrintFFormatType->getPointerAddressSpace() == 0)
354 return;
355
356 // Change address space of vprintf format argument to generic.
357 auto *I8Ptr = llvm::PointerType::get(llvm::Type::getInt8Ty(Context), 0);
358 auto *NewVPrintFType =
359 llvm::FunctionType::get(VPrintF->getReturnType(), {I8Ptr, I8Ptr}, false);
360 auto *NewVPrintF =
361 llvm::Function::Create(NewVPrintFType, VPrintF->getLinkage(), "", Module);
362 NewVPrintF->takeName(VPrintF);
363
364 // Update vprintf callers to pass format arguments in generic address space.
365 Callers.assign(VPrintF->user_begin(), VPrintF->user_end());
366 for (auto &U : Callers) {
367 llvm::CallInst *Call = llvm::dyn_cast<llvm::CallInst>(U);
368 if (!Call)
369 continue;
370
371 llvm::Value *Format = Call->getArgOperand(0);
372 llvm::Type *FormatType = Format->getType();
373 if (FormatType->getPointerAddressSpace() != 0) {
374 // Cast address space to generic.
375 llvm::Type *NewFormatType =
376 FormatType->getPointerElementType()->getPointerTo(0);
377 llvm::AddrSpaceCastInst *FormatASC =
378 new llvm::AddrSpaceCastInst(Format, NewFormatType);
379 FormatASC->insertBefore(Call);
380 Call->setArgOperand(0, FormatASC);
381 }
382 Call->setCalledFunction(NewVPrintF);
383 }
384
385 VPrintF->eraseFromParent();
386}

388// Replace all load users of a scalar global variable with new value.
389static void replaceScalarGlobalVar(llvm::Module *Module, const char *Name,
390 llvm::Value *NewValue) {
391 auto GlobalVar = Module->getGlobalVariable(Name);
392 if (!GlobalVar)
393 return;
394
395 std::vector<llvm::Value *> Users(GlobalVar->user_begin(),
396 GlobalVar->user_end());
397 for (auto *U : Users) {
398 auto Load = llvm::dyn_cast<llvm::LoadInst>(U);
399 assert(Load && "Use of a scalar global variable is not a load");
400 Load->replaceAllUsesWith(NewValue);
401 Load->eraseFromParent();
402 }
403 GlobalVar->eraseFromParent();
404}

406// Add an extra kernel argument for the dimensionality.
407void handleGetWorkDim(llvm::Module *Module, const char *KernelName) {
408 llvm::Function *Function = Module->getFunction(KernelName);
409 if (!Function)
410 POCL_ABORT("[CUDA] ptx-gen: kernel function not found in module\n");
411
412 // Add additional argument for the work item dimensionality.
413 llvm::FunctionType *FunctionType = Function->getFunctionType();
414 std::vector<llvm::Type *> ArgumentTypes(FunctionType->param_begin(),
415 FunctionType->param_end());
416 ArgumentTypes.push_back(llvm::Type::getInt32Ty(Module->getContext()));
417
418 // Create new function.
419 llvm::FunctionType *NewFunctionType =
420 llvm::FunctionType::get(Function->getReturnType(), ArgumentTypes, false);
421 llvm::Function *NewFunction = llvm::Function::Create(
422 NewFunctionType, Function->getLinkage(), Function->getName(), Module);
423 NewFunction->takeName(Function);
424
425 // Map function arguments.
426 llvm::ValueToValueMapTy VV;
427 llvm::Function::arg_iterator OldArg;
428 llvm::Function::arg_iterator NewArg;
429 for (OldArg = Function->arg_begin(), NewArg = NewFunction->arg_begin();
430 OldArg != Function->arg_end(); NewArg++, OldArg++) {
431 NewArg->takeName(&*OldArg);
432 VV[&*OldArg] = &*NewArg;
433 }
434
435 // Clone function.
436 llvm::SmallVector<llvm::ReturnInst *, 1> RI;
437 llvm::CloneFunctionInto(NewFunction, Function, VV, true, RI);
438
439 Function->eraseFromParent();
440
441 auto WorkDimVar = Module->getGlobalVariable("_work_dim");
442 if (!WorkDimVar)
443 return;
444
445 // Replace uses of the global offset variables with the new arguments.
446 NewArg->setName("work_dim");
447 replaceScalarGlobalVar(Module, "_work_dim", (&*NewArg++));
448
449 // TODO: What if get_work_dim() is called from a non-kernel function?
450}

452int findLibDevice(char LibDevicePath[PATH_MAX], const char *Arch) {
453 // Extract numeric portion of SM version.
454 char *End;
455 unsigned long SM = strtoul(Arch + 3, &End, 10);
456 if (!SM || strlen(End)) {
457 POCL_MSG_ERR("[CUDA] invalid GPU architecture %s\n", Arch);
458 return 1;
459 }
460
461 // This mapping from SM version to libdevice library version is given here:
462 // http://docs.nvidia.com/cuda/libdevice-users-guide/basic-usage.html#version-selection
463 // This is no longer needed as of CUDA 9.
464 int LibDeviceSM = 0;
465 if (SM < 30)
466 LibDeviceSM = 20;
467 else if (SM == 30)
468 LibDeviceSM = 30;
469 else if (SM < 35)
470 LibDeviceSM = 20;
471 else if (SM <= 37)
472 LibDeviceSM = 35;
473 else if (SM < 50)
474 LibDeviceSM = 30;
475 else if (SM <= 53)
476 LibDeviceSM = 50;
477 else
478 LibDeviceSM = 30;
479
480 const char *BasePath[] = {
481 pocl_get_string_option("POCL_CUDA_TOOLKIT_PATH", CUDA_TOOLKIT_ROOT_DIR),
482 "/usr/local/lib/cuda",
483 "/usr/local/lib",
484 "/usr/lib",
485 };
486
487 static const char *NVVMPath[] = {
488 "/nvvm",
489 "/nvidia-cuda-toolkit",
490 "",
491 };
492
493 static const char *PathFormat = "%s%s/libdevice/libdevice.10.bc";
494 static const char *OldPathFormat =
495 "%s%s/libdevice/libdevice.compute_%d.10.bc";
496
497 // Search combinations of paths for the libdevice library.
498 for (auto bp : BasePath) {
499 for (auto np : NVVMPath) {
500 // Check for CUDA 9+ libdevice library.
501 size_t ps = snprintf(LibDevicePath, PATH_MAX - 1, PathFormat, bp, np);
502 LibDevicePath[ps] = '\0';
503 POCL_MSG_PRINT2(CUDA, __FUNCTION__, __LINE__,
504 "looking for libdevice at '%s'\n", LibDevicePath);
505 if (pocl_exists(LibDevicePath)) {
506 POCL_MSG_PRINT2(CUDA, __FUNCTION__, __LINE__,
507 "found libdevice at '%s'\n", LibDevicePath);
508 return 0;
509 }
510
511 // Check for pre CUDA 9 libdevice library.
512 ps = snprintf(LibDevicePath, PATH_MAX - 1, OldPathFormat, bp, np,
513 LibDeviceSM);
514 LibDevicePath[ps] = '\0';
515 POCL_MSG_PRINT2(CUDA, __FUNCTION__, __LINE__,
516 "looking for libdevice at '%s'\n", LibDevicePath);
517 if (pocl_exists(LibDevicePath)) {
518 POCL_MSG_PRINT2(CUDA, __FUNCTION__, __LINE__,
519 "found libdevice at '%s'\n", LibDevicePath);
520 return 0;
521 }
522 }
523 }
524
525 return 1;
526}

528// Link CUDA's libdevice bitcode library to provide implementations for most of
529// the OpenCL math functions.
530// TODO: Can we link libdevice into the kernel library at pocl build time?
531// This would remove this runtime dependency on the CUDA toolkit.
532// Had some issues with the earlier pocl LLVM passes crashing on the libdevice
533// code - needs more investigation.
534void linkLibDevice(llvm::Module *Module, const char *KernelName,
535 const char *LibDevicePath) {
536 auto Buffer = llvm::MemoryBuffer::getFile(LibDevicePath);
537 if (!Buffer)
538 POCL_ABORT("[CUDA] failed to open libdevice library file\n");
539
540 POCL_MSG_PRINT_INFO("loading libdevice from '%s'\n", LibDevicePath);
541
542 // Load libdevice bitcode library.
543 llvm::Expected<std::unique_ptr<llvm::Module>> LibDeviceModule =
544 parseBitcodeFile(Buffer->get()->getMemBufferRef(), Module->getContext());
545 if (!LibDeviceModule)
546 POCL_ABORT("[CUDA] failed to load libdevice bitcode\n");
547
548 // Fix triple and data-layout of libdevice module.
549 (*LibDeviceModule)->setTargetTriple(Module->getTargetTriple());
550 (*LibDeviceModule)->setDataLayout(Module->getDataLayout());
551
552 // Link libdevice into module.
553 llvm::Linker Linker(*Module);
554 if (Linker.linkInModule(std::move(LibDeviceModule.get()))) {
555 POCL_ABORT("[CUDA] failed to link to libdevice\n");
556 }
557
558 llvm::legacy::PassManager Passes;
559
560 // Run internalize to mark all non-kernel functions as internal.
561 auto PreserveKernel = [=](const llvm::GlobalValue &GV) {
562 return GV.getName() == KernelName;
563 };
564 Passes.add(llvm::createInternalizePass(PreserveKernel));
565
566 // Add NVVM reflect module flags to set math options.
567 // TODO: Determine correct FTZ value from frontend compiler options.
568 llvm::LLVMContext &Context = Module->getContext();
569 llvm::Type *I32 = llvm::Type::getInt32Ty(Context);
570 llvm::Metadata *FourMD =
571 llvm::ValueAsMetadata::get(llvm::ConstantInt::getSigned(I32, 4));
572 llvm::Metadata *NameMD = llvm::MDString::get(Context, "nvvm-reflect-ftz");
573 llvm::Metadata *OneMD =
574 llvm::ConstantAsMetadata::get(llvm::ConstantInt::getSigned(I32, 1));
575 llvm::MDNode *ReflectFlag =
576 llvm::MDNode::get(Context, {FourMD, NameMD, OneMD});
577 Module->addModuleFlag(ReflectFlag);
578
579 // Run optimization passes to clean up unused functions etc.
580 llvm::PassManagerBuilder Builder;
581 Builder.OptLevel = 3;
582 Builder.SizeLevel = 0;
583 Builder.populateModulePassManager(Passes);
584
585 Passes.run(*Module);
586}

588// This transformation replaces each pointer argument in the specific address
589// space with an integer offset, and then inserts the necessary GEP+BitCast
590// instructions to calculate the new pointers from the provided base global
591// variable.
592void convertPtrArgsToOffsets(llvm::Module *Module, const char *KernelName,
593 unsigned AddrSpace, llvm::GlobalVariable *Base) {
594
595 llvm::LLVMContext &Context = Module->getContext();
596
597 llvm::Function *Function = Module->getFunction(KernelName);
598 if (!Function)
599 POCL_ABORT("[CUDA] ptx-gen: kernel function not found in module\n");
600
601 // Argument info for creating new function.
602 std::vector<llvm::Argument *> Arguments;
603 std::vector<llvm::Type *> ArgumentTypes;
604
605 llvm::ValueToValueMapTy VV;
606 std::vector<std::pair<llvm::Instruction *, llvm::Instruction *>> ToInsert;
607
608 // Loop over arguments.
609 bool NeedsArgOffsets = false;
610 for (auto &Arg : Function->args()) {
611 // Check for local memory pointer.
612 llvm::Type *ArgType = Arg.getType();
613 if (ArgType->isPointerTy() &&
614 ArgType->getPointerAddressSpace() == AddrSpace) {
615 NeedsArgOffsets = true;
616
617 // Create new argument for offset into shared memory allocation.
618 llvm::Type *I32ty = llvm::Type::getInt32Ty(Context);
619 llvm::Argument *Offset =
620 new llvm::Argument(I32ty, Arg.getName() + "_offset");
621 Arguments.push_back(Offset);
622 ArgumentTypes.push_back(I32ty);
623
624 // Insert GEP to add offset.
625 llvm::Value *Zero = llvm::ConstantInt::getSigned(I32ty, 0);
626 llvm::GetElementPtrInst *GEP =
627 llvm::GetElementPtrInst::Create(nullptr, Base, {Zero, Offset});
628
629 // Cast pointer to correct type.
630 llvm::BitCastInst *Cast = new llvm::BitCastInst(GEP, ArgType);
631
632 // Save these instructions to insert into new function later.
633 ToInsert.push_back({GEP, Cast});
634
635 // Map the old local memory argument to the result of this cast.
636 VV[&Arg] = Cast;
637 } else {
638 // No change to other arguments.
639 Arguments.push_back(&Arg);
640 ArgumentTypes.push_back(ArgType);
641 }
642 }
643
644 if (!NeedsArgOffsets)
645 return;
646
647 // Create new function with offsets instead of local memory pointers.
648 llvm::FunctionType *NewFunctionType =
649 llvm::FunctionType::get(Function->getReturnType(), ArgumentTypes, false);
650 llvm::Function *NewFunction = llvm::Function::Create(
651 NewFunctionType, Function->getLinkage(), Function->getName(), Module);
652 NewFunction->takeName(Function);
653
654 // Map function arguments.
655 std::vector<llvm::Argument *>::iterator OldArg;
656 llvm::Function::arg_iterator NewArg;
657 for (OldArg = Arguments.begin(), NewArg = NewFunction->arg_begin();
658 NewArg != NewFunction->arg_end(); NewArg++, OldArg++) {
659 NewArg->takeName(*OldArg);
660 if ((*OldArg)->getParent())
661 VV[*OldArg] = &*NewArg;
662 else {
663 // Manually replace new offset arguments.
664 (*OldArg)->replaceAllUsesWith(&*NewArg);
665 delete *OldArg;
666 }
667 }
668
669 // Clone function.
670 llvm::SmallVector<llvm::ReturnInst *, 1> RI;
671 llvm::CloneFunctionInto(NewFunction, Function, VV, true, RI);
672
673 // Insert offset instructions into new function.
674 for (auto Pair : ToInsert) {
675 Pair.first->insertBefore(&*NewFunction->begin()->begin());
676 Pair.second->insertAfter(Pair.first);
677 }
678
679 Function->eraseFromParent();
680}

682// CUDA doesn't allow constant pointer arguments, so we have to convert them to
683// offsets and manually add them to a global variable base pointer.
684void fixConstantMemArgs(llvm::Module *Module, const char *KernelName) {
685
686 // Calculate total size of automatic constant allocations.
687 size_t TotalAutoConstantSize = 0;
688 for (auto &GlobalVar : Module->globals()) {
689 if (GlobalVar.getType()->getPointerAddressSpace() == 4)
690 TotalAutoConstantSize += Module->getDataLayout().getTypeAllocSize(
691 GlobalVar.getInitializer()->getType());
692 }
693
694 // Create global variable for constant memory allocations.
695 // TODO: Does allocating the maximum amount have a penalty?
696 llvm::Type *ByteArrayType =
697 llvm::ArrayType::get(llvm::Type::getInt8Ty(Module->getContext()),
698 65536 - TotalAutoConstantSize);
699 llvm::GlobalVariable *ConstantMemBase = new llvm::GlobalVariable(
700 *Module, ByteArrayType, false, llvm::GlobalValue::InternalLinkage,
701 llvm::Constant::getNullValue(ByteArrayType), "_constant_memory_region_",
702 NULL, llvm::GlobalValue::NotThreadLocal, 4, false);
703
704 convertPtrArgsToOffsets(Module, KernelName, 4, ConstantMemBase);
705}

707// CUDA doesn't allow multiple local memory arguments or automatic variables, so
708// we have to create a single global variable for local memory allocations, and
709// then manually add offsets to it to get each individual local memory
710// allocation.
711void fixLocalMemArgs(llvm::Module *Module, const char *KernelName) {
712
713 // Create global variable for local memory allocations.
714 llvm::Type *ByteArrayType =
715 llvm::ArrayType::get(llvm::Type::getInt8Ty(Module->getContext()), 0);
716 llvm::GlobalVariable *SharedMemBase = new llvm::GlobalVariable(
717 *Module, ByteArrayType, false, llvm::GlobalValue::ExternalLinkage, NULL,
718 "_shared_memory_region_", NULL, llvm::GlobalValue::NotThreadLocal, 3,
719 false);
720
721 convertPtrArgsToOffsets(Module, KernelName, 3, SharedMemBase);
722}

831int pocl_cuda_get_ptr_arg_alignment(const char *BitcodeFilename,
832 const char *KernelName,
833 size_t *Alignments) {
834 // Create buffer for bitcode file.
835 llvm::ErrorOr<std::unique_ptr<llvm::MemoryBuffer>> Buffer =
836 llvm::MemoryBuffer::getFile(BitcodeFilename);
837 if (!Buffer) {
838 POCL_MSG_ERR("[CUDA] ptx-gen: failed to open bitcode file\n");
839 return 1;
840 }
841
842 // Load the LLVM bitcode module.
843 llvm::LLVMContext Context;
844 llvm::Expected<std::unique_ptr<llvm::Module>> Module =
845 parseBitcodeFile(Buffer->get()->getMemBufferRef(), Context);
846 if (!Module) {
847 POCL_MSG_ERR("[CUDA] ptx-gen: failed to load bitcode\n");
848 return 1;
849 }
850
851 // Get kernel function.
852 llvm::Function *Kernel = (*Module)->getFunction(KernelName);
853 if (!Kernel)
854 POCL_ABORT("[CUDA] kernel function not found in module\n");
855
856 // Calculate alignment for each argument.
857 const llvm::DataLayout &DL = (*Module)->getDataLayout();
858 for (auto &Arg : Kernel->args()) {
859 unsigned i = Arg.getArgNo();
860 llvm::Type *Type = Arg.getType();
861 if (!Type->isPointerTy())
862 Alignments[i] = 0;
863 else {
864 llvm::Type *ElemType = Type->getPointerElementType();
865 Alignments[i] = DL.getTypeAllocSize(ElemType);
866 }
867 }
868
869 return 0;
870}
