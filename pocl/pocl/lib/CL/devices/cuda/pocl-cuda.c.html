92static void
93pocl_cuda_abort_on_error (CUresult result, unsigned line, const char *func,
94 const char *code, const char *api)
95{
96 if (result != CUDA_SUCCESS)
97 {
98 const char *err_name;
99 const char *err_string;
100 cuGetErrorName (result, &err_name);
101 cuGetErrorString (result, &err_string);
102 POCL_MSG_PRINT2 (CUDA, func, line, "Error during %s\n", api);
103 POCL_ABORT ("%s: %s\n", err_name, err_string);
104 }
105}

107static int
108pocl_cuda_error (CUresult result, unsigned line, const char *func,
109 const char *code, const char *api)
110{
111 int err = (result != CUDA_SUCCESS);
112 if (err)
113 {
114 const char *err_name;
115 const char *err_string;
116 cuGetErrorName (result, &err_name);
117 cuGetErrorString (result, &err_string);
118 POCL_MSG_ERR ("CUDA error during %s. %s: %s\n", api, err_name, err_string);
119 }
120 return err;
121}

164cl_int
165pocl_cuda_init (unsigned j, cl_device_id dev, const char *parameters)
166{
167 CUresult result;
168 int ret = CL_SUCCESS;
169
170 if (dev->data)
171 return ret;
172
173 pocl_init_cpu_device_infos (dev);
174
175 dev->vendor = "NVIDIA Corporation";
176 dev->vendor_id = 0x10de; /* the PCIID for NVIDIA */
177
178 dev->type = CL_DEVICE_TYPE_GPU;
179 dev->address_bits = (sizeof (void *) * 8);
180
181 dev->llvm_target_triplet = (sizeof (void *) == 8) ? "nvptx64" : "nvptx";
182
183 dev->spmd = CL_TRUE;
184 dev->workgroup_pass = CL_FALSE;
185 dev->execution_capabilities = CL_EXEC_KERNEL;
186
187 dev->global_as_id = 1;
188 dev->local_as_id = 3;
189 dev->constant_as_id = 1;
190
191 /* TODO: Get images working */
192 dev->image_support = CL_FALSE;
193
194 dev->has_64bit_long = 1;
195
196 pocl_cuda_device_data_t *data = calloc (1, sizeof (pocl_cuda_device_data_t));
197 result = cuDeviceGet (&data->device, j);
198 if (CUDA_CHECK_ERROR (result, "cuDeviceGet"))
199 ret = CL_INVALID_DEVICE;
200
201 /* Get specific device name */
202 dev->long_name = dev->short_name = calloc (256, sizeof (char));
203
204 if (ret != CL_INVALID_DEVICE)
205 cuDeviceGetName (dev->long_name, 256, data->device);
206 else
207 snprintf (dev->long_name, 255, "Unavailable CUDA device #%d", j);
208
209 SETUP_DEVICE_CL_VERSION (CUDA_DEVICE_CL_VERSION_MAJOR,
210 CUDA_DEVICE_CL_VERSION_MINOR);
211
212 /* Get other device properties */
213 if (ret != CL_INVALID_DEVICE)
214 {
215 /* CUDA device attributes (as fetched by cuDeviceGetAttribute) are always (unsigned)
216 * integers, where the OpenCL counterparts are of a variety of (other) integer types.
217 * Fetch the values in an unsigned int and copy it over.
218 * We also OR all return values of cuDeviceGetAttribute, and at the end we will check
219 * if it's not CL_SUCCESS. We miss the exact line that failed this way, but it's
220 * faster than checking after each attribute fetch.
221 */
222 int value = 0;
223#define GET_CU_PROP(key, target) do { \
224 result |= cuDeviceGetAttribute (&value, CU_DEVICE_ATTRIBUTE_##key, data->device); \
225 target = value; \
226} while (0)
227
228 GET_CU_PROP (MAX_THREADS_PER_BLOCK, dev->max_work_group_size);
229 GET_CU_PROP (MAX_BLOCK_DIM_X, dev->max_work_item_sizes[0]);
230 GET_CU_PROP (MAX_BLOCK_DIM_Y, dev->max_work_item_sizes[1]);
231 GET_CU_PROP (MAX_BLOCK_DIM_Z, dev->max_work_item_sizes[2]);
232 GET_CU_PROP (MAX_SHARED_MEMORY_PER_BLOCK, dev->local_mem_size);
233 GET_CU_PROP (MULTIPROCESSOR_COUNT, dev->max_compute_units);
234 GET_CU_PROP (ECC_ENABLED, dev->error_correction_support);
235 GET_CU_PROP (INTEGRATED, dev->host_unified_memory);
236 GET_CU_PROP (TOTAL_CONSTANT_MEMORY, dev->max_constant_buffer_size);
237 GET_CU_PROP (CLOCK_RATE, dev->max_clock_frequency);
238 dev->max_clock_frequency /= 1000;
239 GET_CU_PROP (TEXTURE_ALIGNMENT, dev->mem_base_addr_align);
240 GET_CU_PROP (INTEGRATED, dev->host_unified_memory);
241 }
242 if (CUDA_CHECK_ERROR (result, "cuDeviceGetAttribute"))
243 ret = CL_INVALID_DEVICE;
244
245 dev->preferred_wg_size_multiple = 32;
246 dev->preferred_vector_width_char = 1;
247 dev->preferred_vector_width_short = 1;
248 dev->preferred_vector_width_int = 1;
249 dev->preferred_vector_width_long = 1;
250 dev->preferred_vector_width_float = 1;
251 dev->preferred_vector_width_double = 1;
252 dev->preferred_vector_width_half = 0;
253 dev->native_vector_width_char = 1;
254 dev->native_vector_width_short = 1;
255 dev->native_vector_width_int = 1;
256 dev->native_vector_width_long = 1;
257 dev->native_vector_width_float = 1;
258 dev->native_vector_width_double = 1;
259 dev->native_vector_width_half = 0;
260
261 dev->single_fp_config = CL_FP_ROUND_TO_NEAREST | CL_FP_ROUND_TO_ZERO
262 | CL_FP_ROUND_TO_INF | CL_FP_FMA | CL_FP_INF_NAN
263 | CL_FP_DENORM;
264 dev->double_fp_config = CL_FP_ROUND_TO_NEAREST | CL_FP_ROUND_TO_ZERO
265 | CL_FP_ROUND_TO_INF | CL_FP_FMA | CL_FP_INF_NAN
266 | CL_FP_DENORM;
267
268 dev->local_mem_type = CL_LOCAL;
269
270 /* Get GPU architecture name */
271 int sm_maj = 0, sm_min = 0;
272 if (ret != CL_INVALID_DEVICE)
273 {
274 cuDeviceGetAttribute (&sm_maj, CU_DEVICE_ATTRIBUTE_COMPUTE_CAPABILITY_MAJOR,
275 data->device);
276 cuDeviceGetAttribute (&sm_min, CU_DEVICE_ATTRIBUTE_COMPUTE_CAPABILITY_MINOR,
277 data->device);
278 }
279 char *gpu_arch = calloc (16, sizeof (char));
280 snprintf (gpu_arch, 16, "sm_%d%d", sm_maj, sm_min);
281 dev->llvm_cpu = pocl_get_string_option ("POCL_CUDA_GPU_ARCH", gpu_arch);
282 POCL_MSG_PRINT_INFO ("[CUDA] GPU architecture = %s\n", dev->llvm_cpu);
283
284 /* Find libdevice library */
285 if (findLibDevice (data->libdevice, dev->llvm_cpu))
286 {
287 if (ret != CL_INVALID_DEVICE)
288 {
289 POCL_MSG_ERR ("[CUDA] failed to find libdevice library\n");
290 dev->compiler_available = 0;
291 }
292 }
293
294 dev->device_side_printf = 0;
295
296 /* Create context */
297 if (ret != CL_INVALID_DEVICE)
298 {
299 result = cuCtxCreate (&data->context, CU_CTX_MAP_HOST, data->device);
300 if (CUDA_CHECK_ERROR (result, "cuCtxCreate"))
301 ret = CL_INVALID_DEVICE;
302 }
303
304 /* Create epoch event for timing info */
305 if (ret != CL_INVALID_DEVICE)
306 {
307 result = cuEventCreate (&data->epoch_event, CU_EVENT_DEFAULT);
308 CUDA_CHECK_ERROR (result, "cuEventCreate");
309
310 data->epoch = pocl_gettimemono_ns ();
311
312 result = cuEventRecord (data->epoch_event, 0);
313 result = cuEventSynchronize (data->epoch_event);
314 if (CUDA_CHECK_ERROR (result, "cuEventSynchronize"))
315 ret = CL_INVALID_DEVICE;
316 }
317
318 /* Get global memory size */
319 size_t memfree = 0, memtotal = 0;
320 if (ret != CL_INVALID_DEVICE)
321 result = cuMemGetInfo (&memfree, &memtotal);
322 dev->max_mem_alloc_size = max (memtotal / 4, 128 * 1024 * 1024);
323 dev->global_mem_size = memtotal;
324
325 dev->data = data;
326
327 POCL_INIT_LOCK (data->compile_lock);
328 return ret;
329}

331cl_int
332pocl_cuda_init_queue (cl_command_queue queue)
333{
334 cuCtxSetCurrent (((pocl_cuda_device_data_t *)queue->device->data)->context);
335
336 pocl_cuda_queue_data_t *queue_data
337 = calloc (1, sizeof (pocl_cuda_queue_data_t));
338 queue->data = queue_data;
339 queue_data->queue = queue;
340
341 CUresult result
342 = cuStreamCreate (&queue_data->stream, CU_STREAM_NON_BLOCKING);
343 if (CUDA_CHECK_ERROR (result, "cuStreamCreate"))
344 return CL_OUT_OF_RESOURCES;
345
346 queue_data->use_threads
347 = !pocl_get_bool_option ("POCL_CUDA_DISABLE_QUEUE_THREADS", 0);
348
349 if (queue_data->use_threads)
350 {
351 pthread_mutex_init (&queue_data->lock, NULL);
352 pthread_cond_init (&queue_data->pending_cond, NULL);
353 pthread_cond_init (&queue_data->running_cond, NULL);
354 int err = pthread_create (&queue_data->submit_thread, NULL,
355 pocl_cuda_submit_thread, queue_data);
356 if (err)
357 {
358 POCL_MSG_ERR ("[CUDA] Error creating submit thread: %d\n", err);
359 return CL_OUT_OF_RESOURCES;
360 }
361
362 err = pthread_create (&queue_data->finalize_thread, NULL,
363 pocl_cuda_finalize_thread, queue_data);
364 if (err)
365 {
366 POCL_MSG_ERR ("[CUDA] Error creating finalize thread: %d\n", err);
367 return CL_OUT_OF_RESOURCES;
368 }
369 }
370
371 return CL_SUCCESS;
372}

374void
375pocl_cuda_free_queue (cl_command_queue queue)
376{
377 pocl_cuda_queue_data_t *queue_data = (pocl_cuda_queue_data_t *)queue->data;
378
379 cuCtxSetCurrent (((pocl_cuda_device_data_t *)queue->device->data)->context);
380 cuStreamDestroy (queue_data->stream);
381
382 assert (queue_data->pending_queue == NULL);
383 assert (queue_data->running_queue == NULL);
384
385 /* Kill queue threads */
386 if (queue_data->use_threads)
387 {
388 pthread_mutex_lock (&queue_data->lock);
389 queue_data->queue = NULL;
390 pthread_cond_signal (&queue_data->pending_cond);
391 pthread_cond_signal (&queue_data->running_cond);
392 pthread_mutex_unlock (&queue_data->lock);
393 pthread_join (queue_data->submit_thread, NULL);
394 pthread_join (queue_data->finalize_thread, NULL);
395 }
396}

406unsigned int
407pocl_cuda_probe (struct pocl_device_ops *ops)
408{
409 int env_count = pocl_device_get_env_count (ops->device_name);
410
411 int probe_count = 0;
412 CUresult ret = cuInit (0);
413 if (ret == CUDA_SUCCESS)
414 {
415 ret = cuDeviceGetCount (&probe_count);
416 if (ret != CUDA_SUCCESS)
417 probe_count = 0;
418 }
419
420 /* If the user requested a specific number of CUDA devices,
421 * pretend we only have that many, if we can. If they requested
422 * more than there are, abort informing the user of the issue.
423 */
424 if (env_count >= 0)
425 {
426 if (env_count > probe_count)
427 POCL_ABORT ("[CUDA] %d devices requested, but only %d are available\n",
428 env_count, probe_count);
429 probe_count = env_count;
430 }
431
432 return probe_count;
433}

450cl_int
451pocl_cuda_alloc_mem_obj (cl_device_id device, cl_mem mem_obj, void *host_ptr)
452{
453 cuCtxSetCurrent (((pocl_cuda_device_data_t *)device->data)->context);
454
455 CUresult result;
456 void *b = NULL;
457
458 /* If memory for this global memory is not yet allocated -> do it */
459 if (mem_obj->device_ptrs[device->global_mem_id].mem_ptr == NULL)
460 {
461 cl_mem_flags flags = mem_obj->flags;
462
463 if (flags & CL_MEM_USE_HOST_PTR)
464 {
465#if defined __arm__
466 /* cuMemHostRegister is not supported on ARM.
467 * Allocate device memory and perform explicit copies
468 * before and after running a kernel */
469 result = cuMemAlloc ((CUdeviceptr *)&b, mem_obj->size);
470 CUDA_CHECK (result, "cuMemAlloc");
471#else
472 result = cuMemHostRegister (host_ptr, mem_obj->size,
473 CU_MEMHOSTREGISTER_DEVICEMAP);
474 if (result != CUDA_SUCCESS
475 && result != CUDA_ERROR_HOST_MEMORY_ALREADY_REGISTERED)
476 CUDA_CHECK (result, "cuMemHostRegister");
477 result = cuMemHostGetDevicePointer ((CUdeviceptr *)&b, host_ptr, 0);
478 CUDA_CHECK (result, "cuMemHostGetDevicePointer");
479#endif
480 }
481 else if (flags & CL_MEM_ALLOC_HOST_PTR)
482 {
483 result = cuMemHostAlloc (&mem_obj->mem_host_ptr, mem_obj->size,
484 CU_MEMHOSTREGISTER_DEVICEMAP);
485 CUDA_CHECK (result, "cuMemHostAlloc");
486 result = cuMemHostGetDevicePointer ((CUdeviceptr *)&b,
487 mem_obj->mem_host_ptr, 0);
488 CUDA_CHECK (result, "cuMemHostGetDevicePointer");
489 }
490 else
491 {
492 result = cuMemAlloc ((CUdeviceptr *)&b, mem_obj->size);
493 if (result != CUDA_SUCCESS)
494 {
495 const char *err;
496 cuGetErrorName (result, &err);
497 POCL_MSG_PRINT2 (CUDA, __FUNCTION__, __LINE__,
498 "-> Failed to allocate memory: %s\n", err);
499 return CL_MEM_OBJECT_ALLOCATION_FAILURE;
500 }
501 }
502
503 if (flags & CL_MEM_COPY_HOST_PTR)
504 {
505 result = cuMemcpyHtoD ((CUdeviceptr)b, host_ptr, mem_obj->size);
506 CUDA_CHECK (result, "cuMemcpyHtoD");
507
508 result = cuStreamSynchronize (0);
509 CUDA_CHECK (result, "cuStreamSynchronize");
510 }
511
512 mem_obj->device_ptrs[device->global_mem_id].mem_ptr = b;
513 mem_obj->device_ptrs[device->global_mem_id].global_mem_id
514 = device->global_mem_id;
515 }
516
517 /* Copy allocated global mem info to devices own slot */
518 mem_obj->device_ptrs[device->dev_id]
519 = mem_obj->device_ptrs[device->global_mem_id];
520
521 return CL_SUCCESS;
522}

524void
525pocl_cuda_free (cl_device_id device, cl_mem mem_obj)
526{
527 cuCtxSetCurrent (((pocl_cuda_device_data_t *)device->data)->context);
528
529 if (mem_obj->flags & CL_MEM_ALLOC_HOST_PTR)
530 {
531 cuMemFreeHost (mem_obj->mem_host_ptr);
532 mem_obj->mem_host_ptr = NULL;
533 }
534 else if (mem_obj->flags & CL_MEM_USE_HOST_PTR)
535 {
536 cuMemHostUnregister (mem_obj->mem_host_ptr);
537 mem_obj->mem_host_ptr = NULL;
538 }
539 else
540 {
541 void *ptr = mem_obj->device_ptrs[device->dev_id].mem_ptr;
542 cuMemFree ((CUdeviceptr)ptr);
543 }
544}

574void
575pocl_cuda_submit_copy (CUstream stream, void*__restrict__ src_mem_ptr,
576 size_t src_offset, void *__restrict__ dst_mem_ptr,
577 size_t dst_offset, size_t cb)
578{
579 void *src_ptr = src_mem_ptr + src_offset;
580 void *dst_ptr = dst_mem_ptr + dst_offset;
581
582 if (src_ptr == dst_ptr)
583 return;
584
585 CUresult result;
586 POCL_MSG_PRINT_CUDA ("cuMemcpyDtoDAsync %p -> %p / %zu B \n", src_ptr, dst_ptr, cb);
587 result = cuMemcpyDtoDAsync ((CUdeviceptr)dst_ptr, (CUdeviceptr)src_ptr,
588 cb, stream);
589 CUDA_CHECK (result, "cuMemcpyDtoDAsync");
590}

592void
593pocl_cuda_submit_read_rect (CUstream stream, void *__restrict__ const host_ptr,
594 void *__restrict__ const device_ptr,
595 const size_t *__restrict__ const buffer_origin,
596 const size_t *__restrict__ const host_origin,
597 const size_t *__restrict__ const region,
598 size_t const buffer_row_pitch,
599 size_t const buffer_slice_pitch,
600 size_t const host_row_pitch,
601 size_t const host_slice_pitch)
602{
603 CUDA_MEMCPY3D params = { 0 };
604
605 POCL_MSG_PRINT_CUDA ("cuMemcpy3D / READ_RECT %p -> %p \n", device_ptr, host_ptr);
606
607 params.WidthInBytes = region[0];
608 params.Height = region[1];
609 params.Depth = region[2];
610
611 params.dstMemoryType = CU_MEMORYTYPE_HOST;
612 params.dstHost = host_ptr;
613 params.dstXInBytes = host_origin[0];
614 params.dstY = host_origin[1];
615 params.dstZ = host_origin[2];
616 params.dstPitch = host_row_pitch;
617 params.dstHeight = host_slice_pitch / host_row_pitch;
618
619 params.srcMemoryType = CU_MEMORYTYPE_DEVICE;
620 params.srcDevice = (CUdeviceptr)device_ptr;
621 params.srcXInBytes = buffer_origin[0];
622 params.srcY = buffer_origin[1];
623 params.srcZ = buffer_origin[2];
624 params.srcPitch = buffer_row_pitch;
625 params.srcHeight = buffer_slice_pitch / buffer_row_pitch;
626
627 CUresult result = cuMemcpy3DAsync (&params, stream);
628 CUDA_CHECK (result, "cuMemcpy3DAsync");
629}

631void
632pocl_cuda_submit_write_rect (CUstream stream,
633 const void *__restrict__ const host_ptr,
634 void *__restrict__ const device_ptr,
635 const size_t *__restrict__ const buffer_origin,
636 const size_t *__restrict__ const host_origin,
637 const size_t *__restrict__ const region,
638 size_t const buffer_row_pitch,
639 size_t const buffer_slice_pitch,
640 size_t const host_row_pitch,
641 size_t const host_slice_pitch)
642{
643 CUDA_MEMCPY3D params = { 0 };
644
645 POCL_MSG_PRINT_CUDA ("cuMemcpy3D / WRITE_RECT %p -> %p \n", host_ptr, device_ptr);
646
647 params.WidthInBytes = region[0];
648 params.Height = region[1];
649 params.Depth = region[2];
650
651 params.srcMemoryType = CU_MEMORYTYPE_HOST;
652 params.srcHost = host_ptr;

758static pocl_cuda_kernel_data_t *
759load_or_generate_kernel (cl_kernel kernel, cl_device_id device,
760 int has_offsets, unsigned device_i,
761 _cl_command_node *command, int specialized)
762{
763 CUresult result;
764 pocl_kernel_metadata_t *meta = kernel->meta;
765 /* Check if we already have a compiled kernel function */
766 pocl_cuda_kernel_data_t *kdata
767 = (pocl_cuda_kernel_data_t *)meta->data[device_i];
768 if (kdata)
769 {
770 if ((has_offsets && kdata->kernel_offsets)
771 || (!has_offsets && kdata->kernel))
772 return kdata;
773 }
774 else
775 {
776 /* TODO: when can we release this? */
777 kdata = meta->data[device_i]
778 = (void *)calloc (1, sizeof (pocl_cuda_kernel_data_t));
779 }
780
781 pocl_cuda_device_data_t *ddata = (pocl_cuda_device_data_t *)device->data;
782 cuCtxSetCurrent (ddata->context);
783
784 POCL_LOCK(ddata->compile_lock);
785
786 /* Generate the parallel bitcode file linked with the kernel library */
787 int error = pocl_llvm_generate_workgroup_function (device_i, device, kernel,
788 command, specialized);
789 if (error)
790 {
791 POCL_MSG_PRINT_GENERAL ("pocl_llvm_generate_workgroup_function() failed"
792 " for kernel %s\n", kernel->name);
793 assert (error == 0);
794 }
795
796 char bc_filename[POCL_FILENAME_LENGTH];
797 pocl_cache_work_group_function_path (bc_filename, kernel->program, device_i,
798 kernel, command, specialized);
799
800 char ptx_filename[POCL_FILENAME_LENGTH];
801 strcpy (ptx_filename, bc_filename);
802 strncat (ptx_filename, ".ptx", POCL_FILENAME_LENGTH - 1);
803
804 if (!pocl_exists (ptx_filename))
805 {
806 /* Generate PTX from LLVM bitcode */
807 if (pocl_ptx_gen (bc_filename, ptx_filename, kernel->name,
808 device->llvm_cpu,
809 ((pocl_cuda_device_data_t *)device->data)->libdevice,
810 has_offsets))
811 POCL_ABORT ("pocl-cuda: failed to generate PTX\n");
812 }
813
814 /* Load PTX module */
815 /* TODO: When can we unload the module? */
816 CUmodule module;
817 result = cuModuleLoad (&module, ptx_filename);
818 CUDA_CHECK (result, "cuModuleLoad");
819
820 /* Get kernel function */
821 CUfunction function;
822 result = cuModuleGetFunction (&function, module, kernel->name);
823 CUDA_CHECK (result, "cuModuleGetFunction");
824
825 /* Get pointer aligment */
826 if (!kdata->alignments)
827 {
828 kdata->alignments
829 = calloc (meta->num_args + meta->num_locals + 4, sizeof (size_t));
830 pocl_cuda_get_ptr_arg_alignment (bc_filename, kernel->name,
831 kdata->alignments);
832 }
833
834 if (has_offsets)
835 {
836 kdata->module_offsets = module;
837 kdata->kernel_offsets = function;
838 }
839 else
840 {
841 kdata->module = module;
842 kdata->kernel = function;
843 }
844
845 POCL_UNLOCK (ddata->compile_lock);
846
847 return kdata;
848}

858void
859pocl_cuda_submit_kernel (CUstream stream, _cl_command_node *cmd,
860 cl_device_id device, cl_event event)
861{
862 _cl_command_run run = cmd->command.run;
863 cl_kernel kernel = run.kernel;
864 pocl_argument *arguments = run.arguments;
865 struct pocl_context pc = run.pc;
866 pocl_kernel_metadata_t *meta = kernel->meta;
867
868 /* Check if we need to handle global work offsets */
869 int has_offsets =
870 (pc.global_offset[0] || pc.global_offset[1] || pc.global_offset[2]);
871
872 /* Get kernel function */
873 pocl_cuda_kernel_data_t *kdata = load_or_generate_kernel (
874 kernel, device, has_offsets, run.device_i, cmd, 1);
875 CUmodule module = has_offsets ? kdata->module_offsets : kdata->module;
876 CUfunction function = has_offsets ? kdata->kernel_offsets : kdata->kernel;
877
878 /* Prepare kernel arguments */
879 void *null = NULL;
880 unsigned sharedMemBytes = 0;
881 void *params[meta->num_args + meta->num_locals + 4];
882 unsigned sharedMemOffsets[meta->num_args + meta->num_locals];
883 unsigned constantMemBytes = 0;
884 unsigned constantMemOffsets[meta->num_args];
885 unsigned globalOffsets[3];
886
887 /* Get handle to constant memory buffer */
888 size_t constant_mem_size;
889 CUdeviceptr constant_mem_base = 0;
890 cuModuleGetGlobal (&constant_mem_base, &constant_mem_size, module,
891 "_constant_memory_region_");
892
893 CUresult result;
894 unsigned i;
895 for (i = 0; i < meta->num_args; i++)
896 {
897 pocl_argument_type type = meta->arg_info[i].type;
898 switch (type)
899 {
900 case POCL_ARG_TYPE_NONE:
901 params[i] = arguments[i].value;
902 break;
903 case POCL_ARG_TYPE_POINTER:
904 {
905 if (ARG_IS_LOCAL (meta->arg_info[i]))
906 {
907 size_t size = arguments[i].size;
908 size_t align = kdata->alignments[i];
909
910 /* Pad offset to align memory */
911 if (sharedMemBytes % align)
912 sharedMemBytes += align - (sharedMemBytes % align);
913
914 sharedMemOffsets[i] = sharedMemBytes;
915 params[i] = sharedMemOffsets + i;
916
917 sharedMemBytes += size;
918 }
919 else if (meta->arg_info[i].address_qualifier
920 == CL_KERNEL_ARG_ADDRESS_CONSTANT)
921 {
922 assert (constant_mem_base);
923
924 /* Get device pointer */
925 cl_mem mem = *(void **)arguments[i].value;
926 CUdeviceptr src
927 = (CUdeviceptr)mem->device_ptrs[device->dev_id].mem_ptr
928 + arguments[i].offset;
929
930 size_t align = kdata->alignments[i];
931 if (constantMemBytes % align)
932 {
933 constantMemBytes += align - (constantMemBytes % align);
934 }
935
936 /* Copy to constant buffer at current offset */
937 result
938 = cuMemcpyDtoDAsync (constant_mem_base + constantMemBytes,
939 src, mem->size, stream);
940 CUDA_CHECK (result, "cuMemcpyDtoDAsync");
941
942 constantMemOffsets[i] = constantMemBytes;
943 params[i] = constantMemOffsets + i;
944
945 constantMemBytes += mem->size;
946 }
947 else
948 {
949 if (arguments[i].value)
950 {
951 cl_mem mem = *(void **)arguments[i].value;
952 params[i] = &mem->device_ptrs[device->dev_id].mem_ptr
953 + arguments[i].offset;
954
955#if defined __arm__
956 /* On ARM with USE_HOST_PTR, perform explicit copy to
957 * device */
958 if (mem->flags & CL_MEM_USE_HOST_PTR)
959 {
960 cuMemcpyHtoD (*(CUdeviceptr *)(params[i]),
961 mem->mem_host_ptr, mem->size);
962 cuStreamSynchronize (0);
963 }
964#endif
965 }
966 else
967 {
968 params[i] = &null;
969 }
970 }
971 break;
972 }
973 case POCL_ARG_TYPE_IMAGE:
974 case POCL_ARG_TYPE_SAMPLER:
975 POCL_ABORT ("Unhandled argument type for CUDA\n");
976 break;
977 }
978 }
979
980 if (constantMemBytes > constant_mem_size)
981 POCL_ABORT ("[CUDA] Total constant buffer size %u exceeds %lu allocated\n",
982 constantMemBytes, constant_mem_size);
983
984 unsigned arg_index = meta->num_args;
985
986 /* Deal with automatic local allocations */
987 /* TODO: Would be better to remove arguments and make these static GEPs */
988 for (i = 0; i < meta->num_locals; ++i, ++arg_index)
989 {
990 size_t size = meta->local_sizes[i];
991 size_t align = kdata->alignments[arg_index];
992
993 /* Pad offset to align memory */
994 if (sharedMemBytes % align)
995 sharedMemBytes += align - (sharedMemBytes % align);
996
997 sharedMemOffsets[arg_index] = sharedMemBytes;
998 sharedMemBytes += size;
999 params[arg_index] = sharedMemOffsets + arg_index;
1000 }
1001
1002 /* Add global work dimensionality */
1003 params[arg_index++] = &pc.work_dim;
1004
1005 /* Add global offsets if necessary */
1006 if (has_offsets)
1007 {
1008 globalOffsets[0] = pc.global_offset[0];
1009 globalOffsets[1] = pc.global_offset[1];
1010 globalOffsets[2] = pc.global_offset[2];
1011 params[arg_index++] = globalOffsets + 0;
1012 params[arg_index++] = globalOffsets + 1;
1013 params[arg_index++] = globalOffsets + 2;
1014 }
1015
1016 /* Launch kernel */
1017 result = cuLaunchKernel (function, pc.num_groups[0], pc.num_groups[1],
1018 pc.num_groups[2], pc.local_size[0],
1019 pc.local_size[1], pc.local_size[2], sharedMemBytes,
1020 stream, params, NULL);
1021 CUDA_CHECK (result, "cuLaunchKernel");
1022}

1024void
1025pocl_cuda_submit_node (_cl_command_node *node, cl_command_queue cq, int locked)
1026{
1027 CUresult result;
1028 CUstream stream = ((pocl_cuda_queue_data_t *)cq->data)->stream;
1029
1030 if (!locked)
1031 POCL_LOCK_OBJ (node->event);
1032
1033 pocl_cuda_event_data_t *event_data
1034 = (pocl_cuda_event_data_t *)node->event->data;
1035
1036 /* Process event dependencies */
1037 event_node *dep = NULL;
1038 LL_FOREACH (node->event->wait_list, dep)
1039 {
1040 /* If it is in the process of completing, just skip it */
1041 if (dep->event->status <= CL_COMPLETE)
1042 continue;
1043
1044 /* Add CUDA event dependency */
1045 if (dep->event->command_type != CL_COMMAND_USER
1046 && dep->event->queue->device->ops == cq->device->ops)
1047 {
1048 /* Block stream on event, but only for different queues */
1049 if (dep->event->queue != node->event->queue)
1050 {
1051 pocl_cuda_event_data_t *dep_data
1052 = (pocl_cuda_event_data_t *)dep->event->data;
1053
1054 /* Wait until dependency has finished being submitted */
1055 while (!dep_data->events_ready)
1056 ;
1057
1058 result = cuStreamWaitEvent (stream, dep_data->end, 0);
1059 CUDA_CHECK (result, "cuStreamWaitEvent");
1060 }
1061 }
1062 else
1063 {
1064 if (!((pocl_cuda_queue_data_t *)cq->data)->use_threads)
1065 POCL_ABORT (
1066 "Can't handle non-CUDA dependencies without queue threads\n");
1067
1068 event_data->num_ext_events++;
1069 }
1070 }
1071
1072 /* Wait on flag for external events */
1073 if (event_data->num_ext_events)
1074 {
1075 CUdeviceptr dev_ext_event_flag;
1076 result = cuMemHostAlloc ((void **)&event_data->ext_event_flag, 4,
1077 CU_MEMHOSTALLOC_DEVICEMAP);
1078 CUDA_CHECK (result, "cuMemAllocHost");
1079
1080 *event_data->ext_event_flag = 0;
1081
1082 result = cuMemHostGetDevicePointer (&dev_ext_event_flag,
1083 event_data->ext_event_flag, 0);
1084 CUDA_CHECK (result, "cuMemHostGetDevicePointer");
1085 result = cuStreamWaitValue32 (stream, dev_ext_event_flag, 1,
1086 CU_STREAM_WAIT_VALUE_GEQ);
1087 CUDA_CHECK (result, "cuStreamWaitValue32");
1088 }
1089
1090 /* Create and record event for command start if profiling enabled */
1091 if (cq->properties & CL_QUEUE_PROFILING_ENABLE)
1092 {
1093 result = cuEventCreate (&event_data->start, CU_EVENT_DEFAULT);
1094 CUDA_CHECK (result, "cuEventCreate");
1095 result = cuEventRecord (event_data->start, stream);
1096 CUDA_CHECK (result, "cuEventRecord");
1097 }
1098
1099 pocl_update_event_submitted (node->event);
1100
1101 POCL_UNLOCK_OBJ (node->event);
1102
1103 cl_event event = node->event;
1104 cl_device_id dev = node->device;
1105 _cl_command_t *cmd = &node->command;
1106
1107 switch (node->type)
1108 {
1109 case CL_COMMAND_READ_BUFFER:
1110 pocl_cuda_submit_read (
1111 stream, cmd->read.dst_host_ptr, cmd->read.src_mem_id->mem_ptr,
1112 node->command.read.offset, node->command.read.size);
1113 break;
1114 case CL_COMMAND_WRITE_BUFFER:
1115 pocl_cuda_submit_write (
1116 stream, cmd->write.src_host_ptr, cmd->write.dst_mem_id->mem_ptr,
1117 node->command.write.offset, node->command.write.size);
1118 break;
1119 case CL_COMMAND_COPY_BUFFER:
1120 {
1121 pocl_cuda_submit_copy (
1122 stream, cmd->copy.src_mem_id->mem_ptr, cmd->copy.src_offset,
1123 cmd->copy.dst_mem_id->mem_ptr, cmd->copy.dst_offset, cmd->copy.size);
1124 break;
1125 }
1126 case CL_COMMAND_READ_BUFFER_RECT:
1127 pocl_cuda_submit_read_rect (
1128 stream,
1129 cmd->read_rect.dst_host_ptr,
1130 cmd->read_rect.src_mem_id->mem_ptr,
1131 cmd->read_rect.buffer_origin,
1132 cmd->read_rect.host_origin,
1133 cmd->read_rect.region,
1134 cmd->read_rect.buffer_row_pitch,
1135 cmd->read_rect.buffer_slice_pitch,
1136 cmd->read_rect.host_row_pitch,
1137 cmd->read_rect.host_slice_pitch);
1138 break;
1139 case CL_COMMAND_WRITE_BUFFER_RECT:
1140 pocl_cuda_submit_write_rect (
1141 stream,
1142 cmd->write_rect.src_host_ptr,
1143 cmd->write_rect.dst_mem_id->mem_ptr,
1144 cmd->write_rect.buffer_origin,
1145 cmd->write_rect.host_origin,
1146 cmd->write_rect.region,
1147 cmd->read_rect.buffer_row_pitch,
1148 cmd->read_rect.buffer_slice_pitch,
1149 cmd->read_rect.host_row_pitch,
1150 cmd->read_rect.host_slice_pitch);
1151 break;
1152 case CL_COMMAND_COPY_BUFFER_RECT:
1153 {
1154 pocl_cuda_submit_copy_rect (
1155 stream, dev,
1156 cmd->copy_rect.src_mem_id->mem_ptr,
1157 cmd->copy_rect.dst_mem_id->mem_ptr,
1158 cmd->copy_rect.src_origin,
1159 cmd->copy_rect.dst_origin,
1160 cmd->copy_rect.region,
1161 cmd->copy_rect.src_row_pitch,
1162 cmd->copy_rect.src_slice_pitch,
1163 cmd->copy_rect.dst_row_pitch,
1164 cmd->copy_rect.dst_slice_pitch);
1165 break;
1166 }
1167 case CL_COMMAND_MIGRATE_MEM_OBJECTS:
1168 {
1169 size_t i;
1170 for (i = 0; i < cmd->migrate.num_mem_objects; i++)
1171 {
1172// cl_device_id src_dev = cmd->migrate.source_devices[i];
1173// cl_device_id dst_dev = cq->device;
1174 cl_mem buf = cmd->migrate.mem_objects[i];
1175// if (!src_dev)
1176// src_dev = dst_dev;
1177 // TODO
1178 void *ptr = buf->device_ptrs[dev->dev_id].mem_ptr;
1179 pocl_cuda_submit_copy (stream, ptr, 0, ptr, 0, buf->size);
1180 }
1181 break;
1182 }
1183 case CL_COMMAND_MAP_BUFFER:
1184 {
1185 cl_mem buffer = event->mem_objs[0];
1186 if (!(buffer->flags & (CL_MEM_USE_HOST_PTR | CL_MEM_ALLOC_HOST_PTR)))
1187 pocl_cuda_submit_map_mem (
1188 stream,
1189 cmd->map.mem_id,
1190 cmd->map.mapping->offset,
1191 cmd->map.mapping->size,
1192 cmd->map.mapping->host_ptr);
1193 POCL_LOCK_OBJ (buffer);
1194 buffer->map_count++;
1195 POCL_UNLOCK_OBJ (buffer);
1196 break;
1197 }
1198 case CL_COMMAND_UNMAP_MEM_OBJECT:
1199 {
1200 cl_mem buffer = event->mem_objs[0];
1201 assert (buffer->is_image == CL_FALSE);
1202 pocl_cuda_submit_unmap_mem (
1203 stream,
1204 cmd->unmap.mem_id,
1205 cmd->unmap.mapping->offset,
1206 cmd->unmap.mapping->size,
1207 cmd->unmap.mapping->host_ptr);
1208 break;
1209 }
1210 case CL_COMMAND_NDRANGE_KERNEL:
1211 pocl_cuda_submit_kernel (stream, node, node->device, node->event);
1212 break;
1213
1214 case CL_COMMAND_MARKER:
1215 case CL_COMMAND_BARRIER:
1216 break;
1217
1218 case CL_COMMAND_FILL_BUFFER:
1219 case CL_COMMAND_READ_IMAGE:
1220 case CL_COMMAND_WRITE_IMAGE:
1221 case CL_COMMAND_COPY_IMAGE:
1222 case CL_COMMAND_COPY_BUFFER_TO_IMAGE:
1223 case CL_COMMAND_COPY_IMAGE_TO_BUFFER:
1224 case CL_COMMAND_FILL_IMAGE:
1225 case CL_COMMAND_MAP_IMAGE:
1226 case CL_COMMAND_NATIVE_KERNEL:
1227 case CL_COMMAND_SVM_FREE:
1228 case CL_COMMAND_SVM_MAP:
1229 case CL_COMMAND_SVM_UNMAP:
1230 case CL_COMMAND_SVM_MEMCPY:
1231 case CL_COMMAND_SVM_MEMFILL:
1232 default:
1233 POCL_ABORT_UNIMPLEMENTED (pocl_command_to_str (node->type));
1234 break;
1235 }
1236
1237 /* Create and record event for command end */
1238 if (cq->properties & CL_QUEUE_PROFILING_ENABLE)
1239 result = cuEventCreate (&event_data->end, CU_EVENT_DEFAULT);
1240 else
1241 result = cuEventCreate (&event_data->end, CU_EVENT_DISABLE_TIMING);
1242 CUDA_CHECK (result, "cuEventCreate");
1243 result = cuEventRecord (event_data->end, stream);
1244 CUDA_CHECK (result, "cuEventRecord");
1245
1246 event_data->events_ready = 1;
1247}

1249void
1250pocl_cuda_submit (_cl_command_node *node, cl_command_queue cq)
1251{
1252 /* Allocate CUDA event data */
1253 node->event->data
1254 = (pocl_cuda_event_data_t *)calloc (1, sizeof (pocl_cuda_event_data_t));
1255
1256 if (((pocl_cuda_queue_data_t *)cq->data)->use_threads)
1257 {
1258 /* Add command to work queue */
1259 POCL_UNLOCK_OBJ (node->event);
1260 pocl_cuda_queue_data_t *queue_data = (pocl_cuda_queue_data_t *)cq->data;
1261 pthread_mutex_lock (&queue_data->lock);
1262 DL_APPEND (queue_data->pending_queue, node);
1263 pthread_cond_signal (&queue_data->pending_cond);
1264 pthread_mutex_unlock (&queue_data->lock);
1265 }
1266 else
1267 {
1268 /* Submit command in this thread */
1269 cuCtxSetCurrent (((pocl_cuda_device_data_t *)cq->device->data)->context);
1270 pocl_cuda_submit_node (node, cq, 1);
1271 }
1272}

1274void
1275pocl_cuda_notify (cl_device_id device, cl_event event, cl_event finished)
1276{
1277 /* Ignore CUDA device events, we've already handled these dependencies */
1278 if (finished->queue && finished->queue->device->ops == device->ops)
1279 return;
1280
1281 if (event->status == CL_QUEUED)
1282 return;
1283
1284 pocl_cuda_event_data_t *event_data = (pocl_cuda_event_data_t *)event->data;
1285
1286 assert (event_data);
1287 assert (event_data->num_ext_events > 0);
1288 assert (event_data->ext_event_flag);
1289
1290 /* If dependency failed, so should we */
1291 /* TODO: This isn't true if this is an implicit dependency */
1292 if (finished->status < 0)
1293 event->status = -1;
1294
1295 /* Decrement external event counter */
1296 /* Trigger flag if none left */
1297 if (!--event_data->num_ext_events)
1298 *event_data->ext_event_flag = 1;
1299}

1308pocl_cuda_finalize_command (cl_device_id device, cl_event event)
1309{
1310 CUresult result;
1311 pocl_cuda_event_data_t *event_data = (pocl_cuda_event_data_t *)event->data;
1312
1313 /* Wait for command to finish */
1314 cuCtxSetCurrent (((pocl_cuda_device_data_t *)device->data)->context);
1315 result = cuEventSynchronize (event_data->end);
1316 CUDA_CHECK (result, "cuEventSynchronize");
1317
1318 /* Clean up mapped memory allocations */
1319 if (event->command_type == CL_COMMAND_UNMAP_MEM_OBJECT)
1320 {
1321 cl_mem buffer = event->mem_objs[0];
1322 mem_mapping_t *mapping = event->command->command.unmap.mapping;
1323 if (mapping->host_ptr
1324 && !(buffer->flags
1325 & (CL_MEM_USE_HOST_PTR | CL_MEM_ALLOC_HOST_PTR)))
1326 free (mapping->host_ptr);
1327
1328 POCL_LOCK_OBJ (buffer);
1329 DL_DELETE (buffer->mappings, mapping);
1330 buffer->map_count--;
1331 POCL_UNLOCK_OBJ (buffer);
1332 }
1333
1334 if (event->command_type == CL_COMMAND_NDRANGE_KERNEL
1335 || event->command_type == CL_COMMAND_TASK)
1336 {
1337#if defined __arm__
1338 /* On ARM with USE_HOST_PTR, perform explict copies back from device */
1339 cl_kernel kernel = event->command.run.kernel;
1340 pocl_argument *arguments = event->command.run.arguments;
1341 unsigned i;
1342 for (i = 0; i < meta->num_args; i++)
1343 {
1344 if (meta->arg_info[i].type == POCL_ARG_TYPE_POINTER)
1345 {
1346 if (!ARG_IS_LOCAL (meta->arg_info[i]) && arguments[i].value)
1347 {
1348 cl_mem mem = *(void **)arguments[i].value;
1349 if (mem->flags & CL_MEM_USE_HOST_PTR)
1350 {
1351 CUdeviceptr ptr
1352 = (CUdeviceptr)mem->device_ptrs[device->dev_id]
1353 .mem_ptr;
1354 cuMemcpyDtoH (mem->mem_host_ptr, ptr, mem->size);
1355 cuStreamSynchronize (0);
1356 }
1357 }
1358 }
1359 }
1360#endif
1361
1362 pocl_ndrange_node_cleanup (event->command);
1363 }
1364 else
1365 {
1366 pocl_mem_manager_free_command (event->command);
1367 }
1368
1369 /* Handle failed events */
1370
1371
1372 pocl_update_event_running (event);
1373 if (event->status < 0)
1374 pocl_update_event_failed (event);
1375 else
1376 POCL_UPDATE_EVENT_COMPLETE_MSG (event, "CUDA event");
1377}

1379void
1380pocl_cuda_update_event (cl_device_id device, cl_event event)
1381{
1382 if ((event->status == CL_COMPLETE)
1383 && (event->queue->properties & CL_QUEUE_PROFILING_ENABLE))
1384 {
1385 /* Update timing info with CUDA event timers if profiling enabled */
1386 /* CUDA doesn't provide a way to get event timestamps directly,
1387 * only the elapsed time between two events. We use the elapsed
1388 * time from the epoch event enqueued on device creation to get
1389 * the actual timestamps.
1390 *
1391 * Since the CUDA timer resolution is lower than the host timer,
1392 * this can sometimes result in the start time being before the
1393 * submit time, so we use max() to ensure the timestamps are
1394 * sane. */
1395
1396 float diff;
1397 CUresult result;
1398 pocl_cuda_event_data_t *event_data
1399 = (pocl_cuda_event_data_t *)event->data;
1400 cl_ulong epoch = ((pocl_cuda_device_data_t *)device->data)->epoch;
1401
1402 result = cuEventElapsedTime (
1403 &diff, ((pocl_cuda_device_data_t *)device->data)->epoch_event,
1404 event_data->start);
1405 CUDA_CHECK (result, "cuEventElapsedTime");
1406 event->time_start = (cl_ulong) (epoch + diff * 1e6);
1407 event->time_start = max (event->time_start, event->time_submit + 1);
1408
1409 result = cuEventElapsedTime (
1410 &diff, ((pocl_cuda_device_data_t *)device->data)->epoch_event,
1411 event_data->end);
1412 CUDA_CHECK (result, "cuEventElapsedTime");
1413 event->time_end = (cl_ulong) (epoch + diff * 1e6);
1414 event->time_end = max (event->time_end, event->time_start + 1);
1415 }
1416}

1482void *
1483pocl_cuda_submit_thread (void *data)
1484{
1485 pocl_cuda_queue_data_t *queue_data = (pocl_cuda_queue_data_t *)data;
1486
1487 cl_command_queue queue = queue_data->queue;
1488 if (queue)
1489 cuCtxSetCurrent (
1490 ((pocl_cuda_device_data_t *)queue->device->data)->context);
1491 else
1492 /* This queue has already been released */
1493 return NULL;
1494
1495 while (1)
1496 {
1497 /* Attempt to get next command from work queue */
1498 _cl_command_node *node = NULL;
1499 pthread_mutex_lock (&queue_data->lock);
1500 if (!queue_data->queue)
1501 {
1502 pthread_mutex_unlock (&queue_data->lock);
1503 break;
1504 }
1505 if (!queue_data->pending_queue)
1506 {
1507 pthread_cond_wait (&queue_data->pending_cond, &queue_data->lock);
1508 }
1509 if (queue_data->pending_queue)
1510 {
1511 node = queue_data->pending_queue;
1512 DL_DELETE (queue_data->pending_queue, node);
1513 }
1514 pthread_mutex_unlock (&queue_data->lock);
1515
1516 /* Submit command, if we found one */
1517 if (node)
1518 {
1519 pocl_cuda_submit_node (node, queue_data->queue, 0);
1520
1521 /* Add command to running queue */
1522 pthread_mutex_lock (&queue_data->lock);
1523 DL_APPEND (queue_data->running_queue, node);
1524 pthread_cond_signal (&queue_data->running_cond);
1525 pthread_mutex_unlock (&queue_data->lock);
1526 }
1527 }
1528
1529 return NULL;
1530}

1532void *
1533pocl_cuda_finalize_thread (void *data)
1534{
1535 pocl_cuda_queue_data_t *queue_data = (pocl_cuda_queue_data_t *)data;
1536
1537 cl_command_queue queue = queue_data->queue;
1538 if (queue)
1539 cuCtxSetCurrent (
1540 ((pocl_cuda_device_data_t *)queue->device->data)->context);
1541 else
1542 /* This queue has already been released */
1543 return NULL;
1544
1545 while (1)
1546 {
1547 /* Attempt to get next node from running queue */
1548 _cl_command_node *node = NULL;
1549 pthread_mutex_lock (&queue_data->lock);
1550 if (!queue_data->queue)
1551 {
1552 pthread_mutex_unlock (&queue_data->lock);
1553 break;
1554 }
1555 if (!queue_data->running_queue)
1556 {
1557 pthread_cond_wait (&queue_data->running_cond, &queue_data->lock);
1558 }
1559 if (queue_data->running_queue)
1560 {
1561 node = queue_data->running_queue;
1562 DL_DELETE (queue_data->running_queue, node);
1563 }
1564 pthread_mutex_unlock (&queue_data->lock);
1565
1566 /* Wait for command to finish, if we found one */
1567 if (node)
1568 pocl_cuda_finalize_command (queue->device, node->event);
1569 }
1570
1571 return NULL;
