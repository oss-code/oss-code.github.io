24#ifndef POCL_PTX_GEN_H
25#define POCL_PTX_GEN_H
26
27#include "config.h"
28
29#ifdef __cplusplus
30extern "C"
31{
32#endif
33
34#ifndef PATH_MAX
35#define PATH_MAX 4096
36#endif
37
38/* Search for the libdevice bitcode library for the given GPU architecture. */
39/* Returns zero on success, non-zero on failure. */
40int findLibDevice(char LibDevicePath[PATH_MAX], const char *Arch);
41
42/* Generate a PTX file from an LLVM bitcode file. */
43/* Returns zero on success, non-zero on failure. */
44int pocl_ptx_gen(const char *BitcodeFilename,
45 const char *PTXFilename,
46 const char *KernelName,
47 const char *Arch,
48 const char *LibDevicePath,
49 int HasOffsets);
50
51/* Populate the Alignments array with the required pointer alignments for */
52/* each kernel argument. */
53/* Returns zero on success, non-zero on failure. */
54int pocl_cuda_get_ptr_arg_alignment(const char *BitcodeFilename,
55 const char *KernelName,
56 size_t *Alignments);
57
58#ifdef __cplusplus
59}
60#endif

62#endif /* POCL_PTX_GEN_H */
