81/**
82 * Generate code from the final bitcode using the LLVM
83 * tools.
84 *
85 * Uses an existing (cached) one, if available.
86 */
87
88#ifdef OCS_AVAILABLE
89int
90llvm_codegen (char *output, unsigned device_i, cl_kernel kernel,
91 cl_device_id device, _cl_command_node *command, int specialize)
92{
93 POCL_MEASURE_START (llvm_codegen);
94 int error = 0;
95 void *llvm_module = NULL;
96
97 char tmp_module[POCL_FILENAME_LENGTH];
98 char tmp_objfile[POCL_FILENAME_LENGTH];
99
100 char *objfile = NULL;
101 uint64_t objfile_size = 0;
102
103 cl_program program = kernel->program;
104
105 const char *kernel_name = kernel->name;
106
107 /* $/parallel.bc */
108 char parallel_bc_path[POCL_FILENAME_LENGTH];
109 pocl_cache_work_group_function_path (parallel_bc_path, program, device_i,
110 kernel, command, specialize);
111
112 /* $/kernel.so */
113 char final_binary_path[POCL_FILENAME_LENGTH];
114 pocl_cache_final_binary_path (final_binary_path, program, device_i, kernel,
115 command, specialize);
116
117 if (pocl_exists (final_binary_path))
118 goto FINISH;
119
120 assert (strlen (final_binary_path) < (POCL_FILENAME_LENGTH - 3));
121
122 error = pocl_llvm_generate_workgroup_function_nowrite (
123 device_i, device, kernel, command, &llvm_module, specialize);
124 if (error)
125 {
126 POCL_MSG_PRINT_LLVM ("pocl_llvm_generate_workgroup_function() failed"
127 " for kernel %s\n",
128 kernel_name);
129 goto FINISH;
130 }
131 assert (llvm_module != NULL);
132
133 if (pocl_get_bool_option ("POCL_LEAVE_KERNEL_COMPILER_TEMP_FILES", 0))
134 {
135 POCL_MSG_PRINT_LLVM ("Writing parallel.bc to %s.\n", parallel_bc_path);
136 error = pocl_cache_write_kernel_parallel_bc (
137 llvm_module, program, device_i, kernel, command, specialize);
138 }
139 else
140 {
141 char kernel_parallel_path[POCL_FILENAME_LENGTH];
142 pocl_cache_kernel_cachedir_path (kernel_parallel_path, program, device_i,
143 kernel, "", command, specialize);
144 error = pocl_mkdir_p (kernel_parallel_path);
145 }
146 if (error)
147 {
148 POCL_MSG_PRINT_GENERAL ("writing parallel.bc failed for kernel %s\n",
149 kernel->name);
150 goto FINISH;
151 }
152
153 /* May happen if another thread is building the same program & wins the llvm
154 lock. */
155 if (pocl_exists (final_binary_path))
156 goto FINISH;
157
158 error = pocl_llvm_codegen (device, llvm_module, &objfile, &objfile_size);
159 if (error)
160 {
161 POCL_MSG_PRINT_LLVM ("pocl_llvm_codegen() failed for kernel %s\n",
162 kernel_name);
163 goto FINISH;
164 }
165
166 if (pocl_exists (final_binary_path))
167 goto FINISH;
168
169 /* Write temporary kernel.so.o, required for the final linking step.
170 Use append-write because tmp_objfile is already temporary, thus
171 we don't need to create new temporary... */
172 error = pocl_cache_write_kernel_objfile (tmp_objfile, objfile, objfile_size);
173 if (error)
174 {
175 POCL_MSG_PRINT_LLVM ("writing %s failed for kernel %s\n",
176 tmp_objfile, kernel_name);
177 goto FINISH;
178 }
179 else
180 {
181 POCL_MSG_PRINT_LLVM ("written %s size %zu\n",
182 tmp_objfile, (size_t)objfile_size);
183 }
184
185 /* temporary filename for kernel.so */
186 if (pocl_cache_tempname (tmp_module, ".so", NULL))
187 {
188 POCL_MSG_PRINT_LLVM ("Creating temporary kernel.so file"
189 " for kernel %s FAILED\n",
190 kernel_name);
191 goto FINISH;
192 }
193 else
194 POCL_MSG_PRINT_LLVM ("Temporary kernel.so file"
195 " for kernel %s : %s\n",
196 kernel_name, tmp_module);
197
198 POCL_MSG_PRINT_INFO ("Linking final module\n");
199
200 /* Link through Clang driver interface who knows the correct toolchains
201 for all of its targets. */
202 const char *cmd_line[64] =
203 {CLANG, "-o", tmp_module, tmp_objfile};
204 const char **device_ld_arg = device->final_linkage_flags;
205 const char **pos = &cmd_line[4];
206 while ((*pos++ = *device_ld_arg++)) {}
207
208 error = pocl_invoke_clang (device, cmd_line);
209
210 if (error)
211 {
212 POCL_MSG_PRINT_LLVM ("Linking kernel.so.o -> kernel.so has failed\n");
213 goto FINISH;
214 }
215
216 /* rename temporary kernel.so */
217 error = pocl_rename (tmp_module, final_binary_path);
218 if (error)
219 {
220 POCL_MSG_PRINT_LLVM ("Renaming temporary kernel.so to final has failed.\n");
221 goto FINISH;
222 }
223
224 /* if LEAVE_COMPILER_FILES, rename temporary kernel.so.o, else delete it */
225 if (pocl_get_bool_option ("POCL_LEAVE_KERNEL_COMPILER_TEMP_FILES", 0))
226 {
227 char objfile_path[POCL_FILENAME_LENGTH];
228 strcpy (objfile_path, final_binary_path);
229 strcat (objfile_path, ".o");
230 error = pocl_rename (tmp_objfile, objfile_path);
231 if (error)
232 POCL_MSG_PRINT_LLVM ("Renaming temporary kernel.so.o to final .o has failed.\n");
233 }
234 else
235 {
236 error = pocl_remove (tmp_objfile);
237 if (error)
238 POCL_MSG_PRINT_LLVM ("Removing temporary kernel.so.o has failed.\n");
239 }
240
241FINISH:
242 pocl_destroy_llvm_module (llvm_module);
243 POCL_MEM_FREE (objfile);
244 POCL_MEASURE_FINISH (llvm_codegen);
245
246 if (error)
247 return error;
248 else
249 {
250 memcpy (output, final_binary_path, POCL_FILENAME_LENGTH);
251 return 0;
252 }
253}

257/**
258 * Populates the device specific image data structure used by kernel
259 * from given kernel image argument
260 */
261void
262fill_dev_image_t (dev_image_t* di, struct pocl_argument* parg,
263 cl_device_id device)
264{
265 cl_mem mem = *(cl_mem *)parg->value;
266 di->_width = mem->image_width;
267 di->_height = mem->image_height;
268 di->_depth = mem->image_depth;
269 di->_row_pitch = mem->image_row_pitch;
270 di->_slice_pitch = mem->image_slice_pitch;
271 di->_order = mem->image_channel_order;
272 di->_image_array_size = mem->image_array_size;
273 di->_data_type = mem->image_channel_data_type;
274 pocl_get_image_information (mem->image_channel_order,
275 mem->image_channel_data_type,
276 &(di->_num_channels), &(di->_elem_size));
277
278 IMAGE1D_TO_BUFFER (mem);
279 di->_data = (mem->device_ptrs[device->dev_id].mem_ptr);
280}

282void
283pocl_copy_mem_object (cl_device_id dest_dev, cl_mem dest,
284 size_t dest_offset,
285 cl_device_id source_dev, cl_mem source,
286 size_t source_offset, size_t cb)
287{
288 /* if source_dev is NULL -> src and dest dev must be the same */
289 cl_device_id src_dev = (source_dev) ? source_dev : dest_dev;
290
291 /* if source and destination are on the same global mem */
292 if (src_dev->global_mem_id == dest_dev->global_mem_id)
293 {
294 src_dev->ops->copy
295 (dest_dev->data,
296 &dest->device_ptrs[dest_dev->dev_id],
297 dest,
298 &source->device_ptrs[src_dev->dev_id],
299 source,
300 dest_offset, source_offset,
301 cb);
302 }
303 else
304 {
305 void* tofree = NULL;
306 void* tmp = NULL;
307 if (source->flags & CL_MEM_USE_HOST_PTR)
308 tmp = source->mem_host_ptr;
309 else if (dest->flags & CL_MEM_USE_HOST_PTR)
310 tmp = dest->mem_host_ptr;
311 else
312 {
313 tmp = malloc (dest->size);
314 tofree = tmp;
315 }
316
317 src_dev->ops->read
318 (src_dev->data, tmp,
319 &source->device_ptrs[src_dev->dev_id],
320 source,
321 source_offset, cb);
322 dest_dev->ops->write
323 (dest_dev->data, tmp,
324 &dest->device_ptrs[dest_dev->dev_id],
325 dest, dest_offset,
326 cb);
327 free (tofree);
328 }
329 return;
330}

332void
333pocl_migrate_mem_objects (_cl_command_node * volatile node)
334{
335 size_t i;
336 cl_mem *mem_objects = node->command.migrate.mem_objects;
337
338 for (i = 0; i < node->command.migrate.num_mem_objects; ++i)
339 {
340 pocl_copy_mem_object (node->device,
341 mem_objects[i], 0,
342 node->command.migrate.source_devices[i],
343 mem_objects[i], 0, mem_objects[i]->size);
344
345 return;
346 }
347}

349void
350pocl_ndrange_node_cleanup(_cl_command_node *node)
351{
352 cl_uint i;
353 for (i = 0; i < node->command.run.kernel->meta->num_args; ++i)
354 {
355 pocl_aligned_free (node->command.run.arguments[i].value);
356 }
357 free (node->command.run.arguments);
358
359 POname(clReleaseKernel)(node->command.run.kernel);
360}
361
362void
363pocl_mem_objs_cleanup (cl_event event)
364{
365 size_t i;
366 for (i = 0; i < event->num_buffers; ++i)
367 {
368 assert(event->mem_objs[i] != NULL);
369 POname(clReleaseMemObject) (event->mem_objs[i]);
370 }
371 free (event->mem_objs);
372 event->mem_objs = NULL;
373 event->num_buffers = 0;
374}

376/**
377 * executes given command. Call with node->event UNLOCKED.
378 */
379void
380pocl_exec_command (_cl_command_node * volatile node)
381{
382 unsigned i;
383 /* because of POCL_UPDATE_EVENT_ */
384 cl_event event = node->event;
385 cl_device_id dev = node->device;
386 _cl_command_t *cmd = &node->command;
387 switch (node->type)
388 {
389 case CL_COMMAND_READ_BUFFER:
390 pocl_update_event_running (event);
391 assert (dev->ops->read);
392 dev->ops->read
393 (dev->data,
394 cmd->read.dst_host_ptr,
395 cmd->read.src_mem_id,
396 event->mem_objs[0],
397 cmd->read.offset,
398 cmd->read.size);
399 POCL_UPDATE_EVENT_COMPLETE_MSG (event, "Event Read Buffer ");
400 break;
401
402 case CL_COMMAND_WRITE_BUFFER:
403 pocl_update_event_running (event);
404 assert (dev->ops->write);
405 dev->ops->write
406 (dev->data,
407 cmd->write.src_host_ptr,
408 cmd->write.dst_mem_id,
409 event->mem_objs[0],
410 cmd->write.offset,
411 cmd->write.size);
412 POCL_UPDATE_EVENT_COMPLETE_MSG (event, "Event Write Buffer ");
413 break;
414
415 case CL_COMMAND_COPY_BUFFER:
416 pocl_update_event_running (event);
417 assert (dev->ops->copy);
418 dev->ops->copy
419 (dev->data,
420 cmd->copy.dst_mem_id,
421 event->mem_objs[1],
422 cmd->copy.src_mem_id,
423 event->mem_objs[0],
424 cmd->copy.dst_offset,
425 cmd->copy.src_offset,
426 cmd->copy.size);
427 POCL_UPDATE_EVENT_COMPLETE_MSG (event, "Event Copy Buffer ");
428 break;
429
430 case CL_COMMAND_FILL_BUFFER:
431 pocl_update_event_running (event);
432 assert (dev->ops->memfill);
433 dev->ops->memfill
434 (dev->data,
435 cmd->memfill.dst_mem_id,
436 event->mem_objs[0],
437 cmd->memfill.size,
438 cmd->memfill.offset,
439 cmd->memfill.pattern,
440 cmd->memfill.pattern_size);
441 POCL_UPDATE_EVENT_COMPLETE_MSG (event, "Event Fill Buffer ");
442 pocl_aligned_free (cmd->memfill.pattern);
443 break;
444
445 case CL_COMMAND_READ_BUFFER_RECT:
446 pocl_update_event_running (event);
447 assert (dev->ops->read_rect);
448 dev->ops->read_rect
449 (dev->data,
450 cmd->read_rect.dst_host_ptr,
451 cmd->read_rect.src_mem_id,
452 event->mem_objs[0],
453 cmd->read_rect.buffer_origin,
454 cmd->read_rect.host_origin,
455 cmd->read_rect.region,
456 cmd->read_rect.buffer_row_pitch,
457 cmd->read_rect.buffer_slice_pitch,
458 cmd->read_rect.host_row_pitch,
459 cmd->read_rect.host_slice_pitch);
460 POCL_UPDATE_EVENT_COMPLETE_MSG (event, "Event Read Buffer Rect ");
461 break;
462
463 case CL_COMMAND_COPY_BUFFER_RECT:
464 pocl_update_event_running (event);
465 assert (dev->ops->copy_rect);
466 dev->ops->copy_rect
467 (dev->data,
468 cmd->copy_rect.dst_mem_id,
469 event->mem_objs[1],
470 cmd->copy_rect.src_mem_id,
471 event->mem_objs[0],
472 cmd->copy_rect.dst_origin,
473 cmd->copy_rect.src_origin,
474 cmd->copy_rect.region,
475 cmd->copy_rect.dst_row_pitch,
476 cmd->copy_rect.dst_slice_pitch,
477 cmd->copy_rect.src_row_pitch,
478 cmd->copy_rect.src_slice_pitch);
479 POCL_UPDATE_EVENT_COMPLETE_MSG (event, "Event Copy Buffer Rect ");
480 break;
481
482 case CL_COMMAND_WRITE_BUFFER_RECT:
483 pocl_update_event_running (event);
484 assert (dev->ops->write_rect);
485 dev->ops->write_rect
486 (dev->data,
487 cmd->write_rect.src_host_ptr,
488 cmd->write_rect.dst_mem_id,
489 event->mem_objs[0],
490 cmd->write_rect.buffer_origin,
491 cmd->write_rect.host_origin,
492 cmd->write_rect.region,
493 cmd->write_rect.buffer_row_pitch,
494 cmd->write_rect.buffer_slice_pitch,
495 cmd->write_rect.host_row_pitch,
496 cmd->write_rect.host_slice_pitch);
497 POCL_UPDATE_EVENT_COMPLETE_MSG (event, "Event Write Buffer Rect ");
498 break;
499
500 case CL_COMMAND_MIGRATE_MEM_OBJECTS:
501 pocl_update_event_running (event);
502 pocl_migrate_mem_objects (node);
503 POCL_UPDATE_EVENT_COMPLETE_MSG (event, "Event Migrate Buffer ");
504 break;
505
506 case CL_COMMAND_MAP_BUFFER:
507 pocl_update_event_running (event);
508 assert (dev->ops->map_mem);
509 POCL_LOCK_OBJ (event->mem_objs[0]);
510 dev->ops->map_mem (dev->data,
511 cmd->map.mem_id,
512 event->mem_objs[0],
513 cmd->map.mapping);
514 (event->mem_objs[0])->map_count++;
515 POCL_UNLOCK_OBJ (event->mem_objs[0]);
516 POCL_UPDATE_EVENT_COMPLETE_MSG (event, "Event Map Buffer ");
517 break;
518
519 case CL_COMMAND_COPY_IMAGE_TO_BUFFER:
520 pocl_update_event_running (event);
521 assert (dev->ops->read_image_rect);
522 dev->ops->read_image_rect (
523 dev->data,
524 event->mem_objs[0],
525 cmd->read_image.src_mem_id,
526 NULL,
527 cmd->read_image.dst_mem_id,
528 cmd->read_image.origin,
529 cmd->read_image.region,
530 cmd->read_image.dst_row_pitch,
531 cmd->read_image.dst_slice_pitch,
532 cmd->read_image.dst_offset);
533 POCL_UPDATE_EVENT_COMPLETE_MSG (event, "Event CopyImageToBuffer ");
534 break;
535
536 case CL_COMMAND_READ_IMAGE:
537 pocl_update_event_running (event);
538 assert (dev->ops->read_image_rect);
539 dev->ops->read_image_rect (
540 dev->data,
541 event->mem_objs[0],
542 cmd->read_image.src_mem_id,
543 cmd->read_image.dst_host_ptr,
544 NULL,
545 cmd->read_image.origin,
546 cmd->read_image.region,
547 cmd->read_image.dst_row_pitch,
548 cmd->read_image.dst_slice_pitch,
549 cmd->read_image.dst_offset);
550 POCL_UPDATE_EVENT_COMPLETE_MSG (event, "Event Read Image ");
551 break;
552
553 case CL_COMMAND_COPY_BUFFER_TO_IMAGE:
554 pocl_update_event_running (event);
555 assert (dev->ops->write_image_rect);
556 dev->ops->write_image_rect (
557 dev->data,
558 event->mem_objs[1],
559 cmd->write_image.dst_mem_id,
560 NULL,
561 cmd->write_image.src_mem_id,
562 cmd->write_image.origin,
563 cmd->write_image.region,
564 cmd->write_image.src_row_pitch,
565 cmd->write_image.src_slice_pitch,
566 cmd->write_image.src_offset);
567 POCL_UPDATE_EVENT_COMPLETE_MSG (event, "Event CopyBufferToImage ");
568 break;
569
570 case CL_COMMAND_WRITE_IMAGE:
571 pocl_update_event_running (event);
572 assert (dev->ops->write_image_rect);
573 dev->ops->write_image_rect (
574 dev->data,
575 event->mem_objs[0],
576 cmd->write_image.dst_mem_id,
577 cmd->write_image.src_host_ptr,
578 NULL,
579 cmd->write_image.origin,
580 cmd->write_image.region,
581 cmd->write_image.src_row_pitch,
582 cmd->write_image.src_slice_pitch,
583 cmd->write_image.src_offset);
584 POCL_UPDATE_EVENT_COMPLETE_MSG (event, "Event Write Image ");
585 break;
586
587 case CL_COMMAND_COPY_IMAGE:
588 pocl_update_event_running (event);
589 assert (dev->ops->copy_image_rect);
590 dev->ops->copy_image_rect(
591 dev->data,
592 event->mem_objs[0],
593 event->mem_objs[1],
594 cmd->copy_image.src_mem_id,
595 cmd->copy_image.dst_mem_id,
596 cmd->copy_image.src_origin,
597 cmd->copy_image.dst_origin,
598 cmd->copy_image.region);
599 POCL_UPDATE_EVENT_COMPLETE_MSG (event, "Event Copy Image ");
600 break;
601
602 case CL_COMMAND_FILL_IMAGE:
603 pocl_update_event_running (event);
604 assert (dev->ops->fill_image);
605 dev->ops->fill_image
606 (dev->data,
607 event->mem_objs[0],
608 cmd->fill_image.mem_id,
609 cmd->fill_image.origin,
610 cmd->fill_image.region,
611 cmd->fill_image.fill_pixel,
612 cmd->fill_image.pixel_size);
613 POCL_UPDATE_EVENT_COMPLETE_MSG (event, "Event Fill Image ");
614 POCL_MEM_FREE (cmd->fill_image.fill_pixel);
615 break;
616
617 case CL_COMMAND_MAP_IMAGE:
618 pocl_update_event_running (event);
619 POCL_LOCK_OBJ (event->mem_objs[0]);
620 assert (dev->ops->map_image != NULL);
621 dev->ops->map_image (dev->data,
622 cmd->map.mem_id,
623 event->mem_objs[0],
624 cmd->map.mapping);
625 (event->mem_objs[0])->map_count++;
626 POCL_UNLOCK_OBJ (event->mem_objs[0]);
627 POCL_UPDATE_EVENT_COMPLETE_MSG (event, "Event Map Image ");
628 break;
629
630 case CL_COMMAND_UNMAP_MEM_OBJECT:
631 pocl_update_event_running (event);
632 POCL_LOCK_OBJ (event->mem_objs[0]);
633 if (event->mem_objs[0]->is_image == CL_FALSE
634 || IS_IMAGE1D_BUFFER (event->mem_objs[0]))
635 {
636 assert (dev->ops->unmap_mem != NULL);
637 dev->ops->unmap_mem (dev->data,
638 cmd->unmap.mem_id,
639 event->mem_objs[0],
640 cmd->unmap.mapping);
641 }
642 else
643 {
644 assert (dev->ops->unmap_image != NULL);
645 dev->ops->unmap_image (dev->data,
646 cmd->unmap.mem_id,
647 event->mem_objs[0],
648 cmd->unmap.mapping);
649 }
650 assert ((cmd->unmap.mapping)->unmap_requested > 0);
651 DL_DELETE((event->mem_objs[0])->mappings,
652 cmd->unmap.mapping);
653 (event->mem_objs[0])->map_count--;
654 POCL_MEM_FREE (cmd->unmap.mapping);
655 POCL_UNLOCK_OBJ (event->mem_objs[0]);
656 POCL_UPDATE_EVENT_COMPLETE_MSG (event, "Event Unmap Mem obj ");
657 break;
658
659 case CL_COMMAND_NDRANGE_KERNEL:
660 pocl_update_event_running (event);
661 assert (event == node->event);
662 assert (dev->ops->run);
663 dev->ops->run (dev->data, node);
664 POCL_UPDATE_EVENT_COMPLETE_MSG (event, "Event Enqueue NDRange ");
665 pocl_ndrange_node_cleanup(node);
666 break;
667
668 case CL_COMMAND_NATIVE_KERNEL:
669 pocl_update_event_running (event);
670 assert (dev->ops->run_native);
671 dev->ops->run_native (dev->data, node);
672 POCL_MEM_FREE (node->command.native.args);
673 POCL_UPDATE_EVENT_COMPLETE_MSG (event, "Event Native Kernel ");
674 break;
675
676 case CL_COMMAND_MARKER:
677 pocl_update_event_running (event);
678 POCL_UPDATE_EVENT_COMPLETE(event);
679 break;
680
681 case CL_COMMAND_BARRIER:
682 pocl_update_event_running (event);
683 POCL_UPDATE_EVENT_COMPLETE(event);
684 break;
685
686 case CL_COMMAND_SVM_FREE:
687 pocl_update_event_running (event);
688 if (cmd->svm_free.pfn_free_func)
689 cmd->svm_free.pfn_free_func(
690 cmd->svm_free.queue,
691 cmd->svm_free.num_svm_pointers,
692 cmd->svm_free.svm_pointers,
693 cmd->svm_free.data);
694 else
695 for (i = 0; i < cmd->svm_free.num_svm_pointers; i++)
696 dev->ops->svm_free (dev, cmd->svm_free.svm_pointers[i]);
697 POCL_UPDATE_EVENT_COMPLETE_MSG (event, "Event SVM Free ");
698 break;
699
700 case CL_COMMAND_SVM_MAP:
701 pocl_update_event_running (event);
702 if (DEVICE_MMAP_IS_NOP (dev))
703 ; // no-op
704 else
705 {
706 assert (dev->ops->svm_map);
707 dev->ops->svm_map (dev, cmd->svm_map.svm_ptr);
708 }
709 POCL_UPDATE_EVENT_COMPLETE_MSG (event, "Event SVM Map ");
710 break;
711
712 case CL_COMMAND_SVM_UNMAP:
713 pocl_update_event_running (event);
714 if (DEVICE_MMAP_IS_NOP (dev))
715 ; // no-op
716 else
717 {
718 assert (dev->ops->svm_unmap);
719 dev->ops->svm_unmap (dev, cmd->svm_unmap.svm_ptr);
720 }
721 POCL_UPDATE_EVENT_COMPLETE_MSG (event, "Event SVM Unmap ");
722 break;
723
724 case CL_COMMAND_SVM_MEMCPY:
725 pocl_update_event_running (event);
726 assert (dev->ops->svm_copy);
727 dev->ops->svm_copy (dev,
728 cmd->svm_memcpy.dst,
729 cmd->svm_memcpy.src,
730 cmd->svm_memcpy.size);
731 POCL_UPDATE_EVENT_COMPLETE_MSG (event, "Event SVM Memcpy ");
732 break;
733
734 case CL_COMMAND_SVM_MEMFILL:
735 pocl_update_event_running (event);
736 assert (dev->ops->svm_fill);
737 dev->ops->svm_fill (dev,
738 cmd->svm_fill.svm_ptr,
739 cmd->svm_fill.size,
740 cmd->svm_fill.pattern,
741 cmd->svm_fill.pattern_size);
742 POCL_UPDATE_EVENT_COMPLETE_MSG (event, "Event SVM MemFill ");
743 pocl_aligned_free (cmd->svm_fill.pattern);
744 break;
745
746 default:
747 POCL_ABORT_UNIMPLEMENTED("");
748 break;
749 }
750 pocl_mem_manager_free_command (node);
751}

753/* call with brc_event UNLOCKED. */
754void
755pocl_broadcast (cl_event brc_event)
756{
757 event_node *target;
758 event_node *tmp;
759
760 while ((target = brc_event->notify_list))
761 {
762 pocl_lock_events_inorder (brc_event, target->event);
763 /* remove event from wait list */
764 LL_FOREACH (target->event->wait_list, tmp)
765 {
766 if (tmp->event == brc_event)
767 {
768 LL_DELETE (target->event->wait_list, tmp);
769 pocl_mem_manager_free_event_node (tmp);
770 break;
771 }
772 }
773
774 if ((target->event->status == CL_SUBMITTED)
775 || (target->event->status == CL_QUEUED))
776 {
777 target->event->command->device->ops->notify (
778 target->event->command->device, target->event, brc_event);
779 }
780
781 LL_DELETE (brc_event->notify_list, target);
782 pocl_unlock_events_inorder (brc_event, target->event);
783 pocl_mem_manager_free_event_node (target);
784 }
785}

822/* CPU driver stuff */
823typedef struct pocl_dlhandle_cache_item pocl_dlhandle_cache_item;
824struct pocl_dlhandle_cache_item
825{
826 pocl_kernel_hash_t hash;
827
828 /* The specialization properties. */
829 /* The local dimensions. */
830 size_t local_wgs[3];
831 /* If global offset must be zero for this WG function version. */
832 int goffs_zero;
833 /* Maximum grid dimension this WG function works with. */
834 size_t max_grid_dim_width;
835
836 void *wg;
837 void *dlhandle;
838 pocl_dlhandle_cache_item *next;
839 pocl_dlhandle_cache_item *prev;
840 unsigned ref_count;
841};
842
843static pocl_dlhandle_cache_item *pocl_dlhandle_cache;
844static pocl_lock_t pocl_llvm_codegen_lock;
845static pocl_lock_t pocl_dlhandle_lock;
846static int pocl_dlhandle_cache_initialized;
847
848/* only to be called in basic/pthread/<other cpu driver> init */
849void
850pocl_init_dlhandle_cache ()
851{
852 if (!pocl_dlhandle_cache_initialized)
853 {
854 POCL_INIT_LOCK (pocl_llvm_codegen_lock);
855 POCL_INIT_LOCK (pocl_dlhandle_lock);
856 pocl_dlhandle_cache_initialized = 1;
857 }
858}
859
860static unsigned handle_count = 0;
861#define MAX_CACHE_ITEMS 128
862
863/* must be called with pocl_dlhandle_lock LOCKED */
864static pocl_dlhandle_cache_item *
865get_new_dlhandle_cache_item ()
866{
867 pocl_dlhandle_cache_item *ci = NULL;
868 const char *dl_error = NULL;
869
870 if (pocl_dlhandle_cache)
871 {
872 ci = pocl_dlhandle_cache->prev;
873 while (ci->ref_count > 0 && ci != pocl_dlhandle_cache)
874 ci = ci->prev;
875 }
876
877 if ((handle_count >= MAX_CACHE_ITEMS) && ci && (ci != pocl_dlhandle_cache))
878 {
879 DL_DELETE (pocl_dlhandle_cache, ci);
880 dlclose (ci->dlhandle);
881 dl_error = dlerror ();
882 if (dl_error != NULL)
883 POCL_ABORT ("dlclose() failed with error: %s\n", dl_error);
884 memset (ci, 0, sizeof (pocl_dlhandle_cache_item));
885 }
886 else
887 {
888 ++handle_count;
889 ci = (pocl_dlhandle_cache_item *)calloc (
890 1, sizeof (pocl_dlhandle_cache_item));
891 }
892
893 return ci;
894}
895
896void
897pocl_release_dlhandle_cache (_cl_command_node *cmd)
898{
899 pocl_dlhandle_cache_item *ci = NULL, *found = NULL;
900
901 POCL_LOCK (pocl_dlhandle_lock);
902 DL_FOREACH (pocl_dlhandle_cache, ci)
903 {
904 if ((memcmp (ci->hash, cmd->command.run.hash, sizeof (pocl_kernel_hash_t))
905 == 0)
906 && (ci->local_wgs[0] == cmd->command.run.pc.local_size[0])
907 && (ci->local_wgs[1] == cmd->command.run.pc.local_size[1])
908 && (ci->local_wgs[2] == cmd->command.run.pc.local_size[2]))
909 {
910 found = ci;
911 break;
912 }
913 }
914
915 assert (found != NULL);
916 assert (found->ref_count > 0);
917 --found->ref_count;
918 POCL_UNLOCK (pocl_dlhandle_lock);
919}

921/**
922 * Checks if a built binary is found in the disk for the given kernel command,
923 * if not, builds the kernel, caches it, and returns the file name of the
924 * end result.
925 *
926 * @param command The kernel run command.
927 * @param specialized 1 if should check the per-command specialized one instead
928 * of the generic one.
929 * @returns The filename of the built binary in the disk.
930 */
931char *
932pocl_check_kernel_disk_cache (_cl_command_node *command, int specialized)
933{
934 char *module_fn = NULL;
935 _cl_command_run *run_cmd = &command->command.run;
936 cl_kernel k = run_cmd->kernel;
937 cl_program p = k->program;
938 unsigned dev_i = command->device_i;
939
940 /* First try to find a static WG binary for the local size as they
941 are always more efficient than the dynamic ones. Also, in case
942 of reqd_wg_size, there might not be a dynamic sized one at all. */
943 module_fn = malloc (POCL_FILENAME_LENGTH);
944 pocl_cache_final_binary_path (module_fn, p, dev_i, k, command, specialized);
945
946 if (pocl_exists (module_fn))
947 {
948 POCL_MSG_PRINT_INFO ("Using a cached WG function: %s\n", module_fn);
949 return module_fn;
950 }
951
952 /* static WG binary for the local size does not exist. If we have the LLVM IR
953 * (program.bc), try to compile a new parallel.bc and static binary */
954 if (p->binaries[dev_i])
955 {
956#ifdef OCS_AVAILABLE
957 POCL_LOCK (pocl_llvm_codegen_lock);
958 int error = llvm_codegen (module_fn, dev_i, k, command->device, command,
959 specialized);
960 POCL_UNLOCK (pocl_llvm_codegen_lock);
961 if (error)
962 POCL_ABORT ("Final linking of kernel %s failed.\n", k->name);
963 POCL_MSG_PRINT_INFO ("Built a WG function: %s\n", module_fn);
964 return module_fn;
965#else
966 /* TODO: This should be caught earlier. */
967 if (!p->pocl_binaries[dev_i])
968 POCL_ABORT ("pocl device without online compilation support"
969 " cannot compile LLVM IRs to machine code!\n");
970#endif
971 }
972 else
973 {
974 module_fn = malloc (POCL_FILENAME_LENGTH);
975 /* First try to find a specialized WG binary, if allowed by the
976 command. */
977 if (!run_cmd->force_generic_wg_func)
978 pocl_cache_final_binary_path (module_fn, p, dev_i, k, command, 0);
979
980 if (run_cmd->force_generic_wg_func || !pocl_exists (module_fn))
981 {
982 /* Then check for a dynamic (non-specialized) kernel. */
983 pocl_cache_final_binary_path (module_fn, p, dev_i, k, command, 1);
984 if (!pocl_exists (module_fn))
985 POCL_ABORT ("Generic WG function binary does not exist.\n");
986 POCL_MSG_PRINT_INFO ("Using a cached generic WG function: %s\n",
987 module_fn);
988 }
989 else
990 POCL_MSG_PRINT_INFO ("Using a cached specialized WG function: %s\n",
991 module_fn);
992 }
993 return module_fn;
994}

996/* Returns the width of the widest dimension in the grid of the given
997 run command. */
998size_t
999pocl_cmd_max_grid_dim_width (_cl_command_run *cmd)
1000{
1001 return max (max (cmd->pc.local_size[0] * cmd->pc.num_groups[0],
1002 cmd->pc.local_size[1] * cmd->pc.num_groups[1]),
1003 cmd->pc.local_size[2] * cmd->pc.local_size[2]);
1004}
1005
1006/* Look for a dlhandle in the dlhandle cache for the given kernel command.
1007 If found, push the handle up in the cache to improve cache hit speed,
1008 and return it. Otherwise return NULL. The caller should hold
1009 pocl_dlhandle_lock. */
1010static pocl_dlhandle_cache_item *
1011fetch_dlhandle_cache_item (_cl_command_run *run_cmd)
1012{
1013 pocl_dlhandle_cache_item *ci = NULL, *tmp = NULL;
1014 size_t max_grid_width = pocl_cmd_max_grid_dim_width (run_cmd);
1015 DL_FOREACH_SAFE (pocl_dlhandle_cache, ci, tmp)
1016 {
1017 if ((memcmp (ci->hash, run_cmd->hash, sizeof (pocl_kernel_hash_t)) == 0)
1018 && (ci->local_wgs[0] == run_cmd->pc.local_size[0])
1019 && (ci->local_wgs[1] == run_cmd->pc.local_size[1])
1020 && (ci->local_wgs[2] == run_cmd->pc.local_size[2])
1021 && (max_grid_width <= ci->max_grid_dim_width)
1022 && (!ci->goffs_zero
1023 || (run_cmd->pc.global_offset[0] == 0
1024 && run_cmd->pc.global_offset[1] == 0
1025 && run_cmd->pc.global_offset[2] == 0)))
1026 {
1027 /* move to the front of the line */
1028 DL_DELETE (pocl_dlhandle_cache, ci);
1029 DL_PREPEND (pocl_dlhandle_cache, ci);
1030 ++ci->ref_count;
1031 run_cmd->wg = ci->wg;
1032 return ci;
1033 }
1034 }
1035 return NULL;
1036}

1038/**
1039 * Checks if the kernel command has been built and has been loaded with
1040 * dlopen, and reuses its handle. If not, checks if a built binary is found
1041 * in the disk, if not, builds the kernel and puts it to respective
1042 * caches.
1043 *
1044 * The initial refcount may be 0, in case we're just pre-compiling kernels
1045 * (or compiling them for binaries), and not actually need them immediately.
1046 *
1047 * TODO: This function is really specific to CPU (host) drivers since dlhandles
1048 * imply program loading to the same process as the host. Move to basic.c? */
1049void
1050pocl_check_kernel_dlhandle_cache (_cl_command_node *command,
1051 unsigned initial_refcount, int specialize)
1052{
1053 char workgroup_string[WORKGROUP_STRING_LENGTH];
1054 pocl_dlhandle_cache_item *ci = NULL, *tmp = NULL;
1055 const char *dl_error = NULL;
1056 _cl_command_run *run_cmd = &command->command.run;
1057
1058 POCL_LOCK (pocl_dlhandle_lock);
1059 ci = fetch_dlhandle_cache_item (run_cmd);
1060 if (ci != NULL)
1061 {
1062 POCL_UNLOCK (pocl_dlhandle_lock);
1063 return;
1064 }
1065
1066 /* Not found, build a new kernel and cache its dlhandle. */
1067 ci = get_new_dlhandle_cache_item ();
1068 memcpy (ci->hash, run_cmd->hash, sizeof (pocl_kernel_hash_t));
1069 ci->local_wgs[0] = run_cmd->pc.local_size[0];
1070 ci->local_wgs[1] = run_cmd->pc.local_size[1];
1071 ci->local_wgs[2] = run_cmd->pc.local_size[2];
1072 ci->ref_count = initial_refcount;
1073
1074 ci->goffs_zero = run_cmd->pc.global_offset[0] == 0
1075 && run_cmd->pc.global_offset[1] == 0
1076 && run_cmd->pc.global_offset[2] == 0;
1077
1078 size_t max_grid_width = pocl_cmd_max_grid_dim_width (run_cmd);
1079 ci->max_grid_dim_width = max_grid_width;
1080
1081 char *module_fn = pocl_check_kernel_disk_cache (command, specialize);
1082
1083 ci->dlhandle = dlopen (module_fn, RTLD_NOW | RTLD_LOCAL);
1084 dl_error = dlerror ();
1085
1086 if (ci->dlhandle == NULL || dl_error != NULL)
1087 POCL_ABORT ("dlopen(\"%s\") failed with '%s'.\n"
1088 "note: missing symbols in the kernel binary might be"
1089 " reported as 'file not found' errors.\n",
1090 module_fn, dl_error);
1091
1092 snprintf (workgroup_string, WORKGROUP_STRING_LENGTH,
1093 "_pocl_kernel_%s_workgroup", run_cmd->kernel->name);
1094
1095 ci->wg = dlsym (ci->dlhandle, workgroup_string);
1096 dl_error = dlerror ();
1097
1098 if (ci->wg == NULL || dl_error != NULL)
1099 {
1100 // Older OSX dyld APIs need the name without the underscore.
1101 snprintf (workgroup_string, WORKGROUP_STRING_LENGTH,
1102 "pocl_kernel_%s_workgroup", run_cmd->kernel->name);
1103 ci->wg = dlsym (ci->dlhandle, workgroup_string);
1104 dl_error = dlerror ();
1105
1106 if (ci->wg == NULL || dl_error != NULL)
1107 POCL_ABORT ("dlsym(\"%s\", \"%s\") failed with '%s'.\n"
1108 "note: missing symbols in the kernel binary might be"
1109 " reported as 'file not found' errors.\n",
1110 module_fn, workgroup_string, dl_error);
1111 }
1112
1113 run_cmd->wg = ci->wg;
1114 DL_PREPEND (pocl_dlhandle_cache, ci);
1115
1116 POCL_UNLOCK (pocl_dlhandle_lock);
1117 /***************************************************************************/
1118 POCL_MEM_FREE (module_fn);
1119}

1121#define MIN_MAX_MEM_ALLOC_SIZE (128*1024*1024)
1122
1123/* accounting object for the main memory */
1124static pocl_global_mem_t system_memory = {POCL_LOCK_INITIALIZER, 0, 0, 0};
1125
1126void
1127pocl_setup_device_for_system_memory (cl_device_id device)
1128{
1129 /* set up system memory limits, if required */
1130 if (system_memory.total_alloc_limit == 0)
1131 {
1132 /* global_mem_size contains the entire memory size,
1133 * and we need to leave some available for OS & other programs
1134 * this sets it to 3/4 for systems with <=7gig mem,
1135 * for >7 it sets to (total-2gigs)
1136 */
1137 cl_ulong alloc_limit = device->global_mem_size;
1138 if (alloc_limit > ((cl_ulong)7 << 30))
1139 system_memory.total_alloc_limit = alloc_limit - ((cl_ulong)2 << 30);
1140 else
1141 {
1142 cl_ulong temp = (alloc_limit >> 2);
1143 system_memory.total_alloc_limit = alloc_limit - temp;
1144 }
1145
1146 system_memory.max_ever_allocated =
1147 system_memory.currently_allocated = 0;
1148
1149 /* in some cases (e.g. ARM32 pocl on ARM64 system with >4G ram),
1150 * global memory is correctly reported but larger than can be
1151 * used; limit to pointer size */
1152 if (system_memory.total_alloc_limit > UINTPTR_MAX)
1153 system_memory.total_alloc_limit = UINTPTR_MAX;
1154
1155 /* apply rlimit settings */
1156#ifdef HAVE_GETRLIMIT
1157 struct rlimit limits;
1158 int ret = getrlimit (RLIMIT_DATA, &limits);
1159 if ((ret == 0) && (system_memory.total_alloc_limit > limits.rlim_cur))
1160 system_memory.total_alloc_limit = limits.rlim_cur;
1161#endif
1162 }
1163
1164 device->global_mem_size = system_memory.total_alloc_limit;
1165
1166 int limit_memory_gb = pocl_get_int_option ("POCL_MEMORY_LIMIT", 0);
1167 if (limit_memory_gb > 0)
1168 {
1169 cl_ulong limited_memory = (cl_ulong)limit_memory_gb << 30;
1170 if (device->global_mem_size > limited_memory)
1171 device->global_mem_size = limited_memory;
1172 else
1173 POCL_MSG_WARN ("requested POCL_MEMORY_LIMIT %i GBs is larger than"
1174 " physical memory size (%u) GBs, ignoring\n",
1175 limit_memory_gb,
1176 (unsigned)(device->global_mem_size >> 30));
1177 }
1178
1179 if (device->global_mem_size < MIN_MAX_MEM_ALLOC_SIZE)
1180 POCL_ABORT("Not enough memory to run on this device.\n");
1181
1182 /* Maximum allocation size: we don't have hardware limits, so we
1183 * can potentially allocate the whole memory for a single buffer, unless
1184 * of course there are limits set at the operating system level. Of course
1185 * we still have to respect the OpenCL-commanded minimum */
1186
1187 cl_ulong alloc_limit = pocl_size_ceil2_64 (device->global_mem_size / 4);
1188
1189 if (alloc_limit < MIN_MAX_MEM_ALLOC_SIZE)
1190 alloc_limit = MIN_MAX_MEM_ALLOC_SIZE;
1191
1192 // set up device properties..
1193 device->global_memory = &system_memory;
1194 device->max_mem_alloc_size = alloc_limit;
1195
1196 // TODO in theory now if alloc_limit was > rlim_cur and < rlim_max
1197 // we should try and setrlimit to alloc_limit, or allocations might fail
1198}

1207/* set maximum allocation sizes for buffers and images */
1208void
1209pocl_set_buffer_image_limits(cl_device_id device)
1210{
1211 pocl_setup_device_for_system_memory(device);
1212 /* these aren't set up in pocl_setup_device_for_system_memory,
1213 * because some devices (HSA) set them up themselves
1214 *
1215 * it's max mem alloc / 4 because some programs (conformance test)
1216 * try to allocate max size constant objects and run out of memory
1217 * while trying to fill them. */
1218
1219 cl_ulong s;
1220 if (device->global_mem_cache_size > 0)
1221 s = pocl_size_ceil2_64 (device->global_mem_cache_size / 2);
1222 else
1223 s = pocl_size_ceil2_64 (device->global_mem_size / 256);
1224
1225 device->local_mem_size = device->max_constant_buffer_size = s;
1226
1227 /* We don't have hardware limitations on the buffer-backed image sizes,
1228 * so we set the maximum size in terms of the maximum amount of pixels
1229 * that fix in max_mem_alloc_size. A single pixel can take up to 4 32-bit channels,
1230 * i.e. 16 bytes.
1231 */
1232 size_t max_pixels = device->max_mem_alloc_size/16;
1233 if (max_pixels > device->image_max_buffer_size)
1234 device->image_max_buffer_size = max_pixels;
1235
1236 /* Similarly, we can take the 2D image size limit to be the largest power of 2
1237 * whose square fits in image_max_buffer_size; since the 2D image size limit
1238 * starts at a power of 2, it's a simple matter of doubling.
1239 * This is actually completely arbitrary, another equally valid option
1240 * would be to have each maximum dimension match the image_max_buffer_size.
1241 */
1242 max_pixels = device->image2d_max_width;
1243 // keep doubing until we go over
1244 while (max_pixels <= device->image_max_buffer_size/max_pixels)
1245 max_pixels *= 2;
1246 // halve before assignment
1247 max_pixels /= 2;
1248 if (max_pixels > device->image2d_max_width)
1249 device->image2d_max_width = device->image2d_max_height = max_pixels;
1250
1251 /* Same thing for 3D images, of course with cubes. Again, totally arbitrary. */
1252 max_pixels = device->image3d_max_width;
1253 // keep doubing until we go over
1254 while (max_pixels*max_pixels <= device->image_max_buffer_size/max_pixels)
1255 max_pixels *= 2;
1256 // halve before assignment
1257 max_pixels /= 2;
1258 if (max_pixels > device->image3d_max_width)
1259 device->image3d_max_width = device->image3d_max_height =
1260 device->image3d_max_depth = max_pixels;
1261
1262}

1264void*
1265pocl_aligned_malloc_global_mem(cl_device_id device, size_t align, size_t size)
1266{
1267 pocl_global_mem_t *mem = device->global_memory;
1268 void *retval = NULL;
1269
1270 POCL_LOCK (mem->pocl_lock);
1271 if ((mem->total_alloc_limit - mem->currently_allocated) < size)
1272 goto ERROR;
1273
1274 retval = pocl_aligned_malloc (align, size);
1275 if (!retval)
1276 goto ERROR;
1277
1278 mem->currently_allocated += size;
1279 if (mem->max_ever_allocated < mem->currently_allocated)
1280 mem->max_ever_allocated = mem->currently_allocated;
1281 assert(mem->currently_allocated <= mem->total_alloc_limit);
1282
1283ERROR:
1284 POCL_UNLOCK (mem->pocl_lock);
1285
1286 return retval;
1287}

1318 specialization properties. */
1319void
1320pocl_calculate_kernel_hash (cl_program program, unsigned kernel_i,
1321 unsigned device_i)
1322{
1323 SHA1_CTX hash_ctx;
1324 pocl_SHA1_Init (&hash_ctx);
1325
1326 char *n = program->kernel_meta[kernel_i].name;
1327 pocl_SHA1_Update (&hash_ctx, (uint8_t *)program->build_hash[device_i],
1328 sizeof (SHA1_digest_t));
1329 pocl_SHA1_Update (&hash_ctx, (uint8_t *)n, strlen (n));
1330
1331 uint8_t digest[SHA1_DIGEST_SIZE];
1332 pocl_SHA1_Final (&hash_ctx, digest);
1333
1334 memcpy (program->kernel_meta[kernel_i].build_hash[device_i], digest,
1335 sizeof (pocl_kernel_hash_t));
1336}
