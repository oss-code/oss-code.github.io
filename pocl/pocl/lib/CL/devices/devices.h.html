25#ifndef POCL_DEVICES_H
26#define POCL_DEVICES_H
27
28#include "../pocl_cl.h"
29
30#ifdef __GNUC__
31#pragma GCC visibility push(hidden)
32#endif
33
34#ifdef __cplusplus
35extern "C" {
36#endif
37
38/* The number of available devices. */
39extern unsigned int pocl_num_devices;
40
41/**
42 * Populates the pocl_devices with the wanted device types.
43 *
44 * Should be called before accessing the device list. Can be called repeatedly.
45 * The devices are shared across contexts, thus must implement resource
46 * management internally also across multiple contexts.
47 */
48cl_int pocl_init_devices();
49
50cl_int pocl_uninit_devices ();
51
52/**
53 * \brief Get the count of devices for a specific type
54 * \param device_type the device type for which we want the count of devices
55 * \return the count of devices for this type
56 */
57unsigned int pocl_get_device_type_count(cl_device_type device_type);
58
59/**
60 * \brief Get a certain amount of devices for a specific type
61 * \param type Type of devices wanted
62 * \param devcies Array of pointer to devices
63 * \param num_devices Number of devices queried
64 * \return The real number of devices added to devcies array which match the specified type
65 */
66unsigned int pocl_get_devices(cl_device_type device_type, struct _cl_device_id **devices, unsigned int num_devices);
67
68/**
69 * \brief Return the count of a specific device in the env var POCL_DEVICES
70 * \param dev_type a string describing the device ("basic" for instance)
71 * \return If the env var was not set, return -1, if the env var is specified, return 0
72 * or the number of occurence of dev_type in the env var
73 */
74int pocl_device_get_env_count(const char *dev_type);
75
76
77/**
78 * \brief Unique global memory id for devices with distinct memory from the system memory
79 * \return Unique global mem id, id > 0. Zero is reserved for shared system memory
80 */
81int pocl_get_unique_global_mem_id();
82
83#ifdef __GNUC__
84#pragma GCC visibility pop
85#endif
86
87/* the environment variable that lists the enabled devices */
88#define POCL_DEVICES_ENV "POCL_DEVICES"
89
90#ifdef __cplusplus
91} /* extern "C" */
92#endif
93
94#endif /* POCL_DEVICES_H */
