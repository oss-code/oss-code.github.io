117/**
118 * Get the number of specified devices from environnement
119 */
120int pocl_device_get_env_count(const char *dev_type)
121{
122 const char *dev_env = getenv(POCL_DEVICES_ENV);
123 char *ptr, *saveptr = NULL, *tofree, *token;
124 unsigned int dev_count = 0;
125 if (dev_env == NULL)
126 {
127 return -1;
128 }
129 ptr = tofree = strdup(dev_env);
130 while ((token = strtok_r (ptr, " ", &saveptr)) != NULL)
131 {
132 if(strcmp(token, dev_type) == 0)
133 dev_count++;
134 ptr = NULL;
135 }
136 POCL_MEM_FREE(tofree);
137
138 return dev_count;
139}

141unsigned int
142pocl_get_devices(cl_device_type device_type, struct _cl_device_id **devices, unsigned int num_devices)
143{
144 unsigned int i, dev_added = 0;
145
146 int offline_compile = pocl_get_bool_option("POCL_OFFLINE_COMPILE", 0);
147
148 for (i = 0; i < pocl_num_devices; ++i)
149 {
150 if (!offline_compile && (pocl_devices[i].available != CL_TRUE))
151 continue;
152
153 if (device_type == CL_DEVICE_TYPE_DEFAULT)
154 {
155 devices[dev_added] = &pocl_devices[i];
156 ++dev_added;
157 break;
158 }
159
160 if (pocl_devices[i].type & device_type)
161 {
162 if (dev_added < num_devices)
163 {
164 devices[dev_added] = &pocl_devices[i];
165 ++dev_added;
166 }
167 else
168 {
169 break;
170 }
171 }
172 }
173 return dev_added;
174}

176unsigned int
177pocl_get_device_type_count(cl_device_type device_type)
178{
179 int count = 0;
180 unsigned int i;
181
182 int offline_compile = pocl_get_bool_option("POCL_OFFLINE_COMPILE", 0);
183
184 for (i = 0; i < pocl_num_devices; ++i)
185 {
186 if (!offline_compile && (pocl_devices[i].available != CL_TRUE))
187 continue;
188
189 if (device_type == CL_DEVICE_TYPE_DEFAULT)
190 return 1;
191
192 if (pocl_devices[i].type & device_type)
193 {
194 ++count;
195 }
196 }
197
198 return count;
199}

202static inline void
203str_toupper(char *out, const char *in)
204{
205 int i;
206
207 for (i = 0; in[i] != '\0'; i++)
208 out[i] = toupper(in[i]);
209 out[i] = '\0';

212/* This ugly hack is required because:
213 *
214 * OpenCL 1.2 specification, 6.3 Operators :
215 *
216 * A divide by zero with integer types does not cause an exception
217 * but will result in an unspecified value. Division by zero for
218 * floating-point types will result in ±infinity or NaN as
219 * prescribed by the IEEE-754 standard.
220 *
221 * FPU exceptions are masked by default on x86 linux, but integer divide
222 * is not and there doesn't seem any sane way to mask it.
223 *
224 * This *might* be possible to fix with a LLVM pass (either check divisor
225 * for 0, or perhaps some vector extension has a suitable instruction), but
226 * it's likely to ruin the performance.
227 */
228
229#ifdef ENABLE_HOST_CPU_DEVICES
230#ifdef __linux__
231#ifdef __x86_64__
232
233#define DIV_OPCODE_SIZE 1
234#define DIV_OPCODE_MASK 0xf6
235
236/* F6 /6, F6 /7, F7 /6, F7 /7 */
237#define DIV_OPCODE_1 0xf6
238#define DIV_OPCODE_2 0xf7
239#define DIV_MODRM_OPCODE_EXT_1 0x38 // /7
240#define DIV_MODRM_OPCODE_EXT_2 0x30 // /6
241
242#define MODRM_SIZE 1
243#define MODRM_MASK 0xC0
244#define REG2_MASK 0x38
245#define REG1_MASK 0x07
246#define ADDR_MODE_INDIRECT_ONE_BYTE_OFFSET 0x40
247#define ADDR_MODE_INDIRECT_FOUR_BYTE_OFFSET 0x80
248#define ADDR_MODE_INDIRECT 0x0
249#define ADDR_MODE_REGISTER_ONLY 0xC0
250#define REG_SP 0x4
251#define REG_BP 0x5
252#define SIB_BYTE 1
253#define IP_RELATIVE_INDEXING 4
254
255static struct sigaction sigfpe_action, old_sigfpe_action;
256
257static void
258sigfpe_signal_handler (int signo, siginfo_t *si, void *data)
259{
260 ucontext_t *uc;
261 uc = (ucontext_t *)data;
262 unsigned char *eip = (unsigned char *)(uc->uc_mcontext.gregs[REG_RIP]);
263
264 if ((signo == SIGFPE)
265 && ((si->si_code == FPE_INTDIV) || (si->si_code == FPE_INTOVF)))
266 {
267 /* Luckily for us, div-by-0 exceptions do NOT advance the IP register,
268 * so we have to disassemble the instruction (to know its length)
269 * and move IP past it. */
270 unsigned n = 0;
271
272 /* skip all prefixes */
273 while ((n < 4) && ((eip[n] & DIV_OPCODE_MASK) != DIV_OPCODE_MASK))
274 ++n;
275
276 /* too much prefixes = decoding failed */
277 if (n >= 4)
278 goto ORIGINAL_HANDLER;
279
280 /* check opcode */
281 unsigned opcode = eip[n];
282 if ((opcode != DIV_OPCODE_1) && (opcode != DIV_OPCODE_2))
283 goto ORIGINAL_HANDLER;
284 n += DIV_OPCODE_SIZE;
285
286 unsigned modrm = eip[n];
287 unsigned modmask = modrm & MODRM_MASK;
288 unsigned reg1mask = modrm & REG1_MASK;
289 unsigned reg2mask = modrm & REG2_MASK;
290 /* check opcode extension in ModR/M reg2 */
291 if ((reg2mask != DIV_MODRM_OPCODE_EXT_1)
292 && (reg2mask != DIV_MODRM_OPCODE_EXT_2))
293 goto ORIGINAL_HANDLER;
294 n += MODRM_SIZE;
295
296 /* handle immediates/registers */
297 if (modmask == ADDR_MODE_INDIRECT_ONE_BYTE_OFFSET)
298 n += 1;
299 if (modmask == ADDR_MODE_INDIRECT_FOUR_BYTE_OFFSET)
300 n += 4;
301 if (modmask == ADDR_MODE_INDIRECT)
302 n += 0;
303 if (modmask != ADDR_MODE_REGISTER_ONLY)
304 {
305 if (reg1mask == REG_SP)
306 n += SIB_BYTE;
307 if (reg1mask == REG_BP)
308 n += IP_RELATIVE_INDEXING;
309 }
310
311 uc->uc_mcontext.gregs[REG_RIP] += n;
312 return;
313 }
314 else
315 {
316 ORIGINAL_HANDLER:
317 (*old_sigfpe_action.sa_sigaction) (signo, si, data);
318 }
319}

321#endif
322#endif

325cl_int
326pocl_uninit_devices ()
327{
328 cl_int retval = CL_SUCCESS;
329
330 POCL_LOCK (pocl_init_lock);
331 if ((!devices_active) || (pocl_num_devices == 0))
332 goto FINISH;
333
334 POCL_MSG_PRINT_GENERAL ("UNINIT all devices\n");
335
336 unsigned i, j, dev_index;
337
338 dev_index = 0;
339 cl_device_id d;
340 for (i = 0; i < POCL_NUM_DEVICE_TYPES; ++i)
341 {
342 assert (pocl_device_ops[i].init);
343 for (j = 0; j < device_count[i]; ++j)
344 {
345 d = &pocl_devices[dev_index];
346 if (d->available == 0)
347 continue;
348 if (d->ops->reinit == NULL || d->ops->uninit == NULL)
349 continue;
350 cl_int ret = d->ops->uninit (j, d);
351 if (ret != CL_SUCCESS)
352 {
353 retval = ret;
354 goto FINISH;
355 }
356
357 ++dev_index;
358 }
359 }
360
361FINISH:
362 devices_active = 0;
363 POCL_UNLOCK (pocl_init_lock);
364
365 return retval;
366}

368static cl_int
369pocl_reinit_devices ()
370{
371 assert (first_init_done);
372 cl_int retval = CL_SUCCESS;
373
374 if (devices_active)
375 return retval;
376
377 if (pocl_num_devices == 0)
378 return CL_DEVICE_NOT_FOUND;
379
380 POCL_MSG_WARN ("REINIT all devices\n");
381
382 unsigned i, j, dev_index;
383
384 dev_index = 0;
385 cl_device_id d;
386 /* Init infos for each probed devices */
387 for (i = 0; i < POCL_NUM_DEVICE_TYPES; ++i)
388 {
389 assert (pocl_device_ops[i].init);
390 for (j = 0; j < device_count[i]; ++j)
391 {
392 d = &pocl_devices[dev_index];
393 if (d->available == 0)
394 continue;
395 if (d->ops->reinit == NULL || d->ops->uninit == NULL)
396 continue;
397 cl_int ret = d->ops->reinit (j, d);
398 if (ret != CL_SUCCESS)
399 {
400 retval = ret;
401 goto FINISH;
402 }
403
404 ++dev_index;
405 }
406 }
407
408FINISH:
409
410 devices_active = 1;
411 return retval;
412}

414cl_int
415pocl_init_devices ()
416{
417 int errcode = CL_SUCCESS;
418
419 /* This is a workaround to a nasty problem with libhwloc: When
420 initializing basic, it calls libhwloc to query device info.
421 In case libhwloc has the OpenCL plugin installed, it initializes
422 it and it leads to initializing pocl again which leads to an
423 infinite loop. This only protects against recursive calls of
424 pocl_init_devices(), so must be done without pocl_init_lock held. */
425 if (init_in_progress)
426 return CL_SUCCESS; /* debatable, but what else can we do ? */
427
428 POCL_LOCK (pocl_init_lock);
429 init_in_progress = 1;
430
431 if (first_init_done)
432 {
433 if (!devices_active)
434 {
435 POCL_MSG_PRINT_GENERAL ("FIRST INIT done; REINIT all devices\n");
436 pocl_reinit_devices (); // TODO err check
437 }
438 errcode = pocl_num_devices ? CL_SUCCESS : CL_DEVICE_NOT_FOUND;
439 goto ERROR;
440 }
441
442 /* first time initialization */
443 unsigned i, j, dev_index;
444 char env_name[1024];
445 char dev_name[MAX_DEV_NAME_LEN] = { 0 };
446
447 /* Set a global debug flag, so we don't have to call pocl_get_bool_option
448 * everytime we use the debug macros */
449#ifdef POCL_DEBUG_MESSAGES
450 const char* debug = pocl_get_string_option ("POCL_DEBUG", "0");
451 pocl_debug_messages_setup (debug);
452 stderr_is_a_tty = isatty(fileno(stderr));
453#endif
454
455 POCL_GOTO_ERROR_ON ((pocl_cache_init_topdir ()), CL_DEVICE_NOT_FOUND,
456 "Cache directory initialization failed");
457
458 pocl_tracing_init ();
459
460#ifdef HAVE_SLEEP
461 int delay = pocl_get_int_option ("POCL_STARTUP_DELAY", 0);
462 if (delay > 0)
463 sleep (delay);
464#endif
465
466
467#ifdef ENABLE_HOST_CPU_DEVICES
468#ifdef __linux__
469#ifdef __x86_64__
470
471 if (pocl_get_bool_option ("POCL_SIGFPE_HANDLER", 1))
472 {
473
474#ifdef OCS_AVAILABLE
475 /* This is required to force LLVM to register its signal
476 * handlers, before pocl registers its own SIGFPE handler.
477 * LLVM otherwise calls this via
478 * pocl_llvm_build_program ->
479 * clang::PrintPreprocessedAction ->
480 * CreateOutputFile -> RemoveFileOnSignal
481 * Registering our handlers before LLVM creates its sigaltstack
482 * leads to interesting crashes & bugs later.
483 */
484 char random_empty_file[POCL_FILENAME_LENGTH];
485 pocl_cache_tempname (random_empty_file, NULL, NULL);
486 pocl_llvm_remove_file_on_signal (random_empty_file);
487#endif
488
489 POCL_MSG_PRINT_GENERAL ("Installing SIGFPE handler...\n");
490 sigfpe_action.sa_flags = SA_RESTART | SA_SIGINFO;
491 sigfpe_action.sa_sigaction = sigfpe_signal_handler;
492 int res = sigaction (SIGFPE, &sigfpe_action, &old_sigfpe_action);
493 assert (res == 0);
494 }
495
496#endif
497#endif
498#endif
499
500 /* Init operations */
501 for (i = 0; i < POCL_NUM_DEVICE_TYPES; ++i)
502 {
503 pocl_devices_init_ops[i](&pocl_device_ops[i]);
504 assert(pocl_device_ops[i].device_name != NULL);
505
506 /* Probe and add the result to the number of probed devices */
507 assert(pocl_device_ops[i].probe);
508 device_count[i] = pocl_device_ops[i].probe(&pocl_device_ops[i]);
509 pocl_num_devices += device_count[i];
510 }
511
512 const char *dev_env = pocl_get_string_option ("POCL_DEVICES", NULL);
513 POCL_GOTO_ERROR_ON ((pocl_num_devices == 0), CL_DEVICE_NOT_FOUND,
514 "no devices found. POCL_DEVICES=%s\n", dev_env);
515
516 pocl_devices = (struct _cl_device_id*) calloc(pocl_num_devices, sizeof(struct _cl_device_id));
517 POCL_GOTO_ERROR_ON ((pocl_devices == NULL), CL_OUT_OF_HOST_MEMORY,
518 "Can not allocate memory for devices\n");
519
520 dev_index = 0;
521 /* Init infos for each probed devices */
522 for (i = 0; i < POCL_NUM_DEVICE_TYPES; ++i)
523 {
524 str_toupper (dev_name, pocl_device_ops[i].device_name);
525 assert(pocl_device_ops[i].init);
526 for (j = 0; j < device_count[i]; ++j)
527 {
528 cl_device_id dev = &pocl_devices[dev_index];
529 dev->ops = &pocl_device_ops[i];
530 dev->dev_id = dev_index;
531 POCL_INIT_OBJECT (dev);
532 dev->driver_version = PACKAGE_VERSION;
533 if (dev->version == NULL)
534 dev->version = "OpenCL 2.0 pocl";
535 dev->short_name = strdup (dev->ops->device_name);
536 /* The default value for the global memory space identifier is
537 the same as the device id. The device instance can then override
538 it to point to some other device's global memory id in case of
539 a shared global memory. */
540 pocl_devices[dev_index].global_mem_id = dev_index;
541
542 /* Check if there are device-specific parameters set in the
543 POCL_DEVICEn_PARAMETERS env. */
544 POCL_GOTO_ERROR_ON (
545 (snprintf (env_name, 1024, "POCL_%s%d_PARAMETERS", dev_name, j)
546 < 0),
547 CL_OUT_OF_HOST_MEMORY, "Unable to generate the env string.");
548 errcode = pocl_devices[dev_index].ops->init (
549 j, &pocl_devices[dev_index], getenv (env_name));
550 POCL_GOTO_ERROR_ON ((errcode != CL_SUCCESS), CL_DEVICE_NOT_AVAILABLE,
551 "Device %i / %s initialization failed!", j,
552 dev_name);
553
554 ++dev_index;
555 }
556 }
557
558 first_init_done = 1;
559 devices_active = 1;
560ERROR:
561 init_in_progress = 0;
562 POCL_UNLOCK (pocl_init_lock);
563 return errcode;
564}
