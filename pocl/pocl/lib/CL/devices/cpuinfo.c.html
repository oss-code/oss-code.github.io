72/**
73 * Read the CPU maximum frequency from the Linux cpufreq interface.
74 * If cpufreq is not available on current host processor, returns -1.
75 *
76 * @return CPU maximum frequency in MHz, or -1 if cpufreq is not avaialble.
77 */
78static int
79pocl_cpufreq_get_max()
80{
81 int retval=-1;
82 if (access (cpufreq_file, R_OK) != 0)
83 return -1;
84
85 FILE *f = fopen (cpufreq_file, "r");
86 int nread = fscanf(f, "%d", &retval);
87 fclose(f);
88 if (nread < 1)
89 return -1;
90
91 // KHz to MHz
92 retval /= 1000;
93#ifdef DEBUG_POCL_CPUINFO
94 printf("CPU max frequency (from cpufreq): %d\n", retval);
95#endif
96 return retval;
97}

99/**
100 * Detects the maximum clock frequency of the CPU by parsing the cpuinfo.
101 *
102 * Assumes all cores have the same max clock freq. On some platforms,
103 * /proc/cpuinfo does not provide max CPU frequecny (ARM pandaboard),
104 * others give the *current* frequency, not the max (x86_64).
105 *
106 * @return The clock frequency in MHz, or -1 if couldn't figure it out.
107 */
108static int
109pocl_cpuinfo_detect_max_clock_frequency()
110{
111 int cpufreq=-1;
112
113 // First try to get the result from cpufreq interface.
114 cpufreq = pocl_cpufreq_get_max();
115 if( cpufreq != -1 )
116 return cpufreq;
117
118 if (access (cpuinfo, R_OK) != 0)
119 return -1;
120 else
121 {
122 FILE *f = fopen (cpuinfo, "r");
123 char contents[MAX_CPUINFO_SIZE];
124 int num_read = fread (contents, 1, MAX_CPUINFO_SIZE - 1, f);
125 float freq = 0.0f;
126 fclose (f);
127 contents[num_read] = '\0';
128
129 /* Count the number of times 'processor' keyword is found which
130 should give the number of cores overall in a multiprocessor
131 system. In Meego Harmattan on ARM it prints Processor instead of
132 processor */
133 char* p = contents;
134 if ((p = strstr (p, FREQSTRING)) != NULL &&
135 (p = strstr (p, ": ")) != NULL)
136 {
137 if (sscanf (p, ": %f", &freq) == 0)
138 {
139#ifdef DEBUG_POCL_CPUINFO
140 printf ("could not parse the cpu MHz field %f\n", freq);
141 puts (p);
142#endif
143 return -1;
144 }
145 else
146 {
147#ifdef DEBUG_POCL_CPUINFO
148 printf ("max_freq %d\n", (int)freq);
149#endif
150 return (int)freq;
151 }
152 }
153 }
154 return -1;
155}

158/**
159 * Detects the number of parallel hardware threads supported by
160 * the CPU by parsing the cpuinfo.
161 *
162 * @return The number of hardware threads, or -1 if couldn't figure it out.
163 */
164static int
165pocl_cpuinfo_detect_compute_unit_count()
166{
167 if (access (cpuinfo, R_OK) != 0)
168 return -1;
169 else
170 {
171 FILE *f = fopen (cpuinfo, "r");
172 char contents[MAX_CPUINFO_SIZE];
173 int num_read = fread (contents, 1, MAX_CPUINFO_SIZE - 1, f);
174 int cores = 0;
175 fclose (f);
176 contents[num_read] = '\0';
177
178 /* Count the number of times 'processor' keyword is found which
179 should give the number of cores overall in a multiprocessor
180 system. In Meego Harmattan on ARM it prints Processor instead of
181 processor */
182 char* p = contents;
183 while ((p = strstr (p, "rocessor")) != NULL)
184 {
185 cores++;
186 /* Skip to the end of the line. Otherwise causes two cores
187 to be detected in case of, for example:
188 Processor : ARMv7 Processor rev 2 (v7l) */
189 char* eol = strstr (p, "\n");
190 if (eol != NULL)
191 p = eol;
192 ++p;
193 }
194#ifdef DEBUG_POCL_CPUINFO
195 printf("total cores %d\n", cores);
196#endif
197 if (cores == 0)
198 return -1;
199
200 int cores_per_cpu = 1;
201 p = contents;
202 if ((p = strstr (p, "cpu cores")) != NULL)
203 {
204 if (sscanf (p, ": %d\n", &cores_per_cpu) != 1)
205 cores_per_cpu = 1;
206#ifdef DEBUG_POCL_CPUINFO
207 printf ("cores per cpu %d\n", cores_per_cpu);
208#endif
209 }
210
211 int siblings = 1;
212 p = contents;
213 if ((p = strstr (p, "siblings")) != NULL)
214 {
215 if (sscanf (p, ": %d\n", &siblings) != 1)
216 siblings = cores_per_cpu;
217#ifdef DEBUG_POCL_CPUINFO
218 printf ("siblings %d\n", siblings);
219#endif
220 }
221 if (siblings > cores_per_cpu) {
222#ifdef DEBUG_POCL_CPUINFO
223 printf ("max threads %d\n", cores*(siblings/cores_per_cpu));
224#endif
225 return cores*(siblings/cores_per_cpu); /* hardware threading is on */
226 } else {
227#ifdef DEBUG_POCL_CPUINFO
228 printf ("max threads %d\n", cores);
229#endif
230 return cores; /* only multicore, if not unicore*/
231 }
232 }
233 return -1;
234}

292static void
293pocl_cpuinfo_get_cpu_name_and_vendor(cl_device_id device)
294{
295 /* If something fails here, have this as backup solution.
296 * short_name is in the .data anyways.*/
297 device->long_name = device->short_name;
298
299 /* default vendor and vendor_id, in case it cannot be found by other means */
300 device->vendor = cpuvendor_default;
301 if (device->vendor_id == 0)
302 device->vendor_id = DEFAULTVENDORID;
303
304 /* read contents of /proc/cpuinfo */
305 if (access (cpuinfo, R_OK) != 0)
306 return;
307
308 FILE *f = fopen (cpuinfo, "r");
309 char contents[MAX_CPUINFO_SIZE];
310 int num_read = fread (contents, 1, MAX_CPUINFO_SIZE - 1, f);
311 fclose(f);
312 contents[num_read]='\0';
313
314 char const *start, *end;
315 /* find the vendor_id string an put */
316
317#if __arm__ || __aarch64__
318 unsigned vendor_id = -1;
319 size_t i;
320#endif
321#ifdef VENDORSTRING
322 do {
323 start = strstr(contents, VENDORSTRING"\t: ");
324 if (!start)
325 break;
326 start += strlen(VENDORSTRING"\t: ");
327 end = strchr(start, '\n');
328 if (!end)
329 break;
330
331#if __arm__ || __aarch64__
332 if (1 == sscanf (start, "%x", &vendor_id))
333 {
334 for (i = 0; i < sizeof (vendor_list) / sizeof (vendor_list[0]); ++i)
335 {
336 if (vendor_id == vendor_list[i].id)
337 {
338 start = vendor_list[i].name;
339 end = start + strlen (vendor_list[i].name);
340 break;
341 }
342 }
343 }
344#endif
345
346 char *_vendor = malloc(end-start + 1);
347 if (!_vendor)
348 break;
349 memcpy(_vendor, start, end-start);
350 _vendor[end-start] = '\0';
351 device->vendor = _vendor;
352 } while (0);
353#endif
354
355 /* find the cpu description */
356 start=strstr (contents, MODELSTRING"\t: ");
357 if (start == NULL)
358 return;
359 start += strlen (MODELSTRING"\t: ");
360 end = strchr (start, '\n');
361 if (end == NULL)
362 return;
363
364#if __arm__ || __aarch64__
365 unsigned part_id;
366 if (1 == sscanf (start, "%x", &part_id))
367 {
368 part_tuple_t const *part_list = NULL;
369 size_t part_count = 0;
370
371 switch (vendor_id)
372 {
373 case JEP106_ARM:
374 part_list = part_list_arm;
375 part_count = sizeof (part_list_arm) / sizeof (part_list_arm[0]);
376 break;
377 case JEP106_APM:
378 part_list = part_list_apm;
379 part_count = sizeof (part_list_apm) / sizeof (part_list_apm[0]);
380 break;
381 }
382
383 for (i = 0; i < part_count; ++i)
384 {
385 if (part_id == part_list[i].id)
386 {
387 start = part_list[i].name;
388 end = start + strlen (part_list[i].name);
389 break;
390 }
391 }
392 }
393#endif
394
395 /* create the descriptive long_name for device */
396 int len = strlen (device->short_name) + (end-start) + 2;
397 char *new_name = (char*)malloc (len);
398 snprintf (new_name, len, "%s-%s", device->short_name, start);
399 device->long_name = new_name;
400}

402/*
403 * Expects:
404 * short name
405 *
406 * Sets up:
407 * vendor_id
408 * vendor (name)
409 * long name
410 *
411 * max compute units IF NOT SET ALREADY
412 * max clock freq
413 */
414
415void
416pocl_cpuinfo_detect_device_info(cl_device_id device)
417{
418 int res;
419
420 if (device->max_compute_units == 0) {
421 res = pocl_cpuinfo_detect_compute_unit_count();
422 device->max_compute_units = (res > 0) ? (cl_uint)res : 0;
423 }
424
425 res = pocl_cpuinfo_detect_max_clock_frequency();
426 device->max_clock_frequency = (res > 0) ? (cl_uint)res : 0;
427
428 pocl_cpuinfo_get_cpu_name_and_vendor(device);
429}
