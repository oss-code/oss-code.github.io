24#include "pocl_cl.h"
25#include "pocl_util.h"
26
27CL_API_ENTRY cl_int CL_API_CALL
28POname(clEnqueueSVMMemFill) (cl_command_queue command_queue,
29 void *svm_ptr,
30 const void *pattern,
31 size_t pattern_size,
32 size_t size,
33 cl_uint num_events_in_wait_list,
34 const cl_event *event_wait_list,
35 cl_event *event) CL_API_SUFFIX__VERSION_2_0
36{
37 unsigned i;
38 cl_int errcode;
39
40 POCL_RETURN_ERROR_COND((command_queue == NULL), CL_INVALID_COMMAND_QUEUE);
41
42 POCL_RETURN_ERROR_ON((command_queue->context->svm_allocdev == NULL),
43 CL_INVALID_CONTEXT, "None of the devices in this context is SVM-capable\n");
44
45 POCL_RETURN_ERROR_COND((svm_ptr == NULL), CL_INVALID_VALUE);
46
47 POCL_RETURN_ERROR_COND((pattern_size == 0), CL_INVALID_VALUE);
48
49 POCL_RETURN_ERROR_COND((pattern_size > 128), CL_INVALID_VALUE);
50
51 POCL_RETURN_ERROR_ON((__builtin_popcount(pattern_size) > 1), CL_INVALID_VALUE,
52 "pattern_size (%zu) must be a power-of-2 value\n", pattern_size);
53
54 POCL_RETURN_ERROR_COND((size == 0), CL_INVALID_VALUE);
55
56 POCL_RETURN_ERROR_ON(((intptr_t)svm_ptr % pattern_size > 0), CL_INVALID_VALUE,
57 "svm_ptr must be aligned to pattern_size\n");
58
59 POCL_RETURN_ERROR_ON((size % pattern_size > 0), CL_INVALID_VALUE,
60 "size must be a multiple of pattern_size\n");
61
62 errcode = pocl_check_event_wait_list (command_queue, num_events_in_wait_list,
63 event_wait_list);
64 if (errcode != CL_SUCCESS)
65 return errcode;
66
67 for(i=0; i<num_events_in_wait_list; i++)
68 POCL_RETURN_ERROR_COND((event_wait_list[i] == NULL), CL_INVALID_EVENT_WAIT_LIST);
69
70 _cl_command_node *cmd = NULL;
71
72 errcode = pocl_create_command (&cmd, command_queue, CL_COMMAND_SVM_MEMFILL,
73 event, num_events_in_wait_list,
74 event_wait_list, 0, NULL);
75
76 if (errcode != CL_SUCCESS)
77 {
78 POCL_MEM_FREE(cmd);
79 return errcode;
80 }
81
82 void *p = pocl_aligned_malloc (pattern_size, pattern_size);
83 assert (p);
84 memcpy (p, pattern, pattern_size);
85
86 cmd->command.svm_fill.svm_ptr = svm_ptr;
87 cmd->command.svm_fill.size = size;
88 cmd->command.svm_fill.pattern = p;
89 cmd->command.svm_fill.pattern_size = pattern_size;
90
91 pocl_command_enqueue(command_queue, cmd);
92
93 return CL_SUCCESS;
94}
