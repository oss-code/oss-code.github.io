6POname(clEnqueueCopyImageToBuffer)(cl_command_queue command_queue ,
7 cl_mem src_image ,
8 cl_mem dst_buffer ,
9 const size_t * src_origin ,
10 const size_t * region ,
11 size_t dst_offset ,
12 cl_uint num_events_in_wait_list ,
13 const cl_event * event_wait_list ,
14 cl_event * event ) CL_API_SUFFIX__VERSION_1_0
15{
16 _cl_command_node *cmd = NULL;
17 /* pass dst_origin through in a format pocl_rect_copy understands */
18 const size_t dst_origin[3] = { dst_offset, 0, 0};
19
20 POCL_RETURN_ERROR_COND ((command_queue == NULL), CL_INVALID_COMMAND_QUEUE);
21
22 POCL_RETURN_ERROR_COND ((src_image == NULL), CL_INVALID_MEM_OBJECT);
23
24 if (IS_IMAGE1D_BUFFER (src_image))
25 {
26 /* If src_image is a 1D image or 1D image buffer object, src_origin[1]
27 * and src_origin[2] must be 0 If src_image is a 1D image or 1D image
28 * buffer object, region[1] and region[2] must be 1. */
29 IMAGE1D_ORIG_REG_TO_BYTES (src_image, src_origin, region);
30 return POname (clEnqueueCopyBufferRect (
31 command_queue, src_image->buffer, dst_buffer,
32 i1d_origin, dst_origin, i1d_region,
33 src_image->image_row_pitch, 0,
34 src_image->image_row_pitch, 0,
35 num_events_in_wait_list, event_wait_list, event));
36 }
37
38 POCL_RETURN_ON_SUB_MISALIGN (dst_buffer, command_queue);
39
40 cl_int err = pocl_rect_copy(
41 command_queue,
42 CL_COMMAND_COPY_IMAGE_TO_BUFFER,
43 src_image, CL_TRUE,
44 dst_buffer, CL_FALSE,
45 src_origin, dst_origin, region,
46 0, 0,
47 0, 0,
48 num_events_in_wait_list, event_wait_list,
49 event,
50 &cmd);
51
52 if (err != CL_SUCCESS)
53 return err;
54
55 POCL_CONVERT_SUBBUFFER_OFFSET (dst_buffer, dst_offset);
56
57 POCL_RETURN_ERROR_ON((dst_buffer->size > command_queue->device->max_mem_alloc_size),
58 CL_OUT_OF_RESOURCES,
59 "src is larger than device's MAX_MEM_ALLOC_SIZE\n");
60
61 cl_device_id dev = command_queue->device;
62
63 cmd->command.read_image.src_mem_id = &src_image->device_ptrs[dev->dev_id];
64 cmd->command.read_image.dst_host_ptr = NULL;
65 cmd->command.read_image.dst_mem_id = &dst_buffer->device_ptrs[dev->dev_id];
66
67 cmd->command.read_image.origin[0] = src_origin[0];
68 cmd->command.read_image.origin[1] = src_origin[1];
69 cmd->command.read_image.origin[2] = src_origin[2];
70 cmd->command.read_image.region[0] = region[0];
71 cmd->command.read_image.region[1] = region[1];
72 cmd->command.read_image.region[2] = region[2];
73 // TODO
74 cmd->command.read_image.dst_row_pitch = 0; // src_image->image_row_pitch;
75 cmd->command.read_image.dst_slice_pitch = 0; // src_image->image_slice_pitch;
76 cmd->command.read_image.dst_offset = dst_offset;
77
78 POname (clRetainMemObject) (src_image);
79 src_image->owning_device = dev;
80 POname (clRetainMemObject) (dst_buffer);
81 dst_buffer->owning_device = dev;
82
83 pocl_command_enqueue (command_queue, cmd);
84
85 return CL_SUCCESS;
86}
