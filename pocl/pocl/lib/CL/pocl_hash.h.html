33/* public api for steve reid's public domain SHA-1 implementation */
34/* this file is in the public domain */
35
36#include <stdint.h>
37
38typedef struct {
39 uint32_t state[5];
40 uint32_t count[2];
41 uint8_t buffer[64];
42} SHA1_CTX;
43
44#define SHA1_DIGEST_SIZE 20
45
46void pocl_SHA1_Init(SHA1_CTX* context);
47void pocl_SHA1_Update(SHA1_CTX* context, const uint8_t* data, const size_t len);
48void pocl_SHA1_Final(SHA1_CTX* context, uint8_t digest[SHA1_DIGEST_SIZE]);
49
50#ifdef __GNUC__
51#pragma GCC visibility pop
52#endif

55}
