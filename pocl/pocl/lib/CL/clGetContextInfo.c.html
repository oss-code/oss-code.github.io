25#include "pocl_util.h"

28CL_API_ENTRY cl_int CL_API_CALL
29POname(clGetContextInfo)(cl_context context,
30 cl_context_info param_name,
31 size_t param_value_size,
32 void *param_value,
33 size_t *param_value_size_ret) CL_API_SUFFIX__VERSION_1_0
34{
35 size_t value_size;
36
37 POCL_RETURN_ERROR_COND((context == NULL), CL_INVALID_CONTEXT);
38
39 switch (param_name) {
40 case CL_CONTEXT_REFERENCE_COUNT:
41 {
42 cl_uint refcount = context->pocl_refcount;
43 POCL_RETURN_GETINFO(cl_uint, refcount);
44 }
45 case CL_CONTEXT_DEVICES:
46 value_size = context->num_devices * sizeof(cl_device_id);
47 POCL_RETURN_GETINFO_SIZE(value_size, context->devices);
48 case CL_CONTEXT_NUM_DEVICES:
49 POCL_RETURN_GETINFO(cl_uint, context->num_devices);
50 case CL_CONTEXT_PROPERTIES:
51 if (context->properties)
52 {
53 value_size = (context->num_properties * 2 + 1) * sizeof(cl_context_properties);
54 POCL_RETURN_GETINFO_SIZE(value_size, context->properties);
55 }
56 else
57 {
58 *param_value_size_ret = 0;
59 return CL_SUCCESS;
60 }
61 default:
62 return CL_INVALID_VALUE;
63 }
64
65 return CL_SUCCESS;
66}
