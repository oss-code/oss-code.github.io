7#include "pocl_cl.h"
8#include "pocl_llvm.h"
9#include "pocl_intfn.h"

11
12CL_API_ENTRY cl_int CL_API_CALL
13POname(clCreateKernelsInProgram)(cl_program program ,
14 cl_uint num_kernels ,
15 cl_kernel * kernels ,
16 cl_uint * num_kernels_ret ) CL_API_SUFFIX__VERSION_1_0
17{
18 unsigned idx;
19
20 POCL_RETURN_ERROR_COND((program == NULL), CL_INVALID_PROGRAM);
21
22 POCL_RETURN_ERROR_ON((program->num_devices == 0),
23 CL_INVALID_PROGRAM, "Invalid program (has no devices assigned)\n");
24
25 POCL_RETURN_ERROR_ON((program->build_status == CL_BUILD_NONE),
26 CL_INVALID_PROGRAM_EXECUTABLE, "You must call clBuildProgram first!"
27 " (even for programs created with binaries)\n");
28
29 POCL_RETURN_ERROR_ON((program->build_status != CL_BUILD_SUCCESS),
30 CL_INVALID_PROGRAM_EXECUTABLE, "Last BuildProgram() was not successful\n");
31
32 POCL_RETURN_ERROR_ON((program->llvm_irs == NULL),
33 CL_INVALID_PROGRAM_EXECUTABLE, "No built binaries in program "
34 "(this shouldn't happen...)\n");
35
36 POCL_RETURN_ERROR_ON(((kernels != NULL && num_kernels == 0)
37 || (kernels == NULL && num_kernels != 0)),
38 CL_INVALID_VALUE, "kernels & num_kernels must be "
39 "either both set, or both NULL\n");
40
41 POCL_RETURN_ERROR_ON((kernels && num_kernels < program->num_kernels),
42 CL_INVALID_VALUE,
43 "kernels is not NULL and num_kernels "
44 "is less than the number of kernels in program\n");
45
46 for (idx = 0; idx < num_kernels; idx++)
47 kernels[idx] = NULL;
48
49 if (num_kernels > program->num_kernels)
50 num_kernels = program->num_kernels;
51
52 cl_int error_ret;
53 if (num_kernels > 0 && kernels != NULL)
54 {
55 /* Create the kernels in the 'knames' list */
56 for (idx = 0; idx < num_kernels; idx++)
57 {
58 kernels[idx] = POname(clCreateKernel) (program,
59 program->kernel_meta[idx].name,
60 &error_ret);
61
62 /* Check for errors, clean up & bail.
63 * If we happened to pass a invalid kernel name after all
64 * that should be treated as a pocl bug, not user error.
65 * TODO: what happens if the program is not valid?*/
66 assert(error_ret != CL_INVALID_KERNEL_NAME);
67 assert(error_ret != CL_INVALID_VALUE);
68 if (error_ret != CL_SUCCESS)
69 {
70 for (; idx>0; idx--)
71 POname(clReleaseKernel) (kernels[idx-1]);
72 /* If error_ret is INVALID_KERNEL_DEFINITION, returning it here
73 * is against the specification. But the specs doesn't say what to
74 * do in such a case, and just returning it is the sanest thing
75 * to do. */
76 return error_ret;
77 }
78 }
79 }
80
81 if (num_kernels_ret)
82 *num_kernels_ret = program->num_kernels;
83
84 return CL_SUCCESS;
85}
86POsym(clCreateKernelsInProgram)
