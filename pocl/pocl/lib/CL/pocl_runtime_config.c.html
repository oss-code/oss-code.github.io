31/* TODO: cache the options and to avoid calling getenv more than once
32 per option. */
33
34typedef struct env_data env_data;
35struct env_data
36{
37 char *env;
38 char *value;
39 env_data *next;
40};
41
42static env_data *volatile env_cache = 0;
43static pocl_lock_t lock = POCL_LOCK_INITIALIZER;
44
45static env_data* find_env (env_data* cache, const char* key)
46{
47 env_data* ed;
48 char *value;
49
50 POCL_LOCK(lock);
51 LL_FOREACH(cache, ed)
52 {
53 if (strcmp(ed->env, key) == 0)
54 {
55 POCL_UNLOCK(lock);
56 return ed;
57 }
58 }
59 if ((value = getenv(key)))
60 {
61 ed = (env_data*) malloc (sizeof (env_data));
62 ed->env = strdup (key);
63 ed->value = strdup (value);
64 ed->next = NULL;
65 LL_APPEND(env_cache, ed);
66 POCL_UNLOCK(lock);
67 return ed;
68 }
69
70 POCL_UNLOCK(lock);
71 return NULL;
72}
73/* Can be used to query if the given option has been set by the user. */
74int pocl_is_option_set(const char *key)
75{
76 env_data* ed = NULL;
77 return (ed = find_env (env_cache, key)) ? 1 : 0;
78}
79
80/* Returns an integer value for the option with the given key string. */
81int pocl_get_int_option(const char *key, int default_value)
82{
83 env_data *ed;
84 return (ed = find_env (env_cache, key)) ? atoi(ed->value) : default_value;
85}
86
87/* Returns a boolean value for the option with the given key string. */
88int pocl_get_bool_option(const char *key, int default_value)
89{
90 env_data *ed;
91 if ((ed = find_env(env_cache, key)))
92 return (strncmp(ed->value, "1", 1) == 0);
93 return default_value;
94}
95
96const char* pocl_get_string_option(const char *key, const char *default_value)
97{
98 env_data *ed;
99 return (ed = find_env (env_cache, key)) ? ed->value : default_value;
100}
