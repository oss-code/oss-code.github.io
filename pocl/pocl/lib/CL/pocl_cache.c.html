55static char cache_topdir[POCL_FILENAME_LENGTH];
56static char tempfile_pattern[POCL_FILENAME_LENGTH];
57static char tempdir_pattern[POCL_FILENAME_LENGTH];
58static int cache_topdir_initialized = 0;
59static int use_kernel_cache = 0;

61/* sanity check on SHA1 digest emptiness */
62static unsigned buildhash_is_valid(cl_program program, unsigned device_i)
63{
64 unsigned i, sum = 0;
65 for(i=0; i<sizeof(SHA1_digest_t); i++)
66 sum += program->build_hash[device_i][i];
67 return sum;
68}

70int pocl_cl_device_to_index(cl_program program,
71 cl_device_id device) {
72 unsigned i;
73 assert(program);
74 for (i = 0; i < program->num_devices; i++)
75 if (program->devices[i] == device ||
76 program->devices[i] == device->parent_device)
77 return i;
78 return -1;
79}

81static void program_device_dir(char *path,
82 cl_program program,
83 unsigned device_i,
84 const char* append_path)
85{
86 assert(path);
87 assert(program);
88 assert(device_i < program->num_devices);
89 assert(buildhash_is_valid(program, device_i));
90
91 int bytes_written = snprintf(path, POCL_FILENAME_LENGTH,
92 "%s/%s%s", cache_topdir,
93 program->build_hash[device_i],
94 append_path);
95 assert(bytes_written > 0 && bytes_written < POCL_FILENAME_LENGTH);
96}

113/* Return the cache directory for the given work-group function.
114 If specialized = 1, specialization parameters are derived from run_cmd,
115 otherwise a generic directory name is returned.
116
117 The current specialization parameters are:
118 - local size
119 - if the global offset is zero (in all dimensions) or not
120 - if the grid size in any dimension is smaller than a device
121 specified limit ("smallgrid" specialization)
122*/
123void
124pocl_cache_kernel_cachedir_path (char *kernel_cachedir_path,
125 cl_program program, unsigned device_i,
126 cl_kernel kernel, const char *append_str,
127 _cl_command_node *command, int specialized)
128{
129 int bytes_written;
130 _cl_command_run *run_cmd = &command->command.run;
131 char tempstring[POCL_FILENAME_LENGTH];
132 cl_device_id dev = command->device;
133 size_t max_grid_width = pocl_cmd_max_grid_dim_width (run_cmd);
134 bytes_written = snprintf (
135 tempstring, POCL_FILENAME_LENGTH, "/%s/%zu-%zu-%zu%s%s%s", kernel->name,
136 !specialized ? 0 : run_cmd->pc.local_size[0],
137 !specialized ? 0 : run_cmd->pc.local_size[1],
138 !specialized ? 0 : run_cmd->pc.local_size[2],
139 (specialized && run_cmd->pc.global_offset[0] == 0
140 && run_cmd->pc.global_offset[1] == 0
141 && run_cmd->pc.global_offset[2] == 0)
142 ? "-goffs0"
143 : "",
144 specialized && !run_cmd->force_large_grid_wg_func
145 && max_grid_width < dev->grid_width_specialization_limit
146 ? "-smallgrid"
147 : "",
148 append_str);
149 assert (bytes_written > 0 && bytes_written < POCL_FILENAME_LENGTH);
150
151 program_device_dir (kernel_cachedir_path, program, device_i, tempstring);
152}

154void
155pocl_cache_kernel_cachedir (char *kernel_cachedir_path, cl_program program,
156 unsigned device_i, const char *kernel_name)
157{
158 int bytes_written;
159 char tempstring[POCL_FILENAME_LENGTH];
160 bytes_written
161 = snprintf (tempstring, POCL_FILENAME_LENGTH, "/%s", kernel_name);
162 assert (bytes_written > 0 && bytes_written < POCL_FILENAME_LENGTH);
163 program_device_dir (kernel_cachedir_path, program, device_i, tempstring);
164}

180/* Return the final binary path for the given work-group function.
181 If specialized is 1, find the WG function specialized for the
182 given command's properties, if 0, return the path to a generic version. */
183void
184pocl_cache_final_binary_path (char *final_binary_path, cl_program program,
185 unsigned device_i, cl_kernel kernel,
186 _cl_command_node *command, int specialized)
187{
188 assert (kernel->name);
189
190 /* TODO: This should be probably refactored to either get the binary name
191 from the device itself, or let the device ops call
192 pocl_llvm_generate_workgroup_function() on their own */
193
194 int bytes_written;
195 char final_binary_name[POCL_FILENAME_LENGTH];
196
197 /* FIXME: Why different naming for SPMD and why the .brig suffix? */
198 if (kernel->program->devices[device_i]->spmd)
199 bytes_written = snprintf (final_binary_name, POCL_FILENAME_LENGTH,
200 "%s.brig", POCL_PARALLEL_BC_FILENAME);
201 else
202 bytes_written = snprintf (final_binary_name, POCL_FILENAME_LENGTH,
203 "/%s.so", kernel->name);
204
205 assert (bytes_written > 0 && bytes_written < POCL_FILENAME_LENGTH);
206
207 pocl_cache_kernel_cachedir_path (final_binary_path, program, device_i,
208 kernel, final_binary_name, command,
209 specialized);
210}

283int pocl_cache_write_descriptor(cl_program program,
284 unsigned device_i,
285 const char* kernel_name,
286 const char* content,
287 size_t size) {
288 char devdir[POCL_FILENAME_LENGTH];
289 program_device_dir(devdir, program, device_i, "");
290
291 char descriptor[POCL_FILENAME_LENGTH];
292 int bytes_written = snprintf(descriptor, POCL_FILENAME_LENGTH,
293 "%s/%s", devdir, kernel_name);
294 assert(bytes_written > 0 && bytes_written < POCL_FILENAME_LENGTH);
295 if (pocl_mkdir_p(descriptor))
296 return 1;
297
298 bytes_written = snprintf(descriptor, POCL_FILENAME_LENGTH,
299 "%s/%s/descriptor.so.kernel_obj.c",
300 devdir, kernel_name);
301 assert(bytes_written > 0 && bytes_written < POCL_FILENAME_LENGTH);
302
303 return pocl_write_file(descriptor, content, size, 0, 1);
304}

308char* pocl_cache_read_buildlog(cl_program program,
309 unsigned device_i) {
310 char buildlog_path[POCL_FILENAME_LENGTH];
311 if (program->build_hash[device_i][0] == 0)
312 return NULL;
313 program_device_dir(buildlog_path, program,
314 device_i, POCL_BUILDLOG_FILENAME);
315
316 if (!pocl_exists(buildlog_path))
317 return NULL;
318
319 char* res=NULL;
320 uint64_t filesize;
321 if (pocl_read_file(buildlog_path, &res, &filesize))
322 return NULL;
323 return res;
324}

364static inline void
365build_program_compute_hash (cl_program program, unsigned device_i,
366 const char *preprocessed_source, size_t source_len)
367{
368 SHA1_CTX hash_ctx;
369 unsigned i;
370 cl_device_id device = program->devices[device_i];
371
372 pocl_SHA1_Init(&hash_ctx);
373
374 if (program->source) {
375 assert(preprocessed_source);
376 assert(source_len > 0);
377 pocl_SHA1_Update(&hash_ctx, (uint8_t*)preprocessed_source,
378 source_len);
379 } else if (program->pocl_binary_sizes[device_i] > 0) {
380 /* Program was created with clCreateProgramWithBinary() with
381 a pocl binary */
382 assert(program->pocl_binaries[device_i]);
383 pocl_SHA1_Update(&hash_ctx,
384 (uint8_t*) program->pocl_binaries[device_i],
385 program->pocl_binary_sizes[device_i]);
386 }
387 else if (program->binary_sizes[device_i] > 0)
388 {
389 /* Program was created with clCreateProgramWithBinary() with an LLVM IR
390 * binary */
391 assert (program->binaries[device_i]);
392 pocl_SHA1_Update (&hash_ctx, (uint8_t *)program->binaries[device_i],
393 program->binary_sizes[device_i]);
394 }
395 else
396 {
397 /* Program is linked from binaries, has no source or binary */
398 // assert(program->binary_type == CL_PROGRAM_BIN)
399 assert (preprocessed_source);
400 assert (source_len > 0);
401 pocl_SHA1_Update (&hash_ctx, (uint8_t *)preprocessed_source,
402 source_len);
403 }
404
405 if (program->compiler_options)
406 pocl_SHA1_Update(&hash_ctx, (uint8_t*) program->compiler_options,
407 strlen(program->compiler_options));
408
409 /* The kernel compiler work-group function method affects the
410 produced binary heavily. */
411 const char *wg_method=
412 pocl_get_string_option("POCL_WORK_GROUP_METHOD", "");
413
414 pocl_SHA1_Update(&hash_ctx, (uint8_t*) wg_method, strlen(wg_method));
415 pocl_SHA1_Update(&hash_ctx, (uint8_t*) PACKAGE_VERSION,
416 strlen(PACKAGE_VERSION));
417 pocl_SHA1_Update(&hash_ctx, (uint8_t*) LLVM_VERSION,
418 strlen(LLVM_VERSION));
419 pocl_SHA1_Update(&hash_ctx, (uint8_t*) POCL_BUILD_TIMESTAMP,
420 strlen(POCL_BUILD_TIMESTAMP));
421 pocl_SHA1_Update(&hash_ctx, (const uint8_t *)POCL_KERNELLIB_SHA1,
422 strlen(POCL_KERNELLIB_SHA1));
423 /*devices may include their own information to hash */
424 if (device->ops->build_hash)
425 {
426 char *dev_hash = device->ops->build_hash(device);
427 pocl_SHA1_Update(&hash_ctx, (const uint8_t *)dev_hash, strlen(dev_hash));
428 free(dev_hash);
429 }
430
431 uint8_t digest[SHA1_DIGEST_SIZE];
432 pocl_SHA1_Final(&hash_ctx, digest);
433
434 unsigned char* hashstr = program->build_hash[device_i];
435 for (i=0; i < SHA1_DIGEST_SIZE; i++)
436 {
437 *hashstr++ = (digest[i] & 0x0F) + 65;
438 *hashstr++ = ((digest[i] & 0xF0) >> 4) + 65;
439 }
440 *hashstr = 0;
441
442 program->build_hash[device_i][2] = '/';
443
444}
445#endif

450int
451pocl_cache_init_topdir ()
452{
453 if (cache_topdir_initialized)
454 return 0;
455
456 use_kernel_cache
457 = pocl_get_bool_option ("POCL_KERNEL_CACHE", POCL_KERNEL_CACHE_DEFAULT);
458
459 const char *tmp_path = pocl_get_string_option ("POCL_CACHE_DIR", NULL);
460 int needed;
461
462 if (tmp_path)
463 {
464 needed = snprintf (cache_topdir, POCL_FILENAME_LENGTH, "%s", tmp_path);
465 } else {
466#ifdef __ANDROID__
467 POCL_ABORT ("Please set the POCL_CACHE_DIR env var to your app's cache directory (Context.getCacheDir())\n");
468
469#elif defined(_MSC_VER) || defined(__MINGW32__)
470 tmp_path = getenv("LOCALAPPDATA");
471 if (!tmp_path)
472 tmp_path = getenv("TEMP");
473 assert(tmp_path);
474 needed = snprintf(cache_topdir, POCL_FILENAME_LENGTH,
475 "%s\\pocl", tmp_path);
476#else
477 // "If $XDG_CACHE_HOME is either not set or empty, a default equal to
478 // $HOME/.cache should be used."
479 // http://standards.freedesktop.org/basedir-spec/latest/
480 tmp_path = getenv("XDG_CACHE_HOME");
481 const char *p;
482 if (use_kernel_cache)
483 p = "pocl/kcache";
484 else
485 p = "pocl/uncached";
486
487 if (tmp_path && tmp_path[0] != '\0') {
488 needed = snprintf (cache_topdir, POCL_FILENAME_LENGTH, "%s/%s",
489 tmp_path, p);
490 }
491 else if ((tmp_path = getenv("HOME")) != NULL) {
492 needed = snprintf (cache_topdir, POCL_FILENAME_LENGTH,
493 "%s/.cache/%s", tmp_path, p);
494 }
495 else {
496 needed
497 = snprintf (cache_topdir, POCL_FILENAME_LENGTH, "/tmp/%s", p);
498 }
499#endif
500 }
501
502 if (needed >= POCL_FILENAME_LENGTH)
503 {
504 POCL_MSG_ERR ("pocl: cache path longer than maximum filename length\n");
505 return 1;
506 }
507
508 assert(strlen(cache_topdir) > 0);
509
510 if (pocl_mkdir_p(cache_topdir))
511 {
512 POCL_MSG_ERR (
513 "Could not create top directory (%s) for cache. \n\nNote: "
514 "if you have proper rights to create that directory, and still "
515 "get the error, then most likely pocl and the program you're "
516 "trying to run are linked to different versions of libstdc++ "
517 "library. \nThis is not a bug in pocl and there's nothing we "
518 "can do to fix it - you need both pocl and your program to be"
519 " compiled for your system. This is known to happen with "
520 "Luxmark benchmark binaries dowloaded from website; Luxmark "
521 "installed from your linux distribution's packages should "
522 "work.\n",
523 cache_topdir);
524 return 1;
525 }
526
527 strncpy (tempfile_pattern, cache_topdir, POCL_FILENAME_LENGTH);
528 size_t len = strlen (tempfile_pattern);
529 strncpy (tempfile_pattern + len, "/tempfile",
530 (POCL_FILENAME_LENGTH - len));
531 tempfile_pattern[POCL_FILENAME_LENGTH - 1] = 0;
532 assert (strlen (tempfile_pattern) < POCL_FILENAME_LENGTH);
533
534 int bytes_written;
535 if (use_kernel_cache)
536 {
537 bytes_written = snprintf (tempdir_pattern, POCL_FILENAME_LENGTH,
538 "%s/tempdir", cache_topdir);
539 assert (bytes_written > 0 && bytes_written < POCL_FILENAME_LENGTH);
540 }
541 else
542 {
543 bytes_written = snprintf (tempdir_pattern, POCL_FILENAME_LENGTH,
544 "%s/_UNCACHED", cache_topdir);
545 assert (bytes_written > 0 && bytes_written < POCL_FILENAME_LENGTH);
546 }
547
548 cache_topdir_initialized = 1;
549
550 return 0;
551}

553/* Create the new program cachedir, invalidating the old program
554 * binaries and IRs if the new computed hash is different from the old
555 * one. The source hash is computed from the preprocessed source
556 * if present, from the original source otherwise: this is to ensure
557 * that cache-related functions (which include log retrieval) still
558 * work correctly even if preprocessing fails
559 */
560
561int
562pocl_cache_create_program_cachedir(cl_program program,
563 unsigned device_i,
564 const char* preprocessed_source,
565 size_t source_len,
566 char* program_bc_path)
567{
568 assert(cache_topdir_initialized);
569
570 if (use_kernel_cache)
571 {
572#ifdef OCS_AVAILABLE
573 const char *hash_source = NULL;
574 size_t hs_len = 0;
575
576 if (program->source && preprocessed_source==NULL) {
577 hash_source = program->source;
578 hs_len = strlen(program->source);
579 } else {
580 hash_source = preprocessed_source;
581 hs_len = source_len;
582 }
583
584 build_program_compute_hash(program, device_i, hash_source, hs_len);
585
586#else
587 assert(buildhash_is_valid(program, device_i));
588#endif
589
590 program_device_dir(program_bc_path, program, device_i, "");
591
592 if (pocl_mkdir_p(program_bc_path))
593 return 1;
594 }
595 else
596 {
597 /* if kernel cache is disabled, use a random dir. */
598 char random_dir[POCL_FILENAME_LENGTH];
599 if (pocl_cache_create_tempdir (random_dir))
600 return 1;
601 size_t s = strlen (cache_topdir) + 1;
602 assert (strlen (random_dir) == (s + 16));
603 memcpy (program->build_hash[device_i], random_dir + s, 16);
604 }
605
606 pocl_cache_program_bc_path (program_bc_path, program, device_i);
607
608 return 0;
609}
610
611void pocl_cache_cleanup_cachedir(cl_program program) {
612
613 /* only rm -rf if kernel cache is disabled */
614 if (use_kernel_cache)
615 return;
616
617 unsigned i;
618
619 for (i = 0; i < program->num_devices; i++)
620 {
621 if (!buildhash_is_valid (program, i))
622 continue;
623
624 char cachedir[POCL_FILENAME_LENGTH];
625 program_device_dir (cachedir, program, i, "");
626 pocl_rm_rf (cachedir);
627 }
