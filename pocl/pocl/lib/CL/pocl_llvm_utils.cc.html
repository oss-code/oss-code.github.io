72 llvm::raw_string_ostream sos(str);
73#ifdef LLVM_OLDER_THAN_7_0

81llvm::Module *parseModuleIRMem(const char *input_stream, size_t size) {
82 StringRef input_stream_ref(input_stream, size);
83 std::unique_ptr<MemoryBuffer> buffer =
84 MemoryBuffer::getMemBufferCopy(input_stream_ref);
85
86 auto parsed_module =
87 parseBitcodeFile(buffer->getMemBufferRef(), GlobalContext());
88 if (!parsed_module)
89 return nullptr;
90 return parsed_module.get().release();
91}

93int getModuleTriple(const char *input_stream, size_t size,
94 std::string &triple) {
95 StringRef input_stream_ref(input_stream, size);
96 std::unique_ptr<MemoryBuffer> buffer =
97 MemoryBuffer::getMemBufferCopy(input_stream_ref);
98
99 auto triple_e = getBitcodeTargetTriple(buffer->getMemBufferRef());
100 if (!triple_e)
101 return -1;
102 triple = triple_e.get();
103 return 0;
104}

106char *
107get_llvm_cpu_name ()
108{
109
110 StringRef r = llvm::sys::getHostCPUName();
111
112#ifndef KERNELLIB_HOST_DISTRO_VARIANTS
113 if (r.str() == "generic") {
114 POCL_MSG_WARN("LLVM does not recognize your cpu, trying to use "
115 OCL_KERNEL_TARGET_CPU " for -target-cpu\n");
116 r = llvm::StringRef(OCL_KERNEL_TARGET_CPU);
117 }
118#endif
119
120 assert(r.size() > 0);
121 char *cpu_name = (char *)malloc(r.size() + 1);
122 strncpy(cpu_name, r.data(), r.size());
123 cpu_name[r.size()] = 0;
124 return cpu_name;
125}

127int bitcode_is_spir(const char *bitcode, size_t size) {
128 std::string triple;
129 int err = getModuleTriple(bitcode, size, triple);
130 if (!err)
131 return triple.find("spir") == 0;
132 else
133 return 0;
134}

136#define SPIRV_MAGIC 0x07230203U
137#define OpCapab 0x00020011
138#define KernelExecModel 0x6
139
140int bitcode_is_spirv(const char *bitcode, size_t size, int *is_opencl) {
141 const uint32_t *bc32 = (const uint32_t *)bitcode;
142 unsigned loc = 0;
143 uint32_t magic = htole32(bc32[loc++]);
144
145 if ((size < 20) || (magic != SPIRV_MAGIC))
146 return 0;
147
148 // skip version, generator, bound, schema
149 loc += 4;
150 *is_opencl = 0;
151 uint32_t ins, val;
152 do {
153 ins = htole32(bc32[loc++]);
154 val = htole32(bc32[loc++]);
155 if (val == KernelExecModel)
156 *is_opencl = 1;
157 } while (ins == OpCapab);
158
159 return 1;
160}

171#define VECWIDTH(x) \
172 std::min(std::max((lane_width / (unsigned)(sizeof(x))), 1U), 16U)
173
174void cpu_setup_vector_widths(cl_device_id dev) {
175 StringMap<bool> features;
176 bool res = llvm::sys::getHostCPUFeatures(features);
177 assert(res);
178 unsigned lane_width = 1;
179 if ((features["sse"]) || (features["neon"]))
180 lane_width = 16;
181 if (features["avx"])
182 lane_width = 32;
183 if (features["avx512f"])
184 lane_width = 64;
185
186 dev->native_vector_width_char = dev->preferred_vector_width_char =
187 VECWIDTH(cl_char);
188 dev->native_vector_width_short = dev->preferred_vector_width_short =
189 VECWIDTH(cl_short);
190 dev->native_vector_width_int = dev->preferred_vector_width_int =
191 VECWIDTH(cl_int);
192 dev->native_vector_width_long = dev->preferred_vector_width_long =
193 VECWIDTH(cl_long);
194 dev->native_vector_width_float = dev->preferred_vector_width_float =
195 VECWIDTH(float);
196#ifdef _CL_DISABLE_DOUBLE
197 dev->native_vector_width_double = dev->preferred_vector_width_double = 0;
198#else
199 dev->native_vector_width_double = dev->preferred_vector_width_double =
200 VECWIDTH(double);
201#endif
202
203#ifdef _CL_DISABLE_HALF
204 dev->native_vector_width_half = dev->preferred_vector_width_half = 0;
205#else
206 dev->native_vector_width_half = dev->preferred_vector_width_half =
207 VECWIDTH(cl_short);
208#endif
209}

216/*
217 * Use one global LLVMContext across all LLVM bitcodes. This is because
218 * we want to cache the bitcode IR libraries and reuse them when linking
219 * new kernels. The CloneModule etc. seem to assume we are linking
220 * bitcodes with a same LLVMContext. Unfortunately, this requires serializing
221 * all calls to the LLVM APIs with mutex.
222 * Freeing/deleting the context crashes LLVM 3.2 (at program exit), as a
223 * work-around, allocate this from heap.
224 */
225static LLVMContext *globalContext = NULL;
226static bool LLVMInitialized = false;
227
228static std::string poclDiagString;
229static llvm::raw_string_ostream poclDiagStream(poclDiagString);
230static DiagnosticPrinterRawOStream poclDiagPrinter(poclDiagStream);
231
232static void diagHandler(const DiagnosticInfo &DI, void *Context) {
233 DI.print(poclDiagPrinter);
234}
235
236std::string getDiagString() {
237 poclDiagStream.flush();
238 std::string ret(std::move(poclDiagString));
239 poclDiagString.clear();
240 return ret;
241}
242
243llvm::LLVMContext &GlobalContext() {
244 if (globalContext == NULL) {
245 globalContext = new LLVMContext();
246#ifdef LLVM_OLDER_THAN_6_0
247 globalContext->setDiagnosticHandler(diagHandler, globalContext);
248#else
249 globalContext->setDiagnosticHandlerCallBack(diagHandler, globalContext);
250#endif
251 }
252 return *globalContext;
253}

255/* The LLVM API interface functions are not at the moment not thread safe,
256 * Pocl needs to ensure only one thread is using this layer at the time.
257 */
258static pocl_lock_t kernelCompilerLock = POCL_LOCK_INITIALIZER;

260PoclCompilerMutexGuard::PoclCompilerMutexGuard(void *unused) {
261 POCL_LOCK(kernelCompilerLock);
262}
263
264PoclCompilerMutexGuard::~PoclCompilerMutexGuard() {
265 POCL_UNLOCK(kernelCompilerLock);
266}

270/* must be called with kernelCompilerLock locked */
271void InitializeLLVM() {
272
273 if (LLVMInitialized)
274 return;
275 // We have not initialized any pass managers for any device yet.
276 // Run the global LLVM pass initialization functions.
277 InitializeAllTargets();
278 InitializeAllTargetMCs();
279 InitializeAllAsmPrinters();
280 InitializeAllAsmParsers();
281
282 PassRegistry &Registry = *PassRegistry::getPassRegistry();
283
284 initializeCore(Registry);
285 initializeScalarOpts(Registry);
286 initializeVectorization(Registry);
287 initializeIPO(Registry);
288 initializeAnalysis(Registry);
289 initializeTransformUtils(Registry);
290 initializeInstCombine(Registry);
291 initializeInstrumentation(Registry);
292 initializeTarget(Registry);
293
294// Set the options only once. TODO: fix it so that each
295// device can reset their own options. Now one cannot compile
296// with different options to different devices at one run.
297
298 StringMap<llvm::cl::Option *> &opts = llvm::cl::getRegisteredOptions();
299
300 llvm::cl::Option *O = nullptr;
301
302 currentWgMethod = pocl_get_string_option("POCL_WORK_GROUP_METHOD", "loopvec");
303
304 if (currentWgMethod == "loopvec") {
305
306 O = opts["scalarize-load-store"];
307 assert(O && "could not find LLVM option 'scalarize-load-store'");
308 O->addOccurrence(1, StringRef("scalarize-load-store"), StringRef("1"),
309 false);
310
311 // LLVM inner loop vectorizer does not check whether the loop inside
312 // another loop, in which case even a small trip count loops might be
313 // worthwhile to vectorize.
314 O = opts["vectorizer-min-trip-count"];
315 assert(O && "could not find LLVM option 'vectorizer-min-trip-count'");
316 O->addOccurrence(1, StringRef("vectorizer-min-trip-count"), StringRef("2"),
317 false);
318
319 if (pocl_get_bool_option("POCL_VECTORIZER_REMARKS", 0) == 1) {
320 // Enable diagnostics from the loop vectorizer.
321 O = opts["pass-remarks-missed"];
322 assert(O && "could not find LLVM option 'pass-remarks-missed'");
323 O->addOccurrence(1, StringRef("pass-remarks-missed"),
324 StringRef("loop-vectorize"), false);
325
326 O = opts["pass-remarks-analysis"];
327 assert(O && "could not find LLVM option 'pass-remarks-analysis'");
328 O->addOccurrence(1, StringRef("pass-remarks-analysis"),
329 StringRef("loop-vectorize"), false);
330
331 O = opts["pass-remarks"];
332 assert(O && "could not find LLVM option 'pass-remarks'");
333 O->addOccurrence(1, StringRef("pass-remarks"),
334 StringRef("loop-vectorize"), false);
335 }
336 }
337 if (pocl_get_bool_option("POCL_DEBUG_LLVM_PASSES", 0) == 1) {
338 O = opts["debug"];
339 assert(O && "could not find LLVM option 'debug'");
340 O->addOccurrence(1, StringRef("debug"), StringRef("true"), false);
341 }
342
343 O = opts["unroll-threshold"];
344 assert(O && "could not find LLVM option 'unroll-threshold'");
345 O->addOccurrence(1, StringRef("unroll-threshold"), StringRef("1"), false);
346
347 LLVMInitialized = true;
348}

351// TODO FIXME currently pocl_llvm_release() only works when
352// there are zero programs with IRs, because
353// programs hold references to LLVM IRs
354long numberOfIRs = 0;
355
356void pocl_llvm_release() {
357
358 PoclCompilerMutexGuard lockHolder(NULL);
359
360 assert(numberOfIRs >= 0);
361
362 if (numberOfIRs > 0) {
363 POCL_MSG_PRINT_LLVM("still have references to IRs - not releasing LLVM\n");
364 return;
365 } else {
366 POCL_MSG_PRINT_LLVM("releasing LLVM\n");
367 }
368
369 clearKernelPasses();
370 clearTargetMachines();
371 cleanKernelLibrary();
372
373 delete globalContext;
374 globalContext = nullptr;
375 LLVMInitialized = false;
376}
