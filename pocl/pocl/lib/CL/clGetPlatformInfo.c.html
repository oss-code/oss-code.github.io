27CL_API_ENTRY cl_int CL_API_CALL
28POname(clGetPlatformInfo)(cl_platform_id platform,
29 cl_platform_info param_name,
30 size_t param_value_size,
31 void * param_value,
32 size_t * param_value_size_ret) CL_API_SUFFIX__VERSION_1_0
33{
34 const char *ret;
35 size_t retlen;
36 cl_platform_id tmp_platform;
37
38 // TODO: if we don't have ICD in use, platform==NULL should be valid & point to pocl
39 POCL_RETURN_ERROR_COND((platform == NULL), CL_INVALID_PLATFORM);
40
41 POname(clGetPlatformIDs)(1, &tmp_platform, NULL);
42 POCL_RETURN_ERROR_ON((platform != tmp_platform), CL_INVALID_PLATFORM,
43 "Can only return info about the POCL platform\n");
44
45 switch (param_name)
46 {
47 case CL_PLATFORM_PROFILE:
48 // TODO: figure this out depending on the native execution host.
49 // assume FULL_PROFILE for now.
50 POCL_RETURN_GETINFO_STR("FULL_PROFILE");
51
52 case CL_PLATFORM_VERSION:
53
54 POCL_RETURN_GETINFO_STR ("OpenCL " POCL_CL_VERSION
55 " pocl " PACKAGE_VERSION
56#ifdef OCS_AVAILABLE
57 ", " CMAKE_BUILD_TYPE
58#ifdef POCL_ASSERTS_BUILD
59 "+Asserts"
60#endif
61 ", LLVM " LLVM_VERSION
62#ifdef LLVM_BUILD_MODE_DEBUG
63 " - debug"
64#endif
65
66#ifdef ENABLE_RELOCATION
67 ", RELOC"
68#endif
69
70#ifdef ENABLE_SPIR
71 ", SPIR"
72#endif
73
74#ifdef ENABLE_SPIRV
75 ", SPIR-V"
76#endif
77
78#ifdef ENABLE_SLEEF
79 ", SLEEF"
80#endif
81
82#ifndef _CL_DISABLE_HALF
83 ", FP16"
84#endif
85
86#else
87 ", no online compiler support"
88#endif
89
90#ifdef ENABLE_ASAN
91 ", ASAN"
92#endif
93#ifdef ENABLE_TSAN
94 ", TSAN"
95#endif
96#ifdef ENABLE_LSAN
97 ", LSAN"
98#endif
99#ifdef ENABLE_UBSAN
100 ", UBSAN"
101#endif
102
103#ifdef BUILD_CUDA
104 ", CUDA"
105#endif
106
107#ifdef BUILD_HSA
108 ", HSA"
109#endif
110
111#ifdef TCE_AVAILABLE
112 ", TCE"
113#endif
114
115#ifdef HAVE_LTTNG_UST
116 ", LTTNG"
117#endif
118
119#ifdef KERNELLIB_HOST_DISTRO_VARIANTS
120 ", DISTRO"
121#endif
122
123#ifdef POCL_DEBUG_MESSAGES
124 ", POCL_DEBUG"
125#endif
126 );
127
128 case CL_PLATFORM_NAME:
129 POCL_RETURN_GETINFO_STR("Portable Computing Language");
130
131 case CL_PLATFORM_VENDOR:
132 POCL_RETURN_GETINFO_STR("The pocl project");
133
134 case CL_PLATFORM_EXTENSIONS:
135 // TODO: do we want to list all supported extensions *here*, or in some header?.
136 // TODO: yes, it is better here: available through ICD Loader and headers can be the ones from Khronos Group
137#ifdef BUILD_ICD
138 POCL_RETURN_GETINFO_STR("cl_khr_icd");
139#else
140 POCL_RETURN_GETINFO_STR("");
141#endif
142
143 case CL_PLATFORM_ICD_SUFFIX_KHR:
144 POCL_RETURN_GETINFO_STR("POCL");
145
146 default:
147 return CL_INVALID_VALUE;
148 }
149
150 // the OpenCL API docs *seem* to count the trailing NULL
151 retlen = strlen(ret) + 1;
152
153 if (param_value != NULL)
154 {
155 if (param_value_size < retlen)
156 return CL_INVALID_VALUE;
157
158 memcpy(param_value, ret, retlen);
159 }
160
161 if (param_value_size_ret != NULL)
162 *param_value_size_ret = retlen;
163
164 return CL_SUCCESS;
165}
