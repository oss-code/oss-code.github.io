29CL_API_ENTRY cl_int CL_API_CALL
30POname(clGetKernelInfo)(cl_kernel kernel ,
31 cl_kernel_info param_name ,
32 size_t param_value_size ,
33 void * param_value ,
34 size_t * param_value_size_ret) CL_API_SUFFIX__VERSION_1_0
35{
36 POCL_RETURN_ERROR_COND((kernel == NULL), CL_INVALID_KERNEL);
37
38 switch (param_name) {
39 case CL_KERNEL_FUNCTION_NAME:
40 POCL_RETURN_GETINFO_STR(kernel->name);
41 case CL_KERNEL_NUM_ARGS:
42 POCL_RETURN_GETINFO (cl_uint, kernel->meta->num_args);
43 case CL_KERNEL_REFERENCE_COUNT:
44 POCL_RETURN_GETINFO(cl_uint, kernel->pocl_refcount);
45 case CL_KERNEL_CONTEXT:
46 POCL_RETURN_GETINFO(cl_context, kernel->context);
47 case CL_KERNEL_PROGRAM:
48 POCL_RETURN_GETINFO(cl_program, kernel->program);
49 case CL_KERNEL_ATTRIBUTES:
50 if (kernel->meta->attributes)
51 POCL_RETURN_GETINFO_STR (kernel->meta->attributes);
52 else
53 POCL_RETURN_GETINFO_STR ("");
54 }
55 return CL_INVALID_VALUE;
56}
