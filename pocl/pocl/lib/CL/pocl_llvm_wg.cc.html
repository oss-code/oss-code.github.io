122// Returns the TargetMachine instance or zero if no triple is provided.
123static TargetMachine *GetTargetMachine(cl_device_id device, Triple &triple) {
124
125 if (targetMachines.find(device) != targetMachines.end())
126 return targetMachines[device];
127
128 std::string Error;
129 // Triple TheTriple(device->llvm_target_triplet);
130
131 std::string MCPU = device->llvm_cpu ? device->llvm_cpu : "";
132
133 const Target *TheTarget = TargetRegistry::lookupTarget("", triple, Error);
134
135 // In LLVM 3.4 and earlier, the target registry falls back to
136 // the cpp backend in case a proper match was not found. In
137 // that case simply do not use target info in the compilation
138 // because it can be an off-tree target not registered at
139 // this point (read: TCE).
140 if (!TheTarget || TheTarget->getName() == std::string("cpp")) {
141 return 0;
142 }
143
144#ifdef LLVM_OLDER_THAN_6_0
145 TargetMachine *TM = TheTarget->createTargetMachine(
146 triple.getTriple(), MCPU, StringRef(""), GetTargetOptions(), Reloc::PIC_,
147 CodeModel::Default, CodeGenOpt::Aggressive);
148#else
149 TargetMachine *TM = TheTarget->createTargetMachine(
150 triple.getTriple(), MCPU, StringRef(""), GetTargetOptions(), Reloc::PIC_,
151 CodeModel::Small, CodeGenOpt::Aggressive);
152#endif
153
154 assert(TM != NULL && "llvm target has no targetMachine constructor");
155 if (device->ops->init_target_machine)
156 device->ops->init_target_machine(device->data, TM);
157 targetMachines[device] = TM;
158
159 return TM;
160}

163static PassManager &
164kernel_compiler_passes(cl_device_id device, llvm::Module *input,
165 const std::string &module_data_layout) {
166
167 PassManager *Passes = nullptr;
168 PassRegistry *Registry = nullptr;
169
170 if (kernelPasses.find(device) != kernelPasses.end()) {
171 return *kernelPasses[device];
172 }
173
174 bool SPMDDevice = device->spmd;
175
176 Registry = PassRegistry::getPassRegistry();
177
178 Passes = new PassManager();
179
180 // Need to setup the target info for target specific passes. */
181 Triple triple(device->llvm_target_triplet);
182 TargetMachine *Machine = GetTargetMachine(device, triple);
183
184 if (Machine)
185 Passes->add(
186 createTargetTransformInfoWrapperPass(Machine->getTargetIRAnalysis()));
187
188
189 /* Disables automated generation of libcalls from code patterns.
190 TCE doesn't have a runtime linker which could link the libs later on.
191 Also the libcalls might be harmful for WG autovectorization where we
192 want to try to vectorize the code it converts to e.g. a memset or
193 a memcpy */
194 TargetLibraryInfoImpl TLII(triple);
195 TLII.disableAllFunctions();
196 Passes->add(new TargetLibraryInfoWrapperPass(TLII));
197
198 /* The kernel compiler passes to run, in order.
199
200 Notes about the kernel compiler phase ordering:
201
202 -mem2reg first because we get unoptimized output from Clang where all
203 variables are allocas. Avoid context saving the allocas and make the
204 more readable by calling -mem2reg at the beginning.
205
206 -implicit-cond-barriers after -implicit-loop-barriers because the latter
207 can inject barriers to loops inside conditional regions after which the
208 peeling should be avoided by injecting the implicit conditional barriers.
209
210 -loop-barriers, -barriertails, and -barriers should be ran after the
211 implicit barrier injection passes so they "normalize" the implicit
212 barriers also.
213
214 -phistoallocas before -workitemloops as otherwise it cannot inject context
215 restore code (PHIs need to be at the beginning of the BB and so one cannot
216 context restore them with non-PHI code if the value is needed in another
217 PHI). */
218
219 std::vector<std::string> passes;
220 passes.push_back("remove-optnone");
221 passes.push_back("optimize-wi-func-calls");
222 passes.push_back("handle-samplers");
223 passes.push_back("workitem-handler-chooser");
224 passes.push_back("mem2reg");
225 passes.push_back("domtree");
226 if (device->autolocals_to_args)
227 passes.push_back("automatic-locals");
228
229 if (SPMDDevice) {
230 passes.push_back("flatten-inline-all");
231 passes.push_back("always-inline");
232 } else {
233 passes.push_back("flatten-globals");
234 passes.push_back("flatten-barrier-subs");
235 passes.push_back("always-inline");
236 passes.push_back("inline");
237 }
238
239 // It should be now safe to run -O3 over the single work-item kernel
240 // as the barrier has the attributes preventing illegal motions and
241 // duplication. Let's do it to clean up the code for later passes.
242 // Especially the WI context structures get needlessly bloated in case there
243 // is dead code lying around.
244 passes.push_back("STANDARD_OPTS");
245
246 if (!SPMDDevice) {
247 passes.push_back("simplifycfg");
248 passes.push_back("loop-simplify");
249 passes.push_back("uniformity");
250 passes.push_back("phistoallocas");
251 passes.push_back("isolate-regions");
252 passes.push_back("implicit-loop-barriers");
253 passes.push_back("implicit-cond-barriers");
254 passes.push_back("loop-barriers");
255 passes.push_back("barriertails");
256 passes.push_back("barriers");
257 passes.push_back("isolate-regions");
258 passes.push_back("wi-aa");
259 passes.push_back("workitemrepl");
260 //passes.push_back("print-module");
261 passes.push_back("workitemloops");
262 // Remove the (pseudo) barriers. They have no use anymore due to the
263 // work-item loop control taking care of them.
264 passes.push_back("remove-barriers");
265 }
266
267 // Add the work group launcher functions and privatize the pseudo variable
268 // (local id) accesses. We have to do this late because we rely on aggressive
269 // inlining to expose the _{local,group}_id accesses which will be replaced
270 // with context struct accesses. TODO: A cleaner and a more robust way would
271 // be to add hidden context struct parameters to the builtins that need the
272 // context data and fix the calls early.
273 if (device->workgroup_pass) {
274 passes.push_back("workgroup");
275 passes.push_back("always-inline");
276 }
277
278 // Attempt to move all allocas to the entry block to avoid the need for
279 // dynamic stack which is problematic for some architectures.
280 passes.push_back("allocastoentry");
281
282 // Later passes might get confused (and expose possible bugs in them) due to
283 // UNREACHABLE blocks left by repl. So let's clean up the CFG before running
284 // the standard LLVM optimizations.
285 passes.push_back("simplifycfg");
286
287#if 0
288 passes.push_back("print-module");
289 passes.push_back("dot-cfg");
290#endif
291
292 passes.push_back("STANDARD_OPTS");
293
294 // Due to unfortunate phase-ordering problems with store sinking,
295 // loop deletion does not always apply when executing -O3 only
296 // once. Cherry pick the optimization to rerun here.
297 passes.push_back("loop-deletion");
298
299 passes.push_back("remove-barriers");
300
301 // Now actually add the listed passes to the PassManager.
302 for (unsigned i = 0; i < passes.size(); ++i) {
303 // This is (more or less) -O3.
304 if (passes[i] == "STANDARD_OPTS") {
305 PassManagerBuilder Builder;
306 Builder.OptLevel = 3;
307 Builder.SizeLevel = 0;
308
309 // These need to be setup in addition to invoking the passes
310 // to get the vectorizers initialized properly. Assume SPMD
311 // devices do not want to vectorize intra work-item at this
312 // stage.
313 if (currentWgMethod == "loopvec" && !SPMDDevice) {
314 Builder.LoopVectorize = true;
315 Builder.SLPVectorize = true;
316 } else {
317 Builder.LoopVectorize = false;
318 Builder.SLPVectorize = false;
319 }
320 Builder.VerifyInput = true;
321 Builder.VerifyOutput = true;
322 Builder.populateModulePassManager(*Passes);
323 continue;
324 }
325
326 const PassInfo *PIs = Registry->getPassInfo(StringRef(passes[i]));
327 if (PIs) {
328 // std::cout << "-"<<passes[i] << " ";
329 Pass *thispass = PIs->createPass();
330 Passes->add(thispass);
331 } else {
332 std::cerr << "Failed to create kernel compiler pass " << passes[i]
333 << std::endl;
334 POCL_ABORT("FAIL\n");
335 }
336 }
337
338 kernelPasses[device] = Passes;
339 return *Passes;
340}

368int pocl_llvm_generate_workgroup_function_nowrite(
369 unsigned DeviceI, cl_device_id Device, cl_kernel Kernel,
370 _cl_command_node *Command, void **Output, int Specialize) {
371
372 _cl_command_run *RunCommand = &Command->command.run;
373 cl_program Program = Kernel->program;
374
375 currentPoclDevice = Device;
376
377 PoclCompilerMutexGuard LockHolder(NULL);
378 InitializeLLVM();
379
380 if (Program->llvm_irs[DeviceI] == NULL)
381 pocl_update_program_llvm_irs_unlocked(Program, DeviceI);
382
383 llvm::Module *ProgramBC = (llvm::Module *)Program->llvm_irs[DeviceI];
384
385 // Create an empty Module and copy only the kernel+callgraph from
386 // program.bc.
387 llvm::Module *ParallelBC =
388 new llvm::Module(StringRef("parallel_bc"), GlobalContext());
389
390 ParallelBC->setTargetTriple(ProgramBC->getTargetTriple());
391 ParallelBC->setDataLayout(ProgramBC->getDataLayout());
392
393 copyKernelFromBitcode(Kernel->name, ParallelBC, ProgramBC);
394
395 // Set the specialization properties.
396 if (Specialize) {
397 pocl::WGLocalSizeX = RunCommand->pc.local_size[0];
398 pocl::WGLocalSizeY = RunCommand->pc.local_size[1];
399 pocl::WGLocalSizeZ = RunCommand->pc.local_size[2];
400 pocl::WGDynamicLocalSize = pocl::WGLocalSizeX == 0 &&
401 pocl::WGLocalSizeY == 0 &&
402 pocl::WGLocalSizeZ == 0;
403 pocl::WGAssumeZeroGlobalOffset = RunCommand->pc.global_offset[0] == 0 &&
404 RunCommand->pc.global_offset[1] == 0 &&
405 RunCommand->pc.global_offset[2] == 0;
406 // Compile a smallgrid version or a generic one?
407 if (RunCommand->force_large_grid_wg_func ||
408 pocl_cmd_max_grid_dim_width(RunCommand) >=
409 Device->grid_width_specialization_limit) {
410 pocl::WGMaxGridDimWidth = 0; // The generic / large / unlimited size one.
411 } else {
412 // Limited grid dimension width by the device specific limit.
413 pocl::WGMaxGridDimWidth = Device->grid_width_specialization_limit;
414 }
415 } else {
416 pocl::WGDynamicLocalSize = true;
417 pocl::WGLocalSizeX = pocl::WGLocalSizeY = pocl::WGLocalSizeZ = 0;
418 pocl::WGAssumeZeroGlobalOffset = false;
419 pocl::WGMaxGridDimWidth = 0;
420 }
421
422 KernelName = Kernel->name;
423
424#ifdef DUMP_LLVM_PASS_TIMINGS
425 llvm::TimePassesIsEnabled = true;
426#endif
427 POCL_MEASURE_START(llvm_workgroup_ir_func_gen);
428#ifdef LLVM_OLDER_THAN_3_7
429 kernel_compiler_passes(Device, ParallelBC,
430 ParallelBC->getDataLayout()->getStringRepresentation())
431 .run(*ParallelBC);
432#else
433 kernel_compiler_passes(Device, ParallelBC,
434 ParallelBC->getDataLayout().getStringRepresentation())
435 .run(*ParallelBC);
436#endif
437 POCL_MEASURE_FINISH(llvm_workgroup_ir_func_gen);
438#ifdef DUMP_LLVM_PASS_TIMINGS
439 llvm::reportAndResetTimings();
440#endif
441
442 assert(Output != NULL);
443 *Output = (void *)ParallelBC;
444 ++numberOfIRs;
445 return 0;
446}

448int pocl_llvm_generate_workgroup_function(unsigned DeviceI, cl_device_id Device,
449 cl_kernel Kernel,
450 _cl_command_node *Command,
451 int Specialize) {
452
453 void *Module = NULL;
454
455 char ParallelBCPath[POCL_FILENAME_LENGTH];
456 pocl_cache_work_group_function_path(ParallelBCPath, Kernel->program, DeviceI,
457 Kernel, Command, Specialize);
458
459 if (pocl_exists(ParallelBCPath))
460 return CL_SUCCESS;
461
462 char FinalBinaryPath[POCL_FILENAME_LENGTH];
463 pocl_cache_final_binary_path(FinalBinaryPath, Kernel->program, DeviceI,
464 Kernel, Command, Specialize);
465
466 if (pocl_exists(FinalBinaryPath))
467 return CL_SUCCESS;
468
469 int Error = pocl_llvm_generate_workgroup_function_nowrite(
470 DeviceI, Device, Kernel, Command, &Module, Specialize);
471 if (Error)
472 return Error;
473
474 Error = pocl_cache_write_kernel_parallel_bc(Module, Kernel->program, DeviceI,
475 Kernel, Command, Specialize);
476
477 if (Error) {
478 POCL_MSG_ERR("pocl_cache_write_kernel_parallel_bc() failed with %i\n",
479 Error);
480 return Error;
481 }
482
483 pocl_destroy_llvm_module(Module);
484 return Error;
485}

525void pocl_llvm_update_binaries(cl_program program) {
526
527 PoclCompilerMutexGuard lockHolder(NULL);
528 InitializeLLVM();
529
530 char program_bc_path[POCL_FILENAME_LENGTH];
531 int error;
532
533 // Dump the LLVM IR Modules to memory buffers.
534 assert(program->llvm_irs != NULL);
535#ifdef DEBUG_POCL_LLVM_API
536 printf("### refreshing the binaries of the program %p\n", program);
537#endif
538
539 for (size_t i = 0; i < program->num_devices; ++i) {
540 assert(program->llvm_irs[i] != NULL);
541 if (program->binaries[i])
542 continue;
543
544 pocl_cache_program_bc_path(program_bc_path, program, i);
545 error = pocl_write_module((llvm::Module *)program->llvm_irs[i],
546 program_bc_path, 1);
547 assert(error == 0);
548 if (error)
549 {
550 POCL_MSG_ERR ("pocl_write_module(%s) failed!\n",
551 program_bc_path);
552 continue;
553 }
554
555 std::string content;
556 writeModuleIR((llvm::Module *)program->llvm_irs[i], content);
557
558 size_t n = content.size();
559 if (n < program->binary_sizes[i])
560 POCL_ABORT("binary size doesn't match the expected value\n");
561 if (program->binaries[i])
562 POCL_MEM_FREE(program->binaries[i]);
563 program->binaries[i] = (unsigned char *)malloc(n);
564 std::memcpy(program->binaries[i], content.c_str(), n);
565
566#ifdef DEBUG_POCL_LLVM_API
567 printf("### binary for device %zi was of size %zu\n", i,
568 program->binary_sizes[i]);
569#endif
570 }

582/* Run LLVM codegen on input file (parallel-optimized).
583 * modp = llvm::Module* of parallel.bc
584 * Output native object file (<kernel>.so.o). */
585int pocl_llvm_codegen(cl_device_id Device, void *Modp, char **Output,
586 uint64_t *OutputSize) {
587
588 PoclCompilerMutexGuard LockHolder(nullptr);
589 InitializeLLVM();
590
591 llvm::Module *Input = (llvm::Module *)Modp;
592 assert(Input);
593 *Output = nullptr;
594
595 PassManager PMObj;
596 initPassManagerForCodeGen(PMObj, Device);
597
598 llvm::Triple Triple(Device->llvm_target_triplet);
599 llvm::TargetMachine *Target = GetTargetMachine(Device, Triple);
600
601 // First try direct object code generation from LLVM, if supported by the
602 // LLVM backend for the target.
603 bool LLVMGeneratesObjectFiles = true;
604
605 SmallVector<char, 4096> Data;
606 llvm::raw_svector_ostream SOS(Data);
607 bool cannotEmitFile;
608
609#ifdef LLVM_OLDER_THAN_7_0
610 cannotEmitFile = Target->addPassesToEmitFile(PMObj, SOS,
611 TargetMachine::CGFT_ObjectFile);
612#else
613 cannotEmitFile = Target->addPassesToEmitFile(PMObj, SOS, nullptr,
614 TargetMachine::CGFT_ObjectFile);
615#endif
616
617 LLVMGeneratesObjectFiles = !cannotEmitFile;
618
619 if (LLVMGeneratesObjectFiles) {
620 POCL_MSG_PRINT_LLVM("Generating an object file directly.\n");
621#ifdef DUMP_LLVM_PASS_TIMINGS
622 llvm::TimePassesIsEnabled = true;
623#endif
624 PMObj.run(*Input);
625#ifdef DUMP_LLVM_PASS_TIMINGS
626 llvm::reportAndResetTimings();
627#endif
628 std::string O = SOS.str(); // flush
629 const char *Cstr = O.c_str();
630 size_t S = O.size();
631 *Output = (char *)malloc(S);
632 *OutputSize = S;
633 memcpy(*Output, Cstr, S);
634 return 0;
635 }
636
637 PassManager PMAsm;
638 initPassManagerForCodeGen(PMAsm, Device);
639
640 POCL_MSG_PRINT_LLVM("Generating assembly text.\n");
641
642 // The LLVM target does not implement support for emitting object file directly.
643 // Have to emit the text first and then call the assembler from the command line
644 // to produce the binary.
645#ifdef LLVM_OLDER_THAN_7_0
646 if (Target->addPassesToEmitFile(PMAsm, SOS,
647 TargetMachine::CGFT_AssemblyFile)) {
648 POCL_ABORT("The target supports neither obj nor asm emission!");
649 }
650#else
651 if (Target->addPassesToEmitFile(PMAsm, SOS, nullptr,
652 TargetMachine::CGFT_AssemblyFile)) {
653 POCL_ABORT("The target supports neither obj nor asm emission!");
654 }
655#endif
656
657#ifdef DUMP_LLVM_PASS_TIMINGS
658 llvm::TimePassesIsEnabled = true;
659#endif
660 // This produces the assembly text:
661 PMAsm.run(*Input);
662#ifdef DUMP_LLVM_PASS_TIMINGS
663 llvm::reportAndResetTimings();
664#endif
665
666 // Next call the target's assembler via the Toolchain API indirectly through
667 // the Driver API.
668
669 char AsmFileName[POCL_FILENAME_LENGTH];
670 char ObjFileName[POCL_FILENAME_LENGTH];
671
672 std::string AsmStr = SOS.str().str();
673 pocl_write_tempfile(AsmFileName, "/tmp/pocl-asm", ".s", AsmStr.c_str(),
674 AsmStr.size(), nullptr);
675 pocl_mk_tempname(ObjFileName, "/tmp/pocl-obj", ".o", nullptr);
676
677 const char *Args[] = {CLANG, AsmFileName, "-c", "-o", ObjFileName, nullptr};
678 int Res = pocl_invoke_clang(Device, Args);
679
680 if (Res == 0) {
681 if (pocl_read_file(ObjFileName, Output, OutputSize))
682 POCL_ABORT("Could not read the object file.");
683 }
684
685 pocl_remove(AsmFileName);
686 pocl_remove(ObjFileName);
687 return Res;
688
