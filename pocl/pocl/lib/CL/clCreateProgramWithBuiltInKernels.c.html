31CL_API_ENTRY cl_program CL_API_CALL
32POname (clCreateProgramWithBuiltInKernels) (cl_context context,
33 cl_uint num_devices,
34 const cl_device_id *device_list,
35 const char *kernel_names,
36 cl_int *errcode_ret)
37CL_API_SUFFIX__VERSION_1_2
38{
39 cl_int errcode;
40 char *kernel_names_copy = NULL;
41 cl_program program = NULL;
42 char **builtin_names = NULL;
43 size_t num_kernels = 0;
44 char *save_ptr;
45 char *token;
46
47 POCL_GOTO_ERROR_COND ((kernel_names == NULL), CL_INVALID_VALUE);
48
49 kernel_names_copy = strdup (kernel_names);
50 token = strtok_r (kernel_names_copy, ";", &save_ptr);
51 while (token != NULL)
52 {
53 unsigned num_supported = 0;
54 cl_uint i;
55 for (i = 0; i < num_devices; ++i)
56 {
57 cl_device_id dev = device_list[i];
58 if (dev->ops->supports_builtin_kernel == NULL)
59 continue;
60 if (dev->ops->supports_builtin_kernel (dev->data, token))
61 ++num_supported;
62 }
63 POCL_GOTO_ERROR_ON ((num_supported == 0), CL_INVALID_VALUE,
64 "None of the devices in context supports this"
65 " builtin kernel!\n");
66 ++num_kernels;
67 token = strtok_r (NULL, ";", &save_ptr);
68 }
69 POCL_MEM_FREE (kernel_names_copy);
70
71 builtin_names = (char **)calloc (num_kernels, sizeof (char *));
72 POCL_GOTO_ERROR_COND ((builtin_names == NULL), CL_OUT_OF_HOST_MEMORY);
73
74 kernel_names_copy = strdup (kernel_names);
75 token = strtok_r (kernel_names_copy, ";", &save_ptr);
76 unsigned i;
77 for (i = 0; token != NULL; ++i)
78 {
79 builtin_names[i] = strdup (token);
80 token = strtok_r (NULL, ";", &save_ptr);
81 }
82 POCL_MEM_FREE (kernel_names_copy);
83
84 program = create_program_skeleton (context, num_devices, device_list, NULL,
85 NULL, NULL, &errcode, 1);
86 if (program == NULL)
87 goto ERROR;
88
89 program->num_builtin_kernels = num_kernels;
90 program->builtin_kernel_names = builtin_names;
91
92 if (errcode_ret != NULL)
93 *errcode_ret = CL_SUCCESS;
94
95 return program;
96
97ERROR:
98 POCL_MEM_FREE (kernel_names_copy);
99 POCL_MEM_FREE (builtin_names);
100 if (errcode_ret)
101 *errcode_ret = errcode;
102 return NULL;
103}
