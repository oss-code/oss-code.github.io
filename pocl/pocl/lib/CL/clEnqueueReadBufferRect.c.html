30CL_API_ENTRY cl_int CL_API_CALL
31POname(clEnqueueReadBufferRect)(cl_command_queue command_queue,
32 cl_mem buffer,
33 cl_bool blocking_read,
34 const size_t *buffer_origin,
35 const size_t *host_origin,
36 const size_t *region,
37 size_t buffer_row_pitch,
38 size_t buffer_slice_pitch,
39 size_t host_row_pitch,
40 size_t host_slice_pitch,
41 void *ptr,
42 cl_uint num_events_in_wait_list,
43 const cl_event *event_wait_list,
44 cl_event *event) CL_API_SUFFIX__VERSION_1_1
45{
46 cl_device_id device;
47 unsigned i;
48 _cl_command_node *cmd;
49 int errcode;
50
51 POCL_RETURN_ERROR_COND((command_queue == NULL), CL_INVALID_COMMAND_QUEUE);
52
53 POCL_RETURN_ERROR_COND((buffer == NULL), CL_INVALID_MEM_OBJECT);
54
55 POCL_RETURN_ON_SUB_MISALIGN (buffer, command_queue);
56
57 POCL_RETURN_ERROR_ON((buffer->type != CL_MEM_OBJECT_BUFFER),
58 CL_INVALID_MEM_OBJECT, "buffer is not a CL_MEM_OBJECT_BUFFER\n");
59
60 POCL_RETURN_ERROR_ON((buffer->flags & (CL_MEM_HOST_WRITE_ONLY | CL_MEM_HOST_NO_ACCESS)),
61 CL_INVALID_OPERATION, "buffer has been created with CL_MEM_HOST_WRITE_ONLY "
62 "or CL_MEM_HOST_NO_ACCESS\n");
63
64 POCL_RETURN_ERROR_ON((command_queue->context != buffer->context),
65 CL_INVALID_CONTEXT, "buffer and command_queue are not from the same context\n");
66
67 errcode = pocl_check_event_wait_list (command_queue, num_events_in_wait_list,
68 event_wait_list);
69 if (errcode != CL_SUCCESS)
70 return errcode;
71
72 POCL_RETURN_ERROR_COND((ptr == NULL), CL_INVALID_VALUE);
73 POCL_RETURN_ERROR_COND((buffer_origin == NULL), CL_INVALID_VALUE);
74 POCL_RETURN_ERROR_COND((host_origin == NULL), CL_INVALID_VALUE);
75 POCL_RETURN_ERROR_COND((region == NULL), CL_INVALID_VALUE);
76
77 size_t region_bytes = region[0] * region[1] * region[2];
78 POCL_RETURN_ERROR_ON((region_bytes <= 0), CL_INVALID_VALUE, "All items in region must be >0\n");
79
80 if (pocl_buffer_boundcheck_3d(buffer->size, buffer_origin, region, &buffer_row_pitch,
81 &buffer_slice_pitch, "") != CL_SUCCESS) return CL_INVALID_VALUE;
82
83 if (pocl_buffer_boundcheck_3d(((size_t)-1), host_origin, region, &host_row_pitch,
84 &host_slice_pitch, "") != CL_SUCCESS) return CL_INVALID_VALUE;
85
86 POCL_CHECK_DEV_IN_CMDQ;
87
88 size_t src_offset = 0;
89 POCL_CONVERT_SUBBUFFER_OFFSET (buffer, src_offset);
90
91 POCL_RETURN_ERROR_ON((buffer->size > command_queue->device->max_mem_alloc_size),
92 CL_OUT_OF_RESOURCES,
93 "buffer is larger than device's MAX_MEM_ALLOC_SIZE\n");
94
95 pocl_create_command (&cmd, command_queue, CL_COMMAND_READ_BUFFER_RECT,
96 event, num_events_in_wait_list, event_wait_list, 1,
97 &buffer);
98
99 cmd->command.read_rect.src_mem_id = &buffer->device_ptrs[device->dev_id];
100 cmd->command.read_rect.dst_host_ptr = ptr;
101
102 cmd->command.read_rect.host_origin[0] = host_origin[0];
103 cmd->command.read_rect.host_origin[1] = host_origin[1];
104 cmd->command.read_rect.host_origin[2] = host_origin[2];
105 cmd->command.read_rect.buffer_origin[0] = src_offset + buffer_origin[0];
106 cmd->command.read_rect.buffer_origin[1] = buffer_origin[1];
107 cmd->command.read_rect.buffer_origin[2] = buffer_origin[2];
108 cmd->command.read_rect.region[0] = region[0];
109 cmd->command.read_rect.region[1] = region[1];
110 cmd->command.read_rect.region[2] = region[2];
111
112 cmd->command.read_rect.host_row_pitch = host_row_pitch;
113 cmd->command.read_rect.host_slice_pitch = host_slice_pitch;
114 cmd->command.read_rect.buffer_row_pitch = buffer_row_pitch;
115 cmd->command.read_rect.buffer_slice_pitch = buffer_slice_pitch;
116
117 POname (clRetainMemObject) (buffer);
118 buffer->owning_device = device;
119
120 pocl_command_enqueue (command_queue, cmd);
121
122 if (blocking_read)
123 POname(clFinish)(command_queue);
124
125 return CL_SUCCESS;
126}
