70#define POCL_LOCK(__LOCK__) \
71 do \
72 { \
73 int r = pthread_mutex_lock (&(__LOCK__)); \
74 assert (r == 0); \
75 } \
76 while (0)
77#define POCL_UNLOCK(__LOCK__) \
78 do \
79 { \
80 int r = pthread_mutex_unlock (&(__LOCK__)); \
81 assert (r == 0); \
82 } \
83 while (0)
84#define POCL_INIT_LOCK(__LOCK__) \
85 do \
86 { \
87 int r = pthread_mutex_init (&(__LOCK__), NULL); \
88 assert (r == 0); \
89 } \
90 while (0)
91/* We recycle OpenCL objects by not actually freeing them until the
92 very end. Thus, the lock should not be destoryed at the refcount 0. */
93#define POCL_DESTROY_LOCK(__LOCK__) \
94 do \
95 { \
96 int r = pthread_mutex_destroy (&(__LOCK__)); \
97 assert (r == 0); \
98 } \
99 while (0)

102/* If available, use an Adaptive mutex for locking in the pthread driver,
103 otherwise fallback to simple mutexes */
104#define POCL_FAST_LOCK_T pthread_mutex_t
105#define POCL_FAST_LOCK(l) POCL_LOCK(l)
106#define POCL_FAST_UNLOCK(l) POCL_UNLOCK(l)
107#ifdef PTHREAD_ADAPTIVE_MUTEX_INITIALIZER_NP
108 #define POCL_FAST_INIT(l) \
109 do { \
110 pthread_mutexattr_t attrs; \
111 pthread_mutexattr_init (&attrs); \
112 int r = pthread_mutexattr_settype (&attrs, PTHREAD_MUTEX_ADAPTIVE_NP); \
113 assert (r == 0); \
114 pthread_mutex_init(&l, &attrs); \
115 pthread_mutexattr_destroy(&attrs);\
116 } while (0)
117#else
118 #define POCL_FAST_INIT(l) pthread_mutex_init(&l, NULL);
119#endif
120#define POCL_FAST_DESTROY(l) POCL_DESTROY_LOCK(l)

124#define POCL_LOCK_OBJ(__OBJ__) \
125 do \
126 { \
127 POCL_LOCK ((__OBJ__)->pocl_lock); \
128 assert ((__OBJ__)->pocl_refcount > 0); \
129 } \
130 while (0)
131#define POCL_UNLOCK_OBJ(__OBJ__) \
132 do \
133 { \
134 assert ((__OBJ__)->pocl_refcount >= 0); \
135 POCL_UNLOCK ((__OBJ__)->pocl_lock); \
136 } \
137 while (0)
138
139#define POCL_RELEASE_OBJECT(__OBJ__, __NEW_REFCOUNT__) \
140 do \
141 { \
142 POCL_LOCK_OBJ (__OBJ__); \
143 __NEW_REFCOUNT__ = --(__OBJ__)->pocl_refcount; \
144 POCL_UNLOCK_OBJ (__OBJ__); \
145 } \
146 while (0)
147
148#define POCL_RETAIN_OBJECT_UNLOCKED(__OBJ__) \
149 ++((__OBJ__)->pocl_refcount);
150
151#define POCL_RETAIN_OBJECT_REFCOUNT(__OBJ__, R) \
152 do { \
153 POCL_LOCK_OBJ (__OBJ__); \
154 R = POCL_RETAIN_OBJECT_UNLOCKED (__OBJ__); \
155 POCL_UNLOCK_OBJ (__OBJ__); \
156 } while (0)
157
158#define POCL_RETAIN_OBJECT(__OBJ__) \
159 do { \
160 POCL_LOCK_OBJ (__OBJ__); \
161 POCL_RETAIN_OBJECT_UNLOCKED (__OBJ__); \
162 POCL_UNLOCK_OBJ (__OBJ__); \
163 } while (0)
164
165/* The reference counter is initialized to 1,
166 when it goes to 0 object can be freed. */
167#define POCL_INIT_OBJECT_NO_ICD(__OBJ__) \
168 do { \
169 POCL_INIT_LOCK ((__OBJ__)->pocl_lock); \
170 (__OBJ__)->pocl_refcount = 1; \
171 } while (0)
172
173#define POCL_MEM_FREE(F_PTR) \
174 do { \
175 free((F_PTR)); \
176 (F_PTR) = NULL; \
177 } while (0)
178
179#ifdef BUILD_ICD
180/* Most (all?) object must also initialize the ICD field */
181# define POCL_INIT_OBJECT(__OBJ__) \
182 do { \
183 POCL_INIT_OBJECT_NO_ICD(__OBJ__); \
184 POCL_INIT_ICD_OBJECT(__OBJ__); \
185 } while (0)
186#else
187# define POCL_INIT_OBJECT(__OBJ__) \
188 POCL_INIT_OBJECT_NO_ICD(__OBJ__)
189#endif
190
191#define POCL_DESTROY_OBJECT(__OBJ__) \
192 do \
193 { \
194 POCL_DESTROY_LOCK ((__OBJ__)->pocl_lock); \
195 } \
196 while (0)

305struct pocl_device_ops {
306 const char *device_name;
307
308 /* New driver api extension for out-of-order execution and
309 asynchronous devices.
310 See this for reference: http://URN.fi/URN:NBN:fi:tty-201412051583
311 See basic and pthread driver for reference. */
312
313 /* submit gives the command for the device. The command may be left in the cq
314 or stored to the device driver owning the cq. submit is called
315 with node->event locked, and must return with it unlocked. */
316 void (*submit) (_cl_command_node *node, cl_command_queue cq);
317
318 /* join is called by clFinish and this function blocks until all the enqueued
319 commands are finished. Called by the user thread; see notify_cmdq_finished
320 for the driver thread counterpart. */
321 void (*join) (cl_device_id device, cl_command_queue cq);
322
323 /* flush is called when clFlush is called. This function ensures that
324 commands will be eventually executed. It is up to the device what happens
325 here, if anything. See basic and pthread for reference.*/
326 void (*flush) (cl_device_id device, cl_command_queue cq);
327
328 /* notify is used to communicate to a device driver that an event, it has
329 been waiting, has been completed. Upon call, both events are locked, and
330 must be locked also on return.*/
331 void (*notify) (cl_device_id device, cl_event event, cl_event finished);
332
333 /* broadcast is(has to be) called by the device driver when a command is
334 completed.
335 It is used to broadcast notifications to device drivers waiting
336 this event to complete.
337 There is a default implementation for this. Use it if there is no need
338 to do anything special here.
339 The default implementation calls notify(event, target_event) for the
340 list of events waiting on 'event'. */
341 void (*broadcast) (cl_event event);
342
343 /* wait_event is called by clWaitForEvents() and blocks the execution until
344 * the waited event is complete or failed. Called by user threads; see the
345 * notify_event_finished() callback for driver thread counterpart.
346 * Called (and must return) with unlocked event. */
347 void (*wait_event) (cl_device_id device, cl_event event);
348
349 /* update_event is an extra callback called during handling of event status
350 * changes, useful if something device specific needs to be done. May be
351 * NULL; no need to implement if not needed.
352 *
353 * Called via pocl_update_event_* functions in pocl_util.c
354 * All pocl_update_event_* (except COMPLETE) must be called (and return)
355 * with LOCKED event.
356 */
357 void (*update_event) (cl_device_id device, cl_event event);
358
359 /* free_event_data may be called when event is freed. Event data may only be
360 used by the device driver owning the corresponding command.
361 No need to implement this if the device does not need any event data. */
362 void (*free_event_data) (cl_event event);
363
364 /* Called from driver threads to notify every user thread waiting on
365 * command queue finish. See join() for user counterpart.
366 * Driver may chose to not implement this, which will result in
367 * undefined behaviour in multi-threaded user programs. */
368 void (*notify_cmdq_finished) (cl_command_queue cq);
369
370 /* Called from driver threads to notify every user thread waiting on
371 * a specific event. See wait_event() for user counterpart.
372 * Driver may chose to not implement this, which will result in
373 * undefined behaviour in multi-threaded user programs. */
374 void (*notify_event_finished) (cl_event event);
375
376 /* /New driver api extension */
377
378 /* Detects & returns the number of available devices the driver finds on the system. */
379 unsigned int (*probe) (struct pocl_device_ops *ops);
380 /* Device initialization. Parameters:
381 * j : progressive index for the devices of the same type
382 * device : struct to initialize
383 * parameters : optional environment with device-specific parameters
384 */
385 cl_int (*init) (unsigned j, cl_device_id device, const char *parameters);
386 /* uninitializes the driver for a particular device. May free hardware resources. */
387 cl_int (*uninit) (unsigned j, cl_device_id device);
388 /* reinitializes the driver for a particular device. Called after uninit;
389 * the first initialization is done by 'init'. May be NULL */
390 cl_int (*reinit) (unsigned j, cl_device_id device);
391
392 /* if the driver needs to use hardware resources for command queues, use this */
393 cl_int (*init_queue) (cl_command_queue queue);
394 void (*free_queue) (cl_command_queue queue);
395
396 /* allocate a buffer in device memory */
397 cl_int (*alloc_mem_obj) (cl_device_id device, cl_mem mem_obj, void* host_ptr);
398 /* free a device buffer */
399 void (*free) (cl_device_id device, cl_mem mem_obj);
400
401 /* clEnqueSVMfree - free a SVM memory pointer. May be NULL if device doesn't
402 * support SVM. */
403 void (*svm_free) (cl_device_id dev, void *svm_ptr);
404 void *(*svm_alloc) (cl_device_id dev, cl_svm_mem_flags flags, size_t size);
405 void (*svm_map) (cl_device_id dev, void *svm_ptr);
406 void (*svm_unmap) (cl_device_id dev, void *svm_ptr);
407 /* we can use restrict here, because Spec says overlapping copy should return
408 * with CL_MEM_COPY_OVERLAP error. */
409 void (*svm_copy) (cl_device_id dev, void *__restrict__ dst,
410 const void *__restrict__ src, size_t size);
411 void (*svm_fill) (cl_device_id dev, void *__restrict__ svm_ptr, size_t size,
412 void *__restrict__ pattern, size_t pattern_size);
413
414 /* the following callbacks only deal with buffers (and IMAGE1D_BUFFER which
415 * is backed by a buffer), not images. */
416
417 /* clEnqReadBuffer */
418 void (*read) (void *data,
419 void *__restrict__ dst_host_ptr,
420 pocl_mem_identifier * src_mem_id,
421 cl_mem src_buf,
422 size_t offset,
423 size_t size);
424 /* clEnqReadBufferRect */
425 void (*read_rect) (void *data,
426 void *__restrict__ dst_host_ptr,
427 pocl_mem_identifier * src_mem_id,
428 cl_mem src_buf,
429 const size_t *buffer_origin,
430 const size_t *host_origin,
431 const size_t *region,
432 size_t buffer_row_pitch,
433 size_t buffer_slice_pitch,
434 size_t host_row_pitch,
435 size_t host_slice_pitch);
436 /* clEnqWriteBuffer */
437 void (*write) (void *data,
438 const void *__restrict__ src_host_ptr,
439 pocl_mem_identifier * dst_mem_id,
440 cl_mem dst_buf,
441 size_t offset,
442 size_t size);
443 /* clEnqWriteBufferRect */
444 void (*write_rect) (void *data,
445 const void *__restrict__ src_host_ptr,
446 pocl_mem_identifier * dst_mem_id,
447 cl_mem dst_buf,
448 const size_t *buffer_origin,
449 const size_t *host_origin,
450 const size_t *region,
451 size_t buffer_row_pitch,
452 size_t buffer_slice_pitch,
453 size_t host_row_pitch,
454 size_t host_slice_pitch);
455 /* clEnqCopyBuffer */
456 void (*copy) (void *data,
457 pocl_mem_identifier * dst_mem_id,
458 cl_mem dst_buf,
459 pocl_mem_identifier * src_mem_id,
460 cl_mem src_buf,
461 size_t dst_offset,
462 size_t src_offset,
463 size_t size);
464 /* clEnqCopyBufferRect */
465 void (*copy_rect) (void *data,
466 pocl_mem_identifier * dst_mem_id,
467 cl_mem dst_buf,
468 pocl_mem_identifier * src_mem_id,
469 cl_mem src_buf,
470 const size_t *dst_origin,
471 const size_t *src_origin,
472 const size_t *region,
473 size_t dst_row_pitch,
474 size_t dst_slice_pitch,
475 size_t src_row_pitch,
476 size_t src_slice_pitch);
477
478 /* clEnqFillBuffer */
479 void (*memfill) (void *data,
480 pocl_mem_identifier * dst_mem_id,
481 cl_mem dst_buf,
482 size_t size,
483 size_t offset,
484 const void *__restrict__ pattern,
485 size_t pattern_size);

487 /* Maps 'size' bytes of device global memory at + offset to
