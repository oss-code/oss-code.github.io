24#include "pocl_cl.h"
25#include "utlist.h"
26#include <assert.h>
27#include "pocl_util.h"
28#include "pocl_shared.h"
29
30CL_API_ENTRY void * CL_API_CALL
31POname(clEnqueueMapBuffer)(cl_command_queue command_queue,
32 cl_mem buffer,
33 cl_bool blocking_map,
34 cl_map_flags map_flags,
35 size_t offset,
36 size_t size,
37 cl_uint num_events_in_wait_list,
38 const cl_event * event_wait_list,
39 cl_event * event,
40 cl_int * errcode_ret) CL_API_SUFFIX__VERSION_1_0
41{
42 cl_int errcode = CL_SUCCESS;
43 cl_device_id device = NULL;
44 cl_int mapping_result = CL_FAILED;
45 mem_mapping_t *mapping_info = NULL;
46 unsigned i;
47 _cl_command_node *cmd = NULL;
48 /* need to release the memobject before returning? */
49 int must_release = 0;
50
51 POCL_GOTO_ERROR_COND((command_queue == NULL), CL_INVALID_COMMAND_QUEUE);
52
53 POCL_GOTO_ERROR_COND((buffer == NULL), CL_INVALID_MEM_OBJECT);
54
55 POCL_GOTO_ON_SUB_MISALIGN (buffer, command_queue);
56
57 POCL_GOTO_ERROR_COND((size == 0), CL_INVALID_VALUE);
58
59 POCL_GOTO_ERROR_ON((buffer->type != CL_MEM_OBJECT_BUFFER),
60 CL_INVALID_MEM_OBJECT, "buffer is not a CL_MEM_OBJECT_BUFFER\n");
61
62 POCL_GOTO_ERROR_ON((command_queue->context != buffer->context),
63 CL_INVALID_CONTEXT, "buffer and command_queue are not from the same context\n");
64
65 errcode = pocl_check_event_wait_list (command_queue, num_events_in_wait_list,
66 event_wait_list);
67 if (errcode != CL_SUCCESS)
68 goto ERROR;
69
70 errcode = pocl_buffer_boundcheck(buffer, offset, size);
71 if (errcode != CL_SUCCESS)
72 goto ERROR;
73
74 POCL_GOTO_ERROR_ON((buffer->flags & (CL_MEM_HOST_WRITE_ONLY | CL_MEM_HOST_NO_ACCESS) &&
75 map_flags & CL_MAP_READ), CL_INVALID_OPERATION, "buffer has been created with "
76 "CL_MEM_HOST_WRITE_ONLY or CL_MEM_HOST_NO_ACCESS and CL_MAP_READ is set in map_flags\n");
77
78 POCL_GOTO_ERROR_ON((buffer->flags & (CL_MEM_HOST_READ_ONLY | CL_MEM_HOST_NO_ACCESS) &&
79 map_flags & (CL_MAP_WRITE | CL_MAP_WRITE_INVALIDATE_REGION)), CL_INVALID_OPERATION,
80 "buffer has been created with CL_MEM_HOST_READ_ONL or CL_MEM_HOST_NO_ACCESS "
81 "and CL_MAP_WRITE or CL_MAP_WRITE_INVALIDATE_REGION is set in map_flags\n");
82
83 POCL_CHECK_DEV_IN_CMDQ;
84
85 POCL_CONVERT_SUBBUFFER_OFFSET (buffer, offset);
86
87 POCL_GOTO_ERROR_ON((buffer->size > command_queue->device->max_mem_alloc_size),
88 CL_OUT_OF_RESOURCES,
89 "buffer is larger than device's MAX_MEM_ALLOC_SIZE\n");
90
91 /* Ensure the parent buffer is not freed prematurely. */
92 POname(clRetainMemObject) (buffer);
93 must_release = 1;
94
95 mapping_info = (mem_mapping_t*) calloc (1, sizeof (mem_mapping_t));
96 POCL_GOTO_ERROR_COND ((mapping_info == NULL), CL_OUT_OF_HOST_MEMORY);
97
98 mapping_info->host_ptr = NULL;
99 mapping_info->map_flags = map_flags;
100 mapping_info->offset = offset;
101 mapping_info->size = size;
102
103 // TODO COPY_HOST_PTR
104 if ((buffer->flags & CL_MEM_USE_HOST_PTR)
105 || (buffer->flags & CL_MEM_ALLOC_HOST_PTR))
106 {
107 /* In this case it should use the given host_ptr + offset as
108 the mapping area in the host memory. */
109 assert (buffer->mem_host_ptr != NULL);
110 mapping_info->host_ptr = (char*)buffer->mem_host_ptr + offset;
111 }
112 else
113 {
114 /* The first call to the device driver's map mem tells where
115 the mapping will be stored in the host memory.
116 When return value (mapping_info->host_ptr) is non-NULL, the
117 buffer will be mapped there (assumed it will succeed). */
118 mapping_info->host_ptr = NULL;
119 mapping_result = device->ops->map_mem (
120 device->data,
121 &buffer->device_ptrs[device->dev_id],
122 buffer,
123 mapping_info);
124 }
125
126 POCL_GOTO_ERROR_ON ((mapping_info->host_ptr == NULL), CL_MAP_FAILURE,
127 "device map failed\n");
128
129 errcode = pocl_create_command (&cmd, command_queue, CL_COMMAND_MAP_BUFFER,
130 event, num_events_in_wait_list,
131 event_wait_list, 1, &buffer);
132
133 if (errcode != CL_SUCCESS)
134 goto ERROR;
135
136 cmd->command.map.mem_id = &buffer->device_ptrs[device->dev_id];
137 cmd->command.map.mapping = mapping_info;
138
139 POCL_LOCK_OBJ (buffer);
140 DL_APPEND (buffer->mappings, mapping_info);
141 buffer->owning_device = command_queue->device;
142 POCL_UNLOCK_OBJ (buffer);
143
144 POCL_MSG_PRINT_MEMORY ("Buffer %p New Mapping: host_ptr %p offset %zu\n",
145 buffer, mapping_info->host_ptr, mapping_info->offset);
146
147 pocl_command_enqueue(command_queue, cmd);
148
149 if (blocking_map)
150 {
151 POname (clFinish) (command_queue);
152 }
153
154 if (errcode_ret)
155 *errcode_ret = errcode;
156
157 POCL_SUCCESS ();
158 return mapping_info->host_ptr;
159
160ERROR:
161 if (must_release)
162 POname(clReleaseMemObject)(buffer);
163 if (mapping_result == CL_SUCCESS)
164 device->ops->unmap_mem (device->data,
165 &buffer->device_ptrs[device->dev_id],
166 buffer,
167 mapping_info);
168 POCL_MEM_FREE(mapping_info);
169 POCL_MEM_FREE (cmd);
170 if (errcode_ret)
171 *errcode_ret = errcode;
172
173 return NULL;
174}
