25#include "pocl_cl.h"
26#include "devices.h"
27#include "common.h"
28#include "pocl_util.h"
29
30CL_API_ENTRY cl_mem CL_API_CALL
31POname(clCreateBuffer)(cl_context context,
32 cl_mem_flags flags,
33 size_t size,
34 void *host_ptr,
35 cl_int *errcode_ret) CL_API_SUFFIX__VERSION_1_0
36{
37 cl_mem mem = NULL;
38 cl_device_id device;
39 int errcode;
40 unsigned i, j;
41
42 POCL_GOTO_ERROR_COND((size == 0), CL_INVALID_BUFFER_SIZE);
43
44 POCL_GOTO_ERROR_COND((context == NULL), CL_INVALID_CONTEXT);
45
46 mem = (cl_mem) malloc(sizeof(struct _cl_mem));
47 if (mem == NULL)
48 {
49 errcode = CL_OUT_OF_HOST_MEMORY;
50 goto ERROR;
51 }
52 mem->device_ptrs = NULL;
53
54 if (flags == 0)
55 flags = CL_MEM_READ_WRITE;
56
57 /* validate flags */
58
59 POCL_GOTO_ERROR_ON((flags > (1<<10)-1), CL_INVALID_VALUE, "Flags must "
60 "be < 1024 (there are only 10 flags)\n");
61
62 POCL_GOTO_ERROR_ON(((flags & CL_MEM_READ_WRITE) &&
63 (flags & CL_MEM_WRITE_ONLY || flags & CL_MEM_READ_ONLY)),
64 CL_INVALID_VALUE, "Invalid flags: CL_MEM_READ_WRITE cannot be used "
65 "together with CL_MEM_WRITE_ONLY or CL_MEM_READ_ONLY\n");
66
67 POCL_GOTO_ERROR_ON(((flags & CL_MEM_READ_ONLY) &&
68 (flags & CL_MEM_WRITE_ONLY)), CL_INVALID_VALUE, "Invalid flags: "
69 "can't have both CL_MEM_WRITE_ONLY and CL_MEM_READ_ONLY\n");
70
71 POCL_GOTO_ERROR_ON(((flags & CL_MEM_USE_HOST_PTR) &&
72 (flags & CL_MEM_ALLOC_HOST_PTR || flags & CL_MEM_COPY_HOST_PTR)),
73 CL_INVALID_VALUE, "Invalid flags: CL_MEM_USE_HOST_PTR cannot be used "
74 "together with CL_MEM_ALLOC_HOST_PTR or CL_MEM_COPY_HOST_PTR\n");
75
76 POCL_GOTO_ERROR_ON(((flags & CL_MEM_HOST_WRITE_ONLY) &&
77 (flags & CL_MEM_HOST_READ_ONLY)), CL_INVALID_VALUE, "Invalid flags: "
78 "can't have both CL_MEM_HOST_READ_ONLY and CL_MEM_HOST_WRITE_ONLY\n");
79
80 POCL_GOTO_ERROR_ON(((flags & CL_MEM_HOST_NO_ACCESS) &&
81 ((flags & CL_MEM_HOST_READ_ONLY) || (flags & CL_MEM_HOST_WRITE_ONLY))),
82 CL_INVALID_VALUE, "Invalid flags: CL_MEM_HOST_NO_ACCESS cannot be used "
83 "together with CL_MEM_HOST_READ_ONLY or CL_MEM_HOST_WRITE_ONLY\n");
84
85 if (host_ptr == NULL)
86 {
87 POCL_GOTO_ERROR_ON(((flags & CL_MEM_USE_HOST_PTR) ||
88 (flags & CL_MEM_COPY_HOST_PTR)), CL_INVALID_HOST_PTR,
89 "host_ptr is NULL, but flags specify {COPY|USE}_HOST_PTR\n");
90 }
91 else
92 {
93 POCL_GOTO_ERROR_ON(((~flags & CL_MEM_USE_HOST_PTR) &&
94 (~flags & CL_MEM_COPY_HOST_PTR)), CL_INVALID_HOST_PTR,
95 "host_ptr is not NULL, but flags don't specify {COPY|USE}_HOST_PTR\n");
96 }
97
98 POCL_GOTO_ERROR_ON ((size > context->max_mem_alloc_size),
99 CL_INVALID_BUFFER_SIZE,
100 "Size (%zu) is bigger than max mem alloc size (%zu) "
101 "of all devices in context\n",
102 size, context->max_mem_alloc_size);
103
104 POCL_INIT_OBJECT(mem);
105 mem->parent = NULL;
106 mem->map_count = 0;
107 mem->mappings = NULL;
108 mem->buffer = NULL;
109 mem->destructor_callbacks = NULL;
110 mem->type = CL_MEM_OBJECT_BUFFER;
111 mem->flags = flags;
112 mem->is_image = CL_FALSE;
113 mem->owning_device = NULL;
114 mem->is_pipe = 0;
115 mem->pipe_packet_size = 0;
116 mem->pipe_max_packets = 0;
117
118 /* Store the per device buffer pointers always to a known
119 location in the buffer (dev_id), even though the context
120 might not contain all the devices. */
121 mem->device_ptrs =
122 (pocl_mem_identifier*) calloc(pocl_num_devices,
123 sizeof(pocl_mem_identifier));
124 POCL_GOTO_ERROR_COND((mem->device_ptrs == NULL), CL_OUT_OF_HOST_MEMORY);
125
126 /* init dev pointer structure to ease inter device pointer sharing
127 in ops->alloc_mem_obj */
128 for (i = 0; i < context->num_devices; ++i)
129 {
130 int dev_id = context->devices[i]->dev_id;
131
132 mem->device_ptrs[dev_id].global_mem_id =
133 context->devices[i]->global_mem_id;
134 mem->device_ptrs[i].available = 1;
135 }
136
137 mem->size = size;
138 mem->origin = 0;
139 mem->context = context;
140 mem->mem_host_ptr = host_ptr;
141 mem->shared_mem_allocation_owner = NULL;
142
143 /* if there is a "special needs" device (hsa) operating in the host memory
144 let it alloc memory first for shared memory use */
145 if (context->svm_allocdev)
146 {
147 if (context->svm_allocdev->ops->alloc_mem_obj (context->svm_allocdev, mem, host_ptr) != CL_SUCCESS)
148 {
149 errcode = CL_MEM_OBJECT_ALLOCATION_FAILURE;
150 goto ERROR_CLEAN_MEM_AND_DEVICE;
151 }
152 }
153
154 for (i = 0; i < context->num_devices; ++i)
155 {
156 /* this is already handled iff available */
157 if (context->svm_allocdev == context->devices[i])
158 continue;
159
160 device = context->devices[i];
161 assert (device->ops->alloc_mem_obj != NULL);
162 if (device->ops->alloc_mem_obj (device, mem, host_ptr) != CL_SUCCESS)
163 {
164 errcode = CL_MEM_OBJECT_ALLOCATION_FAILURE;
165 goto ERROR_CLEAN_MEM_AND_DEVICE;
166 }
167 }
168
169 /* Some device driver may already have allocated host accessible memory */
170 if ((flags & CL_MEM_ALLOC_HOST_PTR) && (mem->mem_host_ptr == NULL))
171 {
172 assert(mem->shared_mem_allocation_owner == NULL);
173 mem->mem_host_ptr = pocl_aligned_malloc (MAX_EXTENDED_ALIGNMENT, size);
174 if (mem->mem_host_ptr == NULL)
175 {
176 errcode = CL_OUT_OF_HOST_MEMORY;
177 goto ERROR_CLEAN_MEM_AND_DEVICE;
178 }
179 }
180
181 POCL_RETAIN_OBJECT(context);
182
183 POCL_MSG_PRINT_MEMORY (
184 "Created Buffer %p, HOST_PTR: %p, DEVICE_PTR[0]: %p SIZE %zu \n", mem,
185 mem->mem_host_ptr, mem->device_ptrs[0].mem_ptr, size);
186
187 if (errcode_ret != NULL)
188 *errcode_ret = CL_SUCCESS;
189 return mem;
190
191ERROR_CLEAN_MEM_AND_DEVICE:
192 for (j = 0; j < i; ++j)
193 {
194 device = context->devices[j];
195 device->ops->free(device, mem);
196 }
197ERROR:
198 if (mem)
199 POCL_MEM_FREE (mem->device_ptrs);
200 POCL_MEM_FREE(mem);
201 if(errcode_ret)
202 {
203 *errcode_ret = errcode;
204 }
205 return NULL;
206}
