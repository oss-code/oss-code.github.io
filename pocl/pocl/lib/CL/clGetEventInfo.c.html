26CL_API_ENTRY cl_int CL_API_CALL POname (clGetEventInfo) (
27 cl_event event, cl_event_info param_name, size_t param_value_size,
28 void *param_value, size_t *param_value_size_ret) CL_API_SUFFIX__VERSION_1_0
29{
30 POCL_RETURN_ERROR_COND((event == NULL), CL_INVALID_EVENT);
31 POCL_LOCK_OBJ (event);
32 cl_int s = event->status;
33 cl_command_queue q = event->queue;
34 cl_command_type t = event->command_type;
35 cl_uint r = event->pocl_refcount;
36 cl_context c = event->context;
37 POCL_UNLOCK_OBJ (event);
38
39 switch (param_name)
40 {
41 case CL_EVENT_COMMAND_EXECUTION_STATUS:
42 POCL_RETURN_GETINFO (cl_int, s);
43 case CL_EVENT_COMMAND_QUEUE:
44 POCL_RETURN_GETINFO (cl_command_queue, q);
45 case CL_EVENT_COMMAND_TYPE:
46 POCL_RETURN_GETINFO (cl_command_type, t);
47 case CL_EVENT_REFERENCE_COUNT:
48 POCL_RETURN_GETINFO (cl_uint, r);
49 case CL_EVENT_CONTEXT:
50 POCL_RETURN_GETINFO (cl_context, c);
51 default:
52 break;
53 }
54 return CL_INVALID_VALUE;
55}
