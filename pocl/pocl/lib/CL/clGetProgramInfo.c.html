33static void get_binary_sizes(cl_program program, size_t *sizes)
34{
35#ifdef OCS_AVAILABLE
36 if (program_compile_dynamic_wg_binaries (program) != CL_SUCCESS)
37 {
38 memset(sizes, 0, program->num_devices * sizeof(size_t));
39 return;
40 }
41#endif
42 unsigned i;
43 for (i=0; i < program->num_devices; i++)
44 {
45 if (!program->pocl_binaries[i] && program->binaries[i])
46 program->pocl_binary_sizes[i] = pocl_binary_sizeof_binary(program, i);
47 if (program->pocl_binaries[i])
48 sizes[i] = program->pocl_binary_sizes[i];
49 else
50 sizes[i] = 0;
51 }
52}
53
54static void get_binaries(cl_program program, unsigned char **binaries)
55{
56#ifdef OCS_AVAILABLE
57 if (program_compile_dynamic_wg_binaries (program) != CL_SUCCESS)
58 {
59 memset(binaries, 0, program->num_devices * sizeof(unsigned char*));
60 return;
61 }
62#endif
63 unsigned i;
64 size_t res;
65 for (i=0; i < program->num_devices; i++)
66 {
67 if (!program->pocl_binaries[i] && program->binaries[i])
68 {
69 pocl_binary_serialize(program, i, &res);
70 if (program->pocl_binary_sizes[i])
71 assert(program->pocl_binary_sizes[i] == res);
72 program->pocl_binary_sizes[i] = res;
73 }
74 if (program->pocl_binaries[i])
75 memcpy(binaries[i], program->pocl_binaries[i], program->pocl_binary_sizes[i]);
76 else
77 binaries[i] = NULL;
78 }
79}

83CL_API_ENTRY cl_int CL_API_CALL
84POname(clGetProgramInfo)(cl_program program,
85 cl_program_info param_name,
86 size_t param_value_size,
87 void *param_value,
88 size_t *param_value_size_ret) CL_API_SUFFIX__VERSION_1_0
89{
90 unsigned i;
91
92 POCL_RETURN_ERROR_COND((program == NULL), CL_INVALID_PROGRAM);
93
94 switch (param_name)
95 {
96 case CL_PROGRAM_REFERENCE_COUNT:
97 POCL_RETURN_GETINFO(cl_uint, (cl_uint)program->pocl_refcount);
98 case CL_PROGRAM_CONTEXT:
99 POCL_RETURN_GETINFO(cl_context, program->context);
100
101 case CL_PROGRAM_SOURCE:
102 {
103 const char *source = program->source;
104 if (source == NULL)
105 source = "";
106
107 POCL_RETURN_GETINFO_STR(source);
108 }
109
110 case CL_PROGRAM_BINARY_SIZES:
111 {
112 POCL_RETURN_ERROR_COND(program->build_status != CL_BUILD_SUCCESS,
113 CL_INVALID_PROGRAM);
114 size_t const value_size = sizeof(size_t) * program->num_devices;
115 POCL_RETURN_GETINFO_INNER(value_size, get_binary_sizes(program, param_value));
116 }
117
118 case CL_PROGRAM_BINARIES:
119 {
120 POCL_RETURN_ERROR_COND(program->build_status != CL_BUILD_SUCCESS,
121 CL_INVALID_PROGRAM);
122 size_t const value_size = sizeof(unsigned char *) * program->num_devices;
123 POCL_RETURN_GETINFO_INNER(value_size, get_binaries(program, param_value));
124 }
125
126 case CL_PROGRAM_IL:
127 {
128 POCL_RETURN_GETINFO_INNER (
129 program->program_il_size,
130 memcpy (param_value, program->program_il, program->program_il_size));
131 }
132
133 case CL_PROGRAM_NUM_DEVICES:
134 POCL_RETURN_GETINFO(cl_uint, program->num_devices);
135
136 case CL_PROGRAM_DEVICES:
137 {
138 size_t const value_size = sizeof(cl_device_id) * program->num_devices;
139 POCL_RETURN_GETINFO_SIZE(value_size, program->devices);
140 }
141
142 case CL_PROGRAM_NUM_KERNELS:
143 {
144 POCL_RETURN_ERROR_ON((program->build_status != CL_BUILD_SUCCESS),
145 CL_INVALID_PROGRAM_EXECUTABLE,
146 "This information is only available after a "
147 "successful program executable has been built\n");
148 POCL_RETURN_GETINFO(size_t, program->num_kernels);
149 }
150
151 case CL_PROGRAM_KERNEL_NAMES:
152 {
153 POCL_RETURN_ERROR_ON((program->build_status != CL_BUILD_SUCCESS),
154 CL_INVALID_PROGRAM_EXECUTABLE,
155 "This information is only available after a "
156 "successful program executable has been built\n");
157
158 /* Note: In the specification (2.0) of the other XXXInfo
159 functions, param_value_size_ret is described as follows:
160
161 > param_value_size_ret returns the actual size in bytes of data
162 > being *queried* by param_value.
163
164 while in GetProgramInfo and APIs defined later in the
165 documentation, it is:
166
167 > param_value_size_ret returns the actual size in bytes of data
168 > *copied* to param_value.
169
170 it reads as if the spec allows the implementation to stop copying
171 the string at an arbitrary point where the limit
172 (param_value_size) is reached, but that's not the case. When it
173 happens, it should instead raise an error CL_INVALID_VALUE.
174
175 Also note the specification of the param_value_size_ret to param_name
176 CL_PROGRAM_SOURCE. It says "The actual number of characters that
177 represents[sic] the program source code including the null terminator
178 is returned in param_value_size_ret." By an analogy, it is sane to
179 return the size of entire concatenated string, not the size of
180 bytes copied (partially).
181
182 Also note the specification of GetPlatformInfo +
183 CL_PLATFORM_EXTENSIONS. it refers to "param_value_size_ret" as
184 the actual size in bytes of data being *queried*, and its
185 description of param_value_size is the same.
186
187 --- guicho271828
188 */
189 size_t num_kernels = program->num_kernels;
190 size_t size = 0;
191
192 /* optimized for clarity */
193 for (i = 0; i < num_kernels; ++i)
194 {
195 size += strlen (program->kernel_meta[i].name);
196 if (i != num_kernels - 1)
197 size += 1; /* a semicolon */
198 }
199 size += 1; /* a NULL */
200 if (param_value_size_ret)
201 *param_value_size_ret = size;
202 if (param_value)
203 {
204 /* only when param_value is non-NULL */
205 if (size > param_value_size)
206 return CL_INVALID_VALUE;
207 /* should not break from the switch clause
208 because of POCL_ABORT_UNIMPLEMENTED */
209 for (i = 0; i < num_kernels; ++i)
210 {
211 if (i == 0)
212 strcpy (param_value, program->kernel_meta[i].name); /* copy including NULL */
213 else
214 strcat ((char*)param_value, program->kernel_meta[i].name);
215 if (i != num_kernels - 1)
216 strcat ((char*)param_value, ";");
217 }
218 }
219 return CL_SUCCESS;
220 }
221 default:
222 POCL_RETURN_ERROR_ON(1, CL_INVALID_VALUE,
223 "Parameter %i not implemented\n", param_name);
224 }
