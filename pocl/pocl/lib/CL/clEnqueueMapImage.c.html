24#include "pocl_cl.h"
25#include "pocl_image_util.h"

31CL_API_ENTRY void * CL_API_CALL
32POname(clEnqueueMapImage)(cl_command_queue command_queue,
33 cl_mem image,
34 cl_bool blocking_map,
35 cl_map_flags map_flags,
36 const size_t * origin,
37 const size_t * region,
38 size_t * image_row_pitch,
39 size_t * image_slice_pitch,
40 cl_uint num_events_in_wait_list,
41 const cl_event * event_wait_list,
42 cl_event * event,
43 cl_int * errcode_ret )
44CL_API_SUFFIX__VERSION_1_0
45{
46 cl_int errcode = CL_SUCCESS;
47 cl_int mapping_result = CL_FAILED;
48 cl_device_id device;
49 _cl_command_node *cmd = NULL;
50 unsigned i;
51 mem_mapping_t *mapping_info = NULL;
52 void *retval = NULL;
53 /* need to release the memobject before returning? */
54 int must_release = 0;
55
56 POCL_GOTO_ERROR_COND((command_queue == NULL), CL_INVALID_COMMAND_QUEUE);
57
58 device = command_queue->device;
59
60 POCL_GOTO_ERROR_ON((command_queue->context != image->context),
61 CL_INVALID_CONTEXT, "image and command_queue are not from the same context\n");
62
63 POCL_GOTO_ERROR_COND((image == NULL), CL_INVALID_MEM_OBJECT);
64
65 if (IS_IMAGE1D_BUFFER (image))
66 {
67 IMAGE1D_ORIG_REG_TO_BYTES (image, origin, region)
68 return POname (clEnqueueMapBuffer) (
69 command_queue, image,
70 blocking_map, map_flags,
71 i1d_origin[0], i1d_region[0],
72 num_events_in_wait_list, event_wait_list, event,
73 errcode_ret);
74 }
75
76 POCL_GOTO_ERROR_ON ((!image->is_image), CL_INVALID_MEM_OBJECT,
77 "image argument is not an image\n");
78
79 POCL_GOTO_ON_UNSUPPORTED_IMAGE (image, command_queue->device);
80
81 errcode = pocl_check_event_wait_list (command_queue, num_events_in_wait_list,
82 event_wait_list);
83 if (errcode != CL_SUCCESS)
84 goto ERROR;
85
86 POCL_GOTO_ERROR_COND((image_row_pitch == NULL), CL_INVALID_VALUE);
87
88 errcode = pocl_check_image_origin_region(image, origin, region);
89 if (errcode != CL_SUCCESS)
90 goto ERROR;
91
92 POCL_GOTO_ERROR_ON((image_slice_pitch == NULL &&
93 (image->type == CL_MEM_OBJECT_IMAGE3D ||
94 image->type == CL_MEM_OBJECT_IMAGE1D_ARRAY ||
95 image->type == CL_MEM_OBJECT_IMAGE2D_ARRAY)), CL_INVALID_VALUE,
96 "For a 3D image, 1D, and 2D image array, "
97 "image_slice_pitch must be a non-NULL value\n");
98
99 POCL_CHECK_DEV_IN_CMDQ;
100
101 POname (clRetainMemObject) (image);
102 must_release = 1;
103
104 mapping_info = (mem_mapping_t*) calloc (1, sizeof (mem_mapping_t));
105 POCL_GOTO_ERROR_COND ((mapping_info == NULL), CL_OUT_OF_HOST_MEMORY);
106
107 *image_row_pitch = image->image_row_pitch;
108 if (image_slice_pitch)
109 *image_slice_pitch = image->image_slice_pitch;
110
111 mapping_info->host_ptr = NULL;
112 mapping_info->map_flags = map_flags;
113 mapping_info->origin[0] = origin[0];
114 mapping_info->origin[1] = origin[1];
115 mapping_info->origin[2] = origin[2];
116 mapping_info->region[0] = region[0];
117 mapping_info->region[1] = region[1];
118 mapping_info->region[2] = region[2];
119 mapping_info->row_pitch = image->image_row_pitch;
120 mapping_info->slice_pitch = image->image_slice_pitch;
121
122 size_t px = image->image_elem_size * image->image_channels;
123 mapping_info->offset = origin[2] * mapping_info->slice_pitch
124 + origin[1] * mapping_info->row_pitch
125 + origin[0] * px;
126
127 /* CL_INVALID_OPERATION if buffer has been created with
128 * CL_MEM_HOST_WRITE_ONLY
129 * or CL_MEM_HOST_NO_ACCESS and CL_MAP_READ is set in map_flags or
130 *
131 * if buffer has been created with CL_MEM_HOST_READ_ONL or
132 * CL_MEM_HOST_NO_ACCESS
133 * and CL_MAP_WRITE or CL_MAP_WRITE_INVALIDATE_REGION is set in map_flags.
134 */
135
136 POCL_GOTO_ERROR_COND (
137 ((map_flags & CL_MAP_READ)
138 && (image->flags & (CL_MEM_HOST_WRITE_ONLY | CL_MEM_HOST_NO_ACCESS))),
139 CL_INVALID_OPERATION);
140
141 POCL_GOTO_ERROR_COND (
142 ((map_flags & CL_MAP_WRITE)
143 && (image->flags & (CL_MEM_HOST_READ_ONLY | CL_MEM_HOST_NO_ACCESS))),
144 CL_INVALID_OPERATION);
145
146 if (image->flags & CL_MEM_USE_HOST_PTR)
147 {
148 /* In this case it should use the given host_ptr + offset as
149 the mapping area in the host memory. */
150 assert (image->mem_host_ptr != NULL);
151 mapping_info->host_ptr
152 = (char *)image->mem_host_ptr + mapping_info->offset;
153 }
154 else
155 {
156 /* The first call to the device driver's map mem tells where
157 the mapping will be stored in the host memory.
158 When return value (mapping_info->host_ptr) is non-NULL, the
159 buffer will be mapped there (assumed it will succeed). */
160 mapping_info->host_ptr = NULL;
161 mapping_result = device->ops->map_image
162 (device->data,
163 &image->device_ptrs[device->dev_id],
164 image,
165 mapping_info);
166 }
167
168 retval = mapping_info->host_ptr;
169 POCL_GOTO_ERROR_ON ((retval == NULL), CL_MAP_FAILURE, "device map failed\n");
170
171 errcode = pocl_create_command (&cmd, command_queue, CL_COMMAND_MAP_IMAGE,
172 event, num_events_in_wait_list,
173 event_wait_list, 1, &image);
174 if (errcode != CL_SUCCESS)
175 goto ERROR;
176
177 cmd->command.map.mem_id = &image->device_ptrs[device->dev_id];
178 cmd->command.map.mapping = mapping_info;
179
180 POCL_MSG_PRINT_MEMORY ("Image %p, Mapping: host_ptr %p offset %zu\n", image,
181 mapping_info->host_ptr, mapping_info->offset);
182
183 POCL_LOCK_OBJ (image);
184 DL_APPEND (image->mappings, mapping_info);
185 image->owning_device = device;
186 POCL_UNLOCK_OBJ (image);
187
188 pocl_command_enqueue(command_queue, cmd);
189
190 if (blocking_map)
191 {
192 POname(clFinish) (command_queue);
193 }
194 if (errcode_ret)
195 *errcode_ret = errcode;
196
197 return retval;
198
199ERROR:
200 if (must_release)
201 POname (clReleaseMemObject) (image);
202 if (mapping_result == CL_SUCCESS)
203 device->ops->unmap_image (device->data,
204 &image->device_ptrs[device->dev_id],
205 image,
206 mapping_info);
207 POCL_MEM_FREE(mapping_info);
208 POCL_MEM_FREE (cmd);
209 if (errcode_ret)
210 *errcode_ret = errcode;
211
212 return NULL;
213}
