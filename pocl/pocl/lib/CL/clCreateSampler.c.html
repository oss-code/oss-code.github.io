24#include "devices.h"
25#include "pocl_cl.h"
26#include "pocl_icd.h"

28extern CL_API_ENTRY cl_sampler CL_API_CALL
29POname(clCreateSampler)(cl_context context,
30 cl_bool normalized_coords,
31 cl_addressing_mode addressing_mode,
32 cl_filter_mode filter_mode,
33 cl_int * errcode_ret)
34CL_API_SUFFIX__VERSION_1_0
35{
36 int errcode = CL_SUCCESS;
37 cl_sampler sampler = NULL;
38
39 POCL_GOTO_ERROR_COND ((context == NULL), CL_INVALID_CONTEXT);
40
41 /* at least 1 device must support images */
42 size_t i, any_device_has_images = 0;
43 for (i = 0; i < context->num_devices; i++)
44 any_device_has_images += (size_t)context->devices[i]->image_support;
45 POCL_GOTO_ERROR_ON ((!any_device_has_images), CL_INVALID_OPERATION,
46 "None of the devices within context support images\n");
47
48 /* check requested sampler validity */
49 POCL_GOTO_ERROR_COND (
50 ((normalized_coords != CL_TRUE) && (normalized_coords != CL_FALSE)),
51 CL_INVALID_VALUE);
52 POCL_GOTO_ERROR_COND (((normalized_coords != CL_TRUE)
53 && (addressing_mode == CL_ADDRESS_MIRRORED_REPEAT)),
54 CL_INVALID_VALUE);
55 POCL_GOTO_ERROR_COND (((normalized_coords != CL_TRUE)
56 && (addressing_mode == CL_ADDRESS_REPEAT)),
57 CL_INVALID_VALUE);
58
59 sampler = (cl_sampler) malloc(sizeof(struct _cl_sampler));
60 POCL_GOTO_ERROR_COND ((sampler == NULL), CL_OUT_OF_HOST_MEMORY);
61
62 POCL_INIT_OBJECT (sampler);
63 POname (clRetainContext) (context);
64 sampler->context = context;
65 sampler->normalized_coords = normalized_coords;
66 sampler->addressing_mode = addressing_mode;
67 sampler->filter_mode = filter_mode;
68 sampler->device_data = (void **)calloc (pocl_num_devices, sizeof (void *));
69 for (i = 0; i < context->num_devices; ++i)
70 {
71 cl_device_id dev = context->devices[i];
72 if (dev->image_support == CL_TRUE && dev->ops->create_sampler)
73 sampler->device_data[dev->dev_id]
74 = dev->ops->create_sampler (dev->data, sampler, &errcode);
75 }
76
77ERROR:
78 if(errcode_ret)
79 {
80 *errcode_ret = errcode;
81 }
82 return sampler;
83}
