27#include <assert.h>
28#include "pocl_util.h"

30CL_API_ENTRY cl_int CL_API_CALL
31POname(clEnqueueCopyBuffer)(cl_command_queue command_queue,
32 cl_mem src_buffer,
33 cl_mem dst_buffer,
34 size_t src_offset,
35 size_t dst_offset,
36 size_t size,
37 cl_uint num_events_in_wait_list,
38 const cl_event *event_wait_list,
39 cl_event *event)
40CL_API_SUFFIX__VERSION_1_0
41{
42 cl_device_id device;
43 unsigned i;
44 _cl_command_node *cmd = NULL;
45 int errcode;
46
47 POCL_RETURN_ERROR_COND((command_queue == NULL), CL_INVALID_COMMAND_QUEUE);
48
49 POCL_RETURN_ERROR_COND((src_buffer == NULL), CL_INVALID_MEM_OBJECT);
50
51 POCL_RETURN_ERROR_COND((dst_buffer == NULL), CL_INVALID_MEM_OBJECT);
52
53 POCL_RETURN_ERROR_ON((src_buffer->type != CL_MEM_OBJECT_BUFFER),
54 CL_INVALID_MEM_OBJECT, "src_buffer is not a CL_MEM_OBJECT_BUFFER\n");
55 POCL_RETURN_ERROR_ON((dst_buffer->type != CL_MEM_OBJECT_BUFFER),
56 CL_INVALID_MEM_OBJECT, "dst_buffer is not a CL_MEM_OBJECT_BUFFER\n");
57
58 POCL_RETURN_ON_SUB_MISALIGN (src_buffer, command_queue);
59
60 POCL_RETURN_ON_SUB_MISALIGN (dst_buffer, command_queue);
61
62 POCL_CONVERT_SUBBUFFER_OFFSET (src_buffer, src_offset);
63
64 POCL_CONVERT_SUBBUFFER_OFFSET (dst_buffer, dst_offset);
65
66 POCL_RETURN_ERROR_ON((src_buffer->size > command_queue->device->max_mem_alloc_size),
67 CL_OUT_OF_RESOURCES,
68 "src is larger than device's MAX_MEM_ALLOC_SIZE\n");
69
70 POCL_RETURN_ERROR_ON((dst_buffer->size > command_queue->device->max_mem_alloc_size),
71 CL_OUT_OF_RESOURCES,
72 "src is larger than device's MAX_MEM_ALLOC_SIZE\n");
73
74 POCL_RETURN_ERROR_ON(((command_queue->context != src_buffer->context) ||
75 (command_queue->context != dst_buffer->context)), CL_INVALID_CONTEXT,
76 "src_buffer, dst_buffer and command_queue are not from the same context\n");
77
78 POCL_RETURN_ERROR_COND((size == 0), CL_INVALID_VALUE);
79
80 errcode = pocl_check_event_wait_list (command_queue, num_events_in_wait_list,
81 event_wait_list);
82 if (errcode != CL_SUCCESS)
83 return errcode;
84
85 if (pocl_buffers_boundcheck(src_buffer, dst_buffer, src_offset,
86 dst_offset, size) != CL_SUCCESS) return CL_INVALID_VALUE;
87
88 if (pocl_buffers_overlap(src_buffer, dst_buffer, src_offset,
89 dst_offset, size) != CL_SUCCESS) return CL_MEM_COPY_OVERLAP;
90
91 POCL_CHECK_DEV_IN_CMDQ;
92
93 cl_mem buffers[2] = { src_buffer, dst_buffer };
94
95 errcode = pocl_create_command (&cmd, command_queue, CL_COMMAND_COPY_BUFFER,
96 event, num_events_in_wait_list,
97 event_wait_list, 2, buffers);
98 if (errcode != CL_SUCCESS)
99 return errcode;
100
101 cmd->command.copy.src_mem_id = &src_buffer->device_ptrs[device->dev_id];
102 cmd->command.copy.src_offset = src_offset;
103
104 cmd->command.copy.dst_mem_id = &dst_buffer->device_ptrs[device->dev_id];
105 cmd->command.copy.dst_offset = dst_offset;
106 cmd->command.copy.size = size;
107
108 POname(clRetainMemObject)(src_buffer);
109 src_buffer->owning_device = command_queue->device;
110 POname(clRetainMemObject)(dst_buffer);
111 dst_buffer->owning_device = command_queue->device;
112
113 pocl_command_enqueue(command_queue, cmd);
114
115 return CL_SUCCESS;
116}
