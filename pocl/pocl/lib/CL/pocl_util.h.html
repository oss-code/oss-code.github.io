228/* Common macro for cleaning up "*GetInfo" API call implementations.
229 * All the *GetInfo functions have been specified to look alike,
230 * and have been implemented to use the same variable names, so this
231 * code can be shared.
232 */
233
234#define POCL_RETURN_GETINFO_INNER(__SIZE__, MEMASSIGN) \
235 do \
236 { \
237 if (param_value) \
238 { \
239 POCL_RETURN_ERROR_ON ( \
240 (param_value_size < __SIZE__), CL_INVALID_VALUE, \
241 "param_value_size (%zu) smaller than actual size (%zu)\n", \
242 param_value_size, __SIZE__); \
243 MEMASSIGN; \
244 } \
245 if (param_value_size_ret) \
246 *param_value_size_ret = __SIZE__; \
247 return CL_SUCCESS; \
248 } \
249 while (0)
250
251#define POCL_RETURN_GETINFO_SIZE(__SIZE__, __POINTER__) \
252 POCL_RETURN_GETINFO_INNER(__SIZE__, \
253 memcpy(param_value, __POINTER__, __SIZE__))
254
255#define POCL_RETURN_GETINFO_STR(__STR__) \
256 do \
257 { \
258 size_t const value_size = strlen(__STR__) + 1; \
259 POCL_RETURN_GETINFO_INNER(value_size, \
260 memcpy(param_value, __STR__, value_size)); \
261 } \
262 while (0)
263
264#define POCL_RETURN_GETINFO_STR_FREE(__STR__) \
265 do \
266 { \
267 size_t const value_size = strlen (__STR__) + 1; \
268 if (param_value) \
269 { \
270 if (param_value_size >= value_size) \
271 memcpy (param_value, __STR__, value_size); \
272 POCL_MEM_FREE (__STR__); \
273 if (param_value_size < value_size) \
274 return CL_INVALID_VALUE; \
275 } \
276 else \
277 POCL_MEM_FREE (__STR__); \
278 if (param_value_size_ret) \
279 *param_value_size_ret = value_size; \
280 return CL_SUCCESS; \
281 } \
282 while (0)
283
284#define POCL_RETURN_GETINFO(__TYPE__, __VALUE__) \
285 do \
286 { \
287 size_t const value_size = sizeof(__TYPE__); \
288 POCL_RETURN_GETINFO_INNER(value_size, \
289 *(__TYPE__*)param_value=__VALUE__); \
290 } \
291 while (0)
292
293#define POCL_RETURN_GETINFO_ARRAY(__TYPE__, __NUM__, __VALUE__) \
294 do \
295 { \
296 size_t const value_size = __NUM__*sizeof(__TYPE__); \
297 POCL_RETURN_GETINFO_INNER(value_size, \
298 memcpy(param_value, __VALUE__, value_size)); \
299 } \
300 while (0)
