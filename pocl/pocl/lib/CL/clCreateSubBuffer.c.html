24#include "devices.h"
25#include "pocl_cl.h"
26#include "pocl_util.h"
27
28CL_API_ENTRY cl_mem CL_API_CALL
29POname(clCreateSubBuffer)(cl_mem buffer,
30 cl_mem_flags flags,
31 cl_buffer_create_type buffer_create_type,
32 const void * buffer_create_info,
33 cl_int * errcode_ret) CL_API_SUFFIX__VERSION_1_1
34{
35 cl_mem mem = NULL;
36 int errcode;
37
38 POCL_GOTO_ERROR_COND((buffer == NULL), CL_INVALID_MEM_OBJECT);
39
40 POCL_GOTO_ERROR_ON ((buffer->is_image || IS_IMAGE1D_BUFFER (buffer)),
41 CL_INVALID_MEM_OBJECT,
42 "subbuffers on images not supported\n");
43
44 POCL_GOTO_ERROR_ON((buffer->parent != NULL), CL_INVALID_MEM_OBJECT,
45 "buffer is already a sub-buffer\n");
46
47 POCL_GOTO_ERROR_COND((buffer_create_info == NULL), CL_INVALID_VALUE);
48
49 POCL_GOTO_ERROR_COND((buffer_create_type != CL_BUFFER_CREATE_TYPE_REGION),
50 CL_INVALID_VALUE);
51
52 cl_buffer_region *info = (cl_buffer_region *)buffer_create_info;
53
54 POCL_GOTO_ERROR_ON((info->size == 0), CL_INVALID_BUFFER_SIZE,
55 "buffer_create_info->size == 0\n");
56
57 POCL_GOTO_ERROR_ON((info->size + info->origin > buffer->size), CL_INVALID_VALUE,
58 "buffer_create_info->size+origin > buffer size\n");
59
60 POCL_GOTO_ERROR_ON((buffer->flags & CL_MEM_WRITE_ONLY &&
61 flags & (CL_MEM_READ_WRITE | CL_MEM_READ_ONLY)), CL_INVALID_VALUE,
62 "Invalid flags: buffer is CL_MEM_WRITE_ONLY, requested sub-buffer "
63 "CL_MEM_READ_WRITE or CL_MEM_READ_ONLY\n");
64
65 POCL_GOTO_ERROR_ON((buffer->flags & CL_MEM_READ_ONLY &&
66 flags & (CL_MEM_READ_WRITE | CL_MEM_WRITE_ONLY)), CL_INVALID_VALUE,
67 "Invalid flags: buffer is CL_MEM_READ_ONLY, requested sub-buffer "
68 "CL_MEM_READ_WRITE or CL_MEM_WRITE_ONLY\n");
69
70 POCL_GOTO_ERROR_ON((flags & (CL_MEM_USE_HOST_PTR | CL_MEM_ALLOC_HOST_PTR |
71 CL_MEM_COPY_HOST_PTR)), CL_INVALID_VALUE,
72 "Invalid flags: (CL_MEM_USE_HOST_PTR | CL_MEM_ALLOC_HOST_PTR | "
73 "CL_MEM_COPY_HOST_PTR)\n");
74
75 POCL_GOTO_ERROR_ON((buffer->flags & CL_MEM_HOST_WRITE_ONLY &&
76 flags & CL_MEM_HOST_READ_ONLY), CL_INVALID_VALUE,
77 "Invalid flags: buffer is CL_MEM_HOST_WRITE_ONLY, requested sub-buffer "
78 "CL_MEM_HOST_READ_ONLY\n");
79
80 POCL_GOTO_ERROR_ON((buffer->flags & CL_MEM_HOST_READ_ONLY &&
81 flags & CL_MEM_HOST_WRITE_ONLY), CL_INVALID_VALUE,
82 "Invalid flags: buffer is CL_MEM_HOST_READ_ONLY, requested sub-buffer "
83 "CL_MEM_HOST_WRITE_ONLY\n");
84
85 POCL_GOTO_ERROR_ON((buffer->flags & CL_MEM_HOST_NO_ACCESS &&
86 flags & (CL_MEM_HOST_READ_ONLY | CL_MEM_HOST_WRITE_ONLY)), CL_INVALID_VALUE,
87 "Invalid flags: buffer is CL_MEM_HOST_NO_ACCESS, requested sub-buffer "
88 "(CL_MEM_HOST_READ_ONLY | CL_MEM_HOST_WRITE_ONLY)\n");
89
90 POCL_GOTO_ERROR_ON (
91 ((info->origin % buffer->context->min_buffer_alignment) != 0),
92 CL_MISALIGNED_SUB_BUFFER_OFFSET,
93 "no devices for which the origin value (%zu) is "
94 "aligned to the CL_DEVICE_MEM_BASE_ADDR_ALIGN value (%zu)\n",
95 info->origin, buffer->context->min_buffer_alignment);
96
97 mem = (cl_mem) calloc (1, sizeof (struct _cl_mem));
98 POCL_GOTO_ERROR_COND ((mem == NULL), CL_OUT_OF_HOST_MEMORY);
99
100 POCL_INIT_OBJECT (mem);
101 mem->context = buffer->context;
102 mem->parent = buffer;
103 mem->type = CL_MEM_OBJECT_BUFFER;
104 mem->size = info->size;
105 mem->origin = info->origin;
106 pocl_cl_mem_inherit_flags (mem, buffer, flags);
107 /* all other struct members are NULL (not valid) */
108
109 POCL_RETAIN_OBJECT(mem->parent);
110 POCL_RETAIN_OBJECT(mem->context);
111
112 POCL_MSG_PRINT_INFO ("Created Subbuffer %p, parent %p\n", mem, mem->parent);
113
114 if (errcode_ret != NULL)
115 *errcode_ret = CL_SUCCESS;
116 return mem;
117
118ERROR:
119 POCL_MEM_FREE(mem);
120 if(errcode_ret)
121 {
122 *errcode_ret = errcode;
123 }
124 return NULL;
125}
