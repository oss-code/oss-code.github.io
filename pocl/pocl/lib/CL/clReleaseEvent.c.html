24#include "pocl_cl.h"
25#include "pocl_mem_management.h"
26
27CL_API_ENTRY cl_int CL_API_CALL
28POname(clReleaseEvent)(cl_event event) CL_API_SUFFIX__VERSION_1_0
29{
30 int new_refcount;
31 POCL_RETURN_ERROR_COND((event == NULL), CL_INVALID_EVENT);
32
33 POCL_RETURN_ERROR_COND((event->context == NULL), CL_INVALID_EVENT);
34
35 POCL_RELEASE_OBJECT (event, new_refcount);
36
37 if (new_refcount == 0)
38 {
39 event_callback_item *cb_ptr = NULL;
40 event_callback_item *next = NULL;
41 for (cb_ptr = event->callback_list; cb_ptr; cb_ptr = next)
42 {
43 next = cb_ptr->next;
44 POCL_MEM_FREE (cb_ptr);
45 }
46
47 if (event->command_type == CL_COMMAND_USER)
48 {
49 pocl_user_event_data *p = event->data;
50 pthread_cond_destroy (&p->wakeup_cond);
51 POCL_MEM_FREE (p);
52 }
53
54 POCL_MSG_PRINT_REFCOUNTS ("Free event %d | %p\n", event->id, event);
55 if (event->command_type != CL_COMMAND_USER &&
56 event->queue->device->ops->free_event_data)
57 event->queue->device->ops->free_event_data(event);
58
59 POname(clReleaseContext) (event->context);
60 if (event->queue)
61 POname(clReleaseCommandQueue) (event->queue);
62
63 POCL_DESTROY_OBJECT (event);
64 pocl_mem_manager_free_event (event);
65 }
66
67 return CL_SUCCESS;
68}
