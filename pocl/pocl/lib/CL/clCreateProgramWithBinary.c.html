33/* creates either a program with binaries, or an empty program. The latter
34 * is useful for clLinkProgram() which needs an empty program to put the
35 * compiled results in.
36 */
37cl_program
38create_program_skeleton (cl_context context, cl_uint num_devices,
39 const cl_device_id *device_list,
40 const size_t *lengths, const unsigned char **binaries,
41 cl_int *binary_status, cl_int *errcode_ret,
42 int allow_empty_binaries)
43{
44 cl_program program;
45 unsigned i,j;
46 int errcode;
47 cl_device_id *unique_devlist = NULL;
48
49 POCL_GOTO_ERROR_COND((context == NULL), CL_INVALID_CONTEXT);
50
51 POCL_GOTO_ERROR_COND((device_list == NULL), CL_INVALID_VALUE);
52
53 POCL_GOTO_ERROR_COND((num_devices == 0), CL_INVALID_VALUE);
54
55 if (!allow_empty_binaries)
56 {
57 POCL_GOTO_ERROR_COND ((lengths == NULL), CL_INVALID_VALUE);
58
59 for (i = 0; i < num_devices; ++i)
60 {
61 POCL_GOTO_ERROR_ON ((lengths[i] == 0 || binaries[i] == NULL),
62 CL_INVALID_VALUE,
63 "%i-th binary is NULL or its length==0\n", i);
64 }
65 }
66
67 // check for duplicates in device_list[].
68 for (i = 0; i < context->num_devices; i++)
69 {
70 int count = 0;
71 for (j = 0; j < num_devices; j++)
72 {
73 count += context->devices[i] == device_list[j];
74 }
75 // duplicate devices
76 POCL_GOTO_ERROR_ON((count > 1), CL_INVALID_DEVICE,
77 "device %s specified multiple times\n", context->devices[i]->long_name);
78 }
79
80 // convert subdevices to devices and remove duplicates
81 cl_uint real_num_devices = 0;
82 unique_devlist = pocl_unique_device_list(device_list, num_devices, &real_num_devices);
83 num_devices = real_num_devices;
84 device_list = unique_devlist;
85
86 // check for invalid devices in device_list[].
87 for (i = 0; i < num_devices; i++)
88 {
89 int found = 0;
90 for (j = 0; j < context->num_devices; j++)
91 {
92 found |= context->devices[j] == device_list[i];
93 }
94 POCL_GOTO_ERROR_ON((!found), CL_INVALID_DEVICE,
95 "device not found in the device list of the context\n");
96 }
97
98 if ((program = (cl_program) calloc (1, sizeof (struct _cl_program))) == NULL)
99 {
100 errcode = CL_OUT_OF_HOST_MEMORY;
101 goto ERROR;
102 }
103
104 POCL_INIT_OBJECT(program);
105
106 if ((program->binary_sizes =
107 (size_t*) calloc (num_devices, sizeof(size_t))) == NULL ||
108 (program->binaries = (unsigned char**)
109 calloc (num_devices, sizeof(unsigned char*))) == NULL ||
110 (program->pocl_binaries = (unsigned char**)
111 calloc (num_devices, sizeof(unsigned char*))) == NULL ||
112 (program->pocl_binary_sizes =
113 (size_t*) calloc (num_devices, sizeof(size_t))) == NULL ||
114 (program->build_log = (char**)
115 calloc (num_devices, sizeof(char*))) == NULL ||
116 ((program->llvm_irs =
117 (void**) calloc (num_devices, sizeof(void*))) == NULL) ||
118 ((program->build_hash = (SHA1_digest_t*)
119 calloc (num_devices, sizeof(SHA1_digest_t))) == NULL))
120 {
121 errcode = CL_OUT_OF_HOST_MEMORY;
122 goto ERROR_CLEAN_PROGRAM_AND_BINARIES;
123 }
124
125 program->context = context;
126 program->num_devices = num_devices;
127 program->devices = unique_devlist;
128 program->build_status = CL_BUILD_NONE;
129 program->binary_type = CL_PROGRAM_BINARY_TYPE_NONE;
130 char program_bc_path[POCL_FILENAME_LENGTH];
131
132 if (allow_empty_binaries && (lengths == NULL) && (binaries == NULL))
133 goto SUCCESS;
134
135 for (i = 0; i < num_devices; ++i)
136 {
137#ifdef OCS_AVAILABLE
138 /* LLVM IR */
139 if (!strncmp((const char *)binaries[i], "BC", 2))
140 {
141 program->binary_sizes[i] = lengths[i];
142 program->binaries[i] = (unsigned char*) malloc(lengths[i]);
143 memcpy (program->binaries[i], binaries[i], lengths[i]);
144 if (binary_status != NULL)
145 binary_status[i] = CL_SUCCESS;
146 }
147 else
148#endif
149 /* Poclcc binary */
150 if (pocl_binary_check_binary(device_list[i], binaries[i]))
151 {
152 program->pocl_binary_sizes[i] = lengths[i];
153 program->pocl_binaries[i] = (unsigned char*) malloc (lengths[i]);
154 memcpy (program->pocl_binaries[i], binaries[i], lengths[i]);
155
156 pocl_binary_set_program_buildhash (program, i, binaries[i]);
157 int error = pocl_cache_create_program_cachedir
158 (program, i, NULL, 0, program_bc_path);
159 POCL_GOTO_ERROR_ON((error != 0), CL_BUILD_PROGRAM_FAILURE,
160 "Could not create program cachedir");
161 POCL_GOTO_ERROR_ON(pocl_binary_deserialize (program, i),
162 CL_INVALID_BINARY,
163 "Could not unpack a pocl binary\n");
164 /* read program.bc, can be useful later */
165 if (pocl_exists (program_bc_path))
166 {
167 pocl_read_file (program_bc_path,
168 (char **)(&program->binaries[i]),
169 (uint64_t *)(&program->binary_sizes[i]));
170 }
171
172 if (binary_status != NULL)
173 binary_status[i] = CL_SUCCESS;
174 }
175 /* Unknown binary */
176 else
177 {
178 POCL_MSG_WARN ("Could not recognize binary\n");
179 if (binary_status != NULL)
180 binary_status[i] = CL_INVALID_BINARY;
181 errcode = CL_INVALID_BINARY;
182 goto ERROR_CLEAN_PROGRAM_AND_BINARIES;
183 }
184 }
185
186SUCCESS:
187 POCL_RETAIN_OBJECT(context);
188
189 if (errcode_ret != NULL)
190 *errcode_ret = CL_SUCCESS;
191 return program;
192
193ERROR_CLEAN_PROGRAM_AND_BINARIES:
194 if (program->binaries)
195 for (i = 0; i < num_devices; ++i)
196 POCL_MEM_FREE(program->binaries[i]);
197 POCL_MEM_FREE(program->binaries);
198 POCL_MEM_FREE(program->binary_sizes);
199 if (program->pocl_binaries)
200 for (i = 0; i < num_devices; ++i)
201 POCL_MEM_FREE(program->pocl_binaries[i]);
202 POCL_MEM_FREE(program->pocl_binaries);
203 POCL_MEM_FREE(program->pocl_binary_sizes);
204 POCL_MEM_FREE(program);
205ERROR:
206 POCL_MEM_FREE(unique_devlist);
207 if (errcode_ret != NULL)
208 *errcode_ret = errcode;
209 return NULL;
210}

212CL_API_ENTRY cl_program CL_API_CALL POname (clCreateProgramWithBinary) (
213 cl_context context, cl_uint num_devices, const cl_device_id *device_list,
214 const size_t *lengths, const unsigned char **binaries,
215 cl_int *binary_status, cl_int *errcode_ret) CL_API_SUFFIX__VERSION_1_0
216{
217 return create_program_skeleton (context, num_devices, device_list, lengths,
218 binaries, binary_status, errcode_ret, 0);
219}
