79void
80WorkitemHandler::Initialize(Kernel *K) {
81
82 llvm::Module *M = K->getParent();
83
84 SizeTWidth = currentPoclDevice->address_bits;
85 SizeT = IntegerType::get(M->getContext(), SizeTWidth);
86
87 assert ((SizeTWidth == 32 || SizeTWidth == 64) &&
88 "Only 32 and 64 bit size_t widths supported.");
89
90 llvm::Type *LocalIdType = SizeT;
91 LocalIdZGlobal = M->getOrInsertGlobal(POCL_LOCAL_ID_Z_GLOBAL, LocalIdType);
92 LocalIdYGlobal = M->getOrInsertGlobal(POCL_LOCAL_ID_Y_GLOBAL, LocalIdType);
93 LocalIdXGlobal = M->getOrInsertGlobal(POCL_LOCAL_ID_X_GLOBAL, LocalIdType);
94}

97bool
98WorkitemHandler::dominatesUse
99(llvm::DominatorTreeWrapperPass *DTP, Instruction &I, unsigned i) {
100 DominatorTree *DT = &DTP->getDomTree();
101 Instruction *Op = cast<Instruction>(I.getOperand(i));
102 BasicBlock *OpBlock = Op->getParent();
103 PHINode *PN = dyn_cast<PHINode>(&I);
104
105 // DT can handle non phi instructions for us.
106 if (!PN)
107 {
108 // Definition must dominate use unless use is unreachable!
109 return Op->getParent() == I.getParent() ||
110 DT->dominates(Op, &I);
111 }
112
113 // PHI nodes are more difficult than other nodes because they actually
114 // "use" the value in the predecessor basic blocks they correspond to.
115 unsigned j = PHINode::getIncomingValueNumForOperand(i);
116 BasicBlock *PredBB = PN->getIncomingBlock(j);
117 return (PredBB && DT->dominates(OpBlock, PredBB));
118}

120/* Fixes the undominated variable uses.
121
122 These appear when a conditional barrier kernel is replicated to
123 form a copy of the *same basic block* in the alternative
124 "barrier path".
125
126 E.g., from
127
128 A -> [exit], A -> B -> [exit]
129
130 a replicated CFG as follows, is created:
131
132 A1 -> (T) A2 -> [exit1], A1 -> (F) A2' -> B1, B2 -> [exit2]
133
134 The regions are correct because of the barrier semantics
135 of "all or none". In case any barrier enters the [exit1]
136 from A1, all must (because there's a barrier in the else
137 branch).
138
139 Here at A2 and A2' one creates the same variables.
140 However, B2 does not know which copy
141 to refer to, the ones created in A2 or ones in A2' (correct).
142 The mapping data contains only one possibility, the
143 one that was placed there last. Thus, the instructions in B2
144 might end up referring to the variables defined in A2
145 which do not nominate them.
146
147 The variable references are fixed by exploiting the knowledge
148 of the naming convention of the cloned variables.
149
150 One potential alternative way would be to collect the refmaps per BB,
151 not globally. Then as a final phase traverse through the
152 basic blocks starting from the beginning and propagating the
153 reference data downwards, the data from the new BB overwriting
154 the old one. This should ensure the reachability without
155 the costly dominance analysis.
156*/
157bool
158WorkitemHandler::fixUndominatedVariableUses(llvm::DominatorTreeWrapperPass *DT,
159 llvm::Function &F)
160{
161 bool changed = false;
162 DT->runOnFunction(F);
163
164 for (Function::iterator i = F.begin(), e = F.end(); i != e; ++i)
165 {
166 llvm::BasicBlock *bb = &*i;
167 for (llvm::BasicBlock::iterator ins = bb->begin(), inse = bb->end();
168 ins != inse; ++ins)
169 {
170 for (unsigned opr = 0; opr < ins->getNumOperands(); ++opr)
171 {
172 if (!isa<Instruction>(ins->getOperand(opr))) continue;
173 Instruction *operand = cast<Instruction>(ins->getOperand(opr));
174 if (dominatesUse(DT, *ins, opr))
175 continue;
176#ifdef DEBUG_REFERENCE_FIXING
177 std::cout << "### dominance error!" << std::endl;
178 operand->dump();
179 std::cout << "### does not dominate:" << std::endl;
180 ins->dump();
181#endif
182 StringRef baseName;
183 std::pair< StringRef, StringRef > pieces =
184 operand->getName().rsplit('.');
185 if (pieces.second.startswith("pocl_"))
186 baseName = pieces.first;
187 else
188 baseName = operand->getName();
189
190 Value *alternative = NULL;
191
192 unsigned int copy_i = 0;
193 do {
194 std::ostringstream alternativeName;
195 alternativeName << baseName.str();
196 if (copy_i > 0)
197 alternativeName << ".pocl_" << copy_i;
198
199 alternative =
200 F.getValueSymbolTable()->lookup(alternativeName.str());
201
202 if (alternative != NULL)
203 {
204 ins->setOperand(opr, alternative);
205 if (dominatesUse(DT, *ins, opr))
206 break;
207 }
208
209 if (copy_i > 10000 && alternative == NULL)
210 break; /* ran out of possibilities */
211 ++copy_i;
212 } while (true);
213
214 if (alternative != NULL) {
215#ifdef DEBUG_REFERENCE_FIXING
216 std::cout << "### found the alternative:" << std::endl;
217 alternative->dump();
218#endif
219 changed |= true;
220 } else {
221#ifdef DEBUG_REFERENCE_FIXING
222 std::cout << "### didn't find an alternative for" << std::endl;
223 operand->dump();
224 std::cerr << "### BB:" << std::endl;
225 operand->getParent()->dump();
226 std::cerr << "### the user BB:" << std::endl;
227 ins->getParent()->dump();
228#endif
229 std::cerr << "Could not find a dominating alternative variable." << std::endl;
230 dumpCFG(F, "broken.dot");
231 abort();
232 }
233 }
234 }
235 }
236 return changed;
237}

239/**
240 * Moves the phi nodes in the beginning of the src to the beginning of
241 * the dst.
242 *
243 * MergeBlockIntoPredecessor function from llvm discards the phi nodes
244 * of the replicated BB because it has only one entry.
245 */
246void
247WorkitemHandler::movePhiNodes(llvm::BasicBlock* src, llvm::BasicBlock* dst) {
248 while (PHINode *PN = dyn_cast<PHINode>(src->begin()))
249 PN->moveBefore(dst->getFirstNonPHI());
250}
