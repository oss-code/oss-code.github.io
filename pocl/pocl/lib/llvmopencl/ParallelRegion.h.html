47 // TODO Cleanup: this should not inherit vector but contain it.
48 // It now exposes too much to the clients and leads to hard
49 // to track errors when the API is changed.
50 class ParallelRegion : public std::vector<llvm::BasicBlock *> {
51 public:
52 typedef llvm::SmallVector<ParallelRegion *, 8> ParallelRegionVector;
53
54 ParallelRegion(int forcedRegionId=-1);
55
56 /* BarrierBlock *getEntryBarrier(); */
57 ParallelRegion *replicate(llvm::ValueToValueMapTy &map,
58 const llvm::Twine &suffix);
59 void remap(llvm::ValueToValueMapTy &map);
60 void purge();
61 void chainAfter(ParallelRegion *region);
62 void insertPrologue(unsigned x, unsigned y, unsigned z);
63 static void insertLocalIdInit(llvm::BasicBlock* entry,
64 unsigned x,
65 unsigned y,
66 unsigned z);
67 void dump();
68 void dumpNames();
69 void setEntryBBIndex(std::size_t index) { entryIndex_ = index; }
70 void setExitBBIndex(std::size_t index) { exitIndex_ = index; }
71 void SetExitBB(llvm::BasicBlock *block);
72 void AddBlockBefore(llvm::BasicBlock *block, llvm::BasicBlock *before);
73 void AddBlockAfter(llvm::BasicBlock *block, llvm::BasicBlock *after);
74
75 llvm::BasicBlock* exitBB() { return at(exitIndex_); }
76 llvm::BasicBlock* entryBB() { return at(entryIndex_); }
77 void AddIDMetadata(llvm::LLVMContext& context,
78 std::size_t x = 0,
79 std::size_t y = 0,
80 std::size_t z = 0);
81
82 void AddParallelLoopMetadata(llvm::MDNode *Identifier);
83
84 bool HasBlock(llvm::BasicBlock *bb);
85
86 void InjectRegionPrintF();
87 void InjectVariablePrintouts();
88
89 void InjectPrintF
90 (llvm::Instruction *before, std::string formatStr,
91 std::vector<llvm::Value*>& params);
92
93 static ParallelRegion *
94 Create(const llvm::SmallPtrSet<llvm::BasicBlock *, 8>& bbs,
95 llvm::BasicBlock *entry, llvm::BasicBlock *exit);
96
97 static void GenerateTempNames(llvm::BasicBlock *bb);
98
99 llvm::Instruction* LocalIDXLoad();
100 llvm::Instruction* LocalIDYLoad();
101 llvm::Instruction* LocalIDZLoad();
102
103 void LocalizeIDLoads();
104
105 int GetID() const { return pRegionId; }
106
107 private:
108 llvm::Instruction* LocalIDXLoadInstr;
109 llvm::Instruction* LocalIDYLoadInstr;
110 llvm::Instruction* LocalIDZLoadInstr;
111
112 bool Verify();
113 /// The indices of entry and exit, not pointers, for finding the BBs in the
114 /// replicated PRs too.
115 std::size_t exitIndex_;
116 std::size_t entryIndex_;
117
118 /// Identifier for the parallel region.
119 int pRegionId;
120 static int idGen;
121
122 };
