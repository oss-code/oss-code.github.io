50void
51Kernel::getExitBlocks(SmallVectorImpl<llvm::BasicBlock *> &B)
52{
53 for (iterator i = begin(), e = end(); i != e; ++i) {
54 auto t = i->getTerminator();
55 if (t->getNumSuccessors() == 0) {
56 // All exits must be barrier blocks.
57 llvm::BasicBlock *BB = cast<BasicBlock>(i);
58 if (!Barrier::hasBarrier(BB))
59 Barrier::Create(BB->getTerminator());
60 B.push_back(BB);
61 }
62 }
63}

65ParallelRegion *
66Kernel::createParallelRegionBefore(llvm::BasicBlock *B)
67{
68 SmallVector<BasicBlock *, 4> pending_blocks;
69 SmallPtrSet<BasicBlock *, 8> blocks_in_region;
70 BasicBlock *region_entry_barrier = NULL;
71 BasicBlock *entry = NULL;
72 BasicBlock *exit = B->getSinglePredecessor();
73 add_predecessors(pending_blocks, B);
74
75#ifdef DEBUG_PR_CREATION
76 std::cerr << "createParallelRegionBefore " << B->getName().str() << std::endl;
77#endif
78
79 while (!pending_blocks.empty()) {
80 BasicBlock *current = pending_blocks.back();
81 pending_blocks.pop_back();
82
83#ifdef DEBUG_PR_CREATION
84 std::cerr << "considering " << current->getName().str() << std::endl;
85#endif
86
87 // avoid infinite recursion of loops
88 if (blocks_in_region.count(current) != 0)
89 {
90#ifdef DEBUG_PR_CREATION
91 std::cerr << "already in the region!" << std::endl;
92#endif
93 continue;
94 }
95
96 // If we reach another barrier this must be the
97 // parallel region entry.
98 if (Barrier::hasOnlyBarrier(current)) {
99 if (region_entry_barrier == NULL)
100 region_entry_barrier = current;
101#ifdef DEBUG_PR_CREATION
102 std::cerr << "### it's a barrier!" << std::endl;
103#endif
104 continue;
105 }
106
107
108 if (!verify_no_barriers(current))
109 {
110 assert(verify_no_barriers(current) &&
111 "Barrier found in a non-barrier block! (forgot barrier canonicalization?)");
112 }
113
114#ifdef DEBUG_PR_CREATION
115 std::cerr << "added it to the region" << std::endl;
116#endif
117 // Non-barrier block, this must be on the region.
118 blocks_in_region.insert(current);
119
120 // Add predecessors to pending queue.
121 add_predecessors(pending_blocks, current);
122 }
123
124 if (blocks_in_region.empty())
125 return NULL;
126
127 // Find the entry node.
128 assert (region_entry_barrier != NULL);
129 for (unsigned suc = 0, num = region_entry_barrier->getTerminator()->getNumSuccessors();
130 suc < num; ++suc)
131 {
132 llvm::BasicBlock *entryCandidate =
133 region_entry_barrier->getTerminator()->getSuccessor(suc);
134 if (blocks_in_region.count(entryCandidate) == 0)
135 continue;
136 entry = entryCandidate;
137 break;
138 }
139 assert (blocks_in_region.count(entry) != 0);
140
141 // We got all the blocks in a region, create it.
142 return ParallelRegion::Create(blocks_in_region, entry, exit);
143}

145static void
146add_predecessors(SmallVectorImpl<BasicBlock *> &v, BasicBlock *b)
147{
148 for (pred_iterator i = pred_begin(b), e = pred_end(b);
149 i != e; ++i) {
150 v.push_back(*i);
151 }
152}
153
154static bool
155verify_no_barriers(const BasicBlock *B)
156{
157 for (BasicBlock::const_iterator i = B->begin(), e = B->end(); i != e; ++i) {
158 if (isa<Barrier>(i))
159 return false;
160 }
161
162 return true;
163}

165/**
166 * The main entry to the "parallel region formation", phase which search
167 * for the regions between barriers that can be freely parallelized
168 * across work-items in the work-group.
169 */
170ParallelRegion::ParallelRegionVector *
171Kernel::getParallelRegions(llvm::LoopInfo *LI) {
172 ParallelRegion::ParallelRegionVector *parallel_regions =
173 new ParallelRegion::ParallelRegionVector;
174
175 SmallVector<BasicBlock *, 4> exit_blocks;
176 getExitBlocks(exit_blocks);
177
178 // We need to keep track of traversed barriers to detect back edges.
179 SmallPtrSet<BasicBlock *, 8> found_barriers;
180
181 // First find all the ParallelRegions in the Function.
182 while (!exit_blocks.empty()) {
183
184 // We start on an exit block and process the parallel regions upwards
185 // (finding an execution trace).
186 BasicBlock *exit = exit_blocks.back();
187 exit_blocks.pop_back();
188
189 while (ParallelRegion *PR = createParallelRegionBefore(exit)) {
190 assert(PR != NULL && !PR->empty() &&
191 "Empty parallel region in kernel (contiguous barriers)!");
192
193 found_barriers.insert(exit);
194 exit = NULL;
195 parallel_regions->push_back(PR);
196 BasicBlock *entry = PR->entryBB();
197 int found_predecessors = 0;
198 BasicBlock *loop_barrier = NULL;
199 for (pred_iterator i = pred_begin(entry), e = pred_end(entry);
200 i != e; ++i) {
201 BasicBlock *barrier = (*i);
202 if (!found_barriers.count(barrier)) {
203 /* If this is a loop header block we might have edges from two
204 unprocessed barriers. The one inside the loop (coming from a
205 computation block after a branch block) should be processed
206 first. */
207 std::string bbName = "";
208 const bool IS_IN_THE_SAME_LOOP =
209 LI->getLoopFor(barrier) != NULL &&
210 LI->getLoopFor(entry) != NULL &&
211 LI->getLoopFor(entry) == LI->getLoopFor(barrier);
212
213 if (IS_IN_THE_SAME_LOOP)
214 {
215#ifdef DEBUG_PR_CREATION
216 std::cout << "### found a barrier inside the loop:" << std::endl;
217 std::cout << barrier->getName().str() << std::endl;
218#endif
219 if (loop_barrier != NULL) {
220 // there can be multiple latches and each have their barrier,
221 // save the previously found inner loop barrier
222 exit_blocks.push_back(loop_barrier);
223 }
224 loop_barrier = barrier;
225 }
226 else
227 {
228#ifdef DEBUG_PR_CREATION
229 std::cout << "### found a barrier:" << std::endl;
230 std::cout << barrier->getName().str() << std::endl;
231#endif
232 exit = barrier;
233 }
234 ++found_predecessors;
235 }
236 }
237
238 if (loop_barrier != NULL)
239 {
240 /* The secondary barrier to process in case it was a loop
241 header. Push it for later processing. */
242 if (exit != NULL)
243 exit_blocks.push_back(exit);
244 /* always process the inner loop regions first */
245 if (!found_barriers.count(loop_barrier))
246 exit = loop_barrier;
247 }
248
249#ifdef DEBUG_PR_CREATION
250 std::cout << "### created a ParallelRegion:" << std::endl;
251 PR->dumpNames();
252 std::cout << std::endl;
253#endif
254
255 if (found_predecessors == 0)
256 {
257 /* This path has been traversed and we encountered no more
258 unprocessed regions. It means we have either traversed all
259 paths from the exit or have transformed a loop and thus
260 encountered only a barrier that was seen (and thus
261 processed) before. */
262 break;
263 }
264 assert ((exit != NULL) && "Parallel region without entry barrier!");
265 }
266 }
267
268#ifdef DEBUG_PR_CREATION
269 pocl::dumpCFG(*this, this->getName().str() + ".pregions.dot", parallel_regions);
270#endif
271 return parallel_regions;
272
273}

275void
276Kernel::addLocalSizeInitCode(size_t LocalSizeX, size_t LocalSizeY, size_t LocalSizeZ) {
277
278 IRBuilder<> Builder(getEntryBlock().getFirstNonPHI());
279
280 GlobalVariable *GV;
281
282 llvm::Module* M = getParent();
283
284 llvm::Type *SizeT =
285 IntegerType::get(M->getContext(), currentPoclDevice->address_bits);
286
287 GV = M->getGlobalVariable("_local_size_x");
288 if (GV != NULL)
289 Builder.CreateStore(ConstantInt::get(SizeT, LocalSizeX), GV);
290
291 GV = M->getGlobalVariable("_local_size_y");
292 if (GV != NULL)
293 Builder.CreateStore(ConstantInt::get(SizeT, LocalSizeY), GV);
294
295 GV = M->getGlobalVariable("_local_size_z");
296 if (GV != NULL)
297 Builder.CreateStore(ConstantInt::get(SizeT, LocalSizeZ), GV);
298}
