50void
51PHIsToAllocas::getAnalysisUsage(AnalysisUsage &AU) const {
52 AU.addRequired<pocl::WorkitemHandlerChooser>();
53 AU.addPreserved<pocl::WorkitemHandlerChooser>();
54
55 AU.addRequired<pocl::VariableUniformityAnalysis>();
56 AU.addPreserved<pocl::VariableUniformityAnalysis>();
57}

59bool
60PHIsToAllocas::runOnFunction(Function &F) {
61 if (!Workgroup::isKernelToProcess(F))
62 return false;
63
64 /* Skip PHIsToAllocas when we are not creating the work item loops,
65 as it leads to worse code without benefits for the full replication method.
66 */
67 if (getAnalysis<pocl::WorkitemHandlerChooser>().chosenHandler() !=
68 pocl::WorkitemHandlerChooser::POCL_WIH_LOOPS)
69 return false;
70
71 typedef std::vector<llvm::Instruction* > InstructionVec;
72
73 InstructionVec PHIs;
74
75 for (Function::iterator bb = F.begin(); bb != F.end(); ++bb) {
76 for (BasicBlock::iterator p = bb->begin();
77 p != bb->end(); ++p) {
78 Instruction* instr = &*p;
79 if (isa<PHINode>(instr)) {
80 PHIs.push_back(instr);
81 }
82 }
83 }
84
85 bool changed = false;
86 for (InstructionVec::iterator i = PHIs.begin(); i != PHIs.end();
87 ++i) {
88 Instruction *instr = *i;
89 BreakPHIToAllocas(dyn_cast<PHINode>(instr));
90 changed = true;
91 }
92 return changed;
93}

95/**
96 * Convert a PHI to a read from a stack value and all the sources to
97 * writes to the same stack value.
98 *
99 * Used to fix context save/restore issues with regions with PHI nodes in the
100 * entry node (usually due to the use of work group scope variables such as
101 * B-loop iteration variables). In case of PHI nodes at region entries, we cannot
102 * just insert the context restore code because it is assumed there are no
103 * non-phi Instructions before PHIs which the context restore
104 * code constitutes to. Secondly, in case the PHINode is at a
105 * region entry (e.g. a B-Loop) adding new basic blocks before it would
106 * break the assumption of single entry regions.
107 */
108llvm::Instruction *
109PHIsToAllocas::BreakPHIToAllocas(PHINode* phi) {
110
111 // Loop iteration variables can be detected only when they are
112 // implemented using PHI nodes. Maintain information of the
113 // split PHI nodes in the VUA by first analyzing the function
114 // with the PHIs intact and propagating the uniformity info
115 // of the PHI nodes.
116 VariableUniformityAnalysis &VUA =
117 getAnalysis<VariableUniformityAnalysis>();
118
119 std::string allocaName = std::string(phi->getName().str()) + ".ex_phi";
120
121 llvm::Function *function = phi->getParent()->getParent();
122
123 const bool OriginalPHIWasUniform = VUA.isUniform(function, phi);
124
125 IRBuilder<> builder(&*(function->getEntryBlock().getFirstInsertionPt()));
126
127 llvm::Instruction *alloca =
128 builder.CreateAlloca(phi->getType(), 0, allocaName);
129
130 for (unsigned incoming = 0; incoming < phi->getNumIncomingValues();
131 ++incoming) {
132 Value *val = phi->getIncomingValue(incoming);
133 BasicBlock *incomingBB = phi->getIncomingBlock(incoming);
134 builder.SetInsertPoint(incomingBB->getTerminator());
135 llvm::Instruction *store = builder.CreateStore(val, alloca);
136 if (OriginalPHIWasUniform)
137 VUA.setUniform(function, store);
138 }
139 builder.SetInsertPoint(phi);
140
141 llvm::Instruction *loadedValue = builder.CreateLoad(alloca);
142 phi->replaceAllUsesWith(loadedValue);
143
144 if (OriginalPHIWasUniform) {
145#ifdef DEBUG_PHIS_TO_ALLOCAS
146 std::cout << "PHIsToAllocas: Original PHI was uniform" << std::endl
147 << "original:";
148 phi->dump();
149 std::cout << "alloca:";
150 alloca->dump();
151 std::cout << "loadedValue:";
152 loadedValue->dump();
153#endif
154 VUA.setUniform(function, alloca);
155 VUA.setUniform(function, loadedValue);
156 }
157 phi->eraseFromParent();
158
159 return loadedValue;
160}
