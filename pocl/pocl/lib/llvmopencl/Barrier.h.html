36#define BARRIER_FUNCTION_NAME "pocl.barrier"
37
38namespace pocl {
39
40 class Barrier : public llvm::CallInst {
41
42 public:
43 static void GetBarriers(llvm::SmallVectorImpl<Barrier *> &B,
44 llvm::Module &M) {
45 llvm::Function *F = M.getFunction(BARRIER_FUNCTION_NAME);
46 if (F != NULL) {
47 for (llvm::Function::use_iterator i = F->use_begin(), e = F->use_end();
48 i != e; ++i)
49 B.push_back(llvm::cast<Barrier>(*i));
50 }
51 }
52 /**
53 * Creates a new barrier before the given instruction.
54 *
55 * If there was already a barrier there, returns the old one.
56 */
57 static Barrier *Create(llvm::Instruction *InsertBefore) {
58 llvm::Module *M = InsertBefore->getParent()->getParent()->getParent();
59
60 if (InsertBefore != &InsertBefore->getParent()->front() &&
61 llvm::isa<Barrier>(InsertBefore->getPrevNode()))
62 return llvm::cast<Barrier>(InsertBefore->getPrevNode());
63#ifdef LLVM_OLDER_THAN_9_0
64 llvm::Function *F = llvm::cast<llvm::Function>
65 (M->getOrInsertFunction(BARRIER_FUNCTION_NAME,
66 llvm::Type::getVoidTy(M->getContext())));
67#else
68 llvm::FunctionCallee f =
69 M->getOrInsertFunction(BARRIER_FUNCTION_NAME,
70 llvm::Type::getVoidTy(M->getContext()));
71 llvm::Function *F = llvm::cast<llvm::Function>(f.getCallee());
72#endif
73 F->addFnAttr(llvm::Attribute::NoDuplicate);
74 F->setLinkage(llvm::GlobalValue::LinkOnceAnyLinkage);
75 return llvm::cast<pocl::Barrier>
76 (llvm::CallInst::Create(F, "", InsertBefore));
77 }
78 static bool classof(const Barrier *) { return true; }
79 static bool classof(const llvm::CallInst *C) {
80 return C->getCalledFunction() != NULL &&
81 C->getCalledFunction()->getName() == BARRIER_FUNCTION_NAME;
82 }
83 static bool classof(const Instruction *I) {
84 return (llvm::isa<llvm::CallInst>(I) &&
85 classof(llvm::cast<llvm::CallInst>(I)));
86 }
87 static bool classof(const User *U) {
88 return (llvm::isa<Instruction>(U) &&
89 classof(llvm::cast<llvm::Instruction>(U)));
90 }
91 static bool classof(const Value *V) {
92 return (llvm::isa<User>(V) &&
93 classof(llvm::cast<llvm::User>(V)));
94 }
95
96
97
98 static bool hasOnlyBarrier(const llvm::BasicBlock *bb) {
99 return endsWithBarrier(bb) && bb->size() == 2;
100 }
101
102 static bool hasBarrier(const llvm::BasicBlock *bb)
103 {
104 for (llvm::BasicBlock::const_iterator i = bb->begin(), e = bb->end();
105 i != e; ++i)
106 {
107 if (llvm::isa<Barrier>(i)) return true;
108 }
109 return false;
110 }
111
112 // Returns true in case the given basic block starts with a barrier,
113 // that is, contains a branch instruction after possible PHI nodes.
114 static bool startsWithBarrier(const llvm::BasicBlock *bb) {
115 const llvm::Instruction *i = bb->getFirstNonPHI();
116 if (i == NULL)
117 return false;
118 return llvm::isa<Barrier>(i);
119 }
120
121 // Returns true in case the given basic block ends with a barrier,
122 // that is, contains only a branch instruction after a barrier call.
123 static bool endsWithBarrier(const llvm::BasicBlock *bb) {
124 const llvm::Instruction *t = bb->getTerminator();
125 if (t == NULL)
126 return false;
127 return bb->size() > 1 && t->getPrevNode() != NULL &&
128 llvm::isa<Barrier>(t->getPrevNode());
129 }
130 };
