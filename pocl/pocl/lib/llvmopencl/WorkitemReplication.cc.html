78void
79WorkitemReplication::getAnalysisUsage(AnalysisUsage &AU) const
80{
81 AU.addRequired<DominatorTreeWrapperPass>();
82
83 AU.addRequired<LoopInfoWrapperPass>();
84 AU.addRequired<pocl::WorkitemHandlerChooser>();
85 AU.addPreserved<pocl::VariableUniformityAnalysis>();
86}

88bool
89WorkitemReplication::runOnFunction(Function &F)
90{
91 if (!Workgroup::isKernelToProcess(F))
92 return false;
93
94 if (getAnalysis<pocl::WorkitemHandlerChooser>().chosenHandler() !=
95 pocl::WorkitemHandlerChooser::POCL_WIH_FULL_REPLICATION)
96 return false;
97
98 DTP = &getAnalysis<DominatorTreeWrapperPass>();
99 DT = &DTP->getDomTree();
100
101 LI = &getAnalysis<LoopInfoWrapperPass>();
102
103 bool changed = ProcessFunction(F);
104#ifdef DUMP_RESULT_CFG
105 FunctionPass* cfgPrinter = createCFGPrinterPass();
106 cfgPrinter->runOnFunction(F);
107#endif
108
109 changed |= fixUndominatedVariableUses(DTP, F);
110 return changed;
111}

113bool
114WorkitemReplication::ProcessFunction(Function &F)
115{
116 Module *M = F.getParent();
117
118// F.viewCFG();
119
120 Kernel *K = cast<Kernel> (&F);
121 Initialize(K);
122
123 // Allocate space for workitem reference maps. Workitem 0 does
124 // not need it.
125 unsigned workitem_count = WGLocalSizeZ * WGLocalSizeY * WGLocalSizeX;
126
127 assert (workitem_count > 0);
128
129 BasicBlockVector original_bbs;
130 for (Function::iterator i = F.begin(), e = F.end(); i != e; ++i) {
131 if (!Barrier::hasBarrier(&*i))
132 original_bbs.push_back(&*i);
133 }
134
135 ParallelRegion::ParallelRegionVector* original_parallel_regions =
136 K->getParallelRegions(&LI->getLoopInfo());
137
138 std::vector<ParallelRegion::ParallelRegionVector> parallel_regions(
139 workitem_count);
140
141 parallel_regions[0] = *original_parallel_regions;
142
143 //pocl::dumpCFG(F, F.getName().str() + ".before_repl.dot", original_parallel_regions);
144
145 /* Enable to get region identification printouts */
146#if 0
147 for (ParallelRegion::ParallelRegionVector::iterator
148 i = original_parallel_regions->begin(),
149 e = original_parallel_regions->end();
150 i != e; ++i)
151 {
152 ParallelRegion *region = (*i);
153 region->InjectRegionPrintF();
154 region->InjectVariablePrintouts();
155 }
156#endif
157
158 // Measure the required context (variables alive in more than one region).
159
160 for (SmallVector<ParallelRegion *, 8>::iterator
161 i = original_parallel_regions->begin(),
162 e = original_parallel_regions->end();
163 i != e; ++i) {
164 ParallelRegion *pr = (*i);
165
166 for (ParallelRegion::iterator i2 = pr->begin(), e2 = pr->end();
167 i2 != e2; ++i2) {
168 BasicBlock *bb = (*i2);
169
170 for (BasicBlock::iterator i3 = bb->begin(), e3 = bb->end();
171 i3 != e3; ++i3) {
172 for (Value::use_iterator i4 = i3->use_begin(), e4 = i3->use_end();
173 i4 != e4; ++i4) {
174 // Instructions can only be used by instructions.
175 llvm::Instruction *user = cast<Instruction>(i4->getUser());
176
177 if (find (pr->begin(), pr->end(), user->getParent()) ==
178 pr->end()) {
179 // User is not in the defining region.
180 ++ContextValues;
181 ContextSize += F.getParent()->getDataLayout().getTypeAllocSize(i3->getType());
182 break;
183 }
184 }
185 }
186 }
187 }
188
189 // Then replicate the ParallelRegions.
190 ValueToValueMapTy *const reference_map =
191 new ValueToValueMapTy[workitem_count - 1];
192 for (unsigned z = 0; z < WGLocalSizeZ; ++z) {
193 for (unsigned y = 0; y < WGLocalSizeY; ++y) {
194 for (unsigned x = 0; x < WGLocalSizeX ; ++x) {
195
196 int index =
197 (WGLocalSizeY * WGLocalSizeX * z + WGLocalSizeX * y + x);
198
199 if (index == 0)
200 continue;
201
202 for (SmallVector<ParallelRegion *, 8>::iterator
203 i = original_parallel_regions->begin(),
204 e = original_parallel_regions->end();
205 i != e; ++i) {
206 ParallelRegion *original = (*i);
207 ParallelRegion *replicated =
208 original->replicate
209 (reference_map[index - 1],
210 (".wi_" + Twine(x) + "_" + Twine(y) + "_" + Twine(z)));
211 if (AddWIMetadata)
212 replicated->AddIDMetadata(M->getContext(), x, y, z);
213 parallel_regions[index].push_back(replicated);
214#ifdef DEBUG_PR_REPLICATION
215 std::cerr << "### new replica:" << std::endl;
216 replicated->dump();
217#endif
218 }
219 }
220 }
221 }
222 if (AddWIMetadata) {
223 for (SmallVector<ParallelRegion *, 8>::iterator
224 i = original_parallel_regions->begin(),
225 e = original_parallel_regions->end();
226 i != e; ++i) {
227 ParallelRegion *original = (*i);
228 original->AddIDMetadata(M->getContext(), 0, 0, 0);
229 }
230 }
231
232 for (unsigned z = 0; z < WGLocalSizeZ; ++z) {
233 for (unsigned y = 0; y < WGLocalSizeY; ++y) {
234 for (unsigned x = 0; x < WGLocalSizeX ; ++x) {
235
236 int index =
237 (WGLocalSizeY * WGLocalSizeX * z + WGLocalSizeX * y + x);
238
239 for (unsigned i = 0, e = parallel_regions[index].size(); i != e; ++i) {
240 ParallelRegion *region = parallel_regions[index][i];
241 if (index != 0) {
242 region->remap(reference_map[index - 1]);
243 region->chainAfter(parallel_regions[index - 1][i]);
244#ifdef DEBUG_PR_REPLICATION
245 std::ostringstream s;
246 s << F.getName().str() << ".before_purge_of_" << region->GetID()
247 << ".dot";
248 pocl::dumpCFG
249 (F, s.str(), &parallel_regions[index]);
250#endif
251 region->purge();
252 }
253 region->insertPrologue(x, y, z);
254 }
255 }
256 }
257 }
258
259 // Push the PHI nodes from the entry BBs of the chained
260 // region copys to the entry BB of the first copy to retain
261 // their semantics for branches from outside the parallel
262 // region to the beginning of the region copy chain.
263 for (int z = WGLocalSizeZ - 1; z >= 0; --z) {
264 for (int y = WGLocalSizeY - 1; y >= 0; --y) {
265 for (int x = WGLocalSizeX - 1; x >= 0; --x) {
266
267 int index =
268 (WGLocalSizeY * WGLocalSizeX * z + WGLocalSizeX * y + x);
269
270 if (index == 0)
271 continue;
272
273 for (unsigned i = 0, e = parallel_regions[index].size(); i != e; ++i) {
274 ParallelRegion *firstCopy = parallel_regions[0][i];
275 ParallelRegion *region = parallel_regions[index][i];
276 BasicBlock *entry = region->entryBB();
277#ifdef DEBUG_BB_MERGING
278 std::cerr << "### first entry before hoisting the PHIs " << std::endl;
279 firstCopy->entryBB()->dump();
280#endif
281 movePhiNodes(entry, firstCopy->entryBB());
282#ifdef DEBUG_BB_MERGING
283 std::cerr << "### first entry after hoisting the PHIs " << std::endl;
284 firstCopy->entryBB()->dump();
285#endif
286 }
287 }
288 }
289 }
290
291 // Add the suffixes to original (wi_0_0_0) basic blocks.
292 for (BasicBlockVector::iterator i = original_bbs.begin(),
293 e = original_bbs.end();
294 i != e; ++i)
295 (*i)->setName((*i)->getName() + ".wi_0_0_0");
296
297 // Initialize local size variables (done at the end to avoid unnecessary
298 // replication).
299 K->addLocalSizeInitCode(WGLocalSizeX, WGLocalSizeY, WGLocalSizeZ);
300
301 delete [] reference_map;
302
303 //pocl::dumpCFG(F, F.getName().str() + ".after_repl.dot", original_parallel_regions);
304
305 for (size_t j = 0; j < parallel_regions.size(); ++j) {
306 for (auto i = parallel_regions[j].begin(),
307 e = parallel_regions[j].end();
308 i != e; ++i) {
309 ParallelRegion *p = *i;
310 delete p;
311 }
312 }
313 delete original_parallel_regions;
314 original_parallel_regions = nullptr;
315
316 return true;
317}
