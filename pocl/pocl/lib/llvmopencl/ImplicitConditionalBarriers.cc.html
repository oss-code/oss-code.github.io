69/**
70 * Finds a predecessor that does not come from a back edge.
71 *
72 * This is used to include loops in the conditional parallel region.
73 */
74BasicBlock*
75ImplicitConditionalBarriers::firstNonBackedgePredecessor(
76 llvm::BasicBlock *bb) {
77
78 DominatorTree *DT = &getAnalysis<DominatorTreeWrapperPass>().getDomTree();
79
80 pred_iterator I = pred_begin(bb), E = pred_end(bb);
81 if (I == E) return NULL;
82 while (DT->dominates(bb, *I) && I != E) ++I;
83 if (I == E) return NULL;
84 else return *I;
85}

87bool
88ImplicitConditionalBarriers::runOnFunction(Function &F) {
89{
90 if (!Workgroup::isKernelToProcess(F))
91 return false;
92
93 if (!Workgroup::hasWorkgroupBarriers(F))
94 return false;
95
96 PDT = &getAnalysis<PostDominatorTreeWrapperPass>();
97
98 typedef std::vector<BasicBlock*> BarrierBlockIndex;
99 BarrierBlockIndex conditionalBarriers;
100 for (Function::iterator i = F.begin(), e = F.end(); i != e; ++i) {
101 BasicBlock *b = &*i;
102 if (!Barrier::hasBarrier(b)) continue;
103
104 // Unconditional barrier postdominates the entry node.
105 if (PDT->getPostDomTree().dominates(b, &F.getEntryBlock())) continue;
106
107#ifdef DEBUG_COND_BARRIERS
108 std::cerr << "### found a conditional barrier";
109 b->dump();
110#endif
111 conditionalBarriers.push_back(b);
112 }
113
114 if (conditionalBarriers.size() == 0) return false;
115
116 bool changed = false;
117
118 for (BarrierBlockIndex::const_iterator i = conditionalBarriers.begin();
119 i != conditionalBarriers.end(); ++i) {
120 BasicBlock *b = *i;
121 // Trace upwards from the barrier until one encounters another
122 // barrier or the split point that makes the barrier conditional.
123 // In case of the latter, add a new barrier to both branches of the split point.
124
125 // BB before which to inject the barrier.
126 BasicBlock *pos = b;
127 if (pred_begin(b) == pred_end(b)) {
128#ifdef DEBUG_COND_BARRIERS
129 b->dump();
130#endif
131 assert (pred_begin(b) == pred_end(b));
132 }
133 BasicBlock *pred = firstNonBackedgePredecessor(b);
134
135 while (!Barrier::hasOnlyBarrier(pred) && PDT->getPostDomTree().dominates(b, pred)) {
136
137#ifdef DEBUG_COND_BARRIERS
138 std::cerr << "### looking at BB " << pred->getName().str() << std::endl;
139#endif
140 pos = pred;
141 // If our BB post dominates the given block, we know it is not the
142 // branching block that makes the barrier conditional.
143 pred = firstNonBackedgePredecessor(pred);
144
145 if (pred == b) break; // Traced across a loop edge, skip this case.
146
147 }
148
149 if (Barrier::hasOnlyBarrier(pos)) continue;
150 // Inject a barrier at the beginning of the BB and let the
151 // CanonicalizeBarrier to clean it up (split to a separate BB).
152
153 // mri-q of parboil breaks in case injected at the beginning
154 // TODO: investigate. It might related to the alloca-converted
155 // PHIs. It has a loop that is autoconverted to a b-loop and the
156 // conditional barrier is inserted after the loop short cut check.
157 Barrier::Create(pos->getFirstNonPHI());
158#ifdef DEBUG_COND_BARRIERS
159 std::cerr << "### added an implicit barrier to the BB" << std::endl;
160 pos->dump();
161#endif
162
163 changed = true;
164 }
165
166// F.dump();
167// F.viewCFGOnly();
168
169 return changed;
170}
