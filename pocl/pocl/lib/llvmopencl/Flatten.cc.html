48 class Flatten : public ModulePass {

55 };

59extern cl::opt<std::string> KernelName;
60extern cl_device_id currentPoclDevice;
61
62char Flatten::ID = 0;
63static RegisterPass<Flatten>
64 X("flatten-inline-all",
65 "Kernel function flattening pass - flatten everything");
66
67//#define DEBUG_FLATTEN
68
69bool
70Flatten::runOnModule(Module &M)
71{
72 bool changed = false;
73
74 std::set<std::string> AuxFuncs;
75 if (const char **DevAuxFuncs = currentPoclDevice->device_aux_functions) {
76 const char **Func = DevAuxFuncs;
77 while (*Func != nullptr) {
78 AuxFuncs.insert(*Func++);
79 }
80 }
81
82 for (llvm::Module::iterator i = M.begin(), e = M.end(); i != e; ++i) {
83 llvm::Function *f = &*i;
84 if (f->isDeclaration() || f->getName().startswith("__pocl_print") ||
85 AuxFuncs.find(f->getName().str()) != AuxFuncs.end())
86 continue;
87 if (KernelName == f->getName() && pocl::Workgroup::isKernelToProcess(*f)) {
88 AttributeSet Attrs;
89 f->removeAttributes(AttributeList::FunctionIndex,
90 Attrs.addAttribute(M.getContext(),
91 Attribute::AlwaysInline));
92
93 f->addFnAttr(Attribute::NoInline);
94
95 f->setLinkage(llvm::GlobalValue::ExternalLinkage);
96 changed = true;
97#ifdef DEBUG_FLATTEN
98 std::cerr << "### NoInline for " << f->getName().str() << std::endl;
99#endif
100 } else {
101 AttributeSet Attrs;
102 f->removeAttributes(AttributeList::FunctionIndex,
103 Attrs.addAttribute(M.getContext(),
104 Attribute::NoInline));
105 f->addFnAttr(Attribute::AlwaysInline);
106
107 f->setLinkage(llvm::GlobalValue::InternalLinkage);
108 changed = true;
109#ifdef DEBUG_FLATTEN
110 std::cerr << "### AlwaysInline for " << f->getName().str() << std::endl;
111#endif
112 }
113 }
114 return changed;
115}
