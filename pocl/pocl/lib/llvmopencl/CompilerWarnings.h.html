22#ifndef POCL_COMPILER_WARNINGS_HH
23#define POCL_COMPILER_WARNINGS_HH

25#define DO_PRAGMA(x) _Pragma(#x)
26
27#ifdef __clang__
28
29#define IGNORE_COMPILER_WARNING(X) \
30 DO_PRAGMA(clang diagnostic push) \
31 DO_PRAGMA(clang diagnostic ignored X)
32
33#elif defined(__GNUC__)
34
35#define IGNORE_COMPILER_WARNING(X) \
36 DO_PRAGMA(GCC diagnostic push) \
37 DO_PRAGMA(GCC diagnostic ignored X)
38
39#else
40
41#define IGNORE_COMPILER_WARNING(X)
42
43#endif
44
45
46#ifdef __clang__
47
48#define POP_COMPILER_DIAGS \
49 DO_PRAGMA(clang diagnostic pop)
50
51#elif defined(__GNUC__)
52
53#define POP_COMPILER_DIAGS \
54 DO_PRAGMA(GCC diagnostic pop)
55
56#else
57
58#define POP_COMPILER_DIAGS
59
60#endif

62#endif
