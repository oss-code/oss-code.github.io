56bool
57HandleSamplerInitialization::runOnFunction(Function &F) {
58
59 // Collect the sampler init calls to handle first, do not remove them
60 // instantly as it'd invalidate the iterators.
61 std::set<CallInst*> Calls;
62
63 for (Function::iterator I = F.begin(), E = F.end(); I != E; ++I) {
64 for (BasicBlock::iterator BI = I->begin(), BE = I->end(); BI != BE; ++BI) {
65 Instruction *Instr = dyn_cast<Instruction>(BI);
66 if (!llvm::isa<CallInst>(Instr)) continue;
67 CallInst *CallInstr = dyn_cast<CallInst>(Instr);
68 if (CallInstr->getCalledFunction() != nullptr &&
69 CallInstr->getCalledFunction()->getName() ==
70 "__translate_sampler_initializer") {
71 Calls.insert(CallInstr);
72 }
73 }
74 }
75
76 bool Changed = false;
77 for (auto C : Calls) {
78 llvm::IRBuilder<> Builder(C);
79
80 // get the type of the return value of __translate_sampler
81 // this may not always be opencl.sampler_t, it could be a remapped type.
82 Type *type = C->getCalledValue()->getType();
83 PointerType *pt = dyn_cast<PointerType>(type);
84 FunctionType *ft = dyn_cast<FunctionType>(pt->getPointerElementType());
85 Type *rettype = ft->getReturnType();
86
87 ConstantInt *SamplerValue = dyn_cast<ConstantInt>(C->arg_begin()->get());
88
89 IntegerType *it;
90 if (F.getParent()->getDataLayout().getPointerSizeInBits() == 64)
91 it = Builder.getInt64Ty();
92 else
93 it = Builder.getInt32Ty();
94 // cast the sampler value to intptr_t
95 Constant *sampler = ConstantInt::get(it, SamplerValue->getZExtValue());
96 // then bitcast it to return value (opencl.sampler_t*)
97 Value *val = Builder.CreateBitOrPointerCast(sampler, rettype);
98
99 C->replaceAllUsesWith(val);
100 C->eraseFromParent();
101 Changed = true;
102 }
103
104 return Changed;
105}
