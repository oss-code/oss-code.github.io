61bool
62LoopBarriers::runOnLoop(Loop *L, LPPassManager &LPM) {
63 if (!Workgroup::isKernelToProcess(*L->getHeader()->getParent()))
64 return false;
65
66 if (!Workgroup::hasWorkgroupBarriers(*L->getHeader()->getParent()))
67 return false;
68
69 return ProcessLoop(L, LPM);;
70}

72bool
73LoopBarriers::ProcessLoop(Loop *L, LPPassManager &) {
74 bool isBLoop = false;
75 bool changed = false;
76
77 DT = &getAnalysis<DominatorTreeWrapperPass>().getDomTree();
78
79 for (Loop::block_iterator i = L->block_begin(), e = L->block_end();
80 i != e && !isBLoop; ++i) {
81 for (BasicBlock::iterator j = (*i)->begin(), e = (*i)->end();
82 j != e; ++j) {
83 if (isa<Barrier>(j)) {
84 isBLoop = true;
85 break;
86 }
87 }
88 }
89
90 for (Loop::block_iterator i = L->block_begin(), e = L->block_end();
91 i != e && isBLoop; ++i) {
92 for (BasicBlock::iterator j = (*i)->begin(), e = (*i)->end();
93 j != e; ++j) {
94 if (isa<Barrier>(j)) {
95
96 // Found a barrier in this loop:
97 // 1) add a barrier in the loop header.
98 // 2) add a barrier in the latches
99
100 // Add a barrier on the preheader to ensure all WIs reach
101 // the loop header with all the previous code already
102 // executed.
103 BasicBlock *preheader = L->getLoopPreheader();
104 assert((preheader != NULL) && "Non-canonicalized loop found!\n");
105#ifdef DEBUG_LOOP_BARRIERS
106 std::cerr << "### adding to preheader BB" << std::endl;
107 preheader->dump();
108 std::cerr << "### before instr" << std::endl;
109 preheader->getTerminator()->dump();
110#endif
111 Barrier::Create(preheader->getTerminator());
112 preheader->setName(preheader->getName() + ".loopbarrier");
113
114 // Add a barrier after the PHI nodes on the header (the replicated
115 // headers will be merged afterwards).
116 BasicBlock *header = L->getHeader();
117 if (header->getFirstNonPHI() != &header->front()) {
118 Barrier::Create(header->getFirstNonPHI());
119 header->setName(header->getName() + ".phibarrier");
120 // Split the block to create a replicable region of
121 // the loop contents in case the phi node contains a
122 // branch (that can be to inside the region).
123 // if (header->getTerminator()->getNumSuccessors() > 1)
124 // SplitBlock(header, header->getTerminator(), this);
125 }
126
127 // Add the barriers on the exiting block and the latches,
128 // which might not always be the same if there is computation
129 // after the exit decision.
130 BasicBlock *brexit = L->getExitingBlock();
131 if (brexit != NULL) {
132 Barrier::Create(brexit->getTerminator());
133 brexit->setName(brexit->getName() + ".brexitbarrier");
134 }
135
136 BasicBlock *latch = L->getLoopLatch();
137 if (latch != NULL && brexit != latch) {
138 // This loop has only one latch. Do not check for dominance, we
139 // are probably running before BTR.
140 Barrier::Create(latch->getTerminator());
141 latch->setName(latch->getName() + ".latchbarrier");
142 return changed;
143 }
144
145 // Modified code from llvm::LoopBase::getLoopLatch to
146 // go trough all the latches.
147 BasicBlock *Header = L->getHeader();
148 typedef GraphTraits<Inverse<BasicBlock *> > InvBlockTraits;
149 InvBlockTraits::ChildIteratorType PI =
150 InvBlockTraits::child_begin(Header);
151 InvBlockTraits::ChildIteratorType PE =
152 InvBlockTraits::child_end(Header);
153
154 BasicBlock *Latch = NULL;
155 for (; PI != PE; ++PI) {
156 BasicBlock *N = *PI;
157 if (L->contains(N)) {
158 Latch = N;
159 // Latch found in the loop, see if the barrier dominates it
160 // (otherwise if might no even belong to this "tail", see
161 // forifbarrier1 graph test).
162 if (DT->dominates(j->getParent(), Latch)) {
163 Barrier::Create(Latch->getTerminator());
164 Latch->setName(Latch->getName() + ".latchbarrier");
165 }
166 }
167 }
168 return true;
169 }
170 }
171 }
172
173 /* This is a loop without a barrier. Ensure we have a non-barrier
174 block as a preheader so we can replicate the loop as a whole.
175
176 If the block has proper instructions after the barrier, it
177 will be split in CanonicalizeBarriers. */
178 BasicBlock *preheader = L->getLoopPreheader();
179 assert((preheader != NULL) && "Non-canonicalized loop found!\n");
180
181 Instruction *t = preheader->getTerminator();
182 Instruction *prev = NULL;
183 if (&preheader->front() != t)
184 prev = t->getPrevNode();
185 if (prev && isa<Barrier>(prev)) {
186 BasicBlock *new_b = SplitBlock(preheader, t);
187 new_b->setName(preheader->getName() + ".postbarrier_dummy");
188 return true;
189 }
190
191 return changed;
192}
