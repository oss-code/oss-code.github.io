26#include "llvm/ADT/SmallVector.h"

32namespace pocl {
33
34 class Kernel : public llvm::Function {
35 public:
36 void getExitBlocks(llvm::SmallVectorImpl<llvm::BasicBlock *> &B);
37 ParallelRegion *createParallelRegionBefore(llvm::BasicBlock *B);
38
39 ParallelRegion::ParallelRegionVector*
40 getParallelRegions(llvm::LoopInfo *LI);
41
42 void addLocalSizeInitCode(size_t LocalSizeX, size_t LocalSizeY, size_t LocalSizeZ);
43
44 static bool isKernel(const llvm::Function &F);
45
46 static bool classof(const Kernel *) { return true; }
47 // We assume any function can be a kernel. This could be used
48 // to check for metadata (but would need to be overrideable somehow
49 // to honor the forced kernel name(s) parameter in command line.
50 static bool classof(const llvm::Function *) { return true; }
51 };

53}
