91bool
92WorkitemLoops::runOnFunction(Function &F)
93{
94 if (!Workgroup::isKernelToProcess(F))
95 return false;
96
97 if (getAnalysis<pocl::WorkitemHandlerChooser>().chosenHandler() !=
98 pocl::WorkitemHandlerChooser::POCL_WIH_LOOPS)
99 return false;
100
101 DTP = &getAnalysis<DominatorTreeWrapperPass>();
102 DT = &DTP->getDomTree();
103 LI = &getAnalysis<LoopInfoWrapperPass>();
104
105 PDT = &getAnalysis<PostDominatorTreeWrapperPass>();
106
107 tempInstructionIndex = 0;
108
109// F.viewCFGOnly();
110
111 bool changed = ProcessFunction(F);
112
113#ifdef DUMP_CFGS
114 dumpCFG(F, F.getName().str() + "_after_wiloops.dot",
115 original_parallel_regions);
116#endif
117
118#if 0
119 std::cerr << "### after:" << std::endl;
120 F.viewCFG();
121#endif
122
123 changed |= fixUndominatedVariableUses(DTP, F);
124
125#if 0
126 /* Split large BBs so we can print the Dot without it crashing. */
127 changed |= chopBBs(F, *this);
128 F.viewCFG();
129#endif
130 contextArrays.clear();
131 tempInstructionIds.clear();
132
133 releaseParallelRegions();
134
135 return changed;
136}

138std::pair<llvm::BasicBlock *, llvm::BasicBlock *>
139WorkitemLoops::CreateLoopAround
140(ParallelRegion &region,
141 llvm::BasicBlock *entryBB, llvm::BasicBlock *exitBB,
142 bool peeledFirst, llvm::Value *localIdVar, size_t LocalSizeForDim,
143 bool addIncBlock, llvm::Value *DynamicLocalSize)
144{
145 assert (localIdVar != NULL);
146
147 /*
148
149 Generate a structure like this for each loop level (x,y,z):
150
151 for.init:
152
153 ; if peeledFirst is false:
154 store i32 0, i32* %_local_id_x, align 4
155
156 ; if peeledFirst is true (assume the 0,0,0 iteration has been executed earlier)
157 ; assume _local_id_x_first is is initialized to 1 in the peeled pregion copy
158 store _local_id_x_first, i32* %_local_id_x, align 4
159 store i32 0, %_local_id_x_first
160
161 br label %for.body
162
163 for.body:
164
165 ; the parallel region code here
166
167 br label %for.inc
168
169 for.inc:
170
171 ; Separated inc and cond check blocks for easier loop unrolling later on.
172 ; Can then chain N times for.body+for.inc to unroll.
173
174 %2 = load i32* %_local_id_x, align 4
175 %inc = add nsw i32 %2, 1
176
177 store i32 %inc, i32* %_local_id_x, align 4
178 br label %for.cond
179
180 for.cond:
181
182 ; loop header, compare the id to the local size
183 %0 = load i32* %_local_id_x, align 4
184 %cmp = icmp ult i32 %0, i32 123
185 br i1 %cmp, label %for.body, label %for.end
186
187 for.end:
188
189 OPTIMIZE: Use a separate iteration variable across all the loops to iterate the context
190 data arrays to avoid needing multiplications to find the correct location, and to
191 enable easy vectorization of loading the context data when there are parallel iterations.
192 */
193
194 llvm::BasicBlock *loopBodyEntryBB = entryBB;
195 llvm::LLVMContext &C = loopBodyEntryBB->getContext();
196 llvm::Function *F = loopBodyEntryBB->getParent();
197 loopBodyEntryBB->setName(std::string("pregion_for_entry.") + entryBB->getName().str());
198
199 assert (exitBB->getTerminator()->getNumSuccessors() == 1);
200
201 llvm::BasicBlock *oldExit = exitBB->getTerminator()->getSuccessor(0);
202
203 llvm::BasicBlock *forInitBB =
204 BasicBlock::Create(C, "pregion_for_init", F, loopBodyEntryBB);
205
206 llvm::BasicBlock *loopEndBB =
207 BasicBlock::Create(C, "pregion_for_end", F, exitBB);
208
209 llvm::BasicBlock *forCondBB =
210 BasicBlock::Create(C, "pregion_for_cond", F, exitBB);
211
212
213 DTP->runOnFunction(*F);
214
215 // F->viewCFG();
216 /* Fix the old edges jumping to the region to jump to the basic block
217 that starts the created loop. Back edges should still point to the
218 old basic block so we preserve the old loops. */
219 BasicBlockVector preds;
220 llvm::pred_iterator PI =
221 llvm::pred_begin(entryBB),
222 E = llvm::pred_end(entryBB);
223
224 for (; PI != E; ++PI)
225 {
226 llvm::BasicBlock *bb = *PI;
227 preds.push_back(bb);
228 }
229
230 for (BasicBlockVector::iterator i = preds.begin();
231 i != preds.end(); ++i)
232 {
233 llvm::BasicBlock *bb = *i;
234 /* Do not fix loop edges inside the region. The loop
235 is replicated as a whole to the body of the wi-loop.*/
236 if (DT->dominates(loopBodyEntryBB, bb))
237 continue;
238 bb->getTerminator()->replaceUsesOfWith(loopBodyEntryBB, forInitBB);
239 }
240
241 IRBuilder<> builder(forInitBB);
242
243 if (peeledFirst) {
244 builder.CreateStore(builder.CreateLoad(localIdXFirstVar), localIdVar);
245 builder.CreateStore(ConstantInt::get(SizeT, 0), localIdXFirstVar);
246
247 if (WGDynamicLocalSize) {
248 llvm::Value *cmpResult;
249 cmpResult = builder.CreateICmpULT(builder.CreateLoad(localIdVar),
250 builder.CreateLoad(DynamicLocalSize));
251
252 builder.CreateCondBr(cmpResult, loopBodyEntryBB, loopEndBB);
253 } else {
254 builder.CreateBr(loopBodyEntryBB);
255 }
256 } else {
257 builder.CreateStore(ConstantInt::get(SizeT, 0), localIdVar);
258
259 builder.CreateBr(loopBodyEntryBB);
260 }
261
262 exitBB->getTerminator()->replaceUsesOfWith(oldExit, forCondBB);
263 if (addIncBlock)
264 {
265 AppendIncBlock(exitBB, localIdVar);
266 }
267
268 builder.SetInsertPoint(forCondBB);
269
270 llvm::Value *cmpResult;
271 if (!WGDynamicLocalSize)
272 cmpResult = builder.CreateICmpULT(
273 builder.CreateLoad(localIdVar),
274 ConstantInt::get(SizeT, LocalSizeForDim));
275 else
276 cmpResult = builder.CreateICmpULT(
277 builder.CreateLoad(localIdVar),
278 builder.CreateLoad(DynamicLocalSize));
279
280 Instruction *loopBranch =
281 builder.CreateCondBr(cmpResult, loopBodyEntryBB, loopEndBB);
282
283 /* Add the metadata to mark a parallel loop. The metadata
284 refer to a loop-unique dummy metadata that is not merged
285 automatically. */
286
287 /* This creation of the identifier metadata is copied from
288 LLVM's MDBuilder::createAnonymousTBAARoot(). */
289
290 MDNode *Dummy = MDNode::getTemporary(C, ArrayRef<Metadata*>()).release();
291#ifdef LLVM_OLDER_THAN_8_0
292 MDNode *Root = MDNode::get(C, Dummy);
293#else
294 MDNode *AccessGroupMD = MDNode::getDistinct(C, {});
295 MDNode *ParallelAccessMD = MDNode::get(
296 C, {MDString::get(C, "llvm.loop.parallel_accesses"), AccessGroupMD});
297
298 MDNode *Root = MDNode::get(C, {Dummy, ParallelAccessMD});
299#endif
300
301 // At this point we have
302 // !0 = metadata !{} <- dummy
303 // !1 = metadata !{metadata !0} <- root
304 // Replace the dummy operand with the root node itself and delete the dummy.
305 Root->replaceOperandWith(0, Root);
306 MDNode::deleteTemporary(Dummy);
307 // We now have
308 // !1 = metadata !{metadata !1} <- self-referential root
309 loopBranch->setMetadata("llvm.loop", Root);
310
311#ifdef LLVM_OLDER_THAN_8_0
312 region.AddParallelLoopMetadata(Root);
313#else
314 region.AddParallelLoopMetadata(AccessGroupMD);
315#endif
316
317 builder.SetInsertPoint(loopEndBB);
318 builder.CreateBr(oldExit);
319
320 return std::make_pair(forInitBB, loopEndBB);
321}

323ParallelRegion*
324WorkitemLoops::RegionOfBlock(llvm::BasicBlock *bb)
325{
326 for (ParallelRegion::ParallelRegionVector::iterator
327 i = original_parallel_regions->begin(),
328 e = original_parallel_regions->end();
329 i != e; ++i)
330 {
331 ParallelRegion *region = (*i);
332 if (region->HasBlock(bb)) return region;
333 }
334 return NULL;
335}

338 if (original_parallel_regions) {
339 for (auto i = original_parallel_regions->begin(),
340 e = original_parallel_regions->end();
341 i != e; ++i) {

351bool
352WorkitemLoops::ProcessFunction(Function &F)
353{
354 Kernel *K = cast<Kernel> (&F);
355
356 llvm::Module *M = K->getParent();
357
358 Initialize(K);
359 unsigned workItemCount = WGLocalSizeX*WGLocalSizeY*WGLocalSizeZ;
360
361 if (workItemCount == 1)
362 {
363 K->addLocalSizeInitCode(WGLocalSizeX, WGLocalSizeY, WGLocalSizeZ);
364 ParallelRegion::insertLocalIdInit(&F.getEntryBlock(), 0, 0, 0);
365 return true;
366 }
367
368 releaseParallelRegions();
369
370 original_parallel_regions = K->getParallelRegions(&LI->getLoopInfo());
371
372#ifdef DUMP_CFGS
373 F.dump();
374 dumpCFG(F, F.getName().str() + "_before_wiloops.dot",
375 original_parallel_regions);
376#endif
377
378 IRBuilder<> builder(&*(F.getEntryBlock().getFirstInsertionPt()));
379 localIdXFirstVar = builder.CreateAlloca(SizeT, 0, ".pocl.local_id_x_init");
380
381 // F.viewCFGOnly();
382
383#if 0
384 std::cerr << "### Original" << std::endl;
385 F.viewCFGOnly();
386#endif
387
388#if 0
389 for (ParallelRegion::ParallelRegionVector::iterator
390 i = original_parallel_regions->begin(),
391 e = original_parallel_regions->end();
392 i != e; ++i)
393 {
394 ParallelRegion *region = (*i);
395 region->InjectRegionPrintF();
396 region->InjectVariablePrintouts();
397 }
398#endif
399
400 /* Count how many parallel regions share each entry node to
401 detect diverging regions that need to be peeled. */
402 std::map<llvm::BasicBlock*, int> entryCounts;
403
404 for (ParallelRegion::ParallelRegionVector::iterator
405 i = original_parallel_regions->begin(),
406 e = original_parallel_regions->end();
407 i != e; ++i)
408 {
409 ParallelRegion *region = (*i);
410#ifdef DEBUG_WORK_ITEM_LOOPS
411 std::cerr << "### Adding context save/restore for PR: ";
412 region->dumpNames();
413#endif
414 FixMultiRegionVariables(region);
415 entryCounts[region->entryBB()]++;
416 }
417
418#if 0
419 std::cerr << "### After context code addition:" << std::endl;
420 F.viewCFG();
421#endif
422 std::map<ParallelRegion*, bool> peeledRegion;
423 for (ParallelRegion::ParallelRegionVector::iterator
424 i = original_parallel_regions->begin(),
425 e = original_parallel_regions->end();
426 i != e; ++i)
427 {
428
429 llvm::ValueToValueMapTy reference_map;
430 ParallelRegion *original = (*i);
431
432#ifdef DEBUG_WORK_ITEM_LOOPS
433 std::cerr << "### handling region:" << std::endl;
434 original->dumpNames();
435 //F.viewCFGOnly();
436#endif
437
438 /* In case of conditional barriers, the first iteration
439 has to be peeled so we know which branch to execute
440 with the work item loop. In case there are more than one
441 parallel region sharing an entry BB, it's a diverging
442 region.
443
444 Post dominance of entry by exit does not work in case the
445 region is inside a loop and the exit block is in the path
446 towards the loop exit (and the function exit).
447 */
448 bool peelFirst = entryCounts[original->entryBB()] > 1;
449
450 peeledRegion[original] = peelFirst;
451
452 std::pair<llvm::BasicBlock *, llvm::BasicBlock *> l;
453 // the original predecessor nodes of which successor
454 // should be fixed if not peeling
455 BasicBlockVector preds;
456
457 bool unrolled = false;
458 if (peelFirst)
459 {
460#ifdef DEBUG_WORK_ITEM_LOOPS
461 std::cerr << "### conditional region, peeling the first iteration" << std::endl;
462#endif
463 ParallelRegion *replica =
464 original->replicate(reference_map, ".peeled_wi");
465 replica->chainAfter(original);
466 replica->purge();
467
468 l = std::make_pair(replica->entryBB(), replica->exitBB());
469 }
470 else
471 {
472 llvm::pred_iterator PI =
473 llvm::pred_begin(original->entryBB()),
474 E = llvm::pred_end(original->entryBB());
475
476 for (; PI != E; ++PI)
477 {
478 llvm::BasicBlock *bb = *PI;
479 if (DT->dominates(original->entryBB(), bb) &&
480 (RegionOfBlock(original->entryBB()) ==
481 RegionOfBlock(bb)))
482 continue;
483 preds.push_back(bb);
484 }
485
486 unsigned unrollCount;
487 if (getenv("POCL_WILOOPS_MAX_UNROLL_COUNT") != NULL)
488 unrollCount = atoi(getenv("POCL_WILOOPS_MAX_UNROLL_COUNT"));
489 else
490 unrollCount = 1;
491 /* Find a two's exponent unroll count, if available. */
492 while (unrollCount >= 1)
493 {
494 if (WGLocalSizeX % unrollCount == 0 &&
495 unrollCount <= WGLocalSizeX)
496 {
497 break;
498 }
499 unrollCount /= 2;
500 }
501
502 if (unrollCount > 1) {
503 ParallelRegion *prev = original;
504 llvm::BasicBlock *lastBB =
505 AppendIncBlock(original->exitBB(), LocalIdXGlobal);
506 original->AddBlockAfter(lastBB, original->exitBB());
507 original->SetExitBB(lastBB);
508
509 if (AddWIMetadata)
510 original->AddIDMetadata(F.getContext(), 0);
511
512 for (unsigned c = 1; c < unrollCount; ++c)
513 {
514 ParallelRegion *unrolled =
515 original->replicate(reference_map, ".unrolled_wi");
516 unrolled->chainAfter(prev);
517 prev = unrolled;
518 lastBB = unrolled->exitBB();
519 if (AddWIMetadata)
520 unrolled->AddIDMetadata(F.getContext(), c);
521 }
522 unrolled = true;
523 l = std::make_pair(original->entryBB(), lastBB);
524 } else {
525 l = std::make_pair(original->entryBB(), original->exitBB());
526 }
527 }
528
529 if (WGDynamicLocalSize) {
530 GlobalVariable *gv;
531 gv = M->getGlobalVariable("_local_size_x");
532 if (gv == NULL)
533 gv = new GlobalVariable(*M, SizeT, true, GlobalValue::CommonLinkage,
534 NULL, "_local_size_x", NULL,
535 GlobalValue::ThreadLocalMode::NotThreadLocal,
536 0, true);
537
538 l = CreateLoopAround(*original, l.first, l.second, peelFirst,
539 LocalIdXGlobal, WGLocalSizeX, !unrolled, gv);
540
541 gv = M->getGlobalVariable("_local_size_y");
542 if (gv == NULL)
543 gv = new GlobalVariable(*M, SizeT, false, GlobalValue::CommonLinkage,
544 NULL, "_local_size_y");
545
546 l = CreateLoopAround(*original, l.first, l.second,
547 false, LocalIdYGlobal, WGLocalSizeY, !unrolled, gv);
548
549 gv = M->getGlobalVariable("_local_size_z");
550 if (gv == NULL)
551 gv = new GlobalVariable(*M, SizeT, true, GlobalValue::CommonLinkage,
552 NULL, "_local_size_z", NULL,
553 GlobalValue::ThreadLocalMode::NotThreadLocal,
554 0, true);
555
556 l = CreateLoopAround(*original, l.first, l.second,
557 false, LocalIdZGlobal, WGLocalSizeZ, !unrolled, gv);
558
559 } else {
560 if (WGLocalSizeX > 1) {
561 l = CreateLoopAround(*original, l.first, l.second, peelFirst,
562 LocalIdXGlobal, WGLocalSizeX, !unrolled);
563 }
564
565 if (WGLocalSizeY > 1) {
566 l = CreateLoopAround(*original, l.first, l.second, false,
567 LocalIdYGlobal, WGLocalSizeY);
568 }
569
570 if (WGLocalSizeZ > 1) {
571 l = CreateLoopAround(*original, l.first, l.second, false,
572 LocalIdZGlobal, WGLocalSizeZ);
573 }
574 }
575
576 /* Loop edges coming from another region mean B-loops which means
577 we have to fix the loop edge to jump to the beginning of the wi-loop
578 structure, not its body. This has to be done only for non-peeled
579 blocks as the semantics is correct in the other case (the jump is
580 to the beginning of the peeled iteration). */
581 if (!peelFirst)
582 {
583 for (BasicBlockVector::iterator i = preds.begin();
584 i != preds.end(); ++i)
585 {
586 llvm::BasicBlock *bb = *i;
587 bb->getTerminator()->replaceUsesOfWith
588 (original->entryBB(), l.first);
589 }
590 }
591 }
592
593 // for the peeled regions we need to add a prologue
594 // that initializes the local ids and the first iteration
595 // counter
596 for (ParallelRegion::ParallelRegionVector::iterator
597 i = original_parallel_regions->begin(),
598 e = original_parallel_regions->end();
599 i != e; ++i)
600 {
601 ParallelRegion *pr = (*i);
602
603 if (!peeledRegion[pr]) continue;
604 pr->insertPrologue(0, 0, 0);
605 builder.SetInsertPoint(&*(pr->entryBB()->getFirstInsertionPt()));
606 builder.CreateStore(ConstantInt::get(SizeT, 1), localIdXFirstVar);
607 }
608
609 if (!WGDynamicLocalSize)
610 K->addLocalSizeInitCode(WGLocalSizeX, WGLocalSizeY, WGLocalSizeZ);
611
612 ParallelRegion::insertLocalIdInit(&F.getEntryBlock(), 0, 0, 0);
613
614#if 0
615 F.viewCFG();
616#endif
617
618 return true;
619}

621/*
622 * Add context save/restore code to variables that are defined in
623 * the given region and are used outside the region.
624 *
625 * Each such variable gets a slot in the stack frame. The variable
626 * is restored from the stack whenever it's used.
627 *
628 */
629void
630WorkitemLoops::FixMultiRegionVariables(ParallelRegion *region)
631{
632 InstructionIndex instructionsInRegion;
633 InstructionVec instructionsToFix;
634
635 /* Construct an index of the region's instructions so it's
636 fast to figure out if the variable uses are all
637 in the region. */
638 for (BasicBlockVector::iterator i = region->begin();
639 i != region->end(); ++i)
640 {
641 llvm::BasicBlock *bb = *i;
642 for (llvm::BasicBlock::iterator instr = bb->begin();
643 instr != bb->end(); ++instr)
644 {
645 llvm::Instruction *instruction = &*instr;
646 instructionsInRegion.insert(instruction);
647 }
648 }
649
650 /* Find all the instructions that define new values and
651 check if they need to be context saved. */
652 for (BasicBlockVector::iterator i = region->begin();
653 i != region->end(); ++i)
654 {
655 llvm::BasicBlock *bb = *i;
656 for (llvm::BasicBlock::iterator instr = bb->begin();
657 instr != bb->end(); ++instr)
658 {
659 llvm::Instruction *instruction = &*instr;
660
661 if (ShouldNotBeContextSaved(&*instr)) continue;
662
663 for (Instruction::use_iterator ui = instruction->use_begin(),
664 ue = instruction->use_end();
665 ui != ue; ++ui)
666 {
667 llvm::Instruction *user = dyn_cast<Instruction>(ui->getUser());
668
669 if (user == NULL) continue;
670 // If the instruction is used outside this region inside another
671 // region (not in a regionless BB like the B-loop construct BBs),
672 // need to context save it.
673 // Allocas (private arrays) should be privatized always. Otherwise
674 // we end up reading the same array, but replicating the GEP to that.
675 if (isa<AllocaInst>(instruction) ||
676 (instructionsInRegion.find(user) == instructionsInRegion.end() &&
677 RegionOfBlock(user->getParent()) != NULL))
678 {
679 instructionsToFix.push_back(instruction);
680 break;
681 }
682 }
683 }
684 }
685
686 /* Finally, fix the instructions. */
687 for (InstructionVec::iterator i = instructionsToFix.begin();
688 i != instructionsToFix.end(); ++i)
689 {
690#ifdef DEBUG_WORK_ITEM_LOOPS
691 std::cerr << "### adding context/save restore for" << std::endl;
692 (*i)->dump();
693#endif
694 llvm::Instruction *instructionToFix = *i;
695 AddContextSaveRestore(instructionToFix);
696 }
697}

699llvm::Value *
700WorkitemLoops::GetLinearWiIndex(llvm::IRBuilder<> &builder, llvm::Module *M,
701 ParallelRegion *region)
702{
703 GlobalVariable *LocalSizeXPtr =
704 cast<GlobalVariable>(M->getOrInsertGlobal("_local_size_x", SizeT));
705 GlobalVariable *LocalSizeYPtr =
706 cast<GlobalVariable>(M->getOrInsertGlobal("_local_size_y", SizeT));
707
708 assert(LocalSizeXPtr != NULL && LocalSizeYPtr != NULL);
709
710 LoadInst* LoadX = builder.CreateLoad(LocalSizeXPtr, "ls_x");
711 LoadInst* LoadY = builder.CreateLoad(LocalSizeYPtr, "ls_y");
712
713 /* Form linear index from xyz coordinates:
714 local_size_x * local_size_y * local_id_z (z dimension)
715 + local_size_x * local_id_y (y dimension)
716 + local_id_x (x dimension)
717 */
718 Value* LocalSizeXTimesY =
719 builder.CreateBinOp(Instruction::Mul, LoadX, LoadY, "ls_xy");
720
721 Value* ZPart =
722 builder.CreateBinOp(Instruction::Mul, LocalSizeXTimesY,
723 region->LocalIDZLoad(),
724 "tmp");
725
726 Value* YPart =
727 builder.CreateBinOp(Instruction::Mul, LoadX, region->LocalIDYLoad(),
728 "ls_x_y");
729
730 Value* ZYSum =
731 builder.CreateBinOp(Instruction::Add, ZPart, YPart,
732 "zy_sum");
733
734 return builder.CreateBinOp(Instruction::Add, ZYSum, region->LocalIDXLoad(),
735 "linear_xyz_idx");
736}
737
738llvm::Instruction *
739WorkitemLoops::AddContextSave
740(llvm::Instruction *instruction, llvm::Instruction *alloca)
741{
742
743 if (isa<AllocaInst>(instruction))
744 {
745 /* If the variable to be context saved is itself an alloca,
746 we have created one big alloca that stores the data of all the
747 work-items and return pointers to that array. Thus, we need
748 no initialization code other than the context data alloca itself. */
749 return NULL;
750 }
751
752 /* Save the produced variable to the array. */
753 BasicBlock::iterator definition = (dyn_cast<Instruction>(instruction))->getIterator();
754 ++definition;
755 while (isa<PHINode>(definition)) ++definition;
756
757 IRBuilder<> builder(&*definition);
758 std::vector<llvm::Value *> gepArgs;
759
760 /* Reuse the id loads earlier in the region, if possible, to
761 avoid messy output with lots of redundant loads. */
762 ParallelRegion *region = RegionOfBlock(instruction->getParent());
763 assert ("Adding context save outside any region produces illegal code." &&
764 region != NULL);
765
766 if (WGDynamicLocalSize)
767 {
768 Module *M = alloca->getParent()->getParent()->getParent();
769 gepArgs.push_back(GetLinearWiIndex(builder, M, region));
770 }
771 else
772 {
773 gepArgs.push_back(ConstantInt::get(SizeT, 0));
774 gepArgs.push_back(region->LocalIDZLoad());
775 gepArgs.push_back(region->LocalIDYLoad());
776 gepArgs.push_back(region->LocalIDXLoad());
777 }
778
779 return builder.CreateStore(instruction, builder.CreateGEP(alloca,
780 gepArgs));
781}

783llvm::Instruction *WorkitemLoops::AddContextRestore(llvm::Value *val,
784 llvm::Instruction *alloca,
785 bool PoclWrapperStructAdded,
786 llvm::Instruction *before,
787 bool isAlloca) {
788 assert (val != NULL);
789 assert (alloca != NULL);
790 IRBuilder<> builder(alloca);
791 if (before != NULL)
792 {
793 builder.SetInsertPoint(before);
794 }
795 else if (isa<Instruction>(val))
796 {
797 builder.SetInsertPoint(dyn_cast<Instruction>(val));
798 before = dyn_cast<Instruction>(val);
799 }
800 else
801 {
802 assert (false && "Unknown context restore location!");
803 }
804
805 std::vector<llvm::Value *> gepArgs;
806
807 /* Reuse the id loads earlier in the region, if possible, to
808 avoid messy output with lots of redundant loads. */
809 ParallelRegion *region = RegionOfBlock(before->getParent());
810 assert ("Adding context save outside any region produces illegal code." &&
811 region != NULL);
812
813 if (WGDynamicLocalSize)
814 {
815 Module *M = alloca->getParent()->getParent()->getParent();
816 gepArgs.push_back(GetLinearWiIndex(builder, M, region));
817 }
818 else
819 {
820 gepArgs.push_back(ConstantInt::get(SizeT, 0));
821 gepArgs.push_back(region->LocalIDZLoad());
822 gepArgs.push_back(region->LocalIDYLoad());
823 gepArgs.push_back(region->LocalIDXLoad());
824 }
825
826 if (PoclWrapperStructAdded)
827 gepArgs.push_back(
828 ConstantInt::get(Type::getInt32Ty(alloca->getContext()), 0));
829
830 llvm::Instruction *gep =
831 dyn_cast<Instruction>(builder.CreateGEP(alloca, gepArgs));
832 if (isAlloca) {
833 /* In case the context saved instruction was an alloca, we created a
834 context array with pointed-to elements, and now want to return a
835 pointer to the elements to emulate the original alloca. */
836 return gep;
837 }
838 return builder.CreateLoad(gep);
839}

841/**
842 * Returns the context array (alloca) for the given Value, creates it if not
843 * found.
844 */
845llvm::Instruction *
846WorkitemLoops::GetContextArray(llvm::Instruction *instruction,
847 bool &PoclWrapperStructAdded) {
848 PoclWrapperStructAdded = false;
849 /*
850 * Unnamed temp instructions need a generated name for the
851 * context array. Create one using a running integer.
852 */
853 std::ostringstream var;
854 var << ".";
855
856 if (std::string(instruction->getName().str()) != "")
857 {
858 var << instruction->getName().str();
859 }
860 else if (tempInstructionIds.find(instruction) != tempInstructionIds.end())
861 {
862 var << tempInstructionIds[instruction];
863 }
864 else
865 {
866 tempInstructionIds[instruction] = tempInstructionIndex++;
867 var << tempInstructionIds[instruction];
868 }
869
870 var << ".pocl_context";
871 std::string varName = var.str();
872
873 if (contextArrays.find(varName) != contextArrays.end())
874 return contextArrays[varName];
875
876 BasicBlock &bb = instruction->getParent()->getParent()->getEntryBlock();
877 IRBuilder<> builder(&*(bb.getFirstInsertionPt()));
878
879 llvm::Type *elementType;
880 if (isa<AllocaInst>(instruction))
881 {
882 /* If the variable to be context saved was itself an alloca,
883 create one big alloca that stores the data of all the
884 work-items and directly return pointers to that array.
885 This enables moving all the allocas to the entry node without
886 breaking the parallel loop.
887 Otherwise we would rely on a dynamic alloca to allocate
888 unique stack space to all the work-items when its wiloop
889 iteration is executed. */
890 elementType =
891 dyn_cast<AllocaInst>(instruction)->getType()->getElementType();
892 }
893 else
894 {
895 elementType = instruction->getType();
896 }
897
898 Module* M = instruction->getParent()->getParent()->getParent();
899 const llvm::DataLayout &Layout = M->getDataLayout();
900
901 /* 3D context array. In case the elementType itself is an array or struct,
902 * we must take into account it could be alloca-ed with alignment and loads
903 * or stores might use vectorized instructions expecting proper alignment.
904 * Because of that, we cannot simply allocate x*y*z*(size), we must
905 * enlarge the type to fit the alignment. */
906 Type *AllocType = elementType;
907 AllocaInst *InstCast = dyn_cast<AllocaInst>(instruction);
908 if (InstCast) {
909 unsigned Alignment = InstCast->getAlignment();
910
911 uint64_t StoreSize =
912 Layout.getTypeStoreSize(InstCast->getAllocatedType());
913
914 if ((Alignment > 1) && (StoreSize & (Alignment - 1))) {
915 uint64_t AlignedSize = (StoreSize & (~(Alignment - 1))) + Alignment;
916#ifdef DEBUG_WORK_ITEM_LOOPS
917 std::cerr << "### unaligned type found: aligning " << StoreSize << " to "
918 << AlignedSize << "\n";
919#endif
920 assert(AlignedSize > StoreSize);
921 uint64_t RequiredExtraBytes = AlignedSize - StoreSize;
922
923 if (isa<ArrayType>(elementType)) {
924
925 ArrayType *StructPadding = ArrayType::get(
926 Type::getInt8Ty(M->getContext()), RequiredExtraBytes);
927
928 std::vector<Type *> PaddedStructElements;
929 PaddedStructElements.push_back(elementType);
930 PaddedStructElements.push_back(StructPadding);
931 const ArrayRef<Type *> NewStructElements(PaddedStructElements);
932 AllocType = StructType::get(M->getContext(), NewStructElements, true);
933 PoclWrapperStructAdded = true;
934 uint64_t NewStoreSize = Layout.getTypeStoreSize(AllocType);
935 assert(NewStoreSize == AlignedSize);
936
937 } else if (isa<StructType>(elementType)) {
938 StructType *OldStruct = dyn_cast<StructType>(elementType);
939
940 ArrayType *StructPadding =
941 ArrayType::get(Type::getInt8Ty(M->getContext()), RequiredExtraBytes);
942 std::vector<Type *> PaddedStructElements;
943 for (unsigned j = 0; j < OldStruct->getNumElements(); j++)
944 PaddedStructElements.push_back(OldStruct->getElementType(j));
945 PaddedStructElements.push_back(StructPadding);
946 const ArrayRef<Type *> NewStructElements(PaddedStructElements);
947 AllocType = StructType::get(OldStruct->getContext(), NewStructElements,
948 OldStruct->isPacked());
949 uint64_t NewStoreSize = Layout.getTypeStoreSize(AllocType);
950 assert(NewStoreSize == AlignedSize);
951 }
952 }
953 }
954
955 llvm::AllocaInst *Alloca = nullptr;
956 if (WGDynamicLocalSize)
957 {
958 char GlobalName[32];
959 GlobalVariable* LocalSize;
960 LoadInst* LocalSizeLoad[3];
961 for (int i = 0; i < 3; ++i) {
962 snprintf(GlobalName, 32, "_local_size_%c", 'x' + i);
963 LocalSize =
964 cast<GlobalVariable>(M->getOrInsertGlobal(GlobalName, SizeT));
965 LocalSizeLoad[i] = builder.CreateLoad(LocalSize);
966 }
967
968 Value* LocalXTimesY =
969 builder.CreateBinOp(Instruction::Mul, LocalSizeLoad[0],
970 LocalSizeLoad[1], "tmp");
971 Value* NumberOfWorkItems =
972 builder.CreateBinOp(Instruction::Mul, LocalXTimesY,
973 LocalSizeLoad[2], "num_wi");
974
975 Alloca = builder.CreateAlloca(AllocType, NumberOfWorkItems, varName);
976 }
977 else
978 {
979 llvm::Type *contextArrayType = ArrayType::get(
980 ArrayType::get(ArrayType::get(AllocType, WGLocalSizeX), WGLocalSizeY),
981 WGLocalSizeZ);
982
983 /* Allocate the context data array for the variable. */
984 Alloca = builder.CreateAlloca(contextArrayType, nullptr, varName);
985 }
986
987 /* Align the context arrays to stack to enable wide vectors
988 accesses to them. Also, LLVM 3.3 seems to produce illegal
989 code at least with Core i5 when aligned only at the element
990 size. */
991 Alloca->setAlignment(CONTEXT_ARRAY_ALIGN);
992
993 contextArrays[varName] = Alloca;
994 return Alloca;
995}

997/**
998 * Adds context save/restore code for the value produced by the
999 * given instruction.
1000 *
1001 * TODO: add only one restore per variable per region.
1002 * TODO: add only one load of the id variables per region.
1003 * Could be done by having a context restore BB in the beginning of the
1004 * region and a context save BB at the end.
1005 * TODO: ignore work group variables completely (the iteration variables)
1006 * The LLVM should optimize these away but it would improve
1007 * the readability of the output during debugging.
1008 * TODO: rematerialize some values such as extended values of global
1009 * variables (especially global id which is computed from local id) or kernel
1010 * argument values instead of allocating stack space for them
1011 */
1012void
1013WorkitemLoops::AddContextSaveRestore
1014(llvm::Instruction *instruction) {
1015
1016 /* Allocate the context data array for the variable. */
1017 bool PoclWrapperStructAdded = false;
1018 llvm::Instruction *alloca =
1019 GetContextArray(instruction, PoclWrapperStructAdded);
1020 llvm::Instruction *theStore = AddContextSave(instruction, alloca);
1021
1022 InstructionVec uses;
1023 /* Restore the produced variable before each use to ensure the correct context
1024 copy is used.
1025
1026 We could add the restore only to other regions outside the
1027 variable defining region and use the original variable in the defining
1028 region due to the SSA virtual registers being unique. However,
1029 alloca variables can be redefined also in the same region, thus we
1030 need to ensure the correct alloca context position is written, not
1031 the original unreplicated one. These variables can be generated by
1032 volatile variables, private arrays, and due to the PHIs to allocas
1033 pass.
1034 */
1035
1036 /* Find out the uses to fix first as fixing them invalidates
1037 the iterator. */
1038 for (Instruction::use_iterator ui = instruction->use_begin(),
1039 ue = instruction->use_end();
1040 ui != ue; ++ui)
1041 {
1042 llvm::Instruction *user = cast<Instruction>(ui->getUser());
1043 if (user == NULL) continue;
1044 if (user == theStore) continue;
1045 uses.push_back(user);
1046 }
1047
1048 for (InstructionVec::iterator i = uses.begin(); i != uses.end(); ++i)
1049 {
1050 Instruction *user = *i;
1051 Instruction *contextRestoreLocation = user;
1052 /* If the user is in a block that doesn't belong to a region,
1053 the variable itself must be a "work group variable", that is,
1054 not dependent on the work item. Most likely an iteration
1055 variable of a for loop with a barrier. */
1056 if (RegionOfBlock(user->getParent()) == NULL) continue;
1057
1058 PHINode* phi = dyn_cast<PHINode>(user);
1059 if (phi != NULL)
1060 {
1061 /* In case of PHI nodes, we cannot just insert the context
1062 restore code before it in the same basic block because it is
1063 assumed there are no non-phi Instructions before PHIs which
1064 the context restore code constitutes to. Add the context
1065 restore to the incomingBB instead.
1066
1067 There can be values in the PHINode that are incoming
1068 from another region even though the decision BB is within the region.
1069 For those values we need to add the context restore code in the
1070 incoming BB (which is known to be inside the region due to the
1071 assumption of not having to touch PHI nodes in PRentry BBs).
1072 */
1073
1074 /* PHINodes at region entries are broken down earlier. */
1075 assert ("Cannot add context restore for a PHI node at the region entry!" &&
1076 RegionOfBlock(phi->getParent())->entryBB() != phi->getParent());
1077#ifdef DEBUG_WORK_ITEM_LOOPS
1078 std::cerr << "### adding context restore code before PHI" << std::endl;
1079 user->dump();
1080 std::cerr << "### in BB:" << std::endl;
1081 user->getParent()->dump();
1082#endif
1083 BasicBlock *incomingBB = NULL;
1084 for (unsigned incoming = 0; incoming < phi->getNumIncomingValues();
1085 ++incoming)
1086 {
1087 Value *val = phi->getIncomingValue(incoming);
1088 BasicBlock *bb = phi->getIncomingBlock(incoming);
1089 if (val == instruction) incomingBB = bb;
1090 }
1091 assert (incomingBB != NULL);
1092 contextRestoreLocation = incomingBB->getTerminator();
1093 }
1094 llvm::Value *loadedValue = AddContextRestore(
1095 user, alloca, PoclWrapperStructAdded, contextRestoreLocation,
1096 isa<AllocaInst>(instruction));
1097 user->replaceUsesOfWith(instruction, loadedValue);
1098
1099#ifdef DEBUG_WORK_ITEM_LOOPS
1100 std::cerr << "### done, the user was converted to:" << std::endl;
1101 user->dump();
1102#endif
1103 }
1104}

1106bool
1107WorkitemLoops::ShouldNotBeContextSaved(llvm::Instruction *instr)
1108{
1109 /*
1110 _local_id loads should not be replicated as it leads to
1111 problems in conditional branch case where the header node
1112 of the region is shared across the branches and thus the
1113 header node's ID loads might get context saved which leads
1114 to egg-chicken problems.
1115 */
1116 if (isa<BranchInst>(instr)) return true;
1117
1118 llvm::LoadInst *load = dyn_cast<llvm::LoadInst>(instr);
1119 if (load != NULL &&
1120 (load->getPointerOperand() == LocalIdZGlobal ||
1121 load->getPointerOperand() == LocalIdYGlobal ||
1122 load->getPointerOperand() == LocalIdXGlobal))
1123 return true;
1124
1125 VariableUniformityAnalysis &VUA =
1126 getAnalysis<VariableUniformityAnalysis>();
1127
1128 /* In case of uniform variables (same for all work-items),
1129 there is no point to create a context array slot for them,
1130 but just use the original value everywhere.
1131
1132 Allocas are problematic: they include the de-phi induction
1133 variables of the b-loops. In those case each work item
1134 has a separate loop iteration variable in the LLVM IR but
1135 which is really a parallel region loop invariant. But
1136 because we cannot separate such loop invariant variables
1137 at this point sensibly, let's just replicate the iteration
1138 variable to each work item and hope the latter optimizations
1139 reduce them back to a single induction variable outside the
1140 parallel loop.
1141 */
1142 if (!VUA.shouldBePrivatized(instr->getParent()->getParent(), instr)) {
1143#ifdef DEBUG_WORK_ITEM_LOOPS
1144 std::cerr << "### based on VUA, not context saving:";
1145 instr->dump();
1146#endif
1147 return true;
1148 }
1149
1150 return false;
1151}

1153llvm::BasicBlock *
1154WorkitemLoops::AppendIncBlock
1155(llvm::BasicBlock* after, llvm::Value *localIdVar)
1156{
1157 llvm::LLVMContext &C = after->getContext();
1158
1159 llvm::BasicBlock *oldExit = after->getTerminator()->getSuccessor(0);
1160 assert (oldExit != NULL);
1161
1162 llvm::BasicBlock *forIncBB =
1163 BasicBlock::Create(C, "pregion_for_inc", after->getParent());
1164
1165 after->getTerminator()->replaceUsesOfWith(oldExit, forIncBB);
1166
1167 IRBuilder<> builder(oldExit);
1168
1169 builder.SetInsertPoint(forIncBB);
1170 /* Create the iteration variable increment */
1171 builder.CreateStore(builder.CreateAdd(
1172 builder.CreateLoad(localIdVar),
1173 ConstantInt::get(SizeT, 1)),
1174 localIdVar);
1175
1176 builder.CreateBr(oldExit);
1177
1178 return forIncBB;
1179}
