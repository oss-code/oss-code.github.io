54/// WorkItemAliasAnalysis - This is a simple alias analysis
55/// implementation that uses pocl metadata to make sure memory accesses from
56/// different work items are not aliasing.
57///
58
59// LLVM 3.8+
60
61class WorkItemAAResult : public AAResultBase<WorkItemAAResult> {
62 friend AAResultBase<WorkItemAAResult>;
63
64public:
65 static char ID;
66
67 WorkItemAAResult(const TargetLibraryInfo &)
68 : AAResultBase() {}
69 WorkItemAAResult(const WorkItemAAResult &)
70 : AAResultBase() {}
71 WorkItemAAResult(WorkItemAAResult &&)
72 : AAResultBase() {}
73
74 AliasResult alias(const MemoryLocation &LocA, const MemoryLocation &LocB);
75};

119bool WorkItemAliasAnalysis::runOnFunction(llvm::Function &) {
120 auto &TLIWP = getAnalysis<TargetLibraryInfoWrapperPass>();
121 Result.reset(new WorkItemAAResult(TLIWP.getTLI()));
122 return false;
123}

138/**
139 * Test if memory locations are from different work items from same region.
140 * Then they can not alias.
141 */
142
143AliasResult
144WorkItemAAResult::alias(const Location &LocA, const Location &LocB) {
145 // If either of the memory references is empty, it doesn't matter what the
146 // pointer values are. This allows the code below to ignore this special
147 // case.
148 if (LocA.Size == 0 || LocB.Size == 0)
149 return NoAlias;
150
151 // Pointers from different address spaces do not alias
152 if (cast<PointerType>(LocA.Ptr->getType())->getAddressSpace() !=
153 cast<PointerType>(LocB.Ptr->getType())->getAddressSpace()) {
154 return NoAlias;
155 }
156 // In case code is created by pocl, we can also use metadata.
157 if (isa<Instruction>(LocA.Ptr) && isa<Instruction>(LocB.Ptr)) {
158 const Instruction* valA = dyn_cast<Instruction>(LocA.Ptr);
159 const Instruction* valB = dyn_cast<Instruction>(LocB.Ptr);
160 if (valA->getMetadata("wi") && valB->getMetadata("wi")) {
161 const MDNode* mdA = valA->getMetadata("wi");
162 const MDNode* mdB = valB->getMetadata("wi");
163 // Compare region ID. If they are same, different work items
164 // imply no aliasing. If regions are different or work items
165 // are same anything can happen.
166 // Fall back to other AAs.
167 const MDNode* mdRegionA = dyn_cast<MDNode>(mdA->getOperand(1));
168 const MDNode* mdRegionB = dyn_cast<MDNode>(mdB->getOperand(1));
169 ConstantInt* C1 = dyn_cast<ConstantInt>(
170 dyn_cast<ConstantAsMetadata>(mdRegionA->getOperand(1))->getValue());
171 ConstantInt* C2 = dyn_cast<ConstantInt>(
172 dyn_cast<ConstantAsMetadata>(mdRegionB->getOperand(1))->getValue());
173 if (C1->getValue() == C2->getValue()) {
174 // Now we have both locations from same region. Check for different
175 // work items.
176 MDNode* iXYZ= dyn_cast<MDNode>(mdA->getOperand(2));
177 MDNode* jXYZ= dyn_cast<MDNode>(mdB->getOperand(2));
178 assert(iXYZ->getNumOperands() == 4);
179 assert(jXYZ->getNumOperands() == 4);
180
181 ConstantInt *CIX =
182 dyn_cast<ConstantInt>(
183 dyn_cast<ConstantAsMetadata>(
184 iXYZ->getOperand(1))->getValue());
185 ConstantInt *CJX =
186 dyn_cast<ConstantInt>(
187 dyn_cast<ConstantAsMetadata>(
188 jXYZ->getOperand(1))->getValue());
189
190 ConstantInt *CIY =
191 dyn_cast<ConstantInt>(
192 dyn_cast<ConstantAsMetadata>(
193 iXYZ->getOperand(2))->getValue());
194 ConstantInt *CJY =
195 dyn_cast<ConstantInt>(
196 dyn_cast<ConstantAsMetadata>(
197 jXYZ->getOperand(2))->getValue());
198
199 ConstantInt *CIZ =
200 dyn_cast<ConstantInt>(
201 dyn_cast<ConstantAsMetadata>(
202 iXYZ->getOperand(3))->getValue());
203 ConstantInt *CJZ =
204 dyn_cast<ConstantInt>(
205 dyn_cast<ConstantAsMetadata>(
206 jXYZ->getOperand(3))->getValue());
207
208 if ( !(CIX->getValue() == CJX->getValue()
209 && CIY->getValue() == CJY->getValue()
210 && CIZ->getValue() == CJZ->getValue())) {
211 return NoAlias;
212 }
213 }
214 }
215 }
216
217 return WorkItemAAResult::alias(LocA, LocB);
218}
