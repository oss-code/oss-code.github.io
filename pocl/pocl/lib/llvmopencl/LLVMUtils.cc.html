39/**
40 * Regenerates the metadata that points to the original kernel
41 * (of which finger print was modified) to point to the new
42 * kernel.
43 *
44 * Only checks if the first operand of the metadata is the kernel
45 * function.
46 */
47void
48regenerate_kernel_metadata(llvm::Module &M, FunctionMapping &kernels)
49{
50 // reproduce the opencl.kernel_wg_size_info metadata
51 NamedMDNode *wg_sizes = M.getNamedMetadata("opencl.kernel_wg_size_info");
52 if (wg_sizes != NULL && wg_sizes->getNumOperands() > 0)
53 {
54 for (std::size_t mni = 0; mni < wg_sizes->getNumOperands(); ++mni)
55 {
56 MDNode *wgsizeMD = dyn_cast<MDNode>(wg_sizes->getOperand(mni));
57 for (FunctionMapping::const_iterator i = kernels.begin(),
58 e = kernels.end(); i != e; ++i)
59 {
60 Function *old_kernel = (*i).first;
61 Function *new_kernel = (*i).second;
62 Function *func_from_md;
63 func_from_md = dyn_cast<Function>(
64 dyn_cast<ValueAsMetadata>(wgsizeMD->getOperand(0))->getValue());
65 if (old_kernel == new_kernel || wgsizeMD->getNumOperands() == 0 ||
66 func_from_md != old_kernel)
67 continue;
68 // found a wg size metadata that points to the old kernel, copy its
69 // operands except the first one to a new MDNode
70 SmallVector<Metadata*, 8> operands;
71 operands.push_back(llvm::ValueAsMetadata::get(new_kernel));
72 for (unsigned opr = 1; opr < wgsizeMD->getNumOperands(); ++opr) {
73 operands.push_back(wgsizeMD->getOperand(opr));
74 }
75 MDNode *new_wg_md = MDNode::get(M.getContext(), operands);
76 wg_sizes->addOperand(new_wg_md);
77 }
78 }
79 }
80
81 // reproduce the opencl.kernels metadata, if it exists
82 // unconditionally adding opencl.kernels confuses the
83 // metadata parser in pocl_llvm_metadata.cc, which uses
84 // "opencl.kernels" to distinguish old SPIR format from new
85 NamedMDNode *nmd = M.getNamedMetadata("opencl.kernels");
86 if (nmd) {
87 M.eraseNamedMetadata(nmd);
88
89 nmd = M.getOrInsertNamedMetadata("opencl.kernels");
90 for (FunctionMapping::const_iterator i = kernels.begin(),
91 e = kernels.end();
92 i != e; ++i) {
93 MDNode *md = MDNode::get(M.getContext(), ArrayRef<Metadata *>(
94 llvm::ValueAsMetadata::get((*i).second)));
95 nmd->addOperand(md);
96 }
97 }
98
99}

101void eraseFunctionAndCallers(llvm::Function *Function) {
102 if (!Function)
103 return;
104
105 std::vector<llvm::Value *> Callers(Function->user_begin(),
106 Function->user_end());
107 for (auto &U : Callers) {
108 llvm::CallInst *Call = llvm::dyn_cast<llvm::CallInst>(U);
109 if (!Call)
110 continue;
111 Call->eraseFromParent();
112 }
113 Function->eraseFromParent();
114}

116int getConstantIntMDValue(Metadata *MD) {
117#ifdef LLVM_OLDER_THAN_7_0
118 ConstantInt *CI = mdconst::dyn_extract<ConstantInt>(MD);
119#else
120 ConstantInt *CI = mdconst::extract<ConstantInt>(MD);
121#endif
122 return CI->getLimitedValue();
123}
124
125llvm::Metadata *createConstantIntMD(llvm::LLVMContext &C, int32_t Val) {
126 IntegerType *I32Type = IntegerType::get(C, 32);
127 return ConstantAsMetadata::get(ConstantInt::get(I32Type, Val));
128}

130bool isLocalMemFunctionArg(llvm::Function *F, unsigned ArgIndex) {
131
132 MDNode *MD = F->getMetadata("kernel_arg_addr_space");
133
134 if (MD == nullptr || MD->getNumOperands() <= ArgIndex)
135 return false;
136 else
137 return getConstantIntMDValue(MD->getOperand(ArgIndex)) ==
138 SPIR_ADDRESS_SPACE_LOCAL;
139}
140
141void setFuncArgAddressSpaceMD(llvm::Function *F, unsigned ArgIndex,
142 unsigned AS) {
143
144 unsigned MDKind = F->getContext().getMDKindID("kernel_arg_addr_space");
145 MDNode *OldMD = F->getMetadata(MDKind);
146
147 assert(OldMD == nullptr || OldMD->getNumOperands() >= ArgIndex);
148
149 LLVMContext &C = F->getContext();
150
151 llvm::SmallVector<llvm::Metadata *, 8> AddressQuals;
152 for (unsigned i = 0; i < ArgIndex; ++i) {
153 AddressQuals.push_back(createConstantIntMD(
154 C, OldMD != nullptr ? getConstantIntMDValue(OldMD->getOperand(i))
155 : SPIR_ADDRESS_SPACE_GLOBAL));
156 }
157 AddressQuals.push_back(createConstantIntMD(C, AS));
158 F->setMetadata(MDKind, MDNode::get(F->getContext(), AddressQuals));
159}
