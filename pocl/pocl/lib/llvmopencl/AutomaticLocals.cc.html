75bool
76AutomaticLocals::runOnModule(Module &M)
77{
78 bool changed = false;
79
80 // store the new and old kernel pairs in order to regenerate
81 // all the metadata that used to point to the unmodified
82 // kernels
83 FunctionMapping kernels;
84
85 string ErrorInfo;
86 std::vector<Function*> NewFuncs;
87 for (Module::iterator mi = M.begin(), me = M.end(); mi != me; ++mi) {
88 // This is to prevent recursion with llvm 3.9. The new kernels are
89 // recognized as kernelsToProcess.
90 if (find(NewFuncs.begin(), NewFuncs.end(), &*mi) != NewFuncs.end())
91 continue;
92 if (!Workgroup::isKernelToProcess(*mi))
93 continue;
94
95 Function *F = &*mi;
96
97 Function *new_kernel = processAutomaticLocals(F);
98 if (new_kernel != F)
99 changed = true;
100 kernels[F] = new_kernel;
101 NewFuncs.push_back(new_kernel);
102 }
103
104 if (changed)
105 {
106 regenerate_kernel_metadata(M, kernels);
107 /* Delete the old kernels. */
108 for (FunctionMapping::const_iterator i = kernels.begin(),
109 e = kernels.end(); i != e; ++i)
110 {
111 Function *old_kernel = (*i).first;
112 Function *new_kernel = (*i).second;
113 if (old_kernel == new_kernel) continue;
114 old_kernel->eraseFromParent();
115 }
116 }
117 return changed;
118}

120// Recursively descend a Value's users and convert any constant expressions into
121// regular instructions.
122static void breakConstantExpressions(llvm::Value *Val, llvm::Function *Func) {
123 std::vector<llvm::Value *> Users(Val->user_begin(), Val->user_end());
124 for (auto *U : Users) {
125 if (auto *CE = llvm::dyn_cast<llvm::ConstantExpr>(U)) {
126 // First, make sure no users of this constant expression are themselves
127 // constant expressions.
128 breakConstantExpressions(U, Func);
129
130 // Convert this constant expression to an instruction.
131 llvm::Instruction *I = CE->getAsInstruction();
132 I->insertBefore(&*Func->begin()->begin());
133 CE->replaceAllUsesWith(I);
134 CE->destroyConstant();
135 }
136 }
137}

139Function *
140AutomaticLocals::processAutomaticLocals(Function *F) {
141
142 Module *M = F->getParent();
143
144 SmallVector<GlobalVariable *, 8> Locals;
145
146 SmallVector<Type *, 8> Parameters;
147 for (Function::const_arg_iterator i = F->arg_begin(),
148 e = F->arg_end(); i != e; ++i)
149 Parameters.push_back(i->getType());
150
151 for (Module::global_iterator i = M->global_begin(),
152 e = M->global_end(); i != e; ++i) {
153 std::string FuncName = "";
154 FuncName = F->getName().str();
155 if (isAutomaticLocal(FuncName, *i)) {
156 Locals.push_back(&*i);
157
158 // Add the parameters to the end of the function parameter list.
159 Parameters.push_back(i->getType());
160
161 // Replace any constant expression users with an equivalent instruction.
162 // Otherwise, the IR breaks when we replace the local with an argument.
163 breakConstantExpressions(&*i, F);
164 }
165 }
166
167 if (Locals.empty()) {
168 // This kernel fingerprint has not changed.
169 return F;
170 }
171
172 // Create the new function.
173 FunctionType *FT =
174 FunctionType::get(F->getReturnType(), Parameters, F->isVarArg());
175 Function *NewKernel = Function::Create(FT, F->getLinkage(), "", M);
176 NewKernel->takeName(F);
177
178 ValueToValueMapTy VV;
179 Function::arg_iterator j = NewKernel->arg_begin();
180 for (Function::const_arg_iterator i = F->arg_begin(), e = F->arg_end();
181 i != e; ++i) {
182 j->setName(i->getName());
183 VV[&*i] = &*j;
184 ++j;
185 }
186
187 for (int i = 0; j != NewKernel->arg_end(); ++i, ++j) {
188 j->setName("_local" + Twine(i));
189 VV[Locals[i]] = &*j;
190 }
191
192 SmallVector<ReturnInst *, 1> RI;
193
194 // As of LLVM 5.0 we need to let CFI to make module level changes,
195 // otherwise there will be an assertion. The changes are likely
196 // additional debug info nodes added when cloning the function into
197 // the other. For some reason it doesn't want to reuse the old ones.
198 CloneFunctionInto(NewKernel, F, VV, true, RI);
199
200 for (size_t i = 0; i < Locals.size(); ++i) {
201 setFuncArgAddressSpaceMD(NewKernel, F->arg_size() + i,
202 SPIR_ADDRESS_SPACE_LOCAL);
203 }
204 return NewKernel;
205}
