61bool
62CanonicalizeBarriers::runOnFunction(Function &F)
63{
64 if (!Workgroup::isKernelToProcess(F))
65 return false;
66
67 BasicBlock *entry = &F.getEntryBlock();
68 if (!Barrier::hasOnlyBarrier(entry)) {
69 BasicBlock *effective_entry = SplitBlock(entry,
70 &(entry->front()));
71
72 effective_entry->takeName(entry);
73 entry->setName("entry.barrier");
74 Barrier::Create(entry->getTerminator());
75 }
76
77 for (Function::iterator i = F.begin(), e = F.end(); i != e; ++i) {
78
79 BasicBlock *b = &*i;
80 auto t = b->getTerminator();
81
82 const bool isExitNode =
83 (t->getNumSuccessors() == 0) && (!Barrier::hasOnlyBarrier(b));
84
85 // The function exits should have barriers.
86 if (isExitNode && !Barrier::hasOnlyBarrier(b)) {
87 /* In case the bb is already terminated with a barrier,
88 split before the barrier so we don't create an empty
89 parallel region.
90
91 This is because the assumptions of the other passes in the
92 compilation that are
93 a) exit node is a barrier block
94 b) there are no empty parallel regions (which would be formed
95 between the explicit barrier and the added one). */
96 BasicBlock *exit;
97 if (Barrier::endsWithBarrier(b))
98 exit = SplitBlock(b, t->getPrevNode());
99 else
100 exit = SplitBlock(b, t);
101 exit->setName("exit.barrier");
102 Barrier::Create(t);
103 }
104 }
105
106 DT = &getAnalysis<DominatorTreeWrapperPass>().getDomTree();
107 return ProcessFunction(F);
108}

111// Canonicalize barriers: ensure all barriers are in a separate BB
112// containing only the barrier and the terminator, with just one
113// predecessor. This allows us to use those BBs as markers only,
114// they will not be replicated.
115bool
116CanonicalizeBarriers::ProcessFunction(Function &F) {
117
118 bool changed = false;
119
120 InstructionSet Barriers;
121
122 for (Function::iterator i = F.begin(), e = F.end();
123 i != e; ++i) {
124 BasicBlock *b = &*i;
125 for (BasicBlock::iterator i = b->begin(), e = b->end();
126 i != e; ++i) {
127 if (isa<Barrier>(i)) {
128 Barriers.insert(&*i);
129 }
130 }
131 }
132
133 // Finally add all the split points, now that we are done with the
134 // iterators.
135 for (InstructionSet::iterator i = Barriers.begin(), e = Barriers.end();
136 i != e; ++i) {
137 BasicBlock *b = (*i)->getParent();
138
139 // Split post barrier first cause it does not make the barrier
140 // to belong to another basic block.
141 Instruction *t = b->getTerminator();
142 // if ((t->getNumSuccessors() > 1) ||
143 // (t->getPrevNode() != *i)) {
144 // Change: barriers with several successors are all right
145 // they just start several parallel regions. Simplifies
146 // loop handling.
147
148 const bool HAS_NON_BRANCH_INSTRUCTIONS_AFTER_BARRIER =
149 t->getPrevNode() != *i;
150
151 if (HAS_NON_BRANCH_INSTRUCTIONS_AFTER_BARRIER) {
152 BasicBlock *new_b = SplitBlock(b, (*i)->getNextNode());
153 new_b->setName(b->getName() + ".postbarrier");
154 changed = true;
155 }
156
157 BasicBlock *predecessor = b->getSinglePredecessor();
158 if (predecessor != NULL) {
159 auto pt = predecessor->getTerminator();
160 if ((pt->getNumSuccessors() == 1) &&
161 (&b->front() == (*i))) {
162 // Barrier is at the beginning of the BB,
163 // which has a single predecessor with just
164 // one successor (the barrier itself), thus
165 // no need to split before barrier.
166 continue;
167 }
168 }
169 if ((b == &(b->getParent()->getEntryBlock())) &&
170 (&b->front() == (*i)))
171 continue;
172
173 // If no instructions before barrier, do not split
174 // (allow multiple predecessors, eases loop handling).
175 // if (&b->front() == (*i))
176 // continue;
177 BasicBlock *new_b = SplitBlock(b, *i);
178 new_b->takeName(b);
179 b->setName(new_b->getName() + ".prebarrier");
180 changed = true;
181 }
182
183 // Prune empty regions. That is, if there are two successive
184 // pure barrier blocks without side branches, remove the other one.
185 bool emptyRegionDeleted = false;
186 do {
187 emptyRegionDeleted = false;
188 for (Function::iterator i = F.begin(), e = F.end();
189 i != e; ++i) {
190 BasicBlock *b = &*i;
191 auto t = b->getTerminator();
192 if (!Barrier::endsWithBarrier(b) || t->getNumSuccessors() != 1)
193 continue;
194
195 BasicBlock *successor = t->getSuccessor(0);
196
197 if (Barrier::hasOnlyBarrier(successor) &&
198 successor->getSinglePredecessor() == b) {
199 b->replaceAllUsesWith(successor);
200 b->eraseFromParent();
201 emptyRegionDeleted = true;
202 changed = true;
203 break;
204 }
205 }
206 } while (emptyRegionDeleted);
207
208
209 return changed;
210}
