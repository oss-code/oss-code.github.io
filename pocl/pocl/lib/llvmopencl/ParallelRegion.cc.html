69/**
70 * Ensure all variables are named so they will be replicated and renamed
71 * correctly.
72 */
73void
74ParallelRegion::GenerateTempNames(llvm::BasicBlock *bb)
75{
76 for (llvm::BasicBlock::iterator i = bb->begin(), e = bb->end(); i != e; ++i)
77 {
78 llvm::Instruction *instr = &*i;
79 if (instr->hasName() || !instr->isUsedOutsideOfBlock(bb)) continue;
80 int tempCounter = 0;
81 std::string tempName = "";
82 do {
83 std::ostringstream name;
84 name << ".pocl_temp." << tempCounter;
85 ++tempCounter;
86 tempName = name.str();
87 } while (bb->getParent()->getValueSymbolTable()->lookup(tempName) != NULL);
88 instr->setName(tempName);
89 }
90}

101ParallelRegion *
102ParallelRegion::replicate(ValueToValueMapTy &map,
103 const Twine &suffix = "")
104{
105 ParallelRegion *new_region = new ParallelRegion(pRegionId);
106
107 /* Because ParallelRegions are all replicated before they
108 are attached to the function, it can happen that
109 the same BB is replicated multiple times and it gets
110 the same name (only the BB name will be autorenamed
111 by LLVM). This causes the variable references to become
112 broken. This hack ensures the BB suffixes are unique
113 before cloning so each path gets their own value
114 names. Split points can be such paths.*/
115 static std::map<std::string, int> cloneCounts;
116
117 for (iterator i = begin(), e = end(); i != e; ++i) {
118 BasicBlock *block = *i;
119 GenerateTempNames(block);
120 std::ostringstream suf;
121 suf << suffix.str();
122 std::string block_name = block->getName().str() + "." + suffix.str();
123 if (cloneCounts[block_name] > 0)
124 {
125 suf << ".pocl_" << cloneCounts[block_name];
126 }
127 BasicBlock *new_block = CloneBasicBlock(block, map, suf.str());
128 cloneCounts[block_name]++;
129 // Insert the block itself into the map.
130 map[block] = new_block;
131 new_region->push_back(new_block);
132
133#ifdef DEBUG_REPLICATE
134 std::cerr << "### clonee block:" << std::endl;
135 block->dump();
136 std::cerr << endl << "### cloned block: " << std::endl;
137 new_block->dump();
138#endif
139 }
140
141 new_region->exitIndex_ = exitIndex_;
142 new_region->entryIndex_ = entryIndex_;
143 /* Remap here to get local variables fixed before they
144 are (possibly) overwritten by another clone of the
145 same BB. */
146 new_region->remap(map);
147
148#ifdef DEBUG_REPLICATE
149 Verify();
150#endif
151 LocalizeIDLoads();
152
153 return new_region;
154}

156void
157ParallelRegion::remap(ValueToValueMapTy &map)
158{
159 for (iterator i = begin(), e = end(); i != e; ++i) {
160
161#ifdef DEBUG_REMAP
162 std::cerr << "### block before remap:" << std::endl;
163 (*i)->dump();
164#endif
165
166 for (BasicBlock::iterator ii = (*i)->begin(), ee = (*i)->end();
167 ii != ee; ++ii)
168 RemapInstruction(&*ii, map,
169 RF_IgnoreMissingLocals | RF_NoModuleLevelChanges);
170
171#ifdef DEBUG_REMAP
172 std::cerr << endl << "### block after remap: " << std::endl;
173 (*i)->dump();
174#endif
175 }

178void
179ParallelRegion::chainAfter(ParallelRegion *region)
180{
181 /* If we are replicating a conditional barrier region, the last block can be
182 an unreachable block to mark the impossible path. Skip it and choose the
183 correct branch instead.
184
185 TODO: why have the unreachable block there the first place? Could we just
186 not add it and fix the branch? */
187 BasicBlock *tail = region->exitBB();
188 auto t = tail->getTerminator();
189 if (isa<UnreachableInst>(t))
190 {
191 tail = region->at(region->size() - 2);
192 t = tail->getTerminator();
193 }
194#ifdef LLVM_BUILD_MODE_DEBUG
195 if (t->getNumSuccessors() != 1) {
196 std::cout << "!!! trying to chain region" << std::endl;
197 this->dumpNames();
198 std::cout << "!!! after region" << std::endl;
199 region->dumpNames();
200 t->getParent()->dump();
201
202 assert (t->getNumSuccessors() == 1);
203 }
204#endif
205
206 BasicBlock *successor = t->getSuccessor(0);
207 Function::BasicBlockListType &bb_list =
208 successor->getParent()->getBasicBlockList();
209
210 for (iterator i = begin(), e = end(); i != e; ++i)
211
212 bb_list.insertAfter(tail->getIterator(), *i);
213 t->setSuccessor(0, entryBB());
214
215 t = exitBB()->getTerminator();
216 assert (t->getNumSuccessors() == 1);
217 t->setSuccessor(0, successor);
218}

220/**
221 * Removes known dead side exits from parallel regions.
222 *
223 * These occur with conditional barriers. The head of the path
224 * leading to the conditional barrier is shared by two PRs. The
225 * first work-item defines which path is taken (by definition the
226 * barrier is taken by all or none of the work-items). The blocks
227 * in the branches are in different regions which can contain branches
228 * to blocks that are in known non-taken path. This method replaces
229 * the targets of such branches with undefined BBs so they will be cleaned
230 * up by the optimizer.
231 */
232void
233ParallelRegion::purge()
234{
235 SmallVector<BasicBlock *, 4> new_blocks;
236
237 // Go through all the BBs in the region and check their branch
238 // targets, looking for destinations that are outside the region.
239 // Only the last block in the PR can now contain such branches.
240 for (iterator i = begin(), e = end(); i != e; ++i) {
241
242 // Exit block has a successor out of the region.
243 if (*i == exitBB())
244 continue;
245
246#ifdef DEBUG_PURGE
247 std::cerr << "### block before purge:" << std::endl;
248 (*i)->dump();
249#endif
250 auto t = (*i)->getTerminator();
251 for (unsigned ii = 0, ee = t->getNumSuccessors(); ii != ee; ++ii) {
252 BasicBlock *successor = t->getSuccessor(ii);
253 if (count(begin(), end(), successor) == 0) {
254 // This successor is not on the parallel region, purge.
255#ifdef DEBUG_PURGE
256 std::cerr
257 << "purging a branch to a block "
258 << successor->getName().str() << " outside the region"
259 << std::endl;
260#endif
261
262 BasicBlock *unreachable =
263 BasicBlock::Create((*i)->getContext(),
264 (*i)->getName() + ".unreachable",
265 (*i)->getParent(), back());
266 new UnreachableInst(unreachable->getContext(),
267 unreachable);
268 t->setSuccessor(ii, unreachable);
269 new_blocks.push_back(unreachable);
270 }
271 }
272#ifdef DEBUG_PURGE
273 std::cerr << std::endl << "### block after purge:" << std::endl;
274 (*i)->dump();
275#endif
276 }
277
278 // Add the new "unreachable" blocks to the
279 // region. We cannot do in the loop as it
280 // corrupts iterators.
281 insert(end(), new_blocks.begin(), new_blocks.end());
282}

284void
285ParallelRegion::insertLocalIdInit(llvm::BasicBlock* Entry,
286 unsigned X, unsigned Y, unsigned Z) {
287
288 IRBuilder<> Builder(Entry, Entry->getFirstInsertionPt());
289
290 Module *M = Entry->getParent()->getParent();
291
292 llvm::Type *SizeT =
293 IntegerType::get(M->getContext(), currentPoclDevice->address_bits);
294
295 GlobalVariable *GVX = M->getGlobalVariable(POCL_LOCAL_ID_X_GLOBAL);
296 if (GVX != NULL)
297 Builder.CreateStore(ConstantInt::get(SizeT, X), GVX);
298
299 GlobalVariable *GVY = M->getGlobalVariable(POCL_LOCAL_ID_Y_GLOBAL);
300 if (GVY != NULL)
301 Builder.CreateStore(ConstantInt::get(SizeT, Y), GVY);
302
303 GlobalVariable *GVZ = M->getGlobalVariable(POCL_LOCAL_ID_Z_GLOBAL);
304 if (GVZ != NULL)
305 Builder.CreateStore(ConstantInt::get(SizeT, Z), GVZ);
306}

341ParallelRegion *
342ParallelRegion::Create(const SmallPtrSet<BasicBlock *, 8>& bbs, BasicBlock *entry, BasicBlock *exit)
343{
344 ParallelRegion *new_region = new ParallelRegion();
345
346 assert (entry != NULL);
347 assert (exit != NULL);
348
349 // This is done in two steps so order of the vector
350 // is the same as original function order.
351 Function *F = entry->getParent();
352 for (Function::iterator i = F->begin(), e = F->end(); i != e; ++i) {
353 BasicBlock *b = &*i;
354 for (SmallPtrSetIterator<BasicBlock *> j = bbs.begin(); j != bbs.end(); ++j) {
355 if (*j == b) {
356 new_region->push_back(&*i);
357 if (entry == *j)
358 new_region->setEntryBBIndex(new_region->size() - 1);
359 else if (exit == *j)
360 new_region->setExitBBIndex(new_region->size() - 1);
361 break;
362 }
363 }
364 }
365
366 new_region->LocalizeIDLoads();
367
368 assert(new_region->Verify());
369
370 return new_region;
371}

373bool
374ParallelRegion::Verify()
375{
376 // Parallel region conditions:
377 // 1) Single entry, in entry block.
378 // 2) Single outgoing edge from exit block
379 // (other outgoing edges allowed, will be purged in replicas).
380 // 3) No barriers inside the region.
381
382 int entry_edges = 0;
383
384 for (iterator i = begin(), e = end(); i != e; ++i) {
385 for (pred_iterator ii(*i), ee(*i, true); ii != ee; ++ii) {
386 if (count(begin(), end(), *ii) == 0) {
387 if ((*i) != entryBB()) {
388 dumpNames();
389 std::cerr << "suspicious block: " << (*i)->getName().str() << std::endl;
390 std::cerr << "the entry is: " << entryBB()->getName().str() << std::endl;
391
392 ParallelRegion::ParallelRegionVector prvec;
393 prvec.push_back(this);
394 std::set<llvm::BasicBlock*> highlights;
395 highlights.insert(entryBB());
396 highlights.insert(*i);
397 pocl::dumpCFG(
398 *(*i)->getParent(), (*i)->getParent()->getName().str() + ".dot",
399 &prvec, &highlights);
400 assert(0 && "Incoming edges to non-entry block!");
401 return false;
402 } else if (!Barrier::hasBarrier(*ii)) {
403 (*i)->getParent()->viewCFG();
404 assert (0 && "Entry has edges from non-barrier blocks!");
405 return false;
406 }
407 ++entry_edges;
408 }
409 }
410
411 // if (entry_edges != 1) {
412 // assert(0 && "Parallel regions must be single entry!");
413 // return false;
414 // }
415 if (exitBB()->getTerminator()->getNumSuccessors() != 1) {
416 ParallelRegion::ParallelRegionVector regions;
417 regions.push_back(this);
418
419#ifdef LLVM_BUILD_MODE_DEBUG
420 std::set<llvm::BasicBlock*> highlights;
421 highlights.insert((*i));
422 highlights.insert(exitBB());
423 exitBB()->dump();
424 dumpNames();
425 dumpCFG(*(*i)->getParent(), "broken.dot", &regions, &highlights);
426#endif
427
428 assert(0 && "Multiple outgoing edges from exit block!");
429 return false;
430 }
431
432 for (BasicBlock::iterator ii = (*i)->begin(), ee = (*i)->end();
433 ii != ee; ++ii) {
434 if (isa<Barrier> (ii)) {
435 assert(0 && "Barrier found inside parallel region!");
436 return false;
437 }
438 }
439 }
440
441 return true;
442}

444/**
445 * Adds metadata to all the memory instructions to denote
446 * they originate from a parallel loop.
447 *
448 * Due to nested parallel loops, there can be multiple loop
449 * references.
450 *
451 * Format:
452 * llvm.mem.parallel_loop_access !0
453 *
454 * !0 { metadata !0 }
455 *
456 * In a 2-nested loop:
457 *
458 * llvm.mem.parallel_loop_access !0
459 *
460 * !0 { metadata !1, metadata !2}
461 * !1 { metadata !1 }
462 * !2 { metadata !2 }
463 */
464
465#ifdef LLVM_OLDER_THAN_8_0
466#define PARALLEL_MD_NAME "llvm.mem.parallel_loop_access"
467#else
468#define PARALLEL_MD_NAME "llvm.access.group"
469#endif
470
471void ParallelRegion::AddParallelLoopMetadata(llvm::MDNode *Identifier) {
472 for (iterator i = begin(), e = end(); i != e; ++i) {
473 BasicBlock* bb = *i;
474 for (BasicBlock::iterator ii = bb->begin(), ee = bb->end();
475 ii != ee; ii++) {
476
477 if (ii->mayReadOrWriteMemory()) {
478 MDNode *NewMD = MDNode::get(bb->getContext(), Identifier);
479 MDNode *OldMD = ii->getMetadata(PARALLEL_MD_NAME);
480 if (OldMD != nullptr) {
481 NewMD = llvm::MDNode::concatenate(OldMD, NewMD);
482 }
483 ii->setMetadata(PARALLEL_MD_NAME, NewMD);
484 }
485 }
486 }
487}

489void
490ParallelRegion::AddIDMetadata(
491 llvm::LLVMContext& context,
492 std::size_t x,
493 std::size_t y,
494 std::size_t z) {
495 int counter = 1;
496 Metadata *v1[] = {
497 MDString::get(context, "WI_region"),
498 llvm::ConstantAsMetadata::get(
499 ConstantInt::get(Type::getInt32Ty(context), pRegionId))
500 };
501 MDNode* mdRegion = MDNode::get(context, v1);
502 Metadata *v2[] = {
503 MDString::get(context, "WI_xyz"),
504 llvm::ConstantAsMetadata::get(
505 ConstantInt::get(Type::getInt32Ty(context), x)),
506 llvm::ConstantAsMetadata::get(
507 ConstantInt::get(Type::getInt32Ty(context), y)),
508 llvm::ConstantAsMetadata::get(
509 ConstantInt::get(Type::getInt32Ty(context), z))};
510 MDNode* mdXYZ = MDNode::get(context, v2);
511 Metadata *v[] = {
512 MDString::get(context, "WI_data"),
513 mdRegion,
514 mdXYZ};
515 MDNode* md = MDNode::get(context, v);
516
517 for (iterator i = begin(), e = end(); i != e; ++i) {
518 BasicBlock* bb = *i;
519 for (BasicBlock::iterator ii = bb->begin();
520 ii != bb->end(); ii++) {
521 Metadata *v3[] = {
522 MDString::get(context, "WI_counter"),
523 llvm::ConstantAsMetadata::get(
524 ConstantInt::get(Type::getInt32Ty(context), counter))};
525 MDNode* mdCounter = MDNode::get(context, v3);
526 counter++;
527 ii->setMetadata("wi", md);
528 ii->setMetadata("wi_counter", mdCounter);
529 }
530 }

534/**
535 * Inserts a new basic block to the region, before an old basic block in
536 * the region.
537 *
538 * Assumes the inserted block to be before the other block in control
539 * flow, that is, there should be direct CFG edge from the block to the
540 * other.
541 */
542void
543ParallelRegion::AddBlockBefore(llvm::BasicBlock *block, llvm::BasicBlock *before)
544{
545 llvm::BasicBlock *oldExit = exitBB();
546 ParallelRegion::iterator beforePos = find(begin(), end(), before);
547 ParallelRegion::iterator oldExitPos = find(begin(), end(), oldExit);
548 assert (beforePos != end());
549
550 /* The old exit node might is now pushed further, at most one position.
551 Whether this is the case, depends if the node was inserted before or
552 after that node in the vector. That is, if indexof(before) < indexof(oldExit). */
553 if (beforePos < oldExitPos) ++exitIndex_;
554
555 insert(beforePos, block);
556 /* The entryIndex_ should be still correct. In case the 'before' block
557 was an old entry node, the new one replaces it as an entry node at
558 the same index and the old one gets pushed forward. */
559}
560
561
562void
563ParallelRegion::AddBlockAfter(llvm::BasicBlock *block, llvm::BasicBlock *after)
564{
565 llvm::BasicBlock *oldExit = exitBB();
566 ParallelRegion::iterator afterPos = find(begin(), end(), after);
567 ParallelRegion::iterator oldExitPos = find(begin(), end(), oldExit);
568 assert (afterPos != end());
569
570 /* The old exit node might be pushed further, at most one position.
571 Whether this is the case, depends if the node was inserted before or
572 after that node in the vector. That is, if indexof(before) < indexof(oldExit). */
573 if (afterPos < oldExitPos) ++exitIndex_;
574 afterPos++;
575 insert(afterPos, block);
576}

626void
627ParallelRegion::InjectPrintF
628(llvm::Instruction *before, std::string formatStr,
629 std::vector<Value*>& params)
630{
631 IRBuilder<> builder(before);
632 llvm::Module *M = before->getParent()->getParent()->getParent();
633
634 llvm::Value *stringArg =
635 builder.CreateGlobalString(formatStr);
636
637 /* generated with help from http://llvm.org/demo/index.cgi */
638 Function* printfFunc = M->getFunction("printf");
639 if (printfFunc == NULL) {
640 PointerType* PointerTy_4 = PointerType::get(IntegerType::get(M->getContext(), 8), 0);
641
642 std::vector<Type*> FuncTy_6_args;
643 FuncTy_6_args.push_back(PointerTy_4);
644
645 FunctionType* FuncTy_6 =
646 FunctionType::get
647 (/*Result=*/IntegerType::get(M->getContext(), 32),
648 /*Params=*/FuncTy_6_args,
649 /*isVarArg=*/true);
650
651 printfFunc =
652 Function::Create
653 (/*Type=*/FuncTy_6,
654 /*Linkage=*/GlobalValue::ExternalLinkage,
655 /*Name=*/"printf", M);
656 printfFunc->setCallingConv(CallingConv::C);
657
658 AttributeList func_printf_PAL;
659 {
660 func_printf_PAL.addAttribute( M->getContext(), 1U, Attribute::NoCapture);
661 func_printf_PAL.addAttribute( M->getContext(), 4294967295U, Attribute::NoUnwind);
662 }
663 printfFunc->setAttributes(func_printf_PAL);
664 }
665
666 std::vector<Constant*> const_ptr_8_indices;
667
668 ConstantInt* const_int64_9 = ConstantInt::get(M->getContext(), APInt(64, StringRef("0"), 10));
669 const_ptr_8_indices.push_back(const_int64_9);
670 const_ptr_8_indices.push_back(const_int64_9);
671 assert (isa<Constant>(stringArg));
672 Constant* const_ptr_8 =
673 ConstantExpr::getGetElementPtr
674 (PointerType::getUnqual(Type::getInt8Ty(M->getContext())), cast<Constant>(stringArg), const_ptr_8_indices);
675
676 std::vector<Value*> args;
677 args.push_back(const_ptr_8);
678 args.insert(args.end(), params.begin(), params.end());
679
680 CallInst::Create(printfFunc, args, "", before);
681}

730/**
731 * Adds a printf to the end of the parallel region that prints the
732 * hex contents of all named non-pointer variables.
733 *
734 * Useful for debugging data flow bugs.
735 */
736void
737ParallelRegion::InjectVariablePrintouts()
738{
739 for (ParallelRegion::iterator i = begin();
740 i != end(); ++i)
741 {
742 llvm::BasicBlock *bb = *i;
743 for (llvm::BasicBlock::iterator instr = bb->begin();
744 instr != bb->end(); ++instr)
745 {
746 llvm::Instruction *instruction = &*instr;
747 if (isa<PointerType>(instruction->getType()) ||
748 !instruction->hasName()) continue;
749 std::string name = instruction->getName().str();
750 std::vector<Value*> args;
751 IRBuilder<> builder(exitBB()->getTerminator());
752 args.push_back(builder.CreateGlobalString(name));
753 args.push_back(instruction);
754 InjectPrintF(instruction->getParent()->getTerminator(), "variable %s == %x\n", args);
755 }
756 }
757}

759/**
760 * Localizes all the loads to the the work-item identifiers.
761 *
762 * In case the code inside the region queries the WI id, it
763 * should not (re)use one that is loaded in another region, but
764 * one that is loaded in the same region. Otherwise, it ends
765 * up using the last id the previous PR work-item loop got.
766 * This caused problems in cases where the local id was stored
767 * to a temporary variable in an earlier region and that temp
768 * was reused later.
769 *
770 * The function scans for all loads from the local id variables
771 * and converts them to loads inside the parallel region.
772 */
773void
774ParallelRegion::LocalizeIDLoads()
775{
776 /* The local id loads inside the parallel region. */
777 llvm::Instruction* LocalIDXLoadInstr = LocalIDXLoad();
778 llvm::Instruction* LocalIDYLoadInstr = LocalIDYLoad();
779 llvm::Instruction* LocalIDZLoadInstr = LocalIDZLoad();
780 llvm::Module *M = LocalIDXLoadInstr->getParent()->getParent()->getParent();
781 llvm::Value *localIdZ = M->getNamedGlobal(POCL_LOCAL_ID_Z_GLOBAL);
782 llvm::Value *localIdY = M->getNamedGlobal(POCL_LOCAL_ID_Y_GLOBAL);
783 llvm::Value *localIdX = M->getNamedGlobal(POCL_LOCAL_ID_X_GLOBAL);
784
785 assert (localIdZ != NULL && localIdY != NULL && localIdX != NULL &&
786 "The local id globals were not created.");
787
788 for (ParallelRegion::iterator i = begin();
789 i != end(); ++i)
790 {
791 llvm::BasicBlock *bb = *i;
792 for (llvm::BasicBlock::iterator instrI = bb->begin();
793 instrI != bb->end(); ++instrI)
794 {
795 llvm::Instruction *instr = &*instrI;
796 if (instr == LocalIDXLoadInstr ||
797 instr == LocalIDYLoadInstr ||
798 instr == LocalIDZLoadInstr) continue;
799
800 /* Search all operands of the instruction. If any of them is
801 using a local id, replace it with the intra-PR load from the
802 id variable. */
803 for (unsigned opr = 0; opr < instr->getNumOperands(); ++opr)
804 {
805 llvm::LoadInst *load =
806 dyn_cast<llvm::LoadInst>(instr->getOperand(opr));
807 if (load == NULL) continue;
808 if (load == LocalIDXLoadInstr ||
809 load == LocalIDYLoadInstr ||
810 load == LocalIDZLoadInstr) continue;
811
812 if (load->getPointerOperand() == localIdZ)
813 instr->setOperand(opr, LocalIDZLoadInstr);
814 if (load->getPointerOperand() == localIdY)
815 instr->setOperand(opr, LocalIDYLoadInstr);
816 if (load->getPointerOperand() == localIdX)
817 instr->setOperand(opr, LocalIDXLoadInstr);
818 }
819 }
820 }
