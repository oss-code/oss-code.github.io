50bool
51OptimizeWorkItemFuncCalls::runOnFunction(Function &F) {
52
53 // Let's avoid reoptimizing pocl_printf in the kernel compiler. It should
54 // be optimized already in the bitcode library, and we do not want to
55 // aggressively inline it to the kernel, causing compile time expansion.
56 if (F.getName().startswith("__pocl_print") &&
57 !F.hasFnAttribute(Attribute::OptimizeNone)) {
58 F.addFnAttr(Attribute::OptimizeNone);
59 F.addFnAttr(Attribute::NoInline);
60 }
61
62 if (F.getName().startswith("_") || F.hasFnAttribute(Attribute::OptimizeNone))
63 return false;
64
65 // Find calls to WI functions and unify them to a single call in the
66 // entry to avoid confusing LLVM later with the 'pseudo loads' and to
67 // reduce the inlining bloat.
68
69 typedef std::set<std::string> WIFuncNameVec;
70 const WIFuncNameVec WIFuncNames = {
71 "_Z13get_global_idj",
72 "_Z17get_global_offsetj",
73 "_Z15get_global_sizej",
74 "_Z12get_group_idj",
75 "_Z12get_local_idj",
76 "_Z14get_local_sizej",
77 "_Z14get_num_groupsj",
78 "_Z12get_work_dimv"
79 };
80
81 Function::iterator I = F.begin();
82 Instruction *FirstInsnPt = &*(I++)->getFirstInsertionPt();
83
84 bool Changed = false;
85
86 std::map<std::string, std::vector<CallInst*>> Calls;
87
88 // First collect all calls of interest.
89 for (Function::iterator E = F.end(); I != E; ++I) {
90 for (BasicBlock::iterator BI = I->begin(), BE = I->end(); BI != BE;) {
91
92 CallInst *Call = dyn_cast<CallInst>(BI++);
93
94 if (Call == nullptr) continue;
95
96 if (Call->getCalledFunction() == nullptr) {
97 // The callee can be null in case of asm snippets (TCE).
98 continue;
99 }
100 auto FuncNameI =
101 WIFuncNames.find(Call->getCalledFunction()->getName().str());
102 if (FuncNameI == WIFuncNames.end())
103 continue;
104
105 bool Unsupported = false;
106 // Check that the argument list is something we can handle.
107 for (unsigned I = 0; I < Call->getNumArgOperands(); ++I) {
108 llvm::ConstantInt *CallOperand =
109 dyn_cast<llvm::ConstantInt>(Call->getArgOperand(I));
110 if (CallOperand == nullptr)
111 Unsupported = true;
112 }
113 if (Unsupported) continue;
114 Calls[*FuncNameI].push_back(Call);
115 }
116 }
117
118 // Add single calls for the interesting functions.
119 std::map<std::string, std::vector<CallInst*> > CallsInEntry;
120 for (const auto &Call : Calls) {
121 std::string FuncName = Call.first;
122 auto CallInsts = Call.second;
123
124 for (auto CallInst : CallInsts) {
125 // Try to find a previous call with the same parameters which
126 // we can reuse.
127 std::vector<llvm::CallInst*> &CallsMoved = CallsInEntry[FuncName];
128 llvm::CallInst *PreviousCall = nullptr;
129 for (auto &M : CallsMoved) {
130 llvm::CallInst *MovedCall = dyn_cast<llvm::CallInst>(M);
131
132 // WI functions do not have variable argument lists.
133 assert (MovedCall->getNumArgOperands() ==
134 CallInst->getNumArgOperands());
135
136 bool IsApplicable = true;
137 for (unsigned I = 0; I < MovedCall->getNumArgOperands(); ++I) {
138 llvm::ConstantInt *CallOperand =
139 dyn_cast<llvm::ConstantInt>(CallInst->getArgOperand(I));
140 llvm::ConstantInt *PrevCallOperand =
141 dyn_cast<llvm::ConstantInt>(MovedCall->getArgOperand(I));
142
143 assert (isa<llvm::ConstantInt>(PrevCallOperand));
144
145 if (CallOperand->getValue() != PrevCallOperand->getValue()) {
146 IsApplicable = false;
147 break;
148 }
149 }
150 if (IsApplicable) {
151 // Found a suitable previous call instruction we can reuse.
152 PreviousCall = MovedCall;
153 break;
154 }
155 }
156
157 if (PreviousCall == nullptr) {
158 CallInst->moveBefore(FirstInsnPt);
159 CallsInEntry[FuncName].push_back(CallInst);
160 Changed = true;
161 } else {
162 // Not the first call. Refer to the first call that was moved to
163 // the entry.
164 CallInst->replaceAllUsesWith(PreviousCall);
165 CallInst->eraseFromParent();
166 Changed = true;
167 }
168 }
169 }
170 return Changed;
171}
