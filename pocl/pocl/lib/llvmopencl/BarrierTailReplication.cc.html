70bool
71BarrierTailReplication::runOnFunction(Function &F)
72{
73 if (!Workgroup::isKernelToProcess(F))
74 return false;
75
76#ifdef DEBUG_BARRIER_REPL
77 std::cerr << "### BTR on " << F.getName().str() << std::endl;
78#endif
79
80 DTP = &getAnalysis<DominatorTreeWrapperPass>();
81 DT = &DTP->getDomTree();
82
83 LI = &getAnalysis<LoopInfoWrapperPass>();
84
85 bool changed = ProcessFunction(F);
86
87
88 // In LLVM 7+, it is replaced by some assert(verify()) in LLVM itself
89#ifdef LLVM_OLDER_THAN_7_0
90 DT->verifyDomTree();
91#endif
92
93 LI->verifyAnalysis();
94 /* The created tails might contain PHI nodes with operands
95 referring to the non-predecessor (split point) BB.
96 These must be cleaned to avoid breakage later on.
97 */
98 for (Function::iterator i = F.begin(), e = F.end();
99 i != e; ++i)
100 {
101 llvm::BasicBlock *bb = &*i;
102 changed |= CleanupPHIs(bb);
103 }
104
105 return changed;
106}

108bool
109BarrierTailReplication::ProcessFunction(Function &F)
110{
111 BasicBlockSet processed_bbs;
112
113 return FindBarriersDFS(&F.getEntryBlock(), processed_bbs);
114}

117// Recursively (depht-first) look for barriers in all possible
118// execution paths starting on entry, replicating the barrier
119// successors to ensure there is a separate function exit BB
120// for each combination of traversed barriers. The set
121// processed_bbs stores the
122bool
123BarrierTailReplication::FindBarriersDFS(BasicBlock *bb,
124 BasicBlockSet &processed_bbs)
125{
126 bool changed = false;
127
128 // Check if we already visited this BB (to avoid
129 // infinite recursion in case of unbarriered loops).
130 if (processed_bbs.count(bb) != 0)
131 return changed;
132
133 processed_bbs.insert(bb);
134
135 if (block_has_barrier(bb)) {
136#ifdef DEBUG_BARRIER_REPL
137 std::cerr << "### block " << bb->getName().str() << " has barrier, RJS" << std::endl;
138#endif
139 BasicBlockSet processed_bbs_rjs;
140 changed = ReplicateJoinedSubgraphs(bb, bb, processed_bbs_rjs);
141 }
142
143 auto t = bb->getTerminator();
144
145 // Find barriers in the successors (depth first).
146 for (unsigned i = 0, e = t->getNumSuccessors(); i != e; ++i)
147 changed |= FindBarriersDFS(t->getSuccessor(i), processed_bbs);
148
149 return changed;
150}

153// Only replicate those parts of the subgraph that are not
154// dominated by a (barrier) basic block, to avoid excesive
155// (and confusing) code replication.
156bool
157BarrierTailReplication::ReplicateJoinedSubgraphs(BasicBlock *dominator,
158 BasicBlock *subgraph_entry,
159 BasicBlockSet &processed_bbs)
160{
161 bool changed = false;
162
163 assert(DT->dominates(dominator, subgraph_entry));
164
165 Function *f = dominator->getParent();
166
167 auto t = subgraph_entry->getTerminator();
168 for (int i = 0, e = t->getNumSuccessors(); i != e; ++i) {
169 BasicBlock *b = t->getSuccessor(i);
170#ifdef DEBUG_BARRIER_REPL
171 std::cerr << "### traversing from " << subgraph_entry->getName().str()
172 << " to " << b->getName().str() << std::endl;
173#endif
174
175 // Check if we already handled this BB and all its branches.
176 if (processed_bbs.count(b) != 0)
177 {
178#ifdef DEBUG_BARRIER_REPL
179 std::cerr << "### already processed " << std::endl;
180#endif
181 continue;
182 }
183
184 const bool isBackedge = DT->dominates(b, subgraph_entry);
185 if (isBackedge) {
186 // This is a loop backedge. Do not find subgraphs across
187 // those.
188#ifdef DEBUG_BARRIER_REPL
189 std::cerr << "### a loop backedge, skipping" << std::endl;
190#endif
191 continue;
192 }
193 if (DT->dominates(dominator, b))
194 {
195#ifdef DEBUG_BARRIER_REPL
196 std::cerr << "### " << dominator->getName().str() << " dominates "
197 << b->getName().str() << std::endl;
198#endif
199 changed |= ReplicateJoinedSubgraphs(dominator, b, processed_bbs);
200 }
201 else
202 {
203#ifdef DEBUG_BARRIER_REPL
204 std::cerr << "### " << dominator->getName().str() << " does not dominate "
205 << b->getName().str() << " replicating " << std::endl;
206#endif
207 BasicBlock *replicated_subgraph_entry =
208 ReplicateSubgraph(b, f);
209 t->setSuccessor(i, replicated_subgraph_entry);
210 changed = true;
211 }
212
213 if (changed)
214 {
215 // We have modified the function. Possibly created new loops.
216 // Update analysis passes.
217 DTP->runOnFunction(*f);
218
219 LI->runOnFunction(*f);
220 }
221 }
222 processed_bbs.insert(subgraph_entry);
223 return changed;
224}

226// Removes phi elements for which there are no successors (anymore).
227bool
228BarrierTailReplication::CleanupPHIs(llvm::BasicBlock *BB)
229{
230
231 bool changed = false;
232#ifdef DEBUG_BARRIER_REPL
233 std::cerr << "### CleanupPHIs for BB:" << std::endl;
234 BB->dump();
235#endif
236
237 for (BasicBlock::iterator BI = BB->begin(), BE = BB->end(); BI != BE; )
238 {
239 PHINode *PN = dyn_cast<PHINode>(BI);
240 if (PN == NULL) break;
241
242 bool PHIRemoved = false;
243 for (unsigned i = 0, e = PN->getNumIncomingValues(); i < e; ++i)
244 {
245 bool isSuccessor = false;
246 // find if the predecessor branches to this one (anymore)
247 for (unsigned s = 0,
248 se = PN->getIncomingBlock(i)->getTerminator()->getNumSuccessors();
249 s < se; ++s) {
250 if (PN->getIncomingBlock(i)->getTerminator()->getSuccessor(s) == BB)
251 {
252 isSuccessor = true;
253 break;
254 }
255 }
256 if (!isSuccessor)
257 {
258#ifdef DEBUG_BARRIER_REPL
259 std::cerr << "removing incoming value " << i << " from PHINode:" << std::endl;
260 PN->dump();
261#endif
262 PN->removeIncomingValue(i, true);
263#ifdef DEBUG_BARRIER_REPL
264 std::cerr << "now:" << std::endl;
265 PN->dump();
266#endif
267 changed = true;
268 e--;
269 if (e == 0)
270 {
271 PHIRemoved = true;
272 break;
273 }
274 i = 0;
275 continue;
276 }
277 }
278 if (PHIRemoved)
279 BI = BB->begin();
280 else
281 BI++;
282 }
283 return changed;
284}

286BasicBlock *
287BarrierTailReplication::ReplicateSubgraph(BasicBlock *entry,
288 Function *f)
289{
290 // Find all basic blocks to replicate.
291 BasicBlockVector subgraph;
292 FindSubgraph(subgraph, entry);
293
294 // Replicate subgraph maintaining control flow.
295 BasicBlockVector v;
296
297 ValueToValueMapTy m;
298 ReplicateBasicBlocks(v, m, subgraph, f);
299 UpdateReferences(v, m);
300
301 // Return entry block of replicated subgraph.
302 return cast<BasicBlock>(m[entry]);
303}

306void
307BarrierTailReplication::FindSubgraph(BasicBlockVector &subgraph,
308 BasicBlock *entry)
309{
310 // The subgraph can have internal branches (join points)
311 // avoid replicating these parts multiple times within the
312 // same tail.
313 if (std::count(subgraph.begin(), subgraph.end(), entry) > 0)
314 return;
315
316 subgraph.push_back(entry);
317
318 auto t = entry->getTerminator();
319 for (unsigned i = 0, e = t->getNumSuccessors(); i != e; ++i) {
320 BasicBlock *successor = t->getSuccessor(i);
321 const bool isBackedge = DT->dominates(successor, entry);
322 if (isBackedge) continue;
323 FindSubgraph(subgraph, successor);
324 }
325}

328void
329BarrierTailReplication::ReplicateBasicBlocks(BasicBlockVector &new_graph,
330 ValueToValueMapTy &reference_map,
331 BasicBlockVector &graph,
332 Function *f)
333{
334#ifdef DEBUG_BARRIER_REPL
335 std::cerr << "### ReplicateBasicBlocks: " << std::endl;
336#endif
337 for (BasicBlockVector::const_iterator i = graph.begin(),
338 e = graph.end();
339 i != e; ++i) {
340 BasicBlock *b = *i;
341 BasicBlock *new_b = BasicBlock::Create(b->getContext(),
342 b->getName() + ".btr",
343 f);
344 reference_map.insert(std::make_pair(b, new_b));
345 new_graph.push_back(new_b);
346
347#ifdef DEBUG_BARRIER_REPL
348 std::cerr << "Replicated BB: " << new_b->getName().str() << std::endl;
349#endif
350
351 for (BasicBlock::iterator i2 = b->begin(), e2 = b->end();
352 i2 != e2; ++i2) {
353 Instruction *i = i2->clone();
354 reference_map.insert(std::make_pair(&*i2, i));
355 new_b->getInstList().push_back(i);
356 }
357
358 // Add predicates to PHINodes of basic blocks the replicated
359 // block jumps to (backedges).
360 auto t = new_b->getTerminator();
361 for (unsigned i = 0, e = t->getNumSuccessors(); i != e; ++i) {
362 BasicBlock *successor = t->getSuccessor(i);
363 if (std::count(graph.begin(), graph.end(), successor) == 0) {
364 // Successor is not in the graph, possible backedge.
365 for (BasicBlock::iterator i = successor->begin(), e = successor->end();
366 i != e; ++i) {
367 PHINode *phi = dyn_cast<PHINode>(i);
368 if (phi == NULL)
369 break; // All PHINodes already checked.
370
371 // Get value for original incoming edge and add new predicate.
372 Value *v = phi->getIncomingValueForBlock(b);
373 Value *new_v = reference_map.find(v) == reference_map.end() ?
374 NULL : reference_map[v];
375
376 if (new_v == NULL) {
377 /* This case can happen at least when replicating a latch
378 block in a b-loop. The value produced might be from a common
379 path before the replicated part. Then just use the original value.*/
380 new_v = v;
381#if 0
382 std::cerr << "### could not find a replacement block for phi node ("
383 << b->getName().str() << ")" << std::endl;
384 phi->dump();
385 v->dump();
386 f->viewCFG();
387 assert (0);
388#endif
389 }
390 phi->addIncoming(new_v, new_b);
391 }
392 }
393 }
394 }
395
396#ifdef DEBUG_BARRIER_REPL
397 std::cerr << std::endl;
398#endif

402void
403BarrierTailReplication::UpdateReferences(const BasicBlockVector &graph,
404 ValueToValueMapTy &reference_map)
405{
406 for (BasicBlockVector::const_iterator i = graph.begin(),
407 e = graph.end();
408 i != e; ++i) {
409 BasicBlock *b = *i;
410 for (BasicBlock::iterator i2 = b->begin(), e2 = b->end();
411 i2 != e2; ++i2) {
412 Instruction *i = &*i2;
413 RemapInstruction(i, reference_map,
414 RF_IgnoreMissingLocals | RF_NoModuleLevelChanges);
415 }
416 }
417}

420static bool
421block_has_barrier(const BasicBlock *bb)
422{
423 for (BasicBlock::const_iterator i = bb->begin(), e = bb->end();
424 i != e; ++i) {
425 if (isa<Barrier>(i))
426 return true;
427 }
428
429 return false;
