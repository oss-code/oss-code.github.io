82// Recursively mark the canonical induction variable PHI as uniform.
83// If there's a canonical induction variable in loops, the variable
84// update for each iteration should be uniform. Note: this does not yet
85// imply all the work-items execute the loop same number of times!
86void
87VariableUniformityAnalysis::markInductionVariables(Function &F, llvm::Loop &L) {
88
89 if (llvm::PHINode *inductionVar = L.getCanonicalInductionVariable()) {
90#ifdef DEBUG_UNIFORMITY_ANALYSIS
91 std::cerr << "### canonical induction variable, assuming uniform:";
92 inductionVar->dump();
93#endif
94 setUniform(&F, inductionVar);
95 }
96 for (llvm::Loop *Subloop : L.getSubLoops()) {
97 markInductionVariables(F, *Subloop);
98 }
99}

101bool
102VariableUniformityAnalysis::runOnFunction(Function &F) {
103
104 if (!Workgroup::isKernelToProcess(F))
105 return false;
106
107#ifdef DEBUG_UNIFORMITY_ANALYSIS
108 std::cerr << "### refreshing VUA" << std::endl;
109 dumpCFG(F, F.getName().str() + ".vua.dot");
110 F.dump();
111#endif
112
113 /* Do the actual analysis on-demand except for the basic block
114 divergence analysis. */
115 uniformityCache_[&F].clear();
116
117 llvm::LoopInfo &LI = getAnalysis<LoopInfoWrapperPass>().getLoopInfo();
118
119 for (llvm::LoopInfo::iterator i = LI.begin(), e = LI.end(); i != e; ++i) {
120 llvm::Loop *L = *i;
121 markInductionVariables(F, *L);
122 }
123
124 setUniform(&F, &F.getEntryBlock());
125 analyzeBBDivergence(&F, &F.getEntryBlock(), &F.getEntryBlock());
126 return false;
127}

129/**
130 * Returns true in case the value should be privatized, e.g., a copy
131 * should be created for each parallel work-item.
132 *
133 * This is not the same as !isUniform() because of some of the allocas.
134 * Specifically, the loop iteration variables are sometimes uniform,
135 * that is, each work item sees the same induction variable value at every iteration,
136 * but the variables should be still replicated to avoid multiple increments
137 * of the same induction variable by each work-item.
138 */
139bool
140VariableUniformityAnalysis::shouldBePrivatized
141(llvm::Function *f, llvm::Value *val) {
142 if (!isUniform(f, val)) return true;
143
144 /* Check if the value is stored in stack (is an alloca or writes to an alloca). */
145 /* It should be enough to context save the initial alloca and the stores to
146 make sure each work-item gets their own stack slot and they are updated.
147 How the value (based on which of those allocas) is computed does not matter as
148 we are deadling with uniform computation. */
149
150 if (isa<AllocaInst>(val)) return true;
151
152 if (isa<StoreInst>(val) &&
153 isa<AllocaInst>(dyn_cast<StoreInst>(val)->getPointerOperand())) return true;
154 return false;
155}

158/**
159 * BB divergence analysis.
160 *
161 * Define:
162 * Uniform BB. A basic block which is known to be executed by all or none
163 * of the work-items, that is, a BB where it's known safe to add a barrier.
164 *
165 * Divergent/varying BB. A basic block where work-items *might* diverge.
166 * That is, it cannot be proven that all work-items execute the BB.
167 *
168 * Propagate the information from the entry downwards (breadth first).
169 * This avoids infinite recursion with loop back edges and enables
170 * to keep book of the "last seen" uniform BB.
171 *
172 * The conditions to mark a BB 'uniform':
173 *
174 * a) the function entry, or
175 * b) BBs that post-dominate at least one uniform BB (try the previously
176 * found one), or
177 * c) BBs that are branched to directly from a uniform BB using a uniform branch.
178 * Note: This assumes the CFG is well-formed in a way that there cannot be a divergent
179 * branch to the same BB in that case.
180 *
181 * Otherwise, assume divergent (might not be *proven* to be one!).
182 *
183 */
184void
185VariableUniformityAnalysis::analyzeBBDivergence
186(llvm::Function *f, llvm::BasicBlock *bb, llvm::BasicBlock *previousUniformBB) {
187
188#ifdef DEBUG_UNIFORMITY_ANALYSIS
189 std::cerr << "### Analyzing BB divergence (bb=" << bb->getName().str()
190 << ", prevUniform=" << previousUniformBB->getName().str() << ")"
191 << std::endl;
192#endif
193
194 auto Term = previousUniformBB->getTerminator();
195 if (Term == NULL) {
196 // this is most likely a function with a single basic block, the entry
197 // node, which ends with a ret
198 return;
199 }
200
201 llvm::BranchInst *BrInst = dyn_cast<llvm::BranchInst>(Term);
202 llvm::SwitchInst *SwInst = dyn_cast<llvm::SwitchInst>(Term);
203
204 if (BrInst == nullptr && SwInst == nullptr) {
205 // Can only handle branches and switches for now.
206 return;
207 }
208
209 // The BBs that were found uniform.
210 std::vector<llvm::BasicBlock *> FoundUniforms;
211
212 // Condition c)
213 if ((BrInst && (!BrInst->isConditional() ||
214 isUniform(f, BrInst->getCondition()))) ||
215 (SwInst && isUniform(f, SwInst->getCondition()))) {
216 // This is a branch with a uniform condition, propagate the uniformity
217 // to the BB of interest.
218 for (unsigned suc = 0, end = Term->getNumSuccessors(); suc < end; ++suc) {
219 llvm::BasicBlock *Successor = Term->getSuccessor(suc);
220 // TODO: should we check that there are no divergent entries to this
221 // BB even though if the currently checked condition is uniform?
222 setUniform(f, Successor, true);
223 FoundUniforms.push_back(Successor);
224 }
225 }
226
227 // Condition b)
228 if (FoundUniforms.size() == 0) {
229 llvm::PostDominatorTreeWrapperPass *PDT = &getAnalysis<PostDominatorTreeWrapperPass>();
230 if (PDT->getPostDomTree().dominates(bb, previousUniformBB)) {
231 setUniform(f, bb, true);
232 FoundUniforms.push_back(bb);
233 }
234 }
235
236 /* Assume diverging. */
237 if (!isUniformityAnalyzed(f, bb))
238 setUniform(f, bb, false);
239
240 for (auto UniformBB : FoundUniforms) {
241
242 // Propagate the Uniform BB data downwards.
243 auto NextTerm = UniformBB->getTerminator();
244
245 for (unsigned suc = 0, end = NextTerm->getNumSuccessors(); suc < end;
246 ++suc) {
247 llvm::BasicBlock *NextBB = NextTerm->getSuccessor(suc);
248 if (!isUniformityAnalyzed(f, NextBB)) {
249 analyzeBBDivergence(f, NextBB, UniformBB);
250 }
251 }
252 }

255bool
256VariableUniformityAnalysis::isUniformityAnalyzed(llvm::Function *f, llvm::Value *v) const {
257 UniformityIndex &cache = uniformityCache_[f];
258 UniformityIndex::const_iterator i = cache.find(v);
259 if (i != cache.end()) {
260 return true;
261 }
262 return false;
263}

265/**
266 * Simple uniformity analysis that recursively analyses all the
267 * operands affecting the value.
268 *
269 * Known uniform Values that act as "leafs" in the recursive uniformity
270 * check logic:
271 * a) kernel arguments
272 * b) constants
273 * c) OpenCL C identifiers that are constant for all work-items in a work-group
274 *
275 */
276bool
277VariableUniformityAnalysis::isUniform(llvm::Function *f, llvm::Value* v) {
278
279 UniformityIndex &cache = uniformityCache_[f];
280 UniformityIndex::const_iterator i = cache.find(v);
281 if (i != cache.end()) {
282 return (*i).second;
283 }
284
285 if (llvm::BasicBlock *bb = dyn_cast<llvm::BasicBlock>(v)) {
286 if (bb == &f->getEntryBlock()) {
287 setUniform(f, v, true);
288 return true;
289 }
290 }
291
292 if (isa<llvm::Argument>(v)) {
293 setUniform(f, v, true);
294 return true;
295 }
296
297 if (isa<llvm::ConstantInt>(v)) {
298 setUniform(f, v, true);
299 return true;
300 }
301
302 if (isa<llvm::AllocaInst>(v)) {
303 /* Allocas might or might not be divergent. These are produced
304 from work-item private arrays or the PHIsToAllocas. It depends
305 what is written to them whether they are really divergent.
306
307 We need to figure out if any of the stores to the alloca contain
308 work-item id dependent data. Take a white listing approach that
309 detects the ex-phi allocas of loop iteration variables of non-diverging
310 loops.
311
312 Currently the following case is white listed:
313 a) are scalars, and
314 b) are accessed only with load and stores (e.g. address not taken) from
315 uniform basic blocks, and
316 c) the stored data is uniform
317
318 Because alloca data can be modified in loops and thus be dependent on
319 itself, we need a bit involved mechanism to handle it. First create
320 a copy of the uniformity cache, then assume the alloca itself is uniform,
321 then check if all the stores to the alloca contain uniform data. If
322 our initial assumption was wrong, restore the cache from the backup.
323 */
324 UniformityCache backupCache(uniformityCache_);
325 setUniform(f, v);
326
327 bool isUniformAlloca = true;
328 llvm::Instruction *instruction = dyn_cast<llvm::AllocaInst>(v);
329 for (Instruction::use_iterator ui = instruction->use_begin(),
330 ue = instruction->use_end();
331 ui != ue; ++ui) {
332 llvm::Instruction *user = cast<Instruction>(ui->getUser());
333 if (user == NULL) continue;
334
335 llvm::StoreInst *store = dyn_cast<llvm::StoreInst>(user);
336 if (store) {
337 if (!isUniform(f, store->getValueOperand()) ||
338 !isUniform(f, store->getParent())) {
339 if (!isUniform(f, store->getParent())) {
340#ifdef DEBUG_UNIFORMITY_ANALYSIS
341 std::cerr << "### alloca was written in a non-uniform BB" << std::endl;
342 store->getParent()->dump();
343 /* TODO: This is a problematic chicken-egg situation because the
344 BB uniformity check ends up analyzing allocas in phi-removed code:
345 the loop constructs refer to these allocas and at that point we
346 do not yet know if the BB itself is uniform. This leads to not
347 being able to detect loop iteration variables as uniform. */
348#endif
349 }
350 isUniformAlloca = false;
351 break;
352 }
353 } else if (isa<llvm::LoadInst>(user) || isa<llvm::BitCastInst>(user)) {
354 } else {
355#ifdef DEBUG_UNIFORMITY_ANALYSIS
356 std::cerr << "### alloca has a suspicious user" << std::endl;
357 user->dump();
358#endif
359 isUniformAlloca = false;
360 break;
361 }
362 }
363
364 if (!isUniformAlloca) {
365 // restore the old uniform data as our guess was wrong
366 uniformityCache_ = backupCache;
367 }
368 setUniform(f, v, isUniformAlloca);
369
370 return isUniformAlloca;
371 }
372
373 /* TODO: global memory loads are uniform in case they are accessing
374 the higher scope ids (group_id_?). */
375 if (isa<llvm::LoadInst>(v)) {
376 llvm::LoadInst *load = dyn_cast<llvm::LoadInst>(v);
377 llvm::Value *pointer = load->getPointerOperand();
378 llvm::Module *M = load->getParent()->getParent()->getParent();
379
380 if (pointer == M->getGlobalVariable("_group_id_x") ||
381 pointer == M->getGlobalVariable("_group_id_y") ||
382 pointer == M->getGlobalVariable("_group_id_z") ||
383 pointer == M->getGlobalVariable("_work_dim") ||
384 pointer == M->getGlobalVariable("_num_groups_x") ||
385 pointer == M->getGlobalVariable("_num_groups_y") ||
386 pointer == M->getGlobalVariable("_num_groups_z") ||
387 pointer == M->getGlobalVariable("_global_offset_x") ||
388 pointer == M->getGlobalVariable("_global_offset_y") ||
389 pointer == M->getGlobalVariable("_global_offset_z") ||
390 pointer == M->getGlobalVariable("_local_size_x") ||
391 pointer == M->getGlobalVariable("_local_size_y") ||
392 pointer == M->getGlobalVariable("_local_size_z")) {
393
394 setUniform(f, v, true);
395 return true;
396 }
397 }
398
399 if (isa<llvm::PHINode>(v)) {
400 /* TODO: PHINodes need control flow analysis:
401 even if the values are uniform, the selected
402 value depends on the preceeding basic block which
403 might depend on the ID. Assume they are not uniform
404 for now in general and treat the loop iteration
405 variable as a special case (set externally from a LoopPass).
406
407 TODO: PHINodes can depend (indirectly or directly) on itself in loops
408 so it would need infinite recursion checking.
409 */
410 setUniform(f, v, false);
411 return false;
412 }
413
414 llvm::Instruction *instr = dyn_cast<llvm::Instruction>(v);
415 if (instr == NULL) {
416 setUniform(f, v, false);
417 return false;
418 }
419
420 // Atomic operations might look like uniform if only considering the operands
421 // (access a global memory location of which ordering by default is not
422 // constrained), but their semantics have ordering: Each work-item should get
423 // their own value from that memory location.
424 if (instr->isAtomic()) {
425 setUniform(f, v, false);
426 return false;
427 }
428
429 // not computed previously, scan all operands of the instruction
430 // and figure out their uniformity recursively
431 for (unsigned opr = 0; opr < instr->getNumOperands(); ++opr) {
432 llvm::Value *operand = instr->getOperand(opr);
433 if (!isUniform(f, operand)) {
434 setUniform(f, v, false);
435 return false;
436 }
437 }
438 setUniform(f, v, true);
439 return true;
440}
441
442void
443VariableUniformityAnalysis::setUniform(llvm::Function *f,
444 llvm::Value *v,
445 bool isUniform) {
446
447 UniformityIndex &cache = uniformityCache_[f];
448 cache[v] = isUniform;
449
450#ifdef DEBUG_UNIFORMITY_ANALYSIS
451 std::cerr << "### ";
452 if (isUniform)
453 std::cerr << "uniform ";
454 else
455 std::cerr << "varying ";
456
457 if (isa<llvm::BasicBlock>(v)) {
458 std::cerr << "BB: " << v->getName().str() << std::endl;
459 } else {
460 v->dump();
461 }
462#endif
463}

465bool
466VariableUniformityAnalysis::doFinalization(llvm::Module& /*M*/) {
467 uniformityCache_.clear();
468 return true;
469}
