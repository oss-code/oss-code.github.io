54//
55// Function: hasConstantGEP()
56//
57// Description:
58// This function determines whether the given value is a constant expression
59// that has a constant GEP expression embedded within it.
60//
61// Inputs:
62// V - The value to check.
63//
64// Return value:
65// NULL - This value is not a constant expression with a constant expression
66// GEP within it.
67// ~NULL - A pointer to the value casted into a ConstantExpr is returned.
68//
69static ConstantExpr *
70hasConstantGEP (Value * V) {
71 if (ConstantExpr * CE = dyn_cast<ConstantExpr>(V)) {
72 bool isGEPOrCast =
73 CE->getOpcode() == Instruction::GetElementPtr ||
74 CE->getOpcode() == Instruction::BitCast;
75
76 isGEPOrCast |= CE->getOpcode() == Instruction::AddrSpaceCast;
77 if (isGEPOrCast) {
78 return CE;
79 } else {
80 for (unsigned index = 0; index < CE->getNumOperands(); ++index) {
81 if (hasConstantGEP (CE->getOperand(index)))
82 return CE;
83 }
84 }
85 }
86
87 return 0;
88}

90//
91// Function: convertGEP()
92//
93// Description:
94// Convert a GEP constant expression into a GEP instruction.
95//
96// Inputs:
97// CE - The GEP constant expression.
98// InsertPt - The instruction before which to insert the new GEP instruction.
99//
100// Return value:
101// A pointer to the new GEP instruction is returned.
102//
103static Instruction *
104convertGEP (ConstantExpr * CE, Instruction * InsertPt) {
105 //
106 // Create iterators to the indices of the constant expression.
107 //
108 std::vector<Value *> Indices;
109 for (unsigned index = 1; index < CE->getNumOperands(); ++index) {
110 Indices.push_back (CE->getOperand (index));
111 }
112
113 //
114 // Update the statistics.
115 //
116 ++GEPChanges;
117
118 //
119 // Make the new GEP instruction.
120 //
121 /* The first NULL is the Type. It is not used at all, just asserted
122 * against. And it asserts, no matter what is passed. Except NULL.
123 * Seems this API is still "fluctuation in progress"*/
124 return (GetElementPtrInst::Create (NULL,
125 CE->getOperand(0),
126 Indices,
127 CE->getName(),
128 InsertPt));
129}

131//
132// Function: convertExpression()
133//
134// Description:
135// Convert a constant expression into an instruction. This routine does *not*
136// perform any recursion, so the resulting instruction may have constant
137// expression operands.
138//
139static Instruction *
140convertExpression (ConstantExpr * CE, Instruction * InsertPt) {
141 //
142 // Convert this constant expression into a regular instruction.
143 //
144 Instruction * NewInst = 0;
145 switch (CE->getOpcode()) {
146 case Instruction::GetElementPtr: {
147 NewInst = convertGEP (CE, InsertPt);
148 break;
149 }
150
151 case Instruction::Add:
152 case Instruction::Sub:
153 case Instruction::Mul:
154 case Instruction::UDiv:
155 case Instruction::SDiv:
156 case Instruction::FDiv:
157 case Instruction::URem:
158 case Instruction::SRem:
159 case Instruction::FRem:
160 case Instruction::Shl:
161 case Instruction::LShr:
162 case Instruction::AShr:
163 case Instruction::And:
164 case Instruction::Or:
165 case Instruction::Xor: {
166 Instruction::BinaryOps Op = (Instruction::BinaryOps)(CE->getOpcode());
167 NewInst = BinaryOperator::Create (Op,
168 CE->getOperand(0),
169 CE->getOperand(1),
170 CE->getName(),
171 InsertPt);
172 break;
173 }
174
175 case Instruction::Trunc:
176 case Instruction::ZExt:
177 case Instruction::SExt:
178 case Instruction::FPToUI:
179 case Instruction::FPToSI:
180 case Instruction::UIToFP:
181 case Instruction::SIToFP:
182 case Instruction::FPTrunc:
183 case Instruction::FPExt:
184 case Instruction::PtrToInt:
185 case Instruction::IntToPtr:
186 case Instruction::AddrSpaceCast:
187 case Instruction::BitCast: {
188 Instruction::CastOps Op = (Instruction::CastOps)(CE->getOpcode());
189 NewInst = CastInst::Create (Op,
190 CE->getOperand(0),
191 CE->getType(),
192 CE->getName(),
193 InsertPt);
194 break;
195 }
196
197 case Instruction:: FCmp:
198 case Instruction:: ICmp: {
199 Instruction::OtherOps Op = (Instruction::OtherOps)(CE->getOpcode());
200 NewInst = CmpInst::Create (Op,
201 (llvm::CmpInst::Predicate)CE->getPredicate(),
202 CE->getOperand(0),
203 CE->getOperand(1),
204 CE->getName(),
205 InsertPt);
206 break;
207 }
208
209 case Instruction:: Select:
210 NewInst = SelectInst::Create (CE->getOperand(0),
211 CE->getOperand(1),
212 CE->getOperand(2),
213 CE->getName(),
214 InsertPt);
215 break;
216
217 case Instruction:: ExtractElement:
218 case Instruction:: InsertElement:
219 case Instruction:: ShuffleVector:
220 case Instruction:: InsertValue:
221 default:
222 assert (0 && "Unhandled constant expression!\n");
223 break;
224 }
225
226 //
227 // Update the statistics.
228 //
229 ++TotalChanges;
230
231 return NewInst;
232}

234//
235// Method: runOnFunction()
236//
237// Description:
238// Entry point for this LLVM pass.
239//
240// Return value:
241// true - The function was modified.
242// false - The function was not modified.
243//
244bool
245BreakConstantGEPs::runOnFunction (Function & F) {
246
247 if (!pocl::Workgroup::isKernelToProcess(F)) return false;
248
249 bool modified = false;
250
251 // Worklist of values to check for constant GEP expressions
252 std::vector<Instruction *> Worklist;
253
254 //
255 // Initialize the worklist by finding all instructions that have one or more
256 // operands containing a constant GEP expression.
257 //
258 for (Function::iterator BB = F.begin(); BB != F.end(); ++BB) {
259 for (BasicBlock::iterator i = BB->begin(); i != BB->end(); ++i) {
260 //
261 // Scan through the operands of this instruction. If it is a constant
262 // expression GEP, insert an instruction GEP before the instruction.
263 //
264 Instruction * I = &*i;
265 for (unsigned index = 0; index < I->getNumOperands(); ++index) {
266 if (hasConstantGEP (I->getOperand(index))) {
267 Worklist.push_back (I);
268 }
269 }
270 }
271 }
272
273 //
274 // Determine whether we will modify anything.
275 //
276 if (Worklist.size()) modified = true;
277
278 //
279 // While the worklist is not empty, take an item from it, convert the
280 // operands into instructions if necessary, and determine if the newly
281 // added instructions need to be processed as well.
282 //
283 while (Worklist.size()) {
284 Instruction * I = Worklist.back();
285 Worklist.pop_back();
286
287 //
288 // Scan through the operands of this instruction and convert each into an
289 // instruction. Note that this works a little differently for phi
290 // instructions because the new instruction must be added to the
291 // appropriate predecessor block.
292 //
293 if (PHINode * PHI = dyn_cast<PHINode>(I)) {
294 for (unsigned index = 0; index < PHI->getNumIncomingValues(); ++index) {
295 //
296 // For PHI Nodes, if an operand is a constant expression with a GEP, we
297 // want to insert the new instructions in the predecessor basic block.
298 //
299 // Note: It seems that it's possible for a phi to have the same
300 // incoming basic block listed multiple times; this seems okay as long
301 // the same value is listed for the incoming block.
302 //
303 Instruction * InsertPt = PHI->getIncomingBlock(index)->getTerminator();
304 if (ConstantExpr * CE = hasConstantGEP (PHI->getIncomingValue(index))) {
305 Instruction * NewInst = convertExpression (CE, InsertPt);
306 for (unsigned i2 = index; i2 < PHI->getNumIncomingValues(); ++i2) {
307 if ((PHI->getIncomingBlock (i2)) == PHI->getIncomingBlock (index))
308 PHI->setIncomingValue (i2, NewInst);
309 }
310 Worklist.push_back (NewInst);
311 }
312 }
313 } else {
314 for (unsigned index = 0; index < I->getNumOperands(); ++index) {
315 //
316 // For other instructions, we want to insert instructions replacing
317 // constant expressions immediently before the instruction using the
318 // constant expression.
319 //
320 if (ConstantExpr * CE = hasConstantGEP (I->getOperand(index))) {
321 Instruction * NewInst = convertExpression (CE, I);
322 I->replaceUsesOfWith (CE, NewInst);
323 Worklist.push_back (NewInst);
324 }
325 }
326 }
327 }
328
329 return modified;
330}
