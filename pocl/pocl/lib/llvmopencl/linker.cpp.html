24/* Lightweight bitcode linker to replace llvm::Linker. Unlike the LLVM default
25 linker, this does not link in the entire given module, only the called
26 functions are cloned from the input. This is to speed up the linking of the
27 kernel lib which is so big, that it takes seconds to clone it, even on
28 top-of-the line current processors. */

57/*
58 * Find needle in haystack. O(n) implementation, but n should be
59 * small in our usecases.
60 * Return true if found
61 */
62static bool
63find_from_list(llvm::StringRef needle,
64 std::list<llvm::StringRef> &haystack)
65{
66 std::list<llvm::StringRef>::iterator li,le;
67 for (li=haystack.begin(), le=haystack.end();
68 li != le;
69 li++) {
70 if (needle == *li) {
71 return true;
72 }
73 }
74 return false;
75}

77/* Fixes address space on opencl.imageX_t arguments to be global.
78 * Note this does not change the types in Function->FunctionType
79 * so it's only used inside CopyFunc on kernel library functions */
80static void fixOpenCLimageArguments(llvm::Function *Func) {
81 Function::arg_iterator b = Func->arg_begin();
82 Function::arg_iterator e = Func->arg_end();
83 for (; b != e; b++) {
84 Argument *j = &*b;
85 Type *t = j->getType();
86 if (t->isPointerTy() && t->getPointerElementType()->isStructTy()) {
87 Type *pe_type = t->getPointerElementType();
88 if (pe_type->getStructName().startswith("opencl.image")) {
89 Type *new_t =
90 PointerType::get(pe_type, currentPoclDevice->global_as_id);
91 j->mutateType(new_t);
92 }
93 }
94 }
95}

97/* Fixes opencl.imageX_t type arguments which miss address space global
98 * returns F if no changes are required, or a new cloned F if the arguments
99 * require a fix. To be used on user's kernel code itself, not on kernel library.
100 */
101static llvm::Function *
102CloneFuncFixOpenCLImageT(llvm::Module *Mod, llvm::Function *F)
103{
104 assert(F && "No function to copy");
105 assert(!F->isDeclaration());
106
107 int changed = 0;
108 ValueToValueMapTy VVMap;
109 SmallVector<Type *, 8> sv;
110 for (Function::arg_iterator i = F->arg_begin(), e = F->arg_end();
111 i != e; ++i) {
112 Argument *j = &*i;
113 Type *t = j->getType();
114 Type *new_t = t;
115 if (t->isPointerTy() && t->getPointerElementType()->isStructTy()) {
116 Type *pe_type = t->getPointerElementType();
117 if (pe_type->getStructName().startswith("opencl.image")) {
118
119 if (t->getPointerAddressSpace() != currentPoclDevice->global_as_id) {
120 new_t = PointerType::get(pe_type, currentPoclDevice->global_as_id);
121 changed = 1;
122 }
123 }
124 }
125 sv.push_back(new_t);
126 }
127
128 if (!changed)
129 return F;
130
131 F->removeFromParent();
132
133 FunctionType *NewFT = FunctionType::get(F->getReturnType(),
134 ArrayRef<Type *> (sv),
135 false);
136 assert(NewFT);
137 llvm::Function *DstFunc = nullptr;
138
139 DstFunc = Function::Create(NewFT, F->getLinkage(), F->getName(), Mod);
140
141 Function::arg_iterator j = DstFunc->arg_begin();
142 for (Function::const_arg_iterator i = F->arg_begin(),
143 e = F->arg_end();
144 i != e; ++i) {
145 j->setName(i->getName());
146 VVMap[&*i] = &*j;
147 ++j;
148 }
149
150 DstFunc->copyAttributesFrom(F);
151
152 SmallVector<ReturnInst*, 8> RI; // Ignore returns cloned.
153 CloneFunctionInto(DstFunc, F, VVMap, true, RI);
154 delete F;
155
156 return DstFunc;
157}

159// Find all functions in the calltree of F, append their
160// name to list.
161static inline void
162find_called_functions(llvm::Function *F,
163 std::list<llvm::StringRef> &list)
164{
165 if (F->isDeclaration()) {
166 DB_PRINT("it's a declaration.\n");
167 return;
168 }
169 llvm::Function::iterator fi,fe;
170 for (fi = F->begin(), fe = F->end();
171 fi != fe; fi++) {
172 llvm::BasicBlock::iterator bi, be;
173 for (bi = fi->begin(), be = fi->end();
174 bi != be;
175 bi++) {
176 CallInst *CI = dyn_cast<CallInst>(bi);
177 if (CI == NULL)
178 continue;
179 llvm::Function *callee = CI->getCalledFunction();
180 // this happens with e.g. inline asm calls
181 if (callee == NULL) {
182 DB_PRINT("search: %s callee NULL\n", F->getName().str().c_str());
183 continue;
184 }
185 DB_PRINT("search: %s calls %s\n",
186 F->getName().data(), callee->getName().data());
187 if (find_from_list(callee->getName(), list))
188 continue;
189 else {
190 list.push_back(callee->getName());
191 DB_PRINT("search: recursing into %s\n",
192 callee->getName().data());
193 find_called_functions(callee, list);
194 }
195 }
196 }
197}

199// Copies one function from one module to another
200// does not inspect it for callgraphs
201static void
202CopyFunc(const llvm::StringRef Name,
203 const llvm::Module * From,
204 llvm::Module * To,
205 ValueToValueMapTy & VVMap) {
206
207 llvm::Function *SrcFunc = From->getFunction(Name);
208 // TODO: is this the linker error "not found", and not an assert?
209 assert(SrcFunc && "Did not find function to copy in kernel library");
210 llvm::Function *DstFunc = To->getFunction(Name);
211
212 if (DstFunc == NULL) {
213 DB_PRINT(" %s not found in destination module, creating\n",
214 Name.data());
215 DstFunc =
216 Function::Create(cast<FunctionType>(
217 SrcFunc->getType()->getElementType()),
218 SrcFunc->getLinkage(),
219 SrcFunc->getName(),
220 To);
221 DstFunc->copyAttributesFrom(SrcFunc);
222 } else if (DstFunc->size() > 0) {
223 // We have already encountered and copied this function.
224 return;
225 }
226 VVMap[SrcFunc] = DstFunc;
227
228 Function::arg_iterator j = DstFunc->arg_begin();
229 for (Function::const_arg_iterator i = SrcFunc->arg_begin(),
230 e = SrcFunc->arg_end();
231 i != e; ++i) {
232 j->setName(i->getName());
233 VVMap[&*i] = &*j;
234 ++j;
235 }
236 if (!SrcFunc->isDeclaration()) {
237 SmallVector<ReturnInst*, 8> RI; // Ignore returns cloned.
238 DB_PRINT(" cloning %s\n", Name.data());
239 CloneFunctionInto(DstFunc, SrcFunc, VVMap, true, RI);
240 fixOpenCLimageArguments(DstFunc);
241 } else {
242 DB_PRINT(" found %s, but its a declaration, do nothing\n",
243 Name.data());
244 }
245}

247/* Copy function F and all the functions in its call graph
248 * that are defined in 'from', into 'to', adding the mappings to
249 * 'vvm'.
250 */
251static int
252copy_func_callgraph(const llvm::StringRef func_name,
253 const llvm::Module * from,
254 llvm::Module * to,
255 ValueToValueMapTy & vvm) {
256 std::list<llvm::StringRef> callees;
257 llvm::Function *RootFunc = from->getFunction(func_name);
258 if (RootFunc == NULL)
259 return -1;
260 DB_PRINT("copying function %s with callgraph\n", RootFunc.data());
261
262 find_called_functions(RootFunc, callees);
263
264 // First copy the callees of func, then the function itself.
265 // Recurse into callees to handle the case where kernel library
266 // functions call other kernel library functions.
267 std::list<llvm::StringRef>::iterator ci,ce;
268 for (ci = callees.begin(), ce = callees.end(); ci != ce; ci++) {
269 llvm::Function *SrcFunc = from->getFunction(*ci);
270 if (!SrcFunc->isDeclaration()) {
271 copy_func_callgraph(*ci, from, to, vvm);
272 } else {
273 DB_PRINT("%s is declaration, not recursing into it!\n",
274 SrcFunc->getName().str().c_str());
275 }
276 CopyFunc(*ci, from, to, vvm);
277 }
278 CopyFunc(func_name, from, to, vvm);
279 return 0;
280}

282static inline bool
283stringref_equal(llvm::StringRef a, llvm::StringRef b)
284{
285 return a.equals(b);

288static inline bool
289stringref_cmp(llvm::StringRef a, llvm::StringRef b)
290{
291 return a.str() < b.str();

294static void shared_copy(llvm::Module *program, const llvm::Module *lib,
295 std::string &log, ValueToValueMapTy &vvm) {
296
297 llvm::Module::const_global_iterator gi,ge;
298
299 // copy any aliases to program
300 DB_PRINT("cloning the aliases:\n");
301 llvm::Module::const_alias_iterator ai, ae;
302 for (ai = lib->alias_begin(), ae = lib->alias_end(); ai != ae; ai++) {
303 DB_PRINT(" %s\n", ai->getName().data());
304 GlobalAlias *GA =
305 GlobalAlias::create(
306 ai->getType(), ai->getType()->getAddressSpace(), ai->getLinkage(),
307 ai->getName(), NULL, program);
308
309 GA->copyAttributesFrom(&*ai);
310 vvm[&*ai]=GA;
311 }
312
313 // initialize the globals that were copied
314 for (gi=lib->global_begin(), ge=lib->global_end();
315 gi != ge;
316 gi++) {
317 GlobalVariable *GV=cast<GlobalVariable>(vvm[&*gi]);
318 if (gi->hasInitializer())
319 GV->setInitializer(MapValue(gi->getInitializer(), vvm));
320 }
321
322 // copy metadata
323 DB_PRINT("cloning metadata:\n");
324 llvm::Module::const_named_metadata_iterator mi,me;
325 for (mi=lib->named_metadata_begin(), me=lib->named_metadata_end();
326 mi != me; mi++) {
327 const NamedMDNode &NMD = *mi;
328 // This causes problems with NVidia, and is regenerated by pocl-ptx-gen
329 // anyway.
330 if (NMD.getName() == StringRef("nvvm.annotations"))
331 continue;
332 DB_PRINT(" %s:\n", NMD.getName().data());
333 if (program->getNamedMetadata(NMD.getName())) {
334 // Let's not overwrite existing metadata such as llvm.module.flags and
335 // opencl.ocl.version.
336 continue;
337 }
338 NamedMDNode *NewNMD = program->getOrInsertNamedMetadata(NMD.getName());
339 for (unsigned i = 0, e = NMD.getNumOperands(); i != e; ++i)
340 NewNMD->addOperand(MapMetadata(NMD.getOperand(i), vvm));
341 }
342}

344int link(llvm::Module *program, const llvm::Module *lib, std::string &log) {
345 assert(program);
346 assert(lib);
347 ValueToValueMapTy vvm;
348 std::list<llvm::StringRef> declared;
349
350 // Include auxiliary functions required by the device at hand.
351 if (const char **DevAuxFuncs = currentPoclDevice->device_aux_functions) {
352 const char **Func = DevAuxFuncs;
353 while (*Func != nullptr) {
354 declared.push_back(*Func++);
355 }
356 }
357
358 llvm::Module::iterator fi, fe;
359
360 // Find and fix opencl.imageX_t arguments
361 for (fi = program->begin(), fe = program->end(); fi != fe; fi++) {
362 llvm::Function *f = &*fi;
363 if (f->isDeclaration())
364 continue;
365 // need to restart iteration if we replace a function
366 if (CloneFuncFixOpenCLImageT(program, f) != f) {
367 fi = program->begin();
368 }
369 }
370
371 // Inspect the program, find undefined functions
372 for (fi = program->begin(), fe = program->end(); fi != fe; fi++) {
373 if ((*fi).isDeclaration()) {
374 DB_PRINT("%s is not defined\n", fi->getName().data());
375 declared.push_back(fi->getName());
376 continue;
377 }
378
379 // Find all functions the program source calls
380 // TODO: is there no direct way?
381 find_called_functions(&*fi, declared);
382 }
383 declared.sort(stringref_cmp);
384 declared.unique(stringref_equal);
385
386 // some global variables can have external linkage. Set it to private
387 // otherwise these end up in ELF relocation tables and cause link
388 // failures when linking with -fPIC/PIE
389 llvm::Module::global_iterator gi1, ge1;
390 for (gi1 = program->global_begin(), ge1 = program->global_end(); gi1 != ge1;
391 gi1++) {
392 GlobalValue::LinkageTypes linkage = gi1->getLinkage();
393 if (linkage == GlobalValue::LinkageTypes::ExternalLinkage)
394 gi1->setLinkage(GlobalValue::LinkageTypes::PrivateLinkage);
395 }
396
397 // Copy all the globals from lib to program.
398 // It probably is faster to just copy them all, than to inspect
399 // both program and lib to find which actually are used.
400 DB_PRINT("cloning the global variables:\n");
401 llvm::Module::const_global_iterator gi,ge;
402 for (gi=lib->global_begin(), ge=lib->global_end(); gi != ge; gi++) {
403 DB_PRINT(" %s\n", gi->getName().data());
404 GlobalVariable *GV = new GlobalVariable(
405 *program, gi->getType()->getElementType(), gi->isConstant(),
406 gi->getLinkage(), (Constant*)0, gi->getName(), (GlobalVariable*)0,
407 gi->getThreadLocalMode(), gi->getType()->getAddressSpace());
408 GV->copyAttributesFrom(&*gi);
409 vvm[&*gi]=GV;
410 }
411
412 // For each undefined function in program,
413 // clone it from the lib to the program module,
414 // if found in lib
415 bool found_all_undefined = true;
416
417 // this one is a handled with a special pocl LLVM pass
418 StringRef pocl_sampler_handler("__translate_sampler_initializer");
419 // ignore undefined llvm intrinsics
420 StringRef llvm_intrins("llvm.");
421 std::list<llvm::StringRef>::iterator di,de;
422 for (di = declared.begin(), de = declared.end();
423 di != de; di++) {
424 llvm::StringRef r = *di;
425 if (copy_func_callgraph(r, lib, program, vvm)) {
426 Function *f = program->getFunction(r);
427 if ((f == NULL) ||
428 (f->isDeclaration() &&
429 // A target might want to expose the C99 printf in case not supporting
430 // the OpenCL 1.2 printf.
431 !f->getName().equals("printf") &&
432 !f->getName().equals(pocl_sampler_handler) &&
433 !f->getName().startswith(llvm_intrins))
434 ) {
435 log.append("Cannot find symbol ");
436 log.append(r.str());
437 log.append(" in kernel library\n");
438 found_all_undefined = false;
439 }
440 }
441 }
442 if (!found_all_undefined)
443 return 1;
444
445 shared_copy(program, lib, log, vvm);
446
447 return 0;
448}

450int copyKernelFromBitcode(const char* name, llvm::Module *parallel_bc,
451 const llvm::Module *program) {
452 ValueToValueMapTy vvm;
453
454 // Copy all the globals from lib to program.
455 // It probably is faster to just copy them all, than to inspect
456 // both program and lib to find which actually are used.
457 DB_PRINT("cloning the global variables:\n");
458 llvm::Module::const_global_iterator gi,ge;
459 for (gi=program->global_begin(), ge=program->global_end(); gi != ge; gi++) {
460 DB_PRINT(" %s\n", gi->getName().data());
461 GlobalVariable *GV = new GlobalVariable(
462 *parallel_bc, gi->getType()->getElementType(), gi->isConstant(),
463 gi->getLinkage(), (Constant*)0, gi->getName(), (GlobalVariable*)0,
464 gi->getThreadLocalMode(), gi->getType()->getAddressSpace());
465 GV->copyAttributesFrom(&*gi);
466 vvm[&*gi]=GV;
467 }
468
469 const StringRef kernel_name(name);
470 copy_func_callgraph(kernel_name, program, parallel_bc, vvm);
471
472 if (const char **DevAuxFuncs = currentPoclDevice->device_aux_functions) {
473 const char **Func = DevAuxFuncs;
474 while (*Func != nullptr) {
475 copy_func_callgraph(*Func++, program, parallel_bc, vvm);
476 }
477 }
478
479 std::string log;
480 shared_copy(parallel_bc, program, log, vvm);
481
482 return 0;
483}
