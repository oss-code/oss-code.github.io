51inline bool
52isAutomaticLocal(const std::string &FuncName, llvm::GlobalVariable &Var) {
53 // Without the fake address space IDs, there is no reliable way to figure out
54 // if the address space is local from the bitcode. We could check its AS
55 // against the device's local address space id, but for now lets rely on the
56 // naming convention only. Only relying on the naming convention has the problem
57 // that LLVM can move private const arrays to the global space which make
58 // them look like local arrays (see Github Issue 445). This should be properly
59 // fixed in Clang side with e.g. a naming convention for the local arrays to
60 // detect them robstly without having logical address space info in the IR.
61 return Var.getName().startswith(FuncName + ".") &&
62 llvm::isa<llvm::PointerType>(Var.getType()) && !Var.isConstant();
63}

65inline bool
66is_image_type(const llvm::Type& t)
67{
68 if (t.isPointerTy() && t.getPointerElementType()->isStructTy()) {
69 llvm::StringRef name = t.getPointerElementType()->getStructName();
70 if (name.startswith("opencl.image2d_") || name.startswith("opencl.image3d_") ||
71 name.startswith("opencl.image1d_") || name.startswith("struct._pocl_image"))
72 return true;
73 }
74 return false;
75}
76
77inline bool
78is_sampler_type(const llvm::Type& t)
79{
80 if (t.isPointerTy() && t.getPointerElementType()->isStructTy())
81 {
82 llvm::StringRef name = t.getPointerElementType()->getStructName();
83 if (name.startswith("opencl.sampler_t")) return true;
84 }
85 return false;
86}
