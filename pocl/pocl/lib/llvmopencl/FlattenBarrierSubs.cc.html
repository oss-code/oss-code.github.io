62char FlattenBarrierSubs::ID = 0;
63static RegisterPass<FlattenBarrierSubs>
64 X("flatten-barrier-subs",
65 "Flatten subroutines with barriers and/or local arguments");

67
68static bool recursivelyInlineBarrierUsers(Function *F, bool ChangeInlineFlag) {
69
70 bool BarrierIsCalled = false;
71
72#ifdef DEBUG_FLATTEN_SUBS
73 std::cerr << "### FlattenBarrierSubs: SCANNING " << F->getName().str()
74 << std::endl;
75#endif
76
77 for (Function::iterator I = F->begin(), E = F->end(); I != E; ++I) {
78 for (BasicBlock::iterator BI = I->begin(), BE = I->end(); BI != BE; ++BI) {
79 Instruction *Instr = dyn_cast<Instruction>(BI);
80 if (!llvm::isa<CallInst>(Instr))
81 continue;
82 CallInst *CallInstr = dyn_cast<CallInst>(Instr);
83 Function *Callee = CallInstr->getCalledFunction();
84
85 if ((Callee == nullptr) || Callee->getName().startswith("llvm."))
86 continue;
87
88 if (llvm::isa<pocl::Barrier>(CallInstr))
89 BarrierIsCalled = true;
90 // we cannot break the loop here, since we want to inline all functions
91 // that could call a barrier, not just the first one
92 else if (recursivelyInlineBarrierUsers(Callee, true))
93 BarrierIsCalled = true;
94 }
95 }
96
97 if (ChangeInlineFlag & BarrierIsCalled) {
98 F->addFnAttr(Attribute::AlwaysInline);
99 F->setLinkage(llvm::GlobalValue::InternalLinkage);
100#ifdef DEBUG_FLATTEN_SUBS
101 std::cerr << "### FlattenBarrierSubs: AlwaysInline ENABLED on "
102 << F->getName().str() << std::endl;
103#endif
104 }
105 return BarrierIsCalled;
106}

109bool FlattenBarrierSubs::runOnModule(Module &M) {
110 bool Changed = false;
111 for (llvm::Module::iterator I = M.begin(), E = M.end(); I != E; ++I) {
112 llvm::Function *F = &*I;
113 if (F->isDeclaration())
114 continue;
115
116 if (KernelName == F->getName() || pocl::Workgroup::isKernelToProcess(*F)) {
117
118#ifdef DEBUG_FLATTEN_SUBS
119 std::cerr << "### FlattenBarrierSubs Pass running on " << KernelName
120 << std::endl;
121#endif
122 // we don't want to set alwaysInline on a Kernel, only its subroutines.
123 Changed = recursivelyInlineBarrierUsers(F, false);
124 }
125 }
126 return Changed;
127}
