94 template<bool xcompile> class TypeBuilder<PoclContext, xcompile> {
95 public:
96 static StructType *get(LLVMContext &Context) {
97 if (size_t_width == 64)
98 {
99 SmallVector<Type *, 10> Elements;
100 Elements.push_back(
101 TypeBuilder<types::i<64>[3], xcompile>::get(Context));
102 Elements.push_back(
103 TypeBuilder<types::i<64>[3], xcompile>::get(Context));
104 Elements.push_back(
105 TypeBuilder<types::i<64>[3], xcompile>::get(Context));
106 Elements.push_back(TypeBuilder<types::i<8> *, xcompile>::get(Context));
107 Elements.push_back(TypeBuilder<types::i<32> *, xcompile>::get(Context));
108 Elements.push_back(TypeBuilder<types::i<32>, xcompile>::get(Context));
109 Elements.push_back(TypeBuilder<types::i<32>, xcompile>::get(Context));
110 return StructType::get(Context, Elements);
111 }
112 else if (size_t_width == 32)
113 {
114 SmallVector<Type *, 10> Elements;
115 Elements.push_back(
116 TypeBuilder<types::i<32>[3], xcompile>::get(Context));
117 Elements.push_back(
118 TypeBuilder<types::i<32>[3], xcompile>::get(Context));
119 Elements.push_back(
120 TypeBuilder<types::i<32>[3], xcompile>::get(Context));
121 Elements.push_back(
122 TypeBuilder<types::i<8> *, xcompile>::get(Context));
123 Elements.push_back(
124 TypeBuilder<types::i<32> *, xcompile>::get(Context));
125 Elements.push_back(TypeBuilder<types::i<32>, xcompile>::get(Context));
126 Elements.push_back(
127 TypeBuilder<types::i<32>, xcompile>::get(Context));
128
129 return StructType::get(Context, Elements);
130 }
131 else
132 {
133 assert (false && "Unsupported size_t width.");
134 return NULL;
135 }
136 }

139 size_t_width = width;
140 }

142 private:
143 static int size_t_width;
144 };

165bool
166Workgroup::runOnModule(Module &M) {
167
168 this->M = &M;
169 this->C = &M.getContext();
170
171 HiddenArgs = 0;
172 SizeTWidth = currentPoclDevice->address_bits;
173 SizeT = IntegerType::get(*C, SizeTWidth);
174
175#ifdef LLVM_OLDER_THAN_7_0
176 TypeBuilder<PoclContext, true>::setSizeTWidth(SizeTWidth);
177 PoclContextT = TypeBuilder<PoclContext, true>::get(*C);
178 LauncherFuncT =
179 SizeTWidth == 32
180 ? TypeBuilder<void(types::i<8> *[], PoclContext *, types::i<32>,
181 types::i<32>, types::i<32>),
182 true>::get(M.getContext())
183 : TypeBuilder<void(types::i<8> *[], PoclContext *, types::i<64>,
184 types::i<64>, types::i<64>),
185 true>::get(M.getContext());
186#else
187 // LLVM 8.0 dropped the TypeBuilder API. This is a cleaner version
188 // anyways as it builds the context type using the SizeT directly.
189 llvm::Type *Int32T = Type::getInt32Ty(*C);
190 llvm::Type *Int8T = Type::getInt8Ty(*C);
191 PoclContextT =
192 StructType::get(
193 ArrayType::get(SizeT, 3), // NUM_GROUPS
194 ArrayType::get(SizeT, 3), // GLOBAL_OFFSET
195 ArrayType::get(SizeT, 3), // LOCAL_SIZE
196 PointerType::get(Int8T, 0), // PRINTF_BUFFER
197 PointerType::get(Int32T, 0), // PRINTF_BUFFER_POSITION
198 Int32T, // PRINTF_BUFFER_CAPACITY
199 Int32T); // WORK_DIM
200
201 LauncherFuncT = FunctionType::get(
202 Type::getVoidTy(*C),
203 {PointerType::get(PointerType::get(Type::getInt8Ty(*C), 0),
204 currentPoclDevice->args_as_id),
205 PointerType::get(PoclContextT, currentPoclDevice->context_as_id), SizeT,
206 SizeT, SizeT},
207 false);
208#endif
209
210 assert ((SizeTWidth == 64 || SizeTWidth == 32) &&
211 "Target has an unsupported pointer width.");
212
213 for (Module::iterator i = M.begin(), e = M.end(); i != e; ++i) {
214 // Don't internalize functions starting with "__wrap_" for the use of GNU
215 // linker's switch --wrap=symbol, where calls to the "symbol" are replaced
216 // with "__wrap_symbol" at link time. These functions may not be referenced
217 // until final link and being deleted by LLVM optimizations before it.
218 if (!i->isDeclaration() && !i->getName().startswith("__wrap_"))
219 i->setLinkage(Function::InternalLinkage);
220 }
221
222 // Store the new and old kernel pairs in order to regenerate
223 // all the metadata that used to point to the unmodified
224 // kernels.
225 FunctionMapping kernels;
226
227 // Mapping of all functions which have been transformed to take
228 // extra printf arguments.
229 FunctionMapping printfCache;
230
231 for (Module::iterator i = M.begin(), e = M.end(); i != e; ++i) {
232 Function &OrigKernel = *i;
233 if (!isKernelToProcess(OrigKernel)) continue;
234 Function *L = createWrapper(&OrigKernel, printfCache);
235
236 privatizeContext(L);
237
238 if (currentPoclDevice->arg_buffer_launcher) {
239 Function *WGLauncher =
240 createArgBufferWorkgroupLauncher(L, OrigKernel.getName().str());
241 L->addFnAttr(Attribute::NoInline);
242 L->removeFnAttr(Attribute::AlwaysInline);
243 WGLauncher->addFnAttr(Attribute::AlwaysInline);
244 createGridLauncher(L, WGLauncher, OrigKernel.getName().str());
245 } else if (currentPoclDevice->spmd) {
246 // For SPMD machines there is no need for a WG launcher, the device will
247 // call/handle the single-WI kernel function directly.
248 kernels[&OrigKernel] = L;
249 } else {
250 createDefaultWorkgroupLauncher(L);
251 // This is used only by TCE anymore. TODO: Replace all with the
252 // ArgBuffer one.
253 createFastWorkgroupLauncher(L);
254 }
255 }
256
257 if (!currentPoclDevice->arg_buffer_launcher && currentPoclDevice->spmd) {
258 regenerate_kernel_metadata(M, kernels);
259
260 // Delete the old kernels.
261 for (FunctionMapping::const_iterator i = kernels.begin(),
262 e = kernels.end(); i != e; ++i) {
263 Function *old_kernel = (*i).first;
264 Function *new_kernel = (*i).second;
265 if (old_kernel == new_kernel) continue;
266 old_kernel->eraseFromParent();
267 }
268 }
269 return true;
270}

272// Ensures the given value is not optimized away even if it's not used
273// by LLVM IR.
274void Workgroup::addPlaceHolder(llvm::IRBuilder<> &Builder,
275 llvm::Value *Val,
276 const std::string TypeStr="r") {
277
278 // For the lack of a better holder, add a dummy inline asm that reads the
279 // arg arguments.
280 FunctionType *DummyIAType =
281 FunctionType::get(Type::getVoidTy(M->getContext()), Val->getType(),
282 false);
283
284 llvm::InlineAsm *DummyIA =
285 llvm::InlineAsm::get(DummyIAType, "", TypeStr, false, false);
286 Builder.CreateCall(DummyIA, Val);
287}

289// Adds Range metadata with range [Min, Max] to the given instruction.
290static void addRangeMetadata(llvm::Instruction *Instr, size_t Min, size_t Max) {
291 MDBuilder MDB(Instr->getContext());
292 size_t BitWidth = Instr->getType()->getIntegerBitWidth();
293 MDNode *Range =
294 MDB.createRange(APInt(BitWidth, Min), APInt(BitWidth, Max + 1));
295 Instr->setMetadata(LLVMContext::MD_range, Range);
296}
297
298static void addRangeMetadataForPCField(llvm::Instruction *Instr,
299 int StructFieldIndex,
300 int FieldIndex = -1) {
301 uint64_t Min = 0;
302 uint64_t Max = 0;
303 uint64_t LocalSizes[] = {WGLocalSizeX, WGLocalSizeY, WGLocalSizeZ};
304 switch (StructFieldIndex) {
305 case PC_WORK_DIM:
306 Min = 1;
307 Max = currentPoclDevice->max_work_item_dimensions;
308 break;
309 case PC_NUM_GROUPS:
310 Min = 1;
311 switch (FieldIndex) {
312 case 0:
313 case 1:
314 case 2: {
315 Max = WGMaxGridDimWidth > 0 ? WGMaxGridDimWidth : 0;
316 if (!WGDynamicLocalSize) {
317 // If we know also the local size, we can minimize the known max group
318 // count by dividing by it. Upwards rounding due to 2.0 partial WGs.
319 Max = (Max + LocalSizes[FieldIndex] - 1) / LocalSizes[FieldIndex];
320 }
321 break;
322 }
323 default:
324 llvm_unreachable("More than 3 grid dimensions unsupported.");
325 }
326 break;
327 case PC_GLOBAL_OFFSET:
328 switch (FieldIndex) {
329 case 0:
330 case 1:
331 case 2:
332 // WGAssumeZeroGlobalOffset will be used to convert to a constant 0, so
333 // here we just speculate on the range in case of non-zero offset.
334 Max = WGMaxGridDimWidth;
335 break;
336 default:
337 llvm_unreachable("More than 3 grid dimensions unsupported.");
338 }
339 break;
340 case PC_LOCAL_SIZE:
341 Min = 1;
342 switch (FieldIndex) {
343 case 0:
344 case 1:
345 case 2:
346 if (WGDynamicLocalSize) {
347 Max = (WGMaxGridDimWidth > 0
348 ? WGMaxGridDimWidth
349 : min(currentPoclDevice->max_work_item_sizes[FieldIndex],
350 WGMaxGridDimWidth));
351 } else {
352 // The local size is converted to constant with static WGs, so this is
353 // actually useless.
354 Max = LocalSizes[FieldIndex];
355 }
356 break;
357 default:
358 llvm_unreachable("More than 3 grid dimensions unsupported.");
359 }
360 break;
361 default:
362 break;
363 }
364 if (Max > 0) {
365 addRangeMetadata(Instr, Min, Max);
366#if 0
367 std::cerr << "Added range [" << Min << ", " << Max << "] " << std::endl;
368 std::cerr << StructFieldIndex << " " << FieldIndex << std::endl;
369#endif
370 }
371 return;
372}

374// Creates a load from the hidden context structure argument for
375// the given element.
376llvm::Value *
377Workgroup::createLoadFromContext(
378 IRBuilder<> &Builder, int StructFieldIndex, int FieldIndex=-1) {
379
380 Value *GEP;
381 GEP = Builder.CreateStructGEP(ContextArg->getType()->getPointerElementType(),
382 ContextArg, StructFieldIndex);
383
384 llvm::LoadInst *Load = nullptr;
385 if (SizeTWidth == 64) {
386 if (FieldIndex == -1)
387 Load = Builder.CreateLoad(Builder.CreateConstGEP1_64(GEP, 0));
388 else
389 Load = Builder.CreateLoad(Builder.CreateConstGEP2_64(GEP, 0, FieldIndex));
390 } else {
391 if (FieldIndex == -1)
392 Load = Builder.CreateLoad(Builder.CreateConstGEP1_32(GEP, 0));
393 else
394 Load = Builder.CreateLoad(Builder.CreateConstGEP2_32(
395 GEP->getType()->getPointerElementType(), GEP, 0, FieldIndex));
396 }
397 addRangeMetadataForPCField(Load, StructFieldIndex, FieldIndex);
398 return Load;
399}

401// TODO we should use printf users instead of searching the call tree
402static bool callsPrintf(Function *F) {
403 for (Function::iterator I = F->begin(), E = F->end(); I != E; ++I) {
404 for (BasicBlock::iterator BI = I->begin(), BE = I->end(); BI != BE; ++BI) {
405 Instruction *Instr = dyn_cast<Instruction>(BI);
406 if (!llvm::isa<CallInst>(Instr))
407 continue;
408 CallInst *CallInstr = dyn_cast<CallInst>(Instr);
409 Function *callee = CallInstr->getCalledFunction();
410
411 if (callee->getName().startswith("llvm."))
412 continue;
413 if (callee->getName().equals("printf"))
414 return true;
415 if (callee->getName().equals("__pocl_printf"))
416 return true;
417 if (callsPrintf(callee))
418 return true;
419 }
420 }
421 return false;
422}

424// Clones a function while adding 3 new arguments for printf calls.
425static Function *cloneFunctionWithPrintfArgs(Value *pb, Value *pbp, Value *pbc,
426 Function *F, Module *M) {
427
428 SmallVector<Type *, 8> Parameters;
429
430 Parameters.push_back(pb->getType());
431 Parameters.push_back(pbp->getType());
432 Parameters.push_back(pbc->getType());
433
434 for (Function::const_arg_iterator i = F->arg_begin(), e = F->arg_end();
435 i != e; ++i)
436 Parameters.push_back(i->getType());
437
438 // Create the new function.
439 FunctionType *FT =
440 FunctionType::get(F->getReturnType(), Parameters, F->isVarArg());
441 Function *NewF = Function::Create(FT, F->getLinkage(), "", M);
442 NewF->takeName(F);
443
444 ValueToValueMapTy VV;
445 Function::arg_iterator j = NewF->arg_begin();
446 j->setName("print_buffer");
447 ++j;
448 j->setName("print_buffer_position");
449 ++j;
450 j->setName("print_buffer_capacity");
451 ++j;
452 for (Function::const_arg_iterator i = F->arg_begin(), e = F->arg_end();
453 i != e; ++i) {
454 j->setName(i->getName());
455 VV[&*i] = &*j;
456 ++j;
457 }
458
459 SmallVector<ReturnInst *, 1> RI;
460
461 // As of LLVM 5.0 we need to let CFI to make module level changes,
462 // otherwise there will be an assertion. The changes are likely
463 // additional debug info nodes added when cloning the function into
464 // the other. For some reason it doesn't want to reuse the old ones.
465 CloneFunctionInto(NewF, F, VV, true, RI);
466
467 return NewF;
468}

470// Recursively replace _cl_printf calls with _pocl_printf calls, while
471// propagating the required pocl_context->printf_buffer arguments.
472static void replacePrintfCalls(Value *pb, Value *pbp, Value *pbc, bool isKernel,
473 Function *poclPrintf, Module &M, Function *L,
474 FunctionMapping &printfCache) {
475
476 // If none of the kernels use printf(), it will not be linked into the
477 // module.
478 if (poclPrintf == nullptr)
479 return;
480
481 // For kernel function, we are provided with proper printf arguments;
482 // for non-kernel functions, we assume the function was replaced with
483 // cloneFunctionWithPrintfArgs() and use the first three arguments.
484 if (!isKernel) {
485 auto i = L->arg_begin();
486 pb = &*i;
487 ++i;
488 pbp = &*i;
489 ++i;
490 pbc = &*i;
491 }
492
493 SmallDenseMap<CallInst *, CallInst *> replaceCIMap(16);
494 SmallVector<Value *, 8> ops;
495 SmallVector<CallInst *, 32> callsToCheck;
496
497 // First, replace printf calls in body of L.
498 for (Function::iterator I = L->begin(), E = L->end(); I != E; ++I) {
499 for (BasicBlock::iterator BI = I->begin(), BE = I->end(); BI != BE; ++BI) {
500 Instruction *Instr = dyn_cast<Instruction>(BI);
501 if (!llvm::isa<CallInst>(Instr))
502 continue;
503 CallInst *CallInstr = dyn_cast<CallInst>(Instr);
504 Function *oldF = CallInstr->getCalledFunction();
505
506 // Skip inline asm blocks.
507 if (oldF == nullptr)
508 continue;
509
510 if (oldF->getName().equals("printf")) {
511 ops.clear();
512 ops.push_back(pb);
513 ops.push_back(pbp);
514 ops.push_back(pbc);
515
516 unsigned j = CallInstr->getNumOperands() - 1;
517 for (unsigned i = 0; i < j; ++i)
518 ops.push_back(CallInstr->getOperand(i));
519
520 CallSite CS(CallInstr);
521 CallInst *NewCI = CallInst::Create(poclPrintf, ops);
522 NewCI->setCallingConv(poclPrintf->getCallingConv());
523 NewCI->setTailCall(CS.isTailCall());
524
525 replaceCIMap.insert(
526 std::pair<CallInst *, CallInst *>(CallInstr, NewCI));
527 } else {
528 if (!oldF->getName().startswith("llvm."))
529 callsToCheck.push_back(CallInstr);
530 }
531 }
532 }
533
534 // replace printf calls
535 for (auto it : replaceCIMap) {
536 CallInst *CI = it.first;
537 CallInst *newCI = it.second;
538
539 // LLVM may modify the result type of the called function to void.
540 if (CI->getType()->isVoidTy()) {
541 newCI->insertBefore(CI);
542 CI->eraseFromParent();
543 } else {
544 CI->replaceAllUsesWith(newCI);
545 ReplaceInstWithInst(CI, newCI);
546 }
547 }
548
549 replaceCIMap.clear();
550
551 // check each called function recursively
552 for (auto it : callsToCheck) {
553 CallInst *CI = it;
554 CallInst *NewCI = nullptr;
555 Function *oldF = CI->getCalledFunction();
556 Function *newF = nullptr;
557 bool needsPrintf = false;
558 auto i = printfCache.find(oldF);
559 if (i != printfCache.end()) {
560 // function was already cloned
561 needsPrintf = true;
562 newF = i->second;
563 } else {
564 // create new clone
565 needsPrintf = callsPrintf(oldF);
566 if (needsPrintf) {
567 newF = cloneFunctionWithPrintfArgs(pb, pbp, pbc, oldF, &M);
568 replacePrintfCalls(nullptr, nullptr, nullptr, false, poclPrintf, M,
569 newF, printfCache);
570
571 printfCache.insert(
572 std::pair<llvm::Function *, llvm::Function *>(oldF, newF));
573 }
574 }
575
576 // if the called function calls Printf, replace with newF and add arguments
577 if (needsPrintf) {
578 ops.clear();
579 ops.push_back(pb);
580 ops.push_back(pbp);
581 ops.push_back(pbc);
582 unsigned j = CI->getNumOperands() - 1;
583 for (unsigned i = 0; i < j; ++i)
584 ops.push_back(CI->getOperand(i));
585
586 NewCI = CallInst::Create(newF, ops);
587 replaceCIMap.insert(std::pair<CallInst *, CallInst *>(CI, NewCI));
588 }
589 }
590
591 for (auto it : replaceCIMap) {
592 CallInst *CI = it.first;
593 CallInst *newCI = it.second;
594
595 CI->replaceAllUsesWith(newCI);
596 ReplaceInstWithInst(CI, newCI);
597 }
598}

600// Create a wrapper for the kernel and add pocl-specific hidden arguments.
601// Also inlines the wrapped function to the wrapper.
602Function *
603Workgroup::createWrapper(Function *F, FunctionMapping &printfCache) {
604
605 SmallVector<Type *, 8> sv;
606 LLVMContext &C = M->getContext();
607 for (Function::const_arg_iterator i = F->arg_begin(), e = F->arg_end();
608 i != e; ++i)
609 sv.push_back(i->getType());
610
611 if (!currentPoclDevice->arg_buffer_launcher && currentPoclDevice->spmd) {
612 sv.push_back(
613 PointerType::get(PoclContextT, currentPoclDevice->context_as_id));
614 HiddenArgs = 1;
615 } else {
616 // pocl_context
617 sv.push_back(
618 PointerType::get(PoclContextT, currentPoclDevice->context_as_id));
619 // group_x
620 sv.push_back(SizeT);
621 // group_y
622 sv.push_back(SizeT);
623 // group_z
624 sv.push_back(SizeT);
625
626 // we might not have all of the globals anymore in the module in case the
627 // kernel does not refer to them and they are optimized away
628 HiddenArgs = 4;
629
630 }
631
632 FunctionType *ft =
633 FunctionType::get(Type::getVoidTy(C), ArrayRef<Type *>(sv), false);
634
635 std::string funcName = "";
636 funcName = F->getName().str();
637 Function *L = NULL;
638 if (!currentPoclDevice->arg_buffer_launcher && currentPoclDevice->spmd) {
639 Function *F = M->getFunction(funcName);
640 F->setName(funcName + "_original");
641 L = Function::Create(ft, Function::ExternalLinkage, funcName, M);
642 } else
643 L = Function::Create(ft, Function::ExternalLinkage,
644 "_pocl_kernel_" + funcName, M);
645
646 SmallVector<Value *, 8> arguments;
647 Function::arg_iterator ai = L->arg_begin();
648 for (unsigned i = 0, e = F->arg_size(); i != e; ++i) {
649 arguments.push_back(&*ai);
650 ++ai;
651 }
652
653 ContextArg = &*(ai++);
654 GroupIdArgs.resize(3);
655 GroupIdArgs[0] = &*(ai++);
656 GroupIdArgs[1] = &*(ai++);
657 GroupIdArgs[2] = &*(ai++);
658
659 // Copy the function attributes to transfer noalias etc. from the
660 // original kernel which will be inlined into the launcher.
661 L->setAttributes(F->getAttributes());
662
663 // At least the argument address space metadata is useful. The argument
664 // indices should still hold even though we appended the hidden args.
665 L->copyMetadata(F, 0);
666 // We need to mark the generated function to avoid it being considered a
667 // new kernel to process (which results in infinite recursion). This is
668 // because kernels are detected by the presense of the argument metadata
669 // we just copied from the original kernel function.
670 L->setMetadata("pocl_generated", MDNode::get(C, {createConstantIntMD(C, 1)}));
671
672 IRBuilder<> Builder(BasicBlock::Create(C, "", L));
673
674 Value *pb, *pbp, *pbc;
675 if (currentPoclDevice->device_side_printf) {
676 pb = createLoadFromContext(Builder, PC_PRINTF_BUFFER);
677 pbp = createLoadFromContext(Builder, PC_PRINTF_BUFFER_POSITION);
678 pbc = createLoadFromContext(Builder, PC_PRINTF_BUFFER_CAPACITY);
679 } else {
680 pb = pbp = pbc = nullptr;
681 }
682
683 CallInst *c = Builder.CreateCall(F, ArrayRef<Value*>(arguments));
684 Builder.CreateRetVoid();
685
686 std::set<CallInst *> CallsToRemove;
687
688 // At least with LLVM 4.0, the runtime of AddAliasScopeMetadata of
689 // llvm::InlineFunction explodes in case of kernels with restrict
690 // metadata and a lot of lifetime markers. The issue produces at
691 // least with EinsteinToolkit which has a lot of restrict kernel
692 // args). Remove them here before inlining to speed it up.
693 // TODO: Investigate the root cause.
694
695 for (Function::iterator I = F->begin(), E = F->end(); I != E; ++I) {
696 for (BasicBlock::iterator BI = I->begin(), BE = I->end(); BI != BE; ++BI) {
697 Instruction *Instr = dyn_cast<Instruction>(BI);
698 if (!llvm::isa<CallInst>(Instr)) continue;
699 CallInst *CallInstr = dyn_cast<CallInst>(Instr);
700 Function *Callee = CallInstr->getCalledFunction();
701 // At least with LLVM 4.0, the runtime of AddAliasScopeMetadata of
702 // llvm::InlineFunction explodes in case of kernels with restrict
703 // metadata and a lot of lifetime markers. The issue produces at
704 // least with EinsteinToolkit which has a lot of restrict kernel
705 // args). Remove them here before inlining to speed it up.
706 // TODO: Investigate the root cause.
707 if (Callee != nullptr &&
708 (Callee->getName().startswith("llvm.lifetime.end") ||
709 Callee->getName().startswith("llvm.lifetime.start"))) {
710 CallsToRemove.insert(CallInstr);
711 continue;
712 }
713 }
714 }
715
716 for (auto C : CallsToRemove) {
717 C->eraseFromParent();
718 }
719
720 // needed for printf
721 InlineFunctionInfo IFI;
722 InlineFunction(c, IFI);
723
724 if (currentPoclDevice->device_side_printf) {
725 Function *poclPrintf = M->getFunction("__pocl_printf");
726 replacePrintfCalls(pb, pbp, pbc, true, poclPrintf, *M, L, printfCache);
727 }
728
729 L->setSubprogram(F->getSubprogram());
730
731 // SPMD machines might need a special calling convention to mark the
732 // kernels that should be executed in SPMD fashion. For MIMD/CPU,
733 // we want to use the default calling convention for the work group
734 // function.
735 if (currentPoclDevice->spmd)
736 L->setCallingConv(F->getCallingConv());
737
738 return L;
739}

741// Converts the given global context variable handles to loads from the
742// hidden context struct argument. If there is no reference to the global,
743// the corresponding entry in the returned vector will contain a nullptr.
744std::vector<llvm::Value*>
745Workgroup::globalHandlesToContextStructLoads(
746 IRBuilder<> &Builder,
747 const std::vector<std::string> &&GlobalHandleNames,
748 int StructFieldIndex) {
749
750 std::vector<Value*> StructLoads(GlobalHandleNames.size());
751 for (size_t i = 0; i < GlobalHandleNames.size(); ++i) {
752 if (M->getGlobalVariable(GlobalHandleNames.at(i)) == nullptr) {
753 StructLoads[i] = nullptr;
754 continue;
755 }
756 StructLoads[i] = createLoadFromContext(
757 Builder, StructFieldIndex, GlobalHandleNames.size() == 1 ? -1 : i);

762// Converts uses of the given variable handles (external global variables) to
763// use the given function-private values instead.
764void
765Workgroup::privatizeGlobals(llvm::Function *F, llvm::IRBuilder<> &Builder,
766 const std::vector<std::string> &&GlobalHandleNames,
767 std::vector<llvm::Value*> PrivateValues) {
768
769 for (Function::iterator i = F->begin(), e = F->end();
770 i != e; ++i) {
771 for (BasicBlock::iterator ii = i->begin(), ee = i->end(),
772 Next = std::next(ii); ii != ee; ii = Next) {
773 Next = std::next(ii);
774 for (size_t j = 0; j < GlobalHandleNames.size(); ++j) {
775 if (PrivateValues[j] == nullptr) {
776 continue;
777 }
778 if (!isa<llvm::LoadInst>(ii)) {
779 continue;
780 }
781 llvm::LoadInst *L = cast<llvm::LoadInst>(ii);
782 llvm::GlobalValue *GlobalHandle =
783 M->getGlobalVariable(GlobalHandleNames.at(j));
784
785 if (GlobalHandle == nullptr)
786 continue;
787
788 if (L->getPointerOperand()->stripPointerCasts() != GlobalHandle)
789 continue;
790
791 llvm::Value *Cast =
792 Builder.CreateTruncOrBitCast(PrivateValues[j], L->getType());
793 ii->replaceAllUsesWith(Cast);
794 ii->eraseFromParent();
795 break;
796 }
797 }
798 }

801void
802Workgroup::privatizeContext(Function *F)
803{
804 char TempStr[STRING_LENGTH];
805 IRBuilder<> Builder(F->getEntryBlock().getFirstNonPHI());
806
807 std::vector<GlobalVariable*> LocalIdGlobals(3);
808 std::vector<AllocaInst*> LocalIdAllocas(3);
809 // Privatize _local_id to allocas. They are used as iteration variables in
810 // WorkItemLoops, thus referred to later on.
811 for (int i = 0; i < 3; ++i) {
812 snprintf(TempStr, STRING_LENGTH, "_local_id_%c", 'x' + i);
813 LocalIdGlobals[i] = M->getGlobalVariable(TempStr);
814 if (LocalIdGlobals[i] != NULL) {
815 LocalIdAllocas[i] =
816 Builder.CreateAlloca(LocalIdGlobals[i]->getType()->getElementType(), 0,
817 TempStr);
818 if (LocalIdGlobals[i]->hasInitializer()) {
819 Constant *C = LocalIdGlobals[i]->getInitializer();
820 Builder.CreateStore(C, LocalIdAllocas[i]);
821 }
822 }
823 }
824 for (Function::iterator i = F->begin(), e = F->end(); i != e; ++i) {
825 for (BasicBlock::iterator ii = i->begin(), ee = i->end();
826 ii != ee; ++ii) {
827 for (int j = 0; j < 3; ++j)
828 ii->replaceUsesOfWith(LocalIdGlobals[j], LocalIdAllocas[j]);
829 }
830 }
831
832 std::vector<GlobalVariable*> LocalSizeGlobals(3, nullptr);
833 std::vector<AllocaInst*> LocalSizeAllocas(3, nullptr);
834 // Privatize _local_size* to private allocas.
835 // They are referred to by WorkItemLoops to fetch the WI loop bounds.
836 for (int i = 0; i < 3; ++i) {
837 snprintf(TempStr, STRING_LENGTH, "_local_size_%c", 'x' + i);
838 LocalSizeGlobals[i] = M->getGlobalVariable(TempStr);
839 if (LocalSizeGlobals[i] != NULL) {
840 LocalSizeAllocas[i] =
841 Builder.CreateAlloca(LocalSizeGlobals[i]->getType()->getElementType(),
842 0, TempStr);
843 if (LocalSizeGlobals[i]->hasInitializer()) {
844 Constant *C = LocalSizeGlobals[i]->getInitializer();
845 Builder.CreateStore(C, LocalSizeAllocas[i]);
846 }
847 }
848 }
849 for (Function::iterator i = F->begin(), e = F->end(); i != e; ++i) {
850 for (BasicBlock::iterator ii = i->begin(), ee = i->end();
851 ii != ee; ++ii) {
852 for (int j = 0; j < 3; ++j)
853 ii->replaceUsesOfWith(LocalSizeGlobals[j], LocalSizeAllocas[j]);
854 }
855 }
856
857 if (WGDynamicLocalSize) {
858 if (LocalSizeAllocas[0] != nullptr)
859 Builder.CreateStore(
860 createLoadFromContext(Builder, PC_LOCAL_SIZE, 0),
861 LocalSizeAllocas[0]);
862
863 if (LocalSizeAllocas[1] != nullptr)
864 Builder.CreateStore(
865 createLoadFromContext(Builder, PC_LOCAL_SIZE, 1),
866 LocalSizeAllocas[1]);
867
868 if (LocalSizeAllocas[2] != nullptr)
869 Builder.CreateStore(
870 createLoadFromContext(Builder, PC_LOCAL_SIZE, 2),
871 LocalSizeAllocas[2]);
872 } else {
873 if (LocalSizeAllocas[0] != nullptr)
874 Builder.CreateStore(
875 ConstantInt::get(
876 LocalSizeAllocas[0]->getAllocatedType(), WGLocalSizeX, 0),
877 LocalSizeAllocas[0]);
878
879 if (LocalSizeAllocas[1] != nullptr)
880 Builder.CreateStore(
881 ConstantInt::get(
882 LocalSizeAllocas[1]->getAllocatedType(), WGLocalSizeY, 1),
883 LocalSizeAllocas[1]);
884
885 if (LocalSizeAllocas[2] != nullptr)
886 Builder.CreateStore(
887 ConstantInt::get(
888 LocalSizeAllocas[2]->getAllocatedType(), WGLocalSizeZ, 2),
889 LocalSizeAllocas[2]);
890 }
891
892 privatizeGlobals(
893 F, Builder, {"_group_id_x", "_group_id_y", "_group_id_z"}, GroupIdArgs);
894
895 if (WGAssumeZeroGlobalOffset) {
896 privatizeGlobals(
897 F, Builder,
898 {"_global_offset_x", "_global_offset_y", "_global_offset_z"},
899 {ConstantInt::get(SizeT, 0), ConstantInt::get(SizeT, 0),
900 ConstantInt::get(SizeT, 0)});
901 } else {
902 privatizeGlobals(
903 F, Builder,
904 {"_global_offset_x", "_global_offset_y", "_global_offset_z"},
905 globalHandlesToContextStructLoads(
906 Builder,
907 {"_global_offset_x", "_global_offset_y", "_global_offset_z"},
908 PC_GLOBAL_OFFSET));
909 }
910
911 privatizeGlobals(
912 F, Builder, {"_work_dim"},
913 globalHandlesToContextStructLoads(Builder, {"_work_dim"}, PC_WORK_DIM));
914
915 privatizeGlobals(
916 F, Builder, {"_num_groups_x", "_num_groups_y", "_num_groups_z"},
917 globalHandlesToContextStructLoads(
918 Builder, {"_num_groups_x", "_num_groups_y", "_num_groups_z"},
919 PC_NUM_GROUPS));
920
921 if (currentPoclDevice->device_side_printf) {
922 // Privatize _printf_buffer
923 privatizeGlobals(
924 F, Builder, {"_printf_buffer"}, {
925 createLoadFromContext(
926 Builder, PC_PRINTF_BUFFER)});
927
928 privatizeGlobals(
929 F, Builder, {"_printf_buffer_position"}, {
930 createLoadFromContext(
931 Builder, PC_PRINTF_BUFFER_POSITION)});
932
933 privatizeGlobals(
934 F, Builder, {"_printf_buffer_capacity"}, {
935 createLoadFromContext(
936 Builder, PC_PRINTF_BUFFER_CAPACITY)});
937 }

940// Creates a work group launcher function (called KERNELNAME_workgroup)
941// that assumes kernel pointer arguments are stored as pointers to the
942// actual buffers and that scalar data is loaded from the default memory.
943void
944Workgroup::createDefaultWorkgroupLauncher(llvm::Function *F) {
945
946 IRBuilder<> Builder(M->getContext());
947
948 std::string FuncName = "";
949 FuncName = F->getName().str();
950
951#ifdef LLVM_OLDER_THAN_9_0
952 Function *WorkGroup =
953 dyn_cast<Function>(M->getOrInsertFunction(
954 FuncName + "_workgroup", LauncherFuncT));
955#else
956 FunctionCallee fc = M->getOrInsertFunction(FuncName + "_workgroup", LauncherFuncT);
957 Function *WorkGroup = dyn_cast<Function>(fc.getCallee());
958#endif
959
960 assert(WorkGroup != nullptr);
961 BasicBlock *Block = BasicBlock::Create(M->getContext(), "", WorkGroup);
962 Builder.SetInsertPoint(Block);
963
964 Function::arg_iterator ai = WorkGroup->arg_begin();
965
966 SmallVector<Value *, 8> Arguments;
967 size_t i = 0;
968 for (Function::const_arg_iterator ii = F->arg_begin(), ee = F->arg_end();
969 ii != ee; ++ii) {
970
971 if (i == F->arg_size() - 4)
972 break;
973
974 Type *ArgType = ii->getType();
975
976 Value *GEP = Builder.CreateGEP(
977 &*ai, ConstantInt::get(IntegerType::get(M->getContext(), 32), i));
978 Value *Pointer = Builder.CreateLoad(GEP);
979
980 Value *Arg;
981 if (currentPoclDevice->device_alloca_locals &&
982 isLocalMemFunctionArg(F, i)) {
983 // Generate allocas for the local buffer arguments.
984 PointerType *ParamType = dyn_cast<PointerType>(ArgType);
985 Type *ArgElementType = ParamType->getElementType();
986 if (ArgElementType->isArrayTy()) {
987 // Known static local size (converted automatic local).
988 Arg =
989 new llvm::AllocaInst(ArgElementType, ParamType->getAddressSpace(),
990 ConstantInt::get(IntegerType::get(*C, 32), 1),
991 MAX_EXTENDED_ALIGNMENT, "local_auto", Block);
992 } else {
993 // Dynamic (runtime-set) size local argument.
994
995 const DataLayout &DL = M->getDataLayout();
996 // The size is passed directly instead of the pointer.
997 uint64_t ParamByteSize = DL.getTypeStoreSize(ParamType);
998 uint64_t ElementSize = DL.getTypeStoreSize(ArgElementType);
999 Type *SizeIntType = IntegerType::get(*C, ParamByteSize * 8);
1000 Value *LocalArgByteSize =
1001 Builder.CreatePointerCast(Pointer, SizeIntType);
1002 Value *ElementCount = Builder.CreateUDiv(
1003 LocalArgByteSize, ConstantInt::get(SizeIntType, ElementSize));
1004 Arg = new llvm::AllocaInst(ArgElementType, ParamType->getAddressSpace(),
1005 ElementCount, MAX_EXTENDED_ALIGNMENT,
1006 "local_arg", Block);
1007 }
1008 } else {
1009 // If it's a pass by value pointer argument, we just pass the pointer
1010 // as is to the function, no need to load from it first.
1011 if (ii->hasByValAttr()) {
1012 Arg = Builder.CreatePointerCast(Pointer, ArgType);
1013 } else {
1014 Arg = Builder.CreatePointerCast(Pointer, ArgType->getPointerTo());
1015 Arg = Builder.CreateLoad(Arg);
1016 }
1017 }
1018
1019 Arguments.push_back(Arg);
1020 ++i;
1021 }
1022
1023 ++ai;
1024 Arguments.push_back(&*ai);
1025 ++ai;
1026 Arguments.push_back(&*ai);
1027 ++ai;
1028 Arguments.push_back(&*ai);
1029 ++ai;
1030 Arguments.push_back(&*ai);
1031
1032 Builder.CreateCall(F, ArrayRef<Value *>(Arguments));
1033 Builder.CreateRetVoid();
1034}

1036static inline uint64_t
1037align64(uint64_t value, unsigned alignment)
1038{
1039 return (value + alignment - 1) & ~((uint64_t)alignment - 1);
1040}

1042static bool isByValPtrArgument(llvm::Argument &Arg) {
1043 return Arg.getType()->isPointerTy() && Arg.hasByValAttr();
1044}

1046static size_t getArgumentSize(llvm::Argument &Arg) {
1047 llvm::Type *TypeInBuf = nullptr;
1048 if (Arg.getType()->isPointerTy()) {
1049 if (Arg.hasByValAttr()) {
1050 TypeInBuf = Arg.getType()->getPointerElementType();
1051 } else {
1052 TypeInBuf = Arg.getType();
1053 }
1054 } else {
1055 // Scalar argument.
1056 TypeInBuf = Arg.getType();
1057 }
1058
1059 const DataLayout &DL = Arg.getParent()->getParent()->getDataLayout();
1060 return DL.getTypeStoreSize(TypeInBuf);
1061}

1063static void computeArgBufferOffsets(LLVMValueRef F,
1064 uint64_t *ArgBufferOffsets) {
1065
1066 uint64_t Offset = 0;
1067 uint64_t ArgCount = LLVMCountParams(F);
1068
1069 // Compute the byte offsets of arguments in the arg buffer.
1070 for (size_t i = 0; i < ArgCount; i++) {
1071 LLVMValueRef Param = LLVMGetParam(F, i);
1072 // TODO: This is a target specific type? We would like to get the
1073 // natural size or the "packed size" instead...
1074 uint64_t ByteSize = getArgumentSize(cast<Argument>(*unwrap(Param)));
1075 uint64_t Alignment = ByteSize;
1076
1077 assert(ByteSize > 0 && "Arg type size is zero?");
1078 Offset = align64(Offset, Alignment);
1079 ArgBufferOffsets[i] = Offset;
1080 Offset += ByteSize;
1081 }
1082}

1084static LLVMValueRef
1085createArgBufferLoad(LLVMBuilderRef Builder, LLVMValueRef ArgBufferPtr,
1086 uint64_t *ArgBufferOffsets, LLVMValueRef F,
1087 unsigned ParamIndex) {
1088
1089 LLVMValueRef Param = LLVMGetParam(F, ParamIndex);
1090 LLVMTypeRef ParamType = LLVMTypeOf(Param);
1091
1092 LLVMModuleRef M = LLVMGetGlobalParent(F);
1093 LLVMContextRef LLVMContext = LLVMGetModuleContext(M);
1094
1095 uint64_t ArgPos = ArgBufferOffsets[ParamIndex];
1096 LLVMValueRef Offs =
1097 LLVMConstInt(LLVMInt32TypeInContext(LLVMContext), ArgPos, 0);
1098 LLVMValueRef ArgByteOffset =
1099 LLVMBuildGEP(Builder, ArgBufferPtr, &Offs, 1, "arg_byte_offset");
1100
1101 if (isByValPtrArgument(cast<Argument>(*unwrap(Param)))) {
1102 // In case of byval arguments (private structs), the struct
1103 // is in the arg buffer directly. Just refer to its address.
1104 return LLVMBuildPointerCast(Builder, ArgByteOffset, ParamType,
1105 "inval_arg_ptr");
1106 } else {
1107 LLVMValueRef ArgOffsetBitcast = LLVMBuildPointerCast(
1108 Builder, ArgByteOffset, LLVMPointerType(ParamType, 0), "arg_ptr");
1109 return LLVMBuildLoad(Builder, ArgOffsetBitcast, "");
1110 }

1113/**
1114 * Creates a work group launcher with all the argument data passed
1115 * in a single argument buffer.
1116 *
1117 * All argument values, including pointers are stored directly in the
1118 * argument buffer with natural alignment. The rules for populating the
1119 * buffer are those of the HSA kernel calling convention. The name of
1120 * the generated function is KERNELNAME_workgroup_argbuffer.
1121 */
1122Function*
1123Workgroup::createArgBufferWorkgroupLauncher(Function *Func,
1124 std::string KernName) {
1125
1126 LLVMValueRef F = wrap(Func);
1127 uint64_t ArgCount = LLVMCountParams(F);
1128 uint64_t ArgBufferOffsets[ArgCount];
1129 LLVMModuleRef M = wrap(this->M);
1130
1131 computeArgBufferOffsets(F, ArgBufferOffsets);
1132
1133 LLVMContextRef LLVMContext = LLVMGetModuleContext(M);
1134
1135 LLVMTypeRef Int8Type = LLVMInt8TypeInContext(LLVMContext);
1136 LLVMTypeRef Int32Type = LLVMInt32TypeInContext(LLVMContext);
1137 LLVMTypeRef Int64Type = LLVMInt64TypeInContext(LLVMContext);
1138
1139 LLVMTypeRef ArgsPtrType =
1140 LLVMPointerType(Int8Type, currentPoclDevice->args_as_id);
1141
1142 LLVMTypeRef CtxPtrType =
1143 LLVMPointerType(Int8Type, currentPoclDevice->context_as_id);
1144
1145 std::ostringstream StrStr;
1146 StrStr << KernName;
1147 StrStr << "_workgroup_argbuffer";
1148
1149 std::string FName = StrStr.str();
1150 const char *FunctionName = FName.c_str();
1151
1152 LLVMTypeRef LauncherArgTypes[] = {
1153 ArgsPtrType, // args
1154 CtxPtrType, // pocl_ctx
1155 wrap(SizeT), // group_x
1156 wrap(SizeT), // group_y
1157 wrap(SizeT), // group_z
1158 };
1159
1160 LLVMTypeRef VoidType = LLVMVoidTypeInContext(LLVMContext);
1161 LLVMTypeRef LauncherFuncType =
1162 LLVMFunctionType(VoidType, LauncherArgTypes, 1 + HiddenArgs, 0);
1163
1164 LLVMValueRef WrapperKernel =
1165 LLVMAddFunction(M, FunctionName, LauncherFuncType);
1166
1167 LLVMBasicBlockRef Block =
1168 LLVMAppendBasicBlockInContext(LLVMContext, WrapperKernel, "entry");
1169
1170 LLVMBuilderRef Builder = LLVMCreateBuilderInContext(LLVMContext);
1171 assert(Builder);
1172
1173 LLVMPositionBuilderAtEnd(Builder, Block);
1174
1175 LLVMValueRef Args[ArgCount];
1176 LLVMValueRef ArgBuffer = LLVMGetParam(WrapperKernel, 0);
1177 size_t i = 0;
1178 for (; i < ArgCount - HiddenArgs + 1; ++i) {
1179
1180 if (currentPoclDevice->device_alloca_locals &&
1181 isLocalMemFunctionArg(Func, i)) {
1182
1183 // Generate allocas for the local buffer arguments.
1184
1185 LLVMValueRef Param = LLVMGetParam(F, i);
1186 LLVMTypeRef ParamType = LLVMTypeOf(Param);
1187
1188 LLVMTargetDataRef DataLayout = LLVMGetModuleDataLayout(M);
1189
1190 LLVMTypeRef ArgElementType = LLVMGetElementType(ParamType);
1191 LLVMValueRef LocalArgAlloca = nullptr;
1192
1193 if (LLVMGetTypeKind(ArgElementType) == LLVMArrayTypeKind) {
1194
1195 // Known static local size (converted automatic local).
1196 LocalArgAlloca = wrap(new llvm::AllocaInst(
1197 unwrap(ArgElementType), LLVMGetPointerAddressSpace(ParamType),
1198 unwrap(LLVMConstInt(Int32Type, 1, 0)), MAX_EXTENDED_ALIGNMENT,
1199 "local_auto", unwrap(Block)));
1200 } else {
1201
1202 // Dynamic (runtime-set) size local argument.
1203
1204 uint64_t ParamByteSize = LLVMStoreSizeOfType(DataLayout, ParamType);
1205 LLVMTypeRef ParamIntType = ParamByteSize == 4 ? Int32Type : Int64Type;
1206
1207 uint64_t ArgPos = ArgBufferOffsets[i];
1208 LLVMValueRef Offs = LLVMConstInt(Int32Type, ArgPos, 0);
1209 LLVMValueRef SizeByteOffset =
1210 LLVMBuildGEP(Builder, ArgBuffer, &Offs, 1, "size_byte_offset");
1211 LLVMValueRef SizeOffsetBitcast =
1212 LLVMBuildPointerCast(Builder, SizeByteOffset,
1213 LLVMPointerType(ParamIntType, 0), "size_ptr");
1214
1215 // The buffer size passed from the runtime is a byte size, we
1216 // need to convert it to an element count for the alloca.
1217 LLVMValueRef LocalArgByteSize =
1218 LLVMBuildLoad(Builder, SizeOffsetBitcast, "byte_size");
1219 uint64_t ElementSize = LLVMStoreSizeOfType(DataLayout, ArgElementType);
1220 LLVMValueRef ElementCount =
1221 LLVMBuildUDiv(Builder, LocalArgByteSize,
1222 LLVMConstInt(ParamIntType, ElementSize, 0), "");
1223 LocalArgAlloca = wrap(new llvm::AllocaInst(
1224 unwrap(LLVMGetElementType(ParamType)),
1225 LLVMGetPointerAddressSpace(ParamType), unwrap(ElementCount),
1226 MAX_EXTENDED_ALIGNMENT, "local_arg", unwrap(Block)));
1227 }
1228 Args[i] = LocalArgAlloca;
1229 } else {
1230 Args[i] = createArgBufferLoad(Builder, ArgBuffer, ArgBufferOffsets, F, i);
1231 }
1232 }
1233
1234 size_t Arg = 2;
1235 // Pass the group ids.
1236 Args[i++] = LLVMGetParam(WrapperKernel, Arg++);
1237 Args[i++] = LLVMGetParam(WrapperKernel, Arg++);
1238 Args[i++] = LLVMGetParam(WrapperKernel, Arg++);
1239
1240 assert (i == ArgCount);
1241
1242 // Pass the context object.
1243 LLVMValueRef Call = LLVMBuildCall(Builder, F, Args, ArgCount, "");
1244 LLVMBuildRetVoid(Builder);
1245
1246 llvm::CallInst *CallI = llvm::dyn_cast<llvm::CallInst>(llvm::unwrap(Call));
1247 CallI->setCallingConv(Func->getCallingConv());
1248
1249 return llvm::dyn_cast<llvm::Function>(llvm::unwrap(WrapperKernel));
1250}

1252/**
1253 * Creates a launcher function that executes all work-items in the grid by
1254 * launching a given work-group function for all work-group ids.
1255 *
1256 * The function adheres to the PHSA calling convention where the first two
1257 * arguments are for PHSA's context data, and the third one is the argument
1258 * buffer. The name will be phsa_kernel.KERNELNAME_grid_launcher.
1259 */
1260void
1261Workgroup::createGridLauncher(Function *KernFunc, Function *WGFunc,
1262 std::string KernName) {
1263
1264 LLVMValueRef Kernel = llvm::wrap(KernFunc);
1265 LLVMValueRef WGF = llvm::wrap(WGFunc);
1266 LLVMModuleRef M = llvm::wrap(this->M);
1267 LLVMContextRef LLVMContext = LLVMGetModuleContext(M);
1268
1269 LLVMTypeRef Int8Type = LLVMInt8TypeInContext(LLVMContext);
1270 LLVMTypeRef Int8PtrType = LLVMPointerType(Int8Type, 0);
1271 LLVMTypeRef ArgsPtrType =
1272 LLVMPointerType(Int8Type, currentPoclDevice->args_as_id);
1273
1274 std::ostringstream StrStr("phsa_kernel.", std::ios::ate);
1275 StrStr << KernName;
1276 StrStr << "_grid_launcher";
1277
1278 std::string FName = StrStr.str();
1279 const char *FunctionName = FName.c_str();
1280
1281 LLVMTypeRef LauncherArgTypes[] = {
1282 Int8PtrType /*phsactx0*/, Int8PtrType /*phsactx1*/, ArgsPtrType /*args*/};
1283
1284 LLVMTypeRef VoidType = LLVMVoidTypeInContext(LLVMContext);
1285 LLVMTypeRef LauncherFuncType =
1286 LLVMFunctionType(VoidType, LauncherArgTypes, 3, 0);
1287
1288 LLVMValueRef Launcher =
1289 LLVMAddFunction(M, FunctionName, LauncherFuncType);
1290
1291 LLVMBasicBlockRef Block =
1292 LLVMAppendBasicBlockInContext(LLVMContext, Launcher, "entry");
1293
1294 LLVMBuilderRef Builder = LLVMCreateBuilderInContext(LLVMContext);
1295 assert(Builder);
1296
1297 LLVMPositionBuilderAtEnd(Builder, Block);
1298
1299 LLVMValueRef RunnerFunc = LLVMGetNamedFunction(M, "_pocl_run_all_wgs");
1300 assert (RunnerFunc != nullptr);
1301
1302 LLVMTypeRef ArgTypes[] = {
1303 LLVMTypeOf(LLVMGetParam(RunnerFunc, 0)),
1304 LLVMTypeOf(LLVMGetParam(RunnerFunc, 1)),
1305 LLVMTypeOf(LLVMGetParam(RunnerFunc, 2))};
1306
1307 uint64_t KernArgCount = LLVMCountParams(Kernel);
1308 uint64_t KernArgBufferOffsets[KernArgCount];
1309 computeArgBufferOffsets(Kernel, KernArgBufferOffsets);
1310
1311 // The second argument in the native phsa interface is auxiliary
1312 // driver-specific data that is passed as the last argument to
1313 // the grid launcher.
1314 LLVMValueRef AuxParam = LLVMGetParam(Launcher, 1);
1315 LLVMValueRef ArgBuffer = LLVMGetParam(Launcher, 2);
1316
1317 // Load the pointer to the pocl context (in global memory),
1318 // assuming it is stored as the 4th last argument in the kernel.
1319 LLVMValueRef PoclCtx =
1320 createArgBufferLoad(Builder, ArgBuffer, KernArgBufferOffsets, Kernel,
1321 KernArgCount - HiddenArgs);
1322
1323 LLVMValueRef Args[4] = {
1324 LLVMBuildPointerCast(Builder, WGF, ArgTypes[0], "wg_func"),
1325 LLVMBuildPointerCast(Builder, ArgBuffer, ArgTypes[1], "args"),
1326 LLVMBuildPointerCast(Builder, PoclCtx, ArgTypes[2], "ctx"),
1327 LLVMBuildPointerCast(Builder, AuxParam, ArgTypes[1], "aux")};
1328
1329 LLVMValueRef Call = LLVMBuildCall(Builder, RunnerFunc, Args, 4, "");
1330 LLVMBuildRetVoid(Builder);
1331
1332 InlineFunctionInfo IFI;
1333 InlineFunction(dyn_cast<CallInst>(llvm::unwrap(Call)), IFI);
1334}

1336/**
1337 * Creates a work group launcher more suitable for the heterogeneous
1338 * host-device setup (called KERNELNAME_workgroup_fast).
1339 *
1340 * 1) Pointer arguments are stored directly as pointers to the
1341 * buffers in the argument buffer.
1342 *
1343 * 2) Scalar values are loaded from the global memory address
1344 * space.
1345 *
1346 * This should minimize copying of data and memory allocation
1347 * at the device.
1348 */
1349void
1350Workgroup::createFastWorkgroupLauncher(llvm::Function *F) {
1351
1352 IRBuilder<> builder(M->getContext());
1353
1354 std::string funcName = "";
1355 funcName = F->getName().str();
1356
1357#ifdef LLVM_OLDER_THAN_9_0
1358 Function *WorkGroup =
1359 dyn_cast<Function>(M->getOrInsertFunction(
1360 funcName + "_workgroup_fast", LauncherFuncT));
1361#else
1362 FunctionCallee fc = M->getOrInsertFunction(
1363 funcName + "_workgroup_fast", LauncherFuncT);
1364 Function *WorkGroup = dyn_cast<Function>(fc.getCallee());
1365#endif
1366 assert(WorkGroup != NULL);
1367
1368 builder.SetInsertPoint(BasicBlock::Create(M->getContext(), "", WorkGroup));
1369
1370 Function::arg_iterator ai = WorkGroup->arg_begin();
1371
1372 SmallVector<Value*, 8> arguments;
1373 size_t i = 0;
1374 for (Function::const_arg_iterator ii = F->arg_begin(), ee = F->arg_end();
1375 ii != ee; ++ii, ++i) {
1376
1377 if (i == F->arg_size() - 4)
1378 break;
1379
1380 Type *t = ii->getType();
1381 Value *gep = builder.CreateGEP(&*ai,
1382 ConstantInt::get(IntegerType::get(M->getContext(), 32), i));
1383 Value *pointer = builder.CreateLoad(gep);
1384
1385 if (t->isPointerTy()) {
1386 if (!ii->hasByValAttr()) {
1387 // Assume the pointer is directly in the arg array.
1388 arguments.push_back(builder.CreatePointerCast(pointer, t));
1389 continue;
1390 }
1391
1392 // It's a pass by value pointer argument, use the underlying
1393 // element type in subsequent load.
1394 t = t->getPointerElementType();
1395 }
1396
1397 // If it's a pass by value pointer argument, we just pass the pointer
1398 // as is to the function, no need to load from it first.
1399 Value *value;
1400
1401 if (!ii->hasByValAttr() || ((PointerType*)t)->getAddressSpace() == 1)
1402 value = builder.CreatePointerCast
1403 (pointer, t->getPointerTo(currentPoclDevice->global_as_id));
1404 else
1405 value = builder.CreatePointerCast(pointer, t->getPointerTo());
1406
1407 if (!ii->hasByValAttr()) {
1408 value = builder.CreateLoad(value);
1409 }
1410
1411 arguments.push_back(value);
1412 }
1413
1414 ++ai;
1415 arguments.push_back(&*ai);
1416 ++ai;
1417 arguments.push_back(&*ai);
1418 ++ai;
1419 arguments.push_back(&*ai);
1420 ++ai;
1421 arguments.push_back(&*ai);
1422
1423 builder.CreateCall(F, ArrayRef<Value*>(arguments));
1424 builder.CreateRetVoid();
1425}

1427// Returns true in case the given function is a kernel that
1428// should be processed by the kernel compiler.
1429bool Workgroup::isKernelToProcess(const Function &F) {
1430
1431 const Module *m = F.getParent();
1432
1433 if (F.getMetadata("kernel_arg_access_qual") &&
1434 F.getMetadata("pocl_generated") == nullptr)
1435 return true;
1436
1437 NamedMDNode *kernels = m->getNamedMetadata("opencl.kernels");
1438 if (kernels == NULL) {
1439 if (KernelName == "")
1440 return true;
1441 if (F.getName() == KernelName)
1442 return true;
1443
1444 return false;
1445 }
1446
1447 for (unsigned i = 0, e = kernels->getNumOperands(); i != e; ++i) {
1448 if (kernels->getOperand(i)->getOperand(0) == NULL)
1449 continue; // globaldce might have removed uncalled kernels
1450 Function *k =
1451 cast<Function>(
1452 dyn_cast<ValueAsMetadata>(kernels->getOperand(i)->getOperand(0))
1453 ->getValue());
1454 if (&F == k)
1455 return true;
1456 }
1457
1458 return false;
1459}

1461// Returns true in case the given function is a kernel with work-group
1462// barriers inside it.
1463bool
1464Workgroup::hasWorkgroupBarriers(const Function &F)
1465{
1466 for (llvm::Function::const_iterator i = F.begin(), e = F.end();
1467 i != e; ++i) {
1468 const llvm::BasicBlock* bb = &*i;
1469 if (Barrier::hasBarrier(bb)) {
1470
1471 // Ignore the implicit entry and exit barriers.
1472 if (Barrier::hasOnlyBarrier(bb) && bb == &F.getEntryBlock())
1473 continue;
1474
1475 if (Barrier::hasOnlyBarrier(bb) &&
1476 bb->getTerminator()->getNumSuccessors() == 0)
1477 continue;
1478
1479 return true;
1480 }
1481 }
1482 return false;
1483}
