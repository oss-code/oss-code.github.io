65static const char *workgroup_variables[] = {"_local_id_x",
66 "_local_id_y",
67 "_local_id_z",
68 "_local_size_x",
69 "_local_size_y",
70 "_local_size_z",
71 "_work_dim",
72 "_num_groups_x",
73 "_num_groups_y",
74 "_num_groups_z",
75 "_group_id_x",
76 "_group_id_y",
77 "_group_id_z",
78 "_global_offset_x",
79 "_global_offset_y",
80 "_global_offset_z",
81 NULL};

83bool FlattenGlobals::runOnModule(Module &M) {
84 SmallPtrSet<Function *, 8> functions_to_inline;
85 SmallVector<Value *, 8> pending;
86
87 const char **s = workgroup_variables;
88 while (*s != NULL) {
89 GlobalVariable *gv = M.getGlobalVariable(*s);
90 if (gv != NULL)
91 pending.push_back(gv);
92
93 ++s;
94 }
95
96 while (!pending.empty()) {
97 Value *v = pending.back();
98 pending.pop_back();
99
100 for (Value::use_iterator i = v->use_begin(), e = v->use_end(); i != e;
101 ++i) {
102 llvm::User *user = i->getUser();
103 if (Instruction *ci = dyn_cast<Instruction>(user)) {
104 // Prevent infinite looping on recursive functions
105 // (though OpenCL does not allow this?)
106 Function *f = ci->getParent()->getParent();
107 ;
108 assert(
109 (f != NULL) &&
110 "Per-workgroup global variable used on function with no parent!");
111 if (functions_to_inline.count(f))
112 continue;
113
114 functions_to_inline.insert(f);
115 pending.push_back(f);
116 }
117 }
118 }
119
120 for (SmallPtrSet<Function *, 8>::iterator i = functions_to_inline.begin(),
121 e = functions_to_inline.end();
122 i != e; ++i) {
123 (*i)->removeFnAttr(Attribute::NoInline);
124 (*i)->addFnAttr(Attribute::AlwaysInline);
125 }
126
127 StringRef barrier("_Z7barrierj");
128 for (llvm::Module::iterator i = M.begin(), e = M.end(); i != e; ++i) {
129 llvm::Function *f = &*i;
130 if (f->isDeclaration())
131 continue;
132 if (f->getName().equals(barrier)) {
133 f->removeFnAttr(Attribute::NoInline);
134 f->addFnAttr(Attribute::AlwaysInline);
135 }
136 }
137
138 return true;
139}
