49static std::string getDotBasicBlockID(llvm::BasicBlock* bb) {
50 std::ostringstream namess;
51 namess << "BB" << std::hex << bb;
52 return namess.str();
53}
54
55static void printBranches(
56 llvm::BasicBlock* b, std::ostream& s, bool /*highlighted*/) {
57
58 auto term = b->getTerminator();
59 for (unsigned i = 0; i < term->getNumSuccessors(); ++i)
60 {
61 BasicBlock *succ = term->getSuccessor(i);
62 s << getDotBasicBlockID(b) << " -> " << getDotBasicBlockID(succ) << ";"
63 << std::endl;
64 }
65 s << std::endl;
66}
67
68static void printBasicBlock(
69 llvm::BasicBlock* b, std::ostream& s, bool highlighted) {
70 // if (!Barrier::hasBarrier(b)) continue;
71 s << getDotBasicBlockID(b);
72 s << "[shape=rect,style=";
73 if (Barrier::hasBarrier(b))
74 s << "dotted";
75 else
76 s << "solid";
77
78 if (highlighted) {
79 s << ",color=red,style=filled";
80 }
81 s << ",label=\"" << b->getName().str() << ":\\n";
82
83 // The work-item loop control structures.
84 if (b->getName().startswith("pregion_for_cond")) {
85 s << "wi-loop branch\\n";
86 } else if (b->getName().startswith("pregion_for_inc")) {
87 s << "local_id_* increment\\n";
88 } else if (b->getName().startswith("pregion_for_init")) {
89 s << "wi-loop init\\n";
90 } else if (b->getName().startswith("pregion_for_end")) {
91 s << "wi-loop exit\\n";
92 } else {
93 // analyze the contents of the BB
94 int previousNonBarriers = 0;
95 for (llvm::BasicBlock::iterator instr = b->begin();
96 instr != b->end(); ++instr) {
97
98 if (isa<Barrier>(instr)) {
99 s << "BARRIER\\n";
100 previousNonBarriers = 0;
101 } else if (isa<BranchInst>(instr)) {
102 s << "branch\\n";
103 previousNonBarriers = 0;
104 } else if (isa<PHINode>(instr)) {
105 s << "PHI\\n";
106 previousNonBarriers = 0;
107 } else if (isa<ReturnInst>(instr)) {
108 s << "RETURN\\n";
109 previousNonBarriers = 0;
110 } else if (isa<UnreachableInst>(instr)) {
111 s << "UNREACHABLE\\n";
112 previousNonBarriers = 0;
113 } else {
114 if (previousNonBarriers == 0)
115 s << "...program instructions...\\n";
116 previousNonBarriers++;
117 }
118 }
119 }
120 s << "\"";
121 s << "]";
122 s << ";" << std::endl << std::endl;
123}

125/**
126 * pocl-specific dumping of the LLVM Function as a control flow graph in the
127 * Graphviz dot format.
128 *
129 * @param F the function to dump
130 * @param fname the target file name
131 * @param regions highlight these parallel regions in the graph
132 * @param highlights highlight these basic blocks in the graph
133 */
134void dumpCFG(
135 llvm::Function &F, std::string fname,
136 ParallelRegion::ParallelRegionVector *regions,
137 std::set<llvm::BasicBlock*> *highlights) {
138
139 if (fname == "")
140 fname = std::string("pocl_cfg.") + F.getName().str() + ".dot";
141
142 std::string origName = fname;
143 int counter = 0;
144 while (pocl_exists (fname.c_str())) {
145 std::ostringstream ss;
146 ss << origName << "." << counter;
147 fname = ss.str();
148 ++counter;
149 }
150
151 std::ofstream s;
152 s.open(fname.c_str(), std::ios::trunc);
153 s << "digraph " << F.getName().str() << " {" << std::endl;
154
155 std::set<BasicBlock*> regionBBs;
156
157 if (regions != NULL) {
158
159 for (ParallelRegion::ParallelRegionVector::iterator ri = regions->begin(),
160 re = regions->end(); ri != re; ++ri) {
161
162 ParallelRegion* pr = *ri;
163
164 s << "\tsubgraph cluster" << pr->GetID() << " {" << std::endl;
165
166 for (ParallelRegion::iterator i = pr->begin(), e = pr->end();
167 i != e; ++i) {
168 BasicBlock *b = *i;
169 printBasicBlock(
170 b, s, highlights != NULL &&
171 highlights->find(b) != highlights->end());
172 regionBBs.insert(b);
173 }
174 s << "label=\"Parallel region #" << pr->GetID() << "\";" << std::endl;
175 s << "}" << std::endl;
176 }
177 }
178
179 for (Function::iterator i = F.begin(), e = F.end(); i != e; ++i) {
180 BasicBlock *b = &*i;
181 if (regionBBs.find(b) != regionBBs.end()) continue;
182 printBasicBlock
183 (b, s, highlights != NULL && highlights->find(b) != highlights->end());
184 }
185
186 for (Function::iterator i = F.begin(), e = F.end(); i != e; ++i) {
187 BasicBlock *b = &*i;
188 printBranches
189 (b, s, highlights != NULL && highlights->find(b) != highlights->end());
190 }
191
192 s << "}" << std::endl;
193 s.close();
194 std::cout << "### dumped CFG to " << fname << std::endl;
195}

197bool chopBBs(llvm::Function &F, llvm::Pass &) {
198 bool fchanged = false;
199 const int MAX_INSTRUCTIONS_PER_BB = 70;
200 do {
201 fchanged = false;
202 for (Function::iterator i = F.begin(), e = F.end(); i != e; ++i) {
203 BasicBlock *b = &*i;
204
205 if (b->size() > MAX_INSTRUCTIONS_PER_BB + 1)
206 {
207 int count = 0;
208 BasicBlock::iterator splitPoint = b->begin();
209 while (count < MAX_INSTRUCTIONS_PER_BB || isa<PHINode>(splitPoint))
210 {
211 ++splitPoint;
212 ++count;
213 }
214 SplitBlock(b, &*splitPoint);
215 fchanged = true;
216 break;
217 }
218 }
219
220 } while (fchanged);
221 return fchanged;
222}
