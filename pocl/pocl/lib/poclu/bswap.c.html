28#define GENERIC_BYTESWAP(__DTYPE, __WORD) \
29 do { \
30 unsigned __i; \
31 union _word_union \
32 { \
33 __DTYPE full_word; \
34 unsigned char bytes[sizeof(__DTYPE)]; \
35 } __old, __neww; \
36 \
37 __old.full_word = __WORD; \
38 for (__i = 0; __i < sizeof(__DTYPE); ++__i) \
39 __neww.bytes[__i] = __old.bytes[sizeof(__DTYPE) - __i - 1]; \
40 __WORD = __neww.full_word; \
41 } while(0)
42
43static int
44needs_swap(cl_device_id device)
45{
46 cl_bool deviceLittle;
47 clGetDeviceInfo
48 (device, CL_DEVICE_ENDIAN_LITTLE, sizeof(cl_bool),
49 &deviceLittle, NULL);
50
51#if defined(WORDS_BIGENDIAN) && WORDS_BIGENDIAN == 1
52 return deviceLittle;
53#else
54 return !deviceLittle;
55#endif
56}
57
58cl_int
59poclu_bswap_cl_int(cl_device_id device, cl_int original)
60{
61 if (!needs_swap (device)) return original;
62 GENERIC_BYTESWAP (cl_int, original);
63 return original;
64}

82void
83poclu_bswap_cl_int_array(cl_device_id device, cl_int* array,
84 size_t num_elements)
85{
86 size_t i;
87 if (!needs_swap (device)) return;
88 for (i = 0; i < num_elements; ++i)
89 {
90 GENERIC_BYTESWAP (cl_int, array[i]);
91 }
92}

118void
119poclu_bswap_cl_float2_array(cl_device_id device, cl_float2* array,
120 size_t num_elements)
121{
122 size_t i;
123 if (!needs_swap (device)) return;
124 for (i = 0; i < num_elements; ++i)
125 {
126 GENERIC_BYTESWAP (cl_float, array[i].s[0]);
127 GENERIC_BYTESWAP (cl_float, array[i].s[1]);
128 }
