33cl_context
34poclu_create_any_context ()
35{
36 cl_uint i;
37 cl_platform_id platform;
38
39 clGetPlatformIDs (1, &platform, &i);
40 if (i == 0)
41 return (cl_context) 0;
42
43 cl_context_properties properties[] =
44 {CL_CONTEXT_PLATFORM,
45 (cl_context_properties)platform,
46 0};
47
48 // create the OpenCL context on any available OCL device
49 cl_context context = clCreateContextFromType (properties,
50 CL_DEVICE_TYPE_ALL,
51 NULL, NULL, NULL);
52
53 return context;
54}

64cl_int
65poclu_get_any_device2 (cl_context *context, cl_device_id *device,
66 cl_command_queue *queue, cl_platform_id *platform)
67{
68 cl_int err;
69
70 if (context == NULL ||
71 device == NULL ||
72 queue == NULL ||
73 platform == NULL)
74 return CL_INVALID_VALUE;
75
76 err = clGetPlatformIDs (1, platform, NULL);
77 if (err != CL_SUCCESS)
78 return err;
79
80 err = clGetDeviceIDs (*platform, CL_DEVICE_TYPE_ALL, 1, device, NULL);
81 if (err != CL_SUCCESS)
82 return err;
83
84 *context = clCreateContext (NULL, 1, device, NULL, NULL, &err);
85 if (err != CL_SUCCESS)
86 return err;
87
88 *queue = clCreateCommandQueue (*context, *device,
89 CL_QUEUE_PROFILING_ENABLE, &err);
90 if (err != CL_SUCCESS)
91 return err;
92
93 return CL_SUCCESS;
94}

96cl_int
97poclu_get_multiple_devices (cl_platform_id *platform, cl_context *context,
98 cl_uint *num_devices, cl_device_id **devices,
99 cl_command_queue **queues)
100{
101 cl_int err;
102 *num_devices = 0;
103 size_t i;
104
105 if (context == NULL || devices == NULL || queues == NULL || platform == NULL)
106 return CL_INVALID_VALUE;
107
108 err = clGetPlatformIDs (1, platform, NULL);
109 if (err != CL_SUCCESS)
110 return err;
111
112 err = clGetDeviceIDs (*platform, CL_DEVICE_TYPE_ALL, 0, NULL, num_devices);
113 if (err != CL_SUCCESS)
114 return err;
115
116 cl_device_id *devs = calloc (*num_devices, sizeof (cl_device_id));
117 cl_command_queue *ques = calloc (*num_devices, sizeof (cl_command_queue));
118
119 err = clGetDeviceIDs (*platform, CL_DEVICE_TYPE_ALL, *num_devices, devs,
120 NULL);
121 if (err != CL_SUCCESS)
122 return err;
123
124 *context = clCreateContext (NULL, *num_devices, devs, NULL, NULL, &err);
125 if (err != CL_SUCCESS)
126 return err;
127
128 for (i = 0; i < *num_devices; ++i)
129 {
130 ques[i] = clCreateCommandQueue (*context, devs[i],
131 CL_QUEUE_PROFILING_ENABLE, &err);
132 if (err != CL_SUCCESS)
133 return err;
134 }
135
136 *devices = devs;
137 *queues = ques;
138 return CL_SUCCESS;
139}

141char *
142poclu_read_binfile (const char *filename, size_t *len)
143{
144 FILE *file;
145 char *src;
146
147 file = fopen (filename, "r");
148 if (file == NULL)
149 return NULL;
150
151 fseek (file, 0, SEEK_END);
152 *len = ftell (file);
153 src = (char*)malloc (*len + 1);
154 if (src == NULL)
155 {
156 fclose (file);
157 return NULL;
158 }
159
160 fseek (file, 0, SEEK_SET);
161 fread (src, *len, 1, file);
162 fclose (file);
163
164 return src;
165}

167char *
168poclu_read_file (const char *filename)
169{
170 size_t size;
171 char *res = poclu_read_binfile (filename, &size);
172 if (res)
173 res[size] = 0;
174 return res;
175}

177int
178poclu_write_file (const char *filemane, char *content, size_t size)
179{
180 FILE *file;
181
182 file = fopen (filemane, "w");
183 if (file == NULL)
184 return -1;
185
186 if (fwrite (content, sizeof (char), size, file) < size)
187 {
188 fclose (file);
189 return -1;
190 }
191
192 if (fclose (file))
193 return -1;
194
195 return 0;
196}

276int
277poclu_load_program_multidev (cl_context context, cl_device_id *devices,
278 cl_uint num_devices, const char *basename,
279 int spir, int spirv, int poclbin,
280 const char *explicit_binary,
281 const char *extra_build_opts, cl_program *p)
282{
283 cl_bool little_endian = 0;
284 cl_uint address_bits = 0;
285 char extensions[1024];
286 char path[1024];
287 const char *ext;
288 char final_opts[2048];
289 size_t binary_size = 0;
290 char *binary = NULL;
291 cl_program program;
292 cl_int err = CL_SUCCESS;
293
294 int from_source = (!spir && !spirv && !poclbin);
295 TEST_ASSERT (num_devices > 0);
296 cl_device_id device = devices[0];
297 if (num_devices > 1)
298 TEST_ASSERT (from_source);
299
300 *p = NULL;
301 final_opts[0] = 0;
302 if (extra_build_opts != NULL)
303 strcat(final_opts, extra_build_opts);
304
305 if (spir || spirv)
306 {
307 TEST_ASSERT (device != NULL);
308 strcat (final_opts, " -x spir -spir-std=1.2");
309 err = clGetDeviceInfo (device, CL_DEVICE_EXTENSIONS, 1024, extensions,
310 NULL);
311 CHECK_OPENCL_ERROR_IN ("clGetDeviceInfo extensions");
312
313 if (strstr (extensions, "cl_khr_spir") == NULL)
314 {
315 printf ("SPIR not supported, cannot run the test\n");
316 return -1;
317 }
318
319 err = clGetDeviceInfo (device, CL_DEVICE_ENDIAN_LITTLE, sizeof (cl_bool),
320 &little_endian, NULL);
321 CHECK_OPENCL_ERROR_IN ("clGetDeviceInfo endianness");
322
323 if (little_endian == CL_FALSE)
324 {
325 fprintf (stderr,
326 "SPIR can only be tested on little-endian devices\n");
327 return 1;
328 }
329
330 err = clGetDeviceInfo (device, CL_DEVICE_ADDRESS_BITS, sizeof (cl_uint),
331 &address_bits, NULL);
332 CHECK_OPENCL_ERROR_IN ("clGetDeviceInfo addr bits");
333
334 if (spirv)
335 {
336 if (address_bits < 64)
337 ext = ".spirv32";
338 else
339 ext = ".spirv64";
340 }
341 else
342 {
343 if (address_bits < 64)
344 ext = ".spir32";
345 else
346 ext = ".spir64";
347 }
348 }
349
350 if (poclbin)
351 {
352 ext = ".poclbin";
353 }
354
355 if (from_source)
356 {
357 ext = ".cl";
358 }
359
360 if (explicit_binary)
361 snprintf (path, 1024, "%s", explicit_binary);
362 else
363 {
364 snprintf (path, 1024, "%s%s", basename, ext);
365
366 if (access (path, F_OK))
367 {
368 snprintf (path, 1024, "%s/examples/%s/%s%s", SRCDIR, basename,
369 basename, ext);
370 if (access (path, F_OK))
371 {
372 fprintf (stderr, "Can't find %s%s SPIR / POCLBIN file anywhere\n",
373 basename, ext);
374 return 1;
375 }
376 }
377 }
378
379 if (from_source)
380 {
381 char *src = poclu_read_file (path);
382 TEST_ASSERT (src != NULL);
383
384 program = clCreateProgramWithSource (context, 1, (const char **)&src,
385 NULL, &err);
386 CHECK_OPENCL_ERROR_IN ("clCreateProgramWithSource");
387
388 err = clBuildProgram (program, num_devices, devices, final_opts, NULL,
389 NULL);
390 CHECK_OPENCL_ERROR_IN ("clBuildProgram");
391 free (src);
392 }
393 else if (spirv)
394 {
395 TEST_ASSERT (device != NULL);
396 binary = poclu_read_binfile (path, &binary_size);
397 TEST_ASSERT (binary != NULL);
398
399 program = clCreateProgramWithIL (context, (const void *)binary,
400 binary_size, &err);
401 CHECK_OPENCL_ERROR_IN ("clCreateProgramWithIL");
402
403 err = clBuildProgram (program, 0, NULL, final_opts, NULL, NULL);
404 CHECK_OPENCL_ERROR_IN ("clBuildProgram");
405 free (binary);
406 }
407 else
408 {
409 TEST_ASSERT (device != NULL);
410 binary = poclu_read_binfile (path, &binary_size);
411 TEST_ASSERT (binary != NULL);
412
413 program = clCreateProgramWithBinary (context, 1, &device, &binary_size,
414 (const unsigned char **)&binary,
415 NULL, &err);
416 CHECK_OPENCL_ERROR_IN ("clCreateProgramWithBinary");
417
418 err = clBuildProgram (program, 0, NULL, final_opts, NULL, NULL);
419 CHECK_OPENCL_ERROR_IN ("clBuildProgram");
420 free (binary);
421 }
422
423 *p = program;
424 return err;
425}

427int
428poclu_load_program (cl_context context, cl_device_id device,
429 const char *basename, int spir, int spirv, int poclbin,
430 const char *explicit_binary, const char *extra_build_opts,
431 cl_program *p)
432{
433 return poclu_load_program_multidev (context, &device, 1, basename, spir,
434 spirv, poclbin, explicit_binary,
435 extra_build_opts, p);
436}
