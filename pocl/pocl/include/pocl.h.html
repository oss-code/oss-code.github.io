50typedef struct _mem_mapping mem_mapping_t;
51/* represents a single buffer to host memory mapping */
52struct _mem_mapping {
53 void *host_ptr; /* the location of the mapped buffer chunk in the host memory */
54 size_t offset; /* offset to the beginning of the buffer */
55 size_t size;
56 mem_mapping_t *prev, *next;
57 /* This is required, because two clEnqueueMap() with the same buffer+size+offset,
58 will create two identical mappings in the buffer->mappings LL.
59 Without this flag, both corresponding clEnqUnmap()s will find
60 the same mapping (the first one in mappings LL), which will lead
61 to memory double-free corruption later. */
62 long unmap_requested;
63 cl_map_flags map_flags;
64 /* image mapping data */
65 size_t origin[3];
66 size_t region[3];
67 size_t row_pitch;
68 size_t slice_pitch;
69};

356// one item in the command queue
357typedef struct _cl_command_node _cl_command_node;
358struct _cl_command_node
359{
360 _cl_command_t command;
361 cl_command_type type;
362 _cl_command_node *next; // for linked-list storage
363 _cl_command_node *prev;
364 cl_event event;
365 const cl_event *event_wait_list;
366 cl_device_id device;
367 /* The index of the targeted device in the platform's device list. */
368 unsigned device_i;
369 cl_int ready;
370};
