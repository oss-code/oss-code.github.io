127/* In case cl_err != CL_SUCCESS, prints out the error +
128 * function : line to stderr, and returns 1, otherwise
129 * returns 0
130 */
131POCLU_API int POCLU_CALL check_cl_error (cl_int cl_err, int line,
132 const char *func_name);
133
134#define _POCLU_CHECK_CL_ERROR_INNER(cond, func, line) \
135 do \
136 { \
137 if (check_cl_error (cond, line, func)) \
138 return (EXIT_FAILURE); \
139 } \
140 while (0)
141
142#define CHECK_CL_ERROR(cond) _POCLU_CHECK_CL_ERROR_INNER(cond, __PRETTY_FUNCTION__, __LINE__)
143
144#define CHECK_OPENCL_ERROR_IN(message) _POCLU_CHECK_CL_ERROR_INNER(err, message, __LINE__)
145

147#define TEST_ASSERT(EXP) \
148do { \
149 if (!(EXP)) { \
150 fprintf(stderr, "Assertion: \n" #EXP "\nfailed on %s:%i\n", \
151 __FILE__, __LINE__); \
152 return EXIT_FAILURE; \
153 } \
154} while (0)
155
156#define CHECK_CL_ERROR2(err) \
157 if (check_cl_error (err, __LINE__, __PRETTY_FUNCTION__)) \
158 goto ERROR
