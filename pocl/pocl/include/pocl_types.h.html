24/* This header is designed to be included both from the device and the host.
25 In case compiling OpenCL C sources, __OPENCL_VERSION__ should be set.
26 In case compiling in the host, all but the device-specific types are
27 defined (size_t and others). Devices should avoid including the C
28 stdint.h instead of this one as OpenCL C size_t et al. is allowed to
29 be of different width than when targeting C.
30
31 TODO: replace this header (partially) with Clang's opencl-c.h
32*/
33
34#ifndef POCL_DEVICE_TYPES_H
35#define POCL_DEVICE_TYPES_H
36
37#ifdef __OPENCL_VERSION__
38
39#ifdef __USE_CLANG_OPENCL_C_H
40
41/* Minimal definitions, only the target specific macro overrides,
42 just in case Clang export the C ones which might differ for
43 OpenCL C. */
44
45#ifdef __INTPTR_TYPE__
46#undef __INTPTR_TYPE__
47#endif
48
49#ifdef __UINTPTR_TYPE__
50#undef __UINTPTR_TYPE__
51#endif
52
53#ifdef __SIZE_TYPE__
54#undef __SIZE_TYPE__
55#endif
56
57#ifdef __SIZE_MAX__
58#undef __SIZE_MAX__
59#endif
60
61#if defined(POCL_DEVICE_ADDRESS_BITS) && POCL_DEVICE_ADDRESS_BITS == 32
62#define __SIZE_TYPE__ uint
63#define __SIZE_MAX__ UINT_MAX
64#else
65#define __SIZE_TYPE__ ulong
66#define __SIZE_MAX__ ULONG_MAX
67#endif
68
69#define __INTPTR_TYPE__ __SIZE_TYPE__
70#define __UINTPTR_TYPE__ __INTPTR_TYPE__
71
72#else
73
74/* Compiling Device-specific OpenCL C or builtin library C. */
75
76#if defined cl_khr_fp64 && !defined cl_khr_int64
77#error "cl_khr_fp64 requires cl_khr_int64"
78#endif
79
80/* TODO FIXME We should not use these in OpenCL library's C code at all.
81 * The problem is that 1) these are predefined by glibc, 2) while we can
82 * re-define "ulong", we cannot control the size of "long" at all.
83 * which can lead to "ulong" being 64bit and "long" 32bit, resulting in
84 * mysterious errors and bugs. Therefore OpenCL library's C code should
85 * use the fixed size C types where integer size matters. */
86
87#ifdef __CBUILD__
88
89/* Builtin library C code definitions. */
90
91#define size_t csize_t
92#define uintptr_t cuintptr_t
93
94#include <stdint.h>
95
96#undef size_t
97#undef uintptr_t
98
99typedef uint8_t uchar;
100typedef uint16_t ushort;
101typedef uint32_t uint;
102
103#ifdef cl_khr_int64
104typedef uint64_t ulong;
105#else
106typedef uint32_t ulong;
107#endif
108
109#ifndef cl_khr_fp16
110typedef short half;
111#endif
112
113#endif
114
115/* The definitions below intentionally lead to errors if these types
116 are used when they are not available in the language. This prevents
117 accidentally using them if the compiler does not disable these
118 types, but only e.g. defines them with an incorrect size.*/
119
120#ifndef cl_khr_fp64
121typedef struct error_undefined_type_double error_undefined_type_double;
122#define double error_undefined_type_double
123#endif
124
125#ifdef __SIZE_TYPE__
126#undef __SIZE_TYPE__
127#endif
128
129#ifdef __SIZE_MAX__
130#undef __SIZE_MAX__
131#endif
132
133#if defined(POCL_DEVICE_ADDRESS_BITS) && POCL_DEVICE_ADDRESS_BITS == 32
134#define __SIZE_TYPE__ uint
135#define __SIZE_MAX__ UINT_MAX
136#else
137#define __SIZE_TYPE__ ulong
138#define __SIZE_MAX__ ULONG_MAX
139#endif
140
141typedef __SIZE_TYPE__ size_t;
142typedef __PTRDIFF_TYPE__ ptrdiff_t;
143typedef ptrdiff_t intptr_t;
144typedef size_t uintptr_t;
145
146#endif /* #ifdef __USE_CLANG_OPENCL_C_H */
147
148#else /* #ifdef __OPENCL_VERSION__ */
149
150/* Including from a host source (runtime API implementation). Introduce
151 the fixed width datatypes, but do not override C's size_t and other
152 target specific datatypes. */
153
154typedef unsigned char uchar;
155
156/* FIXME see the above TODO about these types. */
157
158#if !(defined(_SYS_TYPES_H) && defined(__USE_MISC))
159/* glibc, when including sys/types.h, typedefs these. */
160
161typedef unsigned long int ulong;
162typedef unsigned short int ushort;
163typedef unsigned int uint;
164
165#endif
166
167#include <stdint.h>
168
169#endif
170
171#endif
