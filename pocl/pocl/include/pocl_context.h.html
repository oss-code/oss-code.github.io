28#ifndef POCL_CONTEXT_H
29#define POCL_CONTEXT_H
30
31#include "pocl_types.h"
32
33struct pocl_context32 {
34 uint num_groups[3];
35 uint global_offset[3];
36 uint local_size[3];
37 unsigned char *printf_buffer;
38 uint *printf_buffer_position;
39 uint printf_buffer_capacity;
40 uint work_dim;
41};
42
43/* The default pocl_context is 64b. It should be copied to a 32b one
44 when launching a kernel for a 32b target. */
45struct pocl_context {
46 ulong num_groups[3];
47 ulong global_offset[3];
48 ulong local_size[3];
49 uchar *printf_buffer;
50 uint *printf_buffer_position;
51 uint printf_buffer_capacity;
52 uint work_dim;
53};
54
55/* Copy a 64b context struct to a 32b one. */
56#define POCL_CONTEXT_COPY64TO32(__DST, __SRC) \
57 do { \
58 struct pocl_context *__src = (struct pocl_context *)__SRC; \
59 struct pocl_context32 *__dst = (struct pocl_context32 *)__DST; \
60 __dst->work_dim = __src->work_dim; \
61 __dst->num_groups[0] = __src->num_groups[0]; \
62 __dst->num_groups[1] = __src->num_groups[1]; \
63 __dst->num_groups[2] = __src->num_groups[2]; \
64 __dst->global_offset[0] = __src->global_offset[0]; \
65 __dst->global_offset[1] = __src->global_offset[1]; \
66 __dst->global_offset[2] = __src->global_offset[2]; \
67 __dst->local_size[0] = __src->local_size[0]; \
68 __dst->local_size[1] = __src->local_size[1]; \
69 __dst->local_size[2] = __src->local_size[2]; \
70 __dst->printf_buffer = __src->printf_buffer; \
71 __dst->printf_buffer_position = __src->printf_buffer_position; \
72 __dst->printf_buffer_capacity = __src->printf_buffer_capacity; \
73 } while (0)
74
75#define POCL_CONTEXT_SIZE(__BITNESS) \
76 (__BITNESS == 64 ? \
77 sizeof (struct pocl_context) : \
78 sizeof (struct pocl_context32))
79
80#endif
