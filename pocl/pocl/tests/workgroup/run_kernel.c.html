34/**
35 * The test kernels are assumed to:
36 *
37 * 1) called 'test_kernel'
38 * 2) a) have no inputs or outputs, then only work item id printfs to verify
39 * the correct workgroup transformations, or
40 * b) have a single output buffer with the grid size of integers, which
41 * is dumped after executing the test, to avoid the need for relying
42 * on printf or work-item ordering in the verification
43 * 3) executable with any local and global dimensions and sizes
44 *
45 * Usage:
46 *
47 * ./run_kernel somekernel.cl 2 2 3 4
48 *
49 * Where the first integer is the number of work groups to execute and the
50 * rest are the local dimensions.
51 */
52int
53main (int argc, char **argv)
54{
55 char *source;
56 cl_context context = NULL;
57 size_t cb = 0;
58 cl_device_id *devices = NULL;
59 cl_command_queue cmd_queue = NULL;
60 cl_program program = NULL;
61 cl_int err = CL_SUCCESS;
62 cl_kernel kernel = NULL;
63 cl_mem outbuf = NULL;
64 size_t global_work_size[3];
65 size_t local_work_size[3];
66 char kernel_path[2048];
67
68 snprintf (kernel_path, 2048, "%s/%s", SRCDIR, argv[1]);
69
70 source = poclu_read_file (kernel_path);
71 TEST_ASSERT (source != NULL && "Kernel .cl not found.");
72
73 local_work_size[0] = atoi(argv[3]);
74 local_work_size[1] = atoi(argv[4]);
75 local_work_size[2] = atoi(argv[5]);
76
77 global_work_size[0] = local_work_size[0] * atoi(argv[2]);
78 global_work_size[1] = local_work_size[1];
79 global_work_size[2] = local_work_size[2];
80
81 context = poclu_create_any_context();
82 TEST_ASSERT (context != NULL && "clCreateContextFromType call failed\n");
83
84 err = clGetContextInfo (context, CL_CONTEXT_DEVICES, 0, NULL, &cb);
85 CHECK_OPENCL_ERROR_IN ("clGetContextInfo 1\n");
86 devices = (cl_device_id *) malloc(cb);
87 clGetContextInfo (context, CL_CONTEXT_DEVICES, cb, devices, NULL);
88 CHECK_OPENCL_ERROR_IN ("clGetContextInfo 2\n");
89
90 cmd_queue = clCreateCommandQueue (context, devices[0], 0, &err);
91 CHECK_CL_ERROR2 (err);
92
93 program = clCreateProgramWithSource (context, 1, (const char **)&source,
94 NULL, &err);
95 CHECK_CL_ERROR2 (err);
96
97 err = clBuildProgram (program, 0, NULL, NULL, NULL, NULL);
98 CHECK_CL_ERROR2 (err);
99
100 kernel = clCreateKernel (program, "test_kernel", &err);
101 CHECK_CL_ERROR2 (err);
102
103 cl_uint num_args = 0;
104 err = clGetKernelInfo (kernel, CL_KERNEL_NUM_ARGS, sizeof (num_args),
105 &num_args, NULL);
106 CHECK_CL_ERROR2 (err);
107
108 size_t grid_size
109 = global_work_size[0] * global_work_size[1] * global_work_size[2];
110 if (num_args == 1)
111 {
112 cl_int err;
113 outbuf = clCreateBuffer (context, CL_MEM_READ_WRITE,
114 grid_size * sizeof (cl_int), NULL, &err);
115 CHECK_CL_ERROR2 (err);
116 err = clSetKernelArg (kernel, 0, sizeof (outbuf), &outbuf);
117 CHECK_CL_ERROR2 (err);
118 }
119 else if (num_args != 0)
120 assert (num_args == 0 || num_args == 1);
121
122 err = clEnqueueNDRangeKernel (cmd_queue, kernel, 3, NULL, global_work_size,
123 local_work_size, 0, NULL, NULL);
124 CHECK_CL_ERROR2 (err);
125
126 if (num_args == 1)
127 {
128 cl_int kern_output[grid_size];
129 err = clEnqueueReadBuffer (cmd_queue, outbuf, CL_TRUE, 0,
130 grid_size * sizeof (cl_int), kern_output, 0,
131 NULL, NULL);
132 CHECK_CL_ERROR2 (err);
133 size_t i;
134 for (i = 0; i < grid_size; ++i)
135 printf ("%zu: %d\n", i, kern_output[i]);
136 }
137
138 err = clFinish (cmd_queue);
139 CHECK_CL_ERROR2 (err);
140
141ERROR:
142 if (outbuf)
143 clReleaseMemObject (outbuf);
144 if (kernel)
145 clReleaseKernel (kernel);
146 if (program)
147 clReleaseProgram (program);
148 if (cmd_queue)
149 clReleaseCommandQueue (cmd_queue);
150 if (context)
151 clReleaseContext (context);
152
153 return (err == CL_SUCCESS) ? 0 : 1;
154}
