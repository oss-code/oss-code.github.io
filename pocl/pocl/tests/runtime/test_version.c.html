1/* This test is part of pocl.
2 * It is intended to run as the first test of the
3 * testsuite, checking that the tests are
4 * not run against an other installed OpenCL library.
5 */
6
7#include "poclu.h"
8#include <stdio.h>
9#include <string.h>
10#include <stdlib.h>
11#include "config.h"
12
13int main(void)
14{
15 cl_context context;
16 cl_device_id did;
17 cl_platform_id pid;
18 cl_command_queue queue;
19 size_t rvs;
20 char result[1024];
21 char *needle;
22
23 /* Check that the default platform we get from the ICD loader
24 * matches the pocl version string this binary was built against. */
25 CHECK_CL_ERROR (poclu_get_any_device2 (&context, &did, &queue, &pid));
26 TEST_ASSERT( context );
27 TEST_ASSERT( did );
28 TEST_ASSERT( queue );
29
30 CHECK_CL_ERROR(clGetPlatformInfo( pid, CL_PLATFORM_VERSION,
31 sizeof(result), result, &rvs));
32
33 result[rvs]=0; // spec doesn't say it is null-terminated.
34 const char *expected = "OpenCL " POCL_CL_VERSION
35 " pocl " PACKAGE_VERSION;
36 if (strncmp (result, expected, strlen (expected)) != 0)
37 {
38 printf ("Error: platform is: %s\n", result);
39 return 2;
40 }
41
42
43 /* Pocl devices have the form 'type'-'details', if details are
44 * available. If not, they are of the form 'type'.
45 * print here only the type, as the details will be computer
46 * dependent */
47 CHECK_CL_ERROR(clGetDeviceInfo( did, CL_DEVICE_NAME,
48 sizeof(result), result, NULL));
49
50 result[rvs]=0;
51 needle = strchr(result, '-');
52 if( needle != NULL ){
53 *needle=0;
54 }
55 printf("%s\n", result);
56
57 CHECK_CL_ERROR (clReleaseCommandQueue (queue));
58 CHECK_CL_ERROR (clReleaseContext (context));
59 CHECK_CL_ERROR (clUnloadCompiler ());
60
61 return 0;
62}
