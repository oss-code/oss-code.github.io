24#include <stdio.h>
25#include <stdlib.h>
26#include <string.h>
27#include <CL/cl.h>
28#include "poclu.h"
29
30int main(int argc, char **argv)
31{
32 cl_int err;
33 cl_context ctx;
34 cl_command_queue queue;
35 cl_device_id did;
36
37 poclu_get_any_device(&ctx, &did, &queue);
38 TEST_ASSERT(ctx);
39 TEST_ASSERT(did);
40 TEST_ASSERT(queue);
41
42 const size_t buf_size = sizeof(cl_int);
43 cl_mem buf = clCreateBuffer(ctx, CL_MEM_READ_WRITE, buf_size, NULL, &err);
44 CHECK_OPENCL_ERROR_IN("clCreateBuffer");
45
46 cl_image_format img_fmt = {
47 .image_channel_order = CL_RGBA,
48 .image_channel_data_type = CL_UNSIGNED_INT32 };
49 cl_image_desc img_dsc = {
50 .image_type = CL_MEM_OBJECT_IMAGE2D,
51 .image_width = 1,
52 .image_height = 1,
53 .image_depth = 1,
54 .image_array_size = 1,
55 .image_row_pitch = 0,
56 .image_slice_pitch = 0,
57 .num_mip_levels = 0,
58 .num_samples = 0,
59 .buffer = NULL,
60 };
61
62 cl_mem img = clCreateImage(ctx, CL_MEM_READ_WRITE,
63 &img_fmt, &img_dsc, NULL, &err);
64 CHECK_OPENCL_ERROR_IN("clCreateImage");
65
66 /* An invalid waiting list (e.g. a null event in it) should make
67 * associated commands fail without segfaults and without touching any associated
68 * event. Test that this is indeed the case.
69 */
70
71 cl_int *host_ptr = NULL;
72 cl_event no_event = NULL;
73
74 /* We will test both NULL and invalid initial value for map_event */
75 cl_event initial_values[] = { NULL, (cl_event)1 };
76 cl_uint i = 0;
77
78 /***
79 * Test buffer mapping/unmapping
80 */
81
82 /* Test without associated event */
83 host_ptr = clEnqueueMapBuffer(queue, buf, CL_TRUE, CL_MAP_READ, 0, buf_size,
84 1, &no_event, NULL, &err);
85 TEST_ASSERT(err == CL_INVALID_EVENT_WAIT_LIST);
86 TEST_ASSERT(host_ptr == NULL);
87
88 /* Test with map_event = NULL */
89 cl_event map_event = NULL;
90 host_ptr = clEnqueueMapBuffer(queue, buf, CL_TRUE, CL_MAP_READ, 0, buf_size,
91 1, &no_event, &map_event, &err);
92 TEST_ASSERT(err == CL_INVALID_EVENT_WAIT_LIST);
93 TEST_ASSERT(host_ptr == NULL);
94 TEST_ASSERT(map_event == NULL); /* should not have been touched */
95
96 /* Now do an actual mapping to test the unmapping */
97 host_ptr = clEnqueueMapBuffer(queue, buf, CL_TRUE, CL_MAP_READ, 0, buf_size,
98 0, NULL, NULL, &err);
99 CHECK_OPENCL_ERROR_IN("map buffer");
100
101 err = clEnqueueUnmapMemObject(queue, buf, host_ptr, 1, &no_event, NULL);
102 TEST_ASSERT(err == CL_INVALID_EVENT_WAIT_LIST);
103 err = clEnqueueUnmapMemObject(queue, buf, host_ptr, 1, &no_event, &map_event);
104 TEST_ASSERT(err == CL_INVALID_EVENT_WAIT_LIST);
105 TEST_ASSERT(map_event == NULL); /* should not have been touched */
106
107 /* Actually unmap */
108 err = clEnqueueUnmapMemObject(queue, buf, host_ptr, 0, NULL, NULL);
109 CHECK_OPENCL_ERROR_IN("unmap buffer");
110 host_ptr = NULL;
111
112 /* Test with map_event != NULL but invalid */
113 map_event = (cl_event)1;
114 host_ptr = clEnqueueMapBuffer(queue, buf, CL_TRUE, CL_MAP_READ, 0, buf_size,
115 1, &no_event, &map_event, &err);
116 TEST_ASSERT(err == CL_INVALID_EVENT_WAIT_LIST);
117 TEST_ASSERT(host_ptr == NULL);
118 TEST_ASSERT(map_event == (cl_event)1); /* should not have been touched */
119
120 host_ptr = clEnqueueMapBuffer(queue, buf, CL_TRUE, CL_MAP_READ, 0, buf_size,
121 0, NULL, NULL, &err);
122 CHECK_OPENCL_ERROR_IN("map buffer");
123
124 err = clEnqueueUnmapMemObject(queue, buf, host_ptr, 1, &no_event, &map_event);
125 TEST_ASSERT(err == CL_INVALID_EVENT_WAIT_LIST);
126 TEST_ASSERT(map_event == (cl_event)1); /* should not have been touched */
127
128 err = clEnqueueUnmapMemObject(queue, buf, host_ptr, 0, NULL, NULL);
129 CHECK_OPENCL_ERROR_IN("unmap buffer");
130 host_ptr = NULL;
131
132 /***
133 * Test image commands
134 */
135
136 cl_int color = 0;
137 const size_t origin[] = {0, 0, 0};
138 const size_t region[] = {1, 1, 1};
139
140 /* First, no associated event */
141 err = clEnqueueFillImage(queue, img, &color, origin, region,
142 1, &no_event, NULL);
143 TEST_ASSERT(err == CL_INVALID_EVENT_WAIT_LIST);
144 err = clEnqueueWriteImage(queue, img, CL_TRUE, origin, region,
145 0, 0, &color,
146 1, &no_event, NULL);
147 TEST_ASSERT(err == CL_INVALID_EVENT_WAIT_LIST);
148 err = clEnqueueWriteImage(queue, img, CL_TRUE, origin, region,
149 0, 0, &color,
150 1, &no_event, NULL);
151 TEST_ASSERT(err == CL_INVALID_EVENT_WAIT_LIST);
152 err = clEnqueueReadImage(queue, img, CL_TRUE, origin, region,
153 0, 0, &color,
154 1, &no_event, NULL);
155 TEST_ASSERT(err == CL_INVALID_EVENT_WAIT_LIST);
156
157 size_t img_pitch = 0;
158 host_ptr = clEnqueueMapImage(queue, img, CL_TRUE, CL_MAP_READ,
159 origin, region, &img_pitch, NULL,
160 1, &no_event, NULL, &err);
161 TEST_ASSERT(err == CL_INVALID_EVENT_WAIT_LIST);
162 TEST_ASSERT(host_ptr == NULL);
163
164 /* No need to test Unmap, it's the same API call as for buffers */
165
166 /* Next, associated event with initial NULL value and with initial
167 * invalid value */
168
169 for (i = 0; i < 2; ++i) {
170 cl_event initial_value = initial_values[i];
171 map_event = initial_value;
172
173 err = clEnqueueFillImage(queue, img, &color, origin, region,
174 1, &no_event, &map_event);
175 TEST_ASSERT(err == CL_INVALID_EVENT_WAIT_LIST);
176 TEST_ASSERT(map_event == initial_value);
177 err = clEnqueueWriteImage(queue, img, CL_TRUE, origin, region,
178 0, 0, &color,
179 1, &no_event, &map_event);
180 TEST_ASSERT(err == CL_INVALID_EVENT_WAIT_LIST);
181 TEST_ASSERT(map_event == initial_value);
182 err = clEnqueueWriteImage(queue, img, CL_TRUE, origin, region,
183 0, 0, &color,
184 1, &no_event, &map_event);
185 TEST_ASSERT(err == CL_INVALID_EVENT_WAIT_LIST);
186 TEST_ASSERT(map_event == initial_value);
187 err = clEnqueueReadImage(queue, img, CL_TRUE, origin, region,
188 0, 0, &color,
189 1, &no_event, &map_event);
190 TEST_ASSERT(err == CL_INVALID_EVENT_WAIT_LIST);
191 TEST_ASSERT(map_event == initial_value);
192 host_ptr = clEnqueueMapImage(queue, img, CL_TRUE, CL_MAP_READ,
193 origin, region, &img_pitch, NULL,
194 1, &no_event, &map_event, &err);
195 TEST_ASSERT(err == CL_INVALID_EVENT_WAIT_LIST);
196 TEST_ASSERT(host_ptr == NULL);
197 TEST_ASSERT(map_event == initial_value);
198
199 }
200
201 err = clEnqueueMarkerWithWaitList(queue, 1, &no_event, NULL);
202 TEST_ASSERT(err == CL_INVALID_EVENT_WAIT_LIST);
203 for (i = 0; i < 2; ++i) {
204 cl_event initial_value = initial_values[i];
205 map_event = initial_value;
206 err = clEnqueueMarkerWithWaitList(queue, 1, &no_event, &map_event);
207 TEST_ASSERT(err == CL_INVALID_EVENT_WAIT_LIST);
208 TEST_ASSERT(map_event == initial_value);
209 }
210
211 clFinish (queue);
212 clReleaseMemObject (img);
213 clReleaseMemObject (buf);
214
215 CHECK_CL_ERROR (clReleaseCommandQueue (queue));
216 CHECK_CL_ERROR (clReleaseContext (ctx));
217
218 CHECK_CL_ERROR (clUnloadCompiler ());
219
220 return EXIT_SUCCESS;
221
222}
