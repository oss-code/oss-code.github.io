24#include <CL/cl.h>
25#include <string.h>
26#include <stdio.h>
27#include <stdlib.h>
28#include "poclu.h"
29
30
31struct native_kernel_args {
32 double *a;
33 double *b;
34 double *c;
35 int size;
36};
37
38static void
39native_vec_add(void *data)
40{
41 struct native_kernel_args *args = (struct native_kernel_args *) data;
42 int i;
43
44 for(i = 0; i < args->size; i++) {
45 args->c[i] = args->a[i] + args->b[i];
46 }
47
48}
49
50int main(int argc, char **argv) {
51 unsigned int n = 100;
52
53 double *h_a;
54 double *h_b;
55 double *h_c;
56 cl_mem mem_list[3];
57 const void *args_mem_loc[3];
58
59 struct native_kernel_args args;
60
61 cl_mem d_a;
62 cl_mem d_b;
63 cl_mem d_c;
64
65 cl_context ctx;
66 cl_device_id did;
67 cl_command_queue queue;
68
69 size_t bytes = n * sizeof(double);
70
71 h_a = (double *) malloc(bytes);
72 h_b = (double *) malloc(bytes);
73 h_c = (double *) malloc(bytes);
74
75 size_t i;
76 for( i = 0; i < n; i++ )
77 {
78 h_a[i] = (double)i;
79 h_b[i] = (double)i;
80 }
81
82 cl_int err;
83
84 CHECK_CL_ERROR(poclu_get_any_device(&ctx, &did, &queue));
85 TEST_ASSERT( ctx );
86 TEST_ASSERT( did );
87 TEST_ASSERT( queue );
88
89 d_a = clCreateBuffer(ctx, CL_MEM_READ_ONLY | CL_MEM_COPY_HOST_PTR, bytes, h_a, &err);
90 CHECK_OPENCL_ERROR_IN("clCreateBuffer");
91 TEST_ASSERT(d_a);
92
93 d_b = clCreateBuffer(ctx, CL_MEM_READ_ONLY | CL_MEM_COPY_HOST_PTR, bytes, h_b, &err);
94 CHECK_OPENCL_ERROR_IN("clCreateBuffer");
95 TEST_ASSERT(d_b);
96
97 d_c = clCreateBuffer(ctx, CL_MEM_WRITE_ONLY, bytes, NULL, &err);
98 CHECK_OPENCL_ERROR_IN("clCreateBuffer");
99 TEST_ASSERT(d_c);
100
101 args.size = n;
102 args.a = 0;
103 args.b = 0;
104 args.c = 0;
105
106 mem_list[0] = d_a;
107 mem_list[1] = d_b;
108 mem_list[2] = d_c;
109
110 args_mem_loc[0] = &args.a;
111 args_mem_loc[1] = &args.b;
112 args_mem_loc[2] = &args.c;
113
114 err = clEnqueueNativeKernel ( queue, native_vec_add, &args, sizeof(struct native_kernel_args),
115 3, mem_list, args_mem_loc, 0, NULL, NULL);
116 CHECK_OPENCL_ERROR_IN("clEnqueueNativeKernel");
117
118 err = clEnqueueReadBuffer(queue, d_c, CL_TRUE, 0, bytes, h_c, 0, NULL, NULL );
119 CHECK_OPENCL_ERROR_IN("clEnqueueReadBuffer");
120
121 err = clFinish(queue);
122 CHECK_OPENCL_ERROR_IN("clFinish");
123
124 for(i = 0; i < n; i++)
125 if(h_c[i] != 2 * i)
126 {
127 printf("Fail to validate vector\n");
128 goto error;
129 }
130
131 CHECK_CL_ERROR (clReleaseMemObject (d_a));
132 CHECK_CL_ERROR (clReleaseMemObject (d_b));
133 CHECK_CL_ERROR (clReleaseMemObject (d_c));
134 CHECK_CL_ERROR (clReleaseCommandQueue (queue));
135 CHECK_CL_ERROR (clReleaseContext (ctx));
136
137 free(h_a);
138 free(h_b);
139 free(h_c);
140 CHECK_CL_ERROR (clUnloadCompiler ());
141
142 return EXIT_SUCCESS;
143
144error:
145 return EXIT_FAILURE;
146}
