23#include <CL/cl.h>
24#include <string.h>
25#include <stdio.h>
26#include <stdlib.h>
27#include <unistd.h>
28
29#include "poclu.h"
30
31#define ARRAY_SIZE(arr) (sizeof(arr)/sizeof(*arr))
32
33int main()
34{
35 cl_int err;
36 cl_event user_evt = NULL;
37 unsigned i;
38
39 // An user event can be set to either complete or a negative value, indicating error;
40 // additionally, no objects involved in a command that waits on the user event should
41 // be released before the event status is set; however, it should be possible to release
42 // everything even if the status is set to something which is NOT CL_COMPLETE. So
43 // try both CL_COMPLETE and a negative value
44 cl_int status[] = {CL_INVALID_EVENT, CL_COMPLETE };
45
46 // We also query for profiling info of the event, which according to the standard
47 // should return CL_PROFILING_INFO_NOT_AVAILABLE
48 cl_ulong queued, submitted, started, endtime;
49
50 for (i = 0; i < ARRAY_SIZE(status); ++i) {
51 cl_context context;
52 cl_command_queue queue;
53 cl_device_id device;
54
55 CHECK_CL_ERROR (poclu_get_any_device (&context, &device, &queue));
56 TEST_ASSERT( context );
57 TEST_ASSERT( device );
58 TEST_ASSERT( queue );
59
60 user_evt = clCreateUserEvent(context, &err);
61 CHECK_OPENCL_ERROR_IN("clCreateUserEvent");
62 TEST_ASSERT( user_evt );
63
64 CHECK_CL_ERROR(clSetUserEventStatus(user_evt, status[i]));
65
66 err = clGetEventProfilingInfo(user_evt, CL_PROFILING_COMMAND_QUEUED,
67 sizeof(queued), &queued, NULL);
68 TEST_ASSERT(err == CL_PROFILING_INFO_NOT_AVAILABLE);
69 err = clGetEventProfilingInfo(user_evt, CL_PROFILING_COMMAND_SUBMIT,
70 sizeof(submitted), &submitted, NULL);
71 TEST_ASSERT(err == CL_PROFILING_INFO_NOT_AVAILABLE);
72 err = clGetEventProfilingInfo(user_evt, CL_PROFILING_COMMAND_START,
73 sizeof(started), &started, NULL);
74 TEST_ASSERT(err == CL_PROFILING_INFO_NOT_AVAILABLE);
75 err = clGetEventProfilingInfo(user_evt, CL_PROFILING_COMMAND_END,
76 sizeof(endtime), &endtime, NULL);
77 TEST_ASSERT(err == CL_PROFILING_INFO_NOT_AVAILABLE);
78
79 CHECK_CL_ERROR (clReleaseEvent (user_evt));
80 CHECK_CL_ERROR (clReleaseCommandQueue (queue));
81 CHECK_CL_ERROR (clReleaseContext (context));
82 }
83
84 CHECK_CL_ERROR (clUnloadCompiler ());
85
86 return EXIT_SUCCESS;
87
88}
