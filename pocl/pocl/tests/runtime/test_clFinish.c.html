23#include <CL/cl.h>
24#include <string.h>
25#include <stdio.h>
26#include <stdlib.h>
27#include "poclu.h"
28
29char kernelASourceCode[] =
30"kernel \n"
31"void test_kernel(constant char* input) {\n"
32" printf(\"%c\", *input);\n"
33"}\n";
34
35int main()
36{
37 size_t global_work_size[1] = { 1 }, local_work_size[1]= { 1 };
38 cl_int err;
39 cl_platform_id platforms[1];
40 cl_uint nplatforms;
41 cl_device_id devices[1]; // + 1 for duplicate test
42 cl_uint num_devices;
43 cl_program program = NULL;
44 cl_kernel kernelA = NULL;
45 cl_kernel kernelB = NULL;
46 cl_kernel kernelC= NULL;
47 char inputA[] = "A";
48 char inputB[] = "B";
49 char inputC[] = "C";
50 cl_mem inputBufferA = NULL;
51 cl_mem inputBufferB = NULL;
52 cl_mem inputBufferC = NULL;
53 /* command queues */
54 cl_command_queue queueA = NULL;
55 cl_command_queue queueB = NULL;
56 cl_command_queue queueC = NULL;
57 /* events */
58 cl_event eventA1 = NULL;
59 cl_event eventB2 = NULL;
60 cl_event eventA3 = NULL;
61 cl_event eventB4 = NULL;
62 /* event wait lists */
63 cl_event B2_wait_list[1];
64 cl_event A3_wait_list[1];
65 cl_event B4_wait_list[1];
66 cl_event C5_wait_list[2];
67
68 err = clGetPlatformIDs(1, platforms, &nplatforms);
69 CHECK_OPENCL_ERROR_IN("clGetPlatformIDs");
70 if (!nplatforms)
71 return EXIT_FAILURE;
72
73 err = clGetDeviceIDs(platforms[0], CL_DEVICE_TYPE_ALL, 1,
74 devices, &num_devices);
75 CHECK_OPENCL_ERROR_IN("clGetDeviceIDs");
76
77 cl_context context = clCreateContext(NULL, num_devices, devices, NULL,
78 NULL, &err);
79 CHECK_OPENCL_ERROR_IN("clCreateContext");
80
81 err = clGetContextInfo(context, CL_CONTEXT_DEVICES,
82 sizeof(cl_device_id), devices, NULL);
83 CHECK_OPENCL_ERROR_IN("clGetContextInfo");
84
85 queueA = clCreateCommandQueue(context, devices[0], 0, &err);
86 CHECK_OPENCL_ERROR_IN("clCreateCommandQueue");
87 TEST_ASSERT(queueA);
88
89 queueB = clCreateCommandQueue(context, devices[0], 0, &err);
90 CHECK_OPENCL_ERROR_IN("clCreateCommandQueue");
91 TEST_ASSERT(queueB);
92
93 queueC = clCreateCommandQueue(context, devices[0], 0, &err);
94 CHECK_OPENCL_ERROR_IN("clCreateCommandQueue");
95 TEST_ASSERT(queueB);
96
97 inputBufferA = clCreateBuffer(context,
98 CL_MEM_READ_ONLY | CL_MEM_COPY_HOST_PTR,
99 strlen (inputB)+1, (void *) inputA, &err);
100 CHECK_OPENCL_ERROR_IN("clCreateBuffer");
101 TEST_ASSERT(inputBufferA);
102
103 inputBufferB = clCreateBuffer(context,
104 CL_MEM_READ_ONLY | CL_MEM_COPY_HOST_PTR,
105 strlen (inputA)+1, (void *) inputB, &err);
106 CHECK_OPENCL_ERROR_IN("clCreateBuffer");
107 TEST_ASSERT(inputBufferB);
108
109 inputBufferC = clCreateBuffer(context,
110 CL_MEM_READ_ONLY | CL_MEM_COPY_HOST_PTR,
111 strlen (inputA)+1, (void *) inputC, &err);
112 CHECK_OPENCL_ERROR_IN("clCreateBuffer");
113 TEST_ASSERT(inputBufferC);
114
115
116 size_t kernel_size = strlen (kernelASourceCode);
117 char* kernel_buffer = kernelASourceCode;
118
119 program = clCreateProgramWithSource (context, 1,
120 (const char**)&kernel_buffer,
121 &kernel_size, &err);
122 CHECK_OPENCL_ERROR_IN("clCreateProgramWithSource");
123
124 err = clBuildProgram (program, num_devices, devices, NULL, NULL, NULL);
125 CHECK_OPENCL_ERROR_IN("clBuildProgram");
126
127 kernelA = clCreateKernel (program, "test_kernel", NULL);
128 CHECK_OPENCL_ERROR_IN("clCreateKernel");
129 TEST_ASSERT(kernelA);
130
131 kernelB = clCreateKernel (program, "test_kernel", NULL);
132 CHECK_OPENCL_ERROR_IN("clCreateKernel");
133 TEST_ASSERT(kernelB);
134
135 kernelC = clCreateKernel (program, "test_kernel", NULL);
136 CHECK_OPENCL_ERROR_IN("clCreateKernel");
137 TEST_ASSERT(kernelC);
138
139 err = clSetKernelArg (kernelA, 0, sizeof (cl_mem), &inputBufferA);
140 CHECK_OPENCL_ERROR_IN("clSetKernelArg");
141
142 err = clSetKernelArg (kernelB, 0, sizeof (cl_mem), &inputBufferB);
143 CHECK_OPENCL_ERROR_IN("clSetKernelArg");
144
145 err = clSetKernelArg (kernelC, 0, sizeof (cl_mem), &inputBufferC);
146 CHECK_OPENCL_ERROR_IN("clSetKernelArg");
147
148
149 /* first enqueue A1*/
150 err = clEnqueueNDRangeKernel (queueA, kernelA, 1, NULL, global_work_size,
151 local_work_size, 0, NULL, &eventA1);
152 CHECK_OPENCL_ERROR_IN("clEnqueueNDRangeKernel");
153
154 /* enqueue B2 */
155 B2_wait_list[0] = eventA1;
156 err = clEnqueueNDRangeKernel (queueB, kernelB, 1, NULL, global_work_size,
157 local_work_size, 1, B2_wait_list, &eventB2);
158 CHECK_OPENCL_ERROR_IN("clEnqueueNDRangeKernel");
159
160 /* enqueue A3 */
161 A3_wait_list[0] = eventB2;
162 err = clEnqueueNDRangeKernel (queueA, kernelA, 1, NULL, global_work_size,
163 local_work_size, 1, A3_wait_list, &eventA3);
164 CHECK_OPENCL_ERROR_IN("clEnqueueNDRangeKernel");
165
166 /* enqueue B4 */
167 B4_wait_list[0] = eventA3;
168 err = clEnqueueNDRangeKernel (queueB, kernelB, 1, NULL, global_work_size,
169 local_work_size, 1, B4_wait_list, &eventB4);
170 CHECK_OPENCL_ERROR_IN("clEnqueueNDRangeKernel");
171
172 /* enqueue C5 */
173 C5_wait_list[0] = eventA3;
174 C5_wait_list[1] = eventB4;
175 err = clEnqueueNDRangeKernel (queueC, kernelC, 1, NULL, global_work_size,
176 local_work_size, 2, C5_wait_list, NULL);
177 CHECK_OPENCL_ERROR_IN("clEnqueueNDRangeKernel");
178
179 clFinish(queueC);
180 /* TODO some checks */
181
182 CHECK_CL_ERROR (clReleaseEvent (eventA1));
183 CHECK_CL_ERROR (clReleaseEvent (eventB2));
184 CHECK_CL_ERROR (clReleaseEvent (eventA3));
185 CHECK_CL_ERROR (clReleaseEvent (eventB4));
186
187 CHECK_CL_ERROR (clReleaseKernel (kernelA));
188 CHECK_CL_ERROR (clReleaseKernel (kernelB));
189 CHECK_CL_ERROR (clReleaseKernel (kernelC));
190
191 CHECK_CL_ERROR (clReleaseProgram (program));
192
193 CHECK_CL_ERROR (clReleaseCommandQueue (queueA));
194 CHECK_CL_ERROR (clReleaseCommandQueue (queueB));
195 CHECK_CL_ERROR (clReleaseCommandQueue (queueC));
196
197 CHECK_CL_ERROR (clReleaseMemObject (inputBufferA));
198 CHECK_CL_ERROR (clReleaseMemObject (inputBufferB));
199 CHECK_CL_ERROR (clReleaseMemObject (inputBufferC));
200
201 CHECK_CL_ERROR (clReleaseContext (context));
202 CHECK_CL_ERROR (clUnloadCompiler ());
203
204 printf("\n");
205 return EXIT_SUCCESS;
206
207
208}
