24#include <stdio.h>
25#include <stdlib.h>
26#include <string.h>
27#include <unistd.h>
28#include <signal.h>
29#include <CL/cl.h>
30
31#include "poclu.h"
32
33#define MAX_PLATFORMS 32
34#define MAX_DEVICES 32
35
36void timeout(int signum)
37{
38 exit(EXIT_FAILURE);
39}
40
41int
42main(void)
43{
44 cl_int err;
45 cl_platform_id platforms[MAX_PLATFORMS];
46 cl_uint nplatforms;
47 cl_device_id devices[MAX_DEVICES];
48 cl_uint ndevices;
49 cl_uint i, j;
50
51 /* set up a signal handler for ALRM that will kill
52 * the program with EXIT_FAILURE on timeout
53 */
54 struct sigaction sa;
55 memset(&sa, 0, sizeof(sa));
56 sa.sa_handler = timeout;
57 sigaction(SIGALRM, &sa, NULL);
58
59 CHECK_CL_ERROR(clGetPlatformIDs(MAX_PLATFORMS, platforms, &nplatforms));
60
61 for (i = 0; i < nplatforms; i++)
62 {
63 CHECK_CL_ERROR(clGetDeviceIDs(platforms[i], CL_DEVICE_TYPE_ALL, MAX_DEVICES,
64 devices, &ndevices));
65
66 for (j = 0; j < ndevices; j++)
67 {
68 cl_context context = clCreateContext(NULL, 1, &devices[j], NULL, NULL, &err);
69 CHECK_OPENCL_ERROR_IN("clCreateContext");
70 cl_command_queue queue = clCreateCommandQueue(context, devices[j], 0, &err);
71 CHECK_OPENCL_ERROR_IN("clCreateCommandQueue");
72
73 cl_ulong alloc;
74#define MAXALLOC (128*1024U*1024U)
75
76 CHECK_CL_ERROR(clGetDeviceInfo(devices[j], CL_DEVICE_MAX_MEM_ALLOC_SIZE,
77 sizeof(alloc), &alloc, NULL));
78
79 while (alloc > MAXALLOC)
80 alloc /= 2;
81
82 const size_t buf_size = alloc;
83
84 cl_int *host_buf1 = malloc(buf_size);
85 TEST_ASSERT(host_buf1 != NULL);
86 cl_int *host_buf2 = malloc(buf_size);
87 TEST_ASSERT(host_buf2 != NULL);
88
89 memset(host_buf1, 1, buf_size);
90 memset(host_buf2, 2, buf_size);
91
92 cl_mem buf1 = clCreateBuffer(context, CL_MEM_READ_WRITE, buf_size, NULL, &err);
93 CHECK_OPENCL_ERROR_IN("create buf1");
94 cl_mem buf2 = clCreateBuffer(context, CL_MEM_READ_WRITE, buf_size, NULL, &err);
95 CHECK_OPENCL_ERROR_IN("create buf2");
96
97 cl_event buf1_event, bufcp_event, buf2_event;
98
99 /* we test if recycling the wait list leads to neverending loops */
100 cl_event wait_list[1];
101
102 /* Note that this must be CL_TRUE because to trigger the bug the next
103 * command must have a completed event in the waiting lists */
104 CHECK_CL_ERROR(clEnqueueWriteBuffer(queue, buf1, CL_TRUE, 0, buf_size, host_buf1,
105 0, NULL, &buf1_event));
106 CHECK_OPENCL_ERROR_IN("write buf1");
107
108 *wait_list = buf1_event;
109
110 CHECK_CL_ERROR(clEnqueueCopyBuffer(queue, buf1, buf2, 0, 0, buf_size,
111 1, wait_list, &bufcp_event));
112
113 *wait_list = bufcp_event;
114
115 CHECK_CL_ERROR(clEnqueueReadBuffer(queue, buf2, CL_FALSE, 0, buf_size, host_buf2,
116 1, wait_list, &buf2_event));
117
118 /* timeout after 30 seconds: if we're not done by then, timeout() will be
119 * invoked and terminate the program with an EXIT_FAILURE */
120 alarm(30);
121
122 CHECK_CL_ERROR(clFinish(queue));
123
124 TEST_ASSERT(memcmp(host_buf2, host_buf1, buf_size) == 0);
125
126 free(host_buf2);
127 free(host_buf1);
128
129 CHECK_CL_ERROR (clReleaseEvent (buf2_event));
130 CHECK_CL_ERROR (clReleaseEvent (bufcp_event));
131 CHECK_CL_ERROR (clReleaseEvent (buf1_event));
132
133 CHECK_CL_ERROR (clReleaseMemObject (buf1));
134 CHECK_CL_ERROR (clReleaseMemObject (buf2));
135
136 CHECK_CL_ERROR (clReleaseCommandQueue (queue));
137 CHECK_CL_ERROR (clReleaseContext (context));
138 }
139 }
140
141 CHECK_CL_ERROR (clUnloadCompiler ());
142
143 return EXIT_SUCCESS;
144}
