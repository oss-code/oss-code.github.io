24#include <stdio.h>
25#include <stdlib.h>
26#include <string.h>
27#include <CL/opencl.h>
28
29#include "poclu.h"
30
31#define BUFFER_SIZE 1024
32
33// OpenCL kernel. Each work item takes care of one element of c
34const char *kernelSource = "\n"
35 "__kernel void vecAdd( __constant int *a, \n"
36 " __constant int *b, \n"
37 " __global int *c) \n"
38 "{ \n"
39 "unsigned int i = get_global_id(0) * get_global_size(1) \n"
40 " + get_global_id(1); \n"
41 "c[i] = a[i] + b[i] + get_local_id(0) * get_local_size(1) \n"
42 " + get_local_id(1); \n"
43 "} \n";
44
45const char *barrier_kernelSource =
46 "__kernel void barrier_kernel (global int *buffer) \n"
47 "{ \n"
48 " private int a = buffer[get_global_id(0)-1] \n"
49 " + buffer[get_global_id(0)] + buffer[get_global_id(0)+1]; \n"
50 " barrier(CLK_LOCAL_MEM_FENCE); \n"
51 " buffer[get_global_id(0)] = a/3; \n"
52 "} \n";
53
54const char *barrier_kernel_reqd_wg_size_source =
55 "__attribute__((reqd_work_group_size(6, 1, 1)))\n"
56 "__kernel void static_wg_kernel (global int *buffer) \n"
57 "{ \n"
58 " buffer[get_local_id (0)] = 2 + get_global_id (0); \n"
59 "} \n";
60
61int main(void)
62{
63 int *h_a;
64 int *h_b;
65 int *h_c1;
66 int *h_c2;
67 int *h_c3;
68 int *bb1;
69 int *bb2;
70 int *static_wg_buf;
71
72 cl_mem d_a;
73 cl_mem d_b;
74 cl_mem d_c1;
75 cl_mem d_c2;
76 cl_mem d_c3;
77 cl_mem barrier_buffer1;
78 cl_mem barrier_buffer2;
79 cl_mem static_wg_buffer;
80
81 cl_device_id device_id;
82 cl_context context;
83 cl_command_queue queue;
84 cl_program program1, program2;
85 cl_program b_program1, b_program2, static_wg_size_program,
86 static_wg_size_bin_program;
87 cl_kernel kernel1, kernel2, kernel3, barrier_kernel1, barrier_kernel2,
88 static_wg_kernel;
89
90 size_t globalSize[2], localSize[2];
91 cl_int err;
92
93//###################################################################
94//###################################################################
95// Initialize variables
96
97 unsigned bytes = BUFFER_SIZE * sizeof(int);
98
99 h_a = (int*)malloc(bytes);
100 h_b = (int*)malloc(bytes);
101 h_c1 = (int*)malloc(bytes);
102 h_c2 = (int*)malloc(bytes);
103 h_c3 = (int*)malloc(bytes);
104 bb1 = (int*)malloc(bytes);
105 bb2 = (int*)malloc(bytes);
106 static_wg_buf = (int*)malloc(bytes);
107
108 unsigned int i;
109 for( i = 0; i < BUFFER_SIZE; i++ )
110 {
111 h_a[i] = 2*i-1;
112 h_b[i] = -i;
113 }
114
115 for (i = 0; i < 8; ++i)
116 {
117 bb1[i] = bb2[i] = i*3;
118 }
119 memset(h_c1, 0, bytes);
120 memset(h_c2, 0, bytes);
121 memset(h_c3, 0, bytes);
122
123 /* Initialize cl_programs. */
124
125 CHECK_CL_ERROR(poclu_get_any_device(&context, &device_id, &queue));
126 TEST_ASSERT( context );
127 TEST_ASSERT( device_id );
128 TEST_ASSERT( queue );
129
130 program1 = clCreateProgramWithSource(context, 1,
131 (const char **) & kernelSource, NULL, &err);
132 TEST_ASSERT(program1);
133
134 CHECK_CL_ERROR(clBuildProgram(program1, 0, NULL, NULL, NULL, NULL));
135
136 size_t binary_sizes;
137 unsigned char *binary;
138 CHECK_CL_ERROR(clGetProgramInfo(program1, CL_PROGRAM_BINARY_SIZES,
139 sizeof(size_t), &binary_sizes, NULL));
140
141 binary = malloc(sizeof(unsigned char)*binary_sizes);
142 TEST_ASSERT (binary);
143
144 CHECK_CL_ERROR (clGetProgramInfo (program1, CL_PROGRAM_BINARIES,
145 sizeof(unsigned char*), &binary, NULL));
146
147 program2
148 = clCreateProgramWithBinary (context, 1, &device_id, &binary_sizes,
149 (const unsigned char **)(&binary), NULL, &err);
150 CHECK_OPENCL_ERROR_IN ("clCreateProgramWithBinary");
151
152 CHECK_CL_ERROR (clBuildProgram (program2, 0, NULL, NULL, NULL, NULL));
153
154 /* Barrier programs. */
155 b_program1 = clCreateProgramWithSource (context, 1,
156 (const char **)
157 &barrier_kernelSource,
158 NULL, &err);
159
160 CHECK_OPENCL_ERROR_IN("clCreateProgramWithSource");
161 TEST_ASSERT(b_program1);
162 CHECK_CL_ERROR(clBuildProgram(b_program1, 0, NULL, NULL, NULL, NULL));
163
164 static_wg_size_program
165 = clCreateProgramWithSource (context, 1,
166 (const char **)
167 &barrier_kernel_reqd_wg_size_source,
168 NULL, &err);
169
170 CHECK_OPENCL_ERROR_IN ("clCreateProgramWithSource");
171 TEST_ASSERT (static_wg_size_program);
172
173 CHECK_CL_ERROR (clBuildProgram (static_wg_size_program, 0, NULL, NULL, NULL,
174 NULL));
175
176 size_t barrier_binary_sizes;
177 unsigned char *barrier_binary;
178 CHECK_CL_ERROR(clGetProgramInfo(b_program1, CL_PROGRAM_BINARY_SIZES,
179 sizeof(size_t), &barrier_binary_sizes, NULL));
180
181 barrier_binary = malloc(sizeof(unsigned char)*barrier_binary_sizes);
182 TEST_ASSERT (barrier_binary);
183
184 CHECK_CL_ERROR(clGetProgramInfo(b_program1, CL_PROGRAM_BINARIES,
185 sizeof(unsigned char*), &barrier_binary, NULL));
186 b_program2 =
187 clCreateProgramWithBinary(context, 1, &device_id,
188 &barrier_binary_sizes,
189 (const unsigned char **)(&barrier_binary),
190 NULL, &err);
191 CHECK_OPENCL_ERROR_IN("clCreateProgramWithBinary 2");
192
193 CHECK_CL_ERROR(clBuildProgram(b_program2, 0, NULL, NULL, NULL, NULL));
194
195 size_t static_wg_binary_sizes;
196 unsigned char *static_wg_binary;
197 CHECK_CL_ERROR (clGetProgramInfo (static_wg_size_program, CL_PROGRAM_BINARY_SIZES,
198 sizeof (size_t), &static_wg_binary_sizes,
199 NULL));
200
201 static_wg_binary = malloc (sizeof (unsigned char)*static_wg_binary_sizes);
202 TEST_ASSERT (static_wg_binary);
203
204 CHECK_CL_ERROR (clGetProgramInfo (static_wg_size_program, CL_PROGRAM_BINARIES,
205 sizeof (unsigned char*), &static_wg_binary,
206 NULL));
207
208 static_wg_size_bin_program =
209 clCreateProgramWithBinary (context, 1, &device_id,
210 &static_wg_binary_sizes,
211 (const unsigned char **)(&static_wg_binary),
212 NULL, &err);
213 CHECK_OPENCL_ERROR_IN ("clCreateProgramWithBinary (static wg size)");
214
215 CHECK_CL_ERROR (clBuildProgram (static_wg_size_bin_program, 0, NULL, NULL, NULL,
216 NULL));
217
218 /* Set up buffers in memory. */
219
220 d_a = clCreateBuffer(context, CL_MEM_READ_ONLY, bytes, NULL, &err);
221 CHECK_OPENCL_ERROR_IN("clCreateBuffer");
222 TEST_ASSERT(d_a);
223 d_b = clCreateBuffer(context, CL_MEM_READ_ONLY, bytes, NULL, &err);
224 CHECK_OPENCL_ERROR_IN("clCreateBuffer");
225 TEST_ASSERT(d_b);
226
227 d_c1 = clCreateBuffer(context, CL_MEM_WRITE_ONLY, bytes, NULL, &err);
228 CHECK_OPENCL_ERROR_IN("clCreateBuffer");
229 TEST_ASSERT(d_c1);
230 d_c2 = clCreateBuffer(context, CL_MEM_WRITE_ONLY, bytes, NULL, &err);
231 CHECK_OPENCL_ERROR_IN("clCreateBuffer");
232 TEST_ASSERT(d_c2);
233 d_c3 = clCreateBuffer(context, CL_MEM_WRITE_ONLY, bytes, NULL, &err);
234 CHECK_OPENCL_ERROR_IN("clCreateBuffer");
235 TEST_ASSERT(d_c3);
236
237 barrier_buffer1 = clCreateBuffer(context, CL_MEM_COPY_HOST_PTR, bytes, bb1,
238 NULL);
239 CHECK_OPENCL_ERROR_IN("clCreateBuffer");
240 TEST_ASSERT(barrier_buffer1);
241
242 barrier_buffer2 = clCreateBuffer(context, CL_MEM_COPY_HOST_PTR, bytes, bb2,
243 NULL);
244 CHECK_OPENCL_ERROR_IN("clCreateBuffer");
245 TEST_ASSERT(barrier_buffer2);
246
247 static_wg_buffer = clCreateBuffer (context, CL_MEM_COPY_HOST_PTR, bytes,
248 static_wg_buf, NULL);
249 CHECK_OPENCL_ERROR_IN ("clCreateBuffer");
250 TEST_ASSERT (static_wg_buffer);
251
252 CHECK_CL_ERROR(clEnqueueWriteBuffer(queue, d_a, CL_TRUE, 0,
253 bytes, h_a, 0, NULL, NULL));
254 CHECK_CL_ERROR(clEnqueueWriteBuffer(queue, d_b, CL_TRUE, 0,
255 bytes, h_b, 0, NULL, NULL));
256
257 /* Create kernels. */
258
259 kernel1 = clCreateKernel(program1, "vecAdd", &err);
260 CHECK_OPENCL_ERROR_IN("clCreateKernel");
261 TEST_ASSERT(kernel1);
262
263 CHECK_CL_ERROR(clSetKernelArg(kernel1, 0, sizeof(cl_mem), &d_a));
264 CHECK_CL_ERROR(clSetKernelArg(kernel1, 1, sizeof(cl_mem), &d_b));
265 CHECK_CL_ERROR(clSetKernelArg(kernel1, 2, sizeof(cl_mem), &d_c1));
266
267 barrier_kernel1 = clCreateKernel(b_program1, "barrier_kernel", &err);
268 TEST_ASSERT(barrier_kernel1);
269
270 CHECK_CL_ERROR(clSetKernelArg(barrier_kernel1, 0, sizeof(cl_mem), &barrier_buffer1));
271
272 kernel2 = clCreateKernel(program2, "vecAdd", &err);
273 CHECK_OPENCL_ERROR_IN("clCreateKernel");
274 TEST_ASSERT(kernel2);
275
276 CHECK_CL_ERROR(clSetKernelArg(kernel2, 0, sizeof(cl_mem), &d_a));
277 CHECK_CL_ERROR(clSetKernelArg(kernel2, 1, sizeof(cl_mem), &d_b));
278 CHECK_CL_ERROR(clSetKernelArg(kernel2, 2, sizeof(cl_mem), &d_c2));
279
280 barrier_kernel2 = clCreateKernel(b_program2, "barrier_kernel", &err);
281 CHECK_OPENCL_ERROR_IN("clCreateKernel");
282 TEST_ASSERT(barrier_kernel2);
283
284 CHECK_CL_ERROR(clSetKernelArg(barrier_kernel2, 0, sizeof(cl_mem), &barrier_buffer2));
285
286 kernel3 = clCreateKernel(program2, "vecAdd", &err);
287 CHECK_OPENCL_ERROR_IN("clCreateKernel");
288 TEST_ASSERT(kernel3);
289
290 CHECK_CL_ERROR(clSetKernelArg(kernel3, 0, sizeof(cl_mem), &d_a));
291 CHECK_CL_ERROR(clSetKernelArg(kernel3, 1, sizeof(cl_mem), &d_b));
292 CHECK_CL_ERROR(clSetKernelArg(kernel3, 2, sizeof(cl_mem), &d_c3));
293
294 static_wg_kernel
295 = clCreateKernel (static_wg_size_bin_program, "static_wg_kernel", &err);
296 CHECK_OPENCL_ERROR_IN ("clCreateKernel");
297 TEST_ASSERT (static_wg_kernel);
298
299 CHECK_CL_ERROR (clSetKernelArg (static_wg_kernel, 0, sizeof (cl_mem),
300 &static_wg_buffer));
301
302 /* Enqueue kernels. */
303
304 localSize[0] = 4;
305 localSize[1] = 8;
306 globalSize[0] = 16;
307 globalSize[1] = 64;
308 CHECK_CL_ERROR(clEnqueueNDRangeKernel(queue, kernel1, 2, NULL, globalSize, localSize,
309 0, NULL, NULL));
310
311 CHECK_CL_ERROR(clEnqueueNDRangeKernel(queue, kernel2, 2, NULL, globalSize, localSize,
312 0, NULL, NULL));
313
314 localSize[0] = 2;
315 localSize[1] = 8;
316 globalSize[0] = 16;
317 globalSize[1] = 64;
318 CHECK_CL_ERROR(clEnqueueNDRangeKernel(queue, kernel3, 2, NULL, globalSize, localSize,
319 0, NULL, NULL));
320 TEST_ASSERT (!err);
321
322 localSize[0] = 6;
323 globalSize[0] = 6;
324 size_t global_offset = 1;
325 CHECK_CL_ERROR (clEnqueueNDRangeKernel (queue, barrier_kernel1, 1,
326 &global_offset,
327 globalSize,
328 localSize,
329 0, NULL, NULL));
330
331 CHECK_CL_ERROR (clEnqueueNDRangeKernel (queue, barrier_kernel2,
332 1, &global_offset,
333 globalSize,
334 localSize,
335 0, NULL, NULL));
336
337 CHECK_CL_ERROR (clEnqueueNDRangeKernel (queue, static_wg_kernel, 1,
338 &global_offset,
339 globalSize,
340 localSize,
341 0, NULL, NULL));
342
343 /* Read output buffers. */
344
345 CHECK_CL_ERROR (clEnqueueReadBuffer (queue, d_c1, CL_TRUE, 0,
346 bytes, h_c1, 0, NULL, NULL));
347 CHECK_CL_ERROR (clEnqueueReadBuffer (queue, d_c2, CL_TRUE, 0,
348 bytes, h_c2, 0, NULL, NULL));
349 CHECK_CL_ERROR (clEnqueueReadBuffer (queue, d_c3, CL_TRUE, 0,
350 bytes, h_c3, 0, NULL, NULL));
351 CHECK_CL_ERROR (clEnqueueReadBuffer (queue, barrier_buffer1, CL_TRUE, 0,
352 bytes, bb1, 0, NULL, NULL));
353 CHECK_CL_ERROR (clEnqueueReadBuffer (queue, barrier_buffer2, CL_TRUE, 0,
354 bytes, bb2, 0, NULL, NULL));
355 CHECK_CL_ERROR (clEnqueueReadBuffer (queue, static_wg_buffer, CL_TRUE, 0,
356 bytes, static_wg_buf, 0, NULL, NULL));
357 CHECK_CL_ERROR (clFinish (queue));
358
359
360 /* Check output of each kernels. */
361
362 for(i = 0; i < BUFFER_SIZE; i++)
363 {
364 if(h_c1[i] != h_c2[i])
365 {
366 printf("Check failed at offset %d, %i instead of %i\n", i, h_c2[i], h_c1[i]);
367 exit(1);
368 }
369 if ((((i/128)%2) && (h_c1[i]-16 != h_c3[i])))
370 {
371 printf("Check failed at offset %d, %i instead of %i\n", i, h_c3[i], h_c1[i]-16);
372 exit(1);
373 }
374 if (!((i/128)%2) && (h_c1[i] != h_c3[i]))
375 {
376 printf("Check failed at offset %d, %i instead of %i\n", i, h_c3[i], h_c1[i]);
377 exit(1);
378 }
379 }
380
381 for (i = 0; i < 6; ++i)
382 {
383 if (bb1[i] != bb2[i])
384 {
385 printf("barrier kernel failed at index %d: (%d != %d)\n", i, bb1[i],
386 bb2[i]);
387 return EXIT_FAILURE;
388 }
389 if (static_wg_buf[i] != 2 + (int)i + 1)
390 {
391 printf("static wg kernel failed at index %d (%d != %d)\n", i,
392 static_wg_buf[i], 2 + (int)i + 1);
393 return EXIT_FAILURE;
394 }
395 }
396
397 CHECK_CL_ERROR (clReleaseMemObject (d_a));
398 CHECK_CL_ERROR (clReleaseMemObject (d_b));
399 CHECK_CL_ERROR (clReleaseMemObject (d_c1));
400 CHECK_CL_ERROR (clReleaseMemObject (d_c2));
401 CHECK_CL_ERROR (clReleaseMemObject (d_c3));
402 CHECK_CL_ERROR (clReleaseMemObject (barrier_buffer1));
403 CHECK_CL_ERROR (clReleaseMemObject (barrier_buffer2));
404 CHECK_CL_ERROR (clReleaseMemObject (static_wg_buffer));
405
406 CHECK_CL_ERROR (clReleaseKernel (kernel1));
407 CHECK_CL_ERROR (clReleaseKernel (kernel2));
408 CHECK_CL_ERROR (clReleaseKernel (kernel3));
409 CHECK_CL_ERROR (clReleaseKernel (barrier_kernel1));
410 CHECK_CL_ERROR (clReleaseKernel (barrier_kernel2));
411 CHECK_CL_ERROR (clReleaseKernel (static_wg_kernel));
412
413 CHECK_CL_ERROR (clReleaseProgram (program1));
414 CHECK_CL_ERROR (clReleaseProgram (program2));
415 CHECK_CL_ERROR (clReleaseProgram (b_program1));
416 CHECK_CL_ERROR (clReleaseProgram (b_program2));
417 CHECK_CL_ERROR (clReleaseProgram (static_wg_size_bin_program));
418 CHECK_CL_ERROR (clReleaseProgram (static_wg_size_program));
419
420 CHECK_CL_ERROR (clReleaseCommandQueue (queue));
421 CHECK_CL_ERROR (clReleaseContext (context));
422 CHECK_CL_ERROR (clUnloadCompiler ());
423
424 free(bb1);
425 free(bb2);
426 free(h_a);
427 free(h_b);
428 free(h_c1);
429 free(h_c2);
430 free(h_c3);
431 free(binary);
432 free(barrier_binary);
433 free (static_wg_binary);
434 free (static_wg_buf);
435
436 return 0;
437}
