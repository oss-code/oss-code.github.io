24#include <stdio.h>
25#include <stdlib.h>
26#include <string.h>
27#include <CL/cl.h>
28#include "poclu.h"
29
30#define MAX_PLATFORMS 32
31#define MAX_DEVICES 32
32
33/* we will manipulate images with 4 16-bit channels, which are manadated
34 * as necessarily supported by the standard */
35static const cl_image_format img_format = {
36 .image_channel_order = CL_RGBA,
37 .image_channel_data_type = CL_UNSIGNED_INT16
38};
39
40static const size_t pixel_size = 4*sizeof(cl_ushort);
41#define CHANNEL_MAX 0xFFFFU
42
43int
44main(void)
45{
46 cl_int err;
47 cl_platform_id platforms[MAX_PLATFORMS];
48 cl_uint nplatforms;
49 cl_device_id devices[MAX_DEVICES];
50 cl_uint ndevices;
51 cl_uint i, j;
52 size_t el, row, col;
53 cl_context context;
54 cl_command_queue queue;
55 cl_mem buf, img;
56
57 CHECK_CL_ERROR(clGetPlatformIDs(MAX_PLATFORMS, platforms, &nplatforms));
58
59 for (i = 0; i < nplatforms; i++)
60 {
61 CHECK_CL_ERROR(clGetDeviceIDs(platforms[i], CL_DEVICE_TYPE_ALL, MAX_DEVICES,
62 devices, &ndevices));
63
64 /* Only test the devices we actually have room for */
65 if (ndevices > MAX_DEVICES)
66 ndevices = MAX_DEVICES;
67
68 for (j = 0; j < ndevices; j++)
69 {
70 /* skip devices that do not support images */
71 cl_bool has_img;
72 CHECK_CL_ERROR(clGetDeviceInfo(devices[j], CL_DEVICE_IMAGE_SUPPORT, sizeof(has_img), &has_img, NULL));
73 if (!has_img)
74 continue;
75
76 context = clCreateContext (NULL, 1, &devices[j], NULL, NULL, &err);
77 CHECK_OPENCL_ERROR_IN("clCreateContext");
78 queue = clCreateCommandQueue (context, devices[j], 0, &err);
79 CHECK_OPENCL_ERROR_IN("clCreateCommandQueue");
80
81 cl_ulong alloc;
82 size_t max_height;
83 size_t max_width;
84#define MAXALLOC (1024U*1024U)
85
86 CHECK_CL_ERROR(clGetDeviceInfo(devices[j], CL_DEVICE_MAX_MEM_ALLOC_SIZE,
87 sizeof(alloc), &alloc, NULL));
88 CHECK_CL_ERROR(clGetDeviceInfo(devices[j], CL_DEVICE_IMAGE2D_MAX_WIDTH,
89 sizeof(max_width), &max_width, NULL));
90 CHECK_CL_ERROR(clGetDeviceInfo(devices[j], CL_DEVICE_IMAGE2D_MAX_HEIGHT,
91 sizeof(max_height), &max_height, NULL));
92
93
94 while (alloc > MAXALLOC)
95 alloc /= 2;
96
97 // fit at least one max_width inside the alloc (shrink max_width for this)
98 while (max_width*pixel_size > alloc)
99 max_width /= 2;
100
101 // round number of elements to next multiple of max_width elements
102 const size_t nels = (alloc/pixel_size/max_width)*max_width;
103 const size_t buf_size = nels*pixel_size;
104
105 cl_image_desc img_desc;
106 memset(&img_desc, 0, sizeof(img_desc));
107 img_desc.image_type = CL_MEM_OBJECT_IMAGE2D;
108 img_desc.image_width = max_width;
109 img_desc.image_height = nels/max_width;
110 img_desc.image_depth = 1;
111
112 cl_ushort null_pixel[4] = {0, 0, 0, 0};
113 cl_ushort *host_buf = malloc(buf_size);
114 TEST_ASSERT(host_buf);
115
116 for (el = 0; el < nels; el+=4) {
117 host_buf[el] = el & CHANNEL_MAX;
118 host_buf[el+1] = (CHANNEL_MAX - el) & CHANNEL_MAX;
119 host_buf[el+2] = (CHANNEL_MAX/((el & 1) + 1) - el) & CHANNEL_MAX;
120 host_buf[el+3] = (CHANNEL_MAX - el/((el & 1) + 1)) & CHANNEL_MAX;
121 }
122
123 buf = clCreateBuffer (context, CL_MEM_READ_WRITE, buf_size, NULL, &err);
124 CHECK_OPENCL_ERROR_IN("clCreateBuffer");
125 img = clCreateImage (context, CL_MEM_READ_WRITE, &img_format, &img_desc,
126 NULL, &err);
127 CHECK_OPENCL_ERROR_IN("clCreateImage");
128
129 CHECK_CL_ERROR(clEnqueueWriteBuffer(queue, buf, CL_TRUE, 0, buf_size, host_buf, 0, NULL, NULL));
130
131 const size_t offset = 0;
132 const size_t origin[] = {0, 0, 0};
133 const size_t region[] = {img_desc.image_width, img_desc.image_height, 1};
134
135 CHECK_CL_ERROR(clEnqueueCopyBufferToImage(queue, buf, img, offset, origin, region, 0, NULL, NULL));
136
137 size_t row_pitch, slice_pitch;
138 cl_ushort *img_map = clEnqueueMapImage(queue, img, CL_TRUE, CL_MAP_READ, origin, region,
139 &row_pitch, &slice_pitch, 0, NULL, NULL, &err);
140 CHECK_OPENCL_ERROR_IN("clEnqueueMapImage");
141
142 CHECK_CL_ERROR(clFinish(queue));
143
144 for (row = 0; row < img_desc.image_height; ++row) {
145 for (col = 0; col < img_desc.image_width; ++col) {
146 cl_ushort *img_pixel = (cl_ushort*)((char*)img_map + row*row_pitch) + col*4;
147 cl_ushort *buf_pixel = host_buf + (row*img_desc.image_width + col)*4;
148
149 if (memcmp(img_pixel, buf_pixel, pixel_size) != 0)
150 printf("%zu %zu %zu : %x %x %x %x | %x %x %x %x\n",
151 row, col, (size_t)(buf_pixel - host_buf),
152 buf_pixel[0],
153 buf_pixel[1],
154 buf_pixel[2],
155 buf_pixel[3],
156 img_pixel[0],
157 img_pixel[1],
158 img_pixel[2],
159 img_pixel[3]);
160
161 TEST_ASSERT(memcmp(img_pixel, buf_pixel, pixel_size) == 0);
162
163 }
164 }
165
166 CHECK_CL_ERROR(clEnqueueUnmapMemObject(queue, img, img_map, 0, NULL, NULL));
167
168 /* Clear the buffer, and ensure it has been cleared */
169 CHECK_CL_ERROR(clEnqueueFillBuffer(queue, buf, null_pixel, sizeof(null_pixel), 0, buf_size, 0, NULL, NULL));
170 cl_ushort *buf_map = clEnqueueMapBuffer(queue, buf, CL_TRUE, CL_MAP_READ, 0, buf_size, 0, NULL, NULL, &err);
171 CHECK_OPENCL_ERROR_IN("clEnqueueMapBuffer");
172
173 CHECK_CL_ERROR(clFinish(queue));
174
175 for (el = 0; el < nels; ++el) {
176#if 0 // debug
177 if (buf_map[el] != 0) {
178 printf("%zu/%zu => %u\n", el, nels, buf_map[el]);
179 }
180#endif
181 TEST_ASSERT(buf_map[el] == 0);
182 }
183
184 CHECK_CL_ERROR(clEnqueueUnmapMemObject(queue, buf, buf_map, 0, NULL, NULL));
185
186 /* Copy data from image to buffer, and check that it's again equal to the original buffer */
187 CHECK_CL_ERROR(clEnqueueCopyImageToBuffer(queue, img, buf, origin, region, offset, 0, NULL, NULL));
188 buf_map = clEnqueueMapBuffer(queue, buf, CL_TRUE, CL_MAP_READ, 0, buf_size, 0, NULL, NULL, &err);
189 CHECK_CL_ERROR(clFinish(queue));
190
191 TEST_ASSERT(memcmp(buf_map, host_buf, buf_size) == 0);
192
193 CHECK_CL_ERROR (
194 clEnqueueUnmapMemObject (queue, buf, buf_map, 0, NULL, NULL));
195 CHECK_CL_ERROR (clFinish (queue));
196
197 free(host_buf);
198 CHECK_CL_ERROR (clReleaseMemObject (img));
199 CHECK_CL_ERROR (clReleaseMemObject (buf));
200 CHECK_CL_ERROR (clReleaseCommandQueue (queue));
201 CHECK_CL_ERROR (clReleaseContext (context));
202 }
203 }
204
205 CHECK_CL_ERROR (clUnloadCompiler ());
206 return EXIT_SUCCESS;
207}
