32char kernelSourceCode[] =
33"constant sampler_t samp = CLK_NORMALIZED_COORDS_FALSE | CLK_ADDRESS_NONE | CLK_FILTER_NEAREST;\n"
34"typedef float4 tasty_float;\n"
35"kernel \n"
36"__attribute__((reqd_work_group_size(256, 1, 1)))"
37"void test_kernel2(const tasty_float vec1, __read_only image2d_t in, write_only image2d_t out) {\n"
38" const int2 coord = (get_global_id(0),get_global_id(1));\n"
39" float4 b = read_imagef(in, samp, coord);\n"
40" float4 col = b*vec1;\n"
41" write_imagef(out, coord, col);\n"
42"}\n"
43"kernel \n"
44"void test_kernel(constant char* msg, global volatile float* in, global float* out, const float j, local int* c) {\n"
45" printf(\"%s\", msg);\n"
46" const int i = get_global_id(0);\n"
47" out[i] = in[i] * native_sin(j*c[i%50]);\n"
48"}\n";

60int
61test_program (cl_program program, int is_spir)
62{
63
64 cl_int err;
65 size_t retsize;
66 cl_kernel test_kernel = NULL;
67 cl_kernel test_kernel2 = NULL;
68 union {
69 cl_kernel_arg_address_qualifier address;
70 cl_kernel_arg_access_qualifier access;
71 cl_kernel_arg_type_qualifier type;
72 char string[BUF_LEN];
73 } kernel_arg;
74 unsigned i;
75
76 test_kernel = clCreateKernel (program, "test_kernel", &err);
77 CHECK_OPENCL_ERROR_IN("clCreateKernel");
78
79 test_kernel2 = clCreateKernel (program, "test_kernel2", &err);
80 CHECK_OPENCL_ERROR_IN("clCreateKernel");
81
82 /* ADDR SPACE QUALIFIER tests */
83 // constant char* msg, global volatile float* in, global float* out, const float j, local int* c
84 err = clGetKernelArgInfo(test_kernel, 0, CL_KERNEL_ARG_ADDRESS_QUALIFIER,
85 BUF_LEN, &kernel_arg.address, &retsize);
86 CHECK_OPENCL_ERROR_IN("clGetKernelArgInfo");
87 printf ("adr: %x\n", kernel_arg.address);
88 TEST_ASSERT((kernel_arg.address==CL_KERNEL_ARG_ADDRESS_CONSTANT) &&
89 "arg of test_kernel is not CONSTANT addr space");
90
91 err = clGetKernelArgInfo(test_kernel, 1, CL_KERNEL_ARG_ADDRESS_QUALIFIER,
92 BUF_LEN, &kernel_arg.address, &retsize);
93 CHECK_OPENCL_ERROR_IN("clGetKernelArgInfo");
94 TEST_ASSERT((kernel_arg.address==CL_KERNEL_ARG_ADDRESS_GLOBAL) &&
95 "arg of test_kernel is not GLOBAL addr space");
96
97
98 err = clGetKernelArgInfo(test_kernel, 2, CL_KERNEL_ARG_ADDRESS_QUALIFIER,
99 BUF_LEN, &kernel_arg.address, &retsize);
100 CHECK_OPENCL_ERROR_IN("clGetKernelArgInfo");
101 TEST_ASSERT((kernel_arg.address==CL_KERNEL_ARG_ADDRESS_GLOBAL) &&
102 "arg of test_kernel is not GLOBAL addr space");
103
104
105 err = clGetKernelArgInfo(test_kernel, 3, CL_KERNEL_ARG_ADDRESS_QUALIFIER,
106 BUF_LEN, &kernel_arg.address, &retsize);
107 CHECK_OPENCL_ERROR_IN("clGetKernelArgInfo");
108 TEST_ASSERT((kernel_arg.address==CL_KERNEL_ARG_ADDRESS_PRIVATE) &&
109 "arg of test_kernel is not PRIVATE addr space");
110
111
112 err = clGetKernelArgInfo(test_kernel, 4, CL_KERNEL_ARG_ADDRESS_QUALIFIER,
113 BUF_LEN, &kernel_arg.address, &retsize);
114 CHECK_OPENCL_ERROR_IN("clGetKernelArgInfo");
115 TEST_ASSERT((kernel_arg.address==CL_KERNEL_ARG_ADDRESS_LOCAL) &&
116 "arg of test_kernel is not LOCAL addr space");
117
118 /* ACCESS QUALIFIER tests */
119 // constant char* msg, global volatile float* in, global float* out, const float j, local int* c
120
121 for (i = 0; i<5; ++i) {
122 err = clGetKernelArgInfo(test_kernel, i, CL_KERNEL_ARG_ACCESS_QUALIFIER,
123 BUF_LEN, &kernel_arg.access, &retsize);
124 CHECK_OPENCL_ERROR_IN("clGetKernelArgInfo");
125 TEST_ASSERT((kernel_arg.access==CL_KERNEL_ARG_ACCESS_NONE) &&
126 "arg of test_kernel is not NONE access");
127 }
128
129 /* TYPE NAME tests */
130 // constant char* msg, global volatile float* in, global float* out, const float j, local int* c
131
132 err = clGetKernelArgInfo(test_kernel, 0, CL_KERNEL_ARG_TYPE_NAME,
133 BUF_LEN, &kernel_arg.string, &retsize);
134 CHECK_OPENCL_ERROR_IN("clGetKernelArgInfo");
135 TEST_ASSERT ((retsize == 6)
136 && " arg type name size of test_kernel doesn't fit");
137 TEST_ASSERT ((strncmp (kernel_arg.string, "char*", 6) == 0)
138 && " arg type name of test_kernel doesn't compare");
139
140 err = clGetKernelArgInfo(test_kernel, 1, CL_KERNEL_ARG_TYPE_NAME,
141 BUF_LEN, &kernel_arg.string, &retsize);
142 CHECK_OPENCL_ERROR_IN("clGetKernelArgInfo");
143 TEST_ASSERT ((retsize == 7)
144 && " arg type name size of test_kernel doesn't fit");
145 TEST_ASSERT ((strncmp (kernel_arg.string, "float*", 7) == 0)
146 && " arg type name of test_kernel doesn't compare");
147
148 err = clGetKernelArgInfo(test_kernel, 3, CL_KERNEL_ARG_TYPE_NAME,
149 BUF_LEN, &kernel_arg.string, &retsize);
150 CHECK_OPENCL_ERROR_IN("clGetKernelArgInfo");
151 TEST_ASSERT ((retsize == 6)
152 && " arg type name size of test_kernel doesn't fit");
153 TEST_ASSERT ((strncmp (kernel_arg.string, "float", 6) == 0)
154 && " arg type name of test_kernel doesn't compare");
155
156 err = clGetKernelArgInfo(test_kernel, 4, CL_KERNEL_ARG_TYPE_NAME,
157 BUF_LEN, &kernel_arg.string, &retsize);
158 CHECK_OPENCL_ERROR_IN("clGetKernelArgInfo");
159 TEST_ASSERT ((retsize == 5)
160 && " arg type name size of test_kernel doesn't fit");
161 TEST_ASSERT ((strncmp (kernel_arg.string, "int*", 5) == 0)
162 && " arg type name of test_kernel doesn't compare");
163
164 /* TYPE QUALIFIER tests */
165 // constant char* msg, global volatile float* in, global float* out, const float j, local int* c
166
167 err = clGetKernelArgInfo(test_kernel, 0, CL_KERNEL_ARG_TYPE_QUALIFIER,
168 BUF_LEN, &kernel_arg.type, &retsize);
169 CHECK_OPENCL_ERROR_IN("clGetKernelArgInfo");
170 TEST_ASSERT((kernel_arg.type==CL_KERNEL_ARG_TYPE_CONST) &&
171 "type qualifier of arg of test_kernel is not CONST");
172
173 err = clGetKernelArgInfo(test_kernel, 1, CL_KERNEL_ARG_TYPE_QUALIFIER,
174 BUF_LEN, &kernel_arg.type, &retsize);
175 CHECK_OPENCL_ERROR_IN("clGetKernelArgInfo");
176 TEST_ASSERT((kernel_arg.type==CL_KERNEL_ARG_TYPE_VOLATILE) &&
177 "type qualifier of arg of test_kernel is not VOLATILE");
178
179 err = clGetKernelArgInfo(test_kernel, 2, CL_KERNEL_ARG_TYPE_QUALIFIER,
180 BUF_LEN, &kernel_arg.type, &retsize);
181 CHECK_OPENCL_ERROR_IN("clGetKernelArgInfo");
182 TEST_ASSERT((kernel_arg.type==CL_KERNEL_ARG_TYPE_NONE) &&
183 "type qualifier of arg of test_kernel is not NONE");
184
185 if (!is_spir) {
186
187 /* Clang versions before 5 added the const MD also for non-pointer
188 types even though OpenCL specs mandates to put them only in
189 pointer args. This was fixed in Clang r299192 (see below).
190 TODO: update the SPIRs with 5.0+ Clang to drop the metadata. */
191
192 /*
193 r299192 | echuraev | 2017-03-31 13:14:52 +0300 (Fri, 31 Mar 2017) | 26 lines
194
195 [OpenCL] Do not generate "kernel_arg_type_qual" metadata for non-pointer args
196
197 Summary:
198 "kernel_arg_type_qual" metadata should contain const/volatile/restrict
199 tags only for pointer types to match the corresponding requirement of
200 the OpenCL specification.
201
202 OpenCL 2.0 spec 5.9.3 Kernel Object Queries:
203
204 CL_KERNEL_ARG_TYPE_VOLATILE is returned if the argument is a pointer
205 and the referenced type is declared with the volatile qualifier.
206 [...]
207 Similarly, CL_KERNEL_ARG_TYPE_CONST is returned if the argument is a
208 pointer and the referenced type is declared with the restrict or const
209 qualifier.
210 [...]
211 CL_KERNEL_ARG_TYPE_RESTRICT will be returned if the pointer type is
212 marked restrict.
213 */
214 err = clGetKernelArgInfo(test_kernel, 3, CL_KERNEL_ARG_TYPE_QUALIFIER,
215 BUF_LEN, &kernel_arg.type, &retsize);
216 CHECK_OPENCL_ERROR_IN("clGetKernelArgInfo");
217 TEST_ASSERT((kernel_arg.type == CL_KERNEL_ARG_TYPE_NONE));
218 }
219
220 /* NAME tests */
221 // constant char* msg, global volatile float* in, global float* out, const float j, local int* c
222
223 err = clGetKernelArgInfo(test_kernel, 0, CL_KERNEL_ARG_NAME,
224 BUF_LEN, &kernel_arg.string, &retsize);
225 if (is_spir && (err == CL_KERNEL_ARG_INFO_NOT_AVAILABLE)) {
226 printf("arg name not available (this is normal for SPIR)\n");
227 } else {
228 CHECK_OPENCL_ERROR_IN("clGetKernelArgInfo");
229 TEST_ASSERT ((retsize == 4)
230 && " arg name size of test_kernel doesn't fit");
231 TEST_ASSERT ((strncmp (kernel_arg.string, "msg", 4) == 0)
232 && " arg name of test_kernel doesn't compare");
233 }
234
235 err = clGetKernelArgInfo(test_kernel, 1, CL_KERNEL_ARG_NAME,
236 BUF_LEN, &kernel_arg.string, &retsize);
237 if (is_spir && (err == CL_KERNEL_ARG_INFO_NOT_AVAILABLE)) {
238 printf("arg name not available (this is normal for SPIR)\n");
239 } else {
240 CHECK_OPENCL_ERROR_IN("clGetKernelArgInfo");
241 TEST_ASSERT ((retsize == 3)
242 && " arg name size of test_kernel doesn't fit");
243 TEST_ASSERT ((strncmp (kernel_arg.string, "in", 3) == 0)
244 && " arg name of test_kernel doesn't compare");
245 }
246
247 err = clGetKernelArgInfo(test_kernel, 2, CL_KERNEL_ARG_NAME,
248 BUF_LEN, &kernel_arg.string, &retsize);
249 if (is_spir && (err == CL_KERNEL_ARG_INFO_NOT_AVAILABLE)) {
250 printf("arg name not available (this is normal for SPIR)\n");
251 } else {
252 CHECK_OPENCL_ERROR_IN("clGetKernelArgInfo");
253 TEST_ASSERT ((retsize == 4)
254 && " arg name size of test_kernel doesn't fit");
255 TEST_ASSERT ((strncmp (kernel_arg.string, "out", 4) == 0)
256 && " arg name of test_kernel doesn't compare");
257 }
258
259 err = clGetKernelArgInfo(test_kernel, 3, CL_KERNEL_ARG_NAME,
260 BUF_LEN, &kernel_arg.string, &retsize);
261 if (is_spir && (err == CL_KERNEL_ARG_INFO_NOT_AVAILABLE)) {
262 printf("arg name not available (this is normal for SPIR)\n");
263 } else {
264 CHECK_OPENCL_ERROR_IN("clGetKernelArgInfo");
265 TEST_ASSERT ((retsize == 2)
266 && " arg name size of test_kernel doesn't fit");
267 TEST_ASSERT ((strncmp (kernel_arg.string, "j", 2) == 0)
268 && " arg name of test_kernel doesn't compare");
269 }
270
271 /* ACCESS QUALIFIER tests for test_kernel2 */
272 // const float4 vec1, __read_only image2d_t in, write_only image2d_t out
273 err = clGetKernelArgInfo(test_kernel2, 0, CL_KERNEL_ARG_ACCESS_QUALIFIER,
274 BUF_LEN, &kernel_arg.access, &retsize);
275 CHECK_OPENCL_ERROR_IN("clGetKernelArgInfo");
276 TEST_ASSERT((kernel_arg.access==CL_KERNEL_ARG_ACCESS_NONE) && "arg of test_kernel2 is not NONE access");
277
278 err = clGetKernelArgInfo(test_kernel2, 1, CL_KERNEL_ARG_ACCESS_QUALIFIER,
279 BUF_LEN, &kernel_arg.access, &retsize);
280 CHECK_OPENCL_ERROR_IN("clGetKernelArgInfo");
281 TEST_ASSERT((kernel_arg.access==CL_KERNEL_ARG_ACCESS_READ_ONLY) && "arg of test_kernel2 is not READ_ONLY access");
282
283 err = clGetKernelArgInfo(test_kernel2, 2, CL_KERNEL_ARG_ACCESS_QUALIFIER,
284 BUF_LEN, &kernel_arg.access, &retsize);
285 CHECK_OPENCL_ERROR_IN("clGetKernelArgInfo");
286 TEST_ASSERT((kernel_arg.access==CL_KERNEL_ARG_ACCESS_WRITE_ONLY) && "arg of test_kernel2 is not WRITE_ONLY access");
287
288 /* check typedef-ed arg type name */
289
290 err = clGetKernelArgInfo(test_kernel2, 0, CL_KERNEL_ARG_TYPE_NAME,
291 BUF_LEN, &kernel_arg.string, &retsize);
292 CHECK_OPENCL_ERROR_IN("clGetKernelArgInfo");
293 TEST_ASSERT ((retsize == 12)
294 && " arg type name size of test_kernel doesn't fit");
295 TEST_ASSERT ((strncmp (kernel_arg.string, "tasty_float", 12) == 0)
296 && " arg type name of test_kernel2 doesn't compare");
297
298 err = clReleaseKernel(test_kernel);
299 CHECK_OPENCL_ERROR_IN("clReleaseKernel");
300 err = clReleaseKernel(test_kernel2);
301 CHECK_OPENCL_ERROR_IN("clReleaseKernel");
302 return EXIT_SUCCESS;
303}

345int
346spir_program (const char *filename, cl_context ctx, cl_device_id did,
347 cl_program *program)
348{
349 cl_int err;
350
351 size_t program_size;
352 char* program_buffer;
353
354 program_buffer = poclu_read_binfile(filename, &program_size);
355 TEST_ASSERT(program_size > 2000);
356 TEST_ASSERT(program_buffer != NULL);
357 printf("program size: %zi\n", program_size);
358
359 *program = clCreateProgramWithBinary (ctx, 1, &did, &program_size,
360 (const unsigned char**)&program_buffer,
361 NULL, &err);
362 CHECK_OPENCL_ERROR_IN("clCreateProgramWithBinary");
363 TEST_ASSERT(program);
364
365 CHECK_CL_ERROR(clBuildProgram (*program, 1, &did, NULL, NULL, NULL));
366
367 free (program_buffer);
368
369 return EXIT_SUCCESS;
370}

372int main()
373{
374 cl_int err;
375
376 cl_context ctx;
377 cl_device_id did;
378 cl_command_queue queue;
379
380 size_t program_size;
381 char* program_buffer;
382
383 cl_program program = NULL;
384
385 poclu_get_any_device(&ctx, &did, &queue);
386 TEST_ASSERT(ctx);
387 TEST_ASSERT(did);
388 TEST_ASSERT(queue);
389
390 /* regular non-SPIR program */
391
392 program_size = strlen (kernelSourceCode);
393 program_buffer = kernelSourceCode;
394
395 cl_uint address_bits;
396 err = clGetDeviceInfo (did, CL_DEVICE_ADDRESS_BITS, sizeof (address_bits),
397 &address_bits, NULL);
398 CHECK_OPENCL_ERROR_IN ("clGetDeviceInfo");
399
400 program = clCreateProgramWithSource (ctx, 1, (const char **)&program_buffer,
401 &program_size, &err);
402 CHECK_OPENCL_ERROR_IN("clCreateProgramWithBinary");
403 TEST_ASSERT(program);
404
405 err = clBuildProgram (program, 1, &did, NULL, NULL, NULL);
406 CHECK_OPENCL_ERROR_IN("clBuildProgram");
407
408 printf("\nNON-SPIR\n");
409 TEST_ASSERT(test_program(program, 0) == EXIT_SUCCESS);
410
411 CHECK_CL_ERROR(clReleaseProgram(program));
412
413 char extensions[1024];
414 err = clGetDeviceInfo(did, CL_DEVICE_EXTENSIONS, 1024, extensions, NULL);
415 CHECK_OPENCL_ERROR_IN("clGetDeviceInfo");
416 if (strstr(extensions, "cl_khr_spir") == NULL)
417 {
418 printf ("SPIR not supported, skipping SPIR arg info tests\n");
419 goto FINISH;
420 }
421
422 /* SPIR program */
423
424 printf ("\nSPIR with metadata\n");
425
426 const char *filename = (address_bits == 32 ? meta_files[0] : meta_files[1]);
427 TEST_ASSERT (spir_program (filename, ctx, did, &program) == EXIT_SUCCESS);
428
429 TEST_ASSERT (test_program (program, 1) == EXIT_SUCCESS);
430
431 CHECK_CL_ERROR(clReleaseProgram(program));
432
433 /* SPIR program without metadata - currently disabled since Clang seems to
434 * always generate metadata. */
435 /*
436 printf("\nSPIR WITHOUT metadata\n");
437 filename = (address_bits == 32 ? nometa_files[0] : nometa_files[1]);
438 TEST_ASSERT(spir_program(filename, ctx, did, &program) == EXIT_SUCCESS);

445FINISH:
446 CHECK_CL_ERROR (clReleaseCommandQueue (queue));
447 CHECK_CL_ERROR (clReleaseContext (ctx));
448 CHECK_CL_ERROR (clUnloadCompiler ());
449
450 printf("\nOK\n");
451 return EXIT_SUCCESS;
