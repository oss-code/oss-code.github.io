1#include <stdio.h>
2#include <stdlib.h>
3#include <string.h>
4#include <CL/opencl.h>
5#include "poclu.h"
6#include "config.h"
7
8static const char *empty_src = "\n";
9
10int main(int argc, char **argv)
11{
12 cl_int err;
13 const char *krn_src;
14 cl_program empty, program;
15 cl_context ctx;
16 cl_device_id did;
17 cl_command_queue queue;
18 cl_uint num_krn;
19 cl_kernel kernels[2];
20
21 err = poclu_get_any_device(&ctx, &did, &queue);
22 CHECK_OPENCL_ERROR_IN("poclu_get_any_device");
23 TEST_ASSERT( ctx );
24 TEST_ASSERT( did );
25 TEST_ASSERT( queue );
26
27 /* Test creating a program from an empty source */
28 empty = clCreateProgramWithSource(ctx, 1, &empty_src, NULL, &err);
29 CHECK_OPENCL_ERROR_IN("clCreateProgramWithSource");
30 err = clBuildProgram(empty, 0, NULL, NULL, NULL, NULL);
31 CHECK_OPENCL_ERROR_IN("clBuildProgram");
32
33 err = clCreateKernelsInProgram(empty, 0, NULL, &num_krn);
34 CHECK_OPENCL_ERROR_IN("clCreateKernelsInProgram");
35 TEST_ASSERT(num_krn == 0);
36
37 krn_src = poclu_read_file(SRCDIR "/tests/runtime/test_clCreateKernelsInProgram.cl");
38 TEST_ASSERT(krn_src);
39
40 program = clCreateProgramWithSource(ctx, 1, &krn_src, NULL, &err);
41 CHECK_OPENCL_ERROR_IN("clCreateProgramWithSource");
42 err = clBuildProgram(program, 0, NULL, NULL, NULL, NULL);
43 CHECK_OPENCL_ERROR_IN("clBuildProgram");
44
45 err = clCreateKernelsInProgram(program, 0, NULL, &num_krn);
46 CHECK_OPENCL_ERROR_IN("clCreateKernelsInProgram");
47 // test_clCreateKernelsInProgram.cl has two kernel functions.
48 TEST_ASSERT(num_krn == 2);
49
50 err = clCreateKernelsInProgram(program, 2, kernels, NULL);
51 CHECK_OPENCL_ERROR_IN("clCreateKernelsInProgram");
52
53 // make sure the kernels were actually created
54 // Note: nothing in the specification says which kernel function
55 // is kernels[0], which is kernels[1]. For now assume pocl/LLVM
56 // orders these deterministacally
57 err = clEnqueueTask(queue, kernels[0], 0, NULL, NULL);
58 CHECK_OPENCL_ERROR_IN("clEnqueueTask");
59
60 err = clFinish(queue);
61 CHECK_OPENCL_ERROR_IN("clFinish");
62
63 err = clEnqueueTask(queue, kernels[1], 0, NULL, NULL);
64 CHECK_OPENCL_ERROR_IN("clEnqueueTask");
65
66 err = clFinish(queue);
67 CHECK_OPENCL_ERROR_IN("clFinish");
68
69 CHECK_CL_ERROR (clReleaseCommandQueue (queue));
70 CHECK_CL_ERROR (clReleaseKernel (kernels[0]));
71 CHECK_CL_ERROR (clReleaseKernel (kernels[1]));
72 CHECK_CL_ERROR (clReleaseProgram (program));
73 CHECK_CL_ERROR (clReleaseProgram (empty));
74 CHECK_CL_ERROR (clReleaseContext (ctx));
75 CHECK_CL_ERROR (clUnloadCompiler ());
76
77 free ((void *)krn_src);
78
79 return EXIT_SUCCESS;
80}
