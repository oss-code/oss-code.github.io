31/* Two different kernel sources, to ensure that the test for all devices
32 * and the test with only sub-devices do not interphere with each other
33 * (cache-wise)
34 */
35static const char *prog_src_all = "kernel void\n"
36"setidx(global int *buf, int idx) {\n"
37"buf[idx] = idx;\n"
38"}\n";
39
40static const char *prog_src_two = "kernel void\n"
41"setidx(global int *buf, int idx) {\n"
42"buf[idx] = -idx;\n"
43"}\n";
44
45
46int test_context(cl_context ctx, const char *prog_src, int mul,
47 int ndevs, cl_device_id *devs) {
48 cl_int err;
49 cl_command_queue queue[ndevs];
50 cl_program prog;
51 cl_kernel krn;
52 cl_mem buf;
53 cl_event evt[ndevs];
54 cl_int i;
55
56 prog = clCreateProgramWithSource(ctx, 1, &prog_src, NULL, &err);
57 CHECK_OPENCL_ERROR_IN("create program");
58
59 CHECK_CL_ERROR(clBuildProgram(prog, 0, NULL, NULL, NULL, NULL));
60
61 krn = clCreateKernel(prog, "setidx", &err);
62 CHECK_OPENCL_ERROR_IN("create kernel");
63
64 buf = clCreateBuffer(ctx, CL_MEM_ALLOC_HOST_PTR | CL_MEM_READ_WRITE |
65 CL_MEM_HOST_READ_ONLY, ndevs*sizeof(cl_int), NULL, &err);
66 CHECK_OPENCL_ERROR_IN("create buffer");
67
68 CHECK_CL_ERROR(clSetKernelArg(krn, 0, sizeof(cl_mem), &buf));
69
70 /* create one queue per device, and submit task, waiting for all
71 * previous */
72 for (i = 0; i < ndevs; ++i) {
73 queue[i] = clCreateCommandQueue(ctx, devs[i], 0, &err);
74 CHECK_OPENCL_ERROR_IN("create queue");
75 err = clSetKernelArg(krn, 1, sizeof(i), &i);
76 CHECK_OPENCL_ERROR_IN("set kernel arg 1");
77 // no wait list for first (root) device
78 err = clEnqueueTask(queue[i], krn, i, i ? evt : NULL, evt + i);
79 CHECK_OPENCL_ERROR_IN("submit task");
80 }
81
82 /* enqueue map on last */
83 cl_int *buf_host = clEnqueueMapBuffer(queue[ndevs - 1], buf, CL_TRUE,
84 CL_MAP_READ, 0, ndevs*sizeof(cl_int), ndevs, evt, NULL, &err);
85 CHECK_OPENCL_ERROR_IN("map buffer");
86
87 int mismatch = 0;
88 for (i = 0; i < ndevs; ++i) {
89 mismatch += !!(buf_host[i] != i*mul);
90 }
91 TEST_ASSERT(mismatch == 0);
92
93 /* enqueue unmap on first */
94 CHECK_CL_ERROR(clEnqueueUnmapMemObject(queue[0], buf, buf_host, 0, NULL, NULL));
95
96 for (i = 0 ; i < ndevs; ++i) {
97 err = clFinish(queue[i]);
98 err |= clReleaseCommandQueue(queue[i]);
99 err |= clReleaseEvent(evt[i]);
100 }
101
102 err |= clReleaseKernel(krn);
103 err |= clReleaseMemObject(buf);
104 err |= clReleaseProgram(prog);
105 err |= clReleaseContext(ctx);
106
107 CHECK_OPENCL_ERROR_IN("cleanup");
108
109 return CL_SUCCESS;
110
111}
112
113int main(int argc, char **argv)
114{
115 cl_context ctx;
116 cl_command_queue q;
117 // root device, all devices
118#define NUMDEVS 6
119 cl_device_id rootdev, alldevs[NUMDEVS];
120 // pointers to the sub devices of the partitions EQUALLY and BY_COUNTS
121 // respectively
122 cl_device_id
123 *eqdev = alldevs + 1,
124 *countdev = alldevs + 4;
125 cl_uint max_cus, max_subs, split;
126 cl_uint i, j;
127
128 cl_int err = poclu_get_any_device(&ctx, &rootdev, &q);
129 CHECK_OPENCL_ERROR_IN("poclu_get_any_device");
130 TEST_ASSERT( ctx );
131 TEST_ASSERT( rootdev );
132 TEST_ASSERT( q );
133
134 alldevs[0] = rootdev;
135
136 err = clGetDeviceInfo(rootdev, CL_DEVICE_MAX_COMPUTE_UNITS,
137 sizeof(max_cus), &max_cus, NULL);
138 CHECK_OPENCL_ERROR_IN("CL_DEVICE_MAX_COMPUTE_UNITS");
139 if (max_cus < 2)
140 {
141 printf("This test requires a cl device with at least 2 compute units"
142 " (a dual-core or better CPU)\n");
143 return 1;
144 }
145
146 err = clGetDeviceInfo(rootdev, CL_DEVICE_PARTITION_MAX_SUB_DEVICES,
147 sizeof(max_subs), &max_subs, NULL);
148 CHECK_OPENCL_ERROR_IN("CL_DEVICE_PARTITION_MAX_SUB_DEVICES");
149
150 // test fails without possible sub-devices, e.g. with basic pocl device
151 TEST_ASSERT(max_subs > 1);
152
153 cl_device_partition_property *dev_pt;
154 size_t dev_pt_size;
155
156 err = clGetDeviceInfo(rootdev, CL_DEVICE_PARTITION_PROPERTIES,
157 0, NULL, &dev_pt_size);
158 CHECK_OPENCL_ERROR_IN("CL_DEVICE_PARTITION_PROPERTIES size");
159
160 dev_pt = malloc(dev_pt_size);
161 TEST_ASSERT(dev_pt);
162 err = clGetDeviceInfo(rootdev, CL_DEVICE_PARTITION_PROPERTIES,
163 dev_pt_size, dev_pt, NULL);
164 CHECK_OPENCL_ERROR_IN("CL_DEVICE_PARTITION_PROPERTIES");
165
166 j = dev_pt_size / sizeof (*dev_pt); // number of partition types
167
168 // check that partition types EQUALLY and BY_COUNTS are supported
169 int found = 0;
170 for (i = 0; i < j; ++i)
171 {
172 if (dev_pt[i] == CL_DEVICE_PARTITION_EQUALLY
173 || dev_pt[i] == CL_DEVICE_PARTITION_BY_COUNTS)
174 ++found;
175 }
176
177 TEST_ASSERT(found == 2);
178
179 // here we will store the partition types returned by the subdevices
180 cl_device_partition_property *ptype = NULL;
181 size_t ptype_size;
182 cl_uint numdevs = 0;
183
184 cl_device_id parent;
185 cl_uint sub_cus;
186
187 /* CL_DEVICE_PARTITION_EQUALLY */
188
189 printf("Max CUs: %u\n", max_cus);
190
191 /* if the device has 3 CUs, 3 subdevices will be created, otherwise 2. */
192 if (max_cus == 3)
193 split = 3;
194 else
195 split = 2;
196
197 const cl_device_partition_property equal_splitter[] = {
198 CL_DEVICE_PARTITION_EQUALLY, max_cus/split, 0 };
199
200 err = clCreateSubDevices(rootdev, equal_splitter, 0, NULL, &numdevs);
201 CHECK_OPENCL_ERROR_IN("count sub devices");
202 TEST_ASSERT(numdevs == split);
203
204 err = clCreateSubDevices(rootdev, equal_splitter, split, eqdev, NULL);
205 CHECK_OPENCL_ERROR_IN("partition equally");
206 if (split == 2)
207 eqdev[2] = NULL;
208
209 cl_uint refc;
210 err = clGetDeviceInfo (eqdev[0], CL_DEVICE_REFERENCE_COUNT, sizeof (refc),
211 &refc, NULL);
212 CHECK_OPENCL_ERROR_IN ("get refcount");
213 TEST_ASSERT (refc == 1);
214
215 /* First, check that the root device is untouched */
216
217 err = clGetDeviceInfo(rootdev, CL_DEVICE_MAX_COMPUTE_UNITS,
218 sizeof(sub_cus), &sub_cus, NULL);
219 CHECK_OPENCL_ERROR_IN("parenty CU");
220 TEST_ASSERT(sub_cus == max_cus);
221
222 err = clGetDeviceInfo(rootdev, CL_DEVICE_PARENT_DEVICE,
223 sizeof(parent), &parent, NULL);
224 CHECK_OPENCL_ERROR_IN("root parent device");
225 TEST_ASSERT(parent == NULL);
226
227 /* partition type may either be NULL or contain a 0 entry */
228 err = clGetDeviceInfo(rootdev, CL_DEVICE_PARTITION_TYPE,
229 0, NULL, &ptype_size);
230 CHECK_OPENCL_ERROR_IN("root partition type");
231
232 if (ptype_size != 0) {
233 /* abuse dev_pt which should be large enough */
234 TEST_ASSERT(ptype_size == sizeof(cl_device_partition_property));
235 TEST_ASSERT(ptype_size <= dev_pt_size);
236 err = clGetDeviceInfo(rootdev, CL_DEVICE_PARTITION_TYPE,
237 ptype_size, dev_pt, NULL);
238 CHECK_OPENCL_ERROR_IN("root partition type #2");
239 TEST_ASSERT(dev_pt[0] == 0);
240 }
241
242 /* now test the subdevices */
243 for (i = 0; i < split; ++i) {
244 err = clGetDeviceInfo(eqdev[i], CL_DEVICE_MAX_COMPUTE_UNITS,
245 sizeof(sub_cus), &sub_cus, NULL);
246 CHECK_OPENCL_ERROR_IN("sub CU");
247 TEST_ASSERT(sub_cus == max_cus/split);
248
249 err = clGetDeviceInfo(eqdev[i], CL_DEVICE_PARENT_DEVICE,
250 sizeof(parent), &parent, NULL);
251 CHECK_OPENCL_ERROR_IN("sub parent device");
252 TEST_ASSERT(parent == rootdev);
253
254 err = clGetDeviceInfo(eqdev[i], CL_DEVICE_PARTITION_TYPE,
255 0, NULL, &ptype_size);
256 CHECK_OPENCL_ERROR_IN("sub partition type");
257 TEST_ASSERT(ptype_size == sizeof(equal_splitter));
258
259 ptype = malloc(ptype_size);
260 TEST_ASSERT(ptype);
261 err = clGetDeviceInfo(eqdev[i], CL_DEVICE_PARTITION_TYPE,
262 ptype_size, ptype, NULL);
263 CHECK_OPENCL_ERROR_IN("sub partition type #2");
264
265 TEST_ASSERT(memcmp(ptype, equal_splitter, ptype_size) == 0);
266
267 /* free the partition type */
268 free(ptype) ; ptype = NULL;
269 }
270
271 /* CL_DEVICE_PARTITION_BY_COUNTS */
272
273 /* Note that the platform will only read this to the first 0,
274 * which is actually CL_DEVICE_PARTITION_BY_COUNTS_LIST_END;
275 * the test is structured with an additional final 0 intentionally,
276 * to follow the Khoronos doc example
277 */
278 const cl_device_partition_property count_splitter[] = {
279 CL_DEVICE_PARTITION_BY_COUNTS, 1, max_cus - 1,
280 CL_DEVICE_PARTITION_BY_COUNTS_LIST_END, 0 };
281
282 err = clCreateSubDevices(rootdev, count_splitter, 0, NULL, &numdevs);
283 CHECK_OPENCL_ERROR_IN("count sub devices");
284 TEST_ASSERT(numdevs == 2);
285
286 err = clCreateSubDevices(rootdev, count_splitter, 2, countdev, NULL);
287 CHECK_OPENCL_ERROR_IN("partition by counts");
288
289 /* First, check that the root device is untouched */
290
291 err = clGetDeviceInfo(rootdev, CL_DEVICE_MAX_COMPUTE_UNITS,
292 sizeof(sub_cus), &sub_cus, NULL);
293 CHECK_OPENCL_ERROR_IN("parenty CU");
294 TEST_ASSERT(sub_cus == max_cus);
295
296 err = clGetDeviceInfo(rootdev, CL_DEVICE_PARENT_DEVICE,
297 sizeof(parent), &parent, NULL);
298 CHECK_OPENCL_ERROR_IN("root parent device");
299 TEST_ASSERT(parent == NULL);
300
301 /* partition type may either be NULL or contain a 0 entry */
302 err = clGetDeviceInfo(rootdev, CL_DEVICE_PARTITION_TYPE,
303 0, NULL, &ptype_size);
304 CHECK_OPENCL_ERROR_IN("root partition type");
305
306 if (ptype_size != 0) {
307 /* abuse dev_pt which should be large enough */
308 TEST_ASSERT(ptype_size == sizeof(cl_device_partition_property));
309 TEST_ASSERT(ptype_size <= dev_pt_size);
310 err = clGetDeviceInfo(rootdev, CL_DEVICE_PARTITION_TYPE,
311 ptype_size, dev_pt, NULL);
312 CHECK_OPENCL_ERROR_IN("root partition type #2");
313 TEST_ASSERT(dev_pt[0] == 0);
314 }
315
316 // devices might be returned in different order than the counts
317 // in the count_splitter
318
319 int found_cus[2] = {0, 0};
320
321 /* now test the subdevices */
322 for (i = 0; i < 2; ++i) {
323 err = clGetDeviceInfo(countdev[i], CL_DEVICE_MAX_COMPUTE_UNITS,
324 sizeof(sub_cus), &sub_cus, NULL);
325 CHECK_OPENCL_ERROR_IN("sub CU");
326 if (sub_cus == count_splitter[1])
327 found_cus[0] += 1;
328 else if (sub_cus == count_splitter[2])
329 found_cus[1] += 1;
330
331 err = clGetDeviceInfo(countdev[i], CL_DEVICE_PARENT_DEVICE,
332 sizeof(parent), &parent, NULL);
333 CHECK_OPENCL_ERROR_IN("sub parent device");
334 TEST_ASSERT(parent == rootdev);
335
336 /* The partition type returned is up to the first 0,
337 * which happens to be the CL_DEVICE_PARTITION_BY_COUNTS_LIST_END,
338 * not the final terminating 0 in count_splitter, so it has one less
339 * element. It should be otherwise equal */
340 err = clGetDeviceInfo(countdev[i], CL_DEVICE_PARTITION_TYPE,
341 0, NULL, &ptype_size);
342 CHECK_OPENCL_ERROR_IN("sub partition type");
343 TEST_ASSERT(ptype_size == sizeof(count_splitter) - sizeof(*count_splitter));
344
345 ptype = malloc(ptype_size);
346 TEST_ASSERT(ptype);
347 err = clGetDeviceInfo(countdev[i], CL_DEVICE_PARTITION_TYPE,
348 ptype_size, ptype, NULL);
349 CHECK_OPENCL_ERROR_IN("sub partition type #2");
350
351 TEST_ASSERT(memcmp(ptype, count_splitter, ptype_size) == 0);
352
353 /* free the partition type */
354 free(ptype) ; ptype = NULL;
355 }
356
357 /* the previous loop finds 1+1 subdevices only on >dual core systems;
358 * on dual cores, the count_splitter is [1, 1] and the above
359 * "(sub_cus == count_splitter[x])" results in 2+0 subdevices found */
360 if (max_cus > 2)
361 TEST_ASSERT(found_cus[0] == 1 && found_cus[1] == 1);
362 else
363 TEST_ASSERT((found_cus[0] + found_cus[1]) == 2);
364
365 /* So far, so good. Let's now try and use these devices,
366 * by building a program for all of them and launching kernels on them.
367 *
368 * Note that there's a discrepancy in behavior between implementations:
369 * some assume you can treat sub-devices as their parent device, and thus
370 * e.g. using them through any context which includes their parent devices,
371 * other fail miserably if you try this.
372 *
373 * For the time being we will test the stricter behavior, where
374 * sub-devices should be added manually to a context.
375 */
376
377 err = clReleaseCommandQueue(q);
378 CHECK_OPENCL_ERROR_IN("clReleaseCommandQueue");
379 err = clReleaseContext(ctx);
380 CHECK_OPENCL_ERROR_IN("clReleaseContext");
381
382 /* if we split into 2 equal parts, third pointer is NULL. Let's copy the
383 * previous device to it */
384 if (split == 2)
385 eqdev[2] = eqdev[1];
386
387 ctx = clCreateContext(NULL, NUMDEVS, alldevs, NULL, NULL, &err);
388 CHECK_OPENCL_ERROR_IN("clCreateContext");
389 TEST_ASSERT( test_context(ctx, prog_src_all, 1, NUMDEVS, alldevs) == CL_SUCCESS );
390
391 ctx = clCreateContext(NULL, NUMDEVS - 1, alldevs + 1, NULL, NULL, &err);
392 CHECK_OPENCL_ERROR_IN("clCreateContext");
393 TEST_ASSERT( test_context(ctx, prog_src_two, -1, NUMDEVS - 1, alldevs + 1)
394 == CL_SUCCESS );
395
396 /* Don't release the same device twice. clReleaseDevice(NULL) should return
397 * an error but not crash. */
398 if (split == 2)
399 eqdev[2] = NULL;
400
401 for (i = 0; i < NUMDEVS; i++)
402 clReleaseDevice (alldevs[i]);
403
404 CHECK_CL_ERROR (clUnloadCompiler ());
405 free (dev_pt);
406
407 printf ("OK\n");
408
409 return 0;
410}
