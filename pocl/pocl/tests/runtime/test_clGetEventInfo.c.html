1#include <stdio.h>
2#include <stdlib.h>
3#include <CL/cl.h>
4#include "poclu.h"
5
6#define MAX_PLATFORMS 32
7#define MAX_DEVICES 32
8
9int
10main(void)
11{
12 cl_int err;
13 cl_platform_id platforms[MAX_PLATFORMS];
14 cl_uint nplatforms;
15 cl_device_id devices[MAX_DEVICES];
16 cl_uint ndevices;
17 cl_uint i, j;
18 cl_context context;
19 cl_command_queue queue;
20 cl_mem buf;
21 cl_event buf_event;
22 cl_command_queue event_command_queue;
23
24 CHECK_CL_ERROR(clGetPlatformIDs(MAX_PLATFORMS, platforms, &nplatforms));
25
26 for (i = 0; i < nplatforms; i++)
27 {
28 CHECK_CL_ERROR(clGetDeviceIDs(platforms[i], CL_DEVICE_TYPE_ALL, MAX_DEVICES,
29 devices, &ndevices));
30
31 for (j = 0; j < ndevices; j++)
32 {
33 context = clCreateContext (NULL, 1, &devices[j], NULL, NULL, &err);
34 queue = clCreateCommandQueue (context, devices[j], 0, &err);
35
36 const int buf_size = 1024;
37 cl_int host_buf[buf_size];
38
39 buf = clCreateBuffer (context, CL_MEM_READ_WRITE,
40 sizeof (cl_int) * buf_size, NULL, &err);
41 CHECK_CL_ERROR(clEnqueueReadBuffer(queue, buf, CL_TRUE, 0, sizeof(cl_int) * buf_size, &host_buf, 0, NULL, &buf_event));
42 CHECK_CL_ERROR(clFinish(queue));
43 size_t param_val_size_ret;
44 CHECK_CL_ERROR(clGetEventInfo(buf_event, CL_EVENT_COMMAND_QUEUE, sizeof(cl_command_queue), &event_command_queue, &param_val_size_ret));
45 TEST_ASSERT(param_val_size_ret == sizeof(cl_command_queue));
46 TEST_ASSERT(event_command_queue == queue);
47
48 cl_command_type command_type;
49 CHECK_CL_ERROR(clGetEventInfo(buf_event, CL_EVENT_COMMAND_TYPE, sizeof(cl_command_type), &command_type, &param_val_size_ret));
50 TEST_ASSERT(param_val_size_ret == sizeof(cl_command_type));
51 TEST_ASSERT(command_type == CL_COMMAND_READ_BUFFER);
52
53 cl_int execution_status;
54 CHECK_CL_ERROR(clGetEventInfo(buf_event, CL_EVENT_COMMAND_EXECUTION_STATUS, sizeof(cl_int), &execution_status, &param_val_size_ret));
55 TEST_ASSERT(param_val_size_ret == sizeof(cl_int));
56 TEST_ASSERT(execution_status == CL_COMPLETE);
57
58 CHECK_CL_ERROR(clReleaseEvent(buf_event));
59
60 CHECK_CL_ERROR(clReleaseMemObject(buf));
61
62 CHECK_CL_ERROR(clReleaseCommandQueue(queue));
63
64 CHECK_CL_ERROR (clReleaseContext (context));
65 }
66 }
67
68 CHECK_CL_ERROR (clUnloadCompiler ());
69
70 printf ("OK\n");
71 return EXIT_SUCCESS;
72}
