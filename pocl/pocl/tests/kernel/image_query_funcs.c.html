1#include <stdio.h>
2#include <stdlib.h>
3#include <string.h>
4#include <CL/opencl.h>
5#include "poclu.h"
6
7#ifdef _MSC_VER
8# include "vccompat.hpp"
9#endif
10
11int main(int argc, char **argv)
12{
13 /* test name */
14 char name[] = "test_image_query_funcs";
15 size_t global_work_size[1] = { 1 }, local_work_size[1]= { 1 };
16 size_t srcdir_length, name_length, filename_size;
17 char *filename = NULL;
18 char *source = NULL;
19 cl_device_id devices[1];
20 cl_context context = NULL;
21 cl_command_queue queue = NULL;
22 cl_program program = NULL;
23 cl_kernel kernel = NULL;
24 cl_int err;
25
26 /* image parameters */
27 cl_uchar4 *imageData;
28 cl_image_format image_format;
29 cl_image_desc image2_desc, image3_desc;
30
31 printf("Running test %s...\n", name);
32
33 memset(&image2_desc, 0, sizeof(cl_image_desc));
34 image2_desc.image_type = CL_MEM_OBJECT_IMAGE2D;
35 image2_desc.image_width = 2;
36 image2_desc.image_height = 4;
37
38 memset(&image3_desc, 0, sizeof(cl_image_desc));
39 image3_desc.image_type = CL_MEM_OBJECT_IMAGE3D;
40 image3_desc.image_width = 2;
41 image3_desc.image_height = 4;
42 image3_desc.image_depth = 8;
43
44 image_format.image_channel_order = CL_RGBA;
45 image_format.image_channel_data_type = CL_UNSIGNED_INT8;
46 imageData = (cl_uchar4*)malloc (4 * 4 * sizeof(cl_uchar4));
47
48 TEST_ASSERT (imageData != NULL && "out of host memory\n");
49 memset (imageData, 1, 4*4*sizeof(cl_uchar4));
50
51 /* determine file name of kernel source to load */
52 srcdir_length = strlen(SRCDIR);
53 name_length = strlen(name);
54 filename_size = srcdir_length + name_length + 16;
55 filename = (char *)malloc(filename_size + 1);
56 TEST_ASSERT (filename != NULL && "out of host memory\n");
57
58 snprintf(filename, filename_size, "%s/%s.cl", SRCDIR, name);
59
60 /* read source code */
61 source = poclu_read_file (filename);
62 TEST_ASSERT (source != NULL && "Kernel .cl not found.");
63
64 /* setup an OpenCL context and command queue using default device */
65 context = poclu_create_any_context();
66 TEST_ASSERT (context != NULL && "clCreateContextFromType call failed\n");
67
68 cl_sampler external_sampler = clCreateSampler (
69 context, CL_FALSE, CL_ADDRESS_NONE, CL_FILTER_NEAREST, &err);
70 CHECK_OPENCL_ERROR_IN ("clCreateSampler");
71
72 CHECK_CL_ERROR (clGetContextInfo (context, CL_CONTEXT_DEVICES,
73 sizeof (cl_device_id), devices, NULL));
74
75 queue = clCreateCommandQueue (context, devices[0], 0, &err);
76 CHECK_OPENCL_ERROR_IN ("clCreateCommandQueue");
77
78 /* Create image */
79 cl_mem image2
80 = clCreateImage (context, CL_MEM_READ_ONLY | CL_MEM_USE_HOST_PTR,
81 &image_format, &image2_desc, imageData, &err);
82 CHECK_OPENCL_ERROR_IN ("clCreateImage image2");
83
84 cl_mem image3
85 = clCreateImage (context, CL_MEM_READ_ONLY | CL_MEM_USE_HOST_PTR,
86 &image_format, &image3_desc, imageData, &err);
87 CHECK_OPENCL_ERROR_IN ("clCreateImage image3");
88
89 unsigned color[4] = { 2, 9, 11, 7 };
90 size_t orig[3] = { 0, 0, 0 };
91 size_t reg[3] = { 2, 4, 1 };
92 err = clEnqueueFillImage (queue, image2, color, orig, reg, 0, NULL, NULL);
93 CHECK_OPENCL_ERROR_IN ("clCreateImage image3");
94
95 /* create and build program */
96 program = clCreateProgramWithSource (context, 1, (const char **)&source,
97 NULL, &err);
98 CHECK_OPENCL_ERROR_IN ("clCreateProgramWithSource");
99
100 err = clBuildProgram (program, 0, NULL, NULL, NULL, NULL);
101 CHECK_OPENCL_ERROR_IN ("clBuildProgram");
102
103 /* execute the kernel with give name */
104 kernel = clCreateKernel (program, name, NULL);
105 CHECK_OPENCL_ERROR_IN ("clCreateKernel");
106
107 err = clSetKernelArg (kernel, 0, sizeof (cl_mem), &image2);
108 CHECK_OPENCL_ERROR_IN ("clSetKernelArg 0");
109
110 err = clSetKernelArg (kernel, 1, sizeof (cl_mem), &image3);
111 CHECK_OPENCL_ERROR_IN ("clSetKernelArg 1");
112
113 err = clSetKernelArg (kernel, 2, sizeof (cl_sampler), &external_sampler);
114 CHECK_OPENCL_ERROR_IN ("clSetKernelArg 2");
115
116 err = clEnqueueNDRangeKernel (queue, kernel, 1, NULL, global_work_size,
117 local_work_size, 0, NULL, NULL);
118 CHECK_OPENCL_ERROR_IN ("clEnqueueNDRangeKernel");
119
120 err = clFinish (queue);
121 CHECK_OPENCL_ERROR_IN ("clFinish");
122
123 clReleaseMemObject (image2);
124 clReleaseMemObject (image3);
125 clReleaseKernel (kernel);
126 clReleaseProgram (program);
127 clReleaseCommandQueue (queue);
128 clReleaseSampler (external_sampler);
129 clUnloadCompiler ();
130 clReleaseContext (context);
131 free (source);
132 free (filename);
133 free (imageData);
134
135 printf("OK\n");
136 return 0;
137}
