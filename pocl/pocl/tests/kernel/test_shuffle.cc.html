1/* This test is part of pocl.
2 * Author: Kalle Raiskila, 2014
3 *
4 * Test the OpenCL-C 'shuffle' command by looping
5 * over all permutations of data and mask vector
6 * lengths. Only one data type at a time is tested,
7 * the type to test is passed as command line argument
8 * (this allows for more interactive testsuite, and
9 * non-supported types (e.g. half, double) are easier
10 * to filter out).
11 * The data to be shuffled is vectors where the elemnet
12 * data equals the element index (i.e. [0,1,2,3]) but
13 * the shuffle pattern masks are generated with rand().
14 */
15

17#include <cstdio>
18#include <cstring>
19#include <CL/cl.h>
20#include <iostream>
21#include <cmath>
22#include <cstdlib>

24cl_context ctx;
25cl_device_id did;
26cl_platform_id pid;
27cl_command_queue queue;

29#define ERRCHECK() if (check_cl_error(errcode, __LINE__, __FUNCTION__)) abort();
30
31static const unsigned vecelts[] = {2,4,8,16};
32static const int stimuli[] = {4, 2, 69, 4, 5, 0, 45, 16, 4, 6, 1, 18, 28, 14,
33 22, 16, 8, 2, 0, 31, 42, 11, 62, 88, 99, 23, 13};
34
35template <typename D, typename M>
36class TestShuffle {
37 cl_mem mem_in1,
38 mem_in2,
39 mem_out,
40 mem_mask1,
41 mem_mask2;
42
43 cl_program prog;
44
45 D in1 [16] __attribute__ ((aligned (128)));
46 D in2 [16] __attribute__ ((aligned (128)));
47 D out [16] __attribute__ ((aligned (128)));
48 M mask1 [16] __attribute__ ((aligned (128)));
49 M mask2 [16] __attribute__ ((aligned (128)));
50 const char* ocl_type;
51 unsigned size;
52 cl_int errcode;
53
54private:
55 /* Prints into std::string all the OpenCL kernel sources for each n,m combo */
56 void testcase_src(std::string & src) {
57 char buf[1024];
58 int rv;
59 unsigned n, m;
60 const char* mask_type;
61 switch(sizeof(M)) {
62 case 1:
63 mask_type = "uchar"; break;
64 case 2:
65 mask_type = "ushort"; break;
66 case 4:
67 mask_type = "uint"; break;
68 case 8:
69 mask_type = "ulong"; break;
70 default:
71 mask_type = "UNKNOWN_MASK";
72 }
73
74 for(unsigned n_loop=0; n_loop<4; n_loop++) {
75 for(unsigned m_loop=0; m_loop<4; m_loop++) {
76
77 n = vecelts[n_loop];
78 m = vecelts[m_loop];
79 rv = 0;
80 buf[0] = 0;
81 rv=sprintf(buf,
82 "__kernel void test_shuffle_%d_%d("
83 "__global const %s%d *in, __global const %s%d *mask, __global %s%d *out) {\n"
84 "*out = shuffle( *in, *mask);\n}\n",
85 m, n, ocl_type, m, mask_type, n, ocl_type, n);
86 rv+=sprintf(buf+rv,
87 "__kernel void test_shuffle2_%d_%d("
88 "__global const %s%d *in1, __global const %s%d *in2, __global const %s%d *mask, __global %s%d *out) {\n"
89 "*out = shuffle2( *in1, *in2, *mask);\n}\n",
90 m, n, ocl_type, m, ocl_type, m, mask_type, n, ocl_type, n);
91 src.append(buf);
92 }
93 }
94 }
95
96 #define nsize n
97 #define msize m
98 // assume out is filled with 'shuffle(in, mask)' // return true if ok
99 bool output_matches_1(unsigned n, unsigned m)
100 {
101 bool error=false;
102 for(unsigned i=0; i<n; i++)
103 {
104 unsigned mm = mask1[i] % msize;
105 error |= (out[i] != in1[mm]);
106 }
107 return !error;
108 }
109
110 // assume out is filled with 'shuffle2(in1, in2, mask)'
111 // return true if ok
112 bool output_matches_2(unsigned n, unsigned m)
113 {
114 bool error=false;
115 for(unsigned i=0; i<n; i++)
116 {
117 unsigned msk = mask2[i] % (2*msize);
118 D correct = (msk < msize) ? in1[msk] : in2[msk-msize];
119 if (out[i] != correct) {
120 error |= true;
121 printf("element %d should be %d (mask %d), got %d\n", i, (int)correct, (int)mask2[i], (int)out[i]);
122 }
123 }
124 return !error;
125 }

128 // helpers: prints a vector as [0, 1, 2]
129 // cast to int, so vectors of 'char' come out correctly
130 void print_in1(unsigned n, unsigned m)
131 {
132 std::cout << "["<<(int)in1[0];
133 for(unsigned i=1; i<m; i++)
134 std::cout << ", " <<(int)in1[i];
135 std::cout << "]";
136 }
137 void print_in2(unsigned n, unsigned m)
138 {
139 std::cout << "["<<(int)in2[0];
140 for(unsigned i=1; i<m; i++)
141 std::cout << ", " <<(int)in2[i];
142 std::cout << "]";
143 }
144 void print_mask1(unsigned n, unsigned m)
145 {
146 std::cout << "["<<(int)mask1[0];
147 for(unsigned i=1; i<n; i++)
148 std::cout << ", " <<(int)mask1[i];
149 std::cout << "]";
150 }
151 void print_mask2(unsigned n, unsigned m)
152 {
153 std::cout << "["<<(int)mask2[0];
154 for(unsigned i=1; i<n; i++)
155 std::cout << ", " <<(int)mask2[i];
156 std::cout << "]";
157 }
158 void print_out(unsigned n, unsigned m)
159 {
160 std::cout << "["<<(int)out[0];
161 for(unsigned i=1; i<n; i++)
162 std::cout << ", " <<(int)out[i];
163 std::cout << "]";
164 }

166 /* Run one shuffle test, return true if successful*/
167
168 bool run_single_test(unsigned n, unsigned m){
169 bool rv=true;
170 cl_kernel krn, krn2;
171 char kern_name[128], kern_name2[128];
172
173 snprintf(kern_name, 128, "test_shuffle_%d_%d", m, n);
174 krn = clCreateKernel(prog, kern_name, &errcode);
175 ERRCHECK()
176
177 errcode = clSetKernelArg( krn, 0, sizeof(cl_mem), &mem_in1 );
178 ERRCHECK()
179 errcode = clSetKernelArg( krn, 1, sizeof(cl_mem), &mem_mask1 );
180 ERRCHECK()
181 errcode = clSetKernelArg( krn, 2, sizeof(cl_mem), &mem_out );
182 ERRCHECK()
183
184 errcode = clEnqueueTask( queue, krn, 0, NULL, NULL );
185 ERRCHECK()
186 errcode = clEnqueueReadBuffer( queue, mem_out, CL_TRUE, 0, size, out, 0, NULL, NULL );
187 ERRCHECK()
188 errcode = clFinish(queue);
189 ERRCHECK()
190
191 if(!output_matches_1(n, m))
192 {
193 std::cout << "Error in shuffle " << ocl_type << " " << m;
194 std::cout << " => " << ocl_type << " " << n << " :";
195 print_out(n, m);
196 std::cout << " = shuffle( ";
197 print_in1(n, m);
198 std::cout << ", ";
199 print_mask1(n, m);
200 std::cout << ");" << std::endl;
201 rv=false;
202 }
203 clReleaseKernel(krn);
204
205 // Now test shuffle2()
206 snprintf(kern_name2, 128, "test_shuffle2_%d_%d", m, n);
207 krn2 = clCreateKernel(prog, kern_name2, &errcode);
208 ERRCHECK()
209 errcode = clSetKernelArg( krn2, 0, sizeof(cl_mem), &mem_in1 );
210 ERRCHECK()
211 errcode = clSetKernelArg( krn2, 1, sizeof(cl_mem), &mem_in2 );
212 ERRCHECK()
213 errcode = clSetKernelArg( krn2, 2, sizeof(cl_mem), &mem_mask2 );
214 ERRCHECK()
215 errcode = clSetKernelArg( krn2, 3, sizeof(cl_mem), &mem_out );
216 ERRCHECK()
217 errcode = clEnqueueTask( queue, krn2, 0, NULL, NULL );
218 ERRCHECK()
219 errcode = clEnqueueReadBuffer( queue, mem_out, CL_TRUE, 0, size, out, 0, NULL, NULL );
220 ERRCHECK()
221 errcode = clFinish(queue);
222 ERRCHECK()
223
224 if(!output_matches_2(n, m))
225 {
226 std::cout << "Error in shuffle2 " << ocl_type << " " << m;
227 std::cout << " => " << ocl_type << " " << n << " :";
228 print_out(n, m);
229 std::cout << " = shuffle2( ";
230 print_in1(n, m);
231 std::cout << ", ";
232 print_in2(n, m);
233 std::cout << ", ";
234 print_mask2(n, m);
235 std::cout << ");" << std::endl;
236 rv=false;
237 }
238 clReleaseKernel(krn2);
239 return rv;
240 }

246public:
247 TestShuffle(const char* type) {
248 ocl_type = type;
249 size = sizeof(D) * 16;
250 for(unsigned i=0; i<16; i++) {
251 mask1[i] = (M)stimuli[i];
252 mask2[i] = (M)stimuli[i];
253 }
254
255 for (unsigned n_loop = 0; n_loop < 4; n_loop++) {
256 for (unsigned m_loop = 0; m_loop < 4; m_loop++) {
257 unsigned m = vecelts[m_loop];
258 for (unsigned i = 0; i < m; i++) {
259 in2[i] = (D)(i + m);
260 in1[i] = (D)i;
261 }
262 }
263 }
264 }
265
266 unsigned run() {
267
268 // Fixed pseudorandom stimuli to make the test deterministic.
269 // Random stimuli leads to randomly appearing/disappearing
270 // problems which are irritating and hard to reproduce. Values which reduce
271 // to element 3 might produce an undefined value in case of 3 element inputs so
272 // let's not use them in the stimulus.
273
274 mem_in1 = clCreateBuffer(ctx,
275 CL_MEM_READ_ONLY | CL_MEM_USE_HOST_PTR,
276 size, in1, &errcode);
277 ERRCHECK()
278 mem_in2 = clCreateBuffer(ctx,
279 CL_MEM_READ_ONLY | CL_MEM_USE_HOST_PTR,
280 size, in2, &errcode);
281 ERRCHECK()
282 mem_mask1 = clCreateBuffer(ctx,
283 CL_MEM_READ_ONLY | CL_MEM_USE_HOST_PTR,
284 size, mask1, &errcode);
285 ERRCHECK()
286 mem_mask2 = clCreateBuffer(ctx,
287 CL_MEM_READ_ONLY | CL_MEM_USE_HOST_PTR,
288 size, mask2, &errcode);
289 ERRCHECK()
290 mem_out = clCreateBuffer(ctx,
291 CL_MEM_WRITE_ONLY,
292 size, NULL, &errcode);
293 ERRCHECK()
294
295 std::string source;
296 testcase_src(source);
297
298 const char *c_src = source.c_str();
299 size_t srclen = source.size();
300
301 prog = clCreateProgramWithSource(ctx, 1, &c_src, &srclen, &errcode);
302 ERRCHECK()
303 errcode = clBuildProgram(prog, 0, NULL, NULL, NULL, NULL);
304 ERRCHECK()
305
306 unsigned errors = 0;
307 for (unsigned n_loop = 0; n_loop < 4; n_loop++) {
308 for (unsigned m_loop = 0; m_loop < 4; m_loop++) {
309 if (!run_single_test(vecelts[n_loop], vecelts[m_loop]))
310 errors++;
311 }
312 }
313
314 clReleaseMemObject(mem_in1);
315 clReleaseMemObject(mem_in2);
316 clReleaseMemObject(mem_mask1);
317 clReleaseMemObject(mem_mask2);
318 clReleaseMemObject(mem_out);
319 clReleaseProgram(prog);
320
321 return errors;
322 }
323};

330int main( int argc, char *argv[])
331{
332 unsigned num_errors = 0;
333
334 if( argc != 2 ) {
335 std::cout << "give element type"<<std::endl;
336 exit(-1);
337 }
338
339 poclu_get_any_device( &ctx, &did, &queue);
340
341#if (__GNUC__ > 5)
342#pragma GCC diagnostic push
343#pragma GCC diagnostic ignored "-Wignored-attributes"
344#endif
345
346 /* Loop over input (m) and output (n) vector lengths.
347 * The big if-else is needed to pass the string
348 * representation to runtest.
349 * This cannot be fully templated, as there is a
350 * 'typedef short int half', which would cause the
351 * templating mechanism to create the test for shorts instead
352 * of halfs.
353 */
354 if( strcmp("char", argv[1]) == 0 ) {
355
356 TestShuffle<cl_char, cl_uchar> t("char"); num_errors = t.run();
357
358 } else if( strcmp("uchar", argv[1]) == 0 ) {
359
360 TestShuffle<cl_uchar, cl_uchar> t("uchar"); num_errors = t.run();
361
362 } else if( strcmp("short", argv[1]) == 0 ) {
363
364 TestShuffle<cl_short, cl_ushort> t("short"); num_errors = t.run();
365
366 } else if( strcmp("ushort", argv[1]) == 0 ) {
367
368 TestShuffle<cl_ushort, cl_ushort> t("ushort"); num_errors = t.run();
369
370 } else if( strcmp("int", argv[1]) == 0 ) {
371
372 TestShuffle<cl_int, cl_uint> t("int"); num_errors = t.run();
373
374 } else if( strcmp("uint", argv[1]) == 0 ) {
375
376 TestShuffle<cl_uint, cl_uint> t("uint"); num_errors = t.run();
377
378 } else if( strcmp("long", argv[1]) == 0 ) {
379
380 TestShuffle<cl_long, cl_ulong> t("long"); num_errors = t.run();
381
382 } else if( strcmp("ulong", argv[1]) == 0 ) {
383
384 TestShuffle<cl_ulong, cl_ulong> t("ulong"); num_errors = t.run();
385
386 } else if( strcmp("half", argv[1]) == 0 ) {
387
388 TestShuffle<cl_half, cl_ushort> t("half"); num_errors = t.run();
389
390 } else if( strcmp("float", argv[1]) == 0 ) {
391
392 TestShuffle<cl_float, cl_uint> t("float"); num_errors = t.run();
393
394 } else if( strcmp("double", argv[1]) == 0 ) {
395
396 TestShuffle<cl_double, cl_ulong> t("double"); num_errors = t.run();
397
398 } else {
399
400 std::cout << "Error: unknown type " << argv[1] << ": use OCL-C types"<<std::endl;
401 return -1;
402
403 }
404
405#if (__GNUC__ > 5)
406#pragma GCC diagnostic pop
407#endif
408 clReleaseCommandQueue(queue);
409 clUnloadCompiler();
410 clReleaseContext(ctx);
411
412 if( num_errors == 0)
413 std::cout << "OK" << std::endl;
414 return num_errors;
415}
