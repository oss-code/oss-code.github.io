16int call_test(const char *name)
17{
18 size_t global_work_size[1] = { 1 }, local_work_size[1]= { 1 };
19 size_t srcdir_length, name_length, filename_size;
20 char *filename = NULL;
21 char *source = NULL;
22 cl_device_id devices[1];
23 cl_context context = NULL;
24 cl_command_queue queue = NULL;
25 cl_program program = NULL;
26 cl_kernel kernel = NULL;
27 cl_int result;
28 int retval = -1;
29
30 TEST_ASSERT (name != NULL);
31
32 /* determine file name of kernel source to load */
33 srcdir_length = strlen(SRCDIR);
34 name_length = strlen(name);
35 filename_size = srcdir_length + name_length + 16;
36 filename = (char *)malloc(filename_size + 1);
37 if (!filename) {
38 puts("out of memory");
39 goto error;
40 }
41
42 snprintf(filename, filename_size, "%s/%s.cl", SRCDIR, name);
43
44 /* read source code */
45 source = poclu_read_file (filename);
46 TEST_ASSERT (source != NULL && "Kernel .cl not found.");
47
48 /* setup an OpenCL context and command queue using default device */
49 context = poclu_create_any_context();
50 if (!context) {
51 puts("clCreateContextFromType call failed\n");
52 goto error;
53 }
54
55 result = clGetContextInfo(context, CL_CONTEXT_DEVICES,
56 sizeof(cl_device_id), devices, NULL);
57 if (result != CL_SUCCESS) {
58 puts("clGetContextInfo call failed\n");
59 goto error;
60 }
61
62 queue = clCreateCommandQueue(context, devices[0], 0, NULL);
63 if (!queue) {
64 puts("clCreateCommandQueue call failed\n");
65 goto error;
66 }
67
68 /* create and build program */
69 program = clCreateProgramWithSource (context, 1, (const char **)&source,
70 NULL, NULL);
71 if (!program) {
72 puts("clCreateProgramWithSource call failed\n");
73 goto error;
74 }
75
76 result = clBuildProgram(program, 0, NULL, "-I" SRCDIR, NULL, NULL);
77 if (result != CL_SUCCESS) {
78 puts("clBuildProgram call failed\n");
79 goto error;
80 }
81
82 /* execute the kernel with give name */
83 kernel = clCreateKernel(program, name, NULL);
84 if (!kernel) {
85 puts("clCreateKernel call failed\n");
86 goto error;
87 }
88
89 result = clEnqueueNDRangeKernel(queue, kernel, 1, NULL,
90 global_work_size, local_work_size, 0, NULL, NULL);
91 if (result != CL_SUCCESS) {
92 puts("clEnqueueNDRangeKernel call failed\n");
93 goto error;
94 }
95
96 result = clFinish(queue);
97 if (result == CL_SUCCESS)
98 retval = 0;
99
100error:
101
102 if (kernel) {
103 clReleaseKernel(kernel);
104 }
105 if (program) {
106 clReleaseProgram(program);
107 }
108 if (queue) {
109 clReleaseCommandQueue(queue);
110 }
111 if (context) {
112 clUnloadCompiler ();
113 clReleaseContext (context);
114 }
115 if (source) {
116 free(source);
117 }
118 if (filename) {
119 free(filename);
120 }
121
122 return retval;
123}

139{

141
142 if (argc < 2) {
143 /* Run all tests */
144 for (i = 0; i < num_all_tests; ++i) {
145 printf("Running test #%d %s...\n", i, all_tests[i]);
146 fflush(stdout);
147 fflush(stderr);
148 retval = call_test(all_tests[i]);

150 fflush(stderr);

154 printf("Running test %s...\n", argv[1]);
