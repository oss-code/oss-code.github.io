1#include <stdio.h>
2#include <stdlib.h>
3#include <string.h>
4#include <CL/opencl.h>
5#include "poclu.h"
6
7#ifdef _MSC_VER
8# include "vccompat.hpp"
9#endif
10
11int main(int argc, char **argv)
12{
13 /* test name */
14 char name[] = "test_sampler_address_clamp";
15 size_t global_work_size[1] = { 1 }, local_work_size[1]= { 1 };
16 size_t srcdir_length, name_length, filename_size;
17 char *filename = NULL;
18 char *source = NULL;
19 cl_device_id devices[1];
20 cl_context context = NULL;
21 cl_command_queue queue = NULL;
22 cl_program program = NULL;
23 cl_kernel kernel = NULL;
24 cl_int result;
25 int retval = -1;
26
27 /* image parameters */
28 cl_uchar4 *imageData;
29 cl_image_format image_format;
30 cl_image_desc image_desc;
31
32 printf("Running test %s...\n", name);
33 memset(&image_desc, 0, sizeof(cl_image_desc));
34 image_desc.image_type = CL_MEM_OBJECT_IMAGE2D;
35 image_desc.image_width = 4;
36 image_desc.image_height = 4;
37 image_format.image_channel_order = CL_RGBA;
38 image_format.image_channel_data_type = CL_UNSIGNED_INT8;
39 imageData = (cl_uchar4*)malloc (4 * 4 * sizeof(cl_uchar4));
40
41 if (imageData == NULL)
42 {
43 puts("out of host memory\n");
44 goto error;
45 }
46 memset (imageData, 1, 4*4*sizeof(cl_uchar4));
47
48 /* determine file name of kernel source to load */
49 srcdir_length = strlen(SRCDIR);
50 name_length = strlen(name);
51 filename_size = srcdir_length + name_length + 16;
52 filename = (char *)malloc(filename_size + 1);
53 if (!filename)
54 {
55 puts("out of memory");
56 goto error;
57 }
58
59 snprintf(filename, filename_size, "%s/%s.cl", SRCDIR, name);
60
61 /* read source code */
62 source = poclu_read_file (filename);
63 TEST_ASSERT (source != NULL && "Kernel .cl not found.");
64
65 /* setup an OpenCL context and command queue using default device */
66 context = poclu_create_any_context();
67 if (!context)
68 {
69 puts("clCreateContextFromType call failed\n");
70 goto error;
71 }
72
73 result = clGetContextInfo(context, CL_CONTEXT_DEVICES,
74 sizeof(cl_device_id), devices, NULL);
75 if (result != CL_SUCCESS)
76 {
77 puts("clGetContextInfo call failed\n");
78 goto error;
79 }
80
81 queue = clCreateCommandQueue(context, devices[0], 0, NULL);
82 if (!queue)
83 {
84 puts("clCreateCommandQueue call failed\n");
85 goto error;
86 }
87
88 /* Create image */
89
90 cl_mem image = clCreateImage (context, CL_MEM_READ_ONLY | CL_MEM_USE_HOST_PTR,
91 &image_format, &image_desc, imageData, &result);
92 if (result != CL_SUCCESS)
93 {
94 puts("image creation failed\n");
95 goto error;
96 }
97
98
99 /* create and build program */
100 program = clCreateProgramWithSource (context, 1, (const char **)&source,
101 NULL, NULL);
102 if (!program)
103 {
104 puts("clCreateProgramWithSource call failed\n");
105 goto error;
106 }
107
108 result = clBuildProgram(program, 0, NULL, NULL, NULL, NULL);
109 if (result != CL_SUCCESS)
110 {
111 puts("clBuildProgram call failed\n");
112 goto error;
113 }
114
115 /* execute the kernel with give name */
116 kernel = clCreateKernel(program, name, NULL);
117 if (!kernel)
118 {
119 puts("clCreateKernel call failed\n");
120 goto error;
121 }
122
123 result = clSetKernelArg( kernel, 0, sizeof(cl_mem), &image);
124 if (result)
125 {
126 puts("clSetKernelArg failed\n");
127 goto error;
128 }
129
130 result = clEnqueueNDRangeKernel(queue, kernel, 1, NULL, global_work_size,
131 local_work_size, 0, NULL, NULL);
132 if (result != CL_SUCCESS)
133 {
134 puts("clEnqueueNDRangeKernel call failed\n");
135 goto error;
136 }
137
138 result = clFinish(queue);
139 if (result == CL_SUCCESS)
140 retval = 0;
141
142error:
143
144 if (image)
145 {
146 clReleaseMemObject (image);
147 }
148
149 if (kernel)
150 {
151 clReleaseKernel(kernel);
152 }
153 if (program)
154 {
155 clReleaseProgram(program);
156 }
157 if (queue)
158 {
159 clReleaseCommandQueue(queue);
160 }
161 if (context)
162 {
163 clUnloadCompiler ();
164 clReleaseContext (context);
165 }
166 if (source)
167 {
168 free(source);
169 }
170 if (filename)
171 {
172 free(filename);
173 }
174 if (imageData)
175 {
176 free(imageData);
177 }
178
179
180 if (retval)
181 {
182 printf("FAIL\n");
183 return 1;
184 }
185
186 printf("OK\n");
187 return 0;
188}
