2// 0.14rc1 introduced a regression where private kernel local variable (array)
3// was detected as an automatic local address space variable.
4// See https://github.com/pocl/pocl/issues/445
5
6#define CL_HPP_ENABLE_EXCEPTIONS
7#define CL_HPP_MINIMUM_OPENCL_VERSION 120
8#define CL_HPP_TARGET_OPENCL_VERSION 120
9#include <CL/cl2.hpp>
10#include <iostream>
11
12using namespace std;
13
14const char *SOURCE = R"CLC(
15#if __OPENCL_C_VERSION__ < 120
16#pragma OPENCL EXTENSION cl_khr_fp64: enable
17#endif
18
19__kernel void __attribute__ ((reqd_work_group_size(1, 1, 1)))
20private_local_array(__global int *__restrict__ out)
21{
22 int tmp[9] = { 1, 2, 3, 4, 5, 6, 7, 8, 9 };
23
24// for (int j = 0; j <= 2; ++j)
25#pragma no unroll
26 for (int i = 0; i < 9; ++i)
27 out[i] = tmp[i];
28}
29)CLC";
30
31int main(int, char **)
32{
33 try {
34 int N = 9;
35
36 cl::CommandQueue queue((cl_command_queue_properties)0);
37 cl::Program program(SOURCE, true);
38
39 auto kernel = cl::KernelFunctor<cl::Buffer>
40 (program, "private_local_array");
41
42 cl::Buffer buffer(CL_MEM_WRITE_ONLY, N*sizeof(cl_int));
43 kernel(cl::EnqueueArgs(queue, cl::NDRange(1), cl::NDRange(1)), buffer);
44
45 queue.finish();
46
47 cl_int *output = (cl_int*)queue.enqueueMapBuffer(
48 buffer, CL_TRUE, CL_MAP_READ, 0, N*sizeof(int));
49 for (int i = 0; i < N; i++) {
50 if ((int)output[i] != i + 1)
51 std::cout << "FAIL: " << output[i] << " should be " << i + 1
52 << std::endl;
53 }
54 queue.enqueueUnmapMemObject(buffer, output);
55 }
56 catch (cl::Error& err) {
57 std::cout << "FAIL with OpenCL error = " << err.err() << std::endl;
58 }
59 return 0;
60}
