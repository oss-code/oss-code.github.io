24// Enable OpenCL C++ exceptions
25#define CL_HPP_ENABLE_EXCEPTIONS
26#define CL_HPP_MINIMUM_OPENCL_VERSION 120
27#define CL_HPP_TARGET_OPENCL_VERSION 120
28#include <CL/cl2.hpp>
29
30#include <cstdio>
31#include <cstdlib>
32#include <iostream>
33
34#undef SRCDIR
35#include "config.h"
36
37#define WORK_ITEMS 1
38
39// Currently assume these types map the OpenCL types, might be better to use
40// the explicitly-sized types in <cstdint> under C++11 for better portability.
41//
42// We also assume the same packing rules between host and target, so if this
43// regression fails in the future, this might be worth investigating as to
44// the origin of the problem.
45
46struct int_single {
47 cl_int a;
48};
49
50struct int_pair {
51 cl_long a;
52 cl_long b;
53};
54
55struct test_struct {
56 cl_int elementA;
57 cl_int elementB;
58 cl_long elementC;
59 cl_char elementD;
60 cl_long elementE;
61 cl_float elementF;
62 cl_short elementG;
63#ifdef _CL_DISABLE_DOUBLE
64 cl_long elementH;
65#else
66 cl_double elementH;
67#endif
68};
69
70static char
71kernelSourceCode[] =
72"typedef struct int_single {\n"
73" int a; \n"
74"} int_single;\n"
75"typedef struct int_pair {\n"
76" long a;\n"
77" long b;\n"
78"} int_pair;\n"
79"typedef struct test_struct {\n"
80" int elementA;\n"
81" int elementB;\n"
82" long elementC;\n"
83" char elementD;\n"
84" long elementE;\n"
85" float elementF;\n"
86" short elementG;\n"
87"#ifdef cl_khr_fp64\n"
88" double elementH;\n"
89"#else\n"
90" long elementH;\n"
91"#endif\n"
92"} test_struct;\n"
93"\n"
94"kernel void test_single(int_single input, global int* output) {"
95" output[0] = input.a;\n"
96"}\n"
97"kernel void test_pair(int_pair input, global int* output) {"
98" output[0] = (int)input.a;\n"
99" output[1] = (int)input.b;\n"
100"}\n"
101"kernel void test_kernel(test_struct input, global int* output) {"
102" output[0] = input.elementA;\n"
103" output[1] = input.elementB;\n"
104" output[2] = (int)input.elementC;\n"
105" output[3] = (int)input.elementD;\n"
106" output[4] = (int)input.elementE;\n"
107" output[5] = (int)input.elementF;\n"
108" output[6] = (int)input.elementG;\n"
109" output[7] = (int)input.elementH;\n"
110"}\n";
111
112int
113main(void)
114{
115 bool ok = true;
116 int buffer_storage[8];
117
118 int_pair input_single;
119 input_single.a = 1234567;
120
121 int_pair input_pair;
122 input_pair.a = -5588;
123 input_pair.b = 8855;
124
125 test_struct input;
126 input.elementA = 1;
127 input.elementB = 2;
128 input.elementC = 3;
129 input.elementD = 4;
130 input.elementE = 5;
131 input.elementF = 6;
132 input.elementG = 7;
133 input.elementH = 8;
134
135 try {
136 std::vector<cl::Platform> platformList;
137
138 // Pick platform
139 cl::Platform::get(&platformList);
140
141 // Pick first platform
142 cl_context_properties cprops[] = {
143 CL_CONTEXT_PLATFORM, (cl_context_properties)(platformList[0])(), 0};
144 cl::Context context(CL_DEVICE_TYPE_ALL, cprops);
145
146 // Query the set of devices attched to the context
147 std::vector<cl::Device> devices = context.getInfo<CL_CONTEXT_DEVICES>();
148
149 // Create and program from source
150 cl::Program::Sources sources({kernelSourceCode});
151 cl::Program program(context, sources);
152
153 // Build program
154 program.build(devices);
155
156 // Create buffer for that uses the host ptr C
157 cl::Buffer cBuffer = cl::Buffer(
158 context,
159 CL_MEM_WRITE_ONLY | CL_MEM_USE_HOST_PTR,
160 8 * sizeof(int),
161 (void *) &buffer_storage[0]);
162
163 // Create command queue
164 cl::CommandQueue queue(context, devices[0], 0);
165
166 //
167 // int_single
168 //
169
170 // Create kernel object
171 cl::Kernel kernel_single(program, "test_single");
172
173 // Set kernel args
174 kernel_single.setArg(0, sizeof(int_single), &input_single);
175 kernel_single.setArg(1, cBuffer);
176
177 // Do the work
178 queue.enqueueNDRangeKernel(
179 kernel_single,
180 cl::NullRange,
181 cl::NDRange(1),
182 cl::NullRange
183 );
184
185 // Map cBuffer to host pointer. This enforces a sync with
186 // the host backing space, remember we choose GPU device.
187 int* output = (int*)queue.enqueueMapBuffer(
188 cBuffer,
189 CL_TRUE, // block
190 CL_MAP_READ,
191 0,
192 1 * sizeof(int));
193
194 if (*output != 1234567) {
195 std::cout
196 << "Small struct failure - size: 4 bytes expected: 123456 actual: "
197 << *output << std::endl;
198 ok = false;
199 }
200
201 queue.enqueueUnmapMemObject(cBuffer, output);
202
203 //
204 // int_pair
205 //
206
207 // Create kernel object
208 cl::Kernel kernel_pair(program, "test_pair");
209
210 // Set kernel args
211 kernel_pair.setArg(0, sizeof(int_pair), &input_pair);
212 kernel_pair.setArg(1, cBuffer);
213
214 // Do the work
215 queue.enqueueNDRangeKernel(
216 kernel_pair,
217 cl::NullRange,
218 cl::NDRange(1),
219 cl::NullRange
220 );
221
222 // Map cBuffer to host pointer. This enforces a sync with
223 // the host backing space, remember we choose GPU device.
224 output = (int*)queue.enqueueMapBuffer(
225 cBuffer,
226 CL_TRUE, // block
227 CL_MAP_READ,
228 0,
229 2 * sizeof(int));
230
231 if ((output[0] != -5588) || (output[1] != 8855)) {
232 std::cout
233 << "Small struct failure - size: 8 bytes expected: (-5588, 8855) actual: ("
234 << output[0] << ", " << output[1] << ")" << std::endl;
235 ok = false;
236 }
237
238 queue.enqueueUnmapMemObject(cBuffer, output);
239
240 //
241 // test_struct
242 //
243
244 // Create kernel object
245 cl::Kernel kernel(program, "test_kernel");
246
247 // Set kernel args
248 kernel.setArg(0, sizeof(test_struct), &input);
249 kernel.setArg(1, cBuffer);
250
251 // Do the work
252 queue.enqueueNDRangeKernel(
253 kernel,
254 cl::NullRange,
255 cl::NDRange(1),
256 cl::NullRange
257 );
258
259 // Map cBuffer to host pointer. This enforces a sync with
260 // the host backing space, remember we choose GPU device.
261 output = (int*)queue.enqueueMapBuffer(
262 cBuffer,
263 CL_TRUE, // block
264 CL_MAP_READ,
265 0,
266 8 * sizeof(int));
267
268 for (int i = 0; i < 8; i++) {
269 int correct = i + 1;
270 if (output[i] != correct) {
271 std::cout
272 << "F(" << i << ": " << output[i] << " != " << correct
273 << ") ";
274 ok = false;
275 }
276 }
277 if (ok)
278 return EXIT_SUCCESS;
279 else
280 return EXIT_FAILURE;
281 // There is no need to perform a finish on the final unmap
282 // or release any objects as this all happens implicitly with
283 // the C++ Wrapper API.
284 }
285 catch (cl::Error &err) {
286 std::cerr
287 << "ERROR: "
288 << err.what()
289 << "("
290 << err.err()
291 << ")"
292 << std::endl;
293
294 return EXIT_FAILURE;
295 }
296
297 return EXIT_SUCCESS;
298}
