2// Trying to build a faulty program twice results in NULL deref
3// See https://github.com/pocl/pocl/issues/577
4// should print "BUILD ERROR" twice then "OK" once.
5
6#define CL_HPP_ENABLE_EXCEPTIONS
7#define CL_HPP_MINIMUM_OPENCL_VERSION 120
8#define CL_HPP_TARGET_OPENCL_VERSION 120
9#include <CL/cl2.hpp>
10#include <iostream>
11
12const char *SOURCE = R"RAW(
13
14 __kernel void foo(__global int *input) {
15 !@#$%^&*();
16 }
17
18)RAW";
19
20int main(int argc, char *argv[]) {
21 cl_int err;
22 unsigned error_count = 0;
23 cl::Program program(SOURCE, false, &err);
24
25 for (unsigned i = 0; i < 2; i++) {
26 try {
27 program.compile();
28 } catch (cl::BuildError &e) {
29 std::cout << "BUILD ERROR\n";
30 error_count++;
31 }
32 }
33
34 if (error_count == 2) {
35 std::cout << "OK\n";
36 return 0;
37 } else {
38 std::cout << "FAIL\n";
39 return 1;
40 }
41}
