25#include "poclu.h"
26#include <CL/opencl.h>
27#include <stdio.h>
28#include <stdlib.h>
29
30#define NB_WORK_GROUP 32
31#define VEC_SIZE 32
32
33const char *kernelSource =
34 "__kernel void test(__global unsigned * restrict buffer, "
35 " __local unsigned * restrict local_input, "
36 " const unsigned vec_size) "
37 "{ "
38 " unsigned i, j; "
39 " size_t gid = get_global_id(0); "
40 " size_t lid = get_local_id(0); "
41 " size_t lsize = get_local_size(0); "
42 " event_t event_read, event_write; "
43 " event_read = async_work_group_copy(local_input, &buffer[gid*vec_size*lsize], vec_size*lsize, 0);"
44 " for (i=0; i<vec_size; i++) "
45 " { "
46 " if (i == 0) "
47 " wait_group_events(1, &event_read); "
48 " local_input[i*lsize+lid]++; "
49 " } "
50 " event_write = async_work_group_copy(&buffer[gid*vec_size*lsize], local_input, vec_size*lsize, event_write);"
51 " wait_group_events(1, &event_write); "
52 "} ";
53
54int main ()
55{
56 cl_platform_id platform;
57 cl_device_id device;
58 cl_context context;
59 cl_command_queue queue;
60 cl_program program_source, program_binary;
61 cl_kernel kernel;
62 cl_mem buffer;
63 cl_uint vec_size = VEC_SIZE;
64 cl_uint input_buffer[NB_WORK_GROUP * VEC_SIZE] = {0};
65 cl_uint output_buffer[NB_WORK_GROUP * VEC_SIZE] = {0};
66 cl_int err;
67
68 size_t global_size = NB_WORK_GROUP;
69 size_t local_size = 1;
70 size_t sizeof_buffer = global_size * vec_size * sizeof(unsigned);
71 size_t binary_size;
72
73 char *binary;
74
75 unsigned k, i;
76
77 for (k=0; k<global_size; k++)
78 {
79 for (i=0; i<vec_size; i++)
80 {
81 input_buffer[k*vec_size+i]=k*vec_size+i;
82 }
83 }
84
85 err = clGetPlatformIDs(1, &platform, NULL);
86 CHECK_CL_ERROR (err);
87 err = clGetDeviceIDs(platform, CL_DEVICE_TYPE_DEFAULT, 1, &device, NULL);
88 CHECK_CL_ERROR (err);
89 context = clCreateContext(0, 1, &device, NULL, NULL, &err);
90 CHECK_CL_ERROR (err);
91 queue = clCreateCommandQueue(context, device, 0, &err);
92 CHECK_CL_ERROR (err);
93
94 program_source = clCreateProgramWithSource(context, 1, &kernelSource, NULL, &err);
95 CHECK_CL_ERROR (err);
96 err = clBuildProgram(program_source, 0, NULL, NULL, NULL, NULL);
97 CHECK_CL_ERROR (err);
98 err = clGetProgramInfo(program_source, CL_PROGRAM_BINARY_SIZES, sizeof(size_t), &binary_size, NULL);
99 CHECK_CL_ERROR (err);
100 binary = (char *)malloc(sizeof(char)*binary_size);
101 TEST_ASSERT (binary);
102 err = clGetProgramInfo(program_source, CL_PROGRAM_BINARIES, sizeof(char*), &binary, NULL);
103 CHECK_CL_ERROR (err);
104
105 program_binary = clCreateProgramWithBinary(context, 1, &device, &binary_size,
106 (const unsigned char **)&binary, NULL, &err);
107 CHECK_CL_ERROR (err);
108 err = clBuildProgram(program_binary, 0, NULL, NULL, NULL, NULL);
109 CHECK_CL_ERROR (err);
110 kernel = clCreateKernel(program_binary, "test", &err);
111 CHECK_CL_ERROR (err);
112 buffer = clCreateBuffer(context, CL_MEM_READ_WRITE | CL_MEM_COPY_HOST_PTR, sizeof_buffer, &input_buffer, &err);
113 CHECK_CL_ERROR (err);
114 err = clSetKernelArg(kernel, 0, sizeof (cl_mem), &buffer);
115 CHECK_CL_ERROR (err);
116 err = clSetKernelArg(kernel, 1, sizeof (unsigned) * vec_size, NULL);
117 CHECK_CL_ERROR (err);
118 err = clSetKernelArg(kernel, 2, sizeof (cl_uint), &vec_size);
119 CHECK_CL_ERROR (err);
120 err = clEnqueueNDRangeKernel(queue, kernel, 1, NULL, &global_size, &local_size, 0, NULL, NULL);
121 CHECK_CL_ERROR (err);
122 clFinish(queue);
123 err = clEnqueueReadBuffer(queue, buffer, CL_TRUE, 0, sizeof_buffer, output_buffer, 0, NULL, NULL);
124 CHECK_CL_ERROR (err);
125
126 for (k=0; k<global_size; k++)
127 {
128 for (i=0; i<vec_size; i++)
129 {
130 unsigned expected = (input_buffer[k*vec_size+i]+1);
131 if (output_buffer[k*vec_size+i] != expected)
132 {
133 printf("Error at %u %u : %u != %u\n", k, i,
134 output_buffer[k*vec_size+i], expected);
135 return 1;
136 }
137 }
138 }
139
140 clReleaseMemObject(buffer);
141 clReleaseKernel(kernel);
142 clReleaseProgram(program_source);
143 clReleaseProgram(program_binary);
144 clReleaseCommandQueue(queue);
145 clReleaseContext(context);
146 clReleaseDevice(device);
147
148 return 0;
149}
