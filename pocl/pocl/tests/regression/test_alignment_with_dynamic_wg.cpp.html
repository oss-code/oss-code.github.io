10const char *SOURCE = R"RAW(
11
12__kernel void test(global uint *output, global const uint* trialValue){
13 /* Variable declarations */
14 int lid = get_local_id(0);
15
16 int i, j, k;
17
18 uint result[Y][Z];
19
20 __local uint localResult[X][Y][Z];
21
22 for (i = 0; i < Y; ++i)
23 for (j = 0; j < Z; ++j) {
24 result[i][j] = 0;
25 localResult[lid][i][j] = 0;
26 }
27
28 for (i = 0; i < Y; ++i)
29 for (j = 0; j < Z; ++j) {
30 result[i][j] += trialValue[j] * 4;
31 localResult[lid][i][j] += result[i][j];
32 }
33
34 barrier (CLK_LOCAL_MEM_FENCE);
35
36 uint sum = 0;
37
38 for (k = 0; k < X; ++k)
39 for (i = 0; i < Y; ++i)
40 for (j = 0; j < Z; ++j) {
41 sum += localResult[k][i][j];
42 }
43
44 output[lid] = sum;
45}
46
47)RAW";

49bool test_invocation(unsigned x, unsigned y, unsigned z,
50 const std::string &arg_x, const std::string &arg_y,
51 const std::string &arg_z, cl::CommandQueue &queue,
52 cl::Program &program) {
53
54 unsigned expected_sum = x * y * z * 4;
55
56 unsigned local_size = x;
57 assert(local_size > 0);
58 assert(local_size <= 256);
59
60 std::string options = "-cl-std=CL1.2";
61 options += " -DX=" + arg_x + " -DY=" + arg_y + " -DZ=" + arg_z;
62 program.build(options.c_str());
63
64 cl_uint *in1 = new cl_uint[z];
65 cl_uint *out = new cl_uint[x];
66 for (size_t i = 0; i < x; ++i) {
67 out[i] = 0;
68 }
69 for (size_t i = 0; i < z; ++i) {
70 in1[i] = 1;
71 }
72
73 cl::Buffer inbuf((cl_mem_flags)(CL_MEM_READ_ONLY | CL_MEM_USE_HOST_PTR),
74 (z * sizeof(cl_uint)), in1);
75 cl::Buffer outbuf((cl_mem_flags)(CL_MEM_WRITE_ONLY | CL_MEM_USE_HOST_PTR),
76 (x * sizeof(cl_uint)), out);
77
78 // This triggers compilation of dynamic WG binaries.
79 cl::Program::Binaries binaries{};
80 int err = program.getInfo<>(CL_PROGRAM_BINARIES, &binaries);
81 assert(err == CL_SUCCESS);
82
83 auto kernel = cl::KernelFunctor<cl::Buffer, cl::Buffer>(program, "test");
84
85 kernel(
86 cl::EnqueueArgs(queue, cl::NDRange(local_size), cl::NDRange(local_size)),
87 outbuf, inbuf);
88
89 queue.enqueueReadBuffer(outbuf, 1, 0, (x * sizeof(cl_uint)), out);
90
91 queue.finish();
92
93 bool correct = true;
94 for (size_t i = 0; i < x; ++i) {
95 if (out[i] != expected_sum)
96 correct = false;
97 }
98
99 std::cout << (correct ? "OK\n" : "FAIL\n");
100
101 delete[] in1;
102 delete[] out;
103
104 return correct;
105}

107int main(int argc, char *argv[]) {
108 cl::Device device = cl::Device::getDefault();
109 cl::CommandQueue queue = cl::CommandQueue::getDefault();
110 cl::Program program(SOURCE);
111
112 if (argc < 4) {
113 std::cout << "USAGE: $0 X Y Z\n";
114 return 1;
115 }
116
117 std::string arg_x(argv[1]);
118 std::string arg_y(argv[2]);
119 std::string arg_z(argv[3]);
120
121 unsigned x = std::stoi(argv[1]);
122 unsigned y = std::stoi(argv[2]);
123 unsigned z = std::stoi(argv[3]);
124
125 if (!test_invocation(x, y, z, arg_x, arg_y, arg_z, queue, program))
126 return 1;
127
128 if (!test_invocation(y, z, x, arg_y, arg_z, arg_x, queue, program))
129 return 1;
130
131 if (!test_invocation(z, x, y, arg_z, arg_x, arg_y, queue, program))
132 return 1;
133
134 return 0;
135}
