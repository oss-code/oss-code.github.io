31std::string read_text_file(const std::string& path)
32{
33 FILE* f = fopen(path.c_str(), "rb");
34
35 if(!f)
36 {
37 throw std::runtime_error("Unable to open " + path);
38 }
39
40 fseek(f, 0, SEEK_END);
41 size_t sz = ftell(f);
42 fseek(f, 0, SEEK_SET);
43
44 char* data = new char[sz];
45 if(fread(data, 1, sz, f) != sz)
46 {
47 delete [] data;
48 throw std::runtime_error("Unable to read " + path);
49 }
50 fclose(f);
51 std::string ret(data, sz);
52
53 delete [] data;
54 return ret;
55}

57cl::Platform get_platform(unsigned force_platform = 0)
58{
59 std::vector<cl::Platform> platforms;
60 cl::Platform::get(&platforms);
61 if(platforms.empty())
62 throw std::runtime_error("No platforms found!");
63
64 for(unsigned i = 0; i < platforms.size(); ++i)
65 {
66 cl::Platform& p = platforms[i];
67 std::string name = p.getInfo<CL_PLATFORM_NAME>();
68 std::cout << i << ": " << name << std::endl;
69 }
70
71 if(force_platform >= platforms.size()) force_platform = 0;
72
73 return platforms[force_platform];
74}

86void exclusive_scan_cpu(const std::vector<int>& input, std::vector<int>& output)
87{
88 output.resize(input.size());
89 int a = 0;
90 for(unsigned i = 0; i < input.size(); ++i)
91 {
92 output[i] = a;
93 a += input[i];
94 }
95}

97void exclusive_scan_cl(const std::vector<int>& input, std::vector<int>& output)
98{
99 // Fails on POCL, works with AMDGPU-PRO
100 cl::Platform p = get_platform(0);
101 cl::Device d = get_device(p);
102 cl::Context ctx(d);
103 std::string src = read_text_file(SRCDIR "/test_flatten_barrier_subs.cl");
104 cl::Program::Sources sources;
105 sources.push_back({src.c_str(), src.length()});
106 cl::Program program(ctx, sources);
107
108 if(program.build({d}) != CL_SUCCESS)
109 {
110 throw std::runtime_error(program.getBuildInfo<CL_PROGRAM_BUILD_LOG>(d));
111 }
112
113 cl::Buffer input_buf(ctx, CL_MEM_READ_WRITE, sizeof(int)*input.size());
114 cl::Buffer output_buf(ctx, CL_MEM_READ_WRITE, sizeof(int)*input.size());
115 cl::CommandQueue q(ctx, d);
116 q.enqueueWriteBuffer(
117 input_buf, CL_TRUE, 0, sizeof(int)*input.size(), input.data()
118 );
119
120 int numElems = input.size();
121#define WG_SIZE 64
122 int GROUP_BLOCK_SIZE_SCAN = (WG_SIZE << 3);
123 int GROUP_BLOCK_SIZE_DISTRIBUTE = (WG_SIZE << 2);
124
125 int NUM_GROUPS_BOTTOM_LEVEL_SCAN = (numElems + GROUP_BLOCK_SIZE_SCAN - 1) / GROUP_BLOCK_SIZE_SCAN;
126 int NUM_GROUPS_MID_LEVEL_SCAN = (NUM_GROUPS_BOTTOM_LEVEL_SCAN + GROUP_BLOCK_SIZE_SCAN - 1) / GROUP_BLOCK_SIZE_SCAN;
127 int NUM_GROUPS_TOP_LEVEL_SCAN = (NUM_GROUPS_MID_LEVEL_SCAN + GROUP_BLOCK_SIZE_SCAN - 1) / GROUP_BLOCK_SIZE_SCAN;
128
129 int NUM_GROUPS_BOTTOM_LEVEL_DISTRIBUTE = (numElems + GROUP_BLOCK_SIZE_DISTRIBUTE - 1) / GROUP_BLOCK_SIZE_DISTRIBUTE;
130 int NUM_GROUPS_MID_LEVEL_DISTRIBUTE = (NUM_GROUPS_BOTTOM_LEVEL_DISTRIBUTE + GROUP_BLOCK_SIZE_DISTRIBUTE - 1) / GROUP_BLOCK_SIZE_DISTRIBUTE;
131
132 cl::Buffer devicePartSumsBottomLevel(
133 ctx, CL_MEM_READ_WRITE, sizeof(int)*NUM_GROUPS_BOTTOM_LEVEL_SCAN
134 );
135 cl::Buffer devicePartSumsMidLevel(
136 ctx, CL_MEM_READ_WRITE, sizeof(int)*NUM_GROUPS_MID_LEVEL_SCAN
137 );
138 cl::Kernel bottomLevelScan(program, "scan_exclusive_part_int4");
139 cl::Kernel topLevelScan(program, "scan_exclusive_int4");
140 cl::Kernel distributeSums(program, "distribute_part_sum_int4");
141
142 bottomLevelScan.setArg(0, input_buf);
143 bottomLevelScan.setArg(1, output_buf);
144 bottomLevelScan.setArg(2, numElems);
145 bottomLevelScan.setArg(3, devicePartSumsBottomLevel);
146 bottomLevelScan.setArg(4, WG_SIZE * sizeof(cl_int), nullptr);
147 q.enqueueNDRangeKernel(
148 bottomLevelScan,
149 cl::NullRange,
150 cl::NDRange(NUM_GROUPS_BOTTOM_LEVEL_SCAN * WG_SIZE),
151 cl::NDRange(WG_SIZE)
152 );
153
154 bottomLevelScan.setArg(0, devicePartSumsBottomLevel);
155 bottomLevelScan.setArg(1, devicePartSumsBottomLevel);
156 bottomLevelScan.setArg(2, (cl_uint)NUM_GROUPS_BOTTOM_LEVEL_SCAN);
157 bottomLevelScan.setArg(3, devicePartSumsMidLevel);
158 bottomLevelScan.setArg(4, WG_SIZE * sizeof(cl_int), nullptr);
159 q.enqueueNDRangeKernel(
160 bottomLevelScan,
161 cl::NullRange,
162 cl::NDRange(NUM_GROUPS_MID_LEVEL_SCAN * WG_SIZE),
163 cl::NDRange(WG_SIZE)
164 );
165
166 topLevelScan.setArg(0, devicePartSumsMidLevel);
167 topLevelScan.setArg(1, devicePartSumsMidLevel);
168 topLevelScan.setArg(2, (cl_uint)NUM_GROUPS_MID_LEVEL_SCAN);
169 topLevelScan.setArg(3, WG_SIZE * sizeof(cl_int), nullptr);
170 q.enqueueNDRangeKernel(
171 topLevelScan,
172 cl::NullRange,
173 cl::NDRange(NUM_GROUPS_TOP_LEVEL_SCAN * WG_SIZE),
174 cl::NDRange(WG_SIZE)
175 );
176
177 distributeSums.setArg(0, devicePartSumsMidLevel);
178 distributeSums.setArg(1, devicePartSumsBottomLevel);
179 distributeSums.setArg(2, (cl_uint)NUM_GROUPS_BOTTOM_LEVEL_SCAN);
180 q.enqueueNDRangeKernel(
181 distributeSums,
182 cl::NullRange,
183 cl::NDRange(NUM_GROUPS_MID_LEVEL_DISTRIBUTE * WG_SIZE),
184 cl::NDRange(WG_SIZE)
185 );
186
187 distributeSums.setArg(0, devicePartSumsBottomLevel);
188 distributeSums.setArg(1, output_buf);
189 distributeSums.setArg(2, (cl_uint)numElems);
190 q.enqueueNDRangeKernel(
191 distributeSums,
192 cl::NullRange,
193 cl::NDRange(NUM_GROUPS_BOTTOM_LEVEL_DISTRIBUTE * WG_SIZE),
194 cl::NDRange(WG_SIZE)
195 );
196
197 output.resize(input.size());
198 q.enqueueReadBuffer(
199 output_buf, CL_TRUE, 0, sizeof(int)*input.size(), output.data()
200 );
201}

203std::vector<int> generate_hits(unsigned n)
204{
205 std::vector<int> res;
206 res.reserve(n);
207 for(unsigned i = 0; i < n; ++i) res.push_back(random()&1);
208 return res;
209}

211template<typename T>
212void print_vec(const std::vector<T>& t)
213{
214 for(unsigned i = 0; i < t.size(); ++i)
215 {
216 if(i != 0) std::cout << ", ";
217 std::cout << t[i];
218 }
219 std::cout << std::endl;
220}
221
222int main()
223{
224 std::vector<int> hits = generate_hits(100);
225 std::vector<int> indices;
226 exclusive_scan_cpu(hits, indices);
227 print_vec(hits);
228 print_vec(indices);
229
230 std::vector<int> cl_indices;
231 exclusive_scan_cl(hits, cl_indices);
232
233 if(indices == cl_indices)
234 std::cout << "CL gave correct results" << std::endl;
235 else
236 {
237 std::cout << "CL gave wrong results" << std::endl;
238 print_vec(cl_indices);
239 }
240
241 return 0;
242}
